# 客户端数据获取模式

## 1. 概述与背景

### 1.1 什么是客户端数据获取

客户端数据获取是指在浏览器端通过 JavaScript 获取数据的方式。与服务端数据获取不同,客户端数据获取发生在页面加载后,通过异步请求获取数据并更新 UI。

客户端数据获取的核心特点:

- **异步加载**: 页面先渲染,数据后加载
- **交互驱动**: 响应用户操作获取数据
- **实时更新**: 可以定期刷新数据
- **用户特定**: 获取当前用户的个性化数据

### 1.2 为什么需要客户端数据获取

虽然 Next.js 推荐使用服务端数据获取,但某些场景下客户端数据获取更合适:

- **用户特定数据**: 需要用户登录后才能获取的数据
- **实时数据**: 需要频繁更新的数据
- **交互触发**: 用户操作后才需要的数据
- **SEO 不重要**: 不需要搜索引擎索引的数据

### 1.3 客户端 vs 服务端数据获取

**对比表格**:

| 特性     | 客户端数据获取 | 服务端数据获取 |
| :------- | :------------- | :------------- |
| 执行时机 | 页面加载后     | 页面渲染前     |
| SEO      | 不友好         | 友好           |
| 首屏速度 | 快(先显示骨架) | 慢(等待数据)   |
| 实时性   | 好             | 一般           |
| 用户体验 | 需要加载状态   | 直接显示内容   |

```typescript
// hooks/useFetch.ts
import { useEffect, useState } from "react";

interface UseFetchResult<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
  refetch: () => void;
}

export function useFetch<T>(url: string): UseFetchResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [refetchTrigger, setRefetchTrigger] = useState(0);

  useEffect(() => {
    async function fetchData() {
      try {
        setLoading(true);
        setError(null);

        const res = await fetch(url);

        if (!res.ok) {
          throw new Error(`HTTP error! status: ${res.status}`);
        }

        const json = await res.json();
        setData(json);
      } catch (err) {
        setError(err instanceof Error ? err.message : "An error occurred");
      } finally {
        setLoading(false);
      }
    }

    fetchData();
  }, [url, refetchTrigger]);

  const refetch = () => setRefetchTrigger((prev) => prev + 1);

  return { data, loading, error, refetch };
}

// 使用示例

interface Post {

}

export default function Posts() {

    refetch,
  } = useFetch<Post[]>("/api/posts");

  if (error) return <div>Error: {error}</div>;

    <div>
      <button onClick={refetch}>Refresh</button>

          <li key={post.id}>{post.title}</li>
```

```typescript
"use client";

import { useEffect, useState, useRef } from "react";

interface Post {
  id: string;
  title: string;
  content: string;
}

export default function InfiniteScroll() {
  const [posts, setPosts] = useState<Post[]>([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);
  const observerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    async function loadPosts() {
      if (loading || !hasMore) return;

      setLoading(true);

      try {
        const res = await fetch(`/api/posts?page=${page}&limit=10`);
        const data = await res.json();

        if (data.length === 0) {
          setHasMore(false);
        } else {
          setPosts((prev) => [...prev, ...data]);
          setPage((prev) => prev + 1);
        }
      } catch (error) {
        console.error("Failed to load posts:", error);
      } finally {
        setLoading(false);
      }
    }

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          loadPosts();
        }
      },
      { threshold: 1.0 }
    );

    if (observerRef.current) {
      observer.observe(observerRef.current);
    }

    return () => observer.disconnect();
  }, [page, loading, hasMore]);

  return (
    <div>
      {posts.map((post) => (
        <article key={post.id}>
          <h2>{post.title}</h2>
          <p>{post.content}</p>
        </article>
      ))}

      {loading && <div>Loading more...</div>}
      {hasMore && <div ref={observerRef} style={{ height: "20px" }} />}
      {!hasMore && <div>No more posts</div>}
    </div>
  );
}
```

```typescript
import { useEffect, useState } from "react";

export default function RetryFetch() {
  const [data, setData] = useState(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchWithRetry(url: string, maxRetries = 3) {
      for (let i = 0; i < maxRetries; i++) {
        try {
          const res = await fetch(url);

          if (!res.ok) {
            throw new Error(`HTTP error! status: ${res.status}`);
          }

          const data = await res.json();
          setData(data);
          return;
        } catch (err) {
          if (i === maxRetries - 1) {
            setError(err instanceof Error ? err.message : "Failed to fetch");
          } else {
            // 等待后重试
            await new Promise((resolve) => setTimeout(resolve, 1000 * (i + 1)));
          }
        }
      }
    }

    fetchWithRetry("/api/data");
  }, []);

  if (error) return <div>Error: {error}</div>;
  if (!data) return <div>Loading...</div>;

  return <div>{JSON.stringify(data)}</div>;
}
```

```typescript
"use client";

import { useEffect, useState } from "react";

export default function ErrorHandling() {
  const [data, setData] = useState(null);
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchData() {
      try {
        setLoading(true);
        setError(null);

        const res = await fetch("/api/data");

        if (!res.ok) {
          throw new Error(`HTTP error! status: ${res.status}`);
        }

        const json = await res.json();
        setData(json);
      } catch (err) {
        setError(err instanceof Error ? err.message : "An error occurred");
      } finally {
        setLoading(false);
      }
    }

    fetchData();
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;

  return <div>{JSON.stringify(data)}</div>;
}
```

```typescript
"use client";

import { useState, useEffect } from "react";

function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => clearTimeout(timer);
  }, [value, delay]);

  return debouncedValue;
}

export default function DebouncedSearch() {
  const [query, setQuery] = useState("");
  const debouncedQuery = useDebounce(query, 500);
  const [results, setResults] = useState([]);

  useEffect(() => {
    if (!debouncedQuery) return;

    fetch(`/api/search?q=${encodeURIComponent(debouncedQuery)}`)
      .then((res) => res.json())
      .then(setResults);
  }, [debouncedQuery]);

  return (
    <div>
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
      />
      <ul>
        {results.map((result: any) => (
          <li key={result.id}>{result.title}</li>
        ))}
      </ul>
    </div>
  );
}
```

## 7. 常见问题

### 7.1 基础问题

#### 问题一: 客户端数据获取和服务端数据获取有什么区别?

**问题**: 什么时候用客户端数据获取,什么时候用服务端数据获取?

**简短回答**: 服务端数据获取适合 SEO 和首屏数据,客户端数据获取适合用户特定和实时数据。

**详细解释**:

服务端数据获取在页面渲染前完成,对 SEO 友好,但首屏加载慢。客户端数据获取在页面加载后进行,首屏快但需要加载状态。

**代码示例**:

```typescript
// 服务端数据获取
export default async function ServerPage() {
  const data = await fetch("https://api.example.com/data");
  const posts = await data.json();

  return (
    <div>
      {posts.map((p) => (
        <div key={p.id}>{p.title}</div>
      ))}
    </div>
  );
}

// 客户端数据获取
("use client");

import { useEffect, useState } from "react";

export default function ClientPage() {
  const [posts, setPosts] = useState([]);

  useEffect(() => {
    fetch("/api/posts")
      .then((res) => res.json())
      .then(setPosts);
  }, []);

  return (
    <div>
      {posts.map((p) => (
        <div key={p.id}>{p.title}</div>
      ))}
    </div>
  );
}
```

#### 问题二: 如何避免内存泄漏?

**问题**: 组件卸载后如何避免状态更新导致的内存泄漏?

**简短回答**: 使用清理函数或 AbortController 取消请求。

**详细解释**:

在 useEffect 的清理函数中设置标志或取消请求,避免组件卸载后更新状态。

**代码示例**:

```typescript
"use client";

import { useEffect, useState } from "react";

export default function SafeComponent() {
  const [data, setData] = useState(null);

  useEffect(() => {
    const controller = new AbortController();

    fetch("/api/data", { signal: controller.signal })
      .then((res) => res.json())
      .then(setData)
      .catch((err) => {
        if (err.name !== "AbortError") {
          console.error(err);
        }
      });

    return () => controller.abort();
  }, []);

  return <div>{JSON.stringify(data)}</div>;
}
```

#### 问题三: 如何处理加载状态?

**问题**: 如何优雅地显示加载状态?

**简短回答**: 使用 loading 状态和骨架屏。

**详细解释**:

使用 useState 管理 loading 状态,显示加载指示器或骨架屏。

**代码示例**:

```typescript
"use client";

import { useEffect, useState } from "react";

export default function LoadingState() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch("/api/data")
      .then((res) => res.json())
      .then((data) => {
        setData(data);
        setLoading(false);

  if (loading) {
    return (
      <div className="skeleton">
        <div className="skeleton-line" />
        <div className="skeleton-line" />
        <div className="skeleton-line" />
      </div>
    );
  }

  return <div>{JSON.stringify(data)}</div>;
}
```

#### 问题五: 如何实现乐观更新?

**问题**: 如何在请求完成前更新 UI?

**简短回答**: 先更新本地状态,请求失败时回滚。

**详细解释**:

立即更新 UI 提升响应速度,请求失败时恢复原状态。

**代码示例**:

```typescript
"use client";

import { useState } from "react";

export default function OptimisticUpdate() {
  const [count, setCount] = useState(0);

  const handleLike = async () => {
    const previousCount = count;

    // 乐观更新
    setCount(count + 1);

    try {
      await fetch("/api/like", { method: "POST" });
    } catch (error) {
      // 失败时回滚
      setCount(previousCount);
    }
  };

  return (
    <div>
      <p>Likes: {count}</p>
      <button onClick={handleLike}>Like</button>
    </div>
  );
}
```
