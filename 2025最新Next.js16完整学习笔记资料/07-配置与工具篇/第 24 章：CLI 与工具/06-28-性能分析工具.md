## Core Web Vitals

Core Web Vitals 是 Google 定义的核心性能指标,直接影响 SEO 排名和用户体验。

### 三大核心指标

#### 1. LCP (Largest Contentful Paint)

LCP 衡量页面主要内容的加载速度。

**目标值**:

- 优秀: < 2.5 秒
- 需要改进: 2.5-4 秒
- 差: > 4 秒

```javascript
// app/components/LCPMonitor.tsx
"use client";

import { useEffect } from "react";

export default function LCPMonitor() {
  useEffect(() => {
    // 监控 LCP
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const lastEntry = entries[entries.length - 1];

      console.log("LCP:", {
        value: lastEntry.renderTime || lastEntry.loadTime,
        element: lastEntry.element,
        url: lastEntry.url,
      });

      // 发送到分析服务
      sendToAnalytics({
        metric: "LCP",
        value: lastEntry.renderTime || lastEntry.loadTime,
      });
    });

    observer.observe({ type: "largest-contentful-paint", buffered: true });

    return () => observer.disconnect();
  }, []);

  return null;
}

function sendToAnalytics(data: any) {
  // 发送到分析服务
  if (typeof window !== "undefined" && window.gtag) {
    window.gtag("event", "web_vitals", data);
  }
}
```

#### 2. INP (Interaction to Next Paint)

INP 衡量页面对用户交互的响应速度。

**目标值**:

- 优秀: < 200ms
- 需要改进: 200-500ms
- 差: > 500ms

```javascript
// app/components/INPMonitor.tsx
"use client";

import { useEffect } from "react";

export default function INPMonitor() {
  useEffect(() => {
    // 监控 INP
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        console.log("INP:", {
          duration: entry.duration,
          name: entry.name,
          startTime: entry.startTime,
        });

        // 如果交互延迟超过阈值,记录详情
        if (entry.duration > 200) {
          console.warn("慢交互检测:", {
            duration: entry.duration,
            target: entry.target,
            type: entry.name,
          });

          sendToAnalytics({
            metric: "INP",
            value: entry.duration,
            slow: true,
          });
        }
      }
    });

    observer.observe({ type: "event", buffered: true, durationThreshold: 40 });

    return () => observer.disconnect();
  }, []);

  return null;
}

function sendToAnalytics(data: any) {
  if (typeof window !== "undefined" && window.gtag) {
    window.gtag("event", "web_vitals", data);
  }
}
```

#### 3. CLS (Cumulative Layout Shift)

CLS 衡量页面视觉稳定性。

**目标值**:

- 优秀: < 0.1
- 需要改进: 0.1-0.25
- 差: > 0.25

```javascript
// app/components/CLSMonitor.tsx
"use client";

import { useEffect } from "react";

export default function CLSMonitor() {
  useEffect(() => {
    let clsValue = 0;
    let clsEntries = [];

    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        // 只统计非用户输入导致的布局偏移
        if (!entry.hadRecentInput) {
          clsValue += entry.value;
          clsEntries.push(entry);

          console.log("CLS:", {
            value: entry.value,
            累计值: clsValue,
            sources: entry.sources,
          });
        }
      }

      // 发送累计 CLS 值
      sendToAnalytics({
        metric: "CLS",
        value: clsValue,
        entries: clsEntries.length,
      });
    });

    observer.observe({ type: "layout-shift", buffered: true });

    return () => observer.disconnect();
  }, []);

  return null;
}

function sendToAnalytics(data: any) {
  if (typeof window !== "undefined" && window.gtag) {
    window.gtag("event", "web_vitals", data);
  }
}
```

```javascript
"use client";

import { useEffect } from "react";

export default function ResourceTiming() {
  useEffect(() => {
    window.addEventListener("load", () => {
      const resources = performance.getEntriesByType("resource");

      // 按类型分组
      const byType = resources.reduce((acc, resource) => {
        const type = resource.initiatorType;
        if (!acc[type]) acc[type] = [];
        acc[type].push({
          name: resource.name,
          duration: resource.duration,
          size: resource.transferSize,
        });
        return acc;
      }, {});

      console.log("资源加载统计:", byType);

      const slowResources = resources
        .filter((r) => r.duration > 1000)
        .map((r) => ({
          name: r.name,
          duration: r.duration,
          type: r.initiatorType,
        }));

      if (slowResources.length > 0) {
        console.warn("慢资源:", slowResources);
        sendToAnalytics({
          metric: "slow_resources",
          resources: slowResources,
        });
      }
    });
```

```javascript
// app/utils/performance-marks.ts
export class PerformanceMarks {
  static mark(name: string) {
    performance.mark(name);
  }

  static measure(name: string, startMark: string, endMark?: string) {
    if (endMark) {
      performance.measure(name, startMark, endMark);
    } else {
      performance.measure(name, startMark);
    }

    const measure = performance.getEntriesByName(name, "measure")[0];
    console.log(`${name}: ${measure.duration.toFixed(2)}ms`);

    return measure.duration;
  }

  static clearMarks(name?: string) {
    if (name) {
      performance.clearMarks(name);
    } else {
      performance.clearMarks();
    }
  }

  static clearMeasures(name?: string) {
    if (name) {
      performance.clearMeasures(name);
    } else {
      performance.clearMeasures();
    }
  }
}
```

### 自定义性能监控

```javascript
// app/utils/performance-monitor.ts
class PerformanceMonitor {
  private metrics: Map<string, number[]> = new Map()

  track(name: string, value: number) {
    if (!this.metrics.has(name)) {
      this.metrics.set(name, [])
    }
    this.metrics.get(name)!.push(value)
  }

  getStats(name: string) {
    const values = this.metrics.get(name) || []
    if (values.length === 0) return null

    const sorted = [...values].sort((a, b) => a - b)
    return {
      count: values.length,
      min: sorted[0],
      max: sorted[sorted.length - 1],
      avg: values.reduce((a, b) => a + b, 0) / values.length,
      p50: sorted[Math.floor(sorted.length * 0.5)],
      p75: sorted[Math.floor(sorted.length * 0.75)],
      p95: sorted[Math.floor(sorted.length * 0.95)],
      p99: sorted[Math.floor(sorted.length * 0.99)],
    }
  }

  report() {
    const report = {}
    for (const [name, _] of this.metrics) {
      report[name] = this.getStats(name)
    }
    return report
  }

  clear() {
    this.metrics.clear()
  }
}

export const performanceMonitor = new PerformanceMonitor()

// 使用示例
// app/components/MonitoredComponent.tsx

export default function MonitoredComponent() {
```

### 案例 6: 大数据列表性能优化

```javascript
// app/components/VirtualList.tsx
"use client";

import { useState, useEffect, useRef } from "react";

export default function VirtualList({ items, itemHeight = 50 }) {
  const [visibleRange, setVisibleRange] = useState({ start: 0, end: 20 });
  const containerRef = useRef(null);

  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const handleScroll = () => {
      const scrollTop = container.scrollTop;
      const start = Math.floor(scrollTop / itemHeight);
      const end = start + Math.ceil(container.clientHeight / itemHeight);

      setVisibleRange({ start, end });
    };

    container.addEventListener("scroll", handleScroll);
    return () => container.removeEventListener("scroll", handleScroll);
  }, [itemHeight]);

  const visibleItems = items.slice(visibleRange.start, visibleRange.end);
  const offsetY = visibleRange.start * itemHeight;

  return (
    <div ref={containerRef} style={{ height: "600px", overflow: "auto" }}>
      <div style={{ height: items.length * itemHeight, position: "relative" }}>
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {visibleItems.map((item, index) => (
            <div
              key={visibleRange.start + index}
              style={{ height: itemHeight }}
            >
              {item.name}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```

```javascript
import { useState, useEffect, useRef } from "react";
import Image from "next/image";

export default function LazyImage({ src, alt, ...props }) {
  const [isVisible, setIsVisible] = useState(false);
  const imgRef = useRef(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.disconnect();
        }
      },
      { rootMargin: "50px" }
    );

    if (imgRef.current) {
      observer.observe(imgRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <div ref={imgRef}>
      {isVisible ? (
        <Image src={src} alt={alt} {...props} />
      ) : (
        <div
          style={{
            width: props.width,
            height: props.height,
            background: "#f0f0f0",
          }}
        />
      )}
    </div>
  );
}
```

### 案例 8: API 请求缓存优化

```javascript
// app/utils/api-cache.ts
class APICache {
  private cache = new Map()
  private timestamps = new Map()
  private ttl = 5 * 60 * 1000 // 5 分钟

  async get(key: string, fetcher: () => Promise<any>) {
    const now = Date.now()
    const timestamp = this.timestamps.get(key)

    // 检查缓存是否有效
    if (this.cache.has(key) && timestamp && now - timestamp < this.ttl) {
      console.log('从缓存返回:', key)
      return this.cache.get(key)
    }

    // 获取新数据
    console.log('从 API 获取:', key)
    const data = await fetcher()

    this.cache.set(key, data)
    this.timestamps.set(key, now)

    return data
  }

  clear() {
    this.cache.clear()
    this.timestamps.clear()
  }

  delete(key: string) {
    this.cache.delete(key)
    this.timestamps.delete(key)
  }
}

export const apiCache = new APICache()

// 使用示例
async function getUser(id: string) {
  return apiCache.get(`user-${id}`, async () => {
    const res = await fetch(`/api/users/${id}`)
    return res.json()
  })
}
```

```javascript
class AnalyticsBatcher {
  private queue: any[] = []
  private timer: NodeJS.Timeout | null = null

  add(data: any) {
    this.queue.push(data)

    if (this.queue.length >= 10) {
      this.flush()
    } else if (!this.timer) {
      this.timer = setTimeout(() => this.flush(), 5000)
    }
  }

  flush() {
    if (this.queue.length === 0) return

    fetch('/api/analytics/batch', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(this.queue),
    })

    this.queue = []
    if (this.timer) {
      clearTimeout(this.timer)
      this.timer = null
    }
  }
}
```

```javascript
const resources = performance.getEntriesByType("resource");
const slowResources = resources.filter((r) => r.duration > 1000);
console.log("慢资源:", slowResources);
```

### 14. 如何导出性能数据?

**问题**: 需要导出性能数据进行分析

**解决方案**:

```javascript
const perfData = performance.getEntries();
const json = JSON.stringify(perfData, null, 2);
const blob = new Blob([json], { type: "application/json" });
const url = URL.createObjectURL(blob);
const a = document.createElement("a");
a.href = url;
a.download = "performance-data.json";
a.click();
```

### 15. 如何设置性能告警?

**问题**: 性能下降时需要及时通知

**解决方案**:

在监控服务中设置告警规则:

```javascript
// 发送告警
if (lcp > 2500) {
  fetch("/api/alert", {
    method: "POST",
    body: JSON.stringify({
      metric: "LCP",
      value: lcp,
      threshold: 2500,
    }),
  });
}
```

## 总结

性能分析是优化 Web 应用的关键环节。通过掌握各种性能分析工具,可以:

1. **监控核心指标**:使用 Web Vitals 监控 LCP、INP、CLS
2. **自动化测试**:使用 Lighthouse CI 在 CI/CD 中运行性能测试
3. **实时监控**:使用 RUM 工具监控真实用户性能
4. **深入分析**:使用 Performance API 获取详细性能数据
5. **持续优化**:建立性能监控和优化的闭环

关键要点:

- 使用 Core Web Vitals 作为核心性能指标
- 在开发、测试、生产环境都要进行性能监控
- 使用 Lighthouse 进行自动化性能测试
- 使用 Performance API 获取详细的性能数据
- 注意监控开销,使用采样和批量发送
- 保护用户隐私,过滤敏感信息
- 设置性能预算和告警机制
- 定期分析性能数据并优化

记住:性能优化是一个持续的过程。通过建立完善的性能监控体系,及时发现和解决性能问题,为用户提供更好的体验。
