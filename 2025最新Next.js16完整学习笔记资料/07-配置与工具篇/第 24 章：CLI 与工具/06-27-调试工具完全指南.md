import { useEffect, useState } from 'react'
import { DebugLogger } from '@/utils/debug'

export default function UserList() {
  const [users, setUsers] = useState([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    async function fetchUsers() {
      DebugLogger.time('fetchUsers')

      try {
        DebugLogger.log('开始获取用户数据')

        const response = await fetch('/api/users')
        const data = await response.json()

        DebugLogger.group('用户数据详情', () => {
          DebugLogger.log('用户数量:', data.length)
          DebugLogger.table(data)
        })

        setUsers(data)
      } catch (error) {
        DebugLogger.error('获取用户失败', error as Error)
      } finally {
        setLoading(false)
        DebugLogger.timeEnd('fetchUsers')
      }
    }

    fetchUsers()
  }, [])

  return (
    <div>
      {loading ? '加载中...' : users.map(user => (
        <div key={user.id}>{user.name}</div>
      ))}
    </div>
  )

// app/utils/performance-monitor.ts
export class PerformanceMonitor {
  // 开始计时
  static start(name: string) {
    performance.mark(`${name}-start`);
  }

  // 结束计时
  static end(name: string) {
    performance.mark(`${name}-end`);
    performance.measure(name, `${name}-start`, `${name}-end`);

    const measure = performance.getEntriesByName(name)[0];
    console.log(`${name} 耗时: ${measure.duration.toFixed(2)}ms`);

    return measure.duration;
  }

  // 获取所有测量结果
  static getAll() {
    return performance.getEntriesByType("measure");
  }

  // 清除所有标记
  static clear() {
    performance.clearMarks();
    performance.clearMeasures();
  }

  // 导出性能数据
  static export() {
    const measures = performance.getEntriesByType("measure");
    return measures.map((m) => ({
      name: m.name,
      duration: m.duration,
      startTime: m.startTime,
    }));
  }
}

### Error Boundary

```javascript
// app/components/ErrorBoundary.tsx
'use client'

import { Component, ReactNode } from 'react'

interface Props {
  children: ReactNode
  fallback?: ReactNode
}

interface State {
  hasError: boolean
  error: Error | null
  errorInfo: any
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props)
    this.state = {
      hasError: false,
      error: null,
      errorInfo: null
    }
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true }
  }

  componentDidCatch(error: Error, errorInfo: any) {
    // 记录错误详情
    console.error('错误边界捕获错误:', {
      error: error.toString(),
      stack: error.stack,
      componentStack: errorInfo.componentStack,
      timestamp: new Date().toISOString()
    })

    this.setState({
      error,
      errorInfo
    })

    // 发送错误到监控服务
    this.reportError(error, errorInfo)
  }

  reportError(error: Error, errorInfo: any) {
    // 发送到错误追踪服务
    fetch('/api/log-error', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        message: error.message,
        stack: error.stack,
        componentStack: errorInfo.componentStack,
        userAgent: navigator.userAgent,
        url: window.location.href,
        timestamp: new Date().toISOString()
      })
    })
  }

  render() {
    if (this.state.hasError) {
      if (this.props.fallback) {
        return this.props.fallback
      }

      return (
        <div style={{ padding: '20px', border: '1px solid red' }}>
          <h2>出错了</h2>
          <details style={{ whiteSpace: 'pre-wrap' }}>
            <summary>错误详情</summary>
            <p>{this.state.error?.toString()}</p>
            <p>{this.state.error?.stack}</p>
            <p>{this.state.errorInfo?.componentStack}</p>
          </details>
        </div>
      )
    }

    return this.props.children
  }
}

  // 捕获未处理的 Promise 拒绝
  window.addEventListener('unhandledrejection', (event) => {
    console.error('未处理的 Promise 拒绝:', {

    })

      type: 'unhandledrejection',
      message: event.reason?.message || String(event.reason),
      stack: event.reason?.stack
    })

    event.preventDefault()
  })

  // 捕获全局错误
  window.addEventListener('error', (event) => {
    console.error('全局错误:', {
      message: event.message,
      filename: event.filename,
      lineno: event.lineno,
      colno: event.colno,
      error: event.error,
      timestamp: new Date().toISOString()
    })

    reportError({
      type: 'error',
      message: event.message,
      filename: event.filename,
      lineno: event.lineno,
      colno: event.colno,
      stack: event.error?.stack
    })
  })

  // 捕获资源加载错误
  window.addEventListener('error', (event) => {
    if (event.target !== window) {
      console.error('资源加载失败:', {
        tagName: (event.target as HTMLElement).tagName,
        src: (event.target as any).src || (event.target as any).href,
        timestamp: new Date().toISOString()
      })
    }
  }, true)
}

    })

// app/utils/safe-logger.ts
export function safeLog(message: string, data: any) {
  // 过滤敏感信息
  const sanitized = sanitizeData(data);
  console.log(message, sanitized);
}

function sanitizeData(data: any): any {
  if (typeof data !== "object" || data === null) {
    return data;
  }

  const sensitiveKeys = ["password", "token", "secret", "apiKey", "creditCard"];
  const result = { ...data };

  for (const key in result) {
    if (
      sensitiveKeys.some((k) => key.toLowerCase().includes(k.toLowerCase()))
    ) {
      result[key] = "***";
    } else if (typeof result[key] === "object") {
      result[key] = sanitizeData(result[key]);
    }
  }

  return result;
}

) {
  console.log(`${id} 渲染耗时: ${actualDuration}ms`);
}

export default function App() {
  return (
    <Profiler id="App" onRender={onRenderCallback}>
      <YourComponent />
    </Profiler>
  );
}
```

## 总结

调试是开发过程中的重要环节。通过掌握各种调试工具和技巧,可以:

1. **快速定位问题**:使用浏览器 DevTools 和 VSCode 调试器
2. **分析性能**:使用 Performance 和 Profiler 工具
3. **追踪错误**:使用 Error Boundary 和 Sentry
4. **监控生产环境**:使用 LogRocket 和错误追踪服务
5. **优化开发体验**:配置合适的调试工具链

关键要点:

- 熟练使用 Chrome DevTools 的各个面板
- 配置 VSCode 调试器进行断点调试
- 使用 React DevTools 检查组件状态
- 在生产环境使用错误追踪服务
- 注意保护敏感信息和性能开销
- 使用 Source Maps 定位源代码
- 建立完善的日志系统
- 使用性能监控工具优化应用

记住:好的调试习惯能大幅提升开发效率。根据不同场景选择合适的调试工具,建立系统化的调试流程,让问题无处遁形。
