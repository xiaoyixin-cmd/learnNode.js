## 3. 组件组织规范

### 3.1 组件结构

**标准组件结构**:

```typescript
"use client";

import { useState } from "react";
import type { ComponentProps } from "./types";

// 1. 类型定义
interface UserCardProps {
  user: {
    id: string;
    name: string;
    email: string;
  };
  onEdit?: (id: string) => void;
  className?: string;
}

// 2. 常量
const DEFAULT_AVATAR = "/images/avatar.png";

// 3. 主组件
export function UserCard({ user, onEdit, className }: UserCardProps) {
  // 3.1 Hooks
  const [isExpanded, setIsExpanded] = useState(false);

  // 3.2 派生状态
  const displayName = user.name || "Anonymous";

  // 3.3 事件处理
  const handleEdit = () => {
    onEdit?.(user.id);
  };

  const handleToggle = () => {
    setIsExpanded((prev) => !prev);
  };

  // 3.4 渲染逻辑
  return (
    <div className={className}>
      <h3>{displayName}</h3>
      <p>{user.email}</p>
      {isExpanded && (
        <div>
          <button onClick={handleEdit}>编辑</button>
        </div>
      )}
      <button onClick={handleToggle}>{isExpanded ? "收起" : "展开"}</button>
    </div>
  );
}

// 4. 子组件(如果需要)
function UserAvatar({ src }: { src: string }) {
  return <img src={src || DEFAULT_AVATAR} alt="Avatar" />;
}
```

**拆分标准**:

| 指标       | 阈值        | 建议            |
| :--------- | :---------- | :-------------- |

| JSX 嵌套   | > 5 层      | 提取子组件      |
| 重复代码   | > 3 次      | 抽象为组件      |

## 7. 错误处理规范

### 7.1 错误处理模式

**统一错误处理**:

```typescript
// ✅ 自定义错误类
class AppError extends Error {
  constructor(
    message: string,
    public code: string,
    public statusCode: number = 500
  ) {
    super(message);
    this.name = "AppError";
  }
}

class ValidationError extends AppError {
  constructor(message: string) {
    super(message, "VALIDATION_ERROR", 400);
    this.name = "ValidationError";
  }
}

class NotFoundError extends AppError {
  constructor(resource: string) {
    super(`${resource} not found`, "NOT_FOUND", 404);
    this.name = "NotFoundError";
  }
}

// ✅ 使用
async function getUser(id: string): Promise<User> {
  const user = await db.user.findUnique({ where: { id } });

  if (!user) {
    throw new NotFoundError("User");
  }

  return user;
}

// ✅ 统一错误处理
try {
  const user = await getUser(userId);
} catch (error) {
  if (error instanceof NotFoundError) {
    return { error: "User not found" };
  }
  if (error instanceof ValidationError) {
    return { error: error.message };
  }
  throw error;
}
```

### 7.2 Server Actions 错误处理

**Server Actions 模式**:

```typescript
"use server";

import { z } from "zod";

const createUserSchema = z.object({
  name: z.string().min(2),
  email: z.string().email(),
});

export async function createUser(formData: FormData) {
  try {
    // 验证数据
    const data = createUserSchema.parse({
      name: formData.get("name"),
      email: formData.get("email"),
    });

    // 创建用户
    const user = await db.user.create({ data });

    return { success: true, user };
  } catch (error) {
    if (error instanceof z.ZodError) {
      return {
        success: false,
        error: "数据验证失败",
        details: error.errors,
      };
    }

    return {
```

## 11. 常见问题

### 11.1 如何处理导入顺序?

**问题**: 手动排序导入语句很麻烦

**解答**:

使用 ESLint 插件自动排序:

```bash
npm install -D eslint-plugin-import
```

```json
// .eslintrc.json
{
  "plugins": ["import"],
  "rules": {
    "import/order": [
      "error",
      {
        "groups": [
          "builtin",
          "external",
          "internal",
          "parent",
          "sibling",
          "index"
        ],
        "newlines-between": "always",
        "alphabetize": {
          "order": "asc"
        }
      }
    ]
  }
}
```

运行 `npm run lint:fix` 自动修复。

---

### 11.2 何时使用 Interface 何时使用 Type?

**问题**: Interface 和 Type 如何选择

**解答**:

| 场景     | 推荐      | 原因             |
| :------- | :-------- | :--------------- |
| 对象形状 | Interface | 性能更好,可扩展  |
| 联合类型 | Type      | Interface 不支持 |
| 函数类型 | Type      | 更简洁           |
| 工具类型 | Type      | 支持映射类型     |
| 公共 API | Interface | 可声明合并       |
