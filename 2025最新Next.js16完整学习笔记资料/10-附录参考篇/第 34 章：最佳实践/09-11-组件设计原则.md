# 09-11-组件设计原则

## 1. 概述 (Overview)

### 1.1 为什么需要组件设计原则

组件是 React 和 Next.js 应用的基本构建块。良好的组件设计能够提高代码复用性、可维护性和可测试性。在 Next.js 16 中,Server Components 和 Client Components 的区分使得组件设计更加重要。

### 1.2 Next.js 16 组件设计特点

**核心变化**:

- Server Components 默认
- Client Components 需要明确标记
- 组件边界更加清晰
- 性能优化更加重要

### 1.3 设计原则概览

| 原则     | 说明               | 重要性 |
| :------- | :----------------- | :----- |
| 单一职责 | 一个组件只做一件事 | 高     |
| 可组合性 | 组件可以灵活组合   | 高     |
| 可复用性 | 组件可以在多处使用 | 中     |
| 可测试性 | 组件易于测试       | 高     |
| 性能优先 | 考虑渲染性能       | 高     |

## 2. 单一职责原则

### 2.1 概念

每个组件应该只有一个明确的职责,只做一件事情。

### 2.2 反例与正例

**反例 - 职责过多**:

```typescript
// ❌ 组件职责过多
export function UserDashboard() {
  const [user, setUser] = useState(null);
  const [posts, setPosts] = useState([]);
  const [comments, setComments] = useState([]);
  const [notifications, setNotifications] = useState([]);

  useEffect(() => {
    // 获取用户数据
    fetchUser().then(setUser);
    // 获取文章
    fetchPosts().then(setPosts);
    // 获取评论
    fetchComments().then(setComments);
    // 获取通知
    fetchNotifications().then(setNotifications);
  }, []);

  return (
    <div>
      {/* 用户信息 */}
      <div>{user?.name}</div>
      {/* 文章列表 */}
      <div>
        {posts.map((post) => (
          <div key={post.id}>{post.title}</div>
        ))}
      </div>
      {/* 评论列表 */}
      <div>
        {comments.map((comment) => (
          <div key={comment.id}>{comment.text}</div>
        ))}
      </div>
      {/* 通知列表 */}
      <div>
        {notifications.map((n) => (
          <div key={n.id}>{n.message}</div>
        ))}
      </div>
    </div>
  );
}
```

**正例 - 职责分离**:

```typescript
// ✅ 复合组件模式
interface TabsProps {
  children: React.ReactNode
  defaultValue: string
}

export function Tabs({ children, defaultValue }: TabsProps) {
  const [activeTab, setActiveTab] = useState(defaultValue)

  return (
    <TabsContext.Provider value={{ activeTab, setActiveTab }}>
      <div className="tabs">{children}</div>
    </TabsContext.Provider>
  )
}

Tabs.List = function TabsList({ children }: { children: React.ReactNode }) {
  return <div className="tabs-list">{children}</div>
}

Tabs.Trigger = function TabsTrigger({ value, children }: { value: string; children: React.ReactNode }) {
  const { activeTab, setActiveTab } = useTabsContext()
  return (
    <button
      className={activeTab === value ? 'active' : ''}
      onClick={() => setActiveTab(value)}
    >
      {children}
    </button>
  )
}

Tabs.Content = function TabsContent({ value, children }: { value: string; children: React.ReactNode }) {
  const { activeTab } = useTabsContext()
  if (activeTab !== value) return null
  return <div>{children}</div>
}

// 使用
<Tabs defaultValue="tab1">
  <Tabs.List>
    <Tabs.Trigger value="tab1">Tab 1</Tabs.Trigger>
    <Tabs.Trigger value="tab2">Tab 2</Tabs.Trigger>
  </Tabs.List>
  <Tabs.Content value="tab1">Content 1</Tabs.Content>
  <Tabs.Content value="tab2">Content 2</Tabs.Content>
</Tabs>
```

### 3.3 Render Props 模式

**灵活的渲染控制**:

```typescript
// ✅ Render Props 模式
interface DataFetcherProps<T> {
  url: string;
  children: (
    data: T | null,
    loading: boolean,
    error: Error | null
  ) => React.ReactNode;
}

export function DataFetcher<T>({ url, children }: DataFetcherProps<T>) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    fetch(url)
      .then((res) => res.json())
      .then(setData)
      .catch(setError)
      .finally(() => setLoading(false));
  }, [url]);

  return <>{children(data, loading, error)}</>;
}

// 使用
<DataFetcher<User> url="/api/user">
  {(user, loading, error) => {
    if (loading) return <div>加载中...</div>;
    if (error) return <div>错误: {error.message}</div>;
    if (!user) return <div>无数据</div>;
    return <div>{user.name}</div>;
  }}
</DataFetcher>;
```

## 4. Server Components vs Client Components

### 4.1 设计原则

**Server Components 优先**:

```typescript
// ✅ 默认使用 Server Component
// app/users/page.tsx
import { getUsers } from "@/lib/db";
import { UserCard } from "@/components/user-card";

export default async function UsersPage() {
  const users = await getUsers();

  return (
    <div>
      <h1>用户列表</h1>
      {users.map((user) => (
        <UserCard key={user.id} user={user} />
      ))}
    </div>
  );
}
```

**Client Components 按需使用**:

```typescript
// ✅ 需要交互时使用 Client Component
// components/like-button.tsx
"use client";

import { useState } from "react";

export function LikeButton({ postId }: { postId: string }) {
  const [liked, setLiked] = useState(false);
  const [count, setCount] = useState(0);

  const handleLike = async () => {
    setLiked(!liked);
    setCount((prev) => (liked ? prev - 1 : prev + 1));
    await fetch(`/api/posts/${postId}/like`, { method: "POST" });
  };

  return (
    <button onClick={handleLike}>
      {liked ? "已点赞" : "点赞"} ({count})
    </button>
  );
}
```

## 6. 状态管理原则

### 6.1 状态提升

**将状态提升到最近的共同父组件**:

```typescript

  return (
    <div>
      <ChildA />
      <ChildB />
    </div>
  );
}

function ChildA() {
  const [count, setCount] = useState(0);
  return <div>{count}</div>;
}

function ChildB() {
  const [count, setCount] = useState(0);
  return <div>{count}</div>;

// ✅ 状态提升
function Parent() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <ChildA count={count} />
      <ChildB count={count} setCount={setCount} />
    </div>
  );
}

function ChildA({ count }: { count: number }) {
  return <div>{count}</div>;
}

function ChildB({
  count,
  setCount,
}: {
  count: number;
  setCount: (n: number) => void;
}) {
  return <button onClick={() => setCount(count + 1)}>增加</button>;
}
```

## 11. 常见问题

### 11.1 何时使用 Server Component?

**问题**: Server Component 和 Client Component 如何选择

**解答**:

| 场景     | 推荐   | 原因                         |
| :------- | :----- | :--------------------------- |
| 数据获取 | Server | 减少客户端请求               |
| 静态内容 | Server | 更好的性能                   |
| 交互功能 | Client | 需要状态和事件               |
| 表单处理 | 混合   | Server Actions + Client 验证 |

---

### 11.2 如何避免 Props 钻取?

**问题**: Props 层层传递很麻烦

**解答**:

1. 使用 Context API
2. 使用状态管理库(Zustand, Jotai)
3. 组合组件而非嵌套
4. 使用 children props

- 文件超过 200 行

### 11.4 如何设计可复用组件?

**问题**: 组件如何做到真正可复用

**解答**:

```typescript
// ✅ 可复用组件设计
interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: "primary" | "secondary" | "ghost";
  size?: "sm" | "md" | "lg";
  loading?: boolean;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
}

export function Button({
  children,
  variant = "primary",
  size = "md",
  loading = false,
  leftIcon,
  rightIcon,
  disabled,
  ...props
}: ButtonProps) {
  return (
    <button
      {...props}
      disabled={disabled || loading}
      className={`btn btn-${variant} btn-${size}`}
    >
      {loading && <Spinner />}
      {leftIcon && <span>{leftIcon}</span>}
      {children}
      {rightIcon && <span>{rightIcon}</span>}
    </button>
  );
}
```

**关键点**:

### 11.7 组件如何做类型安全?

**问题**: 如何确保组件类型安全

**解答**:

```typescript
// ✅ 完整的类型定义
import type { ReactNode, ButtonHTMLAttributes } from "react";

interface ButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  children: ReactNode;
  variant?: "primary" | "secondary";
  size?: "sm" | "md" | "lg";
  loading?: boolean;
}

export function Button({
  children,
  variant = "primary",
  size = "md",
  loading = false,
  ...props
}: ButtonProps) {
  return (
    <button {...props} disabled={loading || props.disabled}>
      {loading ? "Loading..." : children}
    </button>
  );
}

// 使用时有完整的类型提示
<Button
  variant="primary" // 类型提示
  size="md" // 类型提示
  onClick={(e) => {
    // e 有正确类型
    console.log(e.currentTarget);
  }}
>
  点击
</Button>;
```
