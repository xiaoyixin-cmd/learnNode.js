# 08-10-组件设计模式

组件设计模式是构建可维护、可复用 React 应用的关键。本文介绍 Next.js 中常用的组件设计模式。

## 核心概念

### 组件模式分类

| 模式         | 用途     | 复杂度 | 适用场景   |
| ------------ | -------- | ------ | ---------- |
| 容器组件     | 逻辑处理 | 中     | 数据获取   |
| 展示组件     | UI 渲染  | 低     | 纯展示     |
| 复合组件     | 组件组合 | 高     | 复杂 UI    |
| 高阶组件     | 逻辑复用 | 高     | 横切关注点 |
| Render Props | 渲染逻辑 | 中     | 灵活渲染   |
| Hooks 模式   | 状态逻辑 | 中     | 逻辑复用   |

```typescript
// components/Tabs/Tabs.tsx
"use client";

import { createContext, useContext, useState, ReactNode } from "react";

interface TabsContextValue {
  activeTab: string;
  setActiveTab: (tab: string) => void;
}

const TabsContext = createContext<TabsContextValue | null>(null);

function useTabs() {
  const context = useContext(TabsContext);
  if (!context) throw new Error("useTabs must be used within Tabs");
  return context;
}

interface TabsProps {
  defaultTab: string;
  children: ReactNode;
}

export function Tabs({ defaultTab, children }: TabsProps) {
  const [activeTab, setActiveTab] = useState(defaultTab);

  return (
    <TabsContext.Provider value={{ activeTab, setActiveTab }}>
      <div className="tabs">{children}</div>
    </TabsContext.Provider>
  );
}

interface TabListProps {
  children: ReactNode;
}

Tabs.List = function TabList({ children }: TabListProps) {
  return <div className="tab-list">{children}</div>;
};

interface TabProps {
  value: string;
  children: ReactNode;
}

Tabs.Tab = function Tab({ value, children }: TabProps) {
  const { activeTab, setActiveTab } = useTabs();

  return (
    <button
      className={activeTab === value ? "active" : ""}
      onClick={() => setActiveTab(value)}
    >
      {children}
    </button>
  );
};

interface TabPanelProps {
  value: string;
  children: ReactNode;
}

Tabs.Panel = function TabPanel({ value, children }: TabPanelProps) {
  const { activeTab } = useTabs();

  if (activeTab !== value) return null;

  return <div className="tab-panel">{children}</div>;
};
```

使用示例:

```typescript
// app/page.tsx

        <Tabs.Tab value="tab2">标签2</Tabs.Tab>
      </Tabs.List>
      <Tabs.Panel value="tab1">内容1</Tabs.Panel>
      <Tabs.Panel value="tab2">内容2</Tabs.Panel>
```

```typescript
  toggleTheme: () => void;
}

  if (!context) throw new Error("useTheme must be used within ThemeProvider");
  return context;

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setTheme] = useState<Theme>("light");

  const toggleTheme = () => {
    setTheme((prev) => (prev === "light" ? "dark" : "light"));
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

```typescript
// components/TemperatureCalculator.tsx
"use client";

import { useState } from "react";

function TemperatureInput({
  scale,
  temperature,
  onTemperatureChange,
}: {
  scale: string;
  temperature: string;
  onTemperatureChange: (temp: string) => void;
}) {
  return (
    <fieldset>
      <legend>输入{scale}温度:</legend>
      <input
        value={temperature}
        onChange={(e) => onTemperatureChange(e.target.value)}
      />
    </fieldset>
  );
}

export function TemperatureCalculator() {
  const [temperature, setTemperature] = useState("");
  const [scale, setScale] = useState("c");

  const handleCelsiusChange = (temp: string) => {
    setScale("c");
    setTemperature(temp);
  };

  const handleFahrenheitChange = (temp: string) => {
    setScale("f");
    setTemperature(temp);
  };

  const celsius =
    scale === "f" ? tryConvert(temperature, toCelsius) : temperature;
  const fahrenheit =
    scale === "c" ? tryConvert(temperature, toFahrenheit) : temperature;

  return (
    <div>
      <TemperatureInput
        scale="摄氏度"
        temperature={celsius}
        onTemperatureChange={handleCelsiusChange}
      />
      <TemperatureInput
        scale="华氏度"
        temperature={fahrenheit}
        onTemperatureChange={handleFahrenheitChange}
      />
    </div>
  );
}

function toCelsius(fahrenheit: number) {
  return ((fahrenheit - 32) * 5) / 9;
}

function toFahrenheit(celsius: number) {
  return (celsius * 9) / 5 + 32;
}

function tryConvert(temperature: string, convert: (temp: number) => number) {
  const input = parseFloat(temperature);
  if (Number.isNaN(input)) return "";
  const output = convert(input);
  return Math.round(output * 1000) / 1000 + "";
}
```

```typescript
// hooks/useObserver.ts
import { useEffect } from "react";

type Listener = () => void;

class EventEmitter {
  private listeners: Listener[] = [];

  subscribe(listener: Listener) {
    this.listeners.push(listener);
    return () => {
      this.listeners = this.listeners.filter((l) => l !== listener);
    };
  }

  emit() {
    this.listeners.forEach((listener) => listener());
  }
}

export const emitter = new EventEmitter();

export function useObserver(callback: Listener) {
  useEffect(() => {
    return emitter.subscribe(callback);
  }, [callback]);
}
```

```typescript
interface ValidationHandler {
  setNext(handler: ValidationHandler): ValidationHandler;
  handle(data: any): boolean;
}

class BaseValidator implements ValidationHandler {
  private nextHandler?: ValidationHandler;

  setNext(handler: ValidationHandler): ValidationHandler {
    this.nextHandler = handler;
    return handler;
  }

  handle(data: any): boolean {
    if (this.nextHandler) {
      return this.nextHandler.handle(data);
    }
    return true;
  }
}

class RequiredValidator extends BaseValidator {
  handle(data: any): boolean {
    if (!data) return false;
    return super.handle(data);
  }
}

class LengthValidator extends BaseValidator {
  constructor(private minLength: number) {
    super();
  }

  handle(data: any): boolean {
    if (data.length < this.minLength) return false;
    return super.handle(data);
  }
}
```

```typescript
// utils/iterator.ts
interface Iterator<T> {
  next(): { value: T; done: boolean };
  hasNext(): boolean;
}

class ArrayIterator<T> implements Iterator<T> {
  private index = 0;

  constructor(private items: T[]) {}

  next() {
    if (this.hasNext()) {
      return {
        value: this.items[this.index++],
        done: false,
      };
    }
    return {
      value: null as any,
      done: true,
    };
  }

  hasNext() {
    return this.index < this.items.length;
  }
}
```

## 适用场景

### 模式对比

| 模式         | 适用场景       | 优点       | 缺点         |
| ------------ | -------------- | ---------- | ------------ |
| 容器/展示    | 逻辑与 UI 分离 | 可维护性高 | 组件数量增加 |
| 复合组件     | 复杂 UI 组合   | 灵活性高   | 学习成本高   |
| HOC          | 横切关注点     | 逻辑复用   | Props 冲突   |
| Render Props | 灵活渲染       | 灵活性高   | 嵌套地狱     |
| Hooks        | 状态逻辑       | 简洁易用   | 规则限制     |

## 注意事项

### 1. 选择合适的模式

根据实际需求选择合适的设计模式,避免过度设计。

### 2. 保持组件简单

每个组件应该只做一件事,遵循单一职责原则。

### 3. 避免过度抽象

不要为了使用模式而使用模式,保持代码简洁。

### 4. 考虑性能影响

某些模式可能影响性能,需要权衡利弊。

### 5. 文档化设计决策

记录为什么选择某个模式,方便团队理解。

## 常见问题

### 1. 什么时候使用容器组件?

当需要分离逻辑和 UI 时使用。

### 2. 复合组件的优势是什么?

提供灵活的组件组合方式,API 更清晰。

### 3. HOC 和 Hooks 哪个更好?

Hooks 更现代,推荐优先使用 Hooks。

### 4. 如何避免 Render Props 嵌套?

使用 Hooks 替代 Render Props。

### 5. 什么时候使用 Provider 模式?

需要跨组件共享状态时使用。

### 6. 受控组件和非受控组件的区别?

受控组件状态由 React 管理,非受控组件由 DOM 管理。

### 7. 如何选择组件模式?

根据需求、团队习惯和项目规模选择。

### 8. 组合组件如何实现?

使用静态属性或 Context API。

### 9. 状态提升的时机?

多个组件需要共享状态时。

### 10. 插槽模式的优势?

提供灵活的内容插入点。

### 11. 工厂模式适用场景?

需要根据条件创建不同组件时。

### 12. 观察者模式如何实现?

使用 EventEmitter 或 Context API。

### 13. 策略模式的优势?

易于扩展和维护不同策略。

### 14. 装饰器模式如何使用?

使用 HOC 或组合模式。

### 15. 适配器模式的作用?

适配不同接口,保持兼容性。

### 16. 单例模式的应用?

全局状态管理、日志系统等。

### 17. 命令模式的优势?

支持撤销/重做操作。

### 18. 代理模式的用途?

延迟加载、缓存、权限控制等。

### 19. 责任链模式适用场景?

多步骤验证、中间件等。

### 20. 状态模式的优势?

清晰的状态转换逻辑。

```tsx
export function Modal({
```

```tsx
  error?: Error;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: any) {
    console.error("Error caught by boundary:", error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return (
        this.props.fallback || (
          <div className="p-4 bg-red-50 border border-red-200 rounded">
            <h2 className="text-red-800 font-bold">出错了</h2>
            <p className="text-red-600">{this.state.error?.message}</p>
          </div>
        )
      );
    }

    return this.props.children;
  }
}
```

```tsx
type Theme = "light" | "dark";

  theme: Theme;
  toggleTheme: () => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setTheme] = useState<Theme>("light");

  const toggleTheme = () => {
    setTheme((prev) => (prev === "light" ? "dark" : "light"));
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error("useTheme must be used within ThemeProvider");
  }
  return context;
}
```

### 10. Observer 模式

```tsx
// hooks/useObserver.ts
"use client";

import { useEffect, useState } from "react";

class EventEmitter {
  private events: Map<string, Function[]> = new Map();

  on(event: string, callback: Function) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event)!.push(callback);
  }

  off(event: string, callback: Function) {
    const callbacks = this.events.get(event);
    if (callbacks) {
      const index = callbacks.indexOf(callback);
      if (index > -1) {
        callbacks.splice(index, 1);
      }
    }
  }

  emit(event: string, data?: any) {
    const callbacks = this.events.get(event);
    if (callbacks) {
      callbacks.forEach((callback) => callback(data));
    }
  }
}

export const eventBus = new EventEmitter();

export function useEvent(event: string, callback: Function) {
  useEffect(() => {
    eventBus.on(event, callback);
    return () => eventBus.off(event, callback);
  }, [event, callback]);
}
```
