# 并行路由实现模态对话框

并行路由(Parallel Routes)是 Next.js 16 中的高级路由功能,允许在同一个布局中同时渲染多个页面。这个功能特别适合实现模态对话框、侧边栏、仪表板等复杂的 UI 模式。

## 并行路由基础

### 什么是并行路由

并行路由使用 `@folder` 命名约定创建命名插槽(named slots),可以在同一个布局中同时渲染多个页面:

```
app/
├── layout.tsx
├── page.tsx
├── @modal/
│   └── page.tsx
└── @sidebar/
    └── page.tsx
```

### 基本用法

```typescript
// app/layout.tsx
export default function Layout({
  children,
  modal,
  sidebar,
}: {
  children: React.ReactNode;
  modal: React.ReactNode;
  sidebar: React.ReactNode;
}) {
  return (
    <div>
      <div className="flex">
        <aside>{sidebar}</aside>
        <main>{children}</main>
      </div>
      {modal}
    </div>
  );
}

// app/page.tsx
export default function Page() {
  return <h1>Main Content</h1>;
}

// app/@modal/page.tsx
export default function Modal() {
  return <div>Modal Content</div>;
}

// app/@sidebar/page.tsx
export default function Sidebar() {
  return <div>Sidebar Content</div>;
}
```

## 并行路由与拦截路由

### 拦截路由语法

拦截路由使用特殊的文件夹命名约定:

| 约定       | 说明           |
| ---------- | -------------- |
| `(.)`      | 匹配同级路由   |
| `(..)`     | 匹配上一级路由 |
| `(..)(..)` | 匹配上两级路由 |
| `(...)`    | 匹配根路由     |

### 结合使用

```
app/
├── layout.tsx
├── page.tsx
├── photos/
│   ├── page.tsx
│   └── [id]/
│       └── page.tsx
└── @modal/
    ├── default.tsx
    └── (.)photos/
        └── [id]/
            └── page.tsx
```

```typescript
"use client";

import { useRouter } from "next/navigation";
import { useEffect, useRef } from "react";

export function Modal({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  const dialogRef = useRef<HTMLDialogElement>(null);

  useEffect(() => {
    dialogRef.current?.showModal();
  }, []);

  const closeModal = () => {
    dialogRef.current?.close();
    router.back();
  };

  return (
    <dialog
      ref={dialogRef}
      className="backdrop:bg-black/50 rounded-lg p-0 max-w-5xl"
      onClose={closeModal}
    >
      <div className="relative">
        <button
          onClick={closeModal}
          className="absolute top-4 right-4 text-white bg-black/50 rounded-full w-10 h-10 flex items-center justify-center hover:bg-black/70"

          ✕
        </button>

}
```

### 登录表单组件

```typescript
// components/login-form.tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";

export function LoginForm() {
  const router = useRouter();
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);

    try {
      const res = await fetch("/api/auth/login", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ email, password }),
      });

      if (res.ok) {
        router.push("/dashboard");
        router.refresh();
      } else {
        alert("Login failed");
      }
    } catch (error) {
      alert("An error occurred");
    } finally {
      setLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div>
        <label className="block text-sm font-medium mb-1">Email</label>
        <input
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          className="w-full border rounded px-3 py-2"
          required
        />
      </div>
      <div>
        <label className="block text-sm font-medium mb-1">Password</label>
        <input
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          className="w-full border rounded px-3 py-2"
          required
        />
      </div>
      <button
        type="submit"
        disabled={loading}
        className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700 disabled:opacity-50"
      >
        {loading ? "Logging in..." : "Log In"}
      </button>
    </form>
  );
}
```

```typescript
import { useEffect, useState } from "react";

export default function Notification() {
  const [show, setShow] = useState(false);
  const [message, setMessage] = useState("");

  useEffect(() => {
    // 监听通知事件
    const handleNotification = (e: CustomEvent) => {
      setMessage(e.detail.message);
      setShow(true);
      setTimeout(() => setShow(false), 3000);
    };

    window.addEventListener("notification" as any, handleNotification);
    return () =>
      window.removeEventListener("notification" as any, handleNotification);
  }, []);

  if (!show) return null;

  return (
    <div className="fixed top-4 right-4 bg-green-500 text-white px-6 py-3 rounded shadow-lg">
      {message}
    </div>
  );
}
```

## 适用场景

### 何时使用并行路由

| 场景       | 说明                             | 示例                   |
| ---------- | -------------------------------- | ---------------------- |
| 模态对话框 | 在不离开当前页面的情况下显示内容 | 图片预览、产品快速查看 |
| 侧边栏     | 同时显示主内容和侧边栏           | 仪表板、设置页面       |
| 多面板布局 | 同时显示多个独立内容区域         | 邮件客户端、IDE        |
| 条件渲染   | 根据状态显示不同内容             | 登录/未登录状态        |

### 并行路由 vs 客户端模态框

| 特性       | 并行路由      | 客户端模态框 |
| ---------- | ------------- | ------------ |
| URL 支持   | 有独立 URL    | 无独立 URL   |
| 浏览器历史 | 支持前进/后退 | 不支持       |
| 直接访问   | 可以直接访问  | 不能直接访问 |
| SEO        | 友好          | 不友好       |
| 实现复杂度 | 较高          | 较低         |
| 服务端渲染 | 支持          | 不支持       |

## 常见问题

### 1. 为什么需要 default.tsx?

当并行路由没有匹配的页面时,Next.js 会渲染 `default.tsx`。如果不提供,刷新页面时会出现 404 错误。

```typescript
// app/@modal/default.tsx
export default function Default() {
  return null; // 不显示任何内容
}
```

### 2. 如何在模态框中获取数据?

模态框页面可以像普通页面一样获取数据:

```typescript
// app/@modal/(.)products/[id]/page.tsx
export default async function ProductModal({
  params,
}: {
  params: { id: string };
}) {
  const product = await getProduct(params.id);

  return <Modal>{/* 使用 product 数据 */}</Modal>;
}
```

### 3. 如何处理模态框的元数据?

模态框页面可以有自己的元数据:

```typescript
// app/@modal/(.)products/[id]/page.tsx
export async function generateMetadata({ params }: { params: { id: string } }) {
  const product = await getProduct(params.id);

  return {
    title: `${product.name} - Quick View`,
  };
}
```

### 4. 如何在模态框之间导航?

使用 `Link` 或 `router.push`:

```typescript
// 从一个模态框导航到另一个模态框
<Link href="/products/2">Next Product</Link>
```

## 总结

并行路由是 Next.js 16 中实现复杂 UI 模式的强大工具。通过本文的学习,我们掌握了:

### 核心概念

1. **并行路由**: 使用 `@folder` 创建命名插槽
2. **拦截路由**: 使用 `(.)` 等约定拦截导航
3. **默认插槽**: 使用 `default.tsx` 处理未匹配的路由
4. **客户端导航**: 拦截路由只在客户端导航时生效

### 实战应用

我们通过四个实战场景学习了并行路由的应用:

1. **图片画廊**: 点击图片显示模态框,直接访问显示完整页面
2. **产品快速查看**: 在列表页快速查看产品详情
3. **登录模态框**: 在任何页面弹出登录框
4. **多模态框管理**: 同时管理多个并行路由插槽

### 最佳实践

1. **始终提供 default.tsx**: 避免刷新页面时出错
2. **使用 router.back()**: 关闭模态框时返回上一页
3. **合理使用拦截路由**: 只在需要模态框体验时使用
4. **提供完整页面**: 确保直接访问 URL 时有完整体验
5. **优化性能**: 使用 Suspense 和流式渲染

### 注意事项

1. 拦截路由只在客户端导航时生效
2. 必须提供 default.tsx
3. 模态框关闭使用 router.back()
4. 可以嵌套并行路由
5. 支持加载状态和错误处理

通过合理使用并行路由和拦截路由,可以创建流畅的用户体验,同时保持良好的 SEO 和可访问性。

```typescript
export function Modal({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  const searchParams = useSearchParams();

  useEffect(() => {

    params.delete("modal");

    router.back();
  };

      {children}
    </dialog>
  );
}
```

```typescript
export function Modal({ children }: { children: React.ReactNode }) {

        ${isClosing ? "opacity-0 scale-95" : "opacity-100 scale-100"}
      `}

  );
}
```

### 5. 模态框键盘导航

```typescript
// components/modal.tsx
"use client";

import { useRouter } from "next/navigation";
import { useEffect, useRef } from "react";

export function Modal({
  children,
  onNext,
  onPrev,
}: {
  children: React.ReactNode;
  onNext?: () => void;
  onPrev?: () => void;
}) {
  const router = useRouter();
  const dialogRef = useRef<HTMLDialogElement>(null);

  useEffect(() => {
    dialogRef.current?.showModal();

    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === "Escape") {
        closeModal();
      } else if (e.key === "ArrowRight" && onNext) {
        onNext();
      } else if (e.key === "ArrowLeft" && onPrev) {
        onPrev();
      }
    };

    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [onNext, onPrev]);

  const closeModal = () => {
    dialogRef.current?.close();
    router.back();
  };

  return (
    <dialog ref={dialogRef} onClose={closeModal}>
      {children}
    </dialog>
  );
}

// 使用示例
// app/@modal/(.)photos/[id]/page.tsx
export default async function PhotoModal({
  params,
}: {
  params: { id: string };
}) {
  const photo = await getPhoto(params.id);
  const photos = await getPhotos();
  const currentIndex = photos.findIndex((p) => p.id === params.id);

  return (
    <Modal
      onNext={() => {
        if (currentIndex < photos.length - 1) {
          router.push(`/photos/${photos[currentIndex + 1].id}`);
        }
      }}
      onPrev={() => {
        if (currentIndex > 0) {
          router.push(`/photos/${photos[currentIndex - 1].id}`);
        }
      }}
    >
      <img src={photo.url} alt={photo.title} />
    </Modal>
  );
}
```
