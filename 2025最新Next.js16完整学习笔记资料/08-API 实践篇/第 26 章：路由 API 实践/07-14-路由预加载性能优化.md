# 路由预加载性能优化

路由预加载是提升 Next.js 应用性能的关键技术之一。通过预加载路由,可以在用户点击链接之前就开始加载页面资源,从而实现近乎即时的页面切换。本文将详细介绍 Next.js 16 中的路由预加载机制和优化策略。

## 预加载机制

### Link 组件的预加载

```typescript
// app/page.tsx
import Link from "next/link";

export default function HomePage() {
  return (
    <div>
      {/* 默认启用预加载 */}
      <Link href="/about">About</Link>

      {/* 禁用预加载 */}
      <Link href="/contact" prefetch={false}>
        Contact
      </Link>

      {/* 强制预加载 */}

        Products
      </Link>

```

### 预加载行为

| 渲染方式   | 默认预加载行为 | 说明                   |
| ---------- | -------------- | ---------------------- |
| 静态路由   | 完整预加载     | 预加载整个页面         |
| 动态路由   | 部分预加载     | 只预加载共享布局       |
| 服务端组件 | 30 秒缓存      | 预加载的数据缓存 30 秒 |
| 客户端组件 | 不缓存         | 每次都重新获取         |

## 实战场景一: 智能预加载

### 基于可见性的预加载

```typescript
// components/smart-link.tsx
"use client";

import Link from "next/link";
import { useEffect, useRef, useState } from "react";
import { useRouter } from "next/navigation";

interface SmartLinkProps {
  href: string;
  children: React.ReactNode;
  className?: string;
}

export function SmartLink({ href, children, className }: SmartLinkProps) {
  const router = useRouter();
  const linkRef = useRef<HTMLAnchorElement>(null);
  const [shouldPrefetch, setShouldPrefetch] = useState(false);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setShouldPrefetch(true);
          // 预加载路由
          router.prefetch(href);
          observer.disconnect();
        }
      },
      { threshold: 0.1 }
    );

    if (linkRef.current) {
      observer.observe(linkRef.current);
    }

    return () => observer.disconnect();
  }, [href, router]);

  return (
    <Link
      ref={linkRef}
      href={href}
      prefetch={shouldPrefetch}
      className={className}
    >
      {children}
    </Link>
  );
}
```

```typescript
// components/intent-link.tsx
"use client";

import Link from "next/link";
import { useRouter } from "next/navigation";
import { useRef, useState } from "react";

interface IntentLinkProps {
  href: string;
  children: React.ReactNode;
  className?: string;
}

export function IntentLink({ href, children, className }: IntentLinkProps) {
  const router = useRouter();
  const [isPrefetched, setIsPrefetched] = useState(false);
  const timeoutRef = useRef<NodeJS.Timeout>();

  const handleMouseEnter = () => {
    // 延迟预加载,避免误触
    timeoutRef.current = setTimeout(() => {
      if (!isPrefetched) {
        router.prefetch(href);
        setIsPrefetched(true);
      }
    }, 200); // 200ms 延迟
  };

  const handleMouseLeave = () => {
    // 取消预加载
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }
  };

  return (
    <Link
      href={href}
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
      prefetch={false}
      className={className}
    >
      {children}
    </Link>
  );
}
```

```typescript
import { useEffect } from "use";
import { useRouter } from "next/navigation";

export function usePrefetchOnIdle(routes: string[]) {
  const router = useRouter();

  useEffect(() => {
    // 使用 requestIdleCallback 在浏览器空闲时预加载
    if ("requestIdleCallback" in window) {
      const idleCallback = requestIdleCallback(() => {
        routes.forEach((route) => {
          router.prefetch(route);
        });
      });

      return () => cancelIdleCallback(idleCallback);
    } else {
      // 降级方案:使用 setTimeout
      const timeout = setTimeout(() => {
        routes.forEach((route) => {
          router.prefetch(route);
        });
      }, 1000);

      return () => clearTimeout(timeout);
    }
  }, [routes, router]);
}
```

### 基于网络状态的预加载

```typescript
// hooks/use-adaptive-prefetch.ts
"use client";

import { useEffect, useState } from "react";
import { useRouter } from "next/navigation";

export function useAdaptivePrefetch(routes: string[]) {
  const router = useRouter();
  const [shouldPrefetch, setShouldPrefetch] = useState(true);

  useEffect(() => {
    // 检查网络状态
    if ("connection" in navigator) {
      const connection = (navigator as any).connection;

      // 慢速网络或省流量模式下禁用预加载
      if (
        connection.saveData ||
        connection.effectiveType === "slow-2g" ||
        connection.effectiveType === "2g"
      ) {
        setShouldPrefetch(false);
        return;
      }
    }

    // 检查设备内存
    if ("deviceMemory" in navigator) {
      const memory = (navigator as any).deviceMemory;

      // 低内存设备禁用预加载
      if (memory < 4) {
        setShouldPrefetch(false);
        return;
      }
    }

    // 预加载路由
    if (shouldPrefetch) {
      routes.forEach((route) => {
        router.prefetch(route);
      });
    }
  }, [routes, router, shouldPrefetch]);

  return shouldPrefetch;
}
```

### 批量预加载管理

```typescript
// lib/prefetch-manager.ts
"use client";

class PrefetchManager {
  private prefetchedRoutes = new Set<string>();
  private prefetchQueue: string[] = [];
  private isProcessing = false;

  constructor(private router: any) {}

  // 添加到预加载队列
  add(route: string, priority: "high" | "low" = "low") {
    if (this.prefetchedRoutes.has(route)) {
      return;
    }

    if (priority === "high") {
      this.prefetchQueue.unshift(route);
    } else {
      this.prefetchQueue.push(route);
    }

    this.process();
  }

  // 处理预加载队列
  private async process() {
    if (this.isProcessing || this.prefetchQueue.length === 0) {
      return;
    }

    this.isProcessing = true;

    while (this.prefetchQueue.length > 0) {
      const route = this.prefetchQueue.shift();

      if (route && !this.prefetchedRoutes.has(route)) {
        await this.router.prefetch(route);
        this.prefetchedRoutes.add(route);

        // 避免阻塞主线程
        await new Promise((resolve) => setTimeout(resolve, 100));
      }
    }

    this.isProcessing = false;
  }

  // 清除预加载记录
  clear() {
    this.prefetchedRoutes.clear();
    this.prefetchQueue = [];
  }
}

// 使用示例
import { useRouter } from "next/navigation";
import { useEffect, useMemo } from "react";

export function usePrefetchManager() {
  const router = useRouter();
  const manager = useMemo(() => new PrefetchManager(router), [router]);

  return manager;
}
```

## 性能监控

### 预加载性能追踪

```typescript
// lib/prefetch-tracker.ts
"use client";

interface PrefetchMetrics {
  route: string;
  startTime: number;
  endTime?: number;
  duration?: number;
  success: boolean;
}

class PrefetchTracker {
  private metrics: PrefetchMetrics[] = [];

  start(route: string) {
    this.metrics.push({
      route,
      startTime: performance.now(),
      success: false,
    });
  }

  end(route: string, success: boolean = true) {
    const metric = this.metrics.find((m) => m.route === route && !m.endTime);

    if (metric) {
      metric.endTime = performance.now();
      metric.duration = metric.endTime - metric.startTime;
      metric.success = success;
    }
  }

  getMetrics() {
    return this.metrics;
  }

  getAverageDuration() {
    const successfulMetrics = this.metrics.filter(
      (m) => m.success && m.duration
    );

    if (successfulMetrics.length === 0) {
      return 0;
    }

    const total = successfulMetrics.reduce(
      (sum, m) => sum + (m.duration || 0),
      0
    );
    return total / successfulMetrics.length;
  }

  clear() {
    this.metrics = [];
  }
}

export const prefetchTracker = new PrefetchTracker();

// 使用示例
import { useRouter } from "next/navigation";
import { prefetchTracker } from "@/lib/prefetch-tracker";

export function TrackedLink({
  href,
  children,
}: {
  href: string;
  children: React.ReactNode;
}) {
  const router = useRouter();

  const handlePrefetch = async () => {
    prefetchTracker.start(href);

    try {
      await router.prefetch(href);
      prefetchTracker.end(href, true);
    } catch (error) {
      prefetchTracker.end(href, false);
      console.error("Prefetch failed:", error);
    }
  };

  return (
    <Link href={href} onMouseEnter={handlePrefetch} prefetch={false}>
      {children}
    </Link>
  );
}
```

```typescript
"use client";

import { useEffect, useState } from "react";

interface ResourceTiming {
  name: string;
  duration: number;
  size: number;
}

export function usePrefetchMonitor() {
  const [resources, setResources] = useState<ResourceTiming[]>([]);

  useEffect(() => {
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();

      const resourceTimings = entries
        .filter((entry) => entry.entryType === "resource")
        .map((entry) => {
          const resourceEntry = entry as PerformanceResourceTiming;

          return {
            name: resourceEntry.name,
            duration: resourceEntry.duration,
            size: resourceEntry.transferSize || 0,
          };
        });

      setResources((prev) => [...prev, ...resourceTimings]);
    });

    observer.observe({ entryTypes: ["resource"] });

    return () => observer.disconnect();
  }, []);

  return resources;
}

// 使用示例
export function PrefetchMonitor() {
  const resources = usePrefetchMonitor();

  const prefetchedResources = resources.filter(
    (r) => r.name.includes("_next/data") || r.name.includes("_next/static")
  );

  const totalSize = prefetchedResources.reduce((sum, r) => sum + r.size, 0);
  const averageDuration =
    prefetchedResources.length > 0
      ? prefetchedResources.reduce((sum, r) => sum + r.duration, 0) /
        prefetchedResources.length
      : 0;

    <div>
      <h3>Prefetch Statistics</h3>
      <p>Total Resources: {prefetchedResources.length}</p>
      <p>Total Size: {(totalSize / 1024).toFixed(2)} KB</p>
```

```typescript
// lib/cache-manager.ts
"use client";

interface CacheEntry {
  data: any;
  timestamp: number;
  ttl: number;
}

class CacheManager {
  private cache = new Map<string, CacheEntry>();

  set(key: string, data: any, ttl: number = 30000) {
    this.cache.set(key, {
      data,
      timestamp: Date.now(),
      ttl,
    });
  }

  get(key: string): any | null {
    const entry = this.cache.get(key);

    if (!entry) {
      return null;
    }

    // 检查是否过期
    if (Date.now() - entry.timestamp > entry.ttl) {
      this.cache.delete(key);
      return null;
    }

    return entry.data;
  }

  has(key: string): boolean {
    return this.get(key) !== null;
  }

  clear() {
    this.cache.clear();
  }

  // 清理过期缓存
  cleanup() {
    const now = Date.now();

    for (const [key, entry] of this.cache.entries()) {
      if (now - entry.timestamp > entry.ttl) {
        this.cache.delete(key);
      }
    }
  }
}

export const cacheManager = new CacheManager();

// 定期清理过期缓存
if (typeof window !== "undefined") {
  setInterval(() => {
    cacheManager.cleanup();
  }, 60000); // 每分钟清理一次
}
```

## 适用场景

### 预加载策略选择

| 场景     | 推荐策略         | 说明             |
| -------- | ---------------- | ---------------- |
| 导航菜单 | 鼠标悬停预加载   | 用户意图明确     |
| 相关内容 | 可见性预加载     | 避免过度预加载   |
| 分页内容 | 自动预加载前后页 | 提升浏览体验     |
| 搜索结果 | 禁用预加载       | 内容动态变化     |
| 低端设备 | 自适应预加载     | 根据设备性能调整 |

### 预加载时机选择

| 时机       | 适用场景 | 优点           | 缺点             |
| ---------- | -------- | -------------- | ---------------- |
| 页面加载时 | 关键路由 | 最快的响应速度 | 增加初始加载时间 |
| 鼠标悬停时 | 导航链接 | 平衡性能和体验 | 可能误触         |
| 可见时     | 列表项   | 精准预加载     | 实现复杂         |
| 空闲时     | 次要路由 | 不影响主流程   | 延迟较高         |

## 注意事项

### 1. 避免过度预加载

过度预加载会浪费带宽和资源,特别是在移动设备上:


### 2. 考虑网络条件

```typescript
// hooks/use-network-aware-prefetch.ts
"use client";

import { useEffect, useState } from "react";

export function useNetworkAwarePrefetch() {
  const [shouldPrefetch, setShouldPrefetch] = useState(true);

  useEffect(() => {
    if ("connection" in navigator) {
      const connection = (navigator as any).connection;

      const updatePrefetchStatus = () => {
        const isSlow =
          connection.effectiveType === "slow-2g" ||
          connection.effectiveType === "2g";
        const isSaveData = connection.saveData;

        setShouldPrefetch(!isSlow && !isSaveData);
      };

      updatePrefetchStatus();
      connection.addEventListener("change", updatePrefetchStatus);

      return () =>
        connection.removeEventListener("change", updatePrefetchStatus);
    }
  }, []);

  return shouldPrefetch;
}
```

```typescript
    if (prefetchedRoutes.current.has(route)) {
      return;
    }

    router.prefetch(route);
    prefetchedRoutes.current.add(route);
  };

  return { prefetch };
}
```

在组件卸载时应该清理预加载资源:

## 总结

路由预加载是提升 Next.js 应用性能的重要手段。通过本文的学习,我们掌握了:

### 核心技术

1. **Link 组件预加载**: 使用 `prefetch` 属性控制预加载行为
2. **router.prefetch()**: 编程式预加载路由
3. **智能预加载**: 基于可见性、鼠标悬停、用户意图的预加载
4. **批量预加载**: 预加载关键路由和相关内容
5. **性能监控**: 追踪预加载性能和资源使用

### 最佳实践

1. **避免过度预加载**: 只预加载用户可能访问的路由
2. **考虑网络条件**: 在慢速网络下禁用预加载
3. **处理预加载失败**: 妥善处理预加载错误
4. **避免重复预加载**: 使用缓存避免重复请求
5. **清理预加载资源**: 在组件卸载时清理资源

### 优化策略

1. **优先级预加载**: 根据重要性设置预加载优先级
2. **条件预加载**: 根据用户状态决定是否预加载
3. **自适应预加载**: 根据设备性能和网络状况调整策略
4. **缓存管理**: 合理配置预加载缓存时间
5. **性能监控**: 持续监控预加载效果

通过合理应用这些预加载技术,可以显著提升应用的响应速度和用户体验。

```typescript
        if (nextChapter.pages.length > 0) {
          router.prefetch(`/docs/${nextChapter.pages[0].slug}`);

      }
    }
```

### 3. 性能分析

```typescript
// hooks/use-prefetch-performance.ts
"use client";

import { useEffect, useState } from "react";

interface PerformanceMetrics {
  totalPrefetches: number;
  successfulPrefetches: number;
  failedPrefetches: number;
  averageDuration: number;
  totalSize: number;
}

export function usePrefetchPerformance() {
  const [metrics, setMetrics] = useState<PerformanceMetrics>({
    totalPrefetches: 0,
    successfulPrefetches: 0,
    failedPrefetches: 0,
    averageDuration: 0,
    totalSize: 0,
  });

  useEffect(() => {
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();

      const prefetchEntries = entries.filter(
        (entry) =>
          entry.name.includes("_next/data") ||
          entry.name.includes("_next/static")
      );

      if (prefetchEntries.length > 0) {
        setMetrics((prev) => {
          const totalDuration = prefetchEntries.reduce(
            (sum, entry) => sum + entry.duration,
            0
          );

          const totalSize = prefetchEntries.reduce(
            (sum, entry) => sum + ((entry as any).transferSize || 0),
            0
          );

          return {
            totalPrefetches: prev.totalPrefetches + prefetchEntries.length,
            successfulPrefetches:
              prev.successfulPrefetches + prefetchEntries.length,
            failedPrefetches: prev.failedPrefetches,
            averageDuration: totalDuration / prefetchEntries.length,
            totalSize: prev.totalSize + totalSize,
          };
        });
      }
    });

    observer.observe({ entryTypes: ["resource"] });

    return () => observer.disconnect();
  }, []);

  return metrics;
}
```
