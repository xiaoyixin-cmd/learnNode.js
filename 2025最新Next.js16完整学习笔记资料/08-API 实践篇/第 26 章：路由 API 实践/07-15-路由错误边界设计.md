# 路由错误边界设计

错误边界是 Next.js 应用中处理错误的重要机制。通过合理设计错误边界,可以提升应用的健壮性和用户体验。本文将详细介绍 Next.js 16 中的错误边界设计模式和最佳实践。

```typescript
("use client");

import { NetworkError, TimeoutError, AuthError } from "@/lib/errors";

export default function Error({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  if (error instanceof NetworkError) {
    return (
      <div>
        <h2>Network Error</h2>
        <p>Please check your internet connection.</p>
        <button onClick={reset}>Retry</button>
      </div>
    );

    return (
      <div>
        <h2>Request Timeout</h2>
        <p>The request took too long to complete.</p>
        <button onClick={reset}>Retry</button>
      </div>
    );

    return (
      <div>
        <h2>Authentication Error</h2>
        <p>Please log in to continue.</p>
        <Link href="/login">Go to login</Link>
      </div>
    );
  }

  return (
    <div>
      <h2>Something went wrong</h2>
      <button onClick={reset}>Try again</button>
    </div>
  );
}
```

### 自定义错误追踪

```typescript
// lib/error-tracker.ts
interface ErrorLog {
  message: string;
  stack?: string;
  timestamp: number;
  userAgent: string;
  url: string;
  digest?: string;
}

class ErrorTracker {
  private errors: ErrorLog[] = [];

  track(error: Error & { digest?: string }) {
    const errorLog: ErrorLog = {
      message: error.message,
      stack: error.stack,
      timestamp: Date.now(),
      userAgent: navigator.userAgent,
      url: window.location.href,
      digest: error.digest,
    };

    this.errors.push(errorLog);

    // 发送到服务器
    this.sendToServer(errorLog);
  }

  private async sendToServer(errorLog: ErrorLog) {
    try {
      await fetch("/api/errors", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(errorLog),
      });
    } catch (error) {
      console.error("Failed to send error log:", error);
    }
  }

  getErrors() {
    return this.errors;
  }

  clear() {
    this.errors = [];
  }
}

export const errorTracker = new ErrorTracker();

// app/error.tsx
("use client");

import { useEffect } from "react";
import { errorTracker } from "@/lib/error-tracker";

export default function Error({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  useEffect(() => {
    errorTracker.track(error);
  }, [error]);

  return (
    <div>
      <h2>Something went wrong</h2>
      <button onClick={reset}>Try again</button>
    </div>
  );
}
```

### 错误统计分析

```typescript
// lib/error-analytics.ts
interface ErrorStats {
  totalErrors: number;
  errorsByType: Record<string, number>;
  errorsByPage: Record<string, number>;
  recentErrors: Array<{
    message: string;
    timestamp: number;
    url: string;
  }>;
}

class ErrorAnalytics {
  private stats: ErrorStats = {
    totalErrors: 0,
    errorsByType: {},
    errorsByPage: {},
    recentErrors: [],
  };

  track(error: Error, url: string) {
    this.stats.totalErrors++;

    // 按类型统计
    const errorType = error.name || "Unknown";
    this.stats.errorsByType[errorType] =
      (this.stats.errorsByType[errorType] || 0) + 1;

    // 按页面统计
    this.stats.errorsByPage[url] = (this.stats.errorsByPage[url] || 0) + 1;

    // 记录最近错误
    this.stats.recentErrors.unshift({
      message: error.message,
      timestamp: Date.now(),
      url,
    });

    // 只保留最近100条
    if (this.stats.recentErrors.length > 100) {
      this.stats.recentErrors = this.stats.recentErrors.slice(0, 100);
    }
  }

  getStats() {
    return this.stats;
  }

  getMostCommonErrors(limit: number = 5) {
    return Object.entries(this.stats.errorsByType)
      .sort(([, a], [, b]) => b - a)
      .slice(0, limit);
  }

  getMostProblematicPages(limit: number = 5) {
    return Object.entries(this.stats.errorsByPage)
      .sort(([, a], [, b]) => b - a)
      .slice(0, limit);
  }
}

export const errorAnalytics = new ErrorAnalytics();
```

## 实战场景四: 错误恢复策略

### 自动重试机制

```typescript
// hooks/use-auto-retry.ts
"use client";

import { useState, useEffect } from "react";

export function useAutoRetry(
  error: Error,
  reset: () => void,
  maxRetries: number = 3,
  retryDelay: number = 1000
) {
  const [retryCount, setRetryCount] = useState(0);

  useEffect(() => {
    if (retryCount < maxRetries) {
      const timeout = setTimeout(() => {
        console.log(`Auto retry ${retryCount + 1}/${maxRetries}`);
        setRetryCount((prev) => prev + 1);
        reset();
      }, retryDelay * (retryCount + 1));

      return () => clearTimeout(timeout);
    }
  }, [error, retryCount, maxRetries, retryDelay, reset]);

  return { retryCount, maxRetries };
}

// app/error.tsx
("use client");

import { useAutoRetry } from "@/hooks/use-auto-retry";

export default function Error({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  const { retryCount, maxRetries } = useAutoRetry(error, reset, 3, 2000);

  return (
    <div>
      <h2>Something went wrong</h2>
      <p>{error.message}</p>

      {retryCount < maxRetries ? (
        <p>
          Auto retrying... ({retryCount}/{maxRetries})
        </p>
      ) : (
        <button onClick={reset}>Manual Retry</button>
      )}
    </div>
  );
}
```

### 部分错误处理

```typescript
// components/error-boundary.tsx
"use client";

import { Component, ReactNode } from "react";

interface Props {
  children: ReactNode;
  fallback: ReactNode;
}

interface State {
  hasError: boolean;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError() {
    return { hasError: true };
  }

  componentDidCatch(error: Error, errorInfo: any) {
    console.error("Error caught by boundary:", error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback;
    }

    return this.props.children;
  }
}

// 使用示例
export default function Page() {
  return (
    <div>
      <h1>My Page</h1>

      <ErrorBoundary fallback={<div>Failed to load sidebar</div>}>
        <Sidebar />
      </ErrorBoundary>

      <ErrorBoundary fallback={<div>Failed to load content</div>}>
        <Content />
      </ErrorBoundary>
    </div>
  );
}
```

### 错误边界层级选择

| 场景     | 推荐层级              | 说明                 |
| -------- | --------------------- | -------------------- |
| 全局错误 | app/error.tsx         | 捕获所有未处理的错误 |
| 路由错误 | app/[route]/error.tsx | 特定路由的错误处理   |
| 组件错误 | ErrorBoundary 组件    | 局部组件错误隔离     |
| API 错误 | try-catch             | 精确控制错误处理     |

### 错误类型处理策略

| 错误类型 | 处理策略       | 用户提示     |
| -------- | -------------- | ------------ |
| 网络错误 | 自动重试       | 检查网络连接 |
| 超时错误 | 延迟重试       | 请稍后重试   |
| 404 错误 | 显示 not-found | 资源不存在   |

## 注意事项

### 1. 错误边界只能捕获子组件错误

错误边界无法捕获自身的错误:

```typescript
// 错误:无法捕获自身错误
export default function Error() {
  throw new Error("This will not be caught");

  return <div>Error</div>;
}

// 正确:在子组件中抛出错误
export default function Page() {
  return (
    <ErrorBoundary>
      <ComponentThatMightThrow />
    </ErrorBoundary>
  );
}
```

### 2. 服务端和客户端错误处理不同

```typescript
// 服务端错误:在 page.tsx 中处理
export default async function Page() {
  try {
    const data = await fetchData();
    return <div>{data}</div>;

    // 服务端错误会被 error.tsx 捕获

// 客户端错误:在客户端组件中处理
("use client");

export default function ClientComponent() {

  useEffect(() => {
    fetchData().catch(setError);
  }, []);

  if (error) {
    return <div>Error: {error.message}</div>;

  return <div>Content</div>;
```

### 3. 避免在错误边界中抛出新错误

```typescript
// 不好的做法
export default function Error({ error }: { error: Error }) {
  // 这会导致无限循环
  throw new Error("Error in error boundary");
}

// 好的做法
export default function Error({ error }: { error: Error }) {
  try {
    // 安全地处理错误
    logError(error);
  } catch (loggingError) {
    console.error("Failed to log error:", loggingError);
  }

  return <div>Error occurred</div>;
}
```

### 4. 合理使用 reset 函数

```typescript
// 不好的做法:无限重试
export default function Error({ reset }: { reset: () => void }) {
  useEffect(() => {
    reset(); // 会导致无限循环
  }, [reset]);

  return <div>Error</div>;
}

// 好的做法:有条件地重试
export default function Error({ reset }: { reset: () => void }) {
  const [retryCount, setRetryCount] = useState(0);

  const handleRetry = () => {
    if (retryCount < 3) {
      setRetryCount((prev) => prev + 1);
      reset();
    }
  };

  return (
    <div>
      <button onClick={handleRetry} disabled={retryCount >= 3}>
        Retry ({retryCount}/3)
      </button>
    </div>
  );
}
```

## 常见问题

### 1. error.tsx 和 global-error.tsx 有什么区别?

`error.tsx` 捕获当前路由段及其子路由的错误,而 `global-error.tsx` 捕获根布局中的错误:

```typescript
// app/error.tsx - 捕获大部分错误
"use client";

export default function Error({
  error,
  reset,
}: {
  error: Error;
```

### 6. 如何在错误边界中使用状态管理?

可以使用 Context 或状态管理库:

```typescript
// context/error-context.tsx
"use client";

import { createContext, useContext, useState, ReactNode } from "react";

interface ErrorContextType {
  errors: Error[];
  addError: (error: Error) => void;
  clearErrors: () => void;
}

const ErrorContext = createContext<ErrorContextType | undefined>(undefined);

export function ErrorProvider({ children }: { children: ReactNode }) {
  const [errors, setErrors] = useState<Error[]>([]);

  const addError = (error: Error) => {
    setErrors((prev) => [...prev, error]);
  };

  const clearErrors = () => {
    setErrors([]);
  };

  return (
    <ErrorContext.Provider value={{ errors, addError, clearErrors }}>
      {children}
    </ErrorContext.Provider>
  );
}

export function useErrors() {
  const context = useContext(ErrorContext);
  if (!context) {
    throw new Error("useErrors must be used within ErrorProvider");
  }
  return context;
}

// app/error.tsx
("use client");

import { useErrors } from "@/context/error-context";

export default function Error({
  error,
  reset,
}: {
  error: Error;
  reset: () => void;
}) {
  const { addError } = useErrors();

  useEffect(() => {
    addError(error);
  }, [error, addError]);

  return <div>Error occurred</div>;
}
```

### 7. 如何处理多个错误?

使用错误队列:

```typescript
"use client";

import { useState } from "react";

export default function MultiErrorHandler() {
  const [errors, setErrors] = useState<Error[]>([]);

  const addError = (error: Error) => {
    setErrors((prev) => [...prev, error]);
  };

  const removeError = (index: number) => {
    setErrors((prev) => prev.filter((_, i) => i !== index));
  };

  return (
    <div>
      {errors.map((error, index) => (
        <div key={index} className="error-toast">
          <p>{error.message}</p>
          <button onClick={() => removeError(index)}>Dismiss</button>
        </div>
      ))}
    </div>
  );
}
```

## 总结

错误边界是 Next.js 应用中处理错误的核心机制。通过本文的学习,我们掌握了:

### 核心概念

1. **error.tsx**: 捕获路由段错误
2. **global-error.tsx**: 捕获根布局错误
3. **not-found.tsx**: 处理 404 错误
4. **reset 函数**: 重置错误状态
5. **错误层级**: 多层错误边界设计

### 最佳实践

1. **分层处理**: 在不同层级设置错误边界
2. **用户友好**: 提供清晰的错误提示和恢复建议
3. **错误追踪**: 集成错误监控服务
4. **自动恢复**: 实现自动重试和降级策略
5. **保护隐私**: 在生产环境隐藏敏感信息

### 实战技巧

1. **自定义错误类型**: 根据错误类型提供不同的处理策略
2. **错误本地化**: 支持多语言错误提示
3. **错误统计**: 分析错误模式和频率
4. **部分错误处理**: 使用 ErrorBoundary 组件隔离错误
5. **错误恢复**: 实现缓存降级和自动重试

通过合理设计错误边界,可以显著提升应用的健壮性和用户体验。
