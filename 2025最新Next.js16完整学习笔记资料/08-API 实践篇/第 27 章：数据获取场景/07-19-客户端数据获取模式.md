# 客户端数据获取模式

在 Next.js 中,客户端数据获取主要用于需要实时更新、用户交互或个性化内容的场景。本文将介绍各种客户端数据获取模式及其最佳实践。

## 基础概念

### 客户端数据获取特点

1. **实时性**: 数据在客户端实时获取
2. **交互性**: 响应用户操作
3. **个性化**: 基于用户状态
4. **SEO 影响**: 不利于 SEO

### 数据获取方式对比

| 方式                 | 使用场景           | 优点               | 缺点             |
| -------------------- | ------------------ | ------------------ | ---------------- |
| useEffect + fetch    | 简单数据获取       | 简单直接           | 需要手动处理状态 |
| SWR                  | 需要缓存和重新验证 | 自动缓存、重新验证 | 额外依赖         |
| React Query          | 复杂数据管理       | 功能强大           | 学习成本高       |
| useState + useEffect | 本地状态管理       | 灵活               | 代码冗余         |

## 实战场景一: 基础数据获取

### 使用 useEffect

```typescript
"use client";

import { useState, useEffect } from "react";

interface User {
  id: number;
  name: string;
  email: string;
}

export default function UserProfile({ userId }: { userId: number }) {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchUser() {
      try {
        setLoading(true);
        const res = await fetch(`/api/users/${userId}`);

        if (!res.ok) {
          throw new Error("Failed to fetch user");
        }

        const data = await res.json();
        setUser(data);
      } catch (err) {
        setError(err instanceof Error ? err.message : "Unknown error");
      } finally {
        setLoading(false);
      }
    }

    fetchUser();
  }, [userId]);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!user) return <div>No user found</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}
```

## 实战场景二: 数据变更

### 使用 SWR 变更

```typescript
"use client";

import useSWR, { mutate } from "swr";

const fetcher = (url: string) => fetch(url).then((res) => res.json());

export default function TodoList() {
  const { data: todos } = useSWR("/api/todos", fetcher);

  async function addTodo(title: string) {
    // 乐观更新
    mutate(
      "/api/todos",
      [...(todos || []), { id: Date.now(), title, completed: false }],
      false
    );

    // 发送请求
    await fetch("/api/todos", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ title }),
    });

    // 重新验证
    mutate("/api/todos");
  }

  return (
    <div>
      <ul>
        {todos?.map((todo: any) => (
          <li key={todo.id}>{todo.title}</li>
        ))}
      </ul>
      <button onClick={() => addTodo("New Todo")}>Add</button>
    </div>
  );
}
```

### 使用 React Query 变更

```typescript
"use client";

import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";

export default function TodoList() {
  const queryClient = useQueryClient();

  const { data: todos } = useQuery({
    queryKey: ["todos"],
    queryFn: () => fetch("/api/todos").then((res) => res.json()),
  });

  const mutation = useMutation({
    mutationFn: (title: string) =>
      fetch("/api/todos", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ title }),
      }).then((res) => res.json()),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["todos"] });
    },
  });

  return (
    <div>
      <ul>
        {todos?.map((todo: any) => (
          <li key={todo.id}>{todo.title}</li>
        ))}
      </ul>
      <button onClick={() => mutation.mutate("New Todo")}>
        {mutation.isPending ? "Adding..." : "Add"}
      </button>
    </div>
  );
}
```

### 无限滚动

```typescript
"use client";

import { useInfiniteQuery } from "@tanstack/react-query";
import { useInView } from "react-intersection-observer";
import { useEffect } from "react";

async function fetchProducts({ pageParam = 1 }) {
  const res = await fetch(`/api/products?page=${pageParam}&limit=10`);
  return res.json();
}

export default function InfiniteProductList() {
  const { ref, inView } = useInView();

  const { data, fetchNextPage, hasNextPage, isFetchingNextPage } =
    useInfiniteQuery({
      queryKey: ["products"],
      queryFn: fetchProducts,
      getNextPageParam: (lastPage, pages) =>
        lastPage.hasMore ? pages.length + 1 : undefined,
      initialPageParam: 1,
    });

  useEffect(() => {
    if (inView && hasNextPage) {
      fetchNextPage();
    }
  }, [inView, hasNextPage, fetchNextPage]);

  return (
    <div>
      {data?.pages.map((page, i) => (
        <div key={i}>
          {page.products.map((product: any) => (
            <div key={product.id}>{product.name}</div>
          ))}
        </div>
      ))}

      <div ref={ref}>{isFetchingNextPage && <div>Loading more...</div>}</div>
    </div>
  );
}
```

## 实战场景四: 实时搜索

### 防抖搜索

```typescript
"use client";

import { useState, useEffect } from "react";
import { useDebouncedCallback } from "use-debounce";
import useSWR from "swr";

const fetcher = (url: string) => fetch(url).then((res) => res.json());

export default function SearchProducts() {
  const [query, setQuery] = useState("");
  const [debouncedQuery, setDebouncedQuery] = useState("");

  const debounced = useDebouncedCallback(
    (value: string) => setDebouncedQuery(value),
    300
  );

  useEffect(() => {
    debounced(query);
  }, [query, debounced]);

  const { data, isLoading } = useSWR(
    debouncedQuery ? `/api/search?q=${debouncedQuery}` : null,
    fetcher
  );

  return (
    <div>
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="Search..."
      />

      {isLoading && <div>Searching...</div>}

      <ul>
        {data?.results.map((result: any) => (
          <li key={result.id}>{result.name}</li>
        ))}
      </ul>
    </div>
  );
}
```

## 总结

客户端数据获取是构建交互式应用的关键。通过本文的学习,我们掌握了:

### 核心技术

1. **useEffect + fetch**: 基础数据获取
2. **SWR**: 自动缓存和重新验证
3. **React Query**: 强大的数据管理
4. **无限滚动**: 分页加载优化

### 最佳实践

1. **合理使用缓存**: 减少不必要的请求
2. **错误处理**: 提供友好的错误提示
3. **性能优化**: 使用 memo 和 callback
4. **请求取消**: 避免内存泄漏

### 实战技巧

1. **乐观更新**: 提升用户体验
2. **防抖搜索**: 减少请求次数
3. **条件获取**: 按需加载数据
4. **预加载**: 提前获取数据

### 实战场景七: 错误处理与重试

#### 自定义错误处理

```typescript
"use client";

import useSWR from "swr";

export default function ErrorHandling() {
  const { data, error, isLoading } = useSWR("/api/data", fetcher, {
    onError: (err, key) => {
      console.error(`Error fetching ${key}:`, err);
      // 发送到错误追踪服务
      trackError(err);
    },
    onErrorRetry: (error, key, config, revalidate, { retryCount }) => {
      // 最多重试3次
      if (retryCount >= 3) return;

      // 5秒后重试
      setTimeout(() => revalidate({ retryCount }), 5000);
    },
  });

  if (error) {
    return (
      <div>
        <h2>Error occurred</h2>
        <p>{error.message}</p>
        <button onClick={() => mutate("/api/data")}>Retry</button>
      </div>
    );
  }

  if (isLoading) return <div>Loading...</div>;

  return <div>{data.value}</div>;
}
```

#### 使用 SWR 作为全局状态

```typescript
"use client";

import useSWR, { mutate } from "swr";

// 全局状态
export function useGlobalState<T>(key: string, initialData: T) {
  const { data } = useSWR(key, null, {
    fallbackData: initialData,
  });

  const setState = (value: T | ((prev: T) => T)) => {
    mutate(
      key,
      typeof value === "function" ? (value as (prev: T) => T)(data) : value,
      false
    );
  };

  return [data, setState] as const;
}

// 使用示例
export default function Counter() {
  const [count, setCount] = useGlobalState("count", 0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}
```

"use client";

import useSWR from "swr";
import { useEffect, useState } from "react";

export default function OfflineSupport() {
  const [isOnline, setIsOnline] = useState(true);

  useEffect(() => {
    setIsOnline(navigator.onLine);

    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);

    window.addEventListener("online", handleOnline);
    window.addEventListener("offline", handleOffline);

    return () => {
      window.removeEventListener("online", handleOnline);
      window.removeEventListener("offline", handleOffline);
    };
  }, []);

  const { data } = useSWR(isOnline ? "/api/data" : null, fetcher, {
    fallbackData: { value: "Offline data" },
  });

  return (
    <div>
      {!isOnline && <div>You are offline</div>}
      <div>{data?.value}</div>
    </div>
  );
}

### 实战场景十五: 数据导出

#### 导出为 CSV

```typescript
"use client";

import useSWR from "swr";

export default function ExportData() {
  const { data: products } = useSWR("/api/products", fetcher);

  function exportToCSV() {
    if (!products) return;

    const csv = [
      ["ID", "Name", "Price"],
      ...products.map((p: any) => [p.id, p.name, p.price]),
    ]
      .map((row) => row.join(","))
      .join("\n");

    const blob = new Blob([csv], { type: "text/csv" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "products.csv";
    a.click();
  }

  return (
    <div>
      <button onClick={exportToCSV}>Export to CSV</button>
    </div>
  );
}
```
