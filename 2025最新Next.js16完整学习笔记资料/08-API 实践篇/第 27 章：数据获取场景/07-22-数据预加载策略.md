# 数据预加载策略

数据预加载是提升用户体验的关键技术。通过提前加载用户可能需要的数据,可以显著减少等待时间,让应用感觉更快。

## 核心概念

### 预加载类型

| 类型       | 触发时机   | 适用场景   |
| ---------- | ---------- | ---------- |
| 路由预加载 | 链接可见时 | 页面导航   |
| 数据预加载 | 用户操作前 | 列表详情   |
| 组件预加载 | 即将显示时 | 懒加载组件 |
| 资源预加载 | 页面加载时 | 图片、字体 |

### 预加载时机

预加载的时机选择直接影响效果:

```typescript
"use client";

import { useEffect } from "react";
import { preload } from "swr";

export default function TimingExample() {
  useEffect(() => {
    // 1. 立即预加载 - 页面加载时
    preload("/api/critical-data", fetcher);

    // 2. 延迟预加载 - 1秒后
    setTimeout(() => {
      preload("/api/secondary-data", fetcher);
    }, 1000);

    // 3. 空闲时预加载
    if ("requestIdleCallback" in window) {
      requestIdleCallback(() => {
        preload("/api/low-priority-data", fetcher);
      });
    }
  }, []);

  return <div>{/* 内容 */}</div>;
}
```

```typescript
export default function SmartPrefetch() {
  const [hoveredId, setHoveredId] = useState<string | null>(null);
  const [hoverTime, setHoverTime] = useState(0);

      setHoverTime(0);
    };
  }, [hoveredId, hoverTime]);

}
```

### 基于滚动位置预加载

```typescript
"use client";

import { useEffect, useRef } from "react";
import { preload } from "swr";

export default function ScrollPrefetch({ items }: { items: any[] }) {
  const itemRefs = useRef<(HTMLDivElement | null)[]>([]);

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            const index = itemRefs.current.indexOf(
              entry.target as HTMLDivElement
            );
            // 预加载下一个项目
            if (index < items.length - 1) {
              const nextItem = items[index + 1];
              preload(`/api/items/${nextItem.id}`, fetcher);
            }
          }
        });
      },
      { rootMargin: "100px" }
    );

    itemRefs.current.forEach((ref) => {
      if (ref) observer.observe(ref);
    });

    return () => observer.disconnect();
  }, [items]);

  return (
    <div>
      {items.map((item, index) => (
        <div key={item.id} ref={(el) => (itemRefs.current[index] = el)}>
          {item.title}
        </div>
      ))}
    </div>
  );
}
```

```typescript
import { useEffect, useState } from "react";

export default function PrefetchAnalytics() {
  const [metrics, setMetrics] = useState({
    prefetchCount: 0,
    hitCount: 0,
    missCount: 0,
  });

  useEffect(() => {
    // 记录预加载
    window.addEventListener("prefetch", () => {
      setMetrics((prev) => ({
        ...prev,
        prefetchCount: prev.prefetchCount + 1,
      }));
    });

    // 记录命中
    window.addEventListener("cache-hit", () => {
      setMetrics((prev) => ({
        ...prev,
        hitCount: prev.hitCount + 1,
      }));
    });

    // 记录未命中
    window.addEventListener("cache-miss", () => {
      setMetrics((prev) => ({
        ...prev,
        missCount: prev.missCount + 1,
      }));
    });
  }, []);

  const hitRate =
    metrics.prefetchCount > 0
      ? ((metrics.hitCount / metrics.prefetchCount) * 100).toFixed(2)
      : 0;

  return (
    <div>
      <p>预加载次数: {metrics.prefetchCount}</p>
      <p>命中次数: {metrics.hitCount}</p>
      <p>未命中次数: {metrics.missCount}</p>
      <p>命中率: {hitRate}%</p>
    </div>
  );
}
```

```typescript
import { useEffect, useRef } from "react";

const fetcher = (url: string) => fetch(url).then((r) => r.json());

export default function ProductListWithPrefetch() {
  const router = useRouter();
  const { data } = useSWR("/api/products", fetcher);
  const prefetchedIds = useRef(new Set<string>());

  function handleProductHover(productId: string) {
    // 避免重复预加载
    if (prefetchedIds.current.has(productId)) return;

        img.src = src;

    prefetchedIds.current.add(productId);
  }

    <div className="product-grid">
      {data?.products.map((product: any) => (
        <div
          key={product.id}
          onMouseEnter={() => handleProductHover(product.id)}
          onClick={() => router.push(`/products/${product.id}`)}
        >
```

```typescript
"use client";

import { useEffect, useRef, useState } from "react";
import { preload } from "swr";

export default function NewsListWithPrefetch({
  articles,
}: {
  articles: any[];
}) {
  const [visibleArticles, setVisibleArticles] = useState<Set<string>>(
    new Set()
  );
  const observerRef = useRef<IntersectionObserver>();

  useEffect(() => {
    observerRef.current = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          const articleId = entry.target.getAttribute("data-id");
          if (!articleId) return;

          if (entry.isIntersecting) {
            setVisibleArticles((prev) => new Set(prev).add(articleId));

            // 预加载文章内容
            preload(`/api/articles/${articleId}`, fetcher);

            // 预加载下一篇文章
            const currentIndex = articles.findIndex((a) => a.id === articleId);
            if (currentIndex < articles.length - 1) {
              const nextArticle = articles[currentIndex + 1];
              preload(`/api/articles/${nextArticle.id}`, fetcher);
            }
          }
        });
      },
      { rootMargin: "200px" }
    );

    return () => observerRef.current?.disconnect();
  }, [articles]);

  return (
    <div>
      {articles.map((article) => (
        <article
          key={article.id}
          data-id={article.id}
          ref={(el) => {
            if (el && observerRef.current) {
              observerRef.current.observe(el);
            }
          }}
        >
          <h2>{article.title}</h2>
          <p>{article.summary}</p>
        </article>
      ))}
    </div>
  );
}
```

### 10. 如何实现智能预加载队列?

```typescript
"use client";

import { useEffect, useState } from "react";

class PrefetchQueue {
  private queue: Array<() => Promise<any>> = [];
  private running = 0;
  private maxConcurrent = 2;

  add(task: () => Promise<any>) {
    this.queue.push(task);
    this.process();
  }

  private async process() {
    if (this.running >= this.maxConcurrent || this.queue.length === 0) {
      return;
    }

    this.running++;
    const task = this.queue.shift();

    if (task) {
      try {
        await task();
      } finally {
        this.running--;
        this.process();
      }
    }
  }
}

const prefetchQueue = new PrefetchQueue();

export default function QueuedPrefetch({ ids }: { ids: string[] }) {
  useEffect(() => {
    ids.forEach((id) => {
      prefetchQueue.add(() => fetch(`/api/items/${id}`).then((r) => r.json()));
    });
  }, [ids]);

  return <div>{/* 内容 */}</div>;
}
```

## 总结

数据预加载是提升应用性能的关键策略。通过本文的学习,我们掌握了:

### 核心技术

1. **路由预加载**: Link 组件和 router.prefetch()
2. **数据预加载**: SWR 和 React Query
3. **资源预加载**: preload、prefetch、preconnect
4. **组件预加载**: 动态导入和条件加载

### 最佳实践

1. **智能预加载**: 基于用户行为和设备性能
2. **优先级管理**: 高优先级数据优先加载
3. **错误处理**: 预加载失败不影响主流程
4. **性能监控**: 跟踪预加载效果

### 实战技巧

1. **避免过度预加载**: 限制并发数量
2. **网络感知**: 根据网络状态调整策略
3. **缓存管理**: 避免重复预加载
4. **取消机制**: 及时取消不需要的预加载

通过合理使用预加载策略,可以显著提升应用的响应速度和用户体验。
