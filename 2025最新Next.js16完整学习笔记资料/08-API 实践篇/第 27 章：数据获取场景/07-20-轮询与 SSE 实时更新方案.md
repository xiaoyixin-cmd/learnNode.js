# 轮询与 SSE 实时更新方案

在现代 Web 应用中,实时数据更新是一个常见需求。本文将介绍如何在 Next.js 中实现轮询和 Server-Sent Events (SSE) 两种实时更新方案。

## 基础概念

### 实时更新方案对比

| 方案      | 实现难度 | 服务器压力 | 实时性 | 浏览器兼容性 |
| --------- | -------- | ---------- | ------ | ------------ |
| 轮询      | 低       | 高         | 中     | 好           |
| 长轮询    | 中       | 中         | 高     | 好           |
| SSE       | 中       | 低         | 高     | 好           |
| WebSocket | 高       | 低         | 最高   | 好           |

### 选择建议

| 场景         | 推荐方案  | 原因     |
| ------------ | --------- | -------- |
| 简单数据更新 | 轮询      | 实现简单 |
| 单向推送     | SSE       | 效率高   |
| 双向通信     | WebSocket | 功能强大 |
| 低频更新     | 长轮询    | 平衡性能 |

## 实战场景二: 长轮询

### 基础长轮询

```typescript
"use client";

import { useEffect, useState } from "react";

export default function LongPolling() {
  const [messages, setMessages] = useState<string[]>([]);

  useEffect(() => {
    let active = true;

    async function poll() {
      while (active) {
        try {
          const res = await fetch("/api/long-poll");
          const data = await res.json();

          if (data.message) {
            setMessages((prev) => [...prev, data.message]);
          }
        } catch (error) {
          console.error("Polling error:", error);
          await new Promise((resolve) => setTimeout(resolve, 1000));
        }
      }
    }

    poll();

    return () => {
      active = false;
    };
  }, []);

  return (
    <ul>
      {messages.map((msg, i) => (
        <li key={i}>{msg}</li>
      ))}
    </ul>
  );
}
```

```typescript
async function waitForMessage(timeout: number): Promise<string | null> {
  const start = Date.now();

  while (Date.now() - start < timeout) {
    const message = await checkForNewMessage();
    if (message) return message;

    await new Promise((resolve) => setTimeout(resolve, 100));
  }

  return null;
}
```

### 基础 SSE 实现

```typescript
"use client";

import { useEffect, useState } from "react";

export default function SSEExample() {
  const [messages, setMessages] = useState<string[]>([]);
  const [status, setStatus] = useState("disconnected");

  useEffect(() => {
    const eventSource = new EventSource("/api/events");

    eventSource.onopen = () => {
      setStatus("connected");
    };

    eventSource.onmessage = (event) => {

      setStatus("error");
      eventSource.close();
    };

    return () => {
      eventSource.close();
    };
  }, []);

  return (
    <div>
      <p>Status: {status}</p>
      <ul>
        {messages.map((msg, i) => (
          <li key={i}>{msg}</li>
        ))}
      </ul>
    </div>
  );
}
```

### SSE API 路由

```typescript
// app/api/events/route.ts
export async function GET() {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    async start(controller) {
      // 发送初始消息
      controller.enqueue(
        encoder.encode(`data: ${JSON.stringify({ message: "Connected" })}\n\n`)
      );

      // 定期发送更新
      const interval = setInterval(() => {
        const data = {
          message: `Update at ${new Date().toISOString()}`,
          timestamp: Date.now(),
        };

        controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`));
      }, 1000);

      // 清理
      setTimeout(() => {
        clearInterval(interval);
        controller.close();
      }, 60000); // 60秒后关闭
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      Connection: "keep-alive",
    },
  });
}
```

```typescript
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === "Enter" && sendMessage()}

  );
```

### 聊天 API

```typescript
// app/api/chat/route.ts
const messages: Message[] = [];
const subscribers = new Set<ReadableStreamDefaultController>();

export async function GET() {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    start(controller) {
      subscribers.add(controller);

      // 发送历史消息
      messages.forEach((msg) => {
        controller.enqueue(encoder.encode(`data: ${JSON.stringify(msg)}\n\n`));
      });

      return () => {
        subscribers.delete(controller);
      };
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
    },
  });
}

export async function POST(request: Request) {
  const { text } = await request.json();

  const message: Message = {
    id: Date.now().toString(),
    user: "User",
    text,
    timestamp: Date.now(),
  };

  messages.push(message);

  // 广播给所有订阅者
  const encoder = new TextEncoder();
  const data = encoder.encode(`data: ${JSON.stringify(message)}\n\n`);

  subscribers.forEach((controller) => {
    try {
      controller.enqueue(data);
    } catch (error) {
      subscribers.delete(controller);
    }
  });

  return Response.json({ success: true });
}
```

```typescript
"use client";

import { useEffect, useState } from "react";

interface Notification {
  id: string;
  title: string;
  message: string;
  type: "info" | "success" | "warning" | "error";
}

export default function Notifications() {
  const [notifications, setNotifications] = useState<Notification[]>([]);

  useEffect(() => {
    const eventSource = new EventSource("/api/notifications");

    eventSource.onmessage = (event) => {
      const notification = JSON.parse(event.data);
      setNotifications((prev) => [...prev, notification]);

      // 5秒后自动移除
      setTimeout(() => {
        setNotifications((prev) =>
          prev.filter((n) => n.id !== notification.id)
        );
      }, 5000);
    };

    return () => eventSource.close();
  }, []);

  return (
    <div style={{ position: "fixed", top: 20, right: 20 }}>
      {notifications.map((notif) => (
        <div
          key={notif.id}
          style={{
            padding: "10px",
            margin: "10px 0",
            background: getColor(notif.type),
            borderRadius: "4px",
          }}
        >
          <h4>{notif.title}</h4>
          <p>{notif.message}</p>
        </div>
      ))}
    </div>
  );
}

function getColor(type: string) {
  switch (type) {
    case "success":
      return "#4caf50";
    case "warning":
      return "#ff9800";
    case "error":
      return "#f44336";
    default:
      return "#2196f3";
  }
}
```

### 2. 错误处理

```typescript
"use client";

import { useEffect, useState } from "react";

export default function ErrorHandling() {
  const [error, setError] = useState<string | null>(null);
  const [retryCount, setRetryCount] = useState(0);

  useEffect(() => {
    if (retryCount >= 3) {
      setError("Max retries reached");
      return;
    }

    const eventSource = new EventSource("/api/events");

    eventSource.onerror = () => {
      eventSource.close();
      setRetryCount((prev) => prev + 1);
    };

    return () => eventSource.close();
  }, [retryCount]);

  if (error) return <div>Error: {error}</div>;

  return <div>Connected (Retry: {retryCount})</div>;
}
```

## 常见问题

### 1. 如何选择轮询还是 SSE?

根据需求选择:

- 简单数据更新: 轮询
- 实时推送: SSE
- 双向通信: WebSocket

### 2. SSE 连接断开怎么办?

实现自动重连:

```typescript
function createReconnectingSSE(url: string) {
  let es: EventSource | null = null;

  function connect() {
    es = new EventSource(url);

    es.onerror = () => {
      es?.close();
      setTimeout(connect, 5000);
    };
  }

  connect();

  return () => es?.close();
}
```

## 总结

### 核心技术

1. **轮询**: 简单可靠的实时更新方案
2. **长轮询**: 减少请求次数的优化方案
3. **SSE**: 服务器主动推送的实时方案
4. **自适应策略**: 根据场景选择合适的方案

### 最佳实践

1. **连接管理**: 实现自动重连
2. **错误处理**: 优雅降级
3. **性能优化**: 合理的更新频率
4. **内存管理**: 限制数据量

### 实战技巧

```typescript
import { useEffect, useState } from "react";

interface Stock {
  symbol: string;
  price: number;
  change: number;
}

export default function StockTicker() {
  const [stocks, setStocks] = useState<Stock[]>([]);

  useEffect(() => {
    const es = new EventSource("/api/stocks");

    es.onmessage = (event) => {
      const stock = JSON.parse(event.data);
      setStocks((prev) => {
        const index = prev.findIndex((s) => s.symbol === stock.symbol);
        if (index >= 0) {
          const newStocks = [...prev];
          newStocks[index] = stock;
          return newStocks;
        }
        return [...prev, stock];
      });
    };

    return () => es.close();
  }, []);

  return (
    <table>
      <thead>
        <tr>
          <th>Symbol</th>
          <th>Price</th>
          <th>Change</th>
        </tr>
      </thead>
      <tbody>
        {stocks.map((stock) => (
          <tr key={stock.symbol}>
            <td>{stock.symbol}</td>
            <td>${stock.price.toFixed(2)}</td>
            <td style={{ color: stock.change >= 0 ? "green" : "red" }}>
              {stock.change >= 0 ? "+" : ""}
              {stock.change.toFixed(2)}%
            </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
}
```

```typescript
    <div>
      <h2>Live Locations</h2>

            {loc.name}: ({loc.lat.toFixed(4)}, {loc.lng.toFixed(4)})
          </li>
        ))}

    </div>
```

#### 实时评论

```typescript
"use client";

import { useEffect, useState } from "react";

interface Comment {
  id: string;
  user: string;
  text: string;
  timestamp: number;
}

export default function LiveComments({ postId }: { postId: string }) {
  const [comments, setComments] = useState<Comment[]>([]);

  useEffect(() => {
    const es = new EventSource(`/api/posts/${postId}/comments`);

    es.onmessage = (event) => {
      const comment = JSON.parse(event.data);
      setComments((prev) => [comment, ...prev]);
    };

    return () => es.close();
  }, [postId]);

  return (
    <div>
      <h3>Comments ({comments.length})</h3>
      {comments.map((comment) => (
        <div
          key={comment.id}
          style={{ borderBottom: "1px solid #eee", padding: "10px" }}
        >
          <strong>{comment.user}</strong>
          <p>{comment.text}</p>
          <small>{new Date(comment.timestamp).toLocaleString()}</small>
        </div>
      ))}
    </div>
  );
}
```

### 性能对比

| 方案       | CPU 占用 | 内存占用 | 网络流量 | 延迟   |
| ---------- | -------- | -------- | -------- | ------ |
| 轮询(1 秒) | 高       | 中       | 高       | 1 秒   |
| 长轮询     | 中       | 中       | 中       | <1 秒  |
| SSE        | 低       | 低       | 低       | <100ms |
| WebSocket  | 低       | 低       | 低       | <50ms  |

### 选择决策树

```
需要双向通信?
├─ 是 → WebSocket
└─ 否 → 需要实时性?
    ├─ 高 → SSE
    └─ 低 → 轮询
```
