# Server Actions 表单处理实战

Server Actions 是 Next.js 提供的服务端函数,可以直接在组件中调用,无需创建 API 路由。本文将介绍如何使用 Server Actions 处理表单提交。

## 基础概念

### Server Actions 特点

1. **服务端执行**: 代码在服务端运行
2. **类型安全**: 完整的 TypeScript 支持
3. **自动序列化**: 自动处理数据序列化
4. **渐进增强**: 支持无 JavaScript 环境

### Server Actions vs API Routes

| 特性     | Server Actions     | API Routes   |
| -------- | ------------------ | ------------ |
| 定义位置 | 组件文件或单独文件 | app/api 目录 |
| 调用方式 | 直接调用函数       | fetch 请求   |
| 类型安全 | 完整支持           | 需要手动定义 |
| 表单集成 | 原生支持           | 需要手动处理 |

```typescript
// app/signup/actions.ts
"use server";

import { z } from "zod";
import { redirect } from "next/navigation";

const SignupSchema = z.object({
  username: z.string().min(3).max(20),
  email: z.string().email(),
  password: z.string().min(8),
});

export async function signup(formData: FormData) {
  const data = {
    username: formData.get("username"),
    email: formData.get("email"),
    password: formData.get("password"),
  };

  const result = SignupSchema.safeParse(data);

  if (!result.success) {
    return {
      errors: result.error.flatten().fieldErrors,
    };
  }

  // 创建用户
  await db.user.create({
    data: result.data,
  });

  redirect("/dashboard");
}

// app/signup/page.tsx
("use client");

        {state?.errors?.username && <p>{state.errors.username}</p>}
      </div>
```

### 文件上传

```typescript
// app/upload/actions.ts
"use server";

import { writeFile } from "fs/promises";
import { join } from "path";

export async function uploadFile(formData: FormData) {
  const file = formData.get("file") as File;

  if (!file) {
    return { error: "No file uploaded" };
  }

  const bytes = await file.arrayBuffer();
  const buffer = Buffer.from(bytes);

  const path = join(process.cwd(), "public", "uploads", file.name);
  await writeFile(path, buffer);

  return { success: true, path: `/uploads/${file.name}` };
}

// app/upload/page.tsx
("use client");
```

```typescript
import { revalidatePath } from "next/cache";

export async function addTodo(formData: FormData) {

export async function toggleTodo(id: number) {
  const todo = await db.todo.findUnique({ where: { id } });

  await db.todo.update({
    where: { id },
    data: { completed: !todo.completed },
  });

  revalidatePath("/todos");
}

export async function deleteTodo(id: number) {
  await db.todo.delete({ where: { id } });

  revalidatePath("/todos");
}

// app/todos/page.tsx
("use client");

import { useOptimistic } from "react";
import { addTodo, toggleTodo, deleteTodo } from "./actions";

interface Todo {
  id: number;
  title: string;
  completed: boolean;
}

export default function TodosPage({ todos }: { todos: Todo[] }) {
  const [optimisticTodos, addOptimisticTodo] = useOptimistic(
    todos,
    (state, newTodo: Todo) => [...state, newTodo]
  );

  async function handleAdd(formData: FormData) {
    const title = formData.get("title") as string;

    addOptimisticTodo({
      id: Date.now(),
      title,
      completed: false,
    });

    await addTodo(formData);
  }

  return (
    <div>
      <form action={handleAdd}>
        <input type="text" name="title" />
        <button type="submit">Add</button>
      </form>

      <ul>
        {optimisticTodos.map((todo) => (
          <li key={todo.id}>
            <input
              type="checkbox"
              checked={todo.completed}
              onChange={() => toggleTodo(todo.id)}
            />
            {todo.title}
            <button onClick={() => deleteTodo(todo.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

```typescript
import { cookies } from "next/headers";

export async function saveStep1(formData: FormData) {
  const data = {
    email: formData.get("email"),
    password: formData.get("password"),
  };

  // 保存到 session

export async function saveStep2(formData: FormData) {
  const data = {
    firstName: formData.get("firstName"),
    lastName: formData.get("lastName"),
  };

  cookies().set("register_step2", JSON.stringify(data));

  return { success: true };
}

export async function completeRegistration(formData: FormData) {
  const step1 = JSON.parse(cookies().get("register_step1")?.value || "{}");
  const step2 = JSON.parse(cookies().get("register_step2")?.value || "{}");

  const preferences = {
    newsletter: formData.get("newsletter") === "on",
    notifications: formData.get("notifications") === "on",
  };

  // 创建用户
  await db.user.create({
    data: {
      ...step1,
      ...step2,
      ...preferences,
    },
  });

  // 清除 session
  cookies().delete("register_step1");
  cookies().delete("register_step2");

  redirect("/dashboard");
}
```

## 注意事项

### 1. 必须添加 'use server' 指令

```typescript
// 错误:缺少 'use server'
export async function submitForm(formData: FormData) {
  // 这不是 Server Action
}

// 正确:添加 'use server'
("use server");

export async function submitForm(formData: FormData) {
  // 这是 Server Action
}
```

### 2. 只能传递可序列化的数据

```typescript
// 错误:传递函数
"use server";

export async function action(callback: () => void) {
  // 函数无法序列化
}

// 正确:传递基本类型
("use server");

export async function action(data: string) {
  // 字符串可以序列化
}
```

### 3. 使用 revalidatePath 更新缓存

```typescript
"use server";

import { revalidatePath } from "next/cache";

export async function updateData(formData: FormData) {
  await db.update(formData);

  // 必须手动重新验证
  revalidatePath("/data");
}
```

### 4. 处理文件上传时注意大小限制

```typescript
"use server";

export async function uploadFile(formData: FormData) {
```

## 常见问题

### 1. Server Actions 和 API Routes 有什么区别?

Server Actions 是函数,API Routes 是端点:

```typescript
// Server Action
"use server";
export async function action() {
  return { data: "value" };
}

// API Route
export async function GET() {
  return Response.json({ data: "value" });
}
```

### 2. 如何在 Server Actions 中访问 cookies?

使用 next/headers:

```typescript
"use server";

import { cookies } from "next/headers";

export async function action() {
  const token = cookies().get("token");
  return { token };
}
```

### 3. 如何在 Server Actions 中返回错误?

返回包含错误信息的对象:

```typescript
"use server";

export async function action(formData: FormData) {
  if (!formData.get("name")) {
    return { error: "Name is required" };
  }

  return { success: true };
}
```

### 4. 如何实现表单重置?

使用 ref 调用 reset:

```typescript
"use client";

import { useRef } from "react";

export default function Form() {
  const ref = useRef<HTMLFormElement>(null);

  async function action(formData: FormData) {
    await submitForm(formData);
    ref.current?.reset();
  }

  return (
    <form ref={ref} action={action}>
      ...
    </form>
  );
}
```

### 5. 如何处理多个表单字段?

使用 Object.fromEntries:

```typescript
"use server";

export async function action(formData: FormData) {
  const data = Object.fromEntries(formData);
  // { name: 'value', email: 'value', ... }
}
```

### 6. 如何实现表单验证?

使用 Zod 或其他验证库:

```typescript
"use server";

import { z } from "zod";

const schema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
});

export async function action(formData: FormData) {
  const result = schema.safeParse(Object.fromEntries(formData));

  if (!result.success) {
    return { errors: result.error.flatten() };
  }

  return { success: true };
}
```

### 7. 如何实现文件上传进度?

Server Actions 不支持进度,使用 API Route:

```typescript
// 使用 API Route
export async function POST(request: Request) {
  const formData = await request.formData();
  const file = formData.get("file") as File;

  // 处理上传
}
```

### 8. 如何在 Server Actions 中使用数据库?

直接导入 ORM 客户端:

```typescript
"use server";

import { prisma } from "@/lib/prisma";

export async function action(formData: FormData) {
  await prisma.user.create({
    data: Object.fromEntries(formData),
  });
}
```

### 9. 如何实现防抖?

在客户端使用 debounce:

```typescript
"use client";

import { useDebouncedCallback } from "use-debounce";

export default function Form() {
  const debounced = useDebouncedCallback(async (formData: FormData) => {
    await action(formData);
  }, 300);

  return <form action={debounced}>...</form>;
}
```

### 10. 如何测试 Server Actions?

使用 Jest 或 Vitest:

```typescript
import { action } from "./actions";

describe("action", () => {
  it("should create user", async () => {
    const formData = new FormData();
    formData.set("name", "John");

    const result = await action(formData);

    expect(result.success).toBe(true);
  });

```

## 总结

Server Actions 提供了一种简单的方式来处理表单提交和数据变更。通过本文的学习,我们掌握了:

### 核心功能

1. **表单处理**: 直接使用 action 属性
2. **数据验证**: 使用 Zod 等验证库
3. **文件上传**: 处理 FormData
4. **乐观更新**: 使用 useOptimistic

### 最佳实践

1. **添加验证**: 始终验证用户输入
2. **错误处理**: 返回错误信息
3. **重新验证**: 使用 revalidatePath
4. **类型安全**: 使用 TypeScript

### 实战技巧

1. **多步骤表单**: 使用 cookies 保存状态
2. **加载状态**: 使用 useFormStatus
3. **表单重置**: 使用 ref.current.reset()
4. **批量操作**: 使用 Promise.all

通过合理使用 Server Actions,可以简化表单处理流程,提升开发效率。

```typescript
"use server";

import { prisma } from "@/lib/prisma";

export async function transferMoney(formData: FormData) {
  const fromId = parseInt(formData.get("fromId") as string);
  const toId = parseInt(formData.get("toId") as string);
  const amount = parseFloat(formData.get("amount") as string);

  try {
    await prisma.$transaction(async (tx) => {
      // 扣除发送方余额
      await tx.account.update({
        where: { id: fromId },
        data: { balance: { decrement: amount } },
      });

      // 增加接收方余额
      await tx.account.update({
        where: { id: toId },
        data: { balance: { increment: amount } },
      });

      // 记录交易
      await tx.transaction.create({
        data: { fromId, toId, amount },
      });
    });

    return { success: true };
  } catch (error) {
    return { error: "Transfer failed" };
  }
}
```

```typescript
// app/login/actions.ts
"use server";

import { cookies } from "next/headers";
import { redirect } from "next/navigation";
import bcrypt from "bcrypt";
import jwt from "jsonwebtoken";

export async function login(formData: FormData) {
  const email = formData.get("email") as string;
  const password = formData.get("password") as string;

  // 查找用户
  const user = await db.user.findUnique({
    where: { email },
  });

  if (!user) {
    return { error: "Invalid credentials" };
  }

  // 验证密码
  const valid = await bcrypt.compare(password, user.password);

  if (!valid) {
    return { error: "Invalid credentials" };
  }

  // 生成 token
  const token = jwt.sign({ userId: user.id }, process.env.JWT_SECRET!, {
    expiresIn: "7d",
  });

  // 设置 cookie
  cookies().set("token", token, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 60 * 60 * 24 * 7, // 7 days
  });

  redirect("/dashboard");
}

export async function logout() {
  cookies().delete("token");
  redirect("/login");
}
```

```typescript
// app/admin/actions.ts
"use server";

import { cookies } from "next/headers";
import jwt from "jsonwebtoken";

async function checkAuth() {
  const token = cookies().get("token")?.value;

  if (!token) {
    throw new Error("Unauthorized");
  }

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET!);
    return payload as { userId: number };
  } catch (error) {
    throw new Error("Invalid token");
  }
}

export async function adminAction(formData: FormData) {
  const { userId } = await checkAuth();

  // 检查是否是管理员
  const user = await db.user.findUnique({
    where: { id: userId },
  });

  if (!user?.isAdmin) {
    throw new Error("Forbidden");
  }

  // 执行管理员操作
  await db.admin.create({
    data: Object.fromEntries(formData),
  });

  return { success: true };
}
```

```typescript
  const [query, setQuery] = useState("");
  const [results, setResults] = useState([]);

  const debouncedSearch = useDebouncedCallback(async (value: string) => {
    if (value.length < 2) {
      setResults([]);
      return;
    }

    const results = await search(value);
    setResults(results);
  }, 300);

  useEffect(() => {
    debouncedSearch(query);
  }, [query, debouncedSearch]);

  return (
    <div>
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="Search..."
      />

      <ul>
        {results.map((result: any) => (
          <li key={result.id}>{result.name}</li>
        ))}
      </ul>
    </div>
  );
}
```

```typescript
// app/items/page.tsx

import { useState } from "react";
import { bulkDelete } from "./actions";

export default function ItemsPage({ items }: { items: any[] }) {
  const [selected, setSelected] = useState<number[]>([]);

  async function handleBulkDelete() {
    if (selected.length === 0) return;

    const result = await bulkDelete(selected);
    alert(`Deleted ${result.count} items`);
    setSelected([]);
  }

  function toggleSelect(id: number) {
    setSelected((prev) =>
      prev.includes(id) ? prev.filter((i) => i !== id) : [...prev, id]
    );
  }

  return (
    <div>
      <button onClick={handleBulkDelete} disabled={selected.length === 0}>
        Delete Selected ({selected.length})
      </button>

      <ul>
        {items.map((item) => (
          <li key={item.id}>
            <input
              type="checkbox"
              checked={selected.includes(item.id)}
              onChange={() => toggleSelect(item.id)}
            />
            {item.name}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

```typescript
// app/import/actions.ts
"use server";

import { parse } from "csv-parse/sync";

export async function importCSV(formData: FormData) {
  const file = formData.get("file") as File;

  if (!file) {
    return { error: "No file uploaded" };
  }

  const text = await file.text();
  const records = parse(text, {
    columns: true,
    skip_empty_lines: true,
  });

  let imported = 0;
  let failed = 0;

  for (const record of records) {
    try {
      await db.product.create({
        data: {
          name: record.name,
          price: parseFloat(record.price),
          stock: parseInt(record.stock),
        },
      });
      imported++;
    } catch (error) {
      failed++;
    }
  }

  return { success: true, imported, failed };
}
```
