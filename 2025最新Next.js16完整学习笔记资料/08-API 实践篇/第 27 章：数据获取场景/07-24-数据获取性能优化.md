```typescript
import { useState } from "react";

// 请求队列
let requestQueue: string[] = [];
let timer: NodeJS.Timeout | null = null;

async function batchFetch(ids: string[]) {
  const response = await fetch("/api/batch", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ ids }),
  });

  return response.json();

function queueRequest(id: string): Promise<any> {
  return new Promise((resolve) => {
    requestQueue.push(id);

    if (timer) {
      clearTimeout(timer);
    }

    timer = setTimeout(async () => {
      const ids = [...requestQueue];
      requestQueue = [];

      const results = await batchFetch(ids);

      // 这里简化处理,实际需要维护回调映射
      resolve(results);
    }, 50); // 50ms内的请求合并
  });
}

export default function BatchRequest() {
  const [data, setData] = useState<any[]>([]);

  async function loadData() {
    const results = await Promise.all([
      queueRequest("1"),
      queueRequest("2"),
      queueRequest("3"),

  }

  return (
    <div>
      <button onClick={loadData}>加载数据</button>
```

### DataLoader 模式

```typescript
"use client";

class DataLoader<K, V> {
  private cache = new Map<K, Promise<V>>();
  private queue: K[] = [];
  private timer: NodeJS.Timeout | null = null;

  constructor(
    private batchFn: (keys: K[]) => Promise<V[]>,
    private maxBatchSize = 100,
    private batchDelay = 10
  ) {}

  load(key: K): Promise<V> {
    // 检查缓存
    if (this.cache.has(key)) {
      return this.cache.get(key)!;
    }

    // 创建Promise并加入队列
    const promise = new Promise<V>((resolve, reject) => {
      this.queue.push(key);

      if (this.timer) {
        clearTimeout(this.timer);
      }

      this.timer = setTimeout(() => {
        this.dispatch();
      }, this.batchDelay);
    });

    this.cache.set(key, promise);
    return promise;
  }

  private async dispatch() {
    const keys = this.queue.splice(0, this.maxBatchSize);

    try {
      const values = await this.batchFn(keys);

      keys.forEach((key, index) => {
        const promise = this.cache.get(key);
        // 这里需要实现Promise的resolve
      });
    } catch (error) {
      keys.forEach((key) => {
        this.cache.delete(key);
      });
    }
  }

  clear() {
    this.cache.clear();
    this.queue = [];
  }
}

// 使用示例
const userLoader = new DataLoader<string, any>(async (ids) => {
  const response = await fetch("/api/users/batch", {
    method: "POST",
    body: JSON.stringify({ ids }),
  });
  return response.json();
});
```

### 防止重复请求

```typescript
"use client";

const pendingRequests = new Map<string, Promise<any>>();

async function fetchWithDedup(url: string): Promise<any> {
  // 检查是否有相同的请求正在进行
  if (pendingRequests.has(url)) {
    return pendingRequests.get(url)!;
  }

  // 创建新请求
  const promise = fetch(url)
    .then((r) => r.json())
    .finally(() => {
      // 请求完成后移除
      pendingRequests.delete(url);
    });

  pendingRequests.set(url, promise);
  return promise;
}

export default function DedupExample() {

      fetchWithDedup("/api/data"),
      fetchWithDedup("/api/data"),
      fetchWithDedup("/api/data"),
    ]);

    console.log("所有请求返回相同数据:", data1 === data2 && data2 === data3);
  }

}
```

### 限制并发数

```typescript
"use client";

async function fetchWithConcurrency<T>(
  tasks: (() => Promise<T>)[],
  concurrency: number
): Promise<T[]> {
  const results: T[] = [];
  const executing: Promise<void>[] = [];

  for (const task of tasks) {
    const promise = task().then((result) => {
      results.push(result);
    });

    executing.push(promise);

    if (executing.length >= concurrency) {
      await Promise.race(executing);
      executing.splice(
        executing.findIndex((p) => p === promise),
        1
      );
    }
  }

  await Promise.all(executing);
  return results;
}

export default function ConcurrencyLimit() {
  async function loadData() {
    const tasks = Array.from(
      { length: 100 },
      (_, i) => () => fetch(`/api/items/${i}`).then((r) => r.json())
    );

    // 最多同时3个请求
    const results = await fetchWithConcurrency(tasks, 3);
    console.log(results);
  }

}
```

```typescript
async function fetchWithCache(url: string, ttl = 60000): Promise<any> {
```

```typescript
import { useState } from "react";

export default function StreamingClient() {
  const [data, setData] = useState<any[]>([]);

  async function fetchStream() {
    const response = await fetch("/api/stream");
    const reader = response.body?.getReader();
    const decoder = new TextDecoder();

    if (!reader) return;

    while (true) {
      const { done, value } = await reader.read();

      if (done) break;

      const text = decoder.decode(value);
      const lines = text.split("\n").filter(Boolean);

      lines.forEach((line) => {
        const item = JSON.parse(line);
        setData((prev) => [...prev, item]);
      });
```

## 常见问题

### 1. 如何选择合适的缓存策略?

根据数据特性选择:

- 静态数据: 长时间缓存(24 小时+)
- 用户数据: 中等缓存(5-30 分钟)
- 实时数据: 不缓存或短时间缓存(< 1 分钟)

### 2. 并发请求数量如何设置?

建议根据场景设置:

- HTTP/1.1: 最多 6 个并发
- HTTP/2: 可以更多,但建议不超过 10 个
- 移动端: 建议 3-5 个

### 3. 何时使用请求合并?

适用场景:

- 大量小请求
- 请求间隔很短(< 100ms)
- 服务器支持批量接口

### 10. 如何监控 API 性能?

```typescript
"use client";

class APIMonitor {
  private metrics: Map<string, number[]> = new Map();

  track(url: string, duration: number) {
    if (!this.metrics.has(url)) {
      this.metrics.set(url, []);
    }

    this.metrics.get(url)!.push(duration);
  }

  getStats(url: string) {
    const durations = this.metrics.get(url) || [];

    return {
      count: durations.length,
      avg: durations.reduce((a, b) => a + b, 0) / durations.length,
      min: Math.min(...durations),
      max: Math.max(...durations),
    };
  }
}

const monitor = new APIMonitor();

export default function MonitoredFetch() {
  async function fetchData() {
    const start = performance.now();
    await fetch("/api/data");
    const duration = performance.now() - start;

    monitor.track("/api/data", duration);
    console.log(monitor.getStats("/api/data"));
  }

  return <button onClick={fetchData}>获取数据</button>;
}
```

## 实战场景十二: 完整性能优化方案

### 综合优化策略

```typescript
"use client";

import { useState, useEffect, useCallback } from "react";
import useSWR from "swr";
import { debounce } from "lodash";

// 1. 请求去重
const pendingRequests = new Map();

async function fetchWithDedup(url: string) {
  if (pendingRequests.has(url)) {
    return pendingRequests.get(url);
  }

  const promise = fetch(url)
    .then((r) => r.json())
    .finally(() => {
      pendingRequests.delete(url);
    });

  pendingRequests.set(url, promise);
  return promise;
}

// 2. 内存缓存
const cache = new Map();

function getCached(key: string, ttl = 60000) {
  const cached = cache.get(key);
  if (cached && Date.now() - cached.timestamp < ttl) {
    return cached.data;
  }
  return null;
}

function setCache(key: string, data: any) {
  cache.set(key, { data, timestamp: Date.now() });
}

// 3. 防抖搜索
export default function ComprehensiveOptimization() {
  const [query, setQuery] = useState("");
  const [debouncedQuery, setDebouncedQuery] = useState("");

  const debouncedSearch = useCallback(
    debounce((value: string) => {
      setDebouncedQuery(value);
    }, 300),
    []
  );

  // 4. SWR 自动缓存和重新验证
  const { data, error } = useSWR(
    debouncedQuery ? `/api/search?q=${debouncedQuery}` : null,
    fetchWithDedup,
    {
      revalidateOnFocus: false,
      dedupingInterval: 2000,
    }
  );

  useEffect(() => {
    debouncedSearch(query);
  }, [query, debouncedSearch]);

  return (
    <div>
      <input
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="搜索..."
      />
      {error && <div>加载失败</div>}
      {data && <div>{JSON.stringify(data)}</div>}
    </div>
  );
}
```

### 性能监控仪表板

```typescript
"use client";

import { useState, useEffect } from "react";

interface Metrics {
  requests: number;
  avgDuration: number;
  cacheHits: number;
  cacheMisses: number;
}

export default function PerformanceDashboard() {
  const [metrics, setMetrics] = useState<Metrics>({
    requests: 0,
    avgDuration: 0,
    cacheHits: 0,
    cacheMisses: 0,
  });

  useEffect(() => {
    // 监控性能指标
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.entryType === "resource") {
          const resource = entry as PerformanceResourceTiming;

          setMetrics((prev) => ({
            ...prev,
            requests: prev.requests + 1,
            avgDuration:
              (prev.avgDuration * prev.requests + resource.duration) /
              (prev.requests + 1),
          }));
        }
      }
    });

    observer.observe({ entryTypes: ["resource"] });

    return () => observer.disconnect();
  }, []);

  return (
    <div className="dashboard">
      <h2>性能监控</h2>
      <div>总请求数: {metrics.requests}</div>
      <div>平均耗时: {metrics.avgDuration.toFixed(2)}ms</div>
      <div>缓存命中: {metrics.cacheHits}</div>
      <div>缓存未命中: {metrics.cacheMisses}</div>
      <div>
        缓存命中率:{" "}
        {metrics.requests > 0
          ? ((metrics.cacheHits / metrics.requests) * 100).toFixed(2)
          : 0}
        %
      </div>
    </div>
  );
}
```

```typescript
"use client";

import { useEffect, useState } from "react";

export default function LazyLoadResources() {
  const [nonCriticalData, setNonCriticalData] = useState(null);

  useEffect(() => {
    // 等待主要内容加载完成
    if (document.readyState === "complete") {
      loadNonCriticalData();
    } else {
      window.addEventListener("load", loadNonCriticalData);
      return () => window.removeEventListener("load", loadNonCriticalData);
    }
  }, []);

  async function loadNonCriticalData() {
    // 使用 requestIdleCallback 在空闲时加载
    if ("requestIdleCallback" in window) {
      requestIdleCallback(async () => {
        const response = await fetch("/api/non-critical-data");
        const data = await response.json();
        setNonCriticalData(data);
      });
    } else {
      setTimeout(async () => {
        const response = await fetch("/api/non-critical-data");
        const data = await response.json();
        setNonCriticalData(data);
      }, 1000);
    }
  }

  return (
    <div>
      <div>关键内容</div>
      {nonCriticalData && (
        <div>非关键内容: {JSON.stringify(nonCriticalData)}</div>
      )}
    </div>
  );
}
```
