# 缓存性能测试与分析

## 概述

缓存性能测试是验证缓存策略是否有效的关键步骤。通过系统的性能测试和分析,可以发现缓存配置中的问题,优化缓存策略,提升应用整体性能。

### 为什么需要缓存性能测试

1. **验证缓存效果**: 确认缓存是否真正提升了性能
2. **发现性能瓶颈**: 找出系统中的性能问题
3. **优化缓存配置**: 调整 TTL、缓存层级等参数
4. **监控缓存健康**: 跟踪缓存命中率、响应时间等指标
5. **容量规划**: 确定缓存容量需求

### 测试的关键指标

1. **缓存命中率**: 缓存命中次数 / 总请求次数
2. **响应时间**: 请求的平均响应时间
3. **吞吐量**: 单位时间内处理的请求数
4. **内存使用**: 缓存占用的内存大小
5. **失效率**: 缓存失效的频率

```typescript
// lib/testing/performance.ts

interface PerformanceMetrics {
  totalRequests: number;
  cacheHits: number;
  cacheMisses: number;
  avgResponseTime: number;
  minResponseTime: number;
  maxResponseTime: number;
  p50ResponseTime: number;
  p95ResponseTime: number;
  p99ResponseTime: number;
}

class PerformanceTester {
  private metrics: {
    responseTimes: number[];
    cacheHits: number;
    cacheMisses: number;
  } = {
    responseTimes: [],
    cacheHits: 0,
    cacheMisses: 0,
  };

  // 记录请求
  recordRequest(responseTime: number, cacheHit: boolean) {
    this.metrics.responseTimes.push(responseTime);
    if (cacheHit) {
      this.metrics.cacheHits++;
    } else {
      this.metrics.cacheMisses++;
    }
  }

  // 计算百分位数
  private calculatePercentile(arr: number[], percentile: number): number {
    const sorted = [...arr].sort((a, b) => a - b);
    const index = Math.ceil((percentile / 100) * sorted.length) - 1;
    return sorted[index];
  }

  // 获取测试结果
  getMetrics(): PerformanceMetrics {
    const times = this.metrics.responseTimes;
    const total = this.metrics.cacheHits + this.metrics.cacheMisses;

    return {
      totalRequests: total,
      cacheHits: this.metrics.cacheHits,
      cacheMisses: this.metrics.cacheMisses,
      avgResponseTime: times.reduce((a, b) => a + b, 0) / times.length,
      minResponseTime: Math.min(...times),
      maxResponseTime: Math.max(...times),
      p50ResponseTime: this.calculatePercentile(times, 50),
      p95ResponseTime: this.calculatePercentile(times, 95),
      p99ResponseTime: this.calculatePercentile(times, 99),
    };
  }

  // 打印报告
  printReport() {
    const metrics = this.getMetrics();
    const hitRate = ((metrics.cacheHits / metrics.totalRequests) * 100).toFixed(
      2
    );

    console.log("\n=== 缓存性能测试报告 ===");
    console.log(`总请求数: ${metrics.totalRequests}`);
    console.log(`缓存命中: ${metrics.cacheHits}`);
    console.log(`缓存未命中: ${metrics.cacheMisses}`);
    console.log(`命中率: ${hitRate}%`);
    console.log("\n响应时间统计:");
    console.log(`  平均: ${metrics.avgResponseTime.toFixed(2)}ms`);
    console.log(`  最小: ${metrics.minResponseTime.toFixed(2)}ms`);
    console.log(`  最大: ${metrics.maxResponseTime.toFixed(2)}ms`);
    console.log(`  P50: ${metrics.p50ResponseTime.toFixed(2)}ms`);
    console.log(`  P95: ${metrics.p95ResponseTime.toFixed(2)}ms`);
    console.log(`  P99: ${metrics.p99ResponseTime.toFixed(2)}ms`);
  }

  // 重置统计
  reset() {
    this.metrics = {
      responseTimes: [],
      cacheHits: 0,
      cacheMisses: 0,
    };
  }
}

export const perfTester = new PerformanceTester();
```

```typescript
// 测试单个请求
export async function testSingleRequest(
  fn: () => Promise<any>,
  checkCache: () => boolean
): Promise<{ time: number; cached: boolean }> {
  const start = Date.now();
  await fn();
  const time = Date.now() - start;
  const cached = checkCache();

  return { time, cached };
}

// 批量测试
export async function testBatch(
  fn: () => Promise<any>,
  checkCache: () => boolean,
  count: number
): Promise<void> {
  for (let i = 0; i < count; i++) {
    const { time, cached } = await testSingleRequest(fn, checkCache);
    perfTester.recordRequest(time, cached);
  }
}

// 并发测试
export async function testConcurrent(
  fn: () => Promise<any>,
  checkCache: () => boolean,
  concurrency: number
): Promise<void> {
  const promises = Array(concurrency)
    .fill(null)
    .map(() => testSingleRequest(fn, checkCache));

  const results = await Promise.all(promises);
  results.forEach(({ time, cached }) => {
    perfTester.recordRequest(time, cached);
  });
}
```

## 实战场景二: 缓存命中率分析

### 命中率监控系统

```typescript
// lib/monitoring/cache-monitor.ts

interface CacheHitStats {
  layer: string;
  hits: number;
  misses: number;
  hitRate: number;
  avgHitTime: number;
  avgMissTime: number;
}

class CacheMonitor {
  private stats = new Map<
    string,
    {
      hits: number;
      misses: number;
      hitTimes: number[];
      missTimes: number[];
    }
  >();

  // 记录缓存访问
  record(layer: string, hit: boolean, time: number) {
    if (!this.stats.has(layer)) {
      this.stats.set(layer, {
        hits: 0,
        misses: 0,
        hitTimes: [],
        missTimes: [],
      });
    }

    const stat = this.stats.get(layer)!;
    if (hit) {
      stat.hits++;
      stat.hitTimes.push(time);
    } else {
      stat.misses++;
      stat.missTimes.push(time);
    }
  }

  // 获取统计信息
  getStats(): CacheHitStats[] {
    return Array.from(this.stats.entries()).map(([layer, data]) => {
      const total = data.hits + data.misses;
      const avgHitTime =
        data.hitTimes.length > 0
          ? data.hitTimes.reduce((a, b) => a + b, 0) / data.hitTimes.length
          : 0;
      const avgMissTime =
        data.missTimes.length > 0
          ? data.missTimes.reduce((a, b) => a + b, 0) / data.missTimes.length
          : 0;

      return {
        layer,
        hits: data.hits,
        misses: data.misses,
        hitRate: total > 0 ? (data.hits / total) * 100 : 0,
        avgHitTime,
        avgMissTime,
      };
    });
  }

  // 打印报告
  printReport() {
    const stats = this.getStats();

    console.log("\n=== 缓存命中率报告 ===");
    stats.forEach((stat) => {
      console.log(`\n${stat.layer}:`);
      console.log(`  命中: ${stat.hits}`);
      console.log(`  未命中: ${stat.misses}`);
      console.log(`  命中率: ${stat.hitRate.toFixed(2)}%`);
      console.log(`  命中平均时间: ${stat.avgHitTime.toFixed(2)}ms`);
      console.log(`  未命中平均时间: ${stat.avgMissTime.toFixed(2)}ms`);
    });
  }

  // 重置统计
  reset() {
    this.stats.clear();
  }
}

export const cacheMonitor = new CacheMonitor();
```

```typescript
interface LoadTestConfig {
  duration: number; // 测试持续时间(秒)
  rps: number; // 每秒请求数
  concurrency: number; // 并发数
}

class LoadTester {
  private results: {
    successCount: number;
    errorCount: number;
    responseTimes: number[];
  } = {
    successCount: 0,
    errorCount: 0,
    responseTimes: [],
  };

  async run(fn: () => Promise<any>, config: LoadTestConfig) {
    const startTime = Date.now();
    const endTime = startTime + config.duration * 1000;

    console.log(`\n开始负载测试:`);
    console.log(`  持续时间: ${config.duration}秒`);
    console.log(`  目标RPS: ${config.rps}`);
    console.log(`  并发数: ${config.concurrency}`);

    // 计算每个请求的间隔
    const interval = 1000 / config.rps;

    while (Date.now() < endTime) {
      const batchStart = Date.now();

      // 发送一批并发请求
      const promises = Array(config.concurrency)
        .fill(null)
        .map(async () => {
          const reqStart = Date.now();
          try {
            await fn();
            const reqTime = Date.now() - reqStart;
            this.results.successCount++;
            this.results.responseTimes.push(reqTime);
          } catch (error) {
            this.results.errorCount++;
          }
        });

      await Promise.all(promises);

      // 控制请求速率
      const elapsed = Date.now() - batchStart;
      const waitTime = interval * config.concurrency - elapsed;
      if (waitTime > 0) {
        await new Promise((resolve) => setTimeout(resolve, waitTime));
      }
    }

    this.printResults();
  }

  private printResults() {
    const times = this.results.responseTimes;
    const total = this.results.successCount + this.results.errorCount;
    const avgTime = times.reduce((a, b) => a + b, 0) / times.length;
    const sorted = [...times].sort((a, b) => a - b);
    const p95 = sorted[Math.floor(sorted.length * 0.95)];
    const p99 = sorted[Math.floor(sorted.length * 0.99)];

    console.log("\n=== 负载测试结果 ===");
    console.log(`总请求数: ${total}`);
    console.log(`成功: ${this.results.successCount}`);
    console.log(`失败: ${this.results.errorCount}`);
    console.log(
      `成功率: ${((this.results.successCount / total) * 100).toFixed(2)}%`
    );
    console.log(`平均响应时间: ${avgTime.toFixed(2)}ms`);
    console.log(`P95响应时间: ${p95.toFixed(2)}ms`);
    console.log(`P99响应时间: ${p99.toFixed(2)}ms`);
  }
}
```

```typescript
// lib/monitoring/memory-monitor.ts

class MemoryMonitor {
  private snapshots: Array<{
    timestamp: number;
    heapUsed: number;
    heapTotal: number;
    external: number;
    cacheSize: number;
  }> = [];

  // 记录内存快照
  takeSnapshot(cacheSize: number) {
    const memUsage = process.memoryUsage();

    this.snapshots.push({
      timestamp: Date.now(),
      heapUsed: memUsage.heapUsed,
      heapTotal: memUsage.heapTotal,
      external: memUsage.external,
      cacheSize,
    });
  }

  // 获取内存增长
  getMemoryGrowth() {
    if (this.snapshots.length < 2) return null;

    const first = this.snapshots[0];
    const last = this.snapshots[this.snapshots.length - 1];

    return {
      heapGrowth: last.heapUsed - first.heapUsed,
      cacheGrowth: last.cacheSize - first.cacheSize,
      duration: last.timestamp - first.timestamp,
    };
  }

  // 打印报告
  printReport() {
    if (this.snapshots.length === 0) return;

    console.log("\n=== 内存使用报告 ===");

    const latest = this.snapshots[this.snapshots.length - 1];
    console.log(`当前堆内存: ${(latest.heapUsed / 1024 / 1024).toFixed(2)} MB`);
    console.log(
      `堆内存总量: ${(latest.heapTotal / 1024 / 1024).toFixed(2)} MB`
    );
    console.log(`缓存大小: ${(latest.cacheSize / 1024 / 1024).toFixed(2)} MB`);

    const growth = this.getMemoryGrowth();
    if (growth) {
      console.log(`\n内存增长:`);
      console.log(
        `  堆内存增长: ${(growth.heapGrowth / 1024 / 1024).toFixed(2)} MB`
      );
      console.log(
        `  缓存增长: ${(growth.cacheGrowth / 1024 / 1024).toFixed(2)} MB`
      );
      console.log(`  持续时间: ${(growth.duration / 1000).toFixed(2)}秒`);
    }
  }

  // 重置
  reset() {
    this.snapshots = [];
  }
}

export const memoryMonitor = new MemoryMonitor();
```

```typescript
// tests/benchmark/cache-benchmark.ts

interface BenchmarkResult {
  name: string;
  opsPerSecond: number;
  avgTime: number;
  minTime: number;
  maxTime: number;
}

class Benchmark {
  async run(
    name: string,
    fn: () => Promise<any>,
    duration: number = 5000
  ): Promise<BenchmarkResult> {
    const times: number[] = [];
    const startTime = Date.now();
    let operations = 0;

    while (Date.now() - startTime < duration) {
      const opStart = Date.now();
      await fn();
      times.push(Date.now() - opStart);
      operations++;
    }

    const totalTime = Date.now() - startTime;

    return {
      name,
      opsPerSecond: (operations / totalTime) * 1000,
      avgTime: times.reduce((a, b) => a + b, 0) / times.length,
      minTime: Math.min(...times),
      maxTime: Math.max(...times),
    };
  }

  printResults(results: BenchmarkResult[]) {
    console.log("\n=== 性能基准测试结果 ===");

    results.forEach((result) => {
      console.log(`\n${result.name}:`);
      console.log(`  操作/秒: ${result.opsPerSecond.toFixed(2)}`);
      console.log(`  平均时间: ${result.avgTime.toFixed(2)}ms`);
      console.log(`  最小时间: ${result.minTime.toFixed(2)}ms`);
      console.log(`  最大时间: ${result.maxTime.toFixed(2)}ms`);
    });

    // 对比最快和最慢
    const sorted = [...results].sort((a, b) => b.opsPerSecond - a.opsPerSecond);
    const fastest = sorted[0];
    const slowest = sorted[sorted.length - 1];

    console.log(`\n最快: ${fastest.name}`);
    console.log(`最慢: ${slowest.name}`);
    console.log(
      `性能差异: ${(fastest.opsPerSecond / slowest.opsPerSecond).toFixed(2)}x`
    );
  }
}
```

```typescript
class TrafficSimulator {
  private accessLog: Map<string, number> = new Map();

  // 加载真实访问日志
  loadAccessLog(log: Array<{ productId: string }>) {
    log.forEach((entry) => {
      const count = this.accessLog.get(entry.productId) || 0;
      this.accessLog.set(entry.productId, count + 1);
    });
  }

  // 根据访问频率生成测试请求
  generateRequests(count: number): string[] {
    const requests: string[] = [];
    const products = Array.from(this.accessLog.entries());
    const totalAccess = products.reduce((sum, [, count]) => sum + count, 0);

    for (let i = 0; i < count; i++) {
      const random = Math.random() * totalAccess;
      let cumulative = 0;

      for (const [productId, accessCount] of products) {
        cumulative += accessCount;
        if (random <= cumulative) {
          requests.push(productId);
          break;
        }
      }
    }

    return requests;
  }
}
```

### 10. 如何进行持续性能监控?

**问题**: 如何在生产环境持续监控缓存性能?

```typescript
class PerformanceMonitor {
  private metrics: Array<{
    timestamp: number;
    hitRate: number;
    avgTime: number;
    p95Time: number;
  }> = [];

  // 记录性能指标
  record(hitRate: number, avgTime: number, p95Time: number) {
    this.metrics.push({
      timestamp: Date.now(),
      hitRate,
      avgTime,
      p95Time,
    });

    // 只保留最近24小时的数据
    const oneDayAgo = Date.now() - 24 * 60 * 60 * 1000;
    this.metrics = this.metrics.filter((m) => m.timestamp > oneDayAgo);
  }

  // 检测性能下降
  detectDegradation() {
    if (this.metrics.length < 10) return null;

    const recent = this.metrics.slice(-5);
    const previous = this.metrics.slice(-10, -5);

    const recentAvg =
      recent.reduce((sum, m) => sum + m.avgTime, 0) / recent.length;
    const previousAvg =
      previous.reduce((sum, m) => sum + m.avgTime, 0) / previous.length;

    const degradation = ((recentAvg - previousAvg) / previousAvg) * 100;

    if (degradation > 20) {
      return {
        type: "performance_degradation",
        degradation: `${degradation.toFixed(2)}%`,
        recentAvg,
        previousAvg,
      };
    }

    return null;
  }

  // 生成报告
  generateReport() {
    const last24h = this.metrics;

    return {
      avgHitRate:
        last24h.reduce((sum, m) => sum + m.hitRate, 0) / last24h.length,
      avgResponseTime:
        last24h.reduce((sum, m) => sum + m.avgTime, 0) / last24h.length,
      maxP95: Math.max(...last24h.map((m) => m.p95Time)),
      dataPoints: last24h.length,
    };
  }
}
```
