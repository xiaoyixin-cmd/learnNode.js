# 多级缓存架构示例

## 概述

多级缓存架构是一种将缓存分为多个层次的设计模式,每一层都有不同的特性和用途。通过合理设计多级缓存,可以在性能、成本和数据新鲜度之间找到最佳平衡。

### 多级缓存的优势

1. **性能优化**: 不同层次的缓存提供不同的访问速度
2. **成本控制**: 根据访问频率分配缓存资源
3. **灵活性**: 可以针对不同数据类型使用不同策略
4. **可靠性**: 多层缓存提供冗余保护
5. **可扩展性**: 易于添加新的缓存层

### 常见的缓存层次

```
客户端缓存 (浏览器)
    ↓
CDN 缓存 (边缘节点)
    ↓
应用层缓存 (Next.js)
    ↓
数据库缓存 (Redis/Memcached)
    ↓
数据源 (数据库)
```

```typescript
// lib/cache/layers.ts

// 第一层: 内存缓存 (最快)
const memoryCache = new Map<string, { data: any; expires: number }>();

export function getFromMemory(key: string) {
  const cached = memoryCache.get(key);
  if (!cached) return null;

  if (Date.now() > cached.expires) {
    memoryCache.delete(key);
    return null;
  }

  return cached.data;
}

export function setToMemory(key: string, data: any, ttl: number = 60000) {
  memoryCache.set(key, {
    data,
    expires: Date.now() + ttl,
  });
}

// 第二层: Next.js 缓存 (快)
import { unstable_cache } from "next/cache";

export const getFromNextCache = unstable_cache(
  async (key: string) => {
    // 从数据源获取
    return await fetchFromSource(key);
  },
  ["next-cache"],
  {
    revalidate: 300, // 5分钟
  }
);

// 第三层: 数据源 (慢)
async function fetchFromSource(key: string) {
  // 从数据库或 API 获取数据
  return await db.query(key);
}

// 多层缓存获取函数
export async function getWithMultiLevelCache(key: string) {
  // 1. 尝试从内存缓存获取
  const memoryData = getFromMemory(key);
  if (memoryData) {
    console.log("Hit: Memory cache");
    return memoryData;
  }

  // 2. 尝试从 Next.js 缓存获取
  const nextData = await getFromNextCache(key);
  if (nextData) {
    console.log("Hit: Next.js cache");
    // 回填到内存缓存
    setToMemory(key, nextData);
    return nextData;
  }

  // 3. 从数据源获取
  console.log("Miss: Fetching from source");
  const sourceData = await fetchFromSource(key);

  // 回填到所有缓存层
  setToMemory(key, sourceData);

  return sourceData;
}
```

```typescript
// lib/cache/product.ts
import { unstable_cache } from "next/cache";
import { Redis } from "@upstash/redis";

const redis = new Redis({
  url: process.env.REDIS_URL!,
  token: process.env.REDIS_TOKEN!,
});

// 内存缓存层
const productMemoryCache = new Map<
  string,
  {
    data: any;
    expires: number;
  }
>();

// 获取产品 - 三层缓存
export async function getProduct(slug: string) {
  const cacheKey = `product:${slug}`;

  // 第一层: 内存缓存 (最快, TTL: 1分钟)
  const memoryData = productMemoryCache.get(cacheKey);
  if (memoryData && Date.now() < memoryData.expires) {
    console.log(`[Cache Hit] Memory: ${cacheKey}`);
    return memoryData.data;
  }

  // 第二层: Redis 缓存 (快, TTL: 10分钟)
  const redisData = await redis.get(cacheKey);
  if (redisData) {
    console.log(`[Cache Hit] Redis: ${cacheKey}`);
    // 回填到内存缓存
    productMemoryCache.set(cacheKey, {
      data: redisData,
      expires: Date.now() + 60000, // 1分钟
    });
    return redisData;
  }

  // 第三层: Next.js 缓存 + 数据库 (慢, TTL: 1小时)
  console.log(`[Cache Miss] Fetching from database: ${cacheKey}`);
  const dbData = await fetchProductFromDB(slug);

  if (dbData) {
    // 回填到所有缓存层
    productMemoryCache.set(cacheKey, {
      data: dbData,
      expires: Date.now() + 60000,
    });

    await redis.set(cacheKey, dbData, {
      ex: 600, // 10分钟
    });
  }

  return dbData;
}

// 从数据库获取产品

    const product = await db.product.findUnique({

        category: true,
        images: true,
        reviews: {
          take: 10,
          orderBy: { createdAt: "desc" },
        },
      },
    });
    return product;
  },
  ["product-db"],
  {
    tags: (slug: string) => ["products", `product-${slug}`],
```

### 文章内容缓存

```typescript
// lib/cache/post.ts

// 文章缓存管理器
class PostCacheManager {
  private memoryCache = new Map<string, { data: any; expires: number }>();
  private readonly MEMORY_TTL = 60000; // 1分钟
  private readonly REDIS_TTL = 600; // 10分钟
  private readonly NEXT_TTL = 3600; // 1小时

  // 获取文章
  async getPost(slug: string) {
    const cacheKey = `post:${slug}`;

    // 1. 内存缓存
    const memoryData = this.getFromMemory(cacheKey);
    if (memoryData) {
      console.log(`[Cache Hit] Memory: ${cacheKey}`);
      return memoryData;
    }

    // 2. Redis 缓存
    const redisData = await this.getFromRedis(cacheKey);
    if (redisData) {
      console.log(`[Cache Hit] Redis: ${cacheKey}`);
      this.setToMemory(cacheKey, redisData);
      return redisData;
    }

    // 3. Next.js 缓存 + 数据库
    console.log(`[Cache Miss] Fetching from database: ${cacheKey}`);
    const dbData = await this.getFromDB(slug);

    if (dbData) {
      this.setToMemory(cacheKey, dbData);
      await this.setToRedis(cacheKey, dbData);
    }

    return dbData;
  }

  // 内存缓存操作
  private getFromMemory(key: string) {
    const cached = this.memoryCache.get(key);
    if (!cached) return null;

    if (Date.now() > cached.expires) {
      this.memoryCache.delete(key);
      return null;
    }

    return cached.data;
  }

  private setToMemory(key: string, data: any) {
    this.memoryCache.set(key, {
      data,
      expires: Date.now() + this.MEMORY_TTL,
    });
  }

  // Redis 缓存操作
  private async getFromRedis(key: string) {
    try {
      return await redis.get(key);
    } catch (error) {
      console.error("Redis get error:", error);
      return null;
    }
  }

  private async setToRedis(key: string, data: any) {
    try {
      await redis.set(key, data, { ex: this.REDIS_TTL });
    } catch (error) {
      console.error("Redis set error:", error);
    }
  }

  // 数据库操作
  private getFromDB = unstable_cache(
    async (slug: string) => {
      return await db.post.findUnique({
        where: { slug },
        include: {
          author: {
            select: {
              name: true,
              avatar: true,
            },
          },
          category: true,
          tags: true,
        },
      });
    },
    ["post-db"],
    {
      tags: (slug: string) => ["posts", `post-${slug}`],
      revalidate: this.NEXT_TTL,
    }
  );

  // 清除缓存
  async clearCache(slug: string) {
    const cacheKey = `post:${slug}`;

    // 清除内存缓存
    this.memoryCache.delete(cacheKey);

    // 清除 Redis 缓存
    await redis.del(cacheKey);

    // 清除 Next.js 缓存
    revalidateTag(`post-${slug}`);
  }

  // 预热缓存
  async warmupCache(slugs: string[]) {
    for (const slug of slugs) {
      await this.getPost(slug);
    }
  }
}

export const postCache = new PostCacheManager();
```

## 缓存层性能对比

### 各层缓存的特性

| 缓存层   | 访问速度      | 容量         | TTL            | 成本 | 适用场景   |
| -------- | ------------- | ------------ | -------------- | ---- | ---------- |
| 内存缓存 | 极快 (< 1ms)  | 小 (MB 级)   | 短 (秒-分钟)   | 低   | 热点数据   |
| Redis    | 快 (1-5ms)    | 中 (GB 级)   | 中 (分钟-小时) | 中   | 常用数据   |

| CDN      | 快 (10-100ms) | 极大 (TB 级) | 长 (天-周)     | 高   | 全球分发   |
| 数据库   | 慢 (50-500ms) | 极大 (TB 级) | 永久           | 高   | 持久化存储 |

```typescript
class CacheStats {
  private stats = {
    memory: { hits: 0, misses: 0 },
    redis: { hits: 0, misses: 0 },
    next: { hits: 0, misses: 0 },
  };

  recordHit(layer: "memory" | "redis" | "next") {
    this.stats[layer].hits++;
  }

  recordMiss(layer: "memory" | "redis" | "next") {
    this.stats[layer].misses++;
  }

  getStats() {
    return Object.entries(this.stats).map(([layer, data]) => ({
      layer,
      hits: data.hits,
      misses: data.misses,
      total: data.hits + data.misses,
      hitRate: data.hits / (data.hits + data.misses) || 0,
    }));
  }

  reset() {
    this.stats = {
      memory: { hits: 0, misses: 0 },
      redis: { hits: 0, misses: 0 },
      next: { hits: 0, misses: 0 },
    };
  }
}
```

### 4. 缓存击穿

```typescript
// 使用互斥锁
const locks = new Map<string, Promise<any>>();

export async function getDataWithLock(id: string) {
  const cacheKey = `data:${id}`;

  // 检查缓存
  const cached = await redis.get(cacheKey);
  if (cached) return cached;

  // 检查是否有其他请求正在获取
  if (locks.has(cacheKey)) {
    return await locks.get(cacheKey);
  }

  // 创建锁
  const promise = (async () => {
    const data = await db.findUnique({ where: { id } });
    await redis.set(cacheKey, data, { ex: 600 });
    locks.delete(cacheKey);
    return data;
  })();

  locks.set(cacheKey, promise);
  return await promise;
}
```

### 5. 内存管理

内存缓存需要限制大小:

```typescript
class LRUCache<K, V> {
  private cache = new Map<K, V>();
  private maxSize: number;

  constructor(maxSize: number) {
    this.maxSize = maxSize;
  }

  get(key: K): V | undefined {
    if (!this.cache.has(key)) return undefined;

    // 移到最后(最近使用)
    const value = this.cache.get(key)!;
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  set(key: K, value: V) {
    // 如果已存在,先删除
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }

    // 如果超过大小限制,删除最旧的
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    this.cache.set(key, value);
  }
}
```

## 常见问题

### 1. 如何选择缓存层数?

**问题**: 应该使用几层缓存?

**解答**:

根据业务需求选择:

- 小型应用: 1-2 层(Next.js + 数据库)
- 中型应用: 2-3 层(内存 + Next.js + 数据库)
- 大型应用: 3-4 层(内存 + Redis + Next.js + 数据库)
- 全球应用: 4-5 层(内存 + Redis + Next.js + CDN + 数据库)

### 2. 如何设置各层 TTL?

### 7. 缓存大小如何控制?

**问题**: 如何防止缓存占用过多内存?

**解答**:

使用 LRU 策略:

```typescript
class SizedCache {
  private cache = new Map();
  private maxSize = 100 * 1024 * 1024; // 100MB
  private currentSize = 0;

  set(key: string, value: any) {
    const size = JSON.stringify(value).length;

    // 如果超过限制,清理旧数据
    while (this.currentSize + size > this.maxSize && this.cache.size > 0) {
      const firstKey = this.cache.keys().next().value;
      const firstValue = this.cache.get(firstKey);
      this.currentSize -= JSON.stringify(firstValue).length;
      this.cache.delete(firstKey);
    }

    this.cache.set(key, value);
    this.currentSize += size;
  }
}
```

```typescript
class CacheMonitor {
  private hits = 0;
  private misses = 0;
  private errors = 0;

  recordHit() {
    this.hits++;
  }

  recordMiss() {
    this.misses++;
  }

  recordError() {
    this.errors++;
  }

  getStats() {
    const total = this.hits + this.misses;
    const hitRate = total > 0 ? (this.hits / total) * 100 : 0;

    return {
      hits: this.hits,
      misses: this.misses,
      errors: this.errors,
      total,
      hitRate: hitRate.toFixed(2) + "%",
    };
  }

  reset() {
    this.hits = 0;
    this.misses = 0;
    this.errors = 0;
  }
}

const monitor = new CacheMonitor();

export async function getDataWithMonitoring(id: string) {
  try {
    const cached = await redis.get(`data:${id}`);
    if (cached) {
      monitor.recordHit();
      return JSON.parse(cached);

    monitor.recordMiss();
    const data = await db.findUnique({ where: { id } });
    await redis.set(`data:${id}`, JSON.stringify(data), "EX", 600);
    return data;
  } catch (error) {
    monitor.recordError();
    throw error;
  }
}
```

### 实战场景八: 智能缓存淘汰

#### LRU 缓存实现

```typescript
class LRUCache<T> {
  private cache = new Map<string, T>();
  private maxSize: number;

  constructor(maxSize: number = 1000) {
    this.maxSize = maxSize;
  }

  get(key: string): T | undefined {
    if (!this.cache.has(key)) return undefined;

    // 移到最后(最近使用)
    const value = this.cache.get(key)!;
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  set(key: string, value: T): void {
    // 如果已存在,先删除
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }

    // 如果超过容量,删除最旧的
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    this.cache.set(key, value);
  }

  delete(key: string): boolean {
    return this.cache.delete(key);
  }

  clear(): void {
    this.cache.clear();
  }

  size(): number {
    return this.cache.size;
  }
}

const lruCache = new LRUCache<any>(1000);

export async function getDataWithLRU(id: string) {
  // 1. 检查LRU缓存
  const cached = lruCache.get(id);
  if (cached) return cached;

  // 2. 从数据库获取
  const data = await db.findUnique({ where: { id } });

  // 3. 存入LRU缓存
  lruCache.set(id, data);

  return data;
}
```
