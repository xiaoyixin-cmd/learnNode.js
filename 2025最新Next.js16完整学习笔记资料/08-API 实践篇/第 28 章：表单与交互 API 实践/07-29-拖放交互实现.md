# 拖放交互实现

拖放(Drag and Drop)是现代 Web 应用中常见的交互方式。本文介绍如何在 Next.js 中实现各种拖放功能。

## 核心概念

### 拖放 API

| API          | 说明       | 用途           |
| ------------ | ---------- | -------------- |
| draggable    | 元素可拖拽 | 标记可拖拽元素 |
| onDragStart  | 开始拖拽   | 记录拖拽数据   |
| onDragOver   | 拖拽经过   | 允许放置       |
| onDrop       | 放置元素   | 处理放置逻辑   |
| dataTransfer | 数据传输   | 传递拖拽数据   |

## 实战场景一: 基础拖放

### 简单拖放

```typescript
"use client";

import { useState } from "react";

export default function BasicDragDrop() {
  const [draggedItem, setDraggedItem] = useState<string | null>(null);
  const [droppedItems, setDroppedItems] = useState<string[]>([]);

  function handleDragStart(e: React.DragEvent, item: string) {
    setDraggedItem(item);
    e.dataTransfer.effectAllowed = "move";
  }

  function handleDragOver(e: React.DragEvent) {
    e.preventDefault();
    e.dataTransfer.dropEffect = "move";
  }

  function handleDrop(e: React.DragEvent) {
    e.preventDefault();
    if (draggedItem) {
      setDroppedItems([...droppedItems, draggedItem]);
      setDraggedItem(null);
    }
  }

  const items = ["项目1", "项目2", "项目3"];

  return (
    <div style={{ display: "flex", gap: "20px" }}>
      <div>
        <h3>可拖拽项</h3>
        {items.map((item) => (
          <div
            key={item}
            draggable
            onDragStart={(e) => handleDragStart(e, item)}
            style={{
              padding: "10px",
              margin: "5px",
              backgroundColor: "#f0f0f0",
              cursor: "move",
            }}
          >
            {item}
          </div>
        ))}
      </div>

      <div
        onDragOver={handleDragOver}
        onDrop={handleDrop}
        style={{
          width: "200px",
          minHeight: "200px",
          border: "2px dashed gray",
          padding: "10px",
        }}
      >
        <h3>放置区域</h3>
        {droppedItems.map((item, index) => (
          <div
            key={index}
            style={{ padding: "5px", backgroundColor: "#e0e0e0" }}
          >
            {item}
          </div>
        ))}
      </div>
    </div>
  );
}
```

```typescript
import { useState } from "react";

export default function FileDragDrop() {
  const [files, setFiles] = useState<File[]>([]);
  const [isDragging, setIsDragging] = useState(false);

  function handleDragEnter(e: React.DragEvent) {
    e.preventDefault();
    setIsDragging(true);
  }

  function handleDragLeave(e: React.DragEvent) {
    e.preventDefault();
    setIsDragging(false);
  }

  function handleDragOver(e: React.DragEvent) {
    e.preventDefault();
  }

  function handleDrop(e: React.DragEvent) {
    e.preventDefault();
    setIsDragging(false);

    const droppedFiles = Array.from(e.dataTransfer.files);

  }

  function removeFile(index: number) {
    setFiles(files.filter((_, i) => i !== index));
  }

  return (
    <div>
      <div
        onDragEnter={handleDragEnter}
        onDragLeave={handleDragLeave}
        onDragOver={handleDragOver}
        onDrop={handleDrop}
        style={{
          border: `2px dashed ${isDragging ? "blue" : "gray"}`,
          padding: "40px",
          textAlign: "center",
          backgroundColor: isDragging ? "#f0f0f0" : "white",
          borderRadius: "5px",
        }}
      >
        <p>拖拽文件到这里</p>
      </div>

        <div style={{ marginTop: "20px" }}>

              </li>
            ))}
          </ul>
```

## 实战场景五: 树形结构拖放

### 可拖放树

```typescript
"use client";

import { useState } from "react";

interface TreeNode {
  id: string;
  name: string;
  children?: TreeNode[];
}

export default function DraggableTree() {
  const [tree, setTree] = useState<TreeNode[]>([
    {
      id: "1",
      name: "文件夹1",
      children: [
        { id: "1-1", name: "文件1-1" },
        { id: "1-2", name: "文件1-2" },
      ],
    },
    {
      id: "2",
      name: "文件夹2",
      children: [{ id: "2-1", name: "文件2-1" }],
    },
  ]);

  const [draggedNode, setDraggedNode] = useState<TreeNode | null>(null);

  function handleDragStart(node: TreeNode) {
    setDraggedNode(node);
  }

  function handleDrop(targetNode: TreeNode) {
    if (!draggedNode || draggedNode.id === targetNode.id) return;

    // 移除拖拽节点
    const removeNode = (nodes: TreeNode[]): TreeNode[] => {
      return nodes.filter((node) => {
        if (node.id === draggedNode.id) return false;
        if (node.children) {
          node.children = removeNode(node.children);
        }
        return true;
      });
    };

    // 添加到目标节点
    const addNode = (nodes: TreeNode[]): TreeNode[] => {
      return nodes.map((node) => {
        if (node.id === targetNode.id) {
          return {
            ...node,
            children: [...(node.children || []), draggedNode],
          };
        }
        if (node.children) {
          return { ...node, children: addNode(node.children) };
        }
        return node;
      });
    };

    let newTree = removeNode(tree);
    newTree = addNode(newTree);
    setTree(newTree);
    setDraggedNode(null);
  }

  function renderTree(nodes: TreeNode[], level = 0) {
    return nodes.map((node) => (
      <div key={node.id} style={{ marginLeft: `${level * 20}px` }}>
        <div
          draggable
          onDragStart={() => handleDragStart(node)}
          onDragOver={(e) => e.preventDefault()}
          onDrop={() => handleDrop(node)}
          style={{
            padding: "5px",
            margin: "2px 0",
            backgroundColor: "#f0f0f0",
            cursor: "move",
          }}
        >
          {node.name}
        </div>
        {node.children && renderTree(node.children, level + 1)}
      </div>
    ));
  }

  return <div>{renderTree(tree)}</div>;
}
```

### 6. 如何实现拖拽撤销?

使用历史记录:

```typescript
"use client";

import { useState } from "react";

export default function UndoableDrag() {
  const [items, setItems] = useState(["A", "B", "C"]);
  const [history, setHistory] = useState<string[][]>([["A", "B", "C"]]);
  const [historyIndex, setHistoryIndex] = useState(0);

  function updateItems(newItems: string[]) {
    const newHistory = history.slice(0, historyIndex + 1);
    newHistory.push(newItems);
    setHistory(newHistory);
    setHistoryIndex(newHistory.length - 1);
    setItems(newItems);
  }

  function undo() {
    if (historyIndex > 0) {
      setHistoryIndex(historyIndex - 1);
      setItems(history[historyIndex - 1]);
    }
  }

  function redo() {
    if (historyIndex < history.length - 1) {
      setHistoryIndex(historyIndex + 1);
      setItems(history[historyIndex + 1]);
    }
  }

  return (
    <div>
      <button onClick={undo} disabled={historyIndex === 0}>
        撤销
      </button>
      <button onClick={redo} disabled={historyIndex === history.length - 1}>
        重做
      </button>
    </div>
  );
}
```

```typescript
import { useRef } from "react";

export default function AutoScrollDrag() {
  const containerRef = useRef<HTMLDivElement>(null);

  function handleDragOver(e: React.DragEvent) {
    e.preventDefault();

    const container = containerRef.current;
    if (!container) return;

    const rect = container.getBoundingClientRect();
    const scrollThreshold = 50;

    if (e.clientY < rect.top + scrollThreshold) {
      container.scrollTop -= 10;
    } else if (e.clientY > rect.bottom - scrollThreshold) {
      container.scrollTop += 10;
    }
  }

  return (
    <div
      ref={containerRef}
      onDragOver={handleDragOver}
      style={{ height: "300px", overflow: "auto" }}
    >
      {/* 内容 */}
    </div>
  );
```
