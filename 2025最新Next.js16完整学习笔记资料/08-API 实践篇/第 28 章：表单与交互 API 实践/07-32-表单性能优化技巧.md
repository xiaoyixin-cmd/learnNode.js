## 实战场景一: 防抖和节流

### 优化输入处理

```typescript
"use client";

import { useState, useCallback } from "react";

function debounce<T extends (...args: any[]) => any>(
  func: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timeoutId: NodeJS.Timeout;
  return (...args: Parameters<T>) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func(...args), delay);
  };
}

function throttle<T extends (...args: any[]) => any>(
  func: T,
  limit: number
): (...args: Parameters<T>) => void {
  let inThrottle: boolean;
  return (...args: Parameters<T>) => {
    if (!inThrottle) {
      func(...args);
      inThrottle = true;
      setTimeout(() => (inThrottle = false), limit);
    }
  };
}

export default function DebounceThrottle() {
  const [searchTerm, setSearchTerm] = useState("");
  const [results, setResults] = useState<string[]>([]);

  const debouncedSearch = useCallback(
    debounce((term: string) => {
      console.log("搜索:", term);
      // 模拟API调用
      setResults([`结果1: ${term}`, `结果2: ${term}`]);
    }, 500),
    []
  );

  function handleChange(value: string) {
    setSearchTerm(value);
    debouncedSearch(value);
  }

  return (
    <div>
      <input
        type="text"
        value={searchTerm}
        onChange={(e) => handleChange(e.target.value)}
        placeholder="搜索..."
      />
      <ul>
        {results.map((result, index) => (
          <li key={index}>{result}</li>
        ))}
      </ul>
    </div>
  );
}
```

## 实战场景四: useMemo 优化

### 缓存复杂计算

```typescript
"use client";

import { useState, useMemo } from "react";

export default function MemoOptimization() {
  const [items, setItems] = useState([
    { id: 1, name: "商品1", price: 100, quantity: 2 },
    { id: 2, name: "商品2", price: 200, quantity: 1 },
    { id: 3, name: "商品3", price: 150, quantity: 3 },
  ]);

  const total = useMemo(() => {
    console.log("计算总价");
    return items.reduce((sum, item) => sum + item.price * item.quantity, 0);
  }, [items]);

  const tax = useMemo(() => {
    console.log("计算税费");
    return total * 0.1;
  }, [total]);

  const finalTotal = useMemo(() => {
    console.log("计算最终总价");
    return total + tax;
  }, [total, tax]);

  function updateQuantity(id: number, quantity: number) {
    setItems(
      items.map((item) => (item.id === id ? { ...item, quantity } : item))
    );
  }

  return (
    <div>
      {items.map((item) => (
        <div key={item.id}>
          <span>{item.name}</span>
          <input
            type="number"
            value={item.quantity}
            onChange={(e) => updateQuantity(item.id, Number(e.target.value))}
          />
          <span>小计: {item.price * item.quantity}</span>
        </div>
      ))}
      <div>总价: {total}</div>
      <div>税费: {tax}</div>
      <div>最终总价: {finalTotal}</div>
    </div>
  );
}
```

import { useState, useRef, useEffect } from "react";

const ITEM_HEIGHT = 40;
const VISIBLE_ITEMS = 10;

export default function VirtualScroll() {
  const [items] = useState(() =>
    Array.from({ length: 10000 }, (_, i) => ({ id: i, name: `选项 ${i}` }))
  );
  const [scrollTop, setScrollTop] = useState(0);
  const containerRef = useRef<HTMLDivElement>(null);

  const startIndex = Math.floor(scrollTop / ITEM_HEIGHT);
  const endIndex = Math.min(startIndex + VISIBLE_ITEMS, items.length);
  const visibleItems = items.slice(startIndex, endIndex);

  const totalHeight = items.length * ITEM_HEIGHT;
  const offsetY = startIndex * ITEM_HEIGHT;

  function handleScroll(e: React.UIEvent<HTMLDivElement>) {
    setScrollTop(e.currentTarget.scrollTop);
  }

  return (
    <div
      ref={containerRef}
      onScroll={handleScroll}
      style={{
        height: `${VISIBLE_ITEMS * ITEM_HEIGHT}px`,
        overflow: "auto",
        position: "relative",
      }}
    >
      <div style={{ height: `${totalHeight}px` }}>
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {visibleItems.map((item) => (
            <div
              key={item.id}
              style={{
                height: `${ITEM_HEIGHT}px`,
                display: "flex",
                alignItems: "center",
                padding: "0 10px",
                borderBottom: "1px solid #ccc",

            </div>
          ))}

## 注意事项

### 1. 防抖延迟要合理

根据场景选择延迟时间:

```typescript
// 搜索: 300-500ms
const debouncedSearch = debounce(search, 300);

// 自动保存: 1000-2000ms
const debouncedSave = debounce(save, 1000);

// 验证: 500ms
const debouncedValidate = debounce(validate, 500);
```

### 2. useMemo 不要过度使用

只在计算昂贵时使用:

```typescript
// 不好 - 简单计算不需要memo
const sum = useMemo(() => a + b, [a, b]);

// 好 - 复杂计算需要memo
const expensiveResult = useMemo(() => {
  return items.reduce((acc, item) => {
    // 复杂计算
    return acc + complexCalculation(item);
  }, 0);
}, [items]);
```

### 3. useCallback 配合 memo 使用

单独使用 useCallback 没有意义:

```typescript
// 子组件需要用memo包裹
const ChildComponent = memo(({ onClick }) => {
  return <button onClick={onClick}>点击</button>;
});

// 父组件使用useCallback
function ParentComponent() {
  const handleClick = useCallback(() => {
    console.log("点击");
  }, []);

  return <ChildComponent onClick={handleClick} />;
}
```

### 4. 虚拟滚动要考虑动态高度

固定高度最简单:

```typescript
// 固定高度
const ITEM_HEIGHT = 40;

// 动态高度需要额外处理
const itemHeights = items.map((item) => calculateHeight(item));
const totalHeight = itemHeights.reduce((sum, h) => sum + h, 0);
```

### 5. 表单验证时机要权衡

不同时机的性能影响:

```typescript
// onChange - 实时验证,性能开销大
mode: "onChange";

// onBlur - 失焦验证,性能好
mode: "onBlur";

// onSubmit - 提交验证,性能最好
mode: "onSubmit";
```

## 总结

表单性能优化是提升用户体验的关键。通过本文的学习,我们掌握了:

### 核心技术

1. **防抖节流**: 减少函数执行次数
2. **useMemo**: 缓存复杂计算结果
3. **useCallback**: 缓存函数引用
4. **虚拟滚动**: 只渲染可见项

### 最佳实践

1. **合理选择验证时机**: onBlur vs onChange
2. **状态分离**: 避免不必要的重渲染
3. **代码分割**: 按需加载大型组件
4. **性能监控**: 使用 Profiler 和 DevTools

### 实战技巧

1. **懒加载验证**: 只在需要时验证
2. **表单序列化**: 使用 FormData 高效收集数据
3. **受控 vs 非受控**: 根据场景选择
4. **乐观更新**: 提升用户体验

通过合理使用这些优化技术,可以构建高性能的表单系统,提供流畅的用户体验。记住,性能优化要基于实际测量,不要过早优化。

"use client";

import { Component, ReactNode } from "react";

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
  error: Error | null;
}

class FormErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: any) {
    console.error("表单错误:", error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return (
        <div style={{ color: "red" }}>
          <h2>表单出错了</h2>
          <p>{this.state.error?.message}</p>
          <button
            onClick={() => this.setState({ hasError: false, error: null })}
          >
            重试
          </button>
        </div>
      );
    }

    return this.props.children;
  }
}

export default function FormWithErrorBoundary() {
  return (
    <FormErrorBoundary>
      <form>
        <input name="name" placeholder="姓名" />
        <button type="submit">提交</button>
      </form>
