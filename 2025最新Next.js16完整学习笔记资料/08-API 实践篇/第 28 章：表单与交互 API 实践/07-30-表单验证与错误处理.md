# 表单验证与错误处理

表单验证是 Web 应用的基础功能。本文介绍 Next.js 中的各种验证方案和错误处理策略。

## 核心概念

### 验证方式对比

| 方式        | 时机     | 优点       | 缺点     |
| ----------- | -------- | ---------- | -------- |
| 客户端验证  | 提交前   | 即时反馈   | 可被绕过 |
| 服务端验证  | 提交后   | 安全可靠   | 反馈延迟 |
| Schema 验证 | 两端共用 | 类型安全   | 学习成本 |
| 实时验证    | 输入时   | 用户体验好 | 性能开销 |

## 实战场景一: 基础客户端验证

### HTML5 验证

```typescript
"use client";

import { useState } from "react";

export default function BasicValidation() {
  const [errors, setErrors] = useState<Record<string, string>>({});

  function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();

    const formData = new FormData(e.currentTarget);
    const email = formData.get("email") as string;
    const password = formData.get("password") as string;

    const newErrors: Record<string, string> = {};

    if (!email) {
      newErrors.email = "邮箱不能为空";
    } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
      newErrors.email = "邮箱格式不正确";
    }

    if (!password) {
      newErrors.password = "密码不能为空";
    } else if (password.length < 6) {
      newErrors.password = "密码至少6个字符";
    }

    if (Object.keys(newErrors).length > 0) {
      setErrors(newErrors);
      return;
    }

    // 提交表单
    console.log("提交成功");
  }

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>邮箱:</label>
        <input type="email" name="email" />
        {errors.email && <span style={{ color: "red" }}>{errors.email}</span>}
      </div>

      <div>
        <label>密码:</label>
        <input type="password" name="password" />
        {errors.password && (
          <span style={{ color: "red" }}>{errors.password}</span>
        )}
      </div>

      <button type="submit">提交</button>
    </form>
  );
}
```

## 实战场景二: Zod Schema 验证

### 类型安全验证

```typescript
"use client";

import { useState } from "react";
import { z } from "zod";

const userSchema = z.object({
  email: z.string().email("邮箱格式不正确"),
  password: z.string().min(6, "密码至少6个字符"),
  age: z.number().min(18, "必须年满18岁"),
});

type UserForm = z.infer<typeof userSchema>;

export default function ZodValidation() {
  const [errors, setErrors] = useState<Record<string, string>>({});

  function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();

    const formData = new FormData(e.currentTarget);
    const data = {
      email: formData.get("email") as string,
      password: formData.get("password") as string,
      age: Number(formData.get("age")),
    };

    const result = userSchema.safeParse(data);

    if (!result.success) {
      const newErrors: Record<string, string> = {};
      result.error.errors.forEach((err) => {
        if (err.path[0]) {
          newErrors[err.path[0].toString()] = err.message;
        }
      });
      setErrors(newErrors);
      return;
    }

    console.log("验证通过:", result.data);
  }

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <input type="email" name="email" placeholder="邮箱" />
        {errors.email && <span>{errors.email}</span>}
      </div>

      <div>
        <input type="password" name="password" placeholder="密码" />
        {errors.password && <span>{errors.password}</span>}
      </div>

      <div>
        <input type="number" name="age" placeholder="年龄" />
        {errors.age && <span>{errors.age}</span>}
      </div>

      <button type="submit">提交</button>
    </form>
  );
}
```

```typescript
"use client";

import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";

const schema = z.object({

export default function HookFormValidation() {
```

## 实战场景六: 防抖验证

### 延迟验证

```typescript
"use client";

import { useState, useEffect } from "react";
import { z } from "zod";

const usernameSchema = z.string().min(3);

export default function DebouncedValidation() {
  const [username, setUsername] = useState("");
  const [error, setError] = useState("");
  const [checking, setChecking] = useState(false);

  useEffect(() => {
    if (!username) {
      setError("");
      return;
    }

    setChecking(true);

    const timer = setTimeout(() => {
      const result = usernameSchema.safeParse(username);
      if (!result.success) {
        setError("用户名至少3个字符");
      } else {
        setError("");
      }
      setChecking(false);
    }, 500);

    return () => clearTimeout(timer);
  }, [username]);

  return (
    <div>
      <input
        type="text"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
        placeholder="用户名"
      />
      {checking && <span>检查中...</span>}
      {error && <span style={{ color: "red" }}>{error}</span>}
    </div>
  );
}
```

## 实战场景七: 异步验证

### 检查用户名可用性

```typescript
"use client";

import { useState } from "react";

async function checkUsername(username: string): Promise<boolean> {
  // 模拟API调用
  await new Promise((resolve) => setTimeout(resolve, 1000));
  return username !== "admin";
}

export default function AsyncValidation() {
  const [username, setUsername] = useState("");
  const [error, setError] = useState("");
  const [checking, setChecking] = useState(false);

  async function handleBlur() {
    if (!username) return;

    setChecking(true);
    const available = await checkUsername(username);
    setChecking(false);

    if (!available) {
      setError("用户名已被占用");
    } else {
      setError("");
    }
  }

  return (
    <div>
      <input
        type="text"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
        onBlur={handleBlur}
        placeholder="用户名"
      />
      {checking && <span>检查中...</span>}
      {error && <span style={{ color: "red" }}>{error}</span>}
      {!checking && !error && username && (
        <span style={{ color: "green" }}>用户名可用</span>
      )}
    </div>
  );
}
```

## 实战场景八: 自定义验证器

### 复杂验证规则

```typescript
"use client";

import { useState } from "react";
import { z } from "zod";

const passwordSchema = z
  .string()
  .min(8, "密码至少8个字符")
  .regex(/[A-Z]/, "必须包含大写字母")
  .regex(/[a-z]/, "必须包含小写字母")
  .regex(/[0-9]/, "必须包含数字")
  .regex(/[^A-Za-z0-9]/, "必须包含特殊字符");

export default function CustomValidation() {
  const [password, setPassword] = useState("");
  const [errors, setErrors] = useState<string[]>([]);

  function handleChange(value: string) {
    setPassword(value);

    const result = passwordSchema.safeParse(value);
    if (!result.success) {
      setErrors(result.error.errors.map((e) => e.message));
    } else {
      setErrors([]);
    }
  }

  return (
    <div>
      <input
        type="password"
        value={password}
        onChange={(e) => handleChange(e.target.value)}
        placeholder="密码"
      />
      {errors.length > 0 && (
        <ul style={{ color: "red" }}>
          {errors.map((error, index) => (
            <li key={index}>{error}</li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

```typescript
import { useState } from "react";

const schema = z
```

## 实战场景十: 表单级错误

### 全局错误处理

```typescript
"use client";

import { useState } from "react";
import { useForm } from "react-hook-form";

interface FormData {
  username: string;
  email: string;
}

export default function FormLevelErrors() {
  const {
    register,
    handleSubmit,
    setError,
    formState: { errors },
  } = useForm<FormData>();
  const [globalError, setGlobalError] = useState("");

  async function onSubmit(data: FormData) {
    try {
      // 模拟API调用
      const response = await fetch("/api/register", {
        method: "POST",
        body: JSON.stringify(data),
      });

      if (!response.ok) {
        const error = await response.json();

        if (error.field) {
          // 字段级错误
          setError(error.field, { message: error.message });
        } else {
          // 表单级错误
          setGlobalError(error.message);
        }
      }
    } catch (error) {
      setGlobalError("网络错误,请稍后重试");
    }
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      {globalError && (
        <div style={{ color: "red", marginBottom: "10px" }}>{globalError}</div>
      )}

      <div>
        <input {...register("username")} placeholder="用户名" />
        {errors.username && <span>{errors.username.message}</span>}
      </div>

      <div>
        <input {...register("email")} placeholder="邮箱" />
        {errors.email && <span>{errors.email.message}</span>}
      </div>

      <button type="submit">提交</button>
    </form>
  );
}
```

## 注意事项

### 1. 客户端验证不可靠

必须配合服务端验证:

```typescript
"use server";

export async function submitForm(formData: FormData) {
  // 服务端必须再次验证
  const result = schema.safeParse(data);
  if (!result.success) {
    return { errors: result.error.flatten() };
  }
}
```

### 2. 防抖避免频繁验证

使用防抖减少验证次数:

```typescript
import { useDebouncedCallback } from "use-debounce";

const validate = useDebouncedCallback((value) => {
  // 验证逻辑
}, 500);
```

### 3. 错误信息要清晰

提供具体的错误提示:

```typescript
// 不好
z.string().min(6, "太短");

// 好
z.string().min(6, "密码至少需要6个字符");
```

### 4. 异步验证要有加载状态

显示验证进度:

```typescript
{
  checking && <span>检查中...</span>;
}
{
  !checking && error && <span>{error}</span>;
}
```

### 5. 表单提交时禁用按钮

防止重复提交:

```typescript
const [submitting, setSubmitting] = useState(false)

<button type="submit" disabled={submitting}>
  {submitting ? '提交中...' : '提交'}
</button>
```

## 总结

表单验证是 Web 应用的核心功能。通过本文的学习,我们掌握了:

### 核心技术

1. **客户端验证**: HTML5、自定义验证器
2. **Schema 验证**: Zod、Yup 类型安全验证
3. **React Hook Form**: 高性能表单库
4. **服务端验证**: Server Actions 验证

### 最佳实践

1. **双重验证**: 客户端+服务端
2. **防抖优化**: 减少验证次数
3. **清晰提示**: 具体的错误信息
4. **加载状态**: 异步验证进度

### 实战技巧

1. **实时验证**: 输入时即时反馈
2. **异步验证**: 检查用户名可用性
3. **字段依赖**: 密码确认验证
4. **表单级错误**: 全局错误处理

通过合理使用这些技术,可以构建用户体验优秀、安全可靠的表单验证系统。

```typescript
import { useState } from "react";
```

```typescript
const emailSchema = z.string().email("邮箱格式不正确");
```

```typescript
interface FormData {

    formState: { errors },
  } = useForm<FormData>();

  const [errorMessage, setErrorMessage] = useState("");

      const response = await fetch("/api/register", {

        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(data),
      });

      setSubmitState("success");

      setSubmitState("error");
      setErrorMessage(error instanceof Error ? error.message : "未知错误");

  }

    <div>
      {submitState === "success" ? (
        <div style={{ color: "green" }}>注册成功!</div>
      ) : (
        <form onSubmit={handleSubmit(onSubmit)}>
          {submitState === "error" && (
            <div style={{ color: "red" }}>{errorMessage}</div>
          )}

            <input
              {...register("username", { required: "用户名不能为空" })}

            <input
              {...register("email", { required: "邮箱不能为空" })}
              placeholder="邮箱"
            />
            {errors.email && <span>{errors.email.message}</span>}
```
