# 多步骤表单架构

多步骤表单将复杂表单拆分成多个步骤,提升用户体验。本文介绍 Next.js 中实现多步骤表单的各种方案。

## 核心概念

### 多步骤表单模式

| 模式     | 特点              | 适用场景   |
| -------- | ----------------- | ---------- |
| 状态管理 | useState 管理步骤 | 简单表单   |
| URL 状态 | 路由参数控制      | 可分享链接 |
| Context  | 全局状态共享      | 复杂嵌套   |
| 向导模式 | 线性流程          | 注册流程   |

## 实战场景一: 基础多步骤表单

### useState 管理步骤

```typescript
"use client";

import { useState } from "react";

interface FormData {
  step1: { name: string; email: string };
  step2: { address: string; city: string };
  step3: { cardNumber: string; cvv: string };
}

export default function MultiStepForm() {
  const [currentStep, setCurrentStep] = useState(1);
  const [formData, setFormData] = useState<Partial<FormData>>({});

  function handleNext(data: any) {
    setFormData({ ...formData, [`step${currentStep}`]: data });
    setCurrentStep(currentStep + 1);
  }

  function handleBack() {
    setCurrentStep(currentStep - 1);
  }

  function handleSubmit() {
    console.log("提交数据:", formData);
  }

  return (
    <div>
      <div>步骤 {currentStep} / 3</div>

      {currentStep === 1 && <Step1 onNext={handleNext} />}
      {currentStep === 2 && <Step2 onNext={handleNext} onBack={handleBack} />}
      {currentStep === 3 && (
        <Step3 onSubmit={handleSubmit} onBack={handleBack} />
      )}
    </div>
  );
}

function Step1({ onNext }: { onNext: (data: any) => void }) {
  const [name, setName] = useState("");
  const [email, setEmail] = useState("");

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    onNext({ name, email });
  }

  return (
    <form onSubmit={handleSubmit}>
      <h2>个人信息</h2>
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="姓名"
        required
      />
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="邮箱"
        required
      />
      <button type="submit">下一步</button>
    </form>
  );
}

function Step2({
  onNext,
  onBack,
}: {
  onNext: (data: any) => void;
  onBack: () => void;
}) {
  const [address, setAddress] = useState("");
  const [city, setCity] = useState("");

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    onNext({ address, city });
  }

  return (
    <form onSubmit={handleSubmit}>
      <h2>地址信息</h2>
      <input
        type="text"
        value={address}
        onChange={(e) => setAddress(e.target.value)}
        placeholder="地址"
        required
      />
      <input
        type="text"
        value={city}
        onChange={(e) => setCity(e.target.value)}
        placeholder="城市"
        required
      />
      <button type="button" onClick={onBack}>
        上一步
      </button>
      <button type="submit">下一步</button>
    </form>
  );
}

function Step3({
  onSubmit,
  onBack,
}: {
  onSubmit: () => void;
  onBack: () => void;
}) {
  const [cardNumber, setCardNumber] = useState("");
  const [cvv, setCvv] = useState("");

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    onSubmit();
  }

  return (
    <form onSubmit={handleSubmit}>
      <h2>支付信息</h2>
      <input
        type="text"
        value={cardNumber}
        onChange={(e) => setCardNumber(e.target.value)}
        placeholder="卡号"
        required
      />
      <input
        type="text"
        value={cvv}
        onChange={(e) => setCvv(e.target.value)}
        placeholder="CVV"
        required
      />
      <button type="button" onClick={onBack}>
        上一步
      </button>
      <button type="submit">提交</button>
    </form>
  );
}
```

## 实战场景四: 步骤验证

### 每步验证后才能继续

```typescript
"use client";

import { useState } from "react";
import { z } from "zod";

const step1Schema = z.object({
  name: z.string().min(1, "姓名不能为空"),
  email: z.string().email("邮箱格式不正确"),
});

const step2Schema = z.object({
  address: z.string().min(1, "地址不能为空"),
  city: z.string().min(1, "城市不能为空"),
});

export default function StepValidation() {
  const [currentStep, setCurrentStep] = useState(1);
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    address: "",
    city: "",
  });
  const [errors, setErrors] = useState<Record<string, string>>({});

  function validateStep1() {
    const result = step1Schema.safeParse({
      name: formData.name,
      email: formData.email,
    });

    if (!result.success) {
      const newErrors: Record<string, string> = {};
      result.error.errors.forEach((err) => {
        if (err.path[0]) {
          newErrors[err.path[0].toString()] = err.message;
        }
      });
      setErrors(newErrors);
      return false;
    }

    setErrors({});
    return true;
  }

  function validateStep2() {
    const result = step2Schema.safeParse({
      address: formData.address,
      city: formData.city,
    });

    if (!result.success) {
      const newErrors: Record<string, string> = {};
      result.error.errors.forEach((err) => {
        if (err.path[0]) {
          newErrors[err.path[0].toString()] = err.message;
        }
      });
      setErrors(newErrors);
      return false;
    }

    setErrors({});
    return true;
  }

  function handleNext() {
    if (currentStep === 1 && validateStep1()) {
      setCurrentStep(2);
    } else if (currentStep === 2 && validateStep2()) {
      setCurrentStep(3);
    }
  }

  function handleChange(field: string, value: string) {
    setFormData({ ...formData, [field]: value });
  }

  return (
    <div>
      {currentStep === 1 && (
        <div>
          <h2>个人信息</h2>
          <div>
            <input
              type="text"
              value={formData.name}
              onChange={(e) => handleChange("name", e.target.value)}
              placeholder="姓名"
            />
            {errors.name && <span style={{ color: "red" }}>{errors.name}</span>}
          </div>
          <div>
            <input
              type="email"
              value={formData.email}
              onChange={(e) => handleChange("email", e.target.value)}
              placeholder="邮箱"
            />
            {errors.email && (
              <span style={{ color: "red" }}>{errors.email}</span>
            )}
          </div>
          <button onClick={handleNext}>下一步</button>
        </div>
      )}

      {currentStep === 2 && (
        <div>
          <h2>地址信息</h2>
          <div>
            <input
              type="text"
              value={formData.address}
              onChange={(e) => handleChange("address", e.target.value)}
              placeholder="地址"
            />
            {errors.address && (
              <span style={{ color: "red" }}>{errors.address}</span>
            )}
          </div>
          <div>
            <input
              type="text"
              value={formData.city}
              onChange={(e) => handleChange("city", e.target.value)}
              placeholder="城市"
            />
            {errors.city && <span style={{ color: "red" }}>{errors.city}</span>}
          </div>
          <button onClick={() => setCurrentStep(1)}>上一步</button>
          <button onClick={handleNext}>下一步</button>
        </div>
      )}

      {currentStep === 3 && (
        <div>
          <h2>确认信息</h2>
          <p>姓名: {formData.name}</p>
          <p>邮箱: {formData.email}</p>
          <p>地址: {formData.address}</p>
          <p>城市: {formData.city}</p>
          <button onClick={() => setCurrentStep(2)}>上一步</button>
          <button onClick={() => console.log("提交", formData)}>提交</button>
        </div>
      )}
    </div>
  );
}
```

```typescript
"use client";

import { useState } from "react";

interface WizardStep {
  title: string;
  description: string;
  canSkip?: boolean;

const wizardSteps: WizardStep[] = [
  { title: "欢迎", description: "欢迎使用我们的服务", canSkip: false },
  { title: "个人信息", description: "填写您的基本信息", canSkip: false },

  const [formData, setFormData] = useState({
    name: "",
    email: "",
    notifications: true,
```

```typescript
"use client";

import { useState } from "react";

const steps = ["个人信息", "地址信息", "支付信息", "确认"];

export default function StepNavigation() {
  const [currentStep, setCurrentStep] = useState(0);
  const [completedSteps, setCompletedSteps] = useState<number[]>([]);

  function markStepComplete(step: number) {
    if (!completedSteps.includes(step)) {
      setCompletedSteps([...completedSteps, step]);
    }
  }

  function canNavigateToStep(step: number) {
    if (step === 0) return true;
    return completedSteps.includes(step - 1);
  }

  function handleNext() {
    markStepComplete(currentStep);
    if (currentStep < steps.length - 1) {
      setCurrentStep(currentStep + 1);
    }
  }

  return (
    <div>
      <div style={{ display: "flex", gap: "10px", marginBottom: "20px" }}>
        {steps.map((step, index) => (
          <button
            key={index}
            onClick={() => canNavigateToStep(index) && setCurrentStep(index)}
            disabled={!canNavigateToStep(index)}
            style={{
              backgroundColor:
                currentStep === index
                  ? "#4CAF50"
                  : completedSteps.includes(index)
                  ? "#8BC34A"
                  : "#ccc",
              color: "white",
              padding: "10px",
              border: "none",
              cursor: canNavigateToStep(index) ? "pointer" : "not-allowed",
            }}
          >
            {index + 1}. {step}
            {completedSteps.includes(index) && " ✓"}
          </button>
        ))}
      </div>

      <div>
        <h2>{steps[currentStep]}</h2>
        <p>这是步骤 {currentStep + 1} 的内容</p>
        {currentStep > 0 && (
          <button onClick={() => setCurrentStep(currentStep - 1)}>
            上一步
          </button>
        )}
        <button onClick={handleNext}>
          {currentStep === steps.length - 1 ? "提交" : "下一步"}
        </button>
```

## 总结

多步骤表单是复杂表单的最佳解决方案。通过本文的学习,我们掌握了:

### 核心技术

1. **状态管理**: useState、URL 参数、Context
2. **进度指示**: 可视化步骤进度
3. **步骤验证**: 每步验证后才能继续
4. **数据持久化**: localStorage 保存进度

### 最佳实践

1. **清晰导航**: 显示当前步骤和总步骤数
2. **数据保留**: 返回上一步时保留数据
3. **验证反馈**: 及时显示验证错误
4. **提交确认**: 显示摘要页面

### 实战技巧

1. **条件步骤**: 根据选择显示不同步骤
2. **向导模式**: 线性流程引导
3. **自由导航**: 允许跳转到已完成步骤
4. **Server Actions**: 服务端提交和验证

通过合理使用这些技术,可以构建用户体验优秀的多步骤表单系统。
