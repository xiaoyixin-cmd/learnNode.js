# 自定义编译配置

## 概述

React Compiler 提供了丰富的配置选项,允许开发者根据项目需求自定义编译行为。通过合理的配置,可以优化编译效果、控制编译范围、调整优化策略,甚至扩展编译器功能。本文将详细介绍 React Compiler 的配置系统、常用配置选项、高级配置技巧和实践案例,帮助开发者充分发挥编译器的能力。

## 第一部分:基础配置

### 1.1 Next.js 配置

**启用编译器**

```typescript
// next.config.js - 基础配置

import type { NextConfig } from "next";

const nextConfig: NextConfig = {
  experimental: {
    // 启用 React Compiler
    reactCompiler: true,

    // 或详细配置
    reactCompiler: {
      // 编译目标
      target: "19", // React 版本

      // 编译源
      sources: (filename) => {
        // 只编译 src 目录
        return filename.includes("src/");
      },

      // 排除文件
      exclude: ["**/node_modules/**", "**/.next/**", "**/dist/**"],

      // 优化级别
      optimizationLevel: "default", // 'none' | 'default' | 'aggressive'
    },
  },
};

export default nextConfig;
```

```typescript
interface Transformer {

class PropTypesRemovalTransformer implements Transformer {
  name = "remove-prop-types";

  transform(code: string, filename: string): string {
    // 移除 PropTypes
```

### 6.2 配置验证工具

**配置校验器**

```typescript
// scripts/validate-config.ts
import { NextConfig } from "next";
import fs from "fs";
import path from "path";

interface ValidationResult {
  valid: boolean;
  errors: string[];
  warnings: string[];
}

function validateReactCompilerConfig(config: NextConfig): ValidationResult {
  const errors: string[] = [];
  const warnings: string[] = [];

  const compilerConfig = config.experimental?.reactCompiler;

  if (!compilerConfig) {
    errors.push("React Compiler 未启用");
    return { valid: false, errors, warnings };
  }

  // 检查优化级别
  if (typeof compilerConfig === "object") {
    const level = compilerConfig.optimizationLevel;
    if (level && !["none", "default", "aggressive"].includes(level)) {
      errors.push(`无效的优化级别: ${level}`);
    }

    // 检查日志配置
    if (compilerConfig.logger) {
      const logLevel = compilerConfig.logger.level;
      if (!["error", "warn", "info", "debug", "trace"].includes(logLevel)) {
        errors.push(`无效的日志级别: ${logLevel}`);
      }
    }

    // 检查 source maps
    if (compilerConfig.sourceMaps && process.env.NODE_ENV === "production") {
      warnings.push("生产环境启用了 Source Maps,可能影响性能");
    }
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}

// 运行验证
const configPath = path.join(process.cwd(), "next.config.js");
if (fs.existsSync(configPath)) {
  import(configPath).then((config) => {
    const result = validateReactCompilerConfig(config.default);

    if (!result.valid) {
      console.error("❌ 配置验证失败:");
      result.errors.forEach((err) => console.error(`  - ${err}`));
      process.exit(1);
    }

    if (result.warnings.length > 0) {
      console.warn("⚠️  配置警告:");
      result.warnings.forEach((warn) => console.warn(`  - ${warn}`));
    }

    console.log("✅ 配置验证通过");
  });
}
```

```typescript
  }

  async migrate() {
    const configPath = path.join(process.cwd(), "next.config.js");
    const backupPath = path.join(process.cwd(), "next.config.backup.js");

    // 备份
    fs.copyFileSync(configPath, backupPath);
    console.log(`✅ 配置已备份: ${backupPath}`);

    // 读取配置
    const oldConfig = await import(configPath);

    // 迁移
    const newConfig = this.migrateToV2(oldConfig.default);

    // 写入
    const configContent = `
import type { NextConfig } from 'next';

const nextConfig: NextConfig = ${JSON.stringify(newConfig, null, 2)};

export default nextConfig;
`;

    fs.writeFileSync(configPath, configContent);
    console.log("✅ 配置迁移完成");
  }
}
```
