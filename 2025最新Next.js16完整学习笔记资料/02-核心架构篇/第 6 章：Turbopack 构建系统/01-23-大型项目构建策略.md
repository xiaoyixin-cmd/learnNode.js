# 大型项目构建策略

## 概述

随着项目规模的增长,构建时间和复杂度会呈指数级上升。一个拥有数千个模块的大型 Next.js 项目,如果没有合理的构建策略,可能需要数分钟甚至更长时间才能完成构建。Turbopack 专门为大型项目设计了一套完整的构建优化方案,包括增量构建、并行处理、智能缓存和模块化策略。本文将深入探讨如何在大型 Next.js 项目中实施最佳构建策略,确保开发效率和构建性能。

### 大型项目的挑战

1. **构建时间长**: 数千个模块需要编译和打包
2. **内存占用高**: 大量模块同时处理消耗大量内存
3. **依赖复杂**: 模块间依赖关系错综复杂
4. **团队协作**: 多人同时开发可能产生冲突
5. **维护困难**: 代码库庞大难以理解和维护

### Turbopack 的优势

- **增量构建**: 只构建变化的部分
- **并行处理**: 充分利用多核 CPU
- **持久化缓存**: 跨构建复用结果
- **智能分析**: 精确的依赖追踪
- **可扩展性**: 支持数万个模块

  // 分析模块依赖
  analyzeModuleDependencies(projectRoot: string) {
    const modules = this.scanModules(projectRoot);
    const graph = this.buildDependencyGraph(modules);

    return {
      modules: modules.length,
      dependencies: this.countDependencies(graph),
      circularDeps: this.findCircularDependencies(graph),

    };

  // 扫描所有模块
  private scanModules(root: string): Module[] {
    const modules: Module[] = [];

    // 递归扫描目录
    const scan = (dir: string) => {
      const files = fs.readdirSync(dir);

      files.forEach((file) => {
        const fullPath = path.join(dir, file);
        const stat = fs.statSync(fullPath);

        if (stat.isDirectory()) {
          scan(fullPath);
        } else if (file.endsWith(".ts") || file.endsWith(".tsx")) {
          modules.push(this.parseModule(fullPath));
        }
      });
    };

    scan(path.join(root, "src/modules"));

    return modules;
  }

  // 构建依赖图
  private buildDependencyGraph(modules: Module[]): DependencyGraph {
    const graph = new Map<string, Set<string>>();

    modules.forEach((module) => {
      const deps = this.extractDependencies(module);
      graph.set(module.path, new Set(deps));
    });

    return graph;
  }

  // 查找循环依赖
  private findCircularDependencies(graph: DependencyGraph): string[][] {
    const cycles: string[][] = [];
    const visited = new Set<string>();
    const recursionStack = new Set<string>();

    const dfs = (node: string, path: string[]) => {
      visited.add(node);
      recursionStack.add(node);
      path.push(node);

      const deps = graph.get(node) || new Set();

      for (const dep of deps) {
        if (!visited.has(dep)) {
          dfs(dep, [...path]);
        } else if (recursionStack.has(dep)) {
          // 找到循环
          const cycleStart = path.indexOf(dep);
          cycles.push(path.slice(cycleStart));
        }
      }

      recursionStack.delete(node);
    };

    graph.forEach((_, node) => {
      if (!visited.has(node)) {
        dfs(node, []);
      }
    });

    return cycles;
  }

// 自定义缓存策略

class BuildCache {
  private cacheDir = ".next/cache/turbopack";

  // 缓存模块编译结果
  async cacheModuleResult(moduleId: string, result: CompiledResult) {
    const cacheKey = this.getCacheKey(moduleId, result.hash);
    const cachePath = path.join(this.cacheDir, cacheKey);

    await fs.mkdir(path.dirname(cachePath), { recursive: true });
    await fs.writeFile(cachePath, JSON.stringify(result), "utf-8");
  }

  // 读取缓存
  async getCachedResult(
    moduleId: string,
    contentHash: string
  ): Promise<CompiledResult | null> {
    const cacheKey = this.getCacheKey(moduleId, contentHash);
    const cachePath = path.join(this.cacheDir, cacheKey);

    try {
      const data = await fs.readFile(cachePath, "utf-8");
      return JSON.parse(data);
    } catch {
      return null;
    }
  }

  // 生成缓存键
  private getCacheKey(moduleId: string, hash: string): string {
    return `${moduleId.replace(/[^a-z0-9]/gi, "-")}-${hash}.json`;
  }

  // 清理过期缓存
  async cleanOldCache(maxAge: number = 7 * 24 * 60 * 60 * 1000) {
    const files = await fs.readdir(this.cacheDir);
    const now = Date.now();

    for (const file of files) {
      const filePath = path.join(this.cacheDir, file);
      const stat = await fs.stat(filePath);

      if (now - stat.mtimeMs > maxAge) {
        await fs.unlink(filePath);
      }
    }
  }
}

// 使用 Worker Threads 加速构建

import { Worker } from "worker_threads";

class WorkerPoolBuilder {
  private workers: Worker[] = [];
  private queue: BuildTask[] = [];
  private results = new Map<string, any>();

  constructor(private workerCount: number = os.cpus().length) {
    this.initWorkers();
  }

  private initWorkers() {
    for (let i = 0; i < this.workerCount; i++) {
      const worker = new Worker("./build-worker.js");

      worker.on("message", (result) => {
        this.handleWorkerResult(result);
      });

      worker.on("error", (error) => {
        console.error(`Worker ${i} 错误:`, error);
      });

      this.workers.push(worker);
    }
  }

  // 添加构建任务
  addTask(task: BuildTask) {
    this.queue.push(task);
    this.processQueue();
  }

  // 处理队列
  private processQueue() {
    while (this.queue.length > 0) {
      const availableWorker = this.workers.find((w) => !this.isWorkerBusy(w));

      if (!availableWorker) {
        break;
      }

      const task = this.queue.shift()!;
      availableWorker.postMessage(task);
    }
  }

  private handleWorkerResult(result: any) {
    this.results.set(result.id, result.data);
    this.processQueue();
  }

  private isWorkerBusy(worker: Worker): boolean {
    // 检查 worker 是否繁忙
    return false;
  }

  // 等待所有任务完成
  async waitForCompletion(): Promise<Map<string, any>> {
    return new Promise((resolve) => {
      const check = () => {
        if (this.queue.length === 0 && this.allWorkersIdle()) {
          resolve(this.results);
        } else {
          setTimeout(check, 100);
        }
      };

      check();
    });
  }

  private allWorkersIdle(): boolean {
    return this.workers.every((w) => !this.isWorkerBusy(w));
  }

  // 清理
  destroy() {
    this.workers.forEach((w) => w.terminate());
  }
}

      (a, b) => b.duration - a.duration
    );

    sorted.forEach((metric) => {
      const percentage = ((metric.duration / totalTime) * 100).toFixed(1);

      totalTime: Date.now() - this.startTime,
      phases: Array.from(this.metrics.values()),

class MemoryMonitor {

  private printReport() {
    if (this.samples.length === 0) return;

    const peak = this.samples.reduce((max, sample) =>
      sample.heapUsed > max.heapUsed ? sample : max
    );

    const average =
      this.samples.reduce((sum, sample) => sum + sample.heapUsed, 0) /
      this.samples.length;

    console.log("\n💾 内存使用:");
    console.log(`  峰值: ${formatBytes(peak.heapUsed)}`);
    console.log(`  平均: ${formatBytes(average)}`);
    console.log(`  RSS: ${formatBytes(peak.rss)}`);
  }
}
