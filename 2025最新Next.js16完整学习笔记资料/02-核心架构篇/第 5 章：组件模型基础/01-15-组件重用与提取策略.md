# 组件重用与提取策略

## 概述

在 Next.js 16 开发中,组件的重用与提取是提高开发效率、减少代码重复、提升可维护性的关键策略。合理的组件重用能够加快开发速度,而有效的组件提取则能让代码更加清晰、模块化。本文档将深入探讨组件重用与提取的策略、模式和最佳实践。

### 什么是组件重用与提取

**组件重用**是指将已经开发好的组件在多个地方使用,避免重复编写相同的代码。

**组件提取**是指从现有代码中识别可重用的部分,将其抽取成独立的组件。

### 核心价值

1. **减少重复**:避免编写重复的代码
2. **提高效率**:复用组件加快开发速度
3. **易于维护**:修改一处,处处生效
4. **一致性**:确保 UI 和行为的一致性
5. **可测试性**:独立的组件更易于测试

```typescript
// components/hoc/withAuth.tsx
"use client";

import { useEffect, ComponentType } from "react";
import { useRouter } from "next/navigation";
import { useAuth } from "@/hooks/useAuth";

export function withAuth<P extends object>(
  Component: ComponentType<P>,
  options?: {
    requiredRole?: "admin" | "user";
    redirectTo?: string;
  }
) {
  return function AuthenticatedComponent(props: P) {
    const { isAuthenticated, user, isLoading } = useAuth();
    const router = useRouter();

    useEffect(() => {
      if (!isLoading && !isAuthenticated) {
        router.push(options?.redirectTo || "/login");
      }

      if (!isLoading && isAuthenticated && options?.requiredRole) {
        if (user?.role !== options.requiredRole) {
          router.push("/unauthorized");
        }
      }
    }, [isAuthenticated, user, isLoading, router]);

    if (isLoading) {
      return <div>加载中...</div>;
    }

    if (!isAuthenticated) {
      return null;
    }

    if (options?.requiredRole && user?.role !== options.requiredRole) {
      return null;
    }

    return <Component {...props} />;
  };
}
```

```typescript
// hooks/useLocalStorage.ts
"use client";

import { useState, useEffect } from "react";

export function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    if (typeof window === "undefined") {
      return initialValue;
    }

    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(error);
      return initialValue;
    }
  });

  const setValue = (value: T | ((val: T) => T)) => {
    try {
      const valueToStore =
        value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);

      if (typeof window !== "undefined") {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
      }
    } catch (error) {
      console.error(error);
    }
  };

  return [storedValue, setValue] as const;
}
```

```typescript
// hooks/useDebounce.ts
"use client";

import { useState, useEffect } from "react";

export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}
```

```typescript
// hooks/usePagination.ts
"use client";

import { useState, useMemo } from "react";

export function usePagination<T>(items: T[], itemsPerPage: number) {
  const [currentPage, setCurrentPage] = useState(1);

  const totalPages = Math.ceil(items.length / itemsPerPage);

  const paginatedItems = useMemo(() => {
    const startIndex = (currentPage - 1) * itemsPerPage;
    return items.slice(startIndex, startIndex + itemsPerPage);
  }, [items, currentPage, itemsPerPage]);

  const goToPage = (page: number) => {
    setCurrentPage(Math.max(1, Math.min(page, totalPages)));
  };

  const goToNextPage = () => {
    setCurrentPage((prev) => Math.min(prev + 1, totalPages));
  };

  const goToPreviousPage = () => {
    setCurrentPage((prev) => Math.max(prev - 1, 1));
  };

  return {
    currentPage,
    totalPages,
    paginatedItems,
    goToPage,
    goToNextPage,
    goToPreviousPage,
    hasNextPage: currentPage < totalPages,
    hasPreviousPage: currentPage > 1,
  };
}
```

```typescript
import { useDebounce } from "@/hooks/useDebounce";
import { usePagination } from "@/hooks/usePagination";
import { useLocalStorage } from "@/hooks/useLocalStorage";

interface Product {
  id: string;
  name: string;
  price: number;
}

interface Props {
  products: Product[];
}

export default function ProductList({ products }: Props) {

  const debouncedSearchTerm = useDebounce(searchTerm, 500);
  const [itemsPerPage, setItemsPerPage] = useLocalStorage("itemsPerPage", 10);

  const filteredProducts = products.filter((product) =>
    product.name.toLowerCase().includes(debouncedSearchTerm.toLowerCase())

  const {
    paginatedItems,
    currentPage,
    totalPages,
    goToPage,
    goToNextPage,
    goToPreviousPage,
    hasNextPage,
    hasPreviousPage,
  } = usePagination(filteredProducts, itemsPerPage);

  return (
    <div>
      <input
        type="text"
        value={searchTerm}
        onChange={(e) => setSearchTerm(e.target.value)}
        placeholder="搜索产品..."
      />

        value={itemsPerPage}
        onChange={(e) => setItemsPerPage(Number(e.target.value))}
```

```typescript
"use client";

import { useState, useEffect } from "react";

export function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        const res = await fetch(url);
        const json = await res.json();
        setData(json);
      } catch (err) {
        setError(err as Error);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, [url]);

  return { data, loading, error };
}
```

```typescript
// components/forms/Input.tsx
"use client";

import { InputHTMLAttributes, useState, useEffect } from "react";
import { cn } from "@/lib/utils";

interface InputProps extends InputHTMLAttributes<HTMLInputElement> {
  label?: string;
  error?: string;
  validate?: (value: string) => string | null;
}

export default function Input({
  label,
  error: externalError,
  validate,
  className,
  onBlur,
  onChange,
  ...props
}: InputProps) {
  const [internalError, setInternalError] = useState("");

  const error = externalError || internalError;

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (validate) {
      const error = validate(e.target.value);
      setInternalError(error || "");
    }
    onChange?.(e);
  };

  const handleBlur = (e: React.FocusEvent<HTMLInputElement>) => {
    if (validate) {
      const error = validate(e.target.value);
      setInternalError(error || "");
    }
    onBlur?.(e);
  };

  return (
    <div className="space-y-1">
      {label && (
        <label className="block text-sm font-medium text-gray-700">
          {label}
        </label>
      )}
      <input
        className={cn(
          "block w-full rounded-md shadow-sm",
          error
            ? "border-red-500 focus:border-red-500 focus:ring-red-500"
            : "border-gray-300 focus:border-blue-500 focus:ring-blue-500",
          className
        )}
        onChange={handleChange}
        onBlur={handleBlur}
        {...props}
      />
      {error && <p className="text-sm text-red-600">{error}</p>}
    </div>
  );
}
```

```typescript
import { useState, useMemo } from "react";

interface CartItem {
  id: string;
  name: string;
  price: number;
  quantity: number;
}

export function useCart() {
  const [items, setItems] = useState<CartItem[]>([]);

  const addItem = (item: Omit<CartItem, "quantity">) => {
    setItems((prev) => {
      const existing = prev.find((i) => i.id === item.id);
      if (existing) {
        return prev.map((i) =>
          i.id === item.id ? { ...i, quantity: i.quantity + 1 } : i
        );
      }
      return [...prev, { ...item, quantity: 1 }];
    });
  };

  const removeItem = (id: string) => {
    setItems((prev) => prev.filter((i) => i.id !== id));
  };

  const updateQuantity = (id: string, quantity: number) => {
    if (quantity <= 0) {
      removeItem(id);
      return;
    }
    setItems((prev) => prev.map((i) => (i.id === id ? { ...i, quantity } : i)));
  };

  const clearCart = () => {
    setItems([]);
  };

  const totalPrice = useMemo(
    () => items.reduce((sum, item) => sum + item.price * item.quantity, 0),
    [items]
  );

  const totalItems = useMemo(
    () => items.reduce((sum, item) => sum + item.quantity, 0),
    [items]
  );

  return {
    items,
```

## 第三部分:设计模式应用

### 3.1 Render Props 模式

**创建 DataList 组件**

```typescript
// components/DataList.tsx
"use client";

import { ReactNode } from "react";

interface DataListProps<T> {
  data: T[];
  loading?: boolean;
  error?: Error | null;
  renderItem: (item: T, index: number) => ReactNode;
  renderLoading?: () => ReactNode;
  renderError?: (error: Error) => ReactNode;
  renderEmpty?: () => ReactNode;
}

export default function DataList<T>({
  data,
  loading,
  error,
  renderItem,
  renderLoading,
  renderError,
  renderEmpty,
}: DataListProps<T>) {
  if (loading) {
    return <>{renderLoading ? renderLoading() : <div>加载中...</div>}</>;
  }

  if (error) {
    return (
      <>{renderError ? renderError(error) : <div>错误: {error.message}</div>}</>
    );
  }

  if (data.length === 0) {
    return <>{renderEmpty ? renderEmpty() : <div>暂无数据</div>}</>;
  }

  return (
    <div>
      {data.map((item, index) => (
        <div key={index}>{renderItem(item, index)}</div>
      ))}
    </div>
  );
}
```

### 3.2 Compound Components 模式

**创建 Tabs 组件系统**

```typescript
// components/ui/Tabs/index.tsx
"use client";

import { createContext, useContext, useState, ReactNode } from "react";

interface TabsContextType {
  activeTab: string;
  setActiveTab: (id: string) => void;
}

const TabsContext = createContext<TabsContextType | undefined>(undefined);

export function Tabs({
  children,
  defaultTab,
}: {
  children: ReactNode;
  defaultTab: string;
}) {
  const [activeTab, setActiveTab] = useState(defaultTab);

  return (
    <TabsContext.Provider value={{ activeTab, setActiveTab }}>
      <div className="tabs">{children}</div>
    </TabsContext.Provider>
  );
}

export function TabList({ children }: { children: ReactNode }) {
  return <div className="tab-list flex border-b">{children}</div>;
}

export function Tab({ id, children }: { id: string; children: ReactNode }) {
  const context = useContext(TabsContext);
  if (!context) throw new Error("Tab 必须在 Tabs 内使用");

  const { activeTab, setActiveTab } = context;

  return (
    <button
      className={`tab-button px-4 py-2 ${
        activeTab === id
          ? "border-b-2 border-blue-500 text-blue-600"
          : "text-gray-600"
      }`}
      onClick={() => setActiveTab(id)}
    >
      {children}
    </button>
  );
}

export function TabPanels({ children }: { children: ReactNode }) {
  return <div className="tab-panels mt-4">{children}</div>;
}

export function TabPanel({
  id,
  children,
}: {
  id: string;
  children: ReactNode;
}) {
  const context = useContext(TabsContext);
  if (!context) throw new Error("TabPanel 必须在 Tabs 内使用");

  const { activeTab } = context;

  if (activeTab !== id) return null;

  return <div className="tab-panel">{children}</div>;
}
```

## 常见问题

### Q1: 什么时候应该提取组件?

**A**: 考虑以下因素:

1. **重复使用**: 代码在 2 个或更多地方重复
2. **职责明确**: 组件有明确的单一职责
3. **可测试性**: 独立的组件更易于测试
4. **可读性**: 提取后代码更易理解

### Q2: 如何决定组件的粒度?

**A**: 平衡以下原则:

- **太小**: 过度抽象导致组件碎片化
- **太大**: 组件难以理解和维护
- **合适粒度**: 单一职责、易于理解、可重用

### Q3: HOC 和 Render Props 哪个更好?

**A**: 各有优劣:

**HOC 优点**:

- 语法简洁
- 易于组合多个 HOC
- 类型推导较好

**Render Props 优点**:

- 更灵活
- 避免 Props 冲突
- 更明确的数据流

现代开发推荐使用自定义 Hooks 替代两者。

### Q4: 如何避免过度抽象?

**A**: 遵循以下原则:

1. **YAGNI**: 你不会需要它(You Aren't Gonna Need It)
2. **实际需求**: 基于实际使用场景抽象
3. **三次规则**: 重复三次后再抽象
4. **简单优先**: 先写简单代码,需要时再重构

1. **重用策略**: 基础组件、组合模式、HOC、自定义 Hooks

4. **类型安全**: 使用 TypeScript
