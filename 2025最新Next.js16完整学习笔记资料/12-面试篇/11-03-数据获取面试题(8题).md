## 题目一: Next.js 16 中有哪些数据获取方式?

### 问题描述

请详细说明 Next.js 16 中所有的数据获取方式,并对比它们的优缺点。

### 参考答案

**1. 服务端组件直接获取**:

```typescript
// app/posts/page.tsx
async function PostsPage() {
  const posts = await fetch("https://api.example.com/posts").then((r) =>
    r.json()
  );

  return (
    <div>
      {posts.map((post) => (
        <div key={post.id}>{post.title}</div>
      ))}
    </div>
  );
}

export default PostsPage;
```

优点:

- 简单直观
- 自动缓存
- 类型安全
- 无需额外配置

缺点:

- 只能在服务端组件使用
- 不能使用客户端状态

**2. 使用 fetch 配置缓存**:

```typescript
// 强制缓存
const data = await fetch("https://api.example.com/data", {
  cache: "force-cache",
});

// 不缓存
const data = await fetch("https://api.example.com/data", {
  cache: "no-store",
});

// 定时重新验证
const data = await fetch("https://api.example.com/data", {
  next: { revalidate: 60 },
});

// 标签缓存
const data = await fetch("https://api.example.com/data", {
  next: { tags: ["posts"] },
});
```

**3. 并行数据获取**:

```typescript

    fetch("/api/posts").then((r) => r.json()),
    fetch("/api/users").then((r) => r.json()),
    fetch("/api/comments").then((r) => r.json()),
  ]);

  return <div>...</div>;
}
```

**4. 顺序数据获取**:

**2. 按需重新验证(On-demand Revalidation)**:

```typescript
// app/api/revalidate/route.ts
import { revalidatePath } from "next/cache";
import { NextRequest } from "next/server";

export async function POST(request: NextRequest) {
  const path = request.nextUrl.searchParams.get("path");

  if (path) {
    revalidatePath(path);
    return Response.json({ revalidated: true, now: Date.now() });
  }

  return Response.json({ revalidated: false }, { status: 400 });
}

// 使用
await fetch("/api/revalidate?path=/posts");
```

```typescript
// lib/prefetch.ts
export class PrefetchManager {
  private prefetched = new Set<string>();
  private router: any;

  constructor(router: any) {
    this.router = router;
  }

  prefetch(url: string, priority: "high" | "low" = "low") {
    if (this.prefetched.has(url)) return;

    if (priority === "high") {
      // 立即预取
      this.router.prefetch(url);
      this.prefetched.add(url);
    } else {
      // 延迟预取
      requestIdleCallback(() => {
        this.router.prefetch(url);
        this.prefetched.add(url);
      });
    }
  }

  prefetchBatch(urls: string[]) {
    urls.forEach((url) => this.prefetch(url, "low"));
  }
}

// 使用
const manager = new PrefetchManager(router);
manager.prefetch("/dashboard", "high");
manager.prefetchBatch(["/profile", "/settings"]);
```

```typescript
"use client";

import { useState, useEffect, useRef } from "react";

function InfiniteScroll() {
  const [posts, setPosts] = useState([]);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);
  const [loading, setLoading] = useState(false);
  const observerRef = useRef<HTMLDivElement>(null);

  const loadMore = async () => {
    if (loading || !hasMore) return;

    setLoading(true);
    const data = await fetch(`/api/posts?page=${page}&limit=10`).then((r) =>
      r.json()
    );

    setPosts((prev) => [...prev, ...data.posts]);
    setHasMore(data.hasMore);
    setPage((p) => p + 1);
    setLoading(false);
  };

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          loadMore();
        }
      },
      { threshold: 1.0 }
    );

    if (observerRef.current) {
      observer.observe(observerRef.current);
    }

    return () => observer.disconnect();
  }, [page, hasMore, loading]);

  return (
    <div>
      {posts.map((post) => (
        <div key={post.id}>{post.title}</div>
      ))}

      {hasMore && (
        <div ref={observerRef}>{loading ? "Loading..." : "Load More"}</div>
      )}
    </div>
  );
}
```

```typescript
}
```

**问题八: 类型不安全**

原因: 未定义类型

解决方案:

```typescript
// types/post.ts
export interface Post {
  id: string;
  title: string;
  content: string;
  createdAt: Date;
}

// app/posts/page.tsx
async function PostsPage() {
  const posts: Post[] = await fetch("/api/posts").then((r) => r.json());

  return (
    <div>
      {posts.map((post) => (
        <div key={post.id}>{post.title}</div>
      ))}
    </div>
  );
}
```

**常见问题总结**:

| 数据获取慢 | 顺序获取           | 并行获取、缓存   |

| 错误未处理 | 缺少错误处理       | error.tsx        |
| 类型不安全 | 未定义类型         | TypeScript 接口  |
