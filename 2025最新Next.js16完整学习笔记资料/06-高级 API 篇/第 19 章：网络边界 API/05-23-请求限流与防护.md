# 05-23-请求限流与防护

## 概述

在 Next.js 16 中,请求限流与防护是保护应用免受恶意攻击和滥用的重要机制。通过 Middleware 和 Route Handlers,我们可以实现各种限流策略和安全防护措施。本文将详细介绍如何在 Next.js 16 中实现完整的请求限流与防护系统。

### 核心特性

- **速率限制**: 限制请求频率
- **IP 黑名单**: 阻止恶意 IP
- **CSRF 防护**: 防止跨站请求伪造
- **XSS 防护**: 防止跨站脚本攻击
- **SQL 注入防护**: 防止 SQL 注入攻击
- **DDoS 防护**: 防止分布式拒绝服务攻击
- **请求验证**: 验证请求合法性
- **安全响应头**: 添加安全响应头

### 限流策略

```
┌─────────────┐
│  接收请求   │
└──────┬──────┘
       │
       ▼
┌─────────────┐
│  检查限流   │
└──────┬──────┘
       │
       ├─ 超限 → 拒绝请求(429)
       └─ 正常 → 继续处理
       │
       ▼
┌─────────────┐
│  处理请求   │
└──────┬──────┘
       │
       ▼
┌─────────────┐
│  返回响应   │
└─────────────┘
```

### 限流方式对比

| 方式     | 优点 | 缺点     | 适用场景 |
| -------- | ---- | -------- | -------- |
| 固定窗口 | 简单 | 突发流量 | 一般限流 |
| 滑动窗口 | 精确 | 复杂     | 精确限流 |
| 令牌桶   | 灵活 | 实现复杂 | 突发流量 |
| 漏桶     | 平滑 | 延迟高   | 流量整形 |

## 1. 速率限制

### 1.1 固定窗口限流

```ts
// lib/rate-limit/fixed-window.ts
interface RateLimitStore {
  [key: string]: {
    count: number;
    resetTime: number;
  };
}

const store: RateLimitStore = {};

export interface RateLimitConfig {
  windowMs: number;
  maxRequests: number;
}

export function checkRateLimit(
  identifier: string,
  config: RateLimitConfig
): { allowed: boolean; remaining: number; resetTime: number } {
  const now = Date.now();
  const key = identifier;

  if (!store[key] || store[key].resetTime < now) {
    store[key] = {
      count: 1,
      resetTime: now + config.windowMs,
    };

    return {
      allowed: true,
      remaining: config.maxRequests - 1,
      resetTime: store[key].resetTime,
    };
  }

  if (store[key].count >= config.maxRequests) {
    return {
      allowed: false,
      remaining: 0,
      resetTime: store[key].resetTime,
    };
  }

  store[key].count++;

  return {
    allowed: true,
    remaining: config.maxRequests - store[key].count,
    resetTime: store[key].resetTime,
  };
}
```

### 1.2 限流中间件

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";
import { checkRateLimit } from "@/lib/rate-limit/fixed-window";

export function middleware(request: NextRequest) {
  const ip = request.ip || request.headers.get("x-forwarded-for") || "unknown";

  const result = checkRateLimit(ip, {
    windowMs: 60 * 1000, // 1分钟
    maxRequests: 100, // 最多100个请求
  });

  if (!result.allowed) {
    return NextResponse.json(
      {
        error: "Too many requests",
        retryAfter: Math.ceil((result.resetTime - Date.now()) / 1000),
      },
      {
        status: 429,
        headers: {
          "X-RateLimit-Limit": "100",
          "X-RateLimit-Remaining": "0",
          "X-RateLimit-Reset": result.resetTime.toString(),
          "Retry-After": Math.ceil(
            (result.resetTime - Date.now()) / 1000
          ).toString(),
        },
      }
    );
  }

  const response = NextResponse.next();

  response.headers.set("X-RateLimit-Limit", "100");
  response.headers.set("X-RateLimit-Remaining", result.remaining.toString());
  response.headers.set("X-RateLimit-Reset", result.resetTime.toString());

  return response;
}

export const config = {
  matcher: "/api/:path*",
};
```

### 1.3 滑动窗口限流

```ts
// lib/rate-limit/sliding-window.ts
interface SlidingWindowStore {
  [key: string]: number[];
}

const store: SlidingWindowStore = {};

export function checkSlidingWindowRateLimit(
  identifier: string,
  config: RateLimitConfig
): { allowed: boolean; remaining: number } {
  const now = Date.now();
  const key = identifier;

  if (!store[key]) {
    store[key] = [];
  }

  store[key] = store[key].filter(
    (timestamp) => timestamp > now - config.windowMs
  );

  if (store[key].length >= config.maxRequests) {
    return {
      allowed: false,
      remaining: 0,
    };
  }

  store[key].push(now);

  return {
    allowed: true,
    remaining: config.maxRequests - store[key].length,
  };
}
```

```ts
  [ip: string]: {
    count: number;
    firstAttempt: number;
  };
}

const failedAttempts: FailedAttempts = {};

export function recordFailedAttempt(ip: string): boolean {
  const now = Date.now();
  const windowMs = 15 * 60 * 1000; // 15分钟
  const maxAttempts = 5;

  if (!failedAttempts[ip] || failedAttempts[ip].firstAttempt < now - windowMs) {
    failedAttempts[ip] = {
      count: 1,
      firstAttempt: now,
    };
    return false;
  }

  failedAttempts[ip].count++;

  if (failedAttempts[ip].count >= maxAttempts) {
    addToBlacklist(ip);
    return true;
  }

  return false;
}

export function clearFailedAttempts(ip: string): void {
  delete failedAttempts[ip];
}
```

## 3. CSRF 防护

### 3.1 CSRF Token 生成

```ts
// lib/security/csrf.ts
import crypto from "crypto";

const csrfTokens = new Map<string, { token: string; expiresAt: number }>();

export function generateCSRFToken(sessionId: string): string {
  const token = crypto.randomBytes(32).toString("hex");
  const expiresAt = Date.now() + 60 * 60 * 1000; // 1小时

  csrfTokens.set(sessionId, { token, expiresAt });

  return token;
}

export function verifyCSRFToken(sessionId: string, token: string): boolean {
  const stored = csrfTokens.get(sessionId);

  if (!stored) {
    return false;
  }

  if (stored.expiresAt < Date.now()) {
    csrfTokens.delete(sessionId);
    return false;
  }

  return stored.token === token;
}

export function deleteCSRFToken(sessionId: string): void {
  csrfTokens.delete(sessionId);
}
```

### 3.2 CSRF 中间件

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";
import { verifyCSRFToken } from "@/lib/security/csrf";

export function middleware(request: NextRequest) {
  const { pathname, method } = request.nextUrl;

  if (method !== "GET" && method !== "HEAD" && method !== "OPTIONS") {
    const sessionId = request.cookies.get("session-id")?.value;
    const csrfToken = request.headers.get("x-csrf-token");

    if (!sessionId || !csrfToken || !verifyCSRFToken(sessionId, csrfToken)) {
      return NextResponse.json(
        { error: "Invalid CSRF token" },
        { status: 403 }
      );
    }
  }

  return NextResponse.next();
}

  matcher: "/api/:path*",
};
```

## 4. XSS 防护

### 4.1 输入清理

```ts
// lib/security/xss.ts
import DOMPurify from "isomorphic-dompurify";

export function sanitizeHTML(html: string): string {
  return DOMPurify.sanitize(html, {
    ALLOWED_TAGS: ["p", "br", "strong", "em", "u", "a"],
    ALLOWED_ATTR: ["href", "target"],
  });
}

export function escapeHTML(text: string): string {
  const map: { [key: string]: string } = {
    "&": "&amp;",
    "<": "&lt;",
    ">": "&gt;",
    '"': "&quot;",
    "'": "&#x27;",
    "/": "&#x2F;",
  };

  return text.replace(/[&<>"'/]/g, (char) => map[char]);
}

export function sanitizeInput(input: any): any {
  if (typeof input === "string") {
    return escapeHTML(input);
  }

  if (Array.isArray(input)) {
    return input.map(sanitizeInput);
  }

  if (typeof input === "object" && input !== null) {
    const sanitized: any = {};
    for (const key in input) {
      sanitized[key] = sanitizeInput(input[key]);
    }
    return sanitized;
  }

  return input;
}
```

```ts
// lib/security/csp.ts
export function generateCSP(): string {
  const directives = {
    "default-src": ["'self'"],
    "script-src": ["'self'", "'unsafe-inline'", "https://cdn.example.com"],
    "style-src": ["'self'", "'unsafe-inline'"],
    "img-src": ["'self'", "data:", "https:"],
    "font-src": ["'self'", "https://fonts.gstatic.com"],
    "connect-src": ["'self'", "https://api.example.com"],
    "frame-ancestors": ["'none'"],
    "base-uri": ["'self'"],
    "form-action": ["'self'"],
  };

    .map(([key, values]) => `${key} ${values.join(" ")}`)
    .join("; ");
}
```

## 6. DDoS 防护

### 6.1 连接限制

```ts
// lib/security/ddos.ts
interface ConnectionTracker {
  [ip: string]: {
    connections: number;
    lastReset: number;
  };
}

const connections: ConnectionTracker = {};

export function checkConnectionLimit(
  ip: string,
  maxConnections: number = 10
): boolean {
  const now = Date.now();
  const resetInterval = 60 * 1000; // 1分钟

  if (!connections[ip] || connections[ip].lastReset < now - resetInterval) {
    connections[ip] = {
      connections: 1,
      lastReset: now,
    };
    return true;
  }

  if (connections[ip].connections >= maxConnections) {
    return false;
  }

  connections[ip].connections++;
  return true;
}

export function releaseConnection(ip: string): void {
  if (connections[ip]) {
    connections[ip].connections--;
  }
}
```

```ts
interface RequestTracker {

    startTime: number;
    completed: boolean;
  };
}

const requests: RequestTracker = {};

export function trackRequest(ip: string): void {
  requests[ip] = {
    startTime: Date.now(),
    completed: false,
  };
}

  if (requests[ip]) {
    requests[ip].completed = true;

    return false;

  const elapsed = Date.now() - requests[ip].startTime;

  if (!requests[ip].completed && elapsed > timeout) {
    return true;
  }

  return false;
}
```

## 7. 实战案例

### 7.1 完整的安全中间件

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";
import { checkRateLimit } from "@/lib/rate-limit/fixed-window";
import { isBlacklisted } from "@/lib/security/ip-blacklist";
import { verifyCSRFToken } from "@/lib/security/csrf";
import { checkConnectionLimit } from "@/lib/security/ddos";

export async function middleware(request: NextRequest) {
  const ip = request.ip || request.headers.get("x-forwarded-for") || "unknown";
  const { pathname, method } = request.nextUrl;

  // IP黑名单检查
  if (isBlacklisted(ip)) {
    return NextResponse.json({ error: "Access denied" }, { status: 403 });
  }

  // 连接限制
  if (!checkConnectionLimit(ip)) {
    return NextResponse.json(
      { error: "Too many connections" },
      { status: 429 }
    );
  }

  // 速率限制
  const rateLimit = checkRateLimit(ip, {
    windowMs: 60 * 1000,
    maxRequests: 100,
  });

  if (!rateLimit.allowed) {
    return NextResponse.json(
      { error: "Too many requests" },
      {
        status: 429,
        headers: {
          "Retry-After": Math.ceil(
            (rateLimit.resetTime - Date.now()) / 1000
          ).toString(),
        },
      }
    );
  }

  // CSRF检查
  if (method !== "GET" && method !== "HEAD" && method !== "OPTIONS") {
    const sessionId = request.cookies.get("session-id")?.value;
    const csrfToken = request.headers.get("x-csrf-token");

    if (!sessionId || !csrfToken || !verifyCSRFToken(sessionId, csrfToken)) {
      return NextResponse.json(
        { error: "Invalid CSRF token" },
        { status: 403 }
      );
    }
  }

  // 安全响应头
  const response = NextResponse.next();

  response.headers.set("X-Content-Type-Options", "nosniff");
  response.headers.set("X-Frame-Options", "DENY");
  response.headers.set("X-XSS-Protection", "1; mode=block");
  response.headers.set(
    "Strict-Transport-Security",
    "max-age=31536000; includeSubDomains"
  );

  return response;
}

export const config = {
  matcher: "/api/:path*",
};
```

**解决方案**:

```ts
import { verifyCSRFToken } from "@/lib/security/csrf";

export function middleware(request: NextRequest) {
  if (request.method !== "GET") {
    const sessionId = request.cookies.get("session-id")?.value;
    const csrfToken = request.headers.get("x-csrf-token");

    if (!verifyCSRFToken(sessionId, csrfToken)) {
      return NextResponse.json(
        { error: "Invalid CSRF token" },
        { status: 403 }
      );
    }
  }

  return NextResponse.next();
}
```
