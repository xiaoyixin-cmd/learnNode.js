# 05-21-响应处理与修改

## 概述

在 Next.js 16 中,响应处理与修改是构建动态 Web 应用的关键环节。通过 Middleware 和 Route Handlers,我们可以在响应发送给客户端之前对其进行各种处理和修改。本文将详细介绍如何在 Next.js 16 中实现各种响应处理和修改场景。

### 核心特性

- **响应头修改**: 添加、修改、删除响应头
- **Cookie 操作**: 设置、修改、删除 Cookie
- **响应体转换**: 转换响应数据格式
- **响应压缩**: 压缩响应数据
- **缓存控制**: 设置缓存策略
- **CORS 处理**: 处理跨域请求
- **重定向**: 实现各种重定向逻辑
- **流式响应**: 实现流式数据传输

### 响应处理流程

```
┌─────────────┐
│  路由处理   │
└──────┬──────┘
       │
       ▼
┌─────────────┐
│  Middleware │
│  - 拦截     │
│  - 修改     │
└──────┬──────┘
       │
       ├─ 添加响应头
       ├─ 设置Cookie
       ├─ 转换数据
       └─ 压缩响应
       │
       ▼
┌─────────────┐
│  客户端接收 │
└─────────────┘
```

### 响应修改类型对比

| 类型        | 用途       | 复杂度 | 性能影响 |
| ----------- | ---------- | ------ | -------- |
| 响应头修改  | 添加元信息 | 低     | 低       |
| Cookie 操作 | 状态管理   | 低     | 低       |
| 数据转换    | 格式转换   | 中     | 中       |
| 响应压缩    | 减少传输量 | 中     | 中       |
| 缓存控制    | 优化性能   | 中     | 低       |
| CORS 处理   | 跨域支持   | 低     | 低       |

## 1. 响应头修改

### 1.1 添加自定义响应头

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";

export function middleware(request: NextRequest) {
  const response = NextResponse.next();

  // 添加自定义响应头
  response.headers.set("X-Custom-Header", "value");
  response.headers.set("X-Request-ID", generateRequestId());
  response.headers.set("X-Response-Time", Date.now().toString());

  return response;
}

function generateRequestId(): string {
  return `${Date.now()}-${Math.random().toString(36).substring(7)}`;
}
```

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";

export function middleware(request: NextRequest) {

  response.headers.set("Referrer-Policy", "strict-origin-when-cross-origin");
  response.headers.set(
    "Permissions-Policy",
```

## 2. Cookie 操作

### 2.1 设置 Cookie

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";

export function middleware(request: NextRequest) {
  const response = NextResponse.next();

  // 设置Cookie
  response.cookies.set("session-id", generateSessionId(), {
    httpOnly: true,
    secure: true,
    sameSite: "lax",
    maxAge: 60 * 60 * 24 * 7, // 7天
  });

  response.cookies.set("theme", "dark", {
    maxAge: 60 * 60 * 24 * 365, // 1年
  });

  return response;
}

function generateSessionId(): string {
  return Math.random().toString(36).substring(2, 15);
}
```

### 2.2 修改 Cookie

```ts
  const response = NextResponse.next();

  const visitCount = parseInt(request.cookies.get("visit-count")?.value || "0");

  // 更新访问次数

    maxAge: 60 * 60 * 24 * 365,
  });

  return response;
```

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";

  const response = NextResponse.next();
```

```ts
import { NextRequest, NextResponse } from "next/server";

export async function GET(request: NextRequest) {
  const data = await getDynamicData();

  const response = NextResponse.json(data);

  // 根据数据类型设置不同的缓存策略
  if (data.type === "static") {

}

}
```

### 4.3 ETag 支持

```ts
// app/api/etag/route.ts
import { NextRequest, NextResponse } from "next/server";
import crypto from "crypto";

export async function GET(request: NextRequest) {
  const data = await getData();

  // 生成ETag
  const etag = generateETag(JSON.stringify(data));

  // 检查If-None-Match
  const ifNoneMatch = request.headers.get("If-None-Match");

  if (ifNoneMatch === etag) {
    return new NextResponse(null, { status: 304 });
  }

  const response = NextResponse.json(data);
  response.headers.set("ETag", etag);
  response.headers.set("Cache-Control", "public, max-age=3600");

  return response;
}

function generateETag(content: string): string {
  return crypto.createHash("md5").update(content).digest("hex");
}

async function getData() {
  return { message: "Data with ETag" };
}
```

### 5.2 动态 CORS 配置

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server";

const allowedOrigins = [
  "https://example.com",
  "https://app.example.com",
  "http://localhost:3000",
];

export function middleware(request: NextRequest) {
  const origin = request.headers.get("origin");

  const response = NextResponse.next();

  // 检查来源是否允许
  if (origin && allowedOrigins.includes(origin)) {

    "Access-Control-Allow-Methods",
    "GET, POST, PUT, DELETE, OPTIONS"
  );

    "Access-Control-Allow-Headers",
    "Content-Type, Authorization"

}
```

### 8.3 多格式响应

```ts
// app/api/multi-format/route.ts
import { NextRequest, NextResponse } from "next/server";

export async function GET(request: NextRequest) {
  const { searchParams } = request.nextUrl;
  const format = searchParams.get("format") || "json";

  const data = await getData();

  switch (format) {
    case "json":
      return NextResponse.json(data);

    case "xml":
      const xml = convertToXML(data);
      return new NextResponse(xml, {
        headers: {
          "Content-Type": "application/xml",
        },
      });

    case "csv":
      const csv = convertToCSV(data);
      return new NextResponse(csv, {
        headers: {
          "Content-Type": "text/csv",
          "Content-Disposition": 'attachment; filename="data.csv"',
        },
      });

    default:
      return NextResponse.json(
        { error: "Unsupported format" },
        { status: 400 }
      );
  }
}

async function getData() {
  return [
    { id: 1, name: "Item 1", value: 100 },
    { id: 2, name: "Item 2", value: 200 },
  ];
}

function convertToXML(data: any[]): string {
  let xml = '<?xml version="1.0" encoding="UTF-8"?>\n<items>\n';
  data.forEach((item) => {
    xml += "  <item>\n";
    Object.entries(item).forEach(([key, value]) => {
      xml += `    <${key}>${value}</${key}>\n`;
    });
    xml += "  </item>\n";
  });
  xml += "</items>";
  return xml;
}

function convertToCSV(data: any[]): string {
  if (data.length === 0) return "";

  const headers = Object.keys(data[0]).join(",");
  const rows = data.map((item) => Object.values(item).join(","));

  return [headers, ...rows].join("\n");
}
```

## 10. 注意事项

### 10.1 响应头大小限制

```ts
// ✗ 错误:响应头过大
export function middleware(request: NextRequest) {
  const response = NextResponse.next();

  // 响应头过大可能导致问题
  response.headers.set("X-Large-Data", "a".repeat(10000));

  return response;
}

// ✓ 正确:保持响应头简洁
export function middleware(request: NextRequest) {
  const response = NextResponse.next();

  response.headers.set("X-Request-ID", generateRequestId());

  return response;
}
```

### 10.2 Cookie 安全

```ts
// ✗ 错误:不安全的Cookie
response.cookies.set("session", "value"); // 缺少安全选项

response.cookies.set("session", "value", {
  httpOnly: true,
```

```ts
export async function GET() {
```

```ts
  const stream = new ReadableStream({
    async start(controller) {
      for (let i = 0; i < 10; i++) {

    },
  });

      "Content-Type": "text/event-stream",
    },

}
```

```ts
export async function GET(request: NextRequest) {
  const format = request.nextUrl.searchParams.get("format") || "json";
  const data = await getData();

  switch (format) {
    case "json":
      return NextResponse.json(data);
    case "xml":
      return new NextResponse(convertToXML(data), {
        headers: { "Content-Type": "application/xml" },
      });
    case "csv":

      });

      return NextResponse.json(
        { error: "Unsupported format" },
        { status: 400 }
      );
  }
}
```
