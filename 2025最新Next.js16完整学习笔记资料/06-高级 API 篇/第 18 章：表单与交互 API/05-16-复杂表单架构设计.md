# 05-16-复杂表单架构设计

## 概述

复杂表单是企业级应用的核心组件。Next.js 16 通过 Server Actions 和 React Hooks 提供了强大的表单架构能力。本文将详细介绍如何设计和实现可维护、可扩展的复杂表单系统。

### 核心特性

- **模块化设计**: 可复用的表单组件
- **状态管理**: 集中式表单状态
- **验证系统**: 统一的验证架构
- **错误处理**: 完善的错误处理机制
- **性能优化**: 大型表单性能优化
- **类型安全**: TypeScript 类型支持
- **可测试性**: 易于测试的架构
- **可扩展性**: 支持插件和扩展

### 表单架构对比

| 架构       | 优点     | 缺点       | 适用场景 |
| ---------- | -------- | ---------- | -------- |
| 简单表单   | 实现简单 | 难以维护   | 小型表单 |
| 受控组件   | 状态可控 | 性能问题   | 中型表单 |
| 非受控组件 | 性能好   | 状态难管理 | 简单表单 |
| 表单库     | 功能完善 | 学习成本   | 大型表单 |

```tsx
// app/lib/form/useFormState.ts
import { useState, useCallback } from "react";

export interface FormState<T> {
  values: T;
  errors: Partial<Record<keyof T, string>>;
  touched: Partial<Record<keyof T, boolean>>;
  isSubmitting: boolean;
  isValid: boolean;
}

export function useFormState<T extends Record<string, any>>(initialValues: T) {
  const [state, setState] = useState<FormState<T>>({
    values: initialValues,
    errors: {},
    touched: {},
    isSubmitting: false,
    isValid: true,
  });

  const setFieldValue = useCallback((field: keyof T, value: any) => {
    setState((prev) => ({
      ...prev,
      values: { ...prev.values, [field]: value },
    }));
  }, []);

  const setFieldError = useCallback((field: keyof T, error: string) => {
    setState((prev) => ({
      ...prev,
      errors: { ...prev.errors, [field]: error },
      isValid: false,
    }));
  }, []);

  const setFieldTouched = useCallback(
    (field: keyof T, touched: boolean = true) => {
      setState((prev) => ({
        ...prev,
        touched: { ...prev.touched, [field]: touched },
      }));
    },
    []
  );

  const resetForm = useCallback(() => {
    setState({
      values: initialValues,
      errors: {},
      touched: {},
      isSubmitting: false,
      isValid: true,
    });
  }, [initialValues]);

  return {
    ...state,
    setFieldValue,
    setFieldError,
    setFieldTouched,
    resetForm,
    setState,
  };
}
```

### 1.2 验证系统

```tsx
// app/lib/form/validation.ts
export type ValidationRule<T> = (value: T) => string | undefined;

export interface ValidationSchema<T> {
  [K: string]: ValidationRule<any>[];
}

export function createValidator<T>(schema: ValidationSchema<T>) {
  return function validate(values: T): Partial<Record<keyof T, string>> {
    const errors: any = {};

    Object.keys(schema).forEach((field) => {
      const rules = schema[field];
      const value = values[field as keyof T];

      for (const rule of rules) {
        const error = rule(value);
        if (error) {
          errors[field] = error;
          break;
        }
      }
    });

    return errors;
  };
}

// 常用验证规则
export const required = (message = "此字段必填"): ValidationRule<any> => {
  return (value) => {
    if (!value || (typeof value === "string" && !value.trim())) {
      return message;
    }
  };
};

export const minLength = (
  min: number,
  message?: string
): ValidationRule<string> => {
  return (value) => {
    if (value && value.length < min) {
      return message || `至少${min}个字符`;
    }
  };
};

export const maxLength = (
  max: number,
  message?: string
): ValidationRule<string> => {
  return (value) => {
    if (value && value.length > max) {
      return message || `最多${max}个字符`;
    }
  };
};

export const email = (message = "邮箱格式不正确"): ValidationRule<string> => {
  return (value) => {
    if (value && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
      return message;
    }
  };
};

export const pattern = (
  regex: RegExp,
  message: string
): ValidationRule<string> => {
  return (value) => {
    if (value && !regex.test(value)) {
      return message;
    }
  };
};
```

```tsx
// app/lib/form/FormContext.tsx
import { createContext, useContext, ReactNode } from "react";

interface FormContextValue<T> {
  values: T;
  errors: Partial<Record<keyof T, string>>;
  touched: Partial<Record<keyof T, boolean>>;
  setFieldValue: (field: keyof T, value: any) => void;
  setFieldError: (field: keyof T, error: string) => void;
  setFieldTouched: (field: keyof T, touched: boolean) => void;
  handleSubmit: (e: React.FormEvent) => void;
}

const FormContext = createContext<FormContextValue<any> | null>(null);

export function useFormContext<T>() {
  const context = useContext(FormContext);
  if (!context) {
    throw new Error("useFormContext must be used within FormProvider");
  }
  return context as FormContextValue<T>;
}

export function FormProvider<T>({
  children,
  value,
}: {
  children: ReactNode;
  value: FormContextValue<T>;
}) {
  return <FormContext.Provider value={value}>{children}</FormContext.Provider>;
}
```

```tsx
import { useFormContext } from "@/lib/form/FormContext";

interface FormFieldProps {
  name: string;
  label: string;
  type?: string;
  placeholder?: string;
  required?: boolean;
}

export function FormField({
  name,
  label,
  type = "text",
  placeholder,
  required,
}: FormFieldProps) {

    useFormContext();

  const error = touched[name] && errors[name];

        {required && <span className="text-red-500">*</span>}

      <input
        type={type}
        value={value}
        onChange={(e) => setFieldValue(name, e.target.value)}
        onBlur={() => setFieldTouched(name, true)}
        placeholder={placeholder}
        className={`w-full p-2 border rounded ${
          error ? "border-red-500" : "border-gray-300"
        }`}
      />
      {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
    </div>
  );
}
```

```tsx
// app/components/form/FormCheckbox.tsx
"use client";

import { useFormContext } from "@/lib/form/FormContext";

interface FormCheckboxProps {
  name: string;
  label: string;
  options: { value: string; label: string }[];
}

export function FormCheckbox({ name, label, options }: FormCheckboxProps) {
  const { values, errors, touched, setFieldValue, setFieldTouched } =
    useFormContext();

  const value = values[name] || [];
  const error = touched[name] && errors[name];

  function handleChange(optionValue: string, checked: boolean) {
    const newValue = checked
      ? [...value, optionValue]
      : value.filter((v: string) => v !== optionValue);
    setFieldValue(name, newValue);
  }

  return (
    <div className="mb-4">
      <label className="block mb-2 font-medium">{label}</label>
      <div className="space-y-2">
        {options.map((option) => (
          <label key={option.value} className="flex items-center">
            <input
              type="checkbox"
              checked={value.includes(option.value)}
              onChange={(e) => handleChange(option.value, e.target.checked)}
              onBlur={() => setFieldTouched(name, true)}
              className="mr-2"
            />
            {option.label}
          </label>
        ))}
      </div>
      {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
    </div>
  );
}
```

### 3.1 用户注册表单

```tsx
// app/components/RegisterForm.tsx
"use client";

import { FormProvider } from "@/lib/form/FormContext";
import { useFormState } from "@/lib/form/useFormState";
import {
  createValidator,
  required,
  email,
  minLength,
} from "@/lib/form/validation";
import { FormField } from "@/components/form/FormField";
import { FormSelect } from "@/components/form/FormSelect";
import { FormCheckbox } from "@/components/form/FormCheckbox";

interface RegisterFormData {
  username: string;
  email: string;
  password: string;
  confirmPassword: string;
  gender: string;
  interests: string[];
  agreeTerms: boolean;
}

const validator = createValidator<RegisterFormData>({
  username: [required(), minLength(3)],
  email: [required(), email()],
  password: [required(), minLength(8)],
  confirmPassword: [required()],
  gender: [required()],
  interests: [],
  agreeTerms: [required("请同意服务条款")],
});

export function RegisterForm() {
  const formState = useFormState<RegisterFormData>({
    username: "",
    email: "",
    password: "",
    confirmPassword: "",
    gender: "",
    interests: [],
    agreeTerms: false,
  });

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();

    // 验证
    const errors = validator(formState.values);
    if (Object.keys(errors).length > 0) {
      formState.setState((prev) => ({ ...prev, errors, isValid: false }));
      return;
    }

    // 提交
    formState.setState((prev) => ({ ...prev, isSubmitting: true }));
    try {
      const response = await fetch("/api/register", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(formState.values),
      });

      if (response.ok) {
        alert("注册成功!");
        formState.resetForm();
      } else {
        const data = await response.json();
        formState.setState((prev) => ({
          ...prev,
          errors: data.errors,
          isValid: false,
        }));
      }
    } finally {
      formState.setState((prev) => ({ ...prev, isSubmitting: false }));
    }
  }

  return (
    <FormProvider
      value={{
        ...formState,
        handleSubmit,
      }}
    >
      <form onSubmit={handleSubmit} className="max-w-md mx-auto">
        <h2 className="text-2xl font-bold mb-6">用户注册</h2>

        <FormField name="username" label="用户名" required />
        <FormField name="email" label="邮箱" type="email" required />
        <FormField name="password" label="密码" type="password" required />
        <FormField
          name="confirmPassword"
          label="确认密码"
          type="password"
          required
        />

        <FormSelect
          name="gender"
          label="性别"
          options={[
            { value: "male", label: "男" },
            { value: "female", label: "女" },
            { value: "other", label: "其他" },
          ]}
          required
        />

        <FormCheckbox
          name="interests"
          label="兴趣爱好"
          options={[
            { value: "sports", label: "运动" },
            { value: "music", label: "音乐" },
            { value: "reading", label: "阅读" },
            { value: "travel", label: "旅行" },
          ]}
        />

        <button
          type="submit"
          disabled={formState.isSubmitting}
          className="w-full p-2 bg-blue-500 text-white rounded disabled:bg-gray-400"
        >
          {formState.isSubmitting ? "提交中..." : "注册"}
        </button>
      </form>
    </FormProvider>
  );
}
```

### 6.5 如何实现批量验证?

**问题**: 一次验证所有字段。

**解决方案**:

```tsx
function validateAll() {
  const errors = validator(values);
  setState((prev) => ({
    ...prev,
    errors,
    touched: Object.keys(values).reduce(
      (acc, key) => ({ ...acc, [key]: true }),
      {}
    ),
    isValid: Object.keys(errors).length === 0,
  }));
  return Object.keys(errors).length === 0;
}
```

### 7.1 核心要点

```tsx
        }`}
      />
      {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
    </div>
  );
}
```

```tsx
// app/components/form/FormFileUpload.tsx
"use client";

import { useFormContext } from "@/lib/form/FormContext";

interface FormFileUploadProps {
  name: string;
  label: string;
  required?: boolean;
  accept?: string;
  multiple?: boolean;
}

export function FormFileUpload({
  name,
  label,
  required,
  accept,
  multiple,
}: FormFileUploadProps) {
  const { values, errors, touched, setFieldValue, setFieldTouched } =
    useFormContext();

  const error = touched[name] && errors[name];

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const files = e.target.files;
    if (files) {
      setFieldValue(name, multiple ? Array.from(files) : files[0]);
    }
  }

  return (
    <div className="mb-4">
      <label className="block mb-1 font-medium">
        {label}
        {required && <span className="text-red-500">*</span>}
      </label>
      <input
        type="file"
        onChange={handleChange}
        onBlur={() => setFieldTouched(name, true)}
        accept={accept}
        multiple={multiple}
        className={`w-full p-2 border rounded ${
          error ? "border-red-500" : "border-gray-300"
        }`}
      />
      {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
    </div>
  );
}
```
