# 05-13-表单验证与错误处理

## 1. 概述

表单验证与错误处理是 Web 应用中最重要的功能之一。Next.js 16 提供了多种方式来实现表单验证,包括客户端验证、服务端验证、以及使用第三方库如 Zod、Yup 等。本文将详细介绍如何在 Next.js 16 中实现完整的表单验证与错误处理系统。

### 1.1 核心特性

- **客户端验证**: 即时反馈,提升用户体验
- **服务端验证**: 确保数据安全,防止恶意提交
- **类型安全**: TypeScript + Zod 完整类型推导
- **错误处理**: 字段级和表单级错误管理
- **国际化**: 支持多语言错误消息
- **自定义验证**: 灵活的验证规则

### 1.2 验证策略对比

| 策略        | 优点                     | 缺点                 | 适用场景     |
| ----------- | ------------------------ | -------------------- | ------------ |
| 客户端验证  | 即时反馈、减少服务器负载 | 可被绕过、不够安全   | 用户体验优化 |
| 服务端验证  | 安全可靠、无法绕过       | 延迟较高、服务器负载 | 数据安全保障 |
| 双重验证    | 兼顾体验和安全           | 代码重复、维护成本   | 生产环境推荐 |
| Schema 验证 | 类型安全、可复用         | 学习成本、依赖库     | 大型应用     |

### 1.3 工作原理

```
用户输入 → 客户端验证 → 提交表单 → 服务端验证 → 返回结果
    ↓           ↓           ↓           ↓           ↓
  onChange   实时反馈    FormData   Schema验证   错误/成功
```

```tsx
// app/actions/auth.ts
"use server";

export async function registerUser(formData: FormData) {
  const email = formData.get("email") as string;
  const password = formData.get("password") as string;

  // 服务端验证
  const errors: Record<string, string[]> = {};

  if (!email) {
    errors.email = ["邮箱不能为空"];
  } else if (!/^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i.test(email)) {
    errors.email = ["邮箱格式不正确"];
  }

  if (!password) {
    errors.password = ["密码不能为空"];
  } else if (password.length < 8) {
    errors.password = ["密码至少8个字符"];
  }

  if (Object.keys(errors).length > 0) {
    return { success: false, errors };
  }

  // 处理注册逻辑
  return { success: true };
}
```

```tsx
  const rawData = {
    email: formData.get("email"),
    password: formData.get("password"),
    confirmPassword: formData.get("confirmPassword"),
    age: Number(formData.get("age")),
    terms: formData.get("terms") === "on",

  const result = RegisterSchema.safeParse(rawData);

  if (!result.success) {
    return {
      success: false,
      errors: result.error.flatten().fieldErrors,
    };
  }

  // 处理注册逻辑

  try {
    await createUser({ email, password, age });
    return { success: true };
  } catch (error) {
    return {
      success: false,
      errors: { _form: ["注册失败,请稍后重试"] },
    };
  }
}
```

### 3.3 实时验证

```tsx
// app/components/LiveValidationForm.tsx
"use client";

import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import { useState, useEffect } from "react";

const schema = z.object({
  username: z.string().min(3, "用户名至少3个字符"),
  email: z.string().email("邮箱格式不正确"),
});

type FormData = z.infer<typeof schema>;

export function LiveValidationForm() {
  const {
    register,
    watch,
    formState: { errors },
  } = useForm<FormData>({
    resolver: zodResolver(schema),
    mode: "onChange", // 实时验证
  });

  const username = watch("username");
  const [usernameAvailable, setUsernameAvailable] = useState<boolean | null>(
    null
  );
  const [checking, setChecking] = useState(false);

  useEffect(() => {
    if (username && username.length >= 3) {
      setChecking(true);
      const timer = setTimeout(async () => {
        const available = await checkUsernameAvailability(username);
        setUsernameAvailable(available);
        setChecking(false);
      }, 500);

      return () => clearTimeout(timer);
    }
  }, [username]);

  return (
    <form>
      <div>
        <input {...register("username")} placeholder="用户名" />
        {errors.username && (
          <p className="text-red-500">{errors.username.message}</p>
        )}
        {checking && <p>检查中...</p>}
        {usernameAvailable === true && <p className="text-green-500">✓ 可用</p>}
        {usernameAvailable === false && (
          <p className="text-red-500">✗ 已被占用</p>
        )}
      </div>

      <div>
        <input {...register("email")} placeholder="邮箱" />
        {errors.email && <p className="text-red-500">{errors.email.message}</p>}
      </div>

      <button type="submit">提交</button>
    </form>
  );
}

async function checkUsernameAvailability(username: string): Promise<boolean> {
  const response = await fetch(`/api/check-username?username=${username}`);
  const data = await response.json();
  return data.available;
}
```

```tsx
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import { useState } from "react";

const registerSchema = z
  .object({
    username: z
      .string()
      .min(3, "用户名至少3个字符")
      .max(20, "用户名最多20个字符"),
    email: z.string().email("邮箱格式不正确"),
    password: z.string().min(8, "密码至少8个字符"),
    confirmPassword: z.string(),
    phone: z.string().regex(/^1[3-9]\d{9}$/, "手机号格式不正确"),
    birthdate: z.string().refine((date) => {
      const age = new Date().getFullYear() - new Date(date).getFullYear();
      return age >= 18;
    }, "必须年满18岁"),
    gender: z.enum(["male", "female", "other"]),
    avatar: z.instanceof(File).optional(),
    terms: z.boolean().refine((val) => val === true, "必须同意服务条款"),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "两次密码不一致",
    path: ["confirmPassword"],
  });

type RegisterFormData = z.infer<typeof registerSchema>;

export function CompleteRegisterForm() {
  const [avatarPreview, setAvatarPreview] = useState<string | null>(null);

  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting },
    reset,
  } = useForm<RegisterFormData>({
    resolver: zodResolver(registerSchema),
  });

  async function onSubmit(data: RegisterFormData) {
    const formData = new FormData();
    Object.entries(data).forEach(([key, value]) => {
      if (value instanceof File) {
        formData.append(key, value);
      } else {
        formData.append(key, String(value));
      }
    });

    try {
      const response = await fetch("/api/register", {
        method: "POST",
        body: formData,
      });

      if (response.ok) {
        reset();
        alert("注册成功!");

      console.error("注册失败:", error);

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
      <div>
        <label>用户名</label>
        <input {...register("username")} />
        {errors.username && (
          <p className="text-red-500">{errors.username.message}</p>

        <label>密码</label>

        )}
      </div>

      <div>
        <label>确认密码</label>
        <input {...register("confirmPassword")} type="password" />
        {errors.confirmPassword && (
          <p className="text-red-500">{errors.confirmPassword.message}</p>
        )}
      </div>

      <div>
        <label>手机号</label>
        <input {...register("phone")} />
        {errors.phone && <p className="text-red-500">{errors.phone.message}</p>}
      </div>

      <div>
        <label>出生日期</label>
        <input {...register("birthdate")} type="date" />
        {errors.birthdate && (
          <p className="text-red-500">{errors.birthdate.message}</p>
```

```tsx
// app/components/MultiStepForm.tsx
"use client";

import { useState } from "react";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";

const step1Schema = z.object({
  email: z.string().email("邮箱格式不正确"),

});

type Step1Data = z.infer<typeof step1Schema>;
type Step2Data = z.infer<typeof step2Schema>;
type Step3Data = z.infer<typeof step3Schema>;

export function MultiStepForm() {

  const [formData, setFormData] = useState<

    setFormData((prev) => ({ ...prev, ...data }));

    setStep(3);

              </p>

          </div>
          <div>
            <input
              {...step1Form.register("password")}
              type="password"
              placeholder="密码"
            />
            {step1Form.formState.errors.password && (
              <p className="text-red-500">
                {step1Form.formState.errors.password.message}
              </p>
            )}
          </div>
          <button type="submit">下一步</button>
        </form>
      )}

      {step === 2 && (
        <form onSubmit={step2Form.handleSubmit(handleStep2Submit)}>
          <h2>个人信息</h2>
          <div>
            <input {...step2Form.register("firstName")} placeholder="名字" />
            {step2Form.formState.errors.firstName && (
              <p className="text-red-500">

            上一步

          <button type="submit">下一步</button>

      {step === 3 && (
```

## 6. 注意事项

### 6.1 客户端验证不能替代服务端验证

```tsx
// ✗ 错误:只有客户端验证
"use client";

export function UnsafeForm() {
  const { register, handleSubmit } = useForm();

  async function onSubmit(data) {
    // 直接提交,没有服务端验证
    await fetch("/api/submit", {
      method: "POST",
      body: JSON.stringify(data),
    });
  }

  return <form onSubmit={handleSubmit(onSubmit)}>...</form>;
}

// ✓ 正确:双重验证
("use server");

export async function submitForm(formData: FormData) {
  // 服务端验证
  const result = schema.safeParse(formData);

  if (!result.success) {
    return { errors: result.error.flatten().fieldErrors };
  }

  // 处理数据
  return { success: true };
}
```

### 6.2 避免过度验证

```tsx

  username: z
    .string()
    .min(3)
    .max(20)
    .regex(/^[a-zA-Z0-9_]+$/)
    .refine(async (val) => await checkAvailability(val))
    .refine((val) => !badWords.includes(val))
    .refine((val) => val !== "admin"),
});

// ✓ 更好:合理的验证规则

  username: z
    .string()
    .min(3, "用户名至少3个字符")
    .max(20, "用户名最多20个字符")

```

## 7. 常见问题

### 7.1 如何显示多个错误?

**问题**: 一个字段有多个验证规则失败。

**解决方案**:

```tsx
const schema = z.object({
  password: z
    .string()
    .min(8, "密码至少8个字符")
    .regex(/[A-Z]/, "必须包含大写字母")
    .regex(/[0-9]/, "必须包含数字"),
});

// Zod默认只返回第一个错误
// 如果需要所有错误,使用superRefine
const schema = z.object({
  password: z.string().superRefine((val, ctx) => {
    if (val.length < 8) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        message: "密码至少8个字符",
      });
    }
    if (!/[A-Z]/.test(val)) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        message: "必须包含大写字母",
      });
    }
    if (!/[0-9]/.test(val)) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        message: "必须包含数字",
      });
    }
  }),
});
```

### 7.2 如何处理动态字段?

**问题**: 表单字段数量不固定。

**解决方案**:

```tsx
const schema = z.object({
  items: z
    .array(
      z.object({
        name: z.string().min(1, "名称不能为空"),
        quantity: z.number().min(1, "数量至少为1"),
      })
    )
    .min(1, "至少添加一项"),
});

export function DynamicForm() {
  const {
    register,
    handleSubmit,
    watch,
    formState: { errors },
  } = useForm({
    resolver: zodResolver(schema),
    defaultValues: {
      items: [{ name: "", quantity: 1 }],
    },
  });

  const items = watch("items");

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      {items.map((item, index) => (
        <div key={index}>
          <input {...register(`items.${index}.name`)} />
          {errors.items?.[index]?.name && (
            <p>{errors.items[index].name.message}</p>
          )}
          <input
            {...register(`items.${index}.quantity`, { valueAsNumber: true })}
          />
          {errors.items?.[index]?.quantity && (
            <p>{errors.items[index].quantity.message}</p>
          )}
        </div>
      ))}
      <button type="submit">提交</button>
    </form>
  );
}
```
