# 05-15-实时表单反馈

## 概述

实时表单反馈是提升用户体验的关键功能。Next.js 16 通过 React Hooks 和 Server Actions 提供了强大的实时反馈能力,可以在用户输入时立即验证并显示结果。本文将详细介绍如何实现各种实时表单反馈功能。

### 核心特性

- **即时验证**: 用户输入时立即验证
- **防抖优化**: 避免频繁请求
- **异步验证**: 实时检查数据唯一性
- **实时搜索**: 输入时显示搜索结果
- **自动完成**: 智能建议输入内容
- **字符计数**: 实时显示剩余字符
- **密码强度**: 实时显示密码强度
- **格式化输入**: 自动格式化用户输入

### 实时反馈方式对比

| 方式          | 优点       | 缺点         | 适用场景 |
| ------------- | ---------- | ------------ | -------- |
| onChange 验证 | 即时反馈   | 可能过于频繁 | 简单验证 |
| onBlur 验证   | 用户体验好 | 反馈延迟     | 复杂验证 |
| 防抖验证      | 性能好     | 有延迟       | 异步验证 |
| 节流验证      | 控制频率   | 可能遗漏     | 搜索建议 |

### 工作原理

```
┌─────────────┐
│  用户输入   │
└──────┬──────┘
       │
       │ onChange
       ▼
┌─────────────┐
│  防抖处理   │
└──────┬──────┘
       │
       │ 延迟执行
       ▼
┌─────────────┐
│  验证逻辑   │
└──────┬──────┘
       │
       │ 返回结果
       ▼
┌─────────────┐
│  显示反馈   │
└─────────────┘
```

// app/components/InstantValidation.tsx
"use client";

import { useState } from "react";

export function InstantValidation() {
  const [email, setEmail] = useState("");
  const [error, setError] = useState("");

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value;
    setEmail(value);

    // 即时验证
    if (!value) {
      setError("邮箱不能为空");
    } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
      setError("邮箱格式不正确");
    } else {
      setError("");
    }
  }

  return (
    <div>
      <input
        type="email"
        value={email}
        onChange={handleChange}
        placeholder="请输入邮箱"
      />
      {error && <p className="text-red-500">{error}</p>}
      {!error && email && <p className="text-green-500">✓ 邮箱格式正确</p>}
    </div>
  );
}

// app/hooks/useDebounce.ts
import { useEffect, useState } from "react";

export function useDebounce<T>(value: T, delay: number = 500): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(timer);
    };
  }, [value, delay]);

  return debouncedValue;
}

// app/components/DebouncedValidation.tsx
"use client";

import { useState, useEffect } from "react";
import { useDebounce } from "../hooks/useDebounce";

export function DebouncedValidation() {
  const [username, setUsername] = useState("");
  const [checking, setChecking] = useState(false);
  const [available, setAvailable] = useState<boolean | null>(null);

  const debouncedUsername = useDebounce(username, 500);

  useEffect(() => {
    if (debouncedUsername && debouncedUsername.length >= 3) {
      setChecking(true);
      checkUsername(debouncedUsername).then((result) => {
        setAvailable(result);
        setChecking(false);
      });
    }
  }, [debouncedUsername]);

  return (
    <div>
      <input
        type="text"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
        placeholder="请输入用户名"
      />
      {checking && <p>检查中...</p>}
      {available === true && <p className="text-green-500">✓ 用户名可用</p>}
      {available === false && <p className="text-red-500">✗ 用户名已被占用</p>}
    </div>
  );
}

async function checkUsername(username: string): Promise<boolean> {
  const response = await fetch(`/api/check-username?username=${username}`);
  const data = await response.json();
  return data.available;
}

interface PasswordStrengthResult {

  label: string;
  color: string;
  suggestions: string[];

function calculatePasswordStrength(password: string): PasswordStrengthResult {
  let score = 0;
  const suggestions: string[] = [];

  if (password.length >= 8) score++;
  else suggestions.push("至少8个字符");

  if (/[a-z]/.test(password)) score++;
  else suggestions.push("包含小写字母");

  if (/[A-Z]/.test(password)) score++;
  else suggestions.push("包含大写字母");

  if (/[0-9]/.test(password)) score++;
  else suggestions.push("包含数字");

  if (/[^A-Za-z0-9]/.test(password)) score++;
  else suggestions.push("包含特殊字符");

  const labels = ["很弱", "弱", "一般", "强", "很强"];
  const colors = [
    "bg-red-500",
    "bg-orange-500",
    "bg-yellow-500",
    "bg-blue-500",
    "bg-green-500",
  ];

  return {
    score,
    label: labels[score] || "很弱",
    color: colors[score] || "bg-red-500",
    suggestions,
  };
}

export function PasswordStrength() {
  const [password, setPassword] = useState("");
  const strength = calculatePasswordStrength(password);

  return (
    <div>
      <input
        type="password"
        value={password}

        placeholder="请输入密码"

            <div className="mt-2 text-sm text-gray-600">
              <p>建议:</p>
              <ul className="list-disc list-inside">
                {strength.suggestions.map((suggestion, i) => (
                  <li key={i}>{suggestion}</li>
                ))}
              </ul>
            </div>
          )}

// app/components/LiveSearch.tsx
"use client";

import { useState, useEffect } from "react";
import { useDebounce } from "../hooks/useDebounce";

interface SearchResult {
  id: string;
  title: string;
  description: string;
}

export function LiveSearch() {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<SearchResult[]>([]);
  const [loading, setLoading] = useState(false);

  const debouncedQuery = useDebounce(query, 300);

  useEffect(() => {
    if (debouncedQuery) {
      setLoading(true);
      searchItems(debouncedQuery).then((data) => {
        setResults(data);
        setLoading(false);
      });
    } else {
      setResults([]);
    }
  }, [debouncedQuery]);

  return (
    <div className="relative">
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="搜索..."
        className="w-full p-2 border rounded"
      />

      {loading && <div className="absolute right-2 top-2">搜索中...</div>}

      {results.length > 0 && (
        <div className="absolute w-full mt-1 bg-white border rounded shadow-lg max-h-60 overflow-y-auto">
          {results.map((result) => (
            <div
              key={result.id}
              className="p-2 hover:bg-gray-100 cursor-pointer"
              onClick={() => setQuery(result.title)}
            >
              <div className="font-medium">{result.title}</div>
              <div className="text-sm text-gray-600">{result.description}</div>
            </div>
          ))}
        </div>
      )}

      {query && !loading && results.length === 0 && (
        <div className="absolute w-full mt-1 bg-white border rounded shadow-lg p-2">
          没有找到结果
        </div>
      )}
    </div>
  );
}

async function searchItems(query: string): Promise<SearchResult[]> {
  const response = await fetch(`/api/search?q=${encodeURIComponent(query)}`);
  return response.json();
}

### 2.3 自动完成

```tsx
// app/components/Autocomplete.tsx
"use client";

import { useState, useEffect, useRef } from "react";
import { useDebounce } from "../hooks/useDebounce";

interface AutocompleteProps {
  suggestions: string[];
  onSelect?: (value: string) => void;
}

export function Autocomplete({ suggestions, onSelect }: AutocompleteProps) {
  const [value, setValue] = useState("");
  const [filteredSuggestions, setFilteredSuggestions] = useState<string[]>([]);
  const [showSuggestions, setShowSuggestions] = useState(false);
  const [activeSuggestionIndex, setActiveSuggestionIndex] = useState(0);
  const inputRef = useRef<HTMLInputElement>(null);

  const debouncedValue = useDebounce(value, 200);

  useEffect(() => {
    if (debouncedValue) {
      const filtered = suggestions.filter((suggestion) =>
        suggestion.toLowerCase().includes(debouncedValue.toLowerCase())
      );
      setFilteredSuggestions(filtered);
      setShowSuggestions(true);
    } else {
      setFilteredSuggestions([]);
      setShowSuggestions(false);
    }
  }, [debouncedValue, suggestions]);

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === "ArrowDown") {
      e.preventDefault();
      setActiveSuggestionIndex((prev) =>
        prev < filteredSuggestions.length - 1 ? prev + 1 : prev
      );
    } else if (e.key === "ArrowUp") {
      e.preventDefault();
      setActiveSuggestionIndex((prev) => (prev > 0 ? prev - 1 : 0));
    } else if (e.key === "Enter") {
      e.preventDefault();
      if (filteredSuggestions[activeSuggestionIndex]) {
        selectSuggestion(filteredSuggestions[activeSuggestionIndex]);
      }
    } else if (e.key === "Escape") {
      setShowSuggestions(false);
    }
  }

  function selectSuggestion(suggestion: string) {
    setValue(suggestion);
    setShowSuggestions(false);
    setActiveSuggestionIndex(0);
    onSelect?.(suggestion);
  }

  return (
    <div className="relative">
      <input
        ref={inputRef}
        type="text"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        onKeyDown={handleKeyDown}
        placeholder="输入以获取建议..."
        className="w-full p-2 border rounded"
      />

      {showSuggestions && filteredSuggestions.length > 0 && (
        <div className="absolute w-full mt-1 bg-white border rounded shadow-lg max-h-60 overflow-y-auto z-10">
          {filteredSuggestions.map((suggestion, index) => (
            <div
              key={index}
              className={`p-2 cursor-pointer ${
                index === activeSuggestionIndex
                  ? "bg-blue-100"
                  : "hover:bg-gray-100"
              }`}
              onClick={() => selectSuggestion(suggestion)}
            >
              {suggestion}
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

"use client";

import { useState } from "react";

export function PhoneInput() {
  const [phone, setPhone] = useState("");

  function formatPhone(value: string): string {
    const cleaned = value.replace(/\D/g, "");
    const match = cleaned.match(/^(\d{3})(\d{4})(\d{4})$/);
    if (match) {
      return `${match[1]}-${match[2]}-${match[3]}`;
    }
    return cleaned;
  }

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value.replace(/\D/g, "");
    if (value.length <= 11) {
      setPhone(formatPhone(value));
    }
  }

  return (
    <input
      type="text"
      value={phone}
      onChange={handleChange}
      placeholder="138-0000-0000"
      className="w-full p-2 border rounded"
    />
  );
}

export function CreditCardInput() {
  const [card, setCard] = useState("");

  function formatCard(value: string): string {
    const cleaned = value.replace(/\s/g, "");
    const groups = cleaned.match(/.{1,4}/g);
    return groups ? groups.join(" ") : cleaned;
  }

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value.replace(/\s/g, "");
    if (value.length <= 16 && /^\d*$/.test(value)) {
      setCard(formatCard(value));
    }
  }

  return (
    <input
      type="text"
      value={card}
      onChange={handleChange}
      placeholder="1234 5678 9012 3456"
      className="w-full p-2 border rounded"
    />
  );
}

## 5. 注意事项

### 5.1 避免过度验证

```tsx
// ✗ 错误:每次输入都验证
onChange={(e) => {
  const value = e.target.value;
  checkAvailability(value); // 频繁请求
}}

// ✓ 正确:使用防抖
const debouncedValue = useDebounce(value, 500);
useEffect(() => {
  if (debouncedValue) {
    checkAvailability(debouncedValue);
  }
}, [debouncedValue]);
```

### 5.2 合理的防抖延迟

```tsx
// ✗ 错误:延迟太短
const debouncedValue = useDebounce(value, 50); // 太短,仍然频繁

// ✗ 错误:延迟太长
const debouncedValue = useDebounce(value, 2000); // 太长,用户体验差

// ✓ 正确:合理的延迟
const debouncedValue = useDebounce(value, 300 - 500); // 平衡性能和体验
```

### 6.1 如何实现节流?

**问题**: 防抖和节流的区别。

**解决方案**:

```tsx
// 节流Hook
export function useThrottle<T>(value: T, delay: number = 500): T {
  const [throttledValue, setThrottledValue] = useState<T>(value);
  const lastRun = useRef(Date.now());

  useEffect(() => {
    const handler = setTimeout(() => {
      if (Date.now() - lastRun.current >= delay) {
        setThrottledValue(value);
        lastRun.current = Date.now();
      }
    }, delay - (Date.now() - lastRun.current));

    return () => clearTimeout(handler);
  }, [value, delay]);

  return throttledValue;
}
```

export function useFormValidation(initialValues: any, validationRules: any) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});

  const debouncedValues = useDebounce(values, 500);

  useEffect(() => {
    const newErrors = {};
