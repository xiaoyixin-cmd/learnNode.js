## 🚀 Node.js 22 LTS 新特性

### 1. 内置 SQLite 支持（实验性）

```javascript
// ===== node:sqlite - 内置数据库 =====

import { DatabaseSync } from 'node:sqlite';

// 创建数据库
const db = new DatabaseSync(':memory:');  // 内存数据库
// 或
const db = new DatabaseSync('mydb.db');   // 文件数据库

// 创建表
db.exec(`
  CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    age INTEGER
  )
`);

// 插入数据
const insert = db.prepare(`
  INSERT INTO users (name, email, age) VALUES (?, ?, ?)
`);

insert.run('Alice', 'alice@example.com', 25);
insert.run('Bob', 'bob@example.com', 30);

// 查询数据
const select = db.prepare('SELECT * FROM users WHERE age > ?');
const users = select.all(20);

console.log(users);
// [
//   { id: 1, name: 'Alice', email: 'alice@example.com', age: 25 },
//   { id: 2, name: 'Bob', email: 'bob@example.com', age: 30 }
// ]

// 查询单条
const user = db.prepare('SELECT * FROM users WHERE id = ?').get(1);
console.log(user);  // { id: 1, name: 'Alice', ... }

// 事务
db.exec('BEGIN');
try {
  insert.run('Charlie', 'charlie@example.com', 35);
  insert.run('David', 'david@example.com', 40);
  db.exec('COMMIT');
} catch (error) {
  db.exec('ROLLBACK');
  throw error;
}

// 关闭数据库
db.close();

// ===== 实战：简单的任务管理 =====

class TaskManager {
  constructor(dbPath = ':memory:') {
    this.db = new DatabaseSync(dbPath);
    this.init();
  }
  
  init() {
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS tasks (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        title TEXT NOT NULL,
        completed BOOLEAN DEFAULT 0,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP
      )
    `);
  }
  
  addTask(title) {
    const stmt = this.db.prepare(
      'INSERT INTO tasks (title) VALUES (?)'
    );
    const result = stmt.run(title);
    return result.lastInsertRowid;
  }
  
  getTasks() {
    const stmt = this.db.prepare('SELECT * FROM tasks ORDER BY id DESC');
    return stmt.all();
  }
  
  toggleTask(id) {
    const stmt = this.db.prepare(
      'UPDATE tasks SET completed = NOT completed WHERE id = ?'
    );
    stmt.run(id);
  }
  
  deleteTask(id) {
    const stmt = this.db.prepare('DELETE FROM tasks WHERE id = ?');
    stmt.run(id);
  }
  
  close() {
    this.db.close();
  }
}

// 使用
const tasks = new TaskManager();
tasks.addTask('学习Node.js');
tasks.addTask('完成项目');
console.log(tasks.getTasks());
```

### 2. Permission Model（权限模型）

```javascript
// ===== 启用权限模型 =====

// 运行时指定权限
// node --experimental-permission --allow-fs-read=* --allow-fs-write=./data app.js

// ===== 代码中检查权限 =====

import { permission } from 'node:process';

// 检查文件系统读权限
if (permission.has('fs.read')) {
  console.log('有文件系统读权限');
} else {
  console.log('无文件系统读权限');
}

// 检查子进程权限
if (permission.has('child')) {
  console.log('可以创建子进程');
}

// 检查网络权限
if (permission.has('worker')) {
  console.log('可以创建Worker线程');
}

// ===== 实战：安全的文件读取 =====

import fs from 'fs';

function safeReadFile(path) {
  // 检查权限
  if (!permission.has('fs.read')) {
    throw new Error('No file system read permission');
  }
  
  // 检查路径
  const allowedDir = process.cwd();
  const absolutePath = fs.realpathSync(path);
  
  if (!absolutePath.startsWith(allowedDir)) {
    throw new Error('Access denied: Outside allowed directory');
  }
  
  return fs.readFileSync(path, 'utf8');
}

// 使用
try {
  const content = safeReadFile('./data.txt');
  console.log(content);
} catch (error) {
  console.error('读取失败:', error.message);
}
```

```javascript
// ./myapp arg1 arg2

// ===== 实战：打包CLI工具 =====

// cli.js

import { program } from 'commander';

program
  .name('mytool')
  .description('My CLI Tool')
  .version('1.0.0');

  .command('greet <name>')
  .description('Greet someone')
```

```javascript
// ===== Node.js 24可以直接require ESM模块 =====
```

### 3. 改进的 Test Runner

```javascript
// ===== Node.js 24测试运行器增强 =====

import { test, describe, it, before, after } from 'node:test';
import assert from 'node:assert';

// 新增：测试覆盖率（不需要额外工具）

    // 这些测试会并发运行

  test('test 2', async () => {
    //...
  });

  assert.strictEqual(3 + 3, 6);
```

## 🎯 实战：使用新特性构建应用

```javascript
// ===== 使用Node.js 24新特性的完整应用 =====

// app.js
import { DatabaseSync } from 'node:sqlite';
import { test, describe } from 'node:test';
import assert from 'node:assert';

// 使用内置SQLite
class UserService {
  constructor() {
    this.db = new DatabaseSync(':memory:');
    this.init();
  }
  
  init() {
    this.db.exec(`
      CREATE TABLE users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE NOT NULL
      )
    `);
  }
  
  createUser(name, email) {
    const stmt = this.db.prepare(
      'INSERT INTO users (name, email) VALUES (?, ?)'
    );
    return stmt.run(name, email).lastInsertRowid;
  }
  
  getUser(id) {
    const stmt = this.db.prepare('SELECT * FROM users WHERE id = ?');
    return stmt.get(id);
  }
  
  getAllUsers() {
    const stmt = this.db.prepare('SELECT * FROM users');
    return stmt.all();
  }
}

// 使用改进的测试运行器
describe('UserService', () => {
  let service;
  
  before(() => {
    service = new UserService();
  });
  
  test('should create user', () => {
    const id = service.createUser('Alice', 'alice@example.com');
    assert.ok(id > 0);
  });
  
  test('should get user', () => {
    const id = service.createUser('Bob', 'bob@example.com');
    const user = service.getUser(id);
    
    assert.strictEqual(user.name, 'Bob');
    assert.strictEqual(user.email, 'bob@example.com');
  });
  
  test('should get all users', () => {
    const users = service.getAllUsers();
    assert.ok(users.length >= 0);
  });
});

// 运行：
// node --test --watch app.js
```
