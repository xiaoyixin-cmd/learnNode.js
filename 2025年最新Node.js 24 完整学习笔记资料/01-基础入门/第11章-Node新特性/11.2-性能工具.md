# 11.2 性能优化新工具

> 使用现代化工具优化Node.js应用性能

---

## 🎯 为什么需要性能优化？

### 核心概念

```
性能优化 = 让应用更快、更省资源

性能指标：
• 响应时间（Response Time）- 越短越好
• 吞吐量（Throughput）- 越高越好
• CPU使用率（CPU Usage）- 合理范围
• 内存使用（Memory Usage）- 不泄漏
• 并发数（Concurrency）- 能处理多少请求

性能瓶颈来源：
• 同步I/O操作
• 内存泄漏
• 低效算法
• 过多依赖
• 阻塞事件循环

优化工具：
• 内置工具（Node.js自带）
• Clinic.js（诊断套件）
• 0x（火焰图）
• AutoCannon（压测）
```

## 📊 Node.js 内置性能工具

### 1. Performance Hooks

```javascript
// ===== perf_hooks - 性能测量 =====

import { performance, PerformanceObserver } from 'node:perf_hooks';

// ===== 1. 测量代码执行时间 =====

performance.mark('start');

// 执行一些操作
await someAsyncOperation();

performance.mark('end');

// 测量时间差
performance.measure('Operation Duration', 'start', 'end');

// 获取结果
const measurements = performance.getEntriesByName('Operation Duration');
console.log(`耗时: ${measurements[0].duration}ms`);

// ===== 2. 监听性能条目 =====

const obs = new PerformanceObserver((items) => {
  items.getEntries().forEach((entry) => {
    console.log(`${entry.name}: ${entry.duration}ms`);
  });
});

obs.observe({ entryTypes: ['measure'] });

// ===== 3. 实战：API性能监控 =====

import express from 'express';

const app = express();

// 性能监控中间件
app.use((req, res, next) => {
  const start = `req-${req.url}-start`;
  const end = `req-${req.url}-end`;
  
  performance.mark(start);
  
  // 响应结束时测量
  res.on('finish', () => {
    performance.mark(end);
    performance.measure(`${req.method} ${req.url}`, start, end);
  });
  
  next();
});

// 监听性能数据
const perfObserver = new PerformanceObserver((items) => {
  items.getEntries().forEach((entry) => {
    if (entry.duration > 100) {  // 超过100ms的慢请求
      console.warn(`⚠️ 慢请求: ${entry.name} - ${entry.duration.toFixed(2)}ms`);
    }
  });
});

perfObserver.observe({ entryTypes: ['measure'] });

// ===== 4. 监控事件循环延迟 =====

import { monitorEventLoopDelay } from 'node:perf_hooks';

const h = monitorEventLoopDelay({ resolution: 20 });
h.enable();

setInterval(() => {
  console.log('事件循环延迟统计:');
  console.log(`  最小值: ${h.min}ns`);
  console.log(`  最大值: ${h.max}ns`);
  console.log(`  平均值: ${h.mean}ns`);
  console.log(`  百分位50: ${h.percentile(50)}ns`);
  console.log(`  百分位99: ${h.percentile(99)}ns`);
  
  h.reset();
}, 10000);  // 每10秒统计一次
```

### 2. CPU Profiling

```bash
# ===== 使用 --cpu-prof 生成CPU性能分析 =====

# 运行应用并生成profile
node --cpu-prof app.js

# 生成文件：CPU.20250104.123456.12345.0.001.cpuprofile

# 使用Chrome DevTools查看：
# 1. 打开 chrome://inspect
# 2. 点击 "Open dedicated DevTools for Node"
# 3. 切换到 "Profiler" 标签
# 4. 加载 .cpuprofile 文件
```

```javascript
// ===== 代码中启动/停止profiling =====

import inspector from 'node:inspector';
import fs from 'node:fs';

// 开始profiling
const session = new inspector.Session();
session.connect();

session.post('Profiler.enable', () => {
  session.post('Profiler.start', () => {
    // 执行需要分析的代码
    runExpensiveOperation();
    
    // 停止profiling
    session.post('Profiler.stop', (err, { profile }) => {
      // 保存profile
      fs.writeFileSync('profile.cpuprofile', JSON.stringify(profile));
      console.log('Profile saved to profile.cpuprofile');
    });
  });
});
```

### 2. 实战案例

```javascript
// ===== 性能问题示例 =====

// 问题代码：阻塞事件循环
import express from 'express';

const app = express();

app.get('/slow', (req, res) => {
  // ❌ 同步操作阻塞事件循环
  let sum = 0;
  for (let i = 0; i < 1e9; i++) {
    sum += i;
  }
  res.json({ result: sum });
});

// 诊断：
// clinic doctor -- node app.js
// 然后访问 /slow
// Doctor会显示事件循环延迟警告

// ===== 优化后 =====

import { Worker } from 'worker_threads';

app.get('/fast', (req, res) => {
  // ✅ 使用Worker线程
  const worker = new Worker(`
    const { parentPort } = require('worker_threads');
    let sum = 0;
    for (let i = 0; i < 1e9; i++) {
      sum += i;
    }
    parentPort.postMessage(sum);
  `, { eval: true });
  
  worker.on('message', (sum) => {
    res.json({ result: sum });
  });
});
```

### 2. 编程方式使用

```javascript
// ===== autocannon-script.js =====

import autocannon from 'autocannon';

const instance = autocannon({
  url: 'http://localhost:3000',
  connections: 10,
  duration: 10,
  pipelining: 1
}, (err, result) => {
  if (err) {
    console.error('压测失败:', err);
    return;
  }
  
  console.log('压测结果:');
  console.log(`  请求总数: ${result.requests.total}`);
  console.log(`  平均延迟: ${result.latency.mean}ms`);
  console.log(`  吞吐量: ${result.throughput.mean} bytes/sec`);
  console.log(`  QPS: ${result.requests.mean}`);
});

// 实时输出进度
autocannon.track(instance);
```

## 🎯 实战：完整的性能优化流程

### 1. 诊断性能问题

```javascript
// ===== app.js - 待优化的应用 =====

import express from 'express';
import { performance, PerformanceObserver } from 'node:perf_hooks';

const app = express();
app.use(express.json());

// 性能监控
const obs = new PerformanceObserver((items) => {
  items.getEntries().forEach((entry) => {
    if (entry.duration > 100) {
      console.warn(`⚠️ 慢请求: ${entry.name} - ${entry.duration.toFixed(2)}ms`);
    }
  });
});
obs.observe({ entryTypes: ['measure'] });

// 模拟API
const data = [];

app.get('/api/users', (req, res) => {
  performance.mark('get-users-start');
  
  // ❌ 问题：没有分页，返回所有数据
  const users = data;
  
  performance.mark('get-users-end');
  performance.measure('GET /api/users', 'get-users-start', 'get-users-end');
  
  res.json(users);
});

app.post('/api/users', (req, res) => {
  performance.mark('post-user-start');
  
  const user = req.body;
  data.push(user);
  
  performance.mark('post-user-end');
  performance.measure('POST /api/users', 'post-user-start', 'post-user-end');
  
  res.json(user);
});

app.listen(3000);
```

### 2. 优化代码

```javascript
// ===== app-optimized.js =====

import express from 'express';
import { LRUCache } from 'lru-cache';  // npm install lru-cache

const app = express();
app.use(express.json());

// 添加缓存
const cache = new LRUCache({
  max: 100,
  ttl: 1000 * 60 * 5  // 5分钟
});

const data = [];

app.get('/api/users', (req, res) => {
  // ✅ 添加分页
  const page = parseInt(req.query.page) || 1;
  const limit = parseInt(req.query.limit) || 10;
  const start = (page - 1) * limit;
  const end = start + limit;
  
  // ✅ 使用缓存
  const cacheKey = `users:${page}:${limit}`;
  const cached = cache.get(cacheKey);
  
  if (cached) {
    return res.json(cached);
  }
  
  const users = data.slice(start, end);
  const result = {
    data: users,
    page,
    limit,
    total: data.length
  };
  
  cache.set(cacheKey, result);
  res.json(result);
});

app.post('/api/users', (req, res) => {
  const user = req.body;
  data.push(user);
  
  // ✅ 清除相关缓存
  cache.clear();
  
  res.json(user);
});

app.listen(3000);

// 再次压测：
// autocannon -c 50 -d 30 http://localhost:3000/api/users?page=1&limit=10

// 结果对比：
// 优化前：QPS ~1000
// 优化后：QPS ~5000（提升5倍）
```
