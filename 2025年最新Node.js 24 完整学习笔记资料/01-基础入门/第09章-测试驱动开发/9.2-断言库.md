# 9.2 断言库对比

> 使用 assert 和 Chai 进行测试断言

---

## 🎯 什么是断言？

### 核心概念

```
断言 = 声明某个条件为真

生活中的比喻：
你说："这个苹果是红色的"
检查：苹果真的是红色的吗？
结果：✅ 断言通过 / ❌ 断言失败

代码中的断言：
const result = add(2, 3);
assert.strictEqual(result, 5);  // 断言结果等于5
// ✅ 通过：result确实是5
// ❌ 失败：result不是5，测试失败

常见断言：
• 相等性：a === b
• 类型：typeof x === 'string'
• 真值性：x === true
• 包含：array.includes(item)
• 抛出错误：throws error
```

## 📦 Node.js 内置 assert

### 1. 基础断言

```javascript
import assert from 'node:assert';

// ===== 严格相等 =====

assert.strictEqual(1 + 1, 2);  // ✅ 通过
assert.strictEqual(1 + 1, 3);  // ❌ 失败

// 💡 strictEqual 使用 === 比较
assert.strictEqual('5', 5);    // ❌ 失败（类型不同）

// ===== 深度相等 =====

// 比较对象/数组
assert.deepStrictEqual(
  { name: 'Alice', age: 25 },
  { name: 'Alice', age: 25 }
);  // ✅ 通过

assert.deepStrictEqual([1, 2, 3], [1, 2, 3]);  // ✅ 通过

// 注意：strictEqual 不能比较对象
assert.strictEqual({ a: 1 }, { a: 1 });  // ❌ 失败（不同引用）

// ===== 真值断言 =====

assert.ok(true);         // ✅ 通过
assert.ok(1);            // ✅ 通过（truthy）
assert.ok('hello');      // ✅ 通过（truthy）
assert.ok(false);        // ❌ 失败
assert.ok(0);            // ❌ 失败（falsy）
assert.ok('');           // ❌ 失败（falsy）

// ===== 不相等 =====

assert.notStrictEqual(1, 2);  // ✅ 通过
assert.notStrictEqual(1, 1);  // ❌ 失败
```

### 2. 错误断言

```javascript
import assert from 'node:assert';

// ===== 断言抛出错误 =====

function divide(a, b) {
  if (b === 0) {
    throw new Error('Cannot divide by zero');
  }
  return a / b;
}

// 断言会抛出错误
assert.throws(() => {
  divide(10, 0);
});  // ✅ 通过

// 断言错误消息
assert.throws(
  () => {
    divide(10, 0);
  },
  {
    message: 'Cannot divide by zero'
  }
);  // ✅ 通过

// 断言错误类型
assert.throws(
  () => {
    divide(10, 0);
  },
  Error
);  // ✅ 通过

// ===== 断言不抛出错误 =====

assert.doesNotThrow(() => {
  divide(10, 2);
});  // ✅ 通过

// ===== 异步错误断言 =====

async function fetchData(id) {
  if (id < 0) {
    throw new Error('Invalid ID');
  }
  return { id, data: 'some data' };
}

// 断言异步函数抛出错误
await assert.rejects(
  async () => {
    await fetchData(-1);
  },
  {
    message: 'Invalid ID'
  }
);  // ✅ 通过

// 断言异步函数不抛出错误
await assert.doesNotReject(
  async () => {
    await fetchData(1);
  }
);  // ✅ 通过
```

### 3. 匹配断言

```javascript
import assert from 'node:assert';

// ===== 正则表达式匹配 =====

const email = 'user@example.com';
assert.match(email, /^[^\s@]+@[^\s@]+\.[^\s@]+$/);  // ✅ 通过

const invalid = 'invalid-email';
assert.doesNotMatch(invalid, /@/);  // ✅ 通过（不包含@）
```

## 🎨 assert 完整API

```javascript
import assert from 'node:assert';

// ===== 相等性断言 =====

assert.strictEqual(actual, expected);      // ===
assert.notStrictEqual(actual, expected);   // !==
assert.deepStrictEqual(obj1, obj2);        // 深度比较
assert.notDeepStrictEqual(obj1, obj2);     // 深度不等

// ===== 真值断言 =====

assert.ok(value);       // truthy
assert(value);          // ok() 的简写

// ===== 错误断言 =====

assert.throws(fn, error);              // 同步抛出
assert.doesNotThrow(fn);               // 同步不抛出
assert.rejects(asyncFn, error);        // 异步抛出
assert.doesNotReject(asyncFn);         // 异步不抛出

// ===== 匹配断言 =====

assert.match(string, regexp);          // 匹配
assert.doesNotMatch(string, regexp);   // 不匹配

// ===== 失败断言 =====

assert.fail('Test failed');            // 直接失败
assert.ifError(error);                 // 如果有错误则失败
```

## 🚀 Chai 断言库

### 1. 安装与使用

```bash
# ===== 安装 =====
npm install --save-dev chai
```

```javascript
// ===== 三种断言风格 =====

import { assert, expect, should } from 'chai';

// 1. assert 风格（类似 Node.js assert）
assert.strictEqual(1 + 1, 2);
assert.isTrue(value);
assert.isArray(arr);

// 2. expect 风格（BDD风格，推荐）
expect(1 + 1).to.equal(2);
expect(value).to.be.true;
expect(arr).to.be.an('array');

// 3. should 风格（BDD风格）
should();  // 需要调用以启用
(1 + 1).should.equal(2);
value.should.be.true;
arr.should.be.an('array');
```

### 2. expect 风格（推荐）

```javascript
import { expect } from 'chai';

// ===== 相等性断言 =====

expect(1 + 1).to.equal(2);               // ===
expect('hello').to.equal('hello');

expect({ a: 1 }).to.deep.equal({ a: 1 }); // 深度比较
expect([1, 2, 3]).to.deep.equal([1, 2, 3]);

expect(1).to.not.equal(2);               // 不等

// ===== 类型断言 =====

expect('hello').to.be.a('string');
expect(123).to.be.a('number');
expect(true).to.be.a('boolean');
expect([]).to.be.an('array');
expect({}).to.be.an('object');
expect(null).to.be.null;
expect(undefined).to.be.undefined;

// ===== 真值断言 =====

expect(true).to.be.true;
expect(false).to.be.false;
expect(1).to.be.ok;                      // truthy
expect(0).to.not.be.ok;                  // falsy

// ===== 存在性断言 =====

expect(value).to.exist;                  // 不是 null/undefined
expect(null).to.not.exist;

// ===== 包含断言 =====

expect([1, 2, 3]).to.include(2);
expect('hello world').to.include('world');
expect({ a: 1, b: 2 }).to.include({ a: 1 });

// ===== 长度断言 =====

expect([1, 2, 3]).to.have.lengthOf(3);
expect('hello').to.have.lengthOf(5);

// ===== 范围断言 =====

expect(5).to.be.above(3);                // >
expect(5).to.be.at.least(5);             // >=
expect(5).to.be.below(10);               // <
expect(5).to.be.at.most(5);              // <=
expect(5).to.be.within(1, 10);           // 1 <= x <= 10

// ===== 属性断言 =====

const user = { name: 'Alice', age: 25 };

expect(user).to.have.property('name');
expect(user).to.have.property('age', 25);
expect(user).to.have.all.keys('name', 'age');

// ===== 错误断言 =====

expect(() => {
  throw new Error('oops');
}).to.throw();

expect(() => {
  throw new Error('oops');
}).to.throw('oops');

expect(() => {
  throw new Error('oops');
}).to.throw(Error);

// ===== 异步断言 =====

await expect(
  fetchData()
).to.eventually.equal('data');

await expect(
  fetchError()
).to.be.rejected;

await expect(
  fetchError()
).to.be.rejectedWith('Error message');
```

### 3. 链式断言

```javascript
import { expect } from 'chai';

// ===== 可读性更强的链式断言 =====

expect([1, 2, 3])
  .to.be.an('array')
  .that.includes(2)
  .and.has.lengthOf(3);

expect('hello world')
  .to.be.a('string')
  .that.includes('world')
  .and.has.lengthOf(11);

expect({ name: 'Alice', age: 25 })
  .to.be.an('object')
  .that.has.property('name')
  .that.equals('Alice');

// ===== 否定断言 =====

expect(5).to.not.equal(3);
expect([1, 2, 3]).to.not.include(4);
expect('hello').to.not.be.empty;
```

## 🎯 实战对比

### assert vs Chai

```javascript
// ===== 场景1：基本相等 =====

// assert
import assert from 'node:assert';
assert.strictEqual(1 + 1, 2);

// Chai
import { expect } from 'chai';
expect(1 + 1).to.equal(2);

// ===== 场景2：对象比较 =====

// assert
assert.deepStrictEqual(
  { name: 'Alice' },
  { name: 'Alice' }
);

// Chai（更简洁）
expect({ name: 'Alice' }).to.deep.equal({ name: 'Alice' });

// ===== 场景3：数组包含 =====

// assert（需要手动检查）
const arr = [1, 2, 3];
assert.ok(arr.includes(2));

// Chai（更直观）
expect([1, 2, 3]).to.include(2);

// ===== 场景4：类型检查 =====

// assert
assert.strictEqual(typeof value, 'string');

// Chai（更语义化）
expect(value).to.be.a('string');

// ===== 场景5：错误断言 =====

// assert
assert.throws(
  () => { throw new Error('oops'); },
  { message: 'oops' }
);

// Chai
expect(() => {
  throw new Error('oops');
}).to.throw('oops');

// ===== 场景6：链式断言 =====

// assert（需要多个语句）
assert.ok(Array.isArray(arr));
assert.strictEqual(arr.length, 3);
assert.ok(arr.includes(2));

// Chai（一条链）
expect(arr)
  .to.be.an('array')
  .with.lengthOf(3)
  .that.includes(2);
```

## 💡 选择建议

```
何时使用 assert：
✅ 简单项目
✅ 不想安装额外依赖
✅ 基本的断言就够用

何时使用 Chai：
✅ 需要更好的可读性
✅ 复杂的断言逻辑
✅ 喜欢BDD风格
✅ 需要插件扩展

推荐组合：
• 简单测试：Node.js assert
• 复杂测试：Chai expect
• 团队项目：Chai（更统一）
```

## 🎯 完整示例

```javascript
// ===== 使用 Chai 测试用户类 =====

import { expect } from 'chai';
import { describe, test } from 'node:test';

class User {
  constructor(name, email, age) {
    if (!name || name.length < 2) {
      throw new Error('Name must be at least 2 characters');
    }
    
    if (!email || !email.includes('@')) {
      throw new Error('Invalid email');
    }
    
    if (age < 0 || age > 150) {
      throw new Error('Invalid age');
    }
    
    this.name = name;
    this.email = email;
    this.age = age;
    this.createdAt = new Date();
  }
  
  isAdult() {
    return this.age >= 18;
  }
  
  getDomain() {
    return this.email.split('@')[1];
  }
}

describe('User 类（使用 Chai）', () => {
  test('应该成功创建用户', () => {
    const user = new User('Alice', 'alice@example.com', 25);
    
    expect(user).to.be.an('object');
    expect(user).to.have.property('name', 'Alice');
    expect(user).to.have.property('email', 'alice@example.com');
    expect(user).to.have.property('age', 25);
    expect(user.createdAt).to.be.a('date');
  });
  
  test('应该正确验证成年人', () => {
    const adult = new User('Alice', 'alice@example.com', 25);
    const child = new User('Bob', 'bob@example.com', 10);
    
    expect(adult.isAdult()).to.be.true;
    expect(child.isAdult()).to.be.false;
  });
  
  test('应该正确提取域名', () => {
    const user = new User('Alice', 'alice@example.com', 25);
    
    expect(user.getDomain()).to.equal('example.com');
  });
  
  test('应该拒绝无效的名字', () => {
    expect(() => {
      new User('A', 'alice@example.com', 25);
    }).to.throw('Name must be at least 2 characters');
  });
  
  test('应该拒绝无效的邮箱', () => {
    expect(() => {
      new User('Alice', 'invalid-email', 25);
    }).to.throw('Invalid email');
  });
  
  test('应该拒绝无效的年龄', () => {
    expect(() => {
      new User('Alice', 'alice@example.com', -5);
    }).to.throw('Invalid age');
    
    expect(() => {
      new User('Alice', 'alice@example.com', 200);
    }).to.throw('Invalid age');
  });
});
```
