# 9.4 为CLI工具编写测试

> 测试命令行工具的完整方案

---

## 🎯 CLI测试的挑战

### 核心概念

```
CLI测试 ≠ 普通函数测试

特殊之处：
• 需要执行实际的命令
• 需要捕获stdout/stderr
• 需要测试用户交互
• 需要测试文件操作
• 需要测试退出码

测试层级：
1. 单元测试 - 测试工具函数
2. 集成测试 - 测试命令执行
3. E2E测试 - 测试完整流程
```

// ===== 基础使用 =====

import { execa } from 'execa';

// 执行命令
const { stdout, stderr, exitCode } = await execa('node', ['cli.js', '--version']);

console.log('输出:', stdout);

const result = await execa('node', ['cli.js', 'create', 'my-app'], {
  cwd: '/path/to/directory',     // 工作目录
  env: { NODE_ENV: 'test' },     // 环境变量
  input: 'user input\n',         // 标准输入
  timeout: 5000,                 // 超时（毫秒）
  reject: false                  // 不抛出错误
});

// ===== 捕获错误 =====

try {
  await execa('node', ['cli.js', 'invalid-command']);
} catch (error) {
  console.log('退出码:', error.exitCode);
  console.log('标准输出:', error.stdout);
  console.log('错误输出:', error.stderr);
}

### 测试工具函数

```javascript
// ===== lib/utils.js =====

export function validateProjectName(name) {
  if (!name || name.length === 0) {
    throw new Error('Project name is required');
  }
  
  if (!/^[a-z0-9-_]+$/.test(name)) {
    throw new Error('Invalid project name format');
  }
  
  if (name.length > 50) {
    throw new Error('Project name too long (max 50)');
  }
  
  return true;
}

export function formatBytes(bytes) {
  if (bytes === 0) return '0 B';
  
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  
  return (bytes / Math.pow(k, i)).toFixed(2) + ' ' + sizes[i];
}

// ===== tests/utils.test.js =====

import { describe, test, expect } from 'vitest';
import { validateProjectName, formatBytes } from '../lib/utils.js';

describe('validateProjectName()', () => {
  test('应该接受有效的项目名', () => {
    expect(() => validateProjectName('my-app')).not.toThrow();
    expect(() => validateProjectName('my_app')).not.toThrow();
    expect(() => validateProjectName('app123')).not.toThrow();
  });
  
  test('应该拒绝空名称', () => {
    expect(() => validateProjectName('')).toThrow('required');
    expect(() => validateProjectName(null)).toThrow('required');
  });
  
  test('应该拒绝无效字符', () => {
    expect(() => validateProjectName('My App')).toThrow('Invalid');
    expect(() => validateProjectName('my@app')).toThrow('Invalid');
  });
  
  test('应该拒绝过长的名称', () => {
    const longName = 'a'.repeat(51);
    expect(() => validateProjectName(longName)).toThrow('too long');
  });
});

describe('formatBytes()', () => {
  test('应该正确格式化字节', () => {
    expect(formatBytes(0)).toBe('0 B');
    expect(formatBytes(1024)).toBe('1.00 KB');
    expect(formatBytes(1048576)).toBe('1.00 MB');
    expect(formatBytes(1073741824)).toBe('1.00 GB');
  });
  
  test('应该处理小数', () => {
    expect(formatBytes(1536)).toBe('1.50 KB');
    expect(formatBytes(2621440)).toBe('2.50 MB');
  });
});
```

## 🎮 集成测试

### 测试CLI命令

```javascript
// ===== tests/cli.test.js =====

import { describe, test, expect, beforeEach, afterEach } from 'vitest';
import { execa } from 'execa';
import fs from 'fs-extra';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const CLI_PATH = path.join(__dirname, '../bin/cli.js');
const TEST_DIR = path.join(__dirname, '.test-projects');

describe('CLI 集成测试', () => {
  beforeEach(async () => {
    // 创建测试目录
    await fs.ensureDir(TEST_DIR);
  });
  
  afterEach(async () => {
    // 清理测试目录
    await fs.remove(TEST_DIR);
  });
  
  test('应该显示版本号', async () => {
    const { stdout } = await execa('node', [CLI_PATH, '--version']);
    
    // 检查输出包含版本号
    expect(stdout).toMatch(/\d+\.\d+\.\d+/);
  });
  
  test('应该显示帮助信息', async () => {
    const { stdout } = await execa('node', [CLI_PATH, '--help']);
    
    expect(stdout).toContain('Usage:');
    expect(stdout).toContain('Commands:');
    expect(stdout).toContain('Options:');
  });
  
  test('应该创建项目', async () => {
    const projectName = 'test-app';
    const projectPath = path.join(TEST_DIR, projectName);
    
    // 执行create命令
    const { stdout, exitCode } = await execa('node', [
      CLI_PATH,
      'create',
      projectPath,
      '--template', 'react',
      '--skip-install',
      '--skip-git'
    ]);
    
    // 检查退出码
    expect(exitCode).toBe(0);
    
    // 检查输出
    expect(stdout).toContain('created successfully');
    
    // 检查文件是否创建
    expect(await fs.pathExists(projectPath)).toBe(true);
    expect(await fs.pathExists(path.join(projectPath, 'package.json'))).toBe(true);
    expect(await fs.pathExists(path.join(projectPath, 'README.md'))).toBe(true);
    
    // 检查 package.json 内容
    const pkg = await fs.readJson(path.join(projectPath, 'package.json'));
    expect(pkg.name).toBe(projectName);
    expect(pkg.version).toBe('1.0.0');
  }, 30000);  // 30秒超时
  
  test('应该拒绝无效的项目名', async () => {
    try {
      await execa('node', [CLI_PATH, 'create', 'Invalid Name']);
      // 不应该执行到这里
      expect(true).toBe(false);
    } catch (error) {
      expect(error.exitCode).toBe(1);
      expect(error.stderr).toContain('Invalid');
    }
  });
  
  test('应该拒绝已存在的目录', async () => {
    const projectName = 'existing-app';
    const projectPath = path.join(TEST_DIR, projectName);
    
    // 创建已存在的目录
    await fs.ensureDir(projectPath);
    
    try {
      await execa('node', [
        CLI_PATH,
        'create',
        projectPath,
        '--skip-install',
        '--skip-git'
      ], {
        input: 'n\n'  // 回答 No
      });
      expect(true).toBe(false);
    } catch (error) {
      // 应该取消操作
      expect(error.stdout).toContain('cancelled');
    }
  });
  
  test('应该用force选项覆盖已存在的目录', async () => {
    const projectName = 'force-app';
    const projectPath = path.join(TEST_DIR, projectName);
    
    // 创建已存在的目录和文件
    await fs.ensureDir(projectPath);
    await fs.writeFile(path.join(projectPath, 'old-file.txt'), 'old content');
    
    // 使用 --force 选项
    const { exitCode } = await execa('node', [
      CLI_PATH,
      'create',
      projectPath,
      '--force',
      '--skip-install',
      '--skip-git'
    ]);
    
    expect(exitCode).toBe(0);
    
    // 检查旧文件被删除
    expect(await fs.pathExists(path.join(projectPath, 'old-file.txt'))).toBe(false);
    
    // 检查新文件被创建
    expect(await fs.pathExists(path.join(projectPath, 'package.json'))).toBe(true);
  });
});
```

## 🎯 E2E测试

### 完整流程测试

```javascript
// ===== tests/e2e.test.js =====

import { describe, test, expect, beforeAll, afterAll } from 'vitest';
import { execa } from 'execa';
import fs from 'fs-extra';
import path from 'path';

const CLI_PATH = path.join(__dirname, '../bin/cli.js');
const TEST_DIR = path.join(__dirname, '.e2e-test');

describe('E2E 测试：完整创建流程', () => {
  beforeAll(async () => {
    await fs.ensureDir(TEST_DIR);
  });
  
  afterAll(async () => {
    await fs.remove(TEST_DIR);
  });
  
  test('完整的项目创建流程', async () => {
    const projectName = 'full-test-app';
    const projectPath = path.join(TEST_DIR, projectName);
    
    // 1. 创建项目
    console.log('步骤1: 创建项目...');
    const createResult = await execa('node', [
      CLI_PATH,
      'create',
      projectPath,
      '--template', 'react',
      '--skip-install',
      '--skip-git'
    ]);
    
    expect(createResult.exitCode).toBe(0);
    expect(createResult.stdout).toContain('created successfully');
    
    // 2. 验证文件结构
    console.log('步骤2: 验证文件结构...');
    const files = await fs.readdir(projectPath);
    expect(files).toContain('package.json');
    expect(files).toContain('README.md');
    expect(files).toContain('src');
    
    // 3. 验证 package.json
    console.log('步骤3: 验证 package.json...');
    const pkg = await fs.readJson(path.join(projectPath, 'package.json'));
    expect(pkg.name).toBe(projectName);
    expect(pkg.scripts).toHaveProperty('dev');
    expect(pkg.scripts).toHaveProperty('build');
    
    // 4. 验证 README.md
    console.log('步骤4: 验证 README.md...');
    const readme = await fs.readFile(path.join(projectPath, 'README.md'), 'utf8');
    expect(readme).toContain(projectName);
    expect(readme).toContain('Getting Started');
    
    // 5. 验证源码目录
    console.log('步骤5: 验证源码目录...');
    const srcExists = await fs.pathExists(path.join(projectPath, 'src'));
    expect(srcExists).toBe(true);
    
    console.log('✅ 所有测试通过!');
  }, 60000);  // 60秒超时
});
```
