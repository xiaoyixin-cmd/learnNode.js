# 9.1 Node.js 内置测试运行器

> Node.js 18+ 内置的测试框架 `node:test`

---

## 🎯 为什么需要测试？

### 核心概念

```
测试 = 确保代码按预期工作的自动化检查

没有测试的代码：
❌ 修改代码后不敢重构
❌ 不知道哪里会出bug
❌ 上线后才发现问题
❌ 团队协作困难

有测试的代码：
✅ 修改后立即知道是否破坏了功能
✅ 重构时有信心
✅ 文档作用（测试即文档）
✅ 提高代码质量

测试金字塔：
       /\
      /E2E\      端到端测试（少量）
     /------\
    /集成测试\   集成测试（中量）
   /----------\
  /  单元测试  \  单元测试（大量）
 /--------------\
```


```javascript
// ===== math.js =====

export function add(a, b) {
  return a + b;
}

export function subtract(a, b) {
  return a - b;
}

  return a * b;

export function divide(a, b) {
  if (b === 0) {

  const result = add(2, 3);
  assert.strictEqual(result, 5);
});

  const result = subtract(5, 3);
  assert.strictEqual(result, 2);
});

test('multiply() 应该正确相乘两个数', () => {
  const result = multiply(3, 4);
  assert.strictEqual(result, 12);
});

test('divide() 应该正确相除两个数', () => {
  const result = divide(10, 2);
  assert.strictEqual(result, 5);

  assert.throws(() => {
```

## 🔧 测试API

### 1. test() 函数

```javascript
import { test } from 'node:test';
import assert from 'node:assert';

// ===== 基本用法 =====

test('测试名称', () => {
  // 测试代码
  assert.strictEqual(1 + 1, 2);
});

// ===== 异步测试 =====

test('异步测试', async () => {
  const data = await fetchData();
  assert.strictEqual(data.status, 'ok');
});

// ===== 跳过测试 =====

test('这个测试会被跳过', { skip: true }, () => {
  // 不会执行
});

test.skip('这个也会被跳过', () => {
  // 不会执行
});

// ===== 仅运行这个测试 =====

test('普通测试', () => {
  // 不会运行
});

test.only('只运行这个测试', () => {
  // 只有这个会运行
});

// ===== 待办测试（TODO）=====

test.todo('待实现的功能');
// 会显示为待办，不算失败

// ===== 超时设置 =====

test('长时间运行的测试', { timeout: 5000 }, async () => {
  // 5秒超时
  await longRunningTask();
});
```

### 2. describe() 分组

```javascript
import { describe, test } from 'node:test';
import assert from 'node:assert';

// ===== 测试分组 =====

describe('用户模块', () => {
  test('应该创建用户', () => {
    const user = createUser('Alice');
    assert.strictEqual(user.name, 'Alice');

  test('应该创建订单', () => {
    const order = createOrder(123);
    assert.ok(order.id);
  });
  
  test('应该取消订单', () => {
    const result = cancelOrder(123);
    assert.strictEqual(result, true);
  });
});

// ===== 嵌套分组 =====

describe('数组方法', () => {
  describe('push()', () => {
    test('应该在末尾添加元素', () => {
      const arr = [1, 2, 3];
      arr.push(4);
      assert.deepStrictEqual(arr, [1, 2, 3, 4]);
    });
  });
  
  describe('pop()', () => {
    test('应该删除并返回最后一个元素', () => {
      const arr = [1, 2, 3];
      const last = arr.pop();
      assert.strictEqual(last, 3);
      assert.deepStrictEqual(arr, [1, 2]);
    });
  });
});
```

### 3. before/after 钩子

```javascript
import { describe, test, before, after, beforeEach, afterEach } from 'node:test';
import assert from 'node:assert';

// ===== 生命周期钩子 =====

describe('数据库测试', () => {
  let db;
  
  // 所有测试前执行一次
  before(async () => {
    console.log('连接数据库...');
    db = await connectDatabase();
  });
  
  // 所有测试后执行一次
  after(async () => {
    console.log('关闭数据库...');
    await db.close();
  });
  
  // 每个测试前执行
  beforeEach(async () => {
    console.log('清空测试数据...');
    await db.clear();
  });
  
  // 每个测试后执行
  afterEach(async () => {
    console.log('测试完成');
  });
  
  test('应该插入数据', async () => {
    await db.insert({ name: 'Alice' });
    const count = await db.count();
    assert.strictEqual(count, 1);
  });
  
  test('应该查询数据', async () => {
    await db.insert({ name: 'Bob' });
    const user = await db.findOne({ name: 'Bob' });
    assert.strictEqual(user.name, 'Bob');
  });
});

// 执行顺序：
// 1. before       (1次)
// 2. beforeEach   (测试1)
// 3. test 1
// 4. afterEach    (测试1)
// 5. beforeEach   (测试2)
// 6. test 2
// 7. afterEach    (测试2)
// 8. after        (1次)
```

## 🎯 实战案例

### 案例1：测试工具函数

```javascript
// ===== utils.js =====

export function isEmail(email) {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
}

export function formatDate(date) {
  const d = new Date(date);
  const year = d.getFullYear();
  const month = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');
  return `${year}-${month}-${day}`;
}

export function capitalize(str) {
  if (!str) return '';
  return str.charAt(0).toUpperCase() + str.slice(1).toLowerCase();
}

// ===== utils.test.js =====

import { describe, test } from 'node:test';
import assert from 'node:assert';
import { isEmail, formatDate, capitalize } from './utils.js';

describe('isEmail()', () => {
  test('应该验证有效的邮箱', () => {
    assert.strictEqual(isEmail('user@example.com'), true);
    assert.strictEqual(isEmail('test.user@company.co.uk'), true);
  });
  
  test('应该拒绝无效的邮箱', () => {
    assert.strictEqual(isEmail('invalid'), false);
    assert.strictEqual(isEmail('@example.com'), false);
    assert.strictEqual(isEmail('user@'), false);
    assert.strictEqual(isEmail(''), false);
  });
});

describe('formatDate()', () => {
  test('应该正确格式化日期', () => {
    const result = formatDate('2025-01-05');
    assert.strictEqual(result, '2025-01-05');
  });
  
  test('应该处理单位数的月份和日期', () => {
    const result = formatDate('2025-3-5');
    assert.strictEqual(result, '2025-03-05');
  });
});

describe('capitalize()', () => {
  test('应该首字母大写', () => {
    assert.strictEqual(capitalize('hello'), 'Hello');
    assert.strictEqual(capitalize('WORLD'), 'World');
  });
  
  test('应该处理空字符串', () => {
    assert.strictEqual(capitalize(''), '');
    assert.strictEqual(capitalize(null), '');
  });
});
```

### 案例2：测试异步函数

```javascript
// ===== api.js =====

export async function fetchUser(id) {
  const response = await fetch(`https://api.example.com/users/${id}`);
  
  if (!response.ok) {
    throw new Error(`HTTP ${response.status}`);
  }
  
  return response.json();
}

export async function createUser(data) {
  const response = await fetch('https://api.example.com/users', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });
  
  return response.json();
}

// ===== api.test.js =====

import { describe, test } from 'node:test';
import assert from 'node:assert';
import { fetchUser, createUser } from './api.js';

describe('API 测试', () => {
  test('fetchUser() 应该获取用户信息', async () => {
    const user = await fetchUser(1);
    
    assert.ok(user);
    assert.ok(user.id);
    assert.ok(user.name);
  });
  
  test('fetchUser() 找不到用户时应该抛出错误', async () => {
    await assert.rejects(
      async () => {
        await fetchUser(999999);
      },
      {
        message: /HTTP 404/
      }
    );
  });
  
  test('createUser() 应该创建新用户', async () => {
    const newUser = {
      name: 'Alice',
      email: 'alice@example.com'
    };
    
    const user = await createUser(newUser);
    
    assert.ok(user.id);
    assert.strictEqual(user.name, 'Alice');
    assert.strictEqual(user.email, 'alice@example.com');
  });
});
```

### 案例3：测试类

```javascript
// ===== user.js =====

export class User {
  constructor(name, email) {
    this.name = name;
    this.email = email;
    this.createdAt = new Date();
  }
  
  get displayName() {
    return this.name.toUpperCase();
  }
  
  isAdmin() {
    return this.email.endsWith('@admin.com');
  }
  
  static validate(data) {
    if (!data.name || data.name.length < 2) {
      throw new Error('Name must be at least 2 characters');
    }
    
    if (!data.email || !data.email.includes('@')) {
      throw new Error('Invalid email');
    }
    
    return true;
  }
}

// ===== user.test.js =====

import { describe, test } from 'node:test';
import assert from 'node:assert';
import { User } from './user.js';

describe('User 类', () => {
  test('应该创建用户实例', () => {
    const user = new User('Alice', 'alice@example.com');
    
    assert.strictEqual(user.name, 'Alice');
    assert.strictEqual(user.email, 'alice@example.com');
    assert.ok(user.createdAt instanceof Date);
  });
  
  test('displayName 应该返回大写名字', () => {
    const user = new User('Alice', 'alice@example.com');
    assert.strictEqual(user.displayName, 'ALICE');
  });
  
  test('isAdmin() 应该正确识别管理员', () => {
    const admin = new User('Admin', 'admin@admin.com');
    const user = new User('Alice', 'alice@example.com');
    
    assert.strictEqual(admin.isAdmin(), true);
    assert.strictEqual(user.isAdmin(), false);
  });
  
  describe('User.validate()', () => {
    test('应该验证有效的用户数据', () => {
      const data = { name: 'Alice', email: 'alice@example.com' };
      assert.strictEqual(User.validate(data), true);
    });
    
    test('应该拒绝无效的名字', () => {
      assert.throws(() => {
        User.validate({ name: 'A', email: 'alice@example.com' });
      }, {
        message: 'Name must be at least 2 characters'
      });
    });
    
    test('应该拒绝无效的邮箱', () => {
      assert.throws(() => {
        User.validate({ name: 'Alice', email: 'invalid' });
      }, {
        message: 'Invalid email'
      });
    });
  });
});
```

## 🚀 运行测试

### 1. 命令行选项

```bash
# ===== 基本命令 =====

# 运行单个文件
node --test math.test.js

# 运行目录下所有测试
node --test

# 运行匹配模式的测试
node --test **/*.test.js

# ===== 常用选项 =====

# 显示详细输出
node --test --test-reporter=spec

# 生成 TAP 格式输出
node --test --test-reporter=tap

# 监听模式（文件变化时自动重跑）
node --test --watch

# 并发运行（更快）
node --test --test-concurrency=4

# 指定超时时间（毫秒）
node --test --test-timeout=5000

# 仅运行匹配的测试
node --test --test-name-pattern="用户"

# 查看覆盖率（需要 Node.js 20+）
node --test --experimental-test-coverage
```

### 2. package.json 配置

```json
{
  "scripts": {
    "test": "node --test",
    "test:watch": "node --test --watch",
    "test:coverage": "node --test --experimental-test-coverage"
  }
}
```
