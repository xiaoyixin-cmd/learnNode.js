# 4.2 Stream：流式编程

> 掌握 Node.js 流式处理，高效处理大规模数据

---

## 🌊 Stream 是什么？

### 流的概念

```
Stream（流）= 数据的连续传输
├─ 不需要一次性加载所有数据到内存
├─ 数据以"块"（chunk）为单位处理
├─ 类似水流：一点一点地流动
└─ 适合处理大文件、网络传输等场景
```

### 为什么需要 Stream？

```javascript
// ===== ❌ 不使用 Stream：一次性读取（内存爆炸）=====

import fs from 'fs';

// 读取 1GB 文件
const data = fs.readFileSync('large-file.txt');  // 占用 1GB 内存
console.log(data.length);

// 问题：
// - 内存占用过大
// - 文件过大时会崩溃
// - 无法处理超过内存大小的文件

// ===== ✅ 使用 Stream：分块读取（内存友好）=====

const stream = fs.createReadStream('large-file.txt');

stream.on('data', (chunk) => {
  console.log('Received', chunk.length, 'bytes');
  // 每次只处理一小块数据（默认 64KB）
});

stream.on('end', () => {
  console.log('Done');
});

// 优势：
// ✅ 内存占用固定（只有一个 chunk 的大小）
// ✅ 可以处理任意大小的文件
// ✅ 边读边处理，效率高
```

## 🎯 四种流类型

```
Node.js Stream 类型：

1. Readable    → 可读流（读取数据）
2. Writable    → 可写流（写入数据）
3. Duplex      → 双工流（可读可写）
4. Transform   → 转换流（读取、转换、写入）
```

### 1. Readable Stream（可读流）

```javascript
import fs from 'fs';

// ===== 创建可读流 =====

const readable = fs.createReadStream('input.txt', {
  encoding: 'utf8',
  highWaterMark: 64 * 1024  // 64KB 缓冲区
});

// ===== 事件：'data' =====

readable.on('data', (chunk) => {
  console.log('Received chunk:', chunk);
});

// ===== 事件：'end' =====

readable.on('end', () => {
  console.log('No more data');
});

// ===== 事件：'error' =====

readable.on('error', (err) => {
  console.error('Error:', err);
});

// ===== 事件：'close' =====

readable.on('close', () => {
  console.log('Stream closed');
});

// ===== 暂停和恢复 =====

readable.on('data', (chunk) => {
  console.log('Chunk:', chunk);
  
  // 暂停
  readable.pause();
  
  // 1 秒后恢复
  setTimeout(() => {
    readable.resume();
  }, 1000);
});
```

### 2. Writable Stream（可写流）

```javascript
import fs from 'fs';

// ===== 创建可写流 =====

const writable = fs.createWriteStream('output.txt', {
  encoding: 'utf8'
});

// ===== write() 方法 =====

const canContinue = writable.write('Hello ');
console.log('Can continue?', canContinue);

writable.write('World\n');

// ===== end() 方法：结束写入 =====

writable.end('Final data');

// 等同于：
// writable.write('Final data');
// writable.end();

// ===== 事件：'drain' =====

writable.on('drain', () => {
  console.log('Buffer drained, can write again');
});

// ===== 事件：'finish' =====

writable.on('finish', () => {
  console.log('All data has been written');
});

// ===== 事件：'error' =====

writable.on('error', (err) => {
  console.error('Write error:', err);
});

// ===== 背压处理（重要）=====

function writeData(writer, data, callback) {
  let i = 0;
  
  function write() {
    let ok = true;
    
    while (i < data.length && ok) {
      ok = writer.write(data[i]);
      i++;
    }
    
    if (i < data.length) {
      // 缓冲区满了，等待 drain 事件
      writer.once('drain', write);
    } else {
      writer.end();
      callback();
    }
  }
  
  write();
}

const data = Array.from({ length: 10000 }, (_, i) => `Line ${i}\n`);

writeData(writable, data, () => {
  console.log('All data written');
});
```

### 3. Duplex Stream（双工流）

```javascript
import { Duplex } from 'stream';

// ===== 创建双工流 =====

class MyDuplex extends Duplex {
  constructor(options) {
    super(options);
    this.data = [];
  }
  
  // 实现 _read 方法
  _read(size) {
    if (this.data.length === 0) {
      this.push(null);  // 没有数据了
    } else {
      this.push(this.data.shift());
    }
  }
  
  // 实现 _write 方法
  _write(chunk, encoding, callback) {
    console.log('Writing:', chunk.toString());
    this.data.push(chunk);
    callback();
  }
}

const duplex = new MyDuplex();

// 写入数据
duplex.write('Hello\n');
duplex.write('World\n');
duplex.end();

// 读取数据
duplex.on('data', (chunk) => {
  console.log('Reading:', chunk.toString());
});

// ===== 网络 Socket 是双工流 =====

import net from 'net';

const server = net.createServer((socket) => {
  // socket 是 Duplex Stream
  
  // 读取客户端数据
  socket.on('data', (data) => {
    console.log('Received:', data.toString());
    
    // 写入响应
    socket.write('Echo: ' + data);
  });
});

server.listen(3000);
```

### 4. Transform Stream（转换流）

```javascript
import { Transform } from 'stream';
import fs from 'fs';

// ===== 创建转换流 =====

class UpperCaseTransform extends Transform {
  _transform(chunk, encoding, callback) {
    // 转换数据
    const upperChunk = chunk.toString().toUpperCase();
    this.push(upperChunk);
    callback();
  }
}

// 使用
const readable = fs.createReadStream('input.txt');
const transform = new UpperCaseTransform();
const writable = fs.createWriteStream('output.txt');

readable
  .pipe(transform)
  .pipe(writable);

// ===== 压缩文件（Transform Stream）=====

import zlib from 'zlib';

fs.createReadStream('input.txt')
  .pipe(zlib.createGzip())
  .pipe(fs.createWriteStream('input.txt.gz'));

console.log('File compressed');

// ===== 解压文件 =====

fs.createReadStream('input.txt.gz')
  .pipe(zlib.createGunzip())
  .pipe(fs.createWriteStream('output.txt'));

// ===== 加密文件 =====

import crypto from 'crypto';

const cipher = crypto.createCipher('aes-256-cbc', 'password');

fs.createReadStream('input.txt')
  .pipe(cipher)
  .pipe(fs.createWriteStream('encrypted.txt'));
```

## 🔗 管道（Pipe）操作

### pipe() 方法

```javascript
// ===== 基础管道 =====

import fs from 'fs';

const readable = fs.createReadStream('input.txt');
const writable = fs.createWriteStream('output.txt');

readable.pipe(writable);

// 等同于：
readable.on('data', (chunk) => {
  writable.write(chunk);
});

readable.on('end', () => {
  writable.end();
});

// ===== 链式管道 =====

import zlib from 'zlib';

fs.createReadStream('input.txt')
  .pipe(zlib.createGzip())
  .pipe(fs.createWriteStream('output.txt.gz'));

// ===== 多个目标 =====

const readable = fs.createReadStream('input.txt');

readable.pipe(fs.createWriteStream('copy1.txt'));
readable.pipe(fs.createWriteStream('copy2.txt'));
readable.pipe(process.stdout);  // 同时输出到控制台

// ===== 错误处理 =====

readable
  .on('error', (err) => console.error('Read error:', err))
  .pipe(writable)
  .on('error', (err) => console.error('Write error:', err));
```

### pipeline() 方法（推荐）

```javascript
import { pipeline } from 'stream/promises';
import fs from 'fs';
import zlib from 'zlib';

// ===== 使用 pipeline（自动错误处理）=====

try {
  await pipeline(
    fs.createReadStream('input.txt'),
    zlib.createGzip(),
    fs.createWriteStream('output.txt.gz')
  );
  console.log('Pipeline succeeded');
} catch (err) {
  console.error('Pipeline failed:', err);
}

// ===== 自定义 Transform =====

await pipeline(
  fs.createReadStream('input.txt'),
  async function* (source) {
    for await (const chunk of source) {
      yield chunk.toString().toUpperCase();
    }
  },
  fs.createWriteStream('output.txt')
);

// ===== 复杂管道 =====

import { Transform } from 'stream';

class LineCounter extends Transform {
  constructor() {
    super();
    this.lineCount = 0;
  }
  
  _transform(chunk, encoding, callback) {
    const lines = chunk.toString().split('\n').length - 1;
    this.lineCount += lines;
    this.push(chunk);
    callback();
  }
  
  _flush(callback) {
    console.log(`Total lines: ${this.lineCount}`);
    callback();
  }
}

await pipeline(
  fs.createReadStream('input.txt'),
  new LineCounter(),
  fs.createWriteStream('output.txt')
);
```

## 💧 背压（Backpressure）处理

### 什么是背压？

```
背压（Backpressure）：
├─ 数据生产速度 > 数据消费速度
├─ 导致内存占用不断增长
├─ 最终可能导致内存溢出
└─ 需要暂停生产者，等待消费者
```

### 正确处理背压

```javascript
// ===== ❌ 错误：不处理背压 =====

function badCopy(src, dest) {
  const readable = fs.createReadStream(src);
  const writable = fs.createWriteStream(dest);
  
  readable.on('data', (chunk) => {
    writable.write(chunk);  // 忽略返回值
  });
  
  // 问题：如果写入速度慢，内存会不断增长
}

// ===== ✅ 正确：处理背压 =====

function goodCopy(src, dest) {
  const readable = fs.createReadStream(src);
  const writable = fs.createWriteStream(dest);
  
  readable.on('data', (chunk) => {
    const canContinue = writable.write(chunk);
    
    if (!canContinue) {
      // 缓冲区满了，暂停读取
      readable.pause();
    }
  });
  
  writable.on('drain', () => {
    // 缓冲区空了，恢复读取
    readable.resume();
  });
  
  readable.on('end', () => {
    writable.end();
  });
}

// ===== ✅ 最佳：使用 pipe（自动处理背压）=====

function bestCopy(src, dest) {
  const readable = fs.createReadStream(src);
  const writable = fs.createWriteStream(dest);
  
  readable.pipe(writable);  // 自动处理背压
}
```

## 🎯 实战案例

### 案例1：大文件处理

```javascript
// ===== 统计大文件行数 =====

import fs from 'fs';
import readline from 'readline';

async function countLines(filePath) {
  const fileStream = fs.createReadStream(filePath);
  
  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity
  });
  
  let count = 0;
  
  for await (const line of rl) {
    count++;
  }
  
  return count;
}

const lines = await countLines('large-file.txt');
console.log(`Total lines: ${lines}`);
```

### 案例2：HTTP 文件下载（流式）

```javascript
import http from 'http';
import fs from 'fs';

http.createServer((req, res) => {
  if (req.url === '/download') {
    const stream = fs.createReadStream('large-file.mp4');
    
    res.writeHead(200, {
      'Content-Type': 'video/mp4',
      'Content-Disposition': 'attachment; filename="file.mp4"'
    });
    
    stream.pipe(res);
    
    stream.on('error', (err) => {
      res.statusCode = 500;
      res.end('Server Error');
    });
  }
}).listen(3000);
```

### 案例3：CSV 文件处理

```javascript
import fs from 'fs';
import { Transform } from 'stream';
import { pipeline } from 'stream/promises';

// CSV 解析 Transform
class CSVParser extends Transform {
  constructor() {
    super({ objectMode: true });
    this.buffer = '';
  }
  
  _transform(chunk, encoding, callback) {
    this.buffer += chunk.toString();
    const lines = this.buffer.split('\n');
    
    // 保留最后一行（可能不完整）
    this.buffer = lines.pop();
    
    for (const line of lines) {
      const fields = line.split(',');
      this.push({ fields });
    }
    
    callback();
  }
  
  _flush(callback) {
    if (this.buffer) {
      const fields = this.buffer.split(',');
      this.push({ fields });
    }
    callback();
  }
}

// 数据过滤 Transform
class DataFilter extends Transform {
  constructor() {
    super({ objectMode: true });
  }
  
  _transform(row, encoding, callback) {
    // 只保留第一列大于 100 的行
    if (parseInt(row.fields[0]) > 100) {
      this.push(row);
    }
    callback();
  }
}

// 格式化输出 Transform
class JSONFormatter extends Transform {
  constructor() {
    super({ objectMode: true });
    this.first = true;
  }
  
  _transform(row, encoding, callback) {
    if (this.first) {
      this.push('[\n');
      this.first = false;
    } else {
      this.push(',\n');
    }
    
    this.push(JSON.stringify(row));
    callback();
  }
  
  _flush(callback) {
    this.push('\n]');
    callback();
  }
}

// 使用
await pipeline(
  fs.createReadStream('data.csv'),
  new CSVParser(),
  new DataFilter(),
  new JSONFormatter(),
  fs.createWriteStream('output.json')
);
```

## 💡 最佳实践

```javascript
// ✅ 1. 使用 pipeline 而非 pipe
await pipeline(src, transform, dest);

// ✅ 2. 处理错误
stream.on('error', (err) => {
  console.error(err);
});

// ✅ 3. 正确关闭流
stream.destroy();  // 立即关闭
stream.end();      // 等待数据写完

// ✅ 4. 使用 highWaterMark 控制缓冲区
const stream = fs.createReadStream('file.txt', {
  highWaterMark: 16 * 1024  // 16KB
});

// ✅ 5. 对象模式（处理对象而非 Buffer）
const transform = new Transform({
  objectMode: true,
  transform(obj, encoding, callback) {
    this.push({ ...obj, processed: true });
    callback();
  }
});
```
