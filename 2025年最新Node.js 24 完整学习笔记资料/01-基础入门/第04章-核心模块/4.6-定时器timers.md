# 4.6 定时器（timers）

> 理解 Node.js 定时器与事件循环的关系

---

## ⏰ 定时器基础概念

### 什么是定时器？

```
定时器 = 延迟执行代码的机制

JavaScript 中的定时器：
├─ setTimeout    → 延迟一段时间后执行一次
├─ setInterval   → 每隔一段时间重复执行
├─ setImmediate  → 尽快执行（Node.js 特有）
└─ process.nextTick → 下一个事件循环前执行（Node.js 特有）

用途：
✅ 延迟执行任务
✅ 定时轮询
✅ 防抖和节流
✅ 动画和过渡效果
```

## 📝 setTimeout：延迟执行

### 基础用法

```javascript
// ===== 语法 =====
// setTimeout(回调函数, 延迟时间(毫秒), 参数1, 参数2, ...)

// 示例1：基础延迟
setTimeout(() => {
  console.log('这段代码会在 1 秒后执行');
}, 1000);  // 1000 毫秒 = 1 秒

console.log('这段代码会立即执行');

// 输出顺序：
// 这段代码会立即执行
// 这段代码会在 1 秒后执行

// ===== 示例2：传递参数 =====

function greet(name, age) {
  console.log(`你好，我是 ${name}，今年 ${age} 岁`);
}

// 方式1：通过 setTimeout 的额外参数传递
setTimeout(greet, 1000, 'Alice', 25);
// 1秒后输出：你好，我是 Alice，今年 25 岁

// 方式2：使用箭头函数（更常用）
setTimeout(() => {
  greet('Bob', 30);
}, 1000);

// ===== 示例3：取消定时器 =====

// setTimeout 返回一个定时器 ID
const timerId = setTimeout(() => {
  console.log('这段代码永远不会执行');
}, 5000);

// 在 5 秒之前取消定时器
clearTimeout(timerId);

console.log('定时器已取消');
```

### 零基础理解：setTimeout 的执行顺序

```javascript
// ===== 理解异步执行 =====

console.log('步骤1：开始');

setTimeout(() => {
  console.log('步骤3：定时器执行（1秒后）');
}, 1000);

console.log('步骤2：继续执行');

// 执行顺序：
// 步骤1：开始
// 步骤2：继续执行
// （等待 1 秒）
// 步骤3：定时器执行（1秒后）

// 💡 为什么？
// 因为 setTimeout 是异步的，不会阻塞后续代码执行
// JavaScript 会继续往下执行，1秒后再回来执行定时器里的代码

// ===== 嵌套定时器 =====

console.log('开始');

setTimeout(() => {
  console.log('第一个定时器：1秒后');
  
  setTimeout(() => {
    console.log('第二个定时器：再过1秒（总共2秒后）');
    
    setTimeout(() => {
      console.log('第三个定时器：再过1秒（总共3秒后）');
    }, 1000);
  }, 1000);
}, 1000);

console.log('结束');

// 输出顺序：
// 开始
// 结束
// 第一个定时器：1秒后
// 第二个定时器：再过1秒（总共2秒后）
// 第三个定时器：再过1秒（总共3秒后）
```

### 实用案例：重试机制

```javascript
// ===== 网络请求失败后自动重试 =====

async function fetchWithRetry(url, maxRetries = 3) {
  // maxRetries：最多重试次数
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      console.log(`尝试第 ${i + 1} 次请求...`);
      
      // 模拟网络请求
      const response = await fetch(url);
      
      if (response.ok) {
        console.log('请求成功！');
        return response;
      }
      
    } catch (error) {
      console.log(`请求失败：${error.message}`);
      
      // 如果不是最后一次尝试，等待后重试
      if (i < maxRetries - 1) {
        const delay = 1000 * (i + 1);  // 逐渐增加等待时间
        console.log(`${delay / 1000} 秒后重试...`);
        
        // 使用 setTimeout 的 Promise 版本
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }
  
  throw new Error('请求失败，已达到最大重试次数');
}

// 使用
try {
  await fetchWithRetry('https://api.example.com/data');
} catch (error) {
  console.error('最终失败：', error.message);
}
```

## 🔄 setInterval：重复执行

### 基础用法

```javascript
// ===== 语法 =====
// setInterval(回调函数, 间隔时间(毫秒), 参数1, 参数2, ...)

// 示例1：每秒执行一次
let count = 0;

const intervalId = setInterval(() => {
  count++;
  console.log(`已执行 ${count} 次`);
  
  // 执行 5 次后停止
  if (count >= 5) {
    clearInterval(intervalId);  // 取消重复执行
    console.log('定时器已停止');
  }
}, 1000);  // 每 1000 毫秒（1秒）执行一次

// 输出：
// 已执行 1 次
// 已执行 2 次
// 已执行 3 次
// 已执行 4 次
// 已执行 5 次
// 定时器已停止

// ===== 示例2：倒计时 =====

function countdown(seconds) {
  let remaining = seconds;
  
  console.log(`倒计时 ${remaining} 秒`);
  
  const timer = setInterval(() => {
    remaining--;
    console.log(`剩余 ${remaining} 秒`);
    
    if (remaining <= 0) {
      clearInterval(timer);
      console.log('时间到！');
    }
  }, 1000);
}

countdown(5);
// 输出：
// 倒计时 5 秒
// 剩余 4 秒
// 剩余 3 秒
// 剩余 2 秒
// 剩余 1 秒
// 剩余 0 秒
// 时间到！
```

### ⚠️ setInterval 的陷阱

```javascript
// ===== 陷阱1：任务执行时间超过间隔时间 =====

// ❌ 问题：如果任务需要 2 秒，但间隔只设置 1 秒
setInterval(async () => {
  console.log('开始执行任务...');
  
  // 模拟一个耗时 2 秒的任务
  await new Promise(resolve => setTimeout(resolve, 2000));
  
  console.log('任务完成');
  
  // 问题：新的任务会在前一个任务完成前就开始执行！
  // 可能导致任务堆积
}, 1000);

// ✅ 解决方案：使用递归的 setTimeout
async function repeatTask() {
  console.log('开始执行任务...');
  
  // 执行任务
  await new Promise(resolve => setTimeout(resolve, 2000));
  
  console.log('任务完成');
  
  // 任务完成后，再设置下一次执行
  setTimeout(repeatTask, 1000);
}

repeatTask();  // 开始第一次执行

// 这样可以确保：任务完成 → 等待1秒 → 下一个任务
// 而不会出现任务重叠的情况

// ===== 陷阱2：忘记清除定时器 =====

// ❌ 内存泄漏示例
function createTimer() {
  // 创建了定时器，但没有保存 ID，也没有清除
  setInterval(() => {
    console.log('这个定时器永远不会停止！');
  }, 1000);
}

createTimer();
// 这个定时器会一直运行，即使函数执行完毕
// 导致内存泄漏！

// ✅ 正确做法
class Timer {
  constructor() {
    this.timerId = null;
  }
  
  start() {
    this.timerId = setInterval(() => {
      console.log('定时器执行中...');
    }, 1000);
  }
  
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
      console.log('定时器已停止');
    }
  }
}

const timer = new Timer();
timer.start();

// 5秒后停止
setTimeout(() => {
  timer.stop();
}, 5000);
```

### 实用案例：轮询检查

```javascript
// ===== 定期检查服务器状态 =====

class HealthChecker {
  constructor(url, interval = 30000) {
    this.url = url;              // 要检查的 URL
    this.interval = interval;    // 检查间隔（默认30秒）
    this.timerId = null;         // 定时器 ID
    this.isRunning = false;      // 是否正在运行
  }
  
  // 开始检查
  start() {
    if (this.isRunning) {
      console.log('健康检查已在运行中');
      return;
    }
    
    this.isRunning = true;
    console.log(`开始健康检查，间隔 ${this.interval / 1000} 秒`);
    
    // 立即执行一次
    this.check();
    
    // 然后定期执行
    this.timerId = setInterval(() => {
      this.check();
    }, this.interval);
  }
  
  // 停止检查
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
      this.isRunning = false;
      console.log('健康检查已停止');
    }
  }
  
  // 执行一次检查
  async check() {
    try {
      const start = Date.now();
      const response = await fetch(this.url);
      const duration = Date.now() - start;
      
      if (response.ok) {
        console.log(`✅ 服务正常 (响应时间: ${duration}ms)`);
      } else {
        console.log(`⚠️ 服务异常 (状态码: ${response.status})`);
      }
    } catch (error) {
      console.log(`❌ 服务不可用: ${error.message}`);
    }
  }
}

// 使用
const checker = new HealthChecker('https://api.example.com/health', 10000);
checker.start();

// 1分钟后停止
setTimeout(() => {
  checker.stop();
}, 60000);
```

## ⚡ setImmediate：尽快执行（Node.js 特有）

### 基础概念

```javascript
// ===== setImmediate 是什么？=====

// setImmediate 会在当前事件循环结束后立即执行
// 它比 setTimeout(fn, 0) 更快

console.log('1. 同步代码开始');

setImmediate(() => {
  console.log('3. setImmediate 执行');
});

setTimeout(() => {
  console.log('4. setTimeout 执行');
}, 0);

console.log('2. 同步代码结束');

// 输出顺序（大多数情况下）：
// 1. 同步代码开始
// 2. 同步代码结束
// 3. setImmediate 执行
// 4. setTimeout 执行

// 💡 为什么 setImmediate 比 setTimeout(fn, 0) 快？
// setImmediate 专门设计用于在当前事件循环结束后立即执行
// setTimeout(fn, 0) 还需要经过定时器调度
```

### 实用场景：I/O 操作后的回调

```javascript
import fs from 'fs';

// ===== 在 I/O 操作后使用 setImmediate =====

fs.readFile('file.txt', 'utf8', (err, data) => {
  // 文件读取完成后
  
  console.log('1. 文件读取完成');
  
  // 使用 setImmediate 延迟处理
  setImmediate(() => {
    console.log('2. 使用 setImmediate 处理数据');
    // 处理 data...
  });
  
  console.log('3. 读取回调结束');
});

// 输出顺序：
// 1. 文件读取完成
// 3. 读取回调结束
// 2. 使用 setImmediate 处理数据

// 💡 这样做的好处：
// 让 I/O 回调快速返回，不阻塞其他操作
// 数据处理在下一个事件循环中进行
```

## 🚀 process.nextTick：最高优先级（Node.js 特有）

### 基础概念

```javascript
// ===== process.nextTick 是什么？=====

// process.nextTick 会在当前操作完成后、事件循环继续之前执行
// 它的优先级最高！

console.log('1. 开始');

setTimeout(() => {
  console.log('5. setTimeout');
}, 0);

setImmediate(() => {
  console.log('4. setImmediate');
});

process.nextTick(() => {
  console.log('2. nextTick 1');
});

process.nextTick(() => {
  console.log('3. nextTick 2');
});

console.log('6. 结束');

// 输出顺序：
// 1. 开始
// 6. 结束
// 2. nextTick 1
// 3. nextTick 2
// 4. setImmediate
// 5. setTimeout

// 💡 执行顺序总结：
// 同步代码 > process.nextTick > setImmediate > setTimeout
```

### ⚠️ nextTick 的陷阱

```javascript
// ===== 危险：递归 nextTick 会阻塞事件循环 =====

// ❌ 错误示例：会导致程序卡死
process.nextTick(function recursiveNextTick() {
  console.log('这会无限循环！');
  process.nextTick(recursiveNextTick);  // 递归调用自己
});

// 其他代码永远不会执行，因为 nextTick 队列永远不会清空

// ✅ 正确做法：使用 setImmediate
setImmediate(function recursiveImmediate() {
  console.log('这样是安全的');
  setImmediate(recursiveImmediate);  // 允许其他操作执行
});

// setImmediate 会让其他操作有机会执行
```

### 实用场景：事件发射

```javascript
import { EventEmitter } from 'events';

// ===== 在构造函数中发射事件的正确方式 =====

class Connection extends EventEmitter {
  constructor() {
    super();
    
    // ❌ 错误：立即发射事件
    // this.emit('ready');
    // 问题：此时外部代码还没来得及监听事件
    
    // ✅ 正确：使用 nextTick 延迟发射
    process.nextTick(() => {
      this.emit('ready');
    });
  }
  
  connect() {
    // 连接逻辑...
  }
}

// 使用
const conn = new Connection();

// 这样可以确保监听器已经设置
conn.on('ready', () => {
  console.log('连接已就绪');
});

// 如果没有 nextTick，ready 事件会在监听器设置之前就发射
// 导致监听器永远不会被调用
```

## 🎯 事件循环执行顺序详解

### 完整的执行顺序

```javascript
console.log('1. 同步代码 - 开始');

// 微任务：process.nextTick
process.nextTick(() => {
  console.log('3. nextTick - 第一个');
});

// 微任务：Promise
Promise.resolve().then(() => {
  console.log('4. Promise - 第一个');
});

// 宏任务：setTimeout
setTimeout(() => {
  console.log('6. setTimeout - 0ms');
}, 0);

// 宏任务：setImmediate
setImmediate(() => {
  console.log('7. setImmediate');
});

// 微任务：process.nextTick（再添加一个）
process.nextTick(() => {
  console.log('3. nextTick - 第二个');
});

// 微任务：Promise（再添加一个）
Promise.resolve().then(() => {
  console.log('4. Promise - 第二个');
});

console.log('2. 同步代码 - 结束');

// 输出顺序：
// 1. 同步代码 - 开始
// 2. 同步代码 - 结束
// 3. nextTick - 第一个
// 3. nextTick - 第二个
// 4. Promise - 第一个
// 4. Promise - 第二个
// 6. setTimeout - 0ms
// 7. setImmediate

// 💡 执行顺序规则：
// 1️⃣ 同步代码（最先）
// 2️⃣ process.nextTick（微任务，优先级最高）
// 3️⃣ Promise.then（微任务）
// 4️⃣ setTimeout（宏任务）
// 5️⃣ setImmediate（宏任务，但在 I/O 回调后会优先于 setTimeout）
```

### 零基础理解：事件循环

```
┌─────────────────────────────────────────────┐
│           Node.js 事件循环                     │
└─────────────────────────────────────────────┘

第1步：执行所有同步代码
   console.log('开始')
   console.log('结束')

第2步：执行所有 process.nextTick 回调
   ✓ nextTick 回调1
   ✓ nextTick 回调2

第3步：执行所有 Promise 微任务
   ✓ Promise.then 回调1
   ✓ Promise.then 回调2

第4步：执行 setTimeout 定时器
   ✓ setTimeout 回调

第5步：执行 setImmediate 回调
   ✓ setImmediate 回调

第6步：返回第1步，开始新的循环

💡 简单记忆：
同步 → nextTick → Promise → setTimeout → setImmediate
```

## 📊 定时器对比表

| 定时器 | 执行时机 | 适用场景 | 优先级 |
|--------|---------|----------|--------|
| **同步代码** | 立即 | 普通逻辑 | ⭐⭐⭐⭐⭐ 最高 |
| **process.nextTick** | 当前操作后 | 事件发射、错误处理 | ⭐⭐⭐⭐ 很高 |
| **Promise.then** | 微任务队列 | 异步操作结果 | ⭐⭐⭐ 高 |
| **setImmediate** | 当前循环结束后 | I/O 回调后的处理 | ⭐⭐ 中 |
| **setTimeout** | 指定时间后 | 延迟执行、定时任务 | ⭐ 低 |
| **setInterval** | 周期性执行 | 轮询、定时检查 | ⭐ 低 |

## 🎯 实战案例

### 案例1：防抖（Debounce）

```javascript
// ===== 防抖：在用户停止输入后才执行搜索 =====

function debounce(func, delay) {
  let timerId = null;  // 保存定时器 ID
  
  return function(...args) {
    // 每次调用都清除之前的定时器
    if (timerId) {
      clearTimeout(timerId);
    }
    
    // 设置新的定时器
    timerId = setTimeout(() => {
      func.apply(this, args);
    }, delay);
  };
}

// 使用示例：搜索框
const searchInput = {
  value: '',
  
  // 原始的搜索函数（会频繁调用）
  search(keyword) {
    console.log(`搜索：${keyword}`);
    // 实际的搜索逻辑...
  }
};

// 包装成防抖函数（500ms内不再输入才执行）
const debouncedSearch = debounce((keyword) => {
  searchInput.search(keyword);
}, 500);

// 模拟用户快速输入
console.log('用户快速输入：a');
debouncedSearch('a');

console.log('用户快速输入：ab');
debouncedSearch('ab');

console.log('用户快速输入：abc');
debouncedSearch('abc');

// 只有最后一次输入（abc）会真正执行搜索
// 前面的输入都被取消了

// 500ms后输出：搜索：abc
```

### 案例2：节流（Throttle）

```javascript
// ===== 节流：限制函数执行频率 =====

function throttle(func, limit) {
  let inThrottle = false;  // 是否在节流中
  
  return function(...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      
      setTimeout(() => {
        inThrottle = false;
      }, limit);
    }
  };
}

// 使用示例：滚动事件
let scrollCount = 0;

function handleScroll() {
  scrollCount++;
  console.log(`滚动处理 ${scrollCount} 次`);
}

// 包装成节流函数（每300ms最多执行一次）
const throttledScroll = throttle(handleScroll, 300);

// 模拟快速滚动（每50ms触发一次）
let count = 0;
const scrollInterval = setInterval(() => {
  throttledScroll();
  count++;
  
  if (count >= 20) {
    clearInterval(scrollInterval);
  }
}, 50);

// 虽然触发了20次，但只会执行约7次
// 因为每300ms才能执行一次
```

### 案例3：超时处理

```javascript
// ===== 为异步操作添加超时限制 =====

function withTimeout(promise, timeout) {
  return Promise.race([
    promise,
    new Promise((_, reject) => {
      setTimeout(() => {
        reject(new Error(`操作超时（${timeout}ms）`));
      }, timeout);
    })
  ]);
}

// 使用示例
async function fetchData() {
  // 模拟一个慢速的网络请求
  return new Promise(resolve => {
    setTimeout(() => {
      resolve('数据加载成功');
    }, 3000);  // 需要3秒
  });
}

// 设置2秒超时
try {
  const data = await withTimeout(fetchData(), 2000);
  console.log(data);
} catch (error) {
  console.error(error.message);
  // 输出：操作超时（2000ms）
}
```

## 💡 最佳实践

```javascript
// ✅ 1. 始终清除不再需要的定时器
const timerId = setTimeout(() => {}, 1000);
clearTimeout(timerId);

const intervalId = setInterval(() => {}, 1000);
clearInterval(intervalId);

// ✅ 2. 避免在定时器中执行耗时操作
setTimeout(async () => {
  // 如果这个操作需要很长时间，会阻塞其他代码
  await longRunningOperation();
}, 1000);

// ✅ 3. 使用 Promise 版本的 setTimeout（Node.js 15+）
import { setTimeout as setTimeoutPromise } from 'timers/promises';

await setTimeoutPromise(1000);  // 等待1秒
console.log('1秒后执行');

// ✅ 4. 注意定时器的精度
// setTimeout(fn, 100) 不保证精确在100ms后执行
// 实际延迟可能是 100-110ms 或更多

// ❌ 5. 避免递归 nextTick（会阻塞事件循环）
process.nextTick(function recursive() {
  process.nextTick(recursive);  // ❌ 危险！
});
```
