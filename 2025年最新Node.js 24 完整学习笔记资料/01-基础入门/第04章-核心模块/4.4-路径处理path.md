# 4.4 路径处理（path）

> 跨平台路径操作完全指南

---

## 🛣️ 为什么需要 path 模块？

### 路径问题

```javascript
// ===== 跨平台路径差异 =====

// Windows 路径
'C:\\Users\\Alice\\Documents\\file.txt'

// Unix/Linux/macOS 路径
'/Users/Alice/Documents/file.txt'

// ===== ❌ 错误的做法 =====

const filePath = __dirname + '/' + 'file.txt';  // Unix 风格
const filePath2 = __dirname + '\\' + 'file.txt';  // Windows 风格

// 问题：不同平台使用不同的分隔符

// ===== ✅ 正确的做法 =====

import path from 'path';

const filePath = path.join(__dirname, 'file.txt');
// 自动使用正确的分隔符
```

## 📦 核心方法详解

### path.join()：拼接路径

```javascript
import path from 'path';

// ===== 基础用法 =====

path.join('users', 'alice', 'documents');
// Unix: 'users/alice/documents'
// Windows: 'users\\alice\\documents'

path.join('/users', 'alice', 'documents', 'file.txt');
// Unix: '/users/alice/documents/file.txt'

// ===== 处理 . 和 .. =====

path.join('/users', 'alice', '..', 'bob', './documents');
// '/users/bob/documents'

path.join('a', 'b', 'c', '..', 'd');
// 'a/b/d'

// ===== 空字符串和单点 =====

path.join('');        // '.'
path.join('.', 'file.txt');  // 'file.txt'

// ===== 实际应用 =====

import { fileURLToPath } from 'url';
import { dirname } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

// 读取同目录文件
const configPath = path.join(__dirname, 'config.json');

// 读取上级目录文件
const dataPath = path.join(__dirname, '..', 'data', 'users.json');
```

### path.resolve()：解析为绝对路径

```javascript
// ===== 基础用法 =====

// 从当前工作目录开始
path.resolve('users', 'alice');
// '/current/working/directory/users/alice'

// 遇到绝对路径会重置
path.resolve('users', '/alice', 'documents');
// '/alice/documents'  ← '/alice' 是绝对路径

path.resolve('/a', 'b', 'c');
// '/a/b/c'

// ===== join vs resolve 对比 =====

// join：简单拼接
path.join('a', 'b', 'c');      // 'a/b/c'
path.join('/a', 'b', 'c');     // '/a/b/c'

// resolve：构造绝对路径
path.resolve('a', 'b', 'c');   // '/cwd/a/b/c'  ← 添加了当前目录
path.resolve('/a', 'b', 'c');  // '/a/b/c'

// ===== 实际应用 =====

// 获取配置文件的绝对路径
const configPath = path.resolve(process.cwd(), 'config.json');

// 解析命令行参数中的路径
const filePath = path.resolve(process.argv[2]);
```

### path.dirname()：获取目录名

```javascript
// ===== 基础用法 =====

path.dirname('/users/alice/documents/file.txt');
// '/users/alice/documents'

path.dirname('/users/alice/documents/');
// '/users/alice'

path.dirname('file.txt');
// '.'

// ===== 链式调用 =====

const filePath = '/users/alice/documents/file.txt';

path.dirname(filePath);
// '/users/alice/documents'

path.dirname(path.dirname(filePath));
// '/users/alice'

path.dirname(path.dirname(path.dirname(filePath)));
// '/users'

// ===== 实际应用 =====

import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

console.log('Current directory:', __dirname);
```

### path.basename()：获取文件名

```javascript
// ===== 基础用法 =====

path.basename('/users/alice/documents/file.txt');
// 'file.txt'

path.basename('/users/alice/documents/');
// 'documents'

// ===== 去掉扩展名 =====

path.basename('/users/alice/documents/file.txt', '.txt');
// 'file'

path.basename('image.png', '.png');
// 'image'

// ===== 实际应用 =====

const files = [
  '/path/to/file1.txt',
  '/path/to/file2.js',
  '/path/to/file3.json'
];

const fileNames = files.map(f => path.basename(f));
console.log(fileNames);
// ['file1.txt', 'file2.js', 'file3.json']

// 只获取文件名（无扩展名）
const names = files.map(f => path.basename(f, path.extname(f)));
console.log(names);
// ['file1', 'file2', 'file3']
```

### path.extname()：获取扩展名

```javascript
// ===== 基础用法 =====

path.extname('file.txt');       // '.txt'
path.extname('file.tar.gz');    // '.gz'
path.extname('file.');          // '.'
path.extname('file');           // ''
path.extname('.gitignore');     // ''

// ===== 实际应用 =====

// 根据扩展名分类文件
function categorizeFiles(files) {
  const categories = {};
  
  for (const file of files) {
    const ext = path.extname(file);
    
    if (!categories[ext]) {
      categories[ext] = [];
    }
    
    categories[ext].push(file);
  }
  
  return categories;
}

const files = ['a.txt', 'b.js', 'c.txt', 'd.json'];
const grouped = categorizeFiles(files);
// {
//   '.txt': ['a.txt', 'c.txt'],
//   '.js': ['b.js'],
//   '.json': ['d.json']
// }

// 检查文件类型
function isImageFile(file) {
  const ext = path.extname(file).toLowerCase();
  return ['.jpg', '.jpeg', '.png', '.gif', '.webp'].includes(ext);
}

console.log(isImageFile('photo.jpg'));  // true
console.log(isImageFile('doc.pdf'));    // false
```

### path.parse()：解析路径

```javascript
// ===== 完整解析 =====

const parsed = path.parse('/users/alice/documents/file.txt');

console.log(parsed);
// {
//   root: '/',
//   dir: '/users/alice/documents',
//   base: 'file.txt',
//   ext: '.txt',
//   name: 'file'
// }

// ===== Windows 路径 =====

const winPath = path.win32.parse('C:\\Users\\Alice\\file.txt');
// {
//   root: 'C:\\',
//   dir: 'C:\\Users\\Alice',
//   base: 'file.txt',
//   ext: '.txt',
//   name: 'file'
// }

// ===== 实际应用 =====

function renameFile(filePath, newName) {
  const parsed = path.parse(filePath);
  return path.join(parsed.dir, newName + parsed.ext);
}

const original = '/path/to/old-name.txt';
const renamed = renameFile(original, 'new-name');
console.log(renamed);
// '/path/to/new-name.txt'
```

### path.format()：格式化路径

```javascript
// ===== 从对象构建路径 =====

const formatted = path.format({
  root: '/',
  dir: '/users/alice',
  base: 'file.txt'
});
console.log(formatted);
// '/users/alice/file.txt'

// 使用 name 和 ext（会覆盖 base）
const formatted2 = path.format({
  dir: '/users/alice',
  name: 'file',
  ext: '.txt'
});
console.log(formatted2);
// '/users/alice/file.txt'

// ===== parse 和 format 往返 =====

const original = '/users/alice/file.txt';
const parsed = path.parse(original);
const formatted = path.format(parsed);

console.log(original === formatted);  // true
```

### path.normalize()：规范化路径

```javascript
// ===== 清理路径 =====

path.normalize('/users//alice/../bob/./file.txt');
// '/users/bob/file.txt'

path.normalize('a/b/../c/./d');
// 'a/c/d'

path.normalize('a//b////c');
// 'a/b/c'

// ===== Windows 路径 =====

path.normalize('C:\\Users\\Alice\\..\\Bob\\Documents');
// 'C:\\Users\\Bob\\Documents'

// ===== 注意：不会解析符号链接 =====

// 如需解析符号链接，使用 fs.realpath()
import fs from 'fs/promises';

const realPath = await fs.realpath('symlink-path');
```

### path.relative()：计算相对路径

```javascript
// ===== 基础用法 =====

path.relative('/data/demo', '/data/demo/file.txt');
// 'file.txt'

path.relative('/data/demo', '/data/other/file.txt');
// '../other/file.txt'

path.relative('/users/alice', '/users/bob');
// '../bob'

// ===== 实际应用 =====

// 显示相对于项目根目录的路径
function getRelativePath(filePath) {
  return path.relative(process.cwd(), filePath);
}

const absPath = '/Users/alice/project/src/index.js';
const relPath = getRelativePath(absPath);
console.log(relPath);
// 'src/index.js'（假设 cwd 是 /Users/alice/project）

// 构建 require 路径
const from = '/project/src/components/Button.js';
const to = '/project/src/utils/helpers.js';

const relativePath = path.relative(path.dirname(from), to);
console.log(relativePath);
// '../utils/helpers.js'
```

### path.isAbsolute()：判断绝对路径

```javascript
// ===== Unix/Linux/macOS =====

path.isAbsolute('/users/alice');  // true
path.isAbsolute('users/alice');   // false
path.isAbsolute('./users');       // false
path.isAbsolute('../users');      // false

// ===== Windows =====

path.win32.isAbsolute('C:\\Users');  // true
path.win32.isAbsolute('\\Users');    // true（UNC）
path.win32.isAbsolute('Users');      // false
```

## 🌍 跨平台处理

### 路径分隔符

```javascript
// ===== path.sep：路径分隔符 =====

console.log(path.sep);
// Unix/Linux/macOS: '/'
// Windows: '\\'

// 使用
const parts = '/users/alice/file.txt'.split(path.sep);
console.log(parts);
// Unix: ['', 'users', 'alice', 'file.txt']
// Windows: ['', 'users', 'alice', 'file.txt']

// ===== path.delimiter：环境变量分隔符 =====

console.log(path.delimiter);
// Unix/Linux/macOS: ':'
// Windows: ';'

// 解析 PATH 环境变量
const paths = process.env.PATH.split(path.delimiter);
console.log(paths);
// Unix: ['/usr/local/bin', '/usr/bin', '/bin', ...]
// Windows: ['C:\\Windows', 'C:\\Program Files', ...]
```

### 强制使用特定平台风格

```javascript
// ===== path.posix：强制使用 Unix 风格 =====

path.posix.join('users', 'alice', 'file.txt');
// 'users/alice/file.txt'（在任何平台）

path.posix.sep;  // '/'（在任何平台）

// ===== path.win32：强制使用 Windows 风格 =====

path.win32.join('users', 'alice', 'file.txt');
// 'users\\alice\\file.txt'（在任何平台）

path.win32.sep;  // '\\'（在任何平台）

// ===== 应用场景 =====

// 生成跨平台的 URL 路径（始终使用 /）
function toURLPath(filePath) {
  return filePath.split(path.sep).join('/');
}

// 处理 Git 路径（始终使用 /）
function toGitPath(filePath) {
  return path.posix.normalize(filePath.replace(/\\/g, '/'));
}
```

## 🎯 实战案例

### 案例1：查找项目根目录

```javascript
import fs from 'fs/promises';
import path from 'path';

async function findProjectRoot(startDir = process.cwd()) {
  let currentDir = startDir;
  
  while (true) {
    // 检查是否存在 package.json
    const packageJsonPath = path.join(currentDir, 'package.json');
    
    try {
      await fs.access(packageJsonPath);
      return currentDir;  // 找到了
    } catch {
      // 未找到，继续向上查找
    }
    
    const parentDir = path.dirname(currentDir);
    
    // 已到达根目录
    if (parentDir === currentDir) {
      throw new Error('Project root not found');
    }
    
    currentDir = parentDir;
  }
}

const projectRoot = await findProjectRoot();
console.log('Project root:', projectRoot);
```

### 案例2：构建文件树

```javascript
async function buildFileTree(dir, prefix = '') {
  const entries = await fs.readdir(dir, { withFileTypes: true });
  const tree = [];
  
  for (let i = 0; i < entries.length; i++) {
    const entry = entries[i];
    const isLast = i === entries.length - 1;
    const connector = isLast ? '└── ' : '├── ';
    const childPrefix = isLast ? '    ' : '│   ';
    
    tree.push(prefix + connector + entry.name);
    
    if (entry.isDirectory()) {
      const subDir = path.join(dir, entry.name);
      const subTree = await buildFileTree(subDir, prefix + childPrefix);
      tree.push(...subTree);
    }
  }
  
  return tree;
}

const tree = await buildFileTree('src');
console.log(tree.join('\n'));
// src
// ├── components
// │   ├── Button.tsx
// │   └── Input.tsx
// ├── utils
// │   └── helpers.ts
// └── index.ts
```

### 案例3：安全的路径连接

```javascript
// 防止路径遍历攻击

function safePath(basePath, userPath) {
  const fullPath = path.join(basePath, userPath);
  const normalizedPath = path.normalize(fullPath);
  
  // 确保最终路径在 basePath 内
  if (!normalizedPath.startsWith(basePath)) {
    throw new Error('Path traversal detected');
  }
  
  return normalizedPath;
}

// ✅ 正常使用
const safe = safePath('/var/www', 'uploads/image.jpg');
console.log(safe);
// '/var/www/uploads/image.jpg'

// ❌ 攻击尝试
try {
  const unsafe = safePath('/var/www', '../../../etc/passwd');
} catch (err) {
  console.error(err.message);
  // 'Path traversal detected'
}
```

### 案例4：模块路径解析

```javascript
function resolveModulePath(moduleName, fromPath) {
  const parts = moduleName.split('/');
  const isScoped = moduleName.startsWith('@');
  
  // 确定包名
  const packageName = isScoped
    ? parts.slice(0, 2).join('/')
    : parts[0];
  
  // 构建 node_modules 路径
  const nodeModulesPath = path.join(
    path.dirname(fromPath),
    'node_modules',
    packageName
  );
  
  return nodeModulesPath;
}

const modulePath = resolveModulePath('lodash/debounce', '/project/src/index.js');
console.log(modulePath);
// '/project/src/node_modules/lodash'

const scopedPath = resolveModulePath('@babel/core', '/project/src/index.js');
console.log(scopedPath);
// '/project/src/node_modules/@babel/core'
```
