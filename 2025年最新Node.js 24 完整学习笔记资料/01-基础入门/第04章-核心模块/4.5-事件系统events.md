# 4.5 事件系统（events）

> EventEmitter 原理与自定义事件系统

---

## 🎪 EventEmitter 基础

### 核心概念

```
EventEmitter = Node.js 事件驱动架构的核心
├─ 发布-订阅模式
├─ 一对多的事件通知
├─ 异步编程的基础
└─ 许多 Node.js 核心模块都继承自 EventEmitter
```

### 基础使用

```javascript
import { EventEmitter } from 'events';

// 创建 EventEmitter 实例
const emitter = new EventEmitter();

// ===== 监听事件 =====

emitter.on('data', (data) => {
  console.log('Received data:', data);
});

// ===== 触发事件 =====

emitter.emit('data', { id: 1, name: 'Alice' });
// 输出：Received data: { id: 1, name: 'Alice' }

// ===== 链式调用 =====

emitter
  .on('start', () => console.log('Started'))
  .on('process', () => console.log('Processing'))
  .on('end', () => console.log('Ended'));
```

### 核心方法

```javascript
const emitter = new EventEmitter();

// ===== on()：监听事件（可重复）=====

emitter.on('event', () => console.log('Listener 1'));
emitter.on('event', () => console.log('Listener 2'));

emitter.emit('event');
// 输出：
// Listener 1
// Listener 2

// ===== once()：只监听一次 =====

emitter.once('connect', () => console.log('Connected'));

emitter.emit('connect');  // 输出：Connected
emitter.emit('connect');  // 无输出（已移除）

// ===== off() / removeListener()：移除监听器 =====

function onData(data) {
  console.log('Data:', data);
}

emitter.on('data', onData);
emitter.emit('data', 'test');  // 输出：Data: test

emitter.off('data', onData);   // 移除监听器
emitter.emit('data', 'test');  // 无输出

// ===== removeAllListeners()：移除所有监听器 =====

emitter.on('event', () => {});
emitter.on('event', () => {});

emitter.removeAllListeners('event');  // 移除所有 event 监听器
emitter.removeAllListeners();         // 移除所有事件的所有监听器

// ===== listeners()：获取监听器列表 =====

const listeners = emitter.listeners('event');
console.log('Listener count:', listeners.length);

// ===== listenerCount()：获取监听器数量 =====

const count = emitter.listenerCount('event');
console.log('Listener count:', count);
```

## 🎯 进阶用法

### 传递多个参数

```javascript
emitter.on('user', (name, age, city) => {
  console.log(`${name}, ${age}, from ${city}`);
});

emitter.emit('user', 'Alice', 25, 'Beijing');
// 输出：Alice, 25, from Beijing
```

### 错误处理

```javascript
// ===== 特殊的 error 事件 =====

const emitter = new EventEmitter();

// ❌ 如果没有监听器，会抛出错误并崩溃
emitter.emit('error', new Error('Something went wrong'));
// Error: Uncaught, unspecified "error" event

// ✅ 添加 error 监听器
emitter.on('error', (err) => {
  console.error('Error occurred:', err.message);
});

emitter.emit('error', new Error('Something went wrong'));
// 输出：Error occurred: Something went wrong

// ===== 最佳实践 =====

class MyEmitter extends EventEmitter {
  doSomething() {
    try {
      // 可能抛出错误的操作
      throw new Error('Operation failed');
    } catch (err) {
      this.emit('error', err);
    }
  }
}

const myEmitter = new MyEmitter();

myEmitter.on('error', (err) => {
  console.error('Caught error:', err.message);
});

myEmitter.doSomething();
```

### 监听器优先级

```javascript
// ===== prependListener()：添加到监听器列表开头 =====

emitter.on('event', () => console.log('Second'));
emitter.prependListener('event', () => console.log('First'));

emitter.emit('event');
// 输出：
// First
// Second

// ===== prependOnceListener()：一次性 + 优先 =====

emitter.prependOnceListener('start', () => console.log('Starting...'));
emitter.on('start', () => console.log('Started'));

emitter.emit('start');
// 输出：
// Starting...
// Started

emitter.emit('start');
// 输出：
// Started（Starting... 只执行一次）
```

### 监听器限制

```javascript
// ===== 默认最大监听器数量：10 =====

const emitter = new EventEmitter();

// 添加 11 个监听器会触发警告
for (let i = 0; i < 11; i++) {
  emitter.on('event', () => {});
}
// (node:12345) MaxListenersExceededWarning: Possible EventEmitter memory leak detected

// ===== 设置最大监听器数量 =====

emitter.setMaxListeners(20);

// 取消限制
emitter.setMaxListeners(0);  // 或 Infinity

// 获取当前限制
const max = emitter.getMaxListeners();
console.log('Max listeners:', max);
```

## 🏗️ 自定义 EventEmitter 类

### 基础继承

```javascript
import { EventEmitter } from 'events';

class Database extends EventEmitter {
  connect() {
    // 模拟连接
    setTimeout(() => {
      this.emit('connect');
    }, 1000);
  }
  
  query(sql) {
    // 模拟查询
    this.emit('query', sql);
    
    setTimeout(() => {
      const result = { rows: [] };
      this.emit('result', result);
    }, 500);
  }
  
  close() {
    this.emit('close');
  }
}

// 使用
const db = new Database();

db.on('connect', () => console.log('Database connected'));
db.on('query', (sql) => console.log('Executing:', sql));
db.on('result', (result) => console.log('Result:', result));
db.on('close', () => console.log('Database closed'));

db.connect();
db.query('SELECT * FROM users');
db.close();
```

### 实战：HTTP 服务器

```javascript
import { EventEmitter } from 'events';

class Server extends EventEmitter {
  constructor() {
    super();
    this.connections = [];
  }
  
  listen(port) {
    console.log(`Server listening on port ${port}`);
    this.emit('listening', port);
  }
  
  handleRequest(req, res) {
    this.emit('request', req, res);
  }
  
  handleConnection(socket) {
    this.connections.push(socket);
    this.emit('connection', socket);
    
    socket.on('close', () => {
      const index = this.connections.indexOf(socket);
      this.connections.splice(index, 1);
    });
  }
  
  close() {
    for (const conn of this.connections) {
      conn.destroy();
    }
    
    this.emit('close');
  }
}

// 使用
const server = new Server();

server.on('listening', (port) => {
  console.log(`Listening on ${port}`);
});

server.on('request', (req, res) => {
  console.log(`Request: ${req.method} ${req.url}`);
});

server.on('connection', (socket) => {
  console.log('New connection');
});

server.on('close', () => {
  console.log('Server closed');
});

server.listen(3000);
```

## 🛠️ 手写 EventEmitter

```javascript
class MyEventEmitter {
  constructor() {
    this.events = {};  // { eventName: [listener1, listener2, ...] }
  }
  
  on(eventName, listener) {
    if (!this.events[eventName]) {
      this.events[eventName] = [];
    }
    
    this.events[eventName].push(listener);
    return this;  // 支持链式调用
  }
  
  once(eventName, listener) {
    const onceWrapper = (...args) => {
      listener(...args);
      this.off(eventName, onceWrapper);
    };
    
    // 保存原始 listener 引用（用于 off）
    onceWrapper.listener = listener;
    
    this.on(eventName, onceWrapper);
    return this;
  }
  
  emit(eventName, ...args) {
    const listeners = this.events[eventName];
    
    if (!listeners) {
      return false;
    }
    
    for (const listener of listeners) {
      listener(...args);
    }
    
    return true;
  }
  
  off(eventName, listenerToRemove) {
    const listeners = this.events[eventName];
    
    if (!listeners) {
      return this;
    }
    
    this.events[eventName] = listeners.filter(listener => {
      // 处理 once 包装的监听器
      return listener !== listenerToRemove 
        && listener.listener !== listenerToRemove;
    });
    
    return this;
  }
  
  removeAllListeners(eventName) {
    if (eventName) {
      delete this.events[eventName];
    } else {
      this.events = {};
    }
    
    return this;
  }
  
  listenerCount(eventName) {
    const listeners = this.events[eventName];
    return listeners ? listeners.length : 0;
  }
  
  listeners(eventName) {
    return this.events[eventName] || [];
  }
}

// 测试
const emitter = new MyEventEmitter();

emitter.on('test', (msg) => console.log('Listener 1:', msg));
emitter.on('test', (msg) => console.log('Listener 2:', msg));

emitter.emit('test', 'Hello');
// 输出：
// Listener 1: Hello
// Listener 2: Hello

emitter.once('once', (msg) => console.log('Once:', msg));

emitter.emit('once', 'First');   // 输出：Once: First
emitter.emit('once', 'Second');  // 无输出
```

## 🎯 实战案例

### 案例1：任务队列

```javascript
class TaskQueue extends EventEmitter {
  constructor(concurrency = 1) {
    super();
    this.concurrency = concurrency;
    this.running = 0;
    this.queue = [];
  }
  
  add(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this.emit('add', task);
      this.process();
    });
  }
  
  async process() {
    if (this.running >= this.concurrency || this.queue.length === 0) {
      return;
    }
    
    const { task, resolve, reject } = this.queue.shift();
    this.running++;
    
    this.emit('start', task);
    
    try {
      const result = await task();
      this.emit('success', task, result);
      resolve(result);
    } catch (err) {
      this.emit('error', task, err);
      reject(err);
    } finally {
      this.running--;
      this.emit('complete', task);
      
      if (this.queue.length === 0 && this.running === 0) {
        this.emit('idle');
      }
      
      this.process();
    }
  }
}

// 使用
const queue = new TaskQueue(2);  // 并发数 2

queue.on('add', (task) => console.log('Task added'));
queue.on('start', (task) => console.log('Task started'));
queue.on('success', (task, result) => console.log('Task succeeded:', result));
queue.on('error', (task, err) => console.error('Task failed:', err));
queue.on('complete', (task) => console.log('Task completed'));
queue.on('idle', () => console.log('Queue is idle'));

// 添加任务
for (let i = 0; i < 5; i++) {
  queue.add(async () => {
    await new Promise(resolve => setTimeout(resolve, 1000));
    return `Task ${i} result`;
  });
}
```

### 案例2：实时数据流

```javascript
class DataStream extends EventEmitter {
  constructor() {
    super();
    this.buffer = [];
    this.paused = false;
  }
  
  write(data) {
    if (this.paused) {
      this.buffer.push(data);
      return false;  // 背压信号
    }
    
    this.emit('data', data);
    return true;
  }
  
  pause() {
    this.paused = true;
    this.emit('pause');
  }
  
  resume() {
    this.paused = false;
    this.emit('resume');
    
    // 清空缓冲区
    while (this.buffer.length > 0) {
      const data = this.buffer.shift();
      this.emit('data', data);
    }
  }
  
  end() {
    this.emit('end');
  }
}

// 使用
const stream = new DataStream();

stream.on('data', (data) => {
  console.log('Data:', data);
  
  // 模拟慢速消费者
  if (data.length > 100) {
    stream.pause();
    
    setTimeout(() => {
      stream.resume();
    }, 1000);
  }
});

stream.on('pause', () => console.log('Stream paused'));
stream.on('resume', () => console.log('Stream resumed'));
stream.on('end', () => console.log('Stream ended'));

// 写入数据
stream.write('Small data');
stream.write('X'.repeat(200));  // 触发 pause
stream.end();
```

## 💡 最佳实践

```javascript
// ✅ 1. 始终监听 error 事件
emitter.on('error', (err) => {
  console.error('Error:', err);
});

// ✅ 2. 及时移除不再需要的监听器（避免内存泄漏）
const listener = () => {};
emitter.on('event', listener);
emitter.off('event', listener);

// ✅ 3. 使用 once() 处理一次性事件
emitter.once('connect', () => {
  console.log('Connected');
});

// ✅ 4. 设置合理的最大监听器数量
emitter.setMaxListeners(20);

// ✅ 5. 使用 try-catch 保护监听器
emitter.on('data', (data) => {
  try {
    processData(data);
  } catch (err) {
    emitter.emit('error', err);
  }
});

// ❌ 避免：在监听器中抛出未捕获的错误
emitter.on('data', (data) => {
  throw new Error('Boom!');  // 会导致程序崩溃
});
```
