# 4.1 Buffer：二进制数据处理

> 理解 Buffer 的本质，掌握二进制数据处理

---

## 🎯 Buffer 是什么？

```
Buffer = 二进制数据缓冲区
├─ 类似数组的对象
├─ 固定大小
├─ 存储在 V8 堆外内存
└─ 用于处理 TCP 流、文件系统操作等
```

### 为什么需要 Buffer？

```javascript
// ===== JavaScript 的限制 =====

// JavaScript 原生只支持 Unicode 字符串
const text = "Hello 你好";  // ✅ 支持

// 无法直接处理二进制数据
const binary = [0xFF, 0xFE, 0x00, 0x01];  // ❌ 只是普通数组

// ===== Buffer 的作用 =====

// 处理文件内容（图片、视频等）
const imageBuffer = fs.readFileSync('image.jpg');

// 网络传输
socket.write(Buffer.from('Hello'));

// 加密解密
const encrypted = crypto.encrypt(Buffer.from('secret'));
```

## 📦 创建 Buffer

### 1. Buffer.from()（推荐）

```javascript
// ===== 从字符串创建 =====

const buf1 = Buffer.from('Hello');
console.log(buf1);
// <Buffer 48 65 6c 6c 6f>

// 指定编码
const buf2 = Buffer.from('你好', 'utf8');
console.log(buf2);
// <Buffer e4 bd a0 e5 a5 bd>

// ===== 从数组创建 =====

const buf3 = Buffer.from([0x48, 0x65, 0x6c, 0x6c, 0x6f]);
console.log(buf3.toString());
// 'Hello'

// ===== 从 Buffer 复制 =====

const original = Buffer.from('Hello');
const copy = Buffer.from(original);

console.log(original === copy);  // false（不同的 Buffer）
console.log(original.equals(copy));  // true（内容相同）
```

### 2. Buffer.alloc()（安全）

```javascript
// ===== 创建指定大小的 Buffer（填充 0）=====

const buf = Buffer.alloc(10);
console.log(buf);
// <Buffer 00 00 00 00 00 00 00 00 00 00>

// ===== 创建并填充指定值 =====

const buf2 = Buffer.alloc(10, 0xFF);
console.log(buf2);
// <Buffer ff ff ff ff ff ff ff ff ff ff>

const buf3 = Buffer.alloc(10, 'a');
console.log(buf3);
// <Buffer 61 61 61 61 61 61 61 61 61 61>
```

### 3. Buffer.allocUnsafe()（快速但不安全）

```javascript
// ===== 创建未初始化的 Buffer（可能包含旧数据）=====

const buf = Buffer.allocUnsafe(10);
console.log(buf);
// <Buffer ?? ?? ?? ?? ?? ?? ?? ?? ?? ??>  ← 未知内容

// ⚠️ 必须手动填充或写入数据
buf.fill(0);
console.log(buf);
// <Buffer 00 00 00 00 00 00 00 00 00 00>

// ===== 性能对比 =====

console.time('alloc');
for (let i = 0; i < 100000; i++) {
  Buffer.alloc(1024);
}
console.timeEnd('alloc');
// alloc: 150ms

console.time('allocUnsafe');
for (let i = 0; i < 100000; i++) {
  Buffer.allocUnsafe(1024);
}
console.timeEnd('allocUnsafe');
// allocUnsafe: 50ms  ← 快 3 倍
```

## 🔧 Buffer 操作

### 读取和写入

```javascript
const buf = Buffer.alloc(10);

// ===== 写入数据 =====

buf[0] = 0x48;  // 'H'
buf[1] = 0x65;  // 'e'
buf[2] = 0x6c;  // 'l'
buf[3] = 0x6c;  // 'l'
buf[4] = 0x6f;  // 'o'

console.log(buf.toString('utf8', 0, 5));
// 'Hello'

// ===== write() 方法 =====

const buf2 = Buffer.alloc(20);

buf2.write('Hello', 0, 'utf8');
buf2.write('World', 6, 'utf8');

console.log(buf2.toString());
// 'Hello\u0000World\u0000\u0000\u0000\u0000\u0000\u0000\u0000\u0000'

// ===== 写入数字 =====

const buf3 = Buffer.alloc(4);

buf3.writeUInt8(255, 0);     // 写入 1 字节
buf3.writeUInt16BE(0xFFFF, 1);  // 写入 2 字节（大端序）
buf3.writeUInt8(128, 3);

console.log(buf3);
// <Buffer ff ff ff 80>

// ===== 读取数字 =====

console.log(buf3.readUInt8(0));        // 255
console.log(buf3.readUInt16BE(1));     // 65535
console.log(buf3.readUInt8(3));        // 128
```

### 切片和拷贝

```javascript
// ===== slice()：创建视图（共享内存）=====

const buf = Buffer.from('Hello World');

const slice = buf.slice(0, 5);
console.log(slice.toString());  // 'Hello'

// 修改 slice 会影响原 Buffer
slice[0] = 0x68;  // 'h'
console.log(buf.toString());  // 'hello World'  ← 原 Buffer 被修改

// ===== subarray()：同 slice()（推荐）=====

const sub = buf.subarray(6, 11);
console.log(sub.toString());  // 'World'

// ===== Buffer.from()：创建副本（独立内存）=====

const buf2 = Buffer.from('Hello World');
const copy = Buffer.from(buf2);

copy[0] = 0x68;
console.log(buf2.toString());  // 'Hello World'  ← 不受影响
console.log(copy.toString());  // 'hello World'
```

### 拼接

```javascript
// ===== concat()：拼接多个 Buffer =====

const buf1 = Buffer.from('Hello ');
const buf2 = Buffer.from('World');
const buf3 = Buffer.from('!');

const result = Buffer.concat([buf1, buf2, buf3]);
console.log(result.toString());
// 'Hello World!'

// 指定总长度（可提升性能）
const result2 = Buffer.concat([buf1, buf2, buf3], 20);
console.log(result2.length);  // 20
```

## 🔤 编码转换

### 支持的编码

```javascript
// ===== Node.js 支持的编码 =====

const encodings = [
  'utf8',      // UTF-8 编码（默认）
  'utf16le',   // UTF-16 小端序
  'latin1',    // Latin-1（ISO-8859-1）
  'base64',    // Base64 编码
  'hex',       // 十六进制
  'ascii',     // ASCII（仅支持 0-127）
  'binary',    // Latin-1 的别名
  'ucs2'       // UTF-16 小端序的别名
];

// ===== 编码示例 =====

const text = 'Hello 你好';

// UTF-8（默认）
const utf8 = Buffer.from(text, 'utf8');
console.log(utf8);
// <Buffer 48 65 6c 6c 6f 20 e4 bd a0 e5 a5 bd>

// Base64
const base64 = Buffer.from(text).toString('base64');
console.log(base64);
// 'SGVsbG8g5L2g5aW9'

// Hex
const hex = Buffer.from(text).toString('hex');
console.log(hex);
// '48656c6c6f20e4bda0e5a5bd'

// ===== 解码 =====

console.log(Buffer.from(base64, 'base64').toString());
// 'Hello 你好'

console.log(Buffer.from(hex, 'hex').toString());
// 'Hello 你好'
```

### Base64 编码应用

```javascript
// ===== 文件转 Base64 =====

import fs from 'fs';

const imageBuffer = fs.readFileSync('image.png');
const base64Image = imageBuffer.toString('base64');

// 用于 Data URL
const dataUrl = `data:image/png;base64,${base64Image}`;

// ===== Base64 转文件 =====

const base64Data = 'SGVsbG8gV29ybGQ=';
const buffer = Buffer.from(base64Data, 'base64');

fs.writeFileSync('output.txt', buffer);
```

## ⚡ 性能优化

### 1. Buffer 池化

```javascript
// ===== Buffer.allocUnsafe() + fill =====

// ❌ 慢
function slowWay() {
  const buf = Buffer.alloc(1024);
  // 使用 buf...
  return buf;
}

// ✅ 快
const bufferPool = Buffer.allocUnsafe(10 * 1024);  // 10KB 池
let poolOffset = 0;

function fastWay(size) {
  if (poolOffset + size > bufferPool.length) {
    poolOffset = 0;  // 重置池
  }
  
  const buf = bufferPool.subarray(poolOffset, poolOffset + size);
  buf.fill(0);  // 清空
  poolOffset += size;
  
  return buf;
}
```

### 2. 避免频繁拼接

```javascript
// ===== ❌ 慢：多次 concat =====

let result = Buffer.alloc(0);

for (let i = 0; i < 10000; i++) {
  const chunk = Buffer.from(`chunk${i}`);
  result = Buffer.concat([result, chunk]);  // 每次都创建新 Buffer
}

// ===== ✅ 快：收集后一次拼接 =====

const chunks = [];

for (let i = 0; i < 10000; i++) {
  chunks.push(Buffer.from(`chunk${i}`));
}

const result2 = Buffer.concat(chunks);
```

### 3. 使用 TypedArray

```javascript
// ===== Buffer 与 TypedArray 互转 =====

const buf = Buffer.from([1, 2, 3, 4, 5, 6, 7, 8]);

// Buffer → TypedArray
const uint8 = new Uint8Array(buf.buffer, buf.byteOffset, buf.length);
const uint16 = new Uint16Array(buf.buffer, buf.byteOffset, buf.length / 2);

console.log(uint8);   // Uint8Array(8) [1, 2, 3, 4, 5, 6, 7, 8]
console.log(uint16);  // Uint16Array(4) [513, 1027, 1541, 2055]

// TypedArray → Buffer
const arr = new Uint8Array([1, 2, 3, 4]);
const buf2 = Buffer.from(arr);
```

## 🎯 实战案例

### 案例1：文件完整性校验

```javascript
import crypto from 'crypto';
import fs from 'fs';

function calculateMD5(filePath) {
  const buffer = fs.readFileSync(filePath);
  const hash = crypto.createHash('md5');
  hash.update(buffer);
  return hash.digest('hex');
}

const md5 = calculateMD5('package.json');
console.log('MD5:', md5);
```

### 案例2：图片格式检测

```javascript
function detectImageFormat(buffer) {
  // 检查文件头（Magic Number）
  if (buffer[0] === 0xFF && buffer[1] === 0xD8 && buffer[2] === 0xFF) {
    return 'JPEG';
  }
  
  if (buffer[0] === 0x89 && buffer[1] === 0x50 && buffer[2] === 0x4E && buffer[3] === 0x47) {
    return 'PNG';
  }
  
  if (buffer[0] === 0x47 && buffer[1] === 0x49 && buffer[2] === 0x46) {
    return 'GIF';
  }
  
  return 'Unknown';
}

const imageBuffer = fs.readFileSync('image.jpg');
console.log('Format:', detectImageFormat(imageBuffer));
// 'JPEG'
```

### 案例3：协议解析

```javascript
// ===== 简单的协议：4字节长度 + 数据 =====

function encodeMessage(message) {
  const dataBuffer = Buffer.from(message, 'utf8');
  const lengthBuffer = Buffer.alloc(4);
  
  lengthBuffer.writeUInt32BE(dataBuffer.length, 0);
  
  return Buffer.concat([lengthBuffer, dataBuffer]);
}

function decodeMessage(buffer) {
  const length = buffer.readUInt32BE(0);
  const data = buffer.subarray(4, 4 + length);
  
  return data.toString('utf8');
}

// 使用
const encoded = encodeMessage('Hello World');
console.log(encoded);
// <Buffer 00 00 00 0b 48 65 6c 6c 6f 20 57 6f 72 6c 64>

const decoded = decodeMessage(encoded);
console.log(decoded);
// 'Hello World'
```
