# 4.3 文件系统（fs）

> 掌握 Node.js 文件和目录操作的三种 API 风格

---

## 📁 fs 模块概述

### 三种 API 风格

```javascript
import fs from 'fs';                    // 回调 API
import fsSync from 'fs';                // 同步 API
import fsPromises from 'fs/promises';   // Promise API（推荐）

// 同一个功能的三种写法：

// 1. 回调风格（最原始）
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// 2. 同步风格（阻塞）
try {
  const data = fs.readFileSync('file.txt', 'utf8');
  console.log(data);
} catch (err) {
  console.error(err);
}

// 3. Promise 风格（推荐）
try {
  const data = await fsPromises.readFile('file.txt', 'utf8');
  console.log(data);
} catch (err) {
  console.error(err);
}
```

## 📄 文件读写操作

### 读取文件

```javascript
import fs from 'fs/promises';

// ===== 完整读取 =====

// 读取文本文件
const text = await fs.readFile('file.txt', 'utf8');
console.log(text);

// 读取二进制文件
const buffer = await fs.readFile('image.png');
console.log(buffer);  // <Buffer ...>

// ===== 读取 JSON 文件 =====

const data = JSON.parse(
  await fs.readFile('data.json', 'utf8')
);
console.log(data);

// ===== 流式读取（大文件）=====

import { createReadStream } from 'fs';

const stream = createReadStream('large-file.txt', {
  encoding: 'utf8',
  highWaterMark: 64 * 1024  // 64KB chunks
});

for await (const chunk of stream) {
  console.log('Chunk:', chunk.length);
}

// ===== 按行读取 =====

import readline from 'readline';

const fileStream = createReadStream('file.txt');
const rl = readline.createInterface({
  input: fileStream,
  crlfDelay: Infinity
});

for await (const line of rl) {
  console.log('Line:', line);
}
```

### 写入文件

```javascript
// ===== 完整写入（覆盖）=====

await fs.writeFile('output.txt', 'Hello World');

// 写入 JSON
const data = { name: 'Alice', age: 25 };
await fs.writeFile('data.json', JSON.stringify(data, null, 2));

// ===== 追加写入 =====

await fs.appendFile('log.txt', 'New log entry\n');

// ===== 流式写入（大文件）=====

import { createWriteStream } from 'fs';

const writer = createWriteStream('output.txt');

writer.write('Line 1\n');
writer.write('Line 2\n');
writer.write('Line 3\n');
writer.end();

// 等待写入完成
await new Promise((resolve, reject) => {
  writer.on('finish', resolve);
  writer.on('error', reject);
});

// ===== 原子写入（先写临时文件，再重命名）=====

async function atomicWrite(file, data) {
  const tempFile = `${file}.tmp`;
  await fs.writeFile(tempFile, data);
  await fs.rename(tempFile, file);
}

await atomicWrite('config.json', JSON.stringify(config));
```

### 复制文件

```javascript
// ===== 方式1：copyFile（推荐）=====

await fs.copyFile('source.txt', 'dest.txt');

// 防止覆盖已存在的文件
await fs.copyFile('source.txt', 'dest.txt', fs.constants.COPYFILE_EXCL);

// ===== 方式2：流式复制（大文件）=====

import { pipeline } from 'stream/promises';
import { createReadStream, createWriteStream } from 'fs';

await pipeline(
  createReadStream('large-source.mp4'),
  createWriteStream('large-dest.mp4')
);

// ===== 方式3：cp（Node.js 16.7+，支持目录）=====

await fs.cp('source-dir', 'dest-dir', { recursive: true });
```

## 📂 目录操作

```javascript
// ===== 创建目录 =====

// 创建单级目录
await fs.mkdir('new-dir');

// 创建多级目录（递归）
await fs.mkdir('path/to/nested/dir', { recursive: true });

// ===== 读取目录 =====

// 获取文件名列表
const files = await fs.readdir('.');
console.log(files);  // ['file1.txt', 'file2.txt', 'dir1']

// 获取详细信息
const entries = await fs.readdir('.', { withFileTypes: true });

for (const entry of entries) {
  if (entry.isFile()) {
    console.log('File:', entry.name);
  } else if (entry.isDirectory()) {
    console.log('Directory:', entry.name);
  } else if (entry.isSymbolicLink()) {
    console.log('Symlink:', entry.name);
  }
}

// ===== 递归读取目录树 =====

async function readDirRecursive(dir) {
  const entries = await fs.readdir(dir, { withFileTypes: true });
  const files = [];
  
  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);
    
    if (entry.isDirectory()) {
      files.push(...await readDirRecursive(fullPath));
    } else {
      files.push(fullPath);
    }
  }
  
  return files;
}

const allFiles = await readDirRecursive('src');
console.log(allFiles);

// ===== 删除目录 =====

// 删除空目录
await fs.rmdir('empty-dir');

// 删除目录及其内容（递归）
await fs.rm('dir-with-files', { recursive: true, force: true });

// ===== 重命名/移动 =====

await fs.rename('old-name.txt', 'new-name.txt');
await fs.rename('file.txt', 'subdir/file.txt');  // 移动文件
```

## 🔍 文件信息查询

```javascript
// ===== stat：获取文件信息 =====

const stats = await fs.stat('file.txt');

console.log('Size:', stats.size, 'bytes');
console.log('Created:', stats.birthtime);
console.log('Modified:', stats.mtime);
console.log('Is File:', stats.isFile());
console.log('Is Directory:', stats.isDirectory());
console.log('Is Symlink:', stats.isSymbolicLink());

// ===== 检查文件是否存在 =====

// ❌ 不推荐：使用 fs.exists()（已弃用）

// ✅ 推荐：使用 fs.access()
try {
  await fs.access('file.txt');
  console.log('File exists');
} catch {
  console.log('File does not exist');
}

// 检查读写权限
try {
  await fs.access('file.txt', fs.constants.R_OK | fs.constants.W_OK);
  console.log('File is readable and writable');
} catch {
  console.log('No read/write permission');
}

// ===== 获取文件大小 =====

const { size } = await fs.stat('file.txt');
console.log(`File size: ${(size / 1024).toFixed(2)} KB`);

// ===== 获取绝对路径 =====

const absolutePath = await fs.realpath('relative/path/file.txt');
console.log(absolutePath);
```

## 👁️ 监听文件变化

### fs.watch()（推荐）

```javascript
import fs from 'fs';

// ===== 监听文件 =====

const watcher = fs.watch('file.txt', (eventType, filename) => {
  console.log('Event:', eventType);  // 'rename' or 'change'
  console.log('File:', filename);
});

// 停止监听
setTimeout(() => {
  watcher.close();
}, 60000);

// ===== 监听目录 =====

const dirWatcher = fs.watch('src', { recursive: true }, (eventType, filename) => {
  console.log(`File ${filename} was ${eventType}d`);
});

// ===== 使用 Promise API =====

import { watch } from 'fs/promises';

const ac = new AbortController();
const { signal } = ac;

try {
  const watcher = watch('file.txt', { signal });
  
  for await (const event of watcher) {
    console.log('Event:', event);
  }
} catch (err) {
  if (err.name === 'AbortError') return;
  throw err;
}

// 停止监听
setTimeout(() => ac.abort(), 60000);
```

### fs.watchFile()（性能较低）

```javascript
import fs from 'fs';

// ===== 轮询方式监听 =====

fs.watchFile('file.txt', { interval: 1000 }, (curr, prev) => {
  console.log('Current mtime:', curr.mtime);
  console.log('Previous mtime:', prev.mtime);
  
  if (curr.mtime !== prev.mtime) {
    console.log('File was modified');
  }
});

// 停止监听
fs.unwatchFile('file.txt');
```

## 🎯 实战案例

### 案例1：递归复制目录

```javascript
async function copyDir(src, dest) {
  await fs.mkdir(dest, { recursive: true });
  
  const entries = await fs.readdir(src, { withFileTypes: true });
  
  for (const entry of entries) {
    const srcPath = path.join(src, entry.name);
    const destPath = path.join(dest, entry.name);
    
    if (entry.isDirectory()) {
      await copyDir(srcPath, destPath);
    } else {
      await fs.copyFile(srcPath, destPath);
    }
  }
}

await copyDir('source-dir', 'backup-dir');
```

### 案例2：文件搜索

```javascript
async function findFiles(dir, pattern) {
  const results = [];
  const entries = await fs.readdir(dir, { withFileTypes: true });
  
  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);
    
    if (entry.isDirectory()) {
      results.push(...await findFiles(fullPath, pattern));
    } else if (entry.name.includes(pattern)) {
      results.push(fullPath);
    }
  }
  
  return results;
}

const jsFiles = await findFiles('src', '.js');
console.log('Found files:', jsFiles);
```

### 案例3：文件锁

```javascript
class FileLock {
  constructor(lockFile) {
    this.lockFile = lockFile;
  }
  
  async acquire(timeout = 5000) {
    const startTime = Date.now();
    
    while (true) {
      try {
        // 尝试创建锁文件（EXCL 标志确保不存在时才创建）
        await fs.writeFile(this.lockFile, process.pid.toString(), { flag: 'wx' });
        return;
      } catch (err) {
        if (Date.now() - startTime > timeout) {
          throw new Error('Failed to acquire lock');
        }
        
        // 等待 100ms 后重试
        await new Promise(resolve => setTimeout(resolve, 100));
      }
    }
  }
  
  async release() {
    try {
      await fs.unlink(this.lockFile);
    } catch (err) {
      // 忽略错误（锁文件可能已被删除）
    }
  }
}

// 使用
const lock = new FileLock('app.lock');

try {
  await lock.acquire();
  console.log('Lock acquired');
  
  // 执行需要互斥的操作
  await doSomething();
  
} finally {
  await lock.release();
  console.log('Lock released');
}
```

### 案例4：日志轮转

```javascript
class LogRotator {
  constructor(logFile, maxSize = 10 * 1024 * 1024) {  // 10MB
    this.logFile = logFile;
    this.maxSize = maxSize;
  }
  
  async write(message) {
    const timestamp = new Date().toISOString();
    const entry = `[${timestamp}] ${message}\n`;
    
    // 检查文件大小
    try {
      const stats = await fs.stat(this.logFile);
      
      if (stats.size > this.maxSize) {
        await this.rotate();
      }
    } catch {
      // 文件不存在，忽略
    }
    
    // 追加日志
    await fs.appendFile(this.logFile, entry);
  }
  
  async rotate() {
    // 重命名旧日志文件
    const timestamp = Date.now();
    const rotatedFile = `${this.logFile}.${timestamp}`;
    
    await fs.rename(this.logFile, rotatedFile);
    
    console.log(`Log rotated to ${rotatedFile}`);
  }
}

// 使用
const logger = new LogRotator('app.log');

await logger.write('Application started');
await logger.write('User logged in');
```

## 💡 最佳实践

```javascript
// ✅ 1. 使用 Promise API
import fs from 'fs/promises';

// ✅ 2. 始终处理错误
try {
  await fs.readFile('file.txt');
} catch (err) {
  console.error('Failed to read file:', err);

import { pipeline } from 'stream/promises';

import path from 'path';

const data = fs.readFileSync('file.txt');  // ❌

fs.readFile('file.txt', (err, data) => {
  console.log(data);  // ❌ 未检查 err
```
