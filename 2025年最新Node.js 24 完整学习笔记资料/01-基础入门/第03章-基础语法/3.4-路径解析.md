# 3.4 路径解析：跨平台文件路径处理

> 掌握 Node.js 中的路径处理，正确使用 __dirname、__filename 及其 ESM 替代方案

---

## 📁 CommonJS 中的路径变量

### __dirname 和 __filename

```javascript
// ===== 文件路径：/Users/alice/project/src/utils/helper.js =====

console.log(__filename);
// 输出：/Users/alice/project/src/utils/helper.js
// 含义：当前文件的绝对路径

console.log(__dirname);
// 输出：/Users/alice/project/src/utils
// 含义：当前文件所在目录的绝对路径

// ===== 常见用法 =====

const path = require('path');
const fs = require('fs');

// 读取同目录下的配置文件
const configPath = path.join(__dirname, 'config.json');
const config = JSON.parse(fs.readFileSync(configPath, 'utf8'));

// 读取上级目录的文件
const dataPath = path.join(__dirname, '..', 'data', 'users.json');

// 构建项目根目录路径
const projectRoot = path.join(__dirname, '../..');
```

### 与 process.cwd() 的区别

```javascript
// ===== 目录结构 =====
// /Users/alice/project/
// ├── src/
// │   └── utils/
// │       └── helper.js
// └── app.js

// ===== helper.js =====
console.log('__dirname:', __dirname);
console.log('process.cwd():', process.cwd());

// ===== 场景1：在项目根目录运行 =====
// $ cd /Users/alice/project
// $ node src/utils/helper.js

// 输出：
// __dirname: /Users/alice/project/src/utils
// process.cwd(): /Users/alice/project

// ===== 场景2：在其他目录运行 =====
// $ cd /tmp
// $ node /Users/alice/project/src/utils/helper.js

// 输出：
// __dirname: /Users/alice/project/src/utils  ← 不变
// process.cwd(): /tmp  ← 变了

// ===== 总结 =====
/*
__dirname：
  - 当前文件所在目录的绝对路径
  - 不受运行位置影响
  - 适用于加载模块内部资源

process.cwd()：
  - Node.js 进程启动时的工作目录
  - 受运行位置影响
  - 适用于加载用户提供的文件
*/
```

## ⚡ ESM 中的路径处理

### 问题：ESM 没有 __dirname

```javascript
// ===== 在 ESM 中使用 __dirname =====

// app.js（ESM）
console.log(__dirname);
// ❌ ReferenceError: __dirname is not defined

console.log(__filename);
// ❌ ReferenceError: __filename is not defined
```

### 解决方案1：使用 import.meta.url

```javascript
// ===== 基础转换 =====

import { fileURLToPath } from 'url';
import { dirname } from 'path';

// 获取当前文件路径
const __filename = fileURLToPath(import.meta.url);

// 获取当前目录路径
const __dirname = dirname(__filename);

console.log(__filename);  // /Users/alice/project/src/app.js
console.log(__dirname);   // /Users/alice/project/src

// ===== 封装成辅助函数 =====

// utils/path-helper.js
import { fileURLToPath } from 'url';
import { dirname } from 'path';

export function getDirname(importMetaUrl) {
  return dirname(fileURLToPath(importMetaUrl));
}

export function getFilename(importMetaUrl) {
  return fileURLToPath(importMetaUrl);
}

// app.js
import { getDirname, getFilename } from './utils/path-helper.js';

const __dirname = getDirname(import.meta.url);
const __filename = getFilename(import.meta.url);
```

### 解决方案2：直接使用 URL API

```javascript
// ===== 使用 URL 构建路径 =====

// 读取同目录下的文件
const configUrl = new URL('./config.json', import.meta.url);

console.log(configUrl.href);
// 输出：file:///Users/alice/project/src/config.json

// 转换为文件系统路径
import { fileURLToPath } from 'url';
const configPath = fileURLToPath(configUrl);

console.log(configPath);
// 输出：/Users/alice/project/src/config.json

// ===== 读取文件 =====

import { readFileSync } from 'fs';

const config = JSON.parse(
  readFileSync(new URL('./config.json', import.meta.url), 'utf8')
);

// ===== 构建相对路径 =====

// 读取上级目录的文件
const dataUrl = new URL('../data/users.json', import.meta.url);

// 读取项目根目录的文件
const rootUrl = new URL('../../package.json', import.meta.url);
```

## 🛠️ path 模块详解

### 核心方法

```javascript
import path from 'path';

// ===== path.join()：拼接路径 =====

const joined = path.join('/users', 'alice', 'documents', 'file.txt');
console.log(joined);
// 输出：/users/alice/documents/file.txt

// 处理 .. 和 .
const normalized = path.join('/users', 'alice', '..', 'bob', './file.txt');
console.log(normalized);
// 输出：/users/bob/file.txt

// ===== path.resolve()：解析为绝对路径 =====

// 从右到左处理，直到构造出绝对路径
console.log(path.resolve('users', 'alice', 'file.txt'));
// 输出：/current/working/directory/users/alice/file.txt

console.log(path.resolve('/users', 'alice', 'file.txt'));
// 输出：/users/alice/file.txt  ← 遇到 / 停止

console.log(path.resolve('users', '/alice', 'file.txt'));
// 输出：/alice/file.txt  ← 遇到 / 重置

// ===== join vs resolve 对比 =====

// join：简单拼接
path.join('a', 'b', 'c');      // a/b/c
path.join('/a', 'b', 'c');     // /a/b/c

// resolve：解析绝对路径
path.resolve('a', 'b', 'c');   // /cwd/a/b/c
path.resolve('/a', 'b', 'c');  // /a/b/c

// ===== path.dirname()：获取目录名 =====

console.log(path.dirname('/users/alice/file.txt'));
// 输出：/users/alice

console.log(path.dirname('/users/alice/'));
// 输出：/users

// ===== path.basename()：获取文件名 =====

console.log(path.basename('/users/alice/file.txt'));
// 输出：file.txt

// 去掉扩展名
console.log(path.basename('/users/alice/file.txt', '.txt'));
// 输出：file

// ===== path.extname()：获取扩展名 =====

console.log(path.extname('file.txt'));       // .txt
console.log(path.extname('file.tar.gz'));    // .gz
console.log(path.extname('file.'));          // .
console.log(path.extname('file'));           // ''
console.log(path.extname('.gitignore'));     // ''

// ===== path.parse()：解析路径 =====

const parsed = path.parse('/users/alice/file.txt');
console.log(parsed);
// 输出：
// {
//   root: '/',
//   dir: '/users/alice',
//   base: 'file.txt',
//   ext: '.txt',
//   name: 'file'
// }

// ===== path.format()：格式化路径 =====

const formatted = path.format({
  root: '/',
  dir: '/users/alice',
  base: 'file.txt'
});
console.log(formatted);
// 输出：/users/alice/file.txt

// ===== path.normalize()：规范化路径 =====

console.log(path.normalize('/users//alice/../bob/./file.txt'));
// 输出：/users/bob/file.txt

// ===== path.relative()：计算相对路径 =====

const from = '/users/alice/project';
const to = '/users/bob/documents/file.txt';

console.log(path.relative(from, to));
// 输出：../../bob/documents/file.txt

// ===== path.isAbsolute()：判断是否绝对路径 =====

console.log(path.isAbsolute('/users/alice'));  // true
console.log(path.isAbsolute('users/alice'));   // false
console.log(path.isAbsolute('./users'));       // false
```

### 路径分隔符

```javascript
import path from 'path';

// ===== path.sep：路径分隔符 =====

console.log(path.sep);
// Windows：\
// Unix/Linux/macOS：/

const parts = '/users/alice/file.txt'.split(path.sep);
console.log(parts);
// Unix：['', 'users', 'alice', 'file.txt']

// ===== path.delimiter：环境变量分隔符 =====

console.log(path.delimiter);
// Windows：;
// Unix/Linux/macOS：:

console.log(process.env.PATH.split(path.delimiter));
// Unix：['/usr/local/bin', '/usr/bin', '/bin', ...]
// Windows：['C:\\Windows', 'C:\\Program Files', ...]

// ===== 跨平台路径 =====

// ❌ 不要硬编码分隔符
const badPath = 'users/alice/file.txt';  // Unix 风格
const badPath2 = 'users\\alice\\file.txt';  // Windows 风格

// ✅ 使用 path.join
const goodPath = path.join('users', 'alice', 'file.txt');
// 自动使用正确的分隔符
```

## 🌍 跨平台路径处理

### Windows vs Unix 路径差异

```javascript
// ===== Windows 路径 =====

// 绝对路径
C:\Users\Alice\Documents\file.txt

// UNC 路径（网络路径）
\\server\share\folder\file.txt

// 相对路径
users\alice\file.txt

// 驱动器相对路径
C:file.txt  ← 相对于 C: 盘的当前目录

// ===== Unix/Linux/macOS 路径 =====

// 绝对路径
/Users/Alice/Documents/file.txt

// 相对路径
users/alice/file.txt

// 主目录
~/Documents/file.txt  ← shell 扩展，Node.js 不支持

// ===== Node.js 处理 =====

import path from 'path';

// path.win32：强制使用 Windows 风格
console.log(path.win32.join('users', 'alice', 'file.txt'));
// 输出：users\alice\file.txt（任何平台）

// path.posix：强制使用 Unix 风格
console.log(path.posix.join('users', 'alice', 'file.txt'));
// 输出：users/alice/file.txt（任何平台）

// path：使用当前平台风格
console.log(path.join('users', 'alice', 'file.txt'));
// macOS/Linux：users/alice/file.txt
// Windows：users\alice\file.txt
```

### 处理用户主目录

```javascript
import os from 'os';
import path from 'path';

// ===== 获取用户主目录 =====

const homeDir = os.homedir();
console.log(homeDir);
// macOS：/Users/alice
// Linux：/home/alice
// Windows：C:\Users\Alice

// ===== 构建主目录下的路径 =====

const configPath = path.join(homeDir, '.config', 'myapp', 'config.json');
console.log(configPath);
// macOS：/Users/alice/.config/myapp/config.json
// Windows：C:\Users\Alice\.config\myapp\config.json

// ===== 处理 ~ 符号（需要手动扩展）=====

function expandTilde(filepath) {
  if (filepath.startsWith('~/') || filepath === '~') {
    return path.join(os.homedir(), filepath.slice(1));
  }
  return filepath;
}

console.log(expandTilde('~/Documents/file.txt'));
// 输出：/Users/alice/Documents/file.txt
```

## 🎯 实战案例

### 案例1：配置文件加载

```javascript
// ===== config-loader.js（CommonJS） =====

const path = require('path');
const fs = require('fs');

function loadConfig(filename = 'config.json') {
  // 优先级：项目根目录 > 用户主目录 > 默认配置
  
  const locations = [
    path.join(process.cwd(), filename),
    path.join(os.homedir(), '.myapp', filename),
    path.join(__dirname, 'defaults', filename)
  ];
  
  for (const location of locations) {
    if (fs.existsSync(location)) {
      return JSON.parse(fs.readFileSync(location, 'utf8'));
    }
  }
  
  throw new Error('配置文件未找到');
}

// ===== config-loader.js（ESM） =====

import { fileURLToPath } from 'url';
import { dirname, join } from 'path';
import { readFileSync, existsSync } from 'fs';
import os from 'os';

const __dirname = dirname(fileURLToPath(import.meta.url));

function loadConfig(filename = 'config.json') {
  const locations = [
    join(process.cwd(), filename),
    join(os.homedir(), '.myapp', filename),
    join(__dirname, 'defaults', filename)
  ];
  
  for (const location of locations) {
    if (existsSync(location)) {
      return JSON.parse(readFileSync(location, 'utf8'));
    }
  }
  
  throw new Error('配置文件未找到');
}
```

### 案例2：文件查找

```javascript
// ===== 向上查找文件（类似 Git 查找 .git）=====

import { dirname, join } from 'path';
import { existsSync } from 'fs';

function findUp(filename, startDir = process.cwd()) {
  let currentDir = startDir;
  
  while (true) {
    const filePath = join(currentDir, filename);
    
    if (existsSync(filePath)) {
      return filePath;
    }
    
    const parentDir = dirname(currentDir);
    
    // 到达根目录
    if (parentDir === currentDir) {
      return null;
    }
    
    currentDir = parentDir;
  }
}

// 查找项目根目录的 package.json
const packageJsonPath = findUp('package.json');
console.log('项目根目录:', dirname(packageJsonPath));
```

### 案例3：动态加载模块

```javascript
// ===== 按目录结构自动加载路由 =====

import { readdirSync } from 'fs';
import { join, extname } from 'path';
import { fileURLToPath } from 'url';
import { dirname } from 'path';

const __dirname = dirname(fileURLToPath(import.meta.url));

async function loadRoutes(routesDir) {
  const files = readdirSync(routesDir);
  const routes = [];
  
  for (const file of files) {
    // 只加载 .js 文件
    if (extname(file) !== '.js') continue;
    
    const filePath = join(routesDir, file);
    const route = await import(filePath);
    
    routes.push(route.default);
  }
  
  return routes;
}

// 使用
const routes = await loadRoutes(join(__dirname, 'routes'));
```

## 💡 最佳实践

### ✅ 推荐做法

```javascript
// 1. ESM 中始终定义 __dirname
import { fileURLToPath } from 'url';
import { dirname } from 'path';

const __dirname = dirname(fileURLToPath(import.meta.url));

// 2. 使用 path.join 拼接路径
import path from 'path';

const filePath = path.join(__dirname, 'data', 'users.json');

// 3. 使用 path.resolve 获取绝对路径
const absolutePath = path.resolve('data', 'users.json');

// 4. 跨平台代码不要硬编码路径分隔符
const badPath = __dirname + '/data/file.txt';  // ❌
const goodPath = path.join(__dirname, 'data', 'file.txt');  // ✅

// 5. 使用 URL API 处理模块内部资源
const configUrl = new URL('./config.json', import.meta.url);
```

### ❌ 避免的做法

```javascript
// ❌ 硬编码路径分隔符
const path1 = 'users/alice/file.txt';
const path2 = 'users\\alice\\file.txt';

// ❌ 字符串拼接路径
const path3 = __dirname + '/' + 'file.txt';

// ❌ ESM 中直接使用 __dirname
console.log(__dirname);  // ReferenceError

// ❌ 混用绝对路径和相对路径
const confused = path.join('/absolute', '../relative');

// ❌ 不检查文件是否存在
const data = fs.readFileSync('./config.json');  // 可能报错
```

## 📊 路径处理速查表

| 需求 | CommonJS | ES Modules |
|------|----------|-----------|
| **当前文件路径** | `__filename` | `fileURLToPath(import.meta.url)` |
| **当前目录路径** | `__dirname` | `dirname(fileURLToPath(import.meta.url))` |
| **工作目录** | `process.cwd()` | `process.cwd()` |
| **拼接路径** | `path.join()` | `path.join()` |
| **绝对路径** | `path.resolve()` | `path.resolve()` |
| **相对路径** | `path.relative()` | `path.relative()` |
| **模块内资源** | `path.join(__dirname, 'file')` | `new URL('./file', import.meta.url)` |
