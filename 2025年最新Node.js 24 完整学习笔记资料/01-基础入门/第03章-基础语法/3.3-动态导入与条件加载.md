# 3.3 动态导入与条件加载

## 🚀 动态导入 import()

### 为什么需要动态导入?

```javascript
// ===== 静态导入的局限 =====

// ❌ 不能条件导入
if (isDev) {
  import devConfig from './dev.config.js';  // SyntaxError
}

// ❌ 不能在函数内导入
function loadModule() {
  import fs from 'fs';  // SyntaxError
}

// ❌ 不能使用变量路径
const moduleName = 'express';
import express from moduleName;  // SyntaxError

// ===== 动态导入的解决方案 =====

// ✅ 条件导入
if (isDev) {
  const devConfig = await import('./dev.config.js');
}

// ✅ 函数内导入
async function loadModule() {
  const fs = await import('fs');
}

// ✅ 变量路径
const moduleName = './express.js';
const express = await import(moduleName);
```

### 基础语法

```javascript
// ===== 动态导入返回 Promise =====

import('./module.js')
  .then((module) => {
    console.log(module.default);  // 默认导出
    console.log(module.name);     // 具名导出
  })
  .catch((error) => {
    console.error('加载失败:', error);
  });

// ===== 使用 async/await（推荐） =====

async function loadModule() {
  try {
    const module = await import('./module.js');
    
    console.log(module.default);  // 默认导出
    console.log(module.name);     // 具名导出
  } catch (error) {
    console.error('加载失败:', error);
  }
}

// ===== 解构导入 =====

const { default: defaultExport, name, age } = await import('./user.js');

console.log(defaultExport);  // 默认导出
console.log(name);           // 'Alice'
console.log(age);            // 25
```

// ===== 根据环境加载不同配置 =====

const env = process.env.NODE_ENV || 'development';

let config;
if (env === 'production') {
  config = await import('./config.prod.js');
} else if (env === 'development') {
  config = await import('./config.dev.js');
} else {
  config = await import('./config.test.js');

export default config.default;

async function processData(data, useCache = false) {

    const { Cache } = await import('./cache.js');
    const cache = new Cache();
    
    const cached = cache.get(data.id);
    if (cached) return cached;
  }
  
  // 处理数据...
  return result;
}

// ===== 按需加载调试工具 =====

if (process.env.DEBUG === 'true') {

### 2. 懒加载优化

```javascript
// ===== Express 路由懒加载 =====

import express from 'express';

const app = express();

// 路由懒加载：只在访问时加载对应的处理器
app.get('/users', async (req, res) => {
  const { getUsersHandler } = await import('./handlers/users.js');
  return getUsersHandler(req, res);
});

app.post('/products', async (req, res) => {
  const { createProductHandler } = await import('./handlers/products.js');
  return createProductHandler(req, res);
});

// ===== 大型模块按需加载 =====

async function generatePDF(data) {
  // PDFKit 库很大（~2MB），只在需要时加载
  const PDFDocument = (await import('pdfkit')).default;
  
  const doc = new PDFDocument();
  // 生成 PDF...
  return doc;
}

// ===== 数据处理工具按需加载 =====

async function processLargeDataset(data) {
  if (data.length > 10000) {
    // 大数据集时加载 lodash
    const _ = (await import('lodash')).default;
    return _.chunk(data, 1000);
  } else {
    // 小数据集直接处理
    return [data];
  }
}
```

### 3. 插件系统

```javascript
// ===== 动态加载插件 =====

class PluginManager {
  constructor() {
    this.plugins = [];
  }
  
  async loadPlugin(pluginName) {
    try {
      // 动态加载插件
      const plugin = await import(`./plugins/${pluginName}.js`);
      
      // 初始化插件
      const instance = new plugin.default();
      await instance.init();
      
      this.plugins.push(instance);
      console.log(`插件 ${pluginName} 加载成功`);
    } catch (error) {
      console.error(`插件 ${pluginName} 加载失败:`, error);
    }
  }
  
  async loadAll(pluginNames) {
    await Promise.all(
      pluginNames.map(name => this.loadPlugin(name))
    );
  }
}

// 使用
const manager = new PluginManager();
await manager.loadAll(['logger', 'cache', 'auth']);

// ===== 配置文件动态加载 =====

async function loadConfig(format) {
  const loaders = {
    json: () => import('./loaders/json-loader.js'),
    yaml: () => import('./loaders/yaml-loader.js'),
    toml: () => import('./loaders/toml-loader.js')
  };
  
  const loader = await loaders[format]();
  return loader.load('./config.file');
}

const config = await loadConfig('yaml');
```

### 4. 代码分割

```javascript
// ===== 按功能分割代码 =====

class Application {
  constructor() {
    this.modules = {};
  }
  
  async loadModule(name) {
    if (this.modules[name]) {
      return this.modules[name];
    }
    
    // 按需加载模块
    const module = await import(`./modules/${name}.js`);
    this.modules[name] = module.default;
    
    return module.default;
  }
}

const app = new Application();

// 只加载需要的功能
const userModule = await app.loadModule('user');
const orderModule = await app.loadModule('order');

// ===== CLI 工具命令分割 =====

async function runCommand(command, args) {
  const commands = {
    build: () => import('./commands/build.js'),
    test: () => import('./commands/test.js'),
    deploy: () => import('./commands/deploy.js')
  };
  
  const commandModule = await commands[command]();
  await commandModule.default(args);
}

// 只加载执行的命令
await runCommand('build', ['--production']);
```

## 📍 import.meta

### import.meta.url

```javascript
// ===== 获取当前模块的 URL =====

console.log(import.meta.url);
// 输出：file:///path/to/current/module.js

// ===== 构建相对路径 =====

import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

console.log(__filename);  // /path/to/current/module.js
console.log(__dirname);   // /path/to/current

// 读取同目录文件
const configPath = join(__dirname, 'config.json');

// ===== 动态导入相对路径 =====

// 相对于当前模块的路径
const sibling = await import('./sibling.js');

// 使用 URL API 构建路径
const modulePath = new URL('./utils/helper.js', import.meta.url);
const helper = await import(modulePath.href);
```

### import.meta.resolve()（Node.js 20.6+）

```javascript
// ===== 解析模块路径 =====

// 解析相对路径
const resolved = import.meta.resolve('./module.js');
console.log(resolved);
// 输出：file:///path/to/module.js

// 解析 npm 包
const expressPath = import.meta.resolve('express');
console.log(expressPath);
// 输出：file:///path/to/node_modules/express/index.js

// ===== 检查模块是否存在 =====

async function moduleExists(moduleName) {
  try {
    await import.meta.resolve(moduleName);
    return true;
  } catch {
    return false;
  }
}

const hasExpress = await moduleExists('express');
console.log('Express 已安装:', hasExpress);

// ===== 预加载模块路径 =====

const modulePaths = [
  import.meta.resolve('./module1.js'),
  import.meta.resolve('./module2.js'),
  import.meta.resolve('./module3.js')
];

// 稍后按需加载
const module1 = await import(modulePaths[0]);
```

## ⚡ 性能优化技巧

### 1. 并行加载

```javascript
// ===== ❌ 串行加载（慢）=====

const module1 = await import('./module1.js');
const module2 = await import('./module2.js');
const module3 = await import('./module3.js');

// 总耗时：time1 + time2 + time3

// ===== ✅ 并行加载（快）=====

const [module1, module2, module3] = await Promise.all([
  import('./module1.js'),
  import('./module2.js'),
  import('./module3.js')
]);

// 总耗时：max(time1, time2, time3)
```

### 2. 缓存模块

```javascript
// ===== 模块缓存 =====

const moduleCache = new Map();

async function loadModule(modulePath) {
  // 检查缓存
  if (moduleCache.has(modulePath)) {
    return moduleCache.get(modulePath);
  }
  
  // 加载模块
  const module = await import(modulePath);
  
  // 缓存模块
  moduleCache.set(modulePath, module);
  
  return module;
}

// 多次调用只加载一次
const mod1 = await loadModule('./module.js');
const mod2 = await loadModule('./module.js');  // 从缓存获取

console.log(mod1 === mod2);  // true
```

### 3. 预加载

```javascript
// ===== 预加载关键模块 =====

// 在应用启动时预加载
const preloadModules = [
  import('./critical-module1.js'),
  import('./critical-module2.js')
];

// 等待预加载完成
await Promise.all(preloadModules);

// 稍后使用时已经加载好了
const module1 = await import('./critical-module1.js');  // 立即返回

// ===== 预加载策略 =====

class ModulePreloader {
  constructor() {
    this.preloaded = new Set();
  }
  
  async preload(paths) {
    const promises = paths
      .filter(path => !this.preloaded.has(path))
      .map(async path => {
        await import(path);
        this.preloaded.add(path);
      });
    
    await Promise.all(promises);
  }
  
  async load(path) {
    if (!this.preloaded.has(path)) {
      await this.preload([path]);
    }
    return import(path);
  }
}

const preloader = new ModulePreloader();

// 预加载常用模块
await preloader.preload([
  './user.js',
  './auth.js',
  './db.js'
]);

// 使用时立即可用
const user = await preloader.load('./user.js');
```

### 4. 条件预加载

```javascript
// ===== 基于用户行为预加载 =====

// 鼠标悬停时预加载
button.addEventListener('mouseenter', () => {
  // 预加载下一个页面的代码
  import('./next-page.js');
});

// 空闲时预加载
if ('requestIdleCallback' in window) {
  requestIdleCallback(() => {
    import('./low-priority-module.js');
  });
}

// ===== 基于网络状况 =====

async function smartPreload(modulePath) {
  // 检查网络连接
  if (navigator.connection) {
    const { effectiveType, saveData } = navigator.connection;
    
    // 4G 且未开启数据节省模式
    if (effectiveType === '4g' && !saveData) {
      await import(modulePath);
    }
  }
}

smartPreload('./large-module.js');
```

## 🔄 CommonJS 动态导入对比

```javascript
// ===== CommonJS：同步 require =====

// ✅ 可以在任何地方使用
function loadModule(name) {
  return require(`./${name}.js`);  // 同步加载
}

// ✅ 可以使用变量
const moduleName = 'express';
const express = require(moduleName);

// ✅ 可以条件加载
if (isDev) {
  const devTools = require('./dev-tools.js');
}

// ❌ 但是会阻塞主线程
const largeModule = require('./large-module.js');  // 阻塞

// ===== ES Modules：异步 import() =====

// ✅ 异步加载，不阻塞
const largeModule = await import('./large-module.js');

// ✅ 可以并行加载
const [mod1, mod2] = await Promise.all([
  import('./mod1.js'),
  import('./mod2.js')
]);

// ✅ 支持 Tree Shaking
const { specificFunction } = await import('./utils.js');

// ⚠️ 返回 Promise，需要 await
const module = await import('./module.js');
```

## 💡 最佳实践

### ✅ 推荐做法

```javascript
// 1. 使用 async/await 处理动态导入
async function loadModule() {
  const module = await import('./module.js');
  return module.default;
}

// 2. 错误处理
try {
  const module = await import('./module.js');
} catch (error) {
  console.error('模块加载失败:', error);
}

// 3. 并行加载多个模块
const modules = await Promise.all([
  import('./mod1.js'),
  import('./mod2.js')
]);

// 4. 使用 import.meta.url 构建相对路径
const configPath = new URL('./config.json', import.meta.url);

// 5. 懒加载大型依赖
async function heavyOperation() {
  const { default: heavyLib } = await import('heavy-library');
  return heavyLib.process();
}
```

### ❌ 避免的做法

```javascript
// ❌ 不处理错误
const module = await import('./module.js');  // 可能失败

// ❌ 过度使用动态导入（影响性能）
for (let i = 0; i < 100; i++) {
  await import(`./module${i}.js`);  // 串行加载
}

// ❌ 在循环中动态导入（使用 Promise.all）
const modules = [];
for (let i = 0; i < 10; i++) {
  modules.push(import(`./module${i}.js`));
}
await Promise.all(modules);  // ✅ 这样才对

// ❌ 忘记 await
const module = import('./module.js');  // 返回 Promise，不是模块
module.default();  // ❌ TypeError

// ✅ 正确做法
const module = await import('./module.js');
module.default();
```

### 案例2：国际化按需加载

```javascript
// i18n.js
class I18n {
  constructor() {
    this.locale = 'zh-CN';
    this.messages = null;
  }
  
  async setLocale(locale) {
    this.locale = locale;
    
    // 按需加载对应语言包
    const messages = await import(`./locales/${locale}.js`);
    this.messages = messages.default;
  }
  
  t(key) {
    return this.messages?.[key] || key;
  }
}

const i18n = new I18n();
await i18n.setLocale('zh-CN');

console.log(i18n.t('hello'));  // '你好'

await i18n.setLocale('en-US');
console.log(i18n.t('hello'));  // 'Hello'
```
