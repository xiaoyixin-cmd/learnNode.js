# 3.2 模块系统双轨制：CommonJS vs ES Modules

> 深入理解 Node.js 的两种模块系统，掌握它们的区别和使用场景

---

## 📦 为什么有两种模块系统？

```
历史演进：

2009 年    Node.js 诞生 → 采用 CommonJS 模块系统
   │       └─ 当时 JavaScript 没有官方模块标准
   │
2015 年    ES6 发布 → 引入 ES Modules (ESM)
   │       └─ JavaScript 官方模块标准
   │
2017 年    Node.js 8.5 实验性支持 ESM
   │       └─ 需要 --experimental-modules 标志
   │
2020 年    Node.js 12 正式支持 ESM
   │       └─ 需要 .mjs 扩展名或 type: "module"
   │
2021 年    Node.js 16+ ESM 成为主流
   │       └─ 支持 Top-Level await
   │
2025 年    CommonJS 仍广泛使用，ESM 成为新项目首选
   │       └─ 两种系统共存，Node.js 完全支持
   ↓
```

---

## 🔧 CommonJS 模块系统

### 基础语法

```javascript
// ===== module.exports 导出 =====

// user.js
const name = 'Alice';
const age = 25;

function greet() {
  return `Hello, I'm ${name}`;
}

// 方式1：整体导出
module.exports = {
  name,
  age,
  greet
};

// 方式2：逐个导出
module.exports.name = name;
module.exports.age = age;
module.exports.greet = greet;

// 方式3：使用 exports 简写（注意限制）
exports.name = name;
exports.age = age;
exports.greet = greet;

// ⚠️ 错误用法
// exports = { name, age, greet };  // ❌ 这样不会导出
// 原因：exports 只是 module.exports 的引用
// 重新赋值会断开引用
```

```javascript
// ===== require 导入 =====

// app.js
const user = require('./user.js');

console.log(user.name);     // 'Alice'
console.log(user.age);      // 25
console.log(user.greet());  // 'Hello, I'm Alice'

// 解构导入
const { name, age, greet } = require('./user.js');

console.log(name);    // 'Alice'
console.log(age);     // 25
console.log(greet()); // 'Hello, I'm Alice'
```

### CommonJS 工作原理

```javascript
// ===== Node.js 内部实现（简化版） =====

function require(modulePath) {
  // 1. 解析模块路径
  const absolutePath = resolve(modulePath);
  
  // 2. 检查缓存
  if (require.cache[absolutePath]) {
    return require.cache[absolutePath].exports;
  }
  
  // 3. 创建模块对象
  const module = {
    exports: {},
    id: absolutePath,
    filename: absolutePath,
    loaded: false
  };
  
  // 4. 加载模块代码
  const code = fs.readFileSync(absolutePath, 'utf8');
  
  // 5. 包装成函数
  const wrapper = `(function(exports, require, module, __filename, __dirname) {
    ${code}
  })`;
  
  // 6. 执行函数
  const fn = eval(wrapper);
  fn(module.exports, require, module, absolutePath, path.dirname(absolutePath));
  
  // 7. 标记为已加载
  module.loaded = true;
  
  // 8. 缓存模块
  require.cache[absolutePath] = module;
  
  // 9. 返回 exports
  return module.exports;
}

// ===== 模块包装器 =====

// 你写的代码：
const name = 'Alice';
module.exports = { name };

// Node.js 实际执行的代码：
(function(exports, require, module, __filename, __dirname) {
  const name = 'Alice';
  module.exports = { name };
});

// 这就是为什么模块内可以直接使用：
// - exports
// - require
// - module
// - __filename
// - __dirname
```

### 模块缓存

```javascript
// ===== counter.js =====
let count = 0;

function increment() {
  count++;
}

function getCount() {
  return count;
}

module.exports = { increment, getCount };

// ===== app.js =====
const counter1 = require('./counter.js');
const counter2 = require('./counter.js');

counter1.increment();
counter1.increment();

console.log(counter1.getCount());  // 2
console.log(counter2.getCount());  // 2 ← 共享同一个实例

console.log(counter1 === counter2);  // true

// ===== 清除缓存 =====
delete require.cache[require.resolve('./counter.js')];

const counter3 = require('./counter.js');
console.log(counter3.getCount());  // 0 ← 新实例
```

### 循环依赖

```javascript
// ===== a.js =====
console.log('a.js 开始');
exports.done = false;

const b = require('./b.js');
console.log('在 a.js 中，b.done =', b.done);

exports.done = true;
console.log('a.js 结束');

// ===== b.js =====
console.log('b.js 开始');
exports.done = false;

const a = require('./a.js');
console.log('在 b.js 中，a.done =', a.done);

exports.done = true;
console.log('b.js 结束');

// ===== main.js =====
const a = require('./a.js');
const b = require('./b.js');

console.log('在 main.js 中，a.done =', a.done);
console.log('在 main.js 中，b.done =', b.done);

// ===== 输出 =====
// a.js 开始
// b.js 开始
// 在 b.js 中，a.done = false  ← a.js 还未执行完
// b.js 结束
// 在 a.js 中，b.done = true
// a.js 结束
// 在 main.js 中，a.done = true
// 在 main.js 中，b.done = true
```

## ⚡ ES Modules (ESM)

### 基础语法

```javascript
// ===== export 导出 =====

// user.js
export const name = 'Alice';
export const age = 25;

export function greet() {
  return `Hello, I'm ${name}`;
}

// 或者统一导出
const name = 'Alice';
const age = 25;
function greet() {
  return `Hello, I'm ${name}`;
}

export { name, age, greet };

// ===== export default 默认导出 =====

// config.js
const config = {
  apiUrl: 'https://api.example.com',
  timeout: 5000
};

export default config;

// 或者直接导出
export default {
  apiUrl: 'https://api.example.com',
  timeout: 5000
};

// 函数默认导出
export default function greet(name) {
  return `Hello, ${name}`;
}

// ⚠️ 注意：一个模块只能有一个 default 导出
```

```javascript
// ===== import 导入 =====

// app.js

// 具名导入
import { name, age, greet } from './user.js';

console.log(name);    // 'Alice'
console.log(age);     // 25
console.log(greet()); // 'Hello, I'm Alice'

// 重命名导入
import { name as userName, age as userAge } from './user.js';

// 导入所有
import * as user from './user.js';
console.log(user.name);

// 默认导入
import config from './config.js';
console.log(config.apiUrl);

// 混合导入
import greet, { name, age } from './user.js';

// 只执行模块，不导入
import './side-effects.js';
```

### 启用 ESM

#### 方式1：package.json 配置（推荐）

```json
{
  "type": "module"
}
```

```javascript
// 现在 .js 文件默认是 ESM
// index.js
import express from 'express';

// 如需使用 CommonJS，使用 .cjs 扩展名
// config.cjs
module.exports = { ... };
```

#### 方式2：使用 .mjs 扩展名

```javascript
// index.mjs（自动识别为 ESM）
import express from 'express';
```

#### 方式3：不设置，默认 CommonJS

```json
{
  "type": "commonjs"  // 或不设置
}
```

```javascript
// .js 文件默认是 CommonJS
// index.js
const express = require('express');

// 如需使用 ESM，使用 .mjs 扩展名
// app.mjs
import express from 'express';
```

### ESM 特性

```javascript
// ===== 1. 静态导入（编译时确定） =====

// ✅ 正确：顶层导入
import fs from 'fs';

// ❌ 错误：条件导入
if (condition) {
  import fs from 'fs';  // SyntaxError
}

// ❌ 错误：函数内导入
function loadModule() {
  import fs from 'fs';  // SyntaxError
}

// 💡 动态导入使用 import()（见下节）

// ===== 2. 导入的是只读引用 =====

// counter.js
export let count = 0;

export function increment() {
  count++;
}

// app.js
import { count, increment } from './counter.js';

console.log(count);  // 0
increment();
console.log(count);  // 1 ← 引用会更新

// count = 10;  // ❌ TypeError: Assignment to constant variable

// ===== 3. 严格模式 =====

// ESM 自动启用严格模式
// 'use strict';  ← 不需要手动添加

// 以下在严格模式下会报错：
// - 未声明变量
// - with 语句
// - 八进制字面量
// - 删除变量、函数
```

### Top-Level await（2025 主流）

```javascript
// ===== 传统方式（CommonJS）=====

// ❌ 不能在顶层使用 await
const data = await fetchData();  // SyntaxError

// 必须包装在 async 函数中
async function main() {
  const data = await fetchData();
  console.log(data);
}

main();

// ===== ESM 方式（Node.js 14.8+）=====

// config.mjs（需要 type: "module"）

// ✅ 可以在顶层直接使用 await
const response = await fetch('https://api.example.com/config');
const config = await response.json();

export default config;

// ===== 实际应用 =====

// db.js
import mongoose from 'mongoose';

const DB_URL = process.env.DB_URL;

// 顶层 await：连接数据库
await mongoose.connect(DB_URL);

console.log('数据库连接成功');

export default mongoose;

// app.js
import './db.js';  // db.js 执行完毕后才继续
import express from 'express';

const app = express();
// 这里数据库已经连接成功

app.listen(3000);
```

## ⚖️ CommonJS vs ES Modules 对比

### 语法对比

| 特性 | CommonJS | ES Modules |
|------|----------|-----------|
| **导出** | `module.exports = {}` | `export {}` |
| **导入** | `const m = require()` | `import m from ''` |
| **默认导出** | `module.exports = fn` | `export default fn` |
| **具名导出** | `exports.name = val` | `export const name` |
| **动态导入** | `require()` | `import()` |
| **文件扩展名** | `.js` / `.cjs` | `.js` (type:module) / `.mjs` |
| **顶层 await** | ❌ | ✅ |
| **严格模式** | 可选 | ✅ 强制 |

### 加载机制对比

```javascript
// ===== CommonJS：同步加载，运行时执行 =====

// 1. 可以条件加载
if (isDev) {
  const devConfig = require('./dev.config.js');
}

// 2. 可以动态路径
const moduleName = 'express';
const express = require(moduleName);

// 3. 导入的是值的拷贝
// counter.js
let count = 0;
module.exports = {
  count,
  increment() { count++; }
};

// app.js
const counter = require('./counter.js');
console.log(counter.count);  // 0
counter.increment();
console.log(counter.count);  // 0 ← 不会更新

// ===== ES Modules：异步加载，编译时执行 =====

// 1. 必须顶层导入（静态分析）
// if (isDev) {
//   import devConfig from './dev.config.js';  // ❌ SyntaxError
// }

// 2. 不能使用变量路径
// const moduleName = 'express';
// import express from moduleName;  // ❌ SyntaxError

// 3. 导入的是值的引用
// counter.js
export let count = 0;
export function increment() { count++; }

// app.js
import { count, increment } from './counter.js';
console.log(count);  // 0
increment();
console.log(count);  // 1 ← 会更新
```

### Tree Shaking

```javascript
// ===== CommonJS：无法 Tree Shaking =====

// utils.js
module.exports = {
  add: (a, b) => a + b,
  subtract: (a, b) => a - b,
  multiply: (a, b) => a * b,
  divide: (a, b) => a / b
};

// app.js
const { add } = require('./utils.js');
console.log(add(1, 2));

// 打包后：所有函数都会被包含（包括未使用的）

// ===== ES Modules：支持 Tree Shaking =====

// utils.js
export const add = (a, b) => a + b;
export const subtract = (a, b) => a - b;
export const multiply = (a, b) => a * b;
export const divide = (a, b) => a / b;

// app.js
import { add } from './utils.js';
console.log(add(1, 2));

// 打包后：只包含 add 函数（未使用的函数被移除）
```

## 🔄 两种模块系统互操作

### ESM 中使用 CommonJS

```javascript
// ===== 导入 CommonJS 模块 =====

// cjs-module.js（CommonJS）
module.exports = {
  name: 'Alice',
  greet() {
    return 'Hello';
  }
};

// esm-module.js（ESM）
import cjsModule from './cjs-module.js';

console.log(cjsModule.name);     // 'Alice'
console.log(cjsModule.greet());  // 'Hello'

// ⚠️ 不能使用具名导入
// import { name } from './cjs-module.js';  // ❌ SyntaxError
```

### CommonJS 中使用 ESM

```javascript
// ===== 使用动态 import() =====

// esm-module.js（ESM）
export const name = 'Alice';
export function greet() {
  return 'Hello';
}

// cjs-module.js（CommonJS）
async function loadESM() {
  const esmModule = await import('./esm-module.js');
  
  console.log(esmModule.name);     // 'Alice'
  console.log(esmModule.greet());  // 'Hello'
}

loadESM();

// ⚠️ 不能使用 require
// const esmModule = require('./esm-module.js');  // ❌ Error
```

### 使用 createRequire

```javascript
// ===== ESM 中获取 require =====

import { createRequire } from 'module';

const require = createRequire(import.meta.url);

// 现在可以使用 require
const express = require('express');
const packageJson = require('./package.json');
```

## 🎯 选择建议（2025）

### 新项目：推荐 ESM

```json
// package.json
{
  "type": "module",
  "main": "./src/index.js",
  "exports": {
    ".": "./src/index.js"
  }
}
```

```javascript
// src/index.js
import express from 'express';

const app = express();

app.listen(3000);
```

**理由：**
```
✅ JavaScript 官方标准
✅ 更好的静态分析
✅ 支持 Tree Shaking
✅ 支持 Top-Level await
✅ 更好的 TypeScript 集成
✅ 前后端代码统一
```

### 旧项目：保持 CommonJS

```javascript
// 如果项目已经使用 CommonJS
// 不急于迁移，等合适时机再迁移
const express = require('express');

module.exports = app;
```

**理由：**
```
✅ 稳定可靠
✅ 兼容性好
✅ 大量第三方库仍使用 CommonJS
✅ 迁移成本高
```

### 库开发：双模块支持

```json
// package.json
{
  "type": "module",
  "main": "./dist/index.cjs",      // CommonJS 入口
  "module": "./dist/index.mjs",    // ESM 入口
  "exports": {
    ".": {
      "import": "./dist/index.mjs",
      "require": "./dist/index.cjs"
    }
  }
}
```

## 💡 最佳实践

### ✅ 推荐做法

```javascript
// 1. 明确指定模块类型
// package.json
{
  "type": "module"  // 或 "commonjs"
}

// 2. 使用正确的文件扩展名
// - .js（根据 package.json 的 type）
// - .mjs（强制 ESM）
// - .cjs（强制 CommonJS）

// 3. ESM 中始终包含文件扩展名
import user from './user.js';  // ✅
// import user from './user';  // ❌

// 4. 避免循环依赖
// 如果必须，确保理解加载顺序

// 5. 使用 Top-Level await
const config = await loadConfig();
export default config;
```

### ❌ 避免的做法

```javascript
// ❌ 混用两种导出方式（ESM）
export default { name: 'Alice' };
module.exports = { name: 'Alice' };  // ❌ 会被忽略

// ❌ 重新赋值 exports（CommonJS）
exports = { name: 'Alice' };  // ❌ 不会导出

// ❌ ESM 中省略扩展名
import user from './user';  // ❌ 可能找不到文件

// ❌ 过度使用动态导入
const mod = await import('./mod.js');  // 性能损失
```
