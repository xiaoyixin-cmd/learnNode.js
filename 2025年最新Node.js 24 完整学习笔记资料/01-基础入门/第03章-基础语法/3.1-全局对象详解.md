# 3.1 全局对象详解

> 深入理解 Node.js 的全局对象体系

---

## 🌍 globalThis：统一的全局对象（ES2020+）

### 为什么需要 globalThis？

```javascript
// ===== 跨环境的痛点 =====

// 浏览器环境
console.log(window);  // ✅ 全局对象
console.log(global);  // ❌ undefined

// Node.js 环境
console.log(global);  // ✅ 全局对象
console.log(window);  // ❌ ReferenceError: window is not defined

// Web Worker 环境
console.log(self);    // ✅ 全局对象

// ===== globalThis：统一方案（2020+） =====

console.log(globalThis);  // ✅ 在所有环境都指向全局对象

// 浏览器：globalThis === window
// Node.js：globalThis === global
// Web Worker：globalThis === self
```

### globalThis 使用场景

```javascript
// ===== 场景1：跨平台库开发 =====

// 兼容多环境的全局变量设置
globalThis.myGlobalConfig = {
  apiUrl: 'https://api.example.com',
  timeout: 5000
};

// 任何环境都能访问
console.log(globalThis.myGlobalConfig);

// ===== 场景2：Polyfill 检测 =====

if (!globalThis.fetch) {
  // 浏览器旧版本或 Node.js 17 以下
  globalThis.fetch = require('node-fetch');
}

// ===== 场景3：全局单例 =====

globalThis.db = globalThis.db || createDatabaseConnection();
```

## 🔧 global：Node.js 全局对象

### global 常用属性

```javascript
// ===== process：进程对象（最重要） =====

console.log(process.version);        // Node.js 版本：v22.0.0
console.log(process.platform);       // 操作系统：linux/darwin/win32
console.log(process.arch);           // CPU 架构：x64/arm64
console.log(process.pid);            // 进程 ID
console.log(process.cwd());          // 当前工作目录
console.log(process.env);            // 环境变量对象

// ===== Buffer：二进制数据处理 =====

const buf = Buffer.from('Hello');
console.log(buf);  // <Buffer 48 65 6c 6c 6f>

// ===== 定时器函数（来自浏览器） =====

setTimeout(() => console.log('延迟执行'), 1000);
setInterval(() => console.log('重复执行'), 2000);
setImmediate(() => console.log('立即执行'));

// ===== console：日志输出 =====

console.log('普通日志');
console.error('错误日志');
console.warn('警告日志');
console.info('信息日志');
```

### ⚠️ 不要污染 global

```javascript
// ❌ 不推荐：污染全局作用域
global.myVariable = 'bad practice';

// 任何地方都能访问（容易冲突）
console.log(myVariable);  // 'bad practice'

// ✅ 推荐：使用模块化
// config.js
export const myVariable = 'good practice';

// index.js
import { myVariable } from './config.js';
console.log(myVariable);
```

## 🖥️ process：进程对象（重点）

### 1. 基本信息

```javascript
// ===== 版本信息 =====

console.log(process.version);         // v22.0.0
console.log(process.versions);        // { node, v8, uv, zlib, openssl... }
console.log(process.versions.node);   // 22.0.0
console.log(process.versions.v8);     // 11.8.172.15

// ===== 系统信息 =====

console.log(process.platform);        // linux, darwin, win32
console.log(process.arch);            // x64, arm64, arm
console.log(process.pid);             // 进程 ID：12345
console.log(process.ppid);            // 父进程 ID：12344

// ===== 运行时信息 =====

console.log(process.uptime());        // 运行时长（秒）：123.456
console.log(process.memoryUsage());   // 内存使用情况
// 输出：
// {
//   rss: 36700160,          // 常驻内存集（总内存）
//   heapTotal: 6537216,     // V8 堆总大小
//   heapUsed: 4394032,      // V8 堆已使用
//   external: 1089471,      // C++ 对象绑定的内存
//   arrayBuffers: 17606     // ArrayBuffer 和 SharedArrayBuffer
// }

console.log(process.cpuUsage());      // CPU 使用情况
// 输出：{ user: 38000, system: 6000 }（微秒）
```

### 2. 路径相关

```javascript
// ===== 工作目录 =====

console.log(process.cwd());
// 输出：/Users/username/project

// 改变工作目录
process.chdir('/tmp');
console.log(process.cwd());
// 输出：/tmp

// ===== 执行路径 =====

console.log(process.execPath);
// 输出：/usr/local/bin/node

console.log(process.argv0);
// 输出：node
```

### 3. 命令行参数

```javascript
// ===== process.argv =====

// 运行：node app.js --port 3000 --host localhost

console.log(process.argv);
// 输出：
// [
//   '/usr/local/bin/node',     // Node.js 可执行文件路径
//   '/path/to/app.js',         // 脚本文件路径
//   '--port',
//   '3000',
//   '--host',
//   'localhost'
// ]

// ===== 解析参数 =====

const args = process.argv.slice(2);  // 去掉前两个
console.log(args);
// 输出：['--port', '3000', '--host', 'localhost']

// 手动解析
const parseArgs = (argv) => {
  const result = {};
  for (let i = 0; i < argv.length; i += 2) {
    const key = argv[i].replace(/^--/, '');
    const value = argv[i + 1];
    result[key] = value;
  }
  return result;
};

console.log(parseArgs(args));
// 输出：{ port: '3000', host: 'localhost' }

// ===== 使用 minimist 库（推荐） =====

// npm install minimist
import minimist from 'minimist';

const argv = minimist(process.argv.slice(2));
console.log(argv);
// 输出：{ _: [], port: 3000, host: 'localhost' }

console.log(argv.port);   // 3000
console.log(argv.host);   // 'localhost'
```

### 4. 环境变量

```javascript
// ===== process.env =====

console.log(process.env.NODE_ENV);    // 'development' / 'production'
console.log(process.env.PORT);        // '3000'
console.log(process.env.HOME);        // 用户主目录
console.log(process.env.PATH);        // 系统 PATH

// ===== 设置环境变量 =====

// 方式1：命令行设置
// NODE_ENV=production PORT=8080 node app.js

// 方式2：代码中设置（不推荐）
process.env.MY_VAR = 'value';

// 方式3：使用 .env 文件（推荐）
// .env
// NODE_ENV=development
// PORT=3000

// 加载 .env
import 'dotenv/config';
console.log(process.env.PORT);  // 3000

// ===== 类型转换 =====

// ⚠️ 环境变量都是字符串
console.log(typeof process.env.PORT);  // 'string'

// 需要手动转换
const port = parseInt(process.env.PORT || '3000', 10);
console.log(typeof port);  // 'number'

const debug = process.env.DEBUG === 'true';
console.log(typeof debug);  // 'boolean'
```

### 5. 标准输入输出

```javascript
// ===== process.stdin：标准输入 =====

process.stdin.setEncoding('utf8');

console.log('请输入你的名字：');

process.stdin.on('data', (data) => {
  const name = data.trim();
  console.log(`你好，${name}！`);
  process.exit(0);
});

// ===== process.stdout：标准输出 =====

process.stdout.write('Hello ');
process.stdout.write('World\n');
// 等同于：console.log('Hello World');

// 进度条示例
let progress = 0;
const timer = setInterval(() => {
  progress += 10;
  process.stdout.write(`\r进度：${progress}%`);
  
  if (progress >= 100) {
    clearInterval(timer);
    process.stdout.write('\n完成！\n');
  }
}, 100);

// ===== process.stderr：标准错误 =====

process.stderr.write('这是错误信息\n');
// 等同于：console.error('这是错误信息');
```

### 6. 进程退出

```javascript
// ===== process.exit() =====

// 正常退出
process.exit(0);

// 异常退出
process.exit(1);

// ⚠️ exit() 会立即终止，不执行后续代码
console.log('这行代码会执行');
process.exit(0);
console.log('这行代码不会执行');

// ===== 'exit' 事件 =====

process.on('exit', (code) => {
  console.log(`进程即将退出，退出码：${code}`);
  // 这里只能执行同步代码
  // ❌ 不能使用异步操作（setTimeout、Promise 等）
});

// ===== 'beforeExit' 事件 =====

process.on('beforeExit', (code) => {
  console.log('进程即将退出');
  // ✅ 可以执行异步操作
  // 可以在这里阻止进程退出
});

// ===== 优雅退出示例 =====

async function gracefulShutdown(signal) {
  console.log(`收到 ${signal} 信号，准备退出...`);
  
  // 停止接收新请求
  server.close();
  
  // 等待现有请求处理完毕
  await waitForPendingRequests();
  
  // 关闭数据库连接
  await db.close();
  
  console.log('清理完成，退出进程');
  process.exit(0);
}

process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
process.on('SIGINT', () => gracefulShutdown('SIGINT'));
```

### 7. 信号处理

```javascript
// ===== 常见信号 =====

// SIGINT：Ctrl+C 触发
process.on('SIGINT', () => {
  console.log('收到 SIGINT 信号');
  process.exit(0);
});

// SIGTERM：终止信号（Docker/Kubernetes 常用）
process.on('SIGTERM', () => {
  console.log('收到 SIGTERM 信号');
  process.exit(0);
});

// SIGHUP：终端关闭
process.on('SIGHUP', () => {
  console.log('收到 SIGHUP 信号');
  // 可用于重新加载配置
});

// ===== 发送信号 =====

// 向其他进程发送信号
process.kill(otherProcessPid, 'SIGTERM');

// 向自己发送信号
process.kill(process.pid, 'SIGTERM');
```

### 8. 未捕获异常处理

```javascript
// ===== uncaughtException：未捕获的异常 =====

process.on('uncaughtException', (error, origin) => {
  console.error('未捕获的异常：', error);
  console.error('异常来源：', origin);
  
  // 记录日志
  fs.writeFileSync('./error.log', error.stack);
  
  // ⚠️ 进程处于不稳定状态，应该退出
  process.exit(1);
});

// ===== unhandledRejection：未处理的 Promise 拒绝 =====

process.on('unhandledRejection', (reason, promise) => {
  console.error('未处理的 Promise 拒绝：', reason);
  console.error('Promise：', promise);
  
  // Node.js 15+ 会默认退出进程
  // 早期版本需要手动处理
  process.exit(1);
});

// ===== 示例 =====

// 这会触发 uncaughtException
throw new Error('未捕获的异常');

// 这会触发 unhandledRejection
Promise.reject(new Error('未处理的 Promise'));

// ===== 最佳实践：使用 try/catch =====

try {
  // 可能抛出异常的代码
  riskyOperation();
} catch (error) {
  console.error('捕获到异常：', error);
}

// 异步代码使用 async/await
async function main() {
  try {
    await riskyAsyncOperation();
  } catch (error) {
    console.error('捕获到异步异常：', error);
  }
}
```

## 📝 console：日志输出

### 基础用法

```javascript
// ===== 基本输出 =====

console.log('普通日志');
console.info('信息日志');      // 等同于 log
console.warn('警告日志');      // 黄色输出
console.error('错误日志');     // 红色输出

// ===== 多参数输出 =====

console.log('姓名：', 'Alice', '年龄：', 25);
// 输出：姓名： Alice 年龄： 25

// ===== 格式化输出 =====

console.log('字符串：%s', 'Hello');
console.log('整数：%d', 123);
console.log('浮点数：%f', 3.14);
console.log('对象：%o', { name: 'Alice' });
console.log('JSON：%j', { name: 'Alice' });

// 输出：
// 字符串：Hello
// 整数：123
// 浮点数：3.14
// 对象：{ name: 'Alice' }
// JSON：{"name":"Alice"}
```

### 高级用法

```javascript
// ===== console.dir()：详细展示对象 =====

const obj = {
  name: 'Alice',
  age: 25,
  address: {
    city: 'Beijing'
  }
};

console.log(obj);  // 简单展示
console.dir(obj, { depth: null, colors: true });  // 详细展示

// ===== console.table()：表格展示 =====

const users = [
  { id: 1, name: 'Alice', age: 25 },
  { id: 2, name: 'Bob', age: 30 },
  { id: 3, name: 'Charlie', age: 35 }
];

console.table(users);
// 输出：
// ┌─────────┬────┬──────────┬─────┐
// │ (index) │ id │   name   │ age │
// ├─────────┼────┼──────────┼─────┤
// │    0    │ 1  │ 'Alice'  │ 25  │
// │    1    │ 2  │  'Bob'   │ 30  │
// │    2    │ 3  │'Charlie' │ 35  │
// └─────────┴────┴──────────┴─────┘

// 只显示特定列
console.table(users, ['name', 'age']);

// ===== console.count()：计数器 =====

function processRequest(type) {
  console.count(type);
  // 处理请求...
}

processRequest('GET');   // GET: 1
processRequest('POST');  // POST: 1
processRequest('GET');   // GET: 2
processRequest('GET');   // GET: 3

console.countReset('GET');
processRequest('GET');   // GET: 1

// ===== console.time()：计时器 =====

console.time('数据库查询');
// 执行数据库查询...
await db.query('SELECT * FROM users');
console.timeEnd('数据库查询');
// 输出：数据库查询: 123.456ms

// ===== console.trace()：堆栈跟踪 =====

function a() {
  b();
}

function b() {
  c();
}

function c() {
  console.trace('追踪调用栈');
}

a();
// 输出：
// Trace: 追踪调用栈
//     at c (/path/to/file.js:10:11)
//     at b (/path/to/file.js:6:3)
//     at a (/path/to/file.js:2:3)

// ===== console.assert()：断言 =====

const x = 5;
console.assert(x === 5, 'x 应该等于 5');     // 通过，不输出
console.assert(x === 10, 'x 应该等于 10');   // 失败，输出错误

// ===== console.group()：分组 =====

console.group('用户信息');
console.log('姓名：Alice');
console.log('年龄：25');
console.group('地址');
console.log('城市：Beijing');
console.log('街道：Chaoyang');
console.groupEnd();
console.groupEnd();

// 输出：
// 用户信息
//   姓名：Alice
//   年龄：25
//   地址
//     城市：Beijing
//     街道：Chaoyang
```

## ⏱️ 定时器函数

```javascript
// ===== setTimeout：延迟执行 =====

const timerId = setTimeout(() => {
  console.log('1 秒后执行');
}, 1000);

// 取消定时器
clearTimeout(timerId);

// ===== setInterval：重复执行 =====

let count = 0;
const intervalId = setInterval(() => {
  count++;
  console.log(`执行第 ${count} 次`);
  
  if (count >= 5) {
    clearInterval(intervalId);
  }
}, 1000);

// ===== setImmediate：立即执行（Node.js 特有） =====

console.log('1');

setImmediate(() => {
  console.log('3');
});

console.log('2');

// 输出：1 → 2 → 3

// ===== 执行顺序对比 =====

console.log('1');

setTimeout(() => console.log('4'), 0);
setImmediate(() => console.log('3'));
process.nextTick(() => console.log('2'));

console.log('5');

// 输出：1 → 5 → 2 → 3 → 4
// 解释：
// - 同步代码先执行：1, 5
// - process.nextTick 最先：2
// - setImmediate：3
// - setTimeout：4
```

## 💡 最佳实践

### ✅ 推荐做法

```javascript
// 1. 使用 globalThis 编写跨平台代码
const isBrowser = typeof globalThis.window !== 'undefined';

// 2. 从环境变量读取配置
const config = {
  port: parseInt(process.env.PORT || '3000', 10),
  env: process.env.NODE_ENV || 'development'

// 3. 优雅退出
process.on('SIGTERM', async () => {

import winston from 'winston';
```

```javascript
// ❌ 污染 global
```
