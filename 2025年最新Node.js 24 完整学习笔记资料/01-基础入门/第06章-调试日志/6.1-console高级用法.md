## 🎨 格式化输出

### 占位符

```javascript
// ===== 字符串占位符 %s =====
console.log('用户名：%s', 'Alice');
// 输出：用户名：Alice

console.log('%s 下单了 %s', 'Alice', 'iPhone');
// 输出：Alice 下单了 iPhone

// ===== 数字占位符 %d 或 %i =====
console.log('数量：%d', 42);
// 输出：数量：42

console.log('价格：%d，数量：%d', 99.9, 3);
// 输出：价格：99.9，数量：3

// ===== 浮点数占位符 %f =====
console.log('圆周率：%f', Math.PI);
// 输出：圆周率：3.141592653589793

// ===== 对象占位符 %o（小写）=====
const user = { name: 'Alice', age: 25 };
console.log('用户：%o', user);
// 输出：用户： { name: 'Alice', age: 25 }

// ===== 对象占位符 %O（大写，带类型信息）=====
console.log('用户：%O', user);
// 输出更详细的对象信息

// ===== CSS 样式占位符 %c（仅浏览器）=====
// Node.js 不支持，但了解一下
console.log('%c 这是红色文字', 'color: red; font-size: 20px;');

// ===== 实战：格式化日志 =====

function log(level, message, ...args) {
  const timestamp = new Date().toISOString();
  console.log('[%s] %s: %s', timestamp, level, message, ...args);
}

log('INFO', '用户登录', { userId: 123 });
// [2025-10-04T10:30:00.000Z] INFO: 用户登录 { userId: 123 }

log('ERROR', '数据库连接失败', { error: 'ECONNREFUSED' });
// [2025-10-04T10:30:01.000Z] ERROR: 数据库连接失败 { error: 'ECONNREFUSED' }
```

### console.time() / console.timeEnd()

```javascript
// ===== 基础用法：测量代码执行时间 =====

console.time('操作1');
// 执行一些操作
for (let i = 0; i < 1000000; i++) {
  // 计算
}
console.timeEnd('操作1');
// 输出：操作1: 5.123ms

// ===== 多个计时器同时运行 =====

console.time('步骤1');
await step1();
console.timeEnd('步骤1');
// 步骤1: 100ms

console.time('步骤2');
await step2();
console.timeEnd('步骤2');
// 步骤2: 200ms

console.timeEnd('总耗时');
// 总耗时: 300ms

// ===== console.timeLog() - 中间计时 =====

console.time('长操作');

await task1();
console.timeLog('长操作', '任务1完成');
// 长操作: 100ms 任务1完成

await task2();
console.timeLog('长操作', '任务2完成');
// 长操作: 250ms 任务2完成

await task3();
console.timeEnd('长操作');
// 长操作: 400ms

// ===== 实战：数据库查询性能对比 =====

async function compareQueryMethods() {
  console.log('开始性能对比\n');
  
  // 方法1：逐个查询
  console.time('逐个查询');
  for (const id of [1, 2, 3, 4, 5]) {
    await db.query('SELECT * FROM users WHERE id = ?', [id]);
  }
  console.timeEnd('逐个查询');
  // 逐个查询: 523ms
  
  // 方法2：批量查询
  console.time('批量查询');
  await db.query('SELECT * FROM users WHERE id IN (?,?,?,?,?)', [1,2,3,4,5]);
  console.timeEnd('批量查询');
  // 批量查询: 45ms
  
  console.log('\n✅ 批量查询快了 11 倍！');
}

// ===== 自动性能测量装饰器 =====

function measure(label) {
  return function(target, propertyKey, descriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function(...args) {
      console.time(label || propertyKey);
      try {
        return await originalMethod.apply(this, args);
      } finally {
        console.timeEnd(label || propertyKey);
      }
    };
    
    return descriptor;
  };
}

class UserService {
  @measure('获取用户列表')
  async getUsers() {
    return await db.query('SELECT * FROM users');
  }
  
  @measure()  // 使用方法名
  async createUser(data) {
    return await db.insert('users', data);
  }
}

// 调用时自动计时
await userService.getUsers();
// 获取用户列表: 45ms
```

### 案例2：性能分析报告

```javascript
// ===== 自动性能分析 =====

class PerformanceMonitor {
  constructor() {
    this.metrics = {};
  }
  
  start(name) {
    this.metrics[name] = {
      start: Date.now(),
      count: (this.metrics[name]?.count || 0) + 1
    };
  }
  
  end(name) {
    if (!this.metrics[name]) return;
    
    const duration = Date.now() - this.metrics[name].start;
    
    if (!this.metrics[name].times) {
      this.metrics[name].times = [];
    }
    
    this.metrics[name].times.push(duration);
  }
  
  report() {
    console.log('\n');
    console.log('📊 性能分析报告');
    console.log('='.repeat(60));
    
    const report = Object.entries(this.metrics).map(([name, data]) => {
      const times = data.times || [];
      const avg = times.reduce((a, b) => a + b, 0) / times.length;
      const min = Math.min(...times);
      const max = Math.max(...times);
      
      return {
        操作: name,
        调用次数: data.count,
        平均耗时: `${avg.toFixed(2)}ms`,
        最小耗时: `${min}ms`,
        最大耗时: `${max}ms`
      };
    });
    
    console.table(report);
    console.log('='.repeat(60));
    console.log('\n');
  }
}

// 使用
```

```javascript
console.log('详情2');
console.groupEnd();

// ✅ 5. 使用 trace 追踪调用栈
console.trace('调用来源');

// ✅ 6. 使用 count 统计次数
console.count('事件');

// ⚠️ 7. 生产环境移除调试日志
if (process.env.NODE_ENV !== 'production') {
  console.log('调试信息');
}
```
