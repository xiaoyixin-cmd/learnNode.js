# 6.5 结构化日志最佳实践

> 让日志从"能看"变成"好用" —— 结构化、可搜索、可分析

---

## 🎯 什么是结构化日志？

### 概念对比

```javascript
// ===== 非结构化日志(字符串拼接) =====

console.log('用户 Alice 在 2025-10-04 10:30:00 从 192.168.1.1 登录');
console.log('订单 12345 支付失败,金额 99.9,原因 余额不足');

// ❌ 问题:
// 难以搜索(如何找出所有支付失败的订单?)
// 难以分析(如何统计登录用户数?)
// 难以解析(正则表达式容易出错)

// ===== 结构化日志(JSON格式) =====

logger.info('用户登录', {
  event: 'user_login',
  userId: 'Alice',
  timestamp: '2025-10-04T10:30:00Z',
  ip: '192.168.1.1'
});

logger.error('订单支付失败', {
  event: 'payment_failed',
  orderId: 12345,
  amount: 99.9,
  reason: '余额不足'
});

// ✅ 优势:
// 易搜索: event: 'payment_failed'
// 易分析: 统计 amount 总和
// 易解析: 直接 JSON.parse()
```

### 结构化日志的核心原则

```
1. 固定字段 + 动态数据
   - 固定: level, timestamp, message
   - 动态: userId, orderId, amount

2. 使用语义化的键名
   - ✅ userId, requestId, duration
   - ❌ id, x, tmp

3. 一致的数据类型
   - userId: 始终是字符串
   - amount: 始终是数字
   - timestamp: 始终是 ISO 8601

4. 避免嵌套过深
   - ✅ user_id, user_name
   - ❌ user.profile.personal.name
```

```javascript
// ===== 请求日志中间件 =====

const logger = require('./logger');
const { v4: uuidv4 } = require('uuid');

function requestLogger(req, res, next) {
  // 生成请求ID
  req.id = uuidv4();
  const start = Date.now();
  
  // 创建子日志器
  req.logger = logger.child({
    requestId: req.id,
    ip: req.ip,
    userAgent: req.get('user-agent')
  });
  
  // 记录请求开始
  req.logger.info('请求开始', {
    event: 'request_start',
    method: req.method,
    url: req.url,
    query: req.query,
    headers: {
      host: req.get('host'),
      referer: req.get('referer'),
      contentType: req.get('content-type')
    }
  });
  
  // 劫持 res.end
  const originalEnd = res.end;
  res.end = function(...args) {
    const duration = Date.now() - start;
    
    // 记录请求完成
    req.logger.info('请求完成', {
      event: 'request_end',
      method: req.method,
      url: req.url,
      statusCode: res.statusCode,
      duration,
      contentLength: res.get('content-length')
    });
    
    // 慢请求警告
    if (duration > 1000) {
      req.logger.warn('慢请求', {
        event: 'slow_request',
        method: req.method,
        url: req.url,
        duration
      });
    }
    
    originalEnd.apply(res, args);
  };
  
  next();
}

module.exports = requestLogger;

// ===== 使用 =====

app.use(requestLogger);

// 日志输出:
```

```javascript
// ===== 数据库日志包装器 =====

class DatabaseLogger {
  constructor(db, logger) {
    this.db = db;
    this.logger = logger;
  }
  
  async query(sql, params = []) {
    const queryId = uuidv4();
    const start = Date.now();
    
    // 记录查询开始
    this.logger.debug('SQL 查询开始', {
      event: 'query_start',
      queryId,
      sql: sql.substring(0, 100),  // 只记录前100字符
      params
    });
    
    try {
      const result = await this.db.query(sql, params);
      const duration = Date.now() - start;
      
      // 记录查询成功
      this.logger.info('SQL 查询成功', {
        event: 'query_success',
        queryId,
        duration,
        rowCount: result.length
      });
      
      // 慢查询警告
      if (duration > 500) {
        this.logger.warn('慢查询', {
          event: 'slow_query',
          queryId,
          duration,
          sql,
          params
        });
      }
      
      return result;
      
    } catch (error) {
      const duration = Date.now() - start;
      
      // 记录查询失败
      this.logger.error('SQL 查询失败', {
        event: 'query_error',
        queryId,
        duration,
        sql,
        params,
        errorCode: error.code,
        errorMessage: error.message
      });
      
      throw error;
    }
  }
}

// ===== 使用 =====

const dbLogger = new DatabaseLogger(db, logger);

const users = await dbLogger.query(
  'SELECT * FROM users WHERE age > ?',
  [18]
);
```

```javascript
      timestamp: Date.now(),
      ...data
    });

    });
```

```bash
# ===== 高级查询 =====

cat logs/app.log | jq -s 'map(select(.level == "error")) | length'

# 计算平均响应时间
```

```javascript
// ===== 完整的日志聚合系统 =====

const winston = require('winston');
const { format } = winston;

// 自定义格式化器
const customFormat = format.printf(({ level, message, timestamp, ...meta }) => {
  // 计算日志大小
  const logSize = JSON.stringify(meta).length;
  
  // 标记大日志(超过1KB)
  if (logSize > 1024) {
    meta._largeLog = true;
    meta._size = `${(logSize / 1024).toFixed(2)}KB`;
  }
  
  // 标准化输出
  return JSON.stringify({
    timestamp,
    level,
    message,
    ...meta
  });
});

// 创建聚合日志器
const aggregateLogger = winston.createLogger({
  format: format.combine(
    format.timestamp(),
    format.errors({ stack: true }),
    customFormat
  ),
  transports: [
    // 按级别分离日志文件
    new winston.transports.File({
      filename: 'logs/error.log',
      level: 'error',
      maxsize: 5242880, // 5MB
      maxFiles: 5
    }),
    new winston.transports.File({
      filename: 'logs/combined.log',
      maxsize: 10485760, // 10MB
      maxFiles: 10
    }),
    // 控制台输出(开发环境)
    new winston.transports.Console({
      format: format.combine(
        format.colorize(),
        format.simple()
      )
    })
  ]
});

// ===== 实时日志统计 =====

class LogStatistics {
  constructor() {
    this.stats = {
      total: 0,
      byLevel: {},
      byEvent: {},
      errors: []
    };
  }
  
  record(logEntry) {
    this.stats.total++;
    
    // 按级别统计
    this.stats.byLevel[logEntry.level] = 
      (this.stats.byLevel[logEntry.level] || 0) + 1;
    
    // 按事件统计
    if (logEntry.event) {
      this.stats.byEvent[logEntry.event] = 
        (this.stats.byEvent[logEntry.event] || 0) + 1;
    }
    
    // 收集错误(最近100条)
    if (logEntry.level === 'error') {
      this.stats.errors.unshift(logEntry);
      if (this.stats.errors.length > 100) {
        this.stats.errors.pop();
      }
    }
  }
  
  getReport() {
    return {
      总日志数: this.stats.total,
      按级别统计: this.stats.byLevel,
      按事件统计: this.stats.byEvent,
      最近错误: this.stats.errors.slice(0, 10)
    };
  }
}

const logStats = new LogStatistics();

// 包装日志方法以收集统计
const originalInfo = aggregateLogger.info.bind(aggregateLogger);
const originalError = aggregateLogger.error.bind(aggregateLogger);
```

```javascript
// ===== 分布式追踪系统 =====

const express = require('express');
const { v4: uuidv4 } = require('uuid');
const axios = require('axios');

// 追踪中间件
function distributedTracingMiddleware(req, res, next) {
  // 从请求头获取或生成 traceId
  const traceId = req.get('X-Trace-Id') || `trace-${uuidv4()}`;
  const spanId = `span-${uuidv4()}`;
  const parentSpanId = req.get('X-Parent-Span-Id') || null;
  
  // 存储追踪信息
  req.trace = { traceId, spanId, parentSpanId };
  
  // 创建追踪日志器
  req.logger = logger.child({
    traceId,
    spanId,
    parentSpanId,
    service: 'api-gateway'
  });
  
  // 记录请求开始
  req.logger.info('Span 开始', {
    event: 'span_start',
    spanType: 'http',
    method: req.method,
    url: req.url
  });
  
  // 劫持响应
  const originalSend = res.send;
  res.send = function(data) {
    req.logger.info('Span 结束', {
      event: 'span_end',
      statusCode: res.statusCode
    });
    originalSend.call(this, data);
  };
  
  next();
}

// 创建应用
const app = express();
app.use(distributedTracingMiddleware);

// API 路由
```

### 4. 日志采样(减少日志量)

```javascript
// ===== 高频日志采样 =====

class SamplingLogger {
  constructor(logger, sampleRate = 0.1) {
    this.logger = logger;
    this.sampleRate = sampleRate;  // 10% 采样率
  }
  
  info(message, meta) {
    // 高优先级日志总是记录
    if (meta.priority === 'high') {
      this.logger.info(message, meta);
      return;
    }
    
    // 低优先级日志采样
    if (Math.random() < this.sampleRate) {
      this.logger.info(message, { ...meta, _sampled: true });
    }
  }
}

const samplingLogger = new SamplingLogger(logger, 0.1);

// 只有 10% 的日志被记录
for (let i = 0; i < 10000; i++) {
  samplingLogger.info('心跳检查', { event: 'heartbeat' });
}

// 高优先级日志总是记录
samplingLogger.info('支付成功', { 
  event: 'payment_success', 
  priority: 'high' 
});
```

### 问题1：为什么需要结构化日志？与普通日志有什么区别？

**回答：**

```
结构化日志的核心优势：

1. 可搜索性
   - 普通日志："用户123在2025-10-04登录失败"
   - 结构化日志：{ event: "login_failed", userId: "123", date: "2025-10-04" }
   - 可以轻松查询：SELECT * WHERE event = "login_failed"

2. 可分析性
   - 统计、聚合、可视化
   - 如：计算平均响应时间、错误率、用户活跃度

3. 机器可读
   - JSON 格式可直接被日志分析工具(ELK)处理
   - 不需要复杂的正则表达式解析

4. 统一标准
   - 多个服务使用相同的日志格式
   - 便于集中管理和分析

实际应用场景：
✅ 微服务架构：追踪请求在多个服务间的流动(traceId)
✅ 性能监控：统计慢查询、慢请求
✅ 业务分析：用户行为分析、转化率计算
✅ 故障排查：快速定位错误原因
```

**回答：**

```
生产环境日志管理最佳实践：
```

### 问题4：如何设计一个高性能的日志系统？

**回答：**

```javascript
// ===== 高性能日志系统设计要点 =====

// 1. 使用 Pino(而非 Winston)
const pino = require('pino');
const logger = pino({
  level: 'info',
  // 异步写入(5-10倍性能提升)
  destination: pino.destination({
    dest: './logs/app.log',
    sync: false,
    minLength: 4096  // 4KB 缓冲区
  })
});

// 2. 避免同步 JSON.stringify
// Pino 使用 fast-json-stringify(预编译 schema)
logger.info({
  event: 'user_login',
  userId: '123',
  timestamp: Date.now()
});

// 3. 日志采样(减少 I/O)
class AdaptiveSampler {
  constructor(targetRate = 1000) {  // 每秒 1000 条
    this.targetRate = targetRate;
    this.count = 0;
    this.lastReset = Date.now();
  }
  
  shouldLog() {
    const now = Date.now();
    const elapsed = now - this.lastReset;
    
    // 每秒重置计数
    if (elapsed > 1000) {
      this.count = 0;
      this.lastReset = now;
    }
    
    // 超过目标速率则丢弃
    if (this.count >= this.targetRate) {
      return false;
    }
    
    this.count++;
    return true;
  }
}

const sampler = new AdaptiveSampler(1000);

function smartLog(message, meta) {
  // 错误和警告总是记录
  if (meta.level === 'error' || meta.level === 'warn') {
    logger[meta.level](message, meta);
    return;
  }
  
  // 普通日志采样
  if (sampler.shouldLog()) {
    logger.info(message, meta);
  }
}

// 4. 使用子进程写入(进一步优化)
const { spawn } = require('child_process');

// 启动独立的日志写入进程
const logWriter = spawn('node', ['log-writer.js'], {
  stdio: ['pipe', 'inherit', 'inherit']
});

// 主进程只负责发送日志到子进程
function writeLog(logEntry) {
  logWriter.stdin.write(JSON.stringify(logEntry) + '\n');
}

// 5. 批量写入(减少磁盘 I/O)
class BatchLogger {
  constructor(logger, batchSize = 100, flushInterval = 1000) {
    this.logger = logger;
    this.batchSize = batchSize;
    this.buffer = [];
    
    // 定期刷新缓冲区
    setInterval(() => this.flush(), flushInterval);
  }
  
  log(level, message, meta) {
    this.buffer.push({ level, message, meta, timestamp: Date.now() });
    
    if (this.buffer.length >= this.batchSize) {
      this.flush();
    }
  }
  
  flush() {
    if (this.buffer.length === 0) return;
    
    // 批量写入
    const batch = this.buffer.splice(0);
    this.logger.info({ batch });
  }
}

const batchLogger = new BatchLogger(logger);

// 性能对比:
// Winston(同步):     5,000 logs/s
// Winston(异步):    15,000 logs/s
// Pino(异步):       50,000 logs/s
// Pino(子进程):     80,000 logs/s
// Pino(批量):      100,000 logs/s
```
