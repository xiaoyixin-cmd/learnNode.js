# 6.4 专业日志库选型

> 从 console.log 到企业级日志系统

---

## 🎯 为什么需要日志库？

### console.log 的局限性

```javascript
// ===== console.log 的问题 =====

console.log('用户登录', user.id);
// ❌ 没有时间戳
// ❌ 没有日志级别
// ❌ 不能保存到文件
// ❌ 不能分级控制(只能全部显示或全不显示)
// ❌ 没有结构化格式(难以解析)

// ===== 专业日志库的优势 =====

logger.info('用户登录', { userId: user.id, timestamp: Date.now() });
// ✅ 自动添加时间戳
// ✅ 有日志级别(error/warn/info/debug)
// ✅ 可以输出到文件
// ✅ 可以按级别过滤
// ✅ 结构化输出(JSON格式)
// ✅ 性能更好(异步写入)
```

## 🏆 Winston：功能最全

### 安装与基础使用

```bash
# 安装
npm install winston
```

```javascript
// ===== 1. 最简单的使用 =====

const winston = require('winston');

const logger = winston.createLogger({
  level: 'info',  // 最低日志级别
  format: winston.format.json(),
  transports: [
    new winston.transports.Console()  // 输出到控制台
  ]
});

logger.info('服务器启动');
// {"level":"info","message":"服务器启动"}

logger.error('数据库连接失败', { error: 'ECONNREFUSED' });
// {"level":"error","message":"数据库连接失败","error":"ECONNREFUSED"}

// ===== 2. 日志级别 =====

// Winston 默认级别(优先级从高到低):
// error(0) > warn(1) > info(2) > http(3) > verbose(4) > debug(5) > silly(6)

logger.error('严重错误');    // 总是显示
logger.warn('警告信息');     // level >= warn 时显示
logger.info('普通信息');     // level >= info 时显示
logger.debug('调试信息');    // level >= debug 时显示

// 💡 设置 level: 'warn',只显示 error 和 warn

// ===== 3. 格式化输出 =====

const logger = winston.createLogger({
  format: winston.format.combine(
    winston.format.timestamp(),  // 添加时间戳
    winston.format.errors({ stack: true }),  // 显示错误堆栈
    winston.format.json()  // JSON 格式
  ),
  transports: [
    new winston.transports.Console()
  ]
});

logger.info('用户登录', { userId: 123, ip: '192.168.1.1' });
// 输出:
// {
//   "level": "info",
//   "message": "用户登录",
//   "userId": 123,
//   "ip": "192.168.1.1",
//   "timestamp": "2025-10-04T10:30:00.000Z"
// }
```

### 高级配置

```javascript
// ===== 完整的生产配置 =====

const winston = require('winston');
const path = require('path');

const logger = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  
  // 格式配置
  format: winston.format.combine(
    winston.format.timestamp({
      format: 'YYYY-MM-DD HH:mm:ss'
    }),
    winston.format.errors({ stack: true }),
    winston.format.splat(),  // 支持 %s %d 占位符
    winston.format.json()
  ),
  
  // 默认元数据
  defaultMeta: {
    service: 'my-api',
    environment: process.env.NODE_ENV
  },
  
  // 输出目标
  transports: [
    // 1. 错误日志 -> error.log
    new winston.transports.File({
      filename: 'logs/error.log',
      level: 'error',
      maxsize: 5242880,  // 5MB
      maxFiles: 5  // 保留5个文件
    }),
    
    // 2. 所有日志 -> combined.log
    new winston.transports.File({
      filename: 'logs/combined.log',
      maxsize: 5242880,
      maxFiles: 5
    }),
    
    // 3. 控制台(开发环境)
    new winston.transports.Console({
      format: winston.format.combine(
        winston.format.colorize(),  // 彩色输出
        winston.format.simple()     // 简单格式
      )
    })
  ],
  
  // 处理未捕获的异常
  exceptionHandlers: [
    new winston.transports.File({ filename: 'logs/exceptions.log' })
  ],
  
  // 处理未处理的 Promise 拒绝
  rejectionHandlers: [
    new winston.transports.File({ filename: 'logs/rejections.log' })
  ]
});

// ===== 使用示例 =====

logger.info('服务器启动', { port: 3000 });
// [2025-10-04 10:30:00] info: 服务器启动 {"port":3000,"service":"my-api","environment":"production"}

logger.error('数据库错误', new Error('Connection failed'));
// [2025-10-04 10:30:01] error: 数据库错误
// Error: Connection failed
//     at Object.<anonymous> (app.js:10:25)
//     ...

// ===== 条件输出 =====

// 开发环境：彩色控制台
// 生产环境：文件 + JSON 格式
if (process.env.NODE_ENV !== 'production') {
  logger.add(new winston.transports.Console({
    format: winston.format.combine(
      winston.format.colorize(),
      winston.format.simple()
    )
  }));
}
```

### 自定义格式

```javascript
// ===== 自定义格式 =====

const myFormat = winston.format.printf(({ level, message, timestamp, ...meta }) => {
  let log = `${timestamp} [${level}] ${message}`;
  
  // 如果有额外数据,追加
  if (Object.keys(meta).length > 0) {
    log += ` ${JSON.stringify(meta)}`;
  }
  
  return log;
});

const logger = winston.createLogger({
  format: winston.format.combine(
    winston.format.timestamp(),
    myFormat
  ),
  transports: [
    new winston.transports.Console()
  ]
});

logger.info('用户登录', { userId: 123 });
// 2025-10-04T10:30:00.000Z [info] 用户登录 {"userId":123}
```

### 生产配置

```javascript

const pino = require('pino');
const fs = require('fs');

// 创建日志文件流
const logStream = fs.createWriteStream('logs/app.log', { flags: 'a' });

const logger = pino(
  {
    level: process.env.LOG_LEVEL || 'info',
    
    // 自定义序列化器

      req: (req) => ({
        method: req.method,
        url: req.url,

      env: process.env.NODE_ENV
    }
  },
  logStream  // 输出到文件
);
```

## 🎯 实战案例

### 案例1：统一日志系统

```javascript
// ===== logger.js(日志工具模块) =====

const winston = require('winston');
const path = require('path');

class Logger {
  constructor() {
    this.logger = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: winston.format.combine(
        winston.format.timestamp({
          format: 'YYYY-MM-DD HH:mm:ss'
        }),
        winston.format.errors({ stack: true }),
        winston.format.json()
      ),
      transports: this.createTransports()
    });
  }
  
  createTransports() {
    const transports = [];
    
    // 文件日志
    if (process.env.NODE_ENV === 'production') {
      transports.push(
        new winston.transports.File({
          filename: 'logs/error.log',
          level: 'error',
          maxsize: 5242880,
          maxFiles: 5
        }),
        new winston.transports.File({
          filename: 'logs/combined.log',
          maxsize: 5242880,
          maxFiles: 5
        })
      );
    }
    
    // 控制台日志(开发环境)
    if (process.env.NODE_ENV !== 'production') {
      transports.push(
        new winston.transports.Console({
          format: winston.format.combine(
            winston.format.colorize(),
            winston.format.simple()
          )
        })
      );
    }
    
    return transports;
  }
  
  // 封装常用方法
  info(message, meta = {}) {
    this.logger.info(message, meta);
  }
  
  warn(message, meta = {}) {
    this.logger.warn(message, meta);
  }
  
  error(message, error = null) {
    if (error instanceof Error) {
      this.logger.error(message, {
        error: {
          message: error.message,
          stack: error.stack
        }
      });
    } else {
      this.logger.error(message, error);
    }
  }
  
  debug(message, meta = {}) {
    this.logger.debug(message, meta);
  }
}

module.exports = new Logger();
```

### 案例2：请求日志中间件

```javascript
// ===== 自定义请求日志中间件 =====

const logger = require('./logger');

function requestLogger(req, res, next) {
  const start = Date.now();
  
  // 请求开始
  logger.info('请求开始', {
    method: req.method,
    url: req.url,
    ip: req.ip,
    userAgent: req.get('user-agent')
  });
  
  // 劫持 res.end,记录响应
  const originalEnd = res.end;
  res.end = function(...args) {
    const duration = Date.now() - start;
    
    logger.info('请求完成', {
      method: req.method,
      url: req.url,
      statusCode: res.statusCode,
      duration: `${duration}ms`
    });
    
    originalEnd.apply(res, args);
  };
  
  next();
}

// 使用
app.use(requestLogger);

// 输出示例:
// [2025-10-04 10:30:00] INFO: 请求开始 {"method":"GET","url":"/api/users","ip":"127.0.0.1"}
// [2025-10-04 10:30:00] INFO: 请求完成 {"method":"GET","url":"/api/users","statusCode":200,"duration":"45ms"}
```

### 2. 如何在生产环境优化日志性能？

```javascript
// 答案：

// ✅ 1. 使用异步写入
const pino = require('pino');
const logger = pino(pino.destination({
  dest: './app.log',
  sync: false  // 异步写入
}));

// ✅ 2. 避免过度日志
// ❌ 错误：在循环中记录
for (let i = 0; i < 10000; i++) {
  logger.debug(`处理项目 ${i}`);  // 会产生10000条日志！
}

// ✅ 正确：批量记录
logger.debug(`批量处理 ${items.length} 个项目`);

// ✅ 3. 使用采样
let logCounter = 0;
app.use((req, res, next) => {
  // 每10个请求记录一次
  if (++logCounter % 10 === 0) {
    logger.info('请求采样', { url: req.url });
  }
  next();
});

// ✅ 4. 禁用不必要的日志
if (process.env.NODE_ENV === 'production') {

// 不要直接写本地文件，发送到日志服务器
```

### 3. 如何追踪分布式系统中的请求？

```javascript
// 答案：使用 Trace ID

// ===== 生成全局唯一的 Trace ID =====
const { v4: uuidv4 } = require('uuid');

// 中间件：为每个请求生成 Trace ID
function traceMiddleware(req, res, next) {
  // 从请求头获取或生成新的
  req.traceId = req.get('X-Trace-Id') || `trace-${uuidv4()}`;
  
  // 设置响应头
  res.set('X-Trace-Id', req.traceId);
  
  // 创建带 Trace ID 的日志器
  req.logger = logger.child({ traceId: req.traceId });
  
  next();
}

app.use(traceMiddleware);

// ===== 在所有日志中包含 Trace ID =====
app.get('/api/users/:id', async (req, res) => {
  req.logger.info('获取用户', { userId: req.params.id });
  
  // 调用其他服务时传递 Trace ID
  const orders = await fetch('http://order-service/api/orders', {
    headers: {
      'X-Trace-Id': req.traceId
    }
  });
  
  req.logger.info('用户信息获取成功');
  res.json({ user, orders });
});

// ===== 查询日志 =====
// 使用 Trace ID 查询整个请求链路的所有日志
// grep "trace-abc-123" logs/*.log
```
