# 6.3 VS Code 调试配置

> 最强大的 Node.js 调试体验,图形化界面让调试变简单

---

## 🎯 为什么用 VS Code 调试?

### VS Code vs 命令行调试

```
VS Code 调试 = 图形化 + 智能提示 + 一键操作

命令行调试：
❌ 记命令麻烦
❌ 看代码不方便
❌ 查变量要输入命令

VS Code 调试：
✅ 点击行号设置断点
✅ 鼠标悬停查看变量
✅ 可视化查看调用栈
✅ 内置终端,一体化体验
✅ 支持多种配置(Node/TypeScript/测试等)
```

### 方法1：自动生成配置(推荐)

```javascript
// ===== 步骤1：准备测试文件 =====

// app.js
function greet(name) {
  const message = `Hello, ${name}!`;
  console.log(message);  // 在这行设置断点
  return message;
}

const result = greet('Alice');
console.log('结果：', result);

// ===== 步骤2：打开 VS Code =====

// 1. 打开项目文件夹
// 2. 打开 app.js
// 3. 点击左侧调试图标(或按 F5)
// 4. 选择 "Node.js"
// 5. VS Code 自动生成 .vscode/launch.json

// ===== 步骤3：设置断点 =====

// 点击行号左侧,出现红点即为断点

// ===== 步骤4：开始调试 =====

// 按 F5 或点击 "开始调试"
// 程序会暂停在断点处
```

### 方法2：手动创建配置

```json
// ===== .vscode/launch.json =====

{
  "version": "0.2.0",
  "configurations": [
    {
      "type": "node",
      "request": "launch",
      "name": "启动程序",
      "skipFiles": ["<node_internals>/**"],
      "program": "${workspaceFolder}/app.js"
    }
  ]
}

// 💡 字段说明：
// type: 调试器类型,这里是 node
// request: launch(启动新进程) 或 attach(附加到已运行进程)
// name: 配置名称,显示在调试下拉框
// skipFiles: 跳过的文件(如 Node.js 内部代码)
// program: 要调试的文件路径
```

## 🎮 调试界面详解

### 1. 断点面板

```javascript
// ===== 断点类型 =====

// 1. 普通断点
function normal() {
  const x = 10;  // 👈 点击行号设置断点
  console.log(x);
}

// 2. 条件断点
function conditional(items) {
  for (let i = 0; i < items.length; i++) {
    // 👈 右键行号 → 添加条件断点
    // 条件：i === 5
    // 只有 i === 5 时才暂停
    console.log(items[i]);
  }
}

// 3. 日志点(Logpoint)
function logpoint(user) {
  // 👈 右键行号 → 添加日志点
  // 消息：User {user.name} logged in
  // 不暂停执行,只输出日志
  return user;
}

// 4. 命中次数断点
function hitCount() {
  for (let i = 0; i < 100; i++) {
    // 👈 右键断点 → 编辑断点 → 命中次数
    // 设置：>50
    // 只有第51次及以后才暂停
    console.log(i);
  }
}
```

```javascript
// worker-example.js
const { Worker } = require('worker_threads');

function runWorker() {
  return new Promise((resolve, reject) => {
    const worker = new Worker('./worker.js');
    
    worker.on('message', resolve);
    worker.on('error', reject);
    worker.on('exit', (code) => {
      if (code !== 0) {
        reject(new Error(`Worker stopped with exit code ${code}`));
      }
    });
  });
}

async function main() {
  debugger;  // 主线程断点
  
  const result = await runWorker();
  console.log('Result:', result);
}

main();

// worker.js
const { parentPort } = require('worker_threads');

debugger;  // Worker 线程断点（会自动附加）

parentPort.postMessage({ result: 'Hello from worker' });
```

### 3. skipFiles 配置有什么用？

```json
// 答案：

{
  "skipFiles": [
    "<node_internals>/**",           // 跳过 Node.js 内部代码
    "${workspaceFolder}/node_modules/**",  // 跳过所有依赖
    "!${workspaceFolder}/node_modules/my-lib/**"  // 但不跳过 my-lib
  ]
}

// 作用：
// 1. 单步调试时自动跳过这些文件
// 2. 调用栈中隐藏这些文件
// 3. 提高调试效率，专注于业务代码

// 示例：
function myCode() {
  debugger;
  const result = lodash.map([1, 2, 3], x => x * 2);
  // 按 F11（单步进入）
  // 不会进入 lodash 内部，直接到下一行
  return result;
}
```

// ✅ 1. 使用有意义的配置名称
{
  "name": "🚀 API Server (Dev)",  // 清晰明了

// 所有开发者使用相同的调试配置

"envFile": "${workspaceFolder}/.env.development"

"skipFiles": ["<node_internals>/**", "**/node_modules/**"]

// ✅ 5. 使用条件断点减少干扰
// 右键断点 → 编辑断点 → 表达式: user.id === '123'

// ✅ 6. 善用日志点（Logpoint）
// 不想暂停执行，只想输出日志
// 右键行号 → 添加日志点 → User {user.name} logged in

## 📊 调试工具对比
