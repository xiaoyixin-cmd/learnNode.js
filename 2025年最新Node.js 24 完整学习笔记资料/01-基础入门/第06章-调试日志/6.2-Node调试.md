## 🎯 什么是 Node.js 调试器?

### 基础概念

```
Node.js 调试器 = 内置的命令行调试工具

类比：
汽车修理 = 打开引擎盖,逐个检查零件
代码调试 = 暂停程序运行,逐行检查代码

核心功能：
✅ 设置断点(暂停执行的位置)
✅ 单步执行(一行一行走)
✅ 查看变量值
✅ 监视表达式
✅ 调用栈追踪
```

function calculateTotal(price, quantity) {

// ===== performance-debug.js =====

// 慢函数
function slowFunction(n) {
  debugger;  // 断点1
  
  let result = 0;
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      result += i * j;  // O(n²) 复杂度
    }
  }
  
  debugger;  // 断点2
  return result;
}

console.time('slow');
const result = slowFunction(10000);
console.timeEnd('slow');
// slow: 500ms（太慢了！）

// ===== 优化后 =====

function fastFunction(n) {
  let result = 0;
  const sum = (n * (n - 1)) / 2;
  result = sum * sum;  // O(1) 复杂度
  return result;
}

console.time('fast');
const result2 = fastFunction(10000);
console.timeEnd('fast');
// fast: 0.1ms（快了5000倍！）

const EventEmitter = require('events');

class OrderService extends EventEmitter {
  createOrder(orderData) {
    debugger;  // 断点1
    
    // 验证订单
    if (!orderData.amount || orderData.amount <= 0) {
      this.emit('error', new Error('Invalid amount'));

## 📚 调试技巧总结

### 快捷技巧

```javascript
// ✅ 1. 使用 debugger 语句快速设置断点
function myFunction() {
  debugger;  // 程序会在这里暂停
}

// ✅ 2. 使用 console.trace() 查看调用栈
function deepFunction() {
  console.trace('调用栈');
}

// ✅ 3. 使用 console.time/timeEnd 测量性能
console.time('operation');
expensiveOperation();
console.timeEnd('operation');

// ✅ 4. 使用 process.memoryUsage() 监控内存
setInterval(() => {
  const usage = process.memoryUsage();
  console.log({
    heapUsed: `${(usage.heapUsed / 1024 / 1024).toFixed(2)} MB`,
    heapTotal: `${(usage.heapTotal / 1024 / 1024).toFixed(2)} MB`
  });
}, 1000);

// ✅ 5. 使用 Error.captureStackTrace 获取堆栈
const err = {};
Error.captureStackTrace(err);
console.log(err.stack);
```

## 🎓 面试常见问题

### 1. debugger 语句的作用？

```javascript
// 答案：
// debugger 是 JavaScript 关键字，用于在代码中设置断点
// 当调试器运行时，程序会在 debugger 处暂停
// 如果没有调试器，debugger 语句会被忽略

function example() {
  const x = 10;
  debugger;  // 在这里暂停
  const y = x * 2;
  return y;
}

// 💡 生产环境应该移除 debugger
if (process.env.NODE_ENV !== 'production') {
  debugger;
}
```

### 2. 如何调试运行中的 Node.js 进程？

```bash
# 答案：
# 方法1：发送 SIGUSR1 信号
kill -SIGUSR1 <pid>

# 方法2：使用 --inspect 参数启动
node --inspect app.js

# 方法3：使用 process._debugProcess()
node -e "process._debugProcess(12345)"

# 然后打开 Chrome DevTools 连接
# chrome://inspect
```

### 3. 如何在生产环境调试？

```javascript
// 答案：
// ❌ 不要在生产环境使用断点（会阻塞服务）
// ✅ 使用结构化日志
// ✅ 使用性能监控工具（如 New Relic, DataDog）
// ✅ 使用 APM（Application Performance Monitoring）
// ✅ 使用远程日志收集（ELK Stack）

// 生产环境安全的调试方法：
const logger = require('./logger');

function processOrder(order) {
  logger.info('处理订单', {
    orderId: order.id,
    amount: order.amount,
    timestamp: Date.now()
  });
  
  try {
    // 业务逻辑
    validateOrder(order);
    saveOrder(order);
    
    logger.info('订单处理成功', { orderId: order.id });
  } catch (error) {
    logger.error('订单处理失败', {
      orderId: order.id,
      error: error.message,
      stack: error.stack
    });
    throw error;
  }
}
```
