## 🎯 Node.js 的适用场景

### 适合场景（⭐⭐⭐⭐⭐）

```
✅ I/O 密集型应用
   ├─ Web API 服务
   ├─ 实时通信（WebSocket）
   ├─ 数据库操作
   └─ 文件处理

✅ 前后端同构
   ├─ SSR (服务端渲染)
   ├─ 同构应用
   └─ 全栈 JavaScript

✅ 工具链开发
   ├─ 构建工具 (Webpack, Vite)
   ├─ CLI 工具
   └─ 脚本自动化

✅ 微服务架构
   ├─ API 网关
   ├─ BFF 层
   └─ 轻量级服务
```

### 不适合场景（❌）

```
❌ CPU 密集型应用
   └─ 图像/视频处理
   └─ 大数据计算
   └─ 机器学习训练
   └─ 密码学运算

❌ 低延迟实时系统
   └─ 游戏服务器 (要求 <1ms)
   └─ 高频交易系统
   └─ 实时音视频编码

替代方案：
CPU 密集型 → Go, Rust, C++
低延迟 → Rust, C++
```

### 实战：3 分钟搭建 RESTful API

```javascript
// app.js
const express = require('express');
const app = express();

// 中间件
app.use(express.json());

// 模拟数据库
let users = [
  { id: 1, name: 'Alice', age: 25, email: 'alice@example.com' },
  { id: 2, name: 'Bob', age: 30, email: 'bob@example.com' }
];

// ===== CRUD 操作 =====

// 1. GET - 查询所有用户
app.get('/api/users', (req, res) => {
  res.json({
    success: true,
    data: users,
    total: users.length
  });
});

// 2. GET - 查询单个用户
app.get('/api/users/:id', (req, res) => {
  const user = users.find(u => u.id === parseInt(req.params.id));
  
  if (!user) {
    return res.status(404).json({
      success: false,
      message: 'User not found'
    });
  }
  
  res.json({
    success: true,
    data: user
  });
});

// 3. POST - 创建用户
app.post('/api/users', (req, res) => {
  const { name, age, email } = req.body;
  
  // 数据验证
  if (!name || !age || !email) {
    return res.status(400).json({
      success: false,
      message: 'Missing required fields'
    });
  }
  
  const newUser = {
    id: users.length + 1,
    name,
    age,
    email
  };
  
  users.push(newUser);
  
  res.status(201).json({
    success: true,
    data: newUser,
    message: 'User created successfully'
  });
});

// 4. PUT - 更新用户
app.put('/api/users/:id', (req, res) => {
  const user = users.find(u => u.id === parseInt(req.params.id));
  
  if (!user) {
    return res.status(404).json({
      success: false,
      message: 'User not found'
    });
  }
  
  // 更新用户信息
  Object.assign(user, req.body);
  
  res.json({
    success: true,
    data: user,
    message: 'User updated successfully'
  });
});

// 5. DELETE - 删除用户
app.delete('/api/users/:id', (req, res) => {
  const index = users.findIndex(u => u.id === parseInt(req.params.id));
  
  if (index === -1) {
    return res.status(404).json({
      success: false,
      message: 'User not found'
    });
  }
  
  users.splice(index, 1);
  
  res.json({
    success: true,
    message: 'User deleted successfully'
  });
});

// 错误处理中间件
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({
    success: false,
    message: 'Internal Server Error'
  });
});

// 启动服务器
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`✅ API Server running on http://localhost:${PORT}`);
  console.log(`📖 API Docs: http://localhost:${PORT}/api/users`);
});
```

### 实战：开发文件批量重命名工具

```javascript
#!/usr/bin/env node
// rename-tool.js

const fs = require('fs');
const path = require('path');
const { program } = require('commander');
const chalk = require('chalk');
const ora = require('ora');
const inquirer = require('inquirer');

// ===== 命令行配置 =====
program
  .name('rename-batch')
  .description('批量重命名文件工具')
  .version('1.0.0')
  .argument('[dir]', '目标目录', '.')
  .option('-p, --prefix <prefix>', '文件名前缀')
  .option('-s, --suffix <suffix>', '文件名后缀')
  .option('-e, --ext <ext>', '文件扩展名筛选')
  .option('-d, --dry-run', '预览模式（不实际重命名）')
  .action(async (dir, options) => {
    try {
      await renameFiles(dir, options);
    } catch (error) {
      console.error(chalk.red(`❌ 错误：${error.message}`));
      process.exit(1);
    }
  });

program.parse();

// ===== 核心功能 =====
async function renameFiles(dir, options) {
  const spinner = ora('正在扫描文件...').start();
  
  // 1. 获取文件列表
  let files = fs.readdirSync(dir);
  
  // 2. 筛选文件
  if (options.ext) {
    files = files.filter(f => f.endsWith(`.${options.ext}`));
  }
  
  spinner.succeed(`找到 ${files.length} 个文件`);
  
  if (files.length === 0) {
    console.log(chalk.yellow('⚠️  没有找到匹配的文件'));
    return;
  }
  
  // 3. 交互式确认
  const { confirmed } = await inquirer.prompt([
    {
      type: 'confirm',
      name: 'confirmed',
      message: `确认重命名 ${files.length} 个文件吗？`,
      default: false
    }
  ]);
  
  if (!confirmed) {
    console.log(chalk.yellow('⚠️  操作已取消'));
    return;
  }
  
  // 4. 执行重命名
  const renameSpinner = ora('正在重命名文件...').start();
  
  let count = 0;
  files.forEach((file, index) => {
    const oldPath = path.join(dir, file);
    const ext = path.extname(file);
    const basename = path.basename(file, ext);
    
    // 生成新文件名
    let newName = basename;
    if (options.prefix) newName = options.prefix + newName;
    if (options.suffix) newName = newName + options.suffix;
    newName = `${newName}-${index + 1}${ext}`;
    
    const newPath = path.join(dir, newName);
    
    // 预览模式
    if (options.dryRun) {
      console.log(chalk.gray(`  ${file} → ${newName}`));
    } else {
      fs.renameSync(oldPath, newPath);
      count++;
    }
  });
  
  renameSpinner.succeed(
    options.dryRun
      ? chalk.blue(`📝 预览完成（未实际重命名）`)
      : chalk.green(`✅ 成功重命名 ${count} 个文件！`)
  );
}
```

### 实战：价格监控爬虫 + 邮件通知

```javascript
// price-monitor.js
const axios = require('axios');
const cheerio = require('cheerio');
const nodemailer = require('nodemailer');
const cron = require('node-cron');
const fs = require('fs').promises;

// ===== 配置 =====
const CONFIG = {
  targetPrice: 500,
  productUrl: 'https://example.com/product/12345',
  checkInterval: '0 10 * * *', // 每天 10:00
  emailConfig: {
    service: 'gmail',
    auth: {
      user: process.env.EMAIL_USER,
      pass: process.env.EMAIL_PASS
    }
  }
};

// ===== 爬取商品价格 =====
async function fetchPrice(url) {
  try {
    const { data } = await axios.get(url, {
      headers: {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
      }
    });
    
    const $ = cheerio.load(data);
    const priceText = $('.price').first().text();
    const price = parseFloat(priceText.replace(/[^0-9.]/g, ''));
    
    return price;
  } catch (error) {
    console.error('爬取价格失败：', error.message);
    throw error;
  }
}

// ===== 发送邮件通知 =====
async function sendEmail(price) {
  const transporter = nodemailer.createTransport(CONFIG.emailConfig);
  
  const mailOptions = {
    from: CONFIG.emailConfig.auth.user,
    to: CONFIG.emailConfig.auth.user,
    subject: '🎉 价格提醒：商品降价了！',
    html: `
      <h2>您关注的商品降价了！</h2>
      <p><strong>当前价格：</strong>¥${price}</p>
      <p><strong>目标价格：</strong>¥${CONFIG.targetPrice}</p>
      <p><a href="${CONFIG.productUrl}">立即购买</a></p>
    `
  };
  
  await transporter.sendMail(mailOptions);
  console.log('📧 邮件已发送');
}

// ===== 记录价格历史 =====
async function logPrice(price) {
  const log = {
    timestamp: new Date().toISOString(),
    price
  };
  
  let history = [];
  try {
    const data = await fs.readFile('price-history.json', 'utf-8');
    history = JSON.parse(data);
  } catch (error) {
    // 文件不存在，创建新历史
  }
  
  history.push(log);
  await fs.writeFile('price-history.json', JSON.stringify(history, null, 2));
}

// ===== 主任务 =====
async function checkPrice() {
  console.log('⏰', new Date().toLocaleString(), '- 开始检查价格...');
  
  try {
    const price = await fetchPrice(CONFIG.productUrl);
    console.log(`📊 当前价格：¥${price}`);
    
    await logPrice(price);
    
    if (price <= CONFIG.targetPrice) {
      console.log('🎉 价格达到目标！发送通知...');
      await sendEmail(price);
    } else {
      console.log(`💰 价格高于目标 ¥${CONFIG.targetPrice}，继续监控...`);
    }
  } catch (error) {
    console.error('检查失败：', error.message);
  }
}

// ===== 定时任务 =====
cron.schedule(CONFIG.checkInterval, checkPrice);

console.log('✅ 价格监控已启动！');
console.log(`📅 检查时间：${CONFIG.checkInterval}`);
console.log(`🎯 目标价格：¥${CONFIG.targetPrice}`);

// 立即执行一次
checkPrice();
```

### 实战：实时聊天室

```javascript
// server.js
const express = require('express');
const { createServer } = require('http');
const { Server } = require('socket.io');
const path = require('path');

const app = express();
const httpServer = createServer(app);
const io = new Server(httpServer, {
  cors: {
    origin: "*"
  }
});

// 静态文件
app.use(express.static(path.join(__dirname, 'public')));

// ===== 在线用户管理 =====
const users = new Map(); // { socketId: { username, room } }
const rooms = new Map(); // { roomName: Set<socketId> }

// ===== Socket.IO 事件处理 =====
io.on('connection', (socket) => {
  console.log(`✅ 用户连接：${socket.id}`);
  
  // 1. 用户加入房间
  socket.on('join-room', ({ username, room }) => {
    // 保存用户信息
    users.set(socket.id, { username, room });
    
    // 加入房间
    socket.join(room);
    
    // 更新房间用户列表
    if (!rooms.has(room)) {
      rooms.set(room, new Set());
    }
    rooms.get(room).add(socket.id);
    
    // 通知房间内所有用户
    io.to(room).emit('user-joined', {
      username,
      userCount: rooms.get(room).size,
      timestamp: Date.now()
    });
    
    // 发送欢迎消息
    socket.emit('message', {
      type: 'system',
      content: `欢迎加入 ${room} 房间！`,
      timestamp: Date.now()
    });
    
    console.log(`👤 ${username} 加入房间 ${room}`);
  });
  
  // 2. 接收消息
  socket.on('message', (data) => {
    const user = users.get(socket.id);
    if (!user) return;
    
    const message = {
      type: 'user',
      username: user.username,
      content: data.content,
      timestamp: Date.now()
    };
    
    // 广播到房间
    io.to(user.room).emit('message', message);
    
    console.log(`💬 [${user.room}] ${user.username}: ${data.content}`);
  });
  
  // 3. 用户正在输入
  socket.on('typing', () => {
    const user = users.get(socket.id);
    if (!user) return;
    
    socket.to(user.room).emit('user-typing', {
      username: user.username
    });
  });
  
  // 4. 用户停止输入
  socket.on('stop-typing', () => {
    const user = users.get(socket.id);
    if (!user) return;
    
    socket.to(user.room).emit('user-stop-typing', {
      username: user.username
    });
  });
  
  // 5. 用户离开
  socket.on('disconnect', () => {
    const user = users.get(socket.id);
    if (!user) return;
    
    // 从房间移除
    rooms.get(user.room)?.delete(socket.id);
    
    // 通知其他用户
    io.to(user.room).emit('user-left', {
      username: user.username,
      userCount: rooms.get(user.room)?.size || 0,
      timestamp: Date.now()
    });
    
    // 清理用户信息
    users.delete(socket.id);
    
    console.log(`❌ ${user.username} 离开房间 ${user.room}`);
  });
});

// ===== 启动服务器 =====
const PORT = process.env.PORT || 3000;
httpServer.listen(PORT, () => {
  console.log(`💬 聊天服务器运行在 http://localhost:${PORT}`);
});
```

## 💡 常见问题（FAQ）

**Q1：Node.js 真的不适合 CPU 密集型任务吗？**  
A：不完全是。可以使用 Worker Threads 处理 CPU 密集型任务，但不如 Go、Rust 等语言高效。

**Q2：为什么 Node.js 做实时应用这么流行？**  
A：事件驱动 + 非阻塞 I/O 天然适合长连接，单进程可支持数万并发连接。

**Q3：Node.js 适合做大型企业应用吗？**  
A：适合！Netflix、PayPal、LinkedIn 等大公司都在用。关键是合理的架构设计。
