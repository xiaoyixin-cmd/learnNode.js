## 🎯 为什么使用 TypeScript？

### TypeScript 解决的问题

```javascript
// ===== JavaScript 的痛点 =====

// 问题1：类型错误运行时才发现
function add(a, b) {
  return a + b;
}

add(1, 2);        // 3 ✅
add('1', '2');    // '12' ❌ 本想得到数字3
add(1, '2');      // '12' ❌ 隐式转换

// 问题2：IDE 无法智能提示
const user = {
  name: 'Alice',
  age: 25
};

user.nam;  // 拼写错误，运行时才发现
user.email;  // 属性不存在，运行时才发现

// 问题3：重构困难
function processUser(user) {
  // user 有哪些属性？不知道！
  // 修改 user 结构后，哪些地方需要更新？不清楚！
}
```

```typescript
// ===== TypeScript 的解决方案 =====

// 优势1：编译时类型检查
function add(a: number, b: number): number {
  return a + b;
}

add(1, 2);        // 3 ✅
add('1', '2');    // ❌ 编译错误：类型不匹配
add(1, '2');      // ❌ 编译错误：类型不匹配

// 优势2：智能提示和自动补全
interface User {
  name: string;
  age: number;
}

const user: User = {
  name: 'Alice',
  age: 25
};

user.n  // IDE 自动提示 name
//      ↑ 输入时自动补全

user.email;  // ❌ 编译错误：属性不存在

// 优势3：安全重构
function processUser(user: User) {
  // user 的结构清晰明确
  // 修改 User 接口后，所有使用处都会报错提示
}
```

### TypeScript vs JavaScript

| 特性 | JavaScript | TypeScript |
|------|-----------|-----------|
| **类型检查** | ❌ 运行时 | ✅ 编译时 |
| **IDE 支持** | ⭐⭐⭐ | ⭐⭐⭐⭐⭐ |
| **重构安全** | ⭐⭐ | ⭐⭐⭐⭐⭐ |
| **学习曲线** | ⭐ 简单 | ⭐⭐⭐ 中等 |
| **运行速度** | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐⭐（编译后） |
| **生态支持** | ⭐⭐⭐⭐⭐ | ⭐⭐⭐⭐⭐ |


## ⚙️ tsconfig.json 配置详解

### 基础配置（2025 推荐）

```json
{
  "compilerOptions": {
    /* 语言和环境 */
    "target": "ES2022",                    // 编译目标（Node.js 20+ 支持 ES2022）
    "lib": ["ES2022"],                     // 标准库
    "module": "NodeNext",                  // 模块系统（ESM + CommonJS）
    "moduleResolution": "NodeNext",        // 模块解析策略
    
    /* 输出 */
    "outDir": "./dist",                    // 输出目录
    "rootDir": "./src",                    // 源码目录
    "sourceMap": true,                     // 生成 source map（调试用）
    "declaration": true,                   // 生成 .d.ts 类型声明文件
    "declarationMap": true,                // 生成 .d.ts.map
    "removeComments": false,               // 保留注释
    
    /* 类型检查 */
    "strict": true,                        // 启用所有严格检查
    "noUnusedLocals": true,                // 检查未使用的局部变量
    "noUnusedParameters": true,            // 检查未使用的参数
    "noImplicitReturns": true,             // 检查函数所有路径都有返回值
    "noFallthroughCasesInSwitch": true,    // 检查 switch 的 fallthrough
    
    /* 模块解析 */
    "esModuleInterop": true,               // ESM/CommonJS 互操作
    "allowSyntheticDefaultImports": true,  // 允许合成默认导入
    "resolveJsonModule": true,             // 允许导入 JSON
    "isolatedModules": true,               // 确保每个文件可独立编译
    
    /* 其他 */
    "skipLibCheck": true,                  // 跳过库文件的类型检查（提升速度）
    "forceConsistentCasingInFileNames": true  // 强制文件名大小写一致
  },
  "include": ["src/**/*"],                 // 包含的文件
  "exclude": ["node_modules", "dist"]      // 排除的文件
}
```

### 字段详解

```json
{
  "compilerOptions": {
    /* ===== target：编译目标 ===== */
    "target": "ES2022",
    // ES2015 (ES6) → 支持 class, let/const, 箭头函数
    // ES2020 → 支持 Optional Chaining (?.)、Nullish Coalescing (??)
    // ES2022 → 支持 Top-Level await、class 字段
    // ESNext → 最新特性（不推荐生产）
    
    /* ===== module：模块系统 ===== */
    "module": "NodeNext",
    // CommonJS → require/module.exports
    // ES2020/ES2022 → import/export
    // NodeNext → 根据 package.json 的 type 自动选择（推荐）
    
    /* ===== strict：严格模式 ===== */
    "strict": true,
    // 等同于启用以下所有选项：
    // - strictNullChecks: true        // null/undefined 严格检查
    // - strictFunctionTypes: true     // 函数类型严格检查
    // - strictBindCallApply: true     // bind/call/apply 严格检查
    // - strictPropertyInitialization  // 类属性初始化检查
    // - noImplicitAny: true           // 禁止隐式 any
    // - noImplicitThis: true          // 禁止隐式 this
    
    /* ===== esModuleInterop：互操作性 ===== */
    "esModuleInterop": true,
    // 允许：import express from 'express';
    // 而不是：import * as express from 'express';
  }
}
```

## 🔧 运行 TypeScript

### 方式1：编译后运行（传统）

```bash
# 编译
npx tsc

# 运行
node dist/index.js
```

```json
// package.json
{
  "scripts": {
    "build": "tsc",
    "start": "node dist/index.js",
    "dev": "tsc --watch"  // 监听文件变化，自动编译
  }
}
```

### 方式2：ts-node（实时编译）

```bash
# 安装
npm install -D ts-node

# 运行
npx ts-node src/index.ts
```

```json
// package.json
{
  "scripts": {
    "dev": "ts-node src/index.ts",
    "dev:watch": "nodemon --exec ts-node src/index.ts"
  }
}
```

⚠️ **注意**：ts-node 对 ESM 支持不完善（2025）

### 方式3：tsx（推荐，2025 最佳）

```bash
# 安装
npm install -D tsx

# 运行
npx tsx src/index.ts

# 监听模式
npx tsx watch src/index.ts
```

```json
// package.json
{
  "scripts": {
    "dev": "tsx watch src/index.ts",
    "build": "tsc",
    "start": "node dist/index.js"
  }
}
```

**tsx 优势：**

```
✅ 支持 ESM 和 CommonJS
✅ 比 ts-node 快 2-3 倍
✅ 原生支持 watch 模式
✅ 支持 .env 文件
✅ 零配置
```

```bash

```

```json

  }
```

## 🎓 TypeScript 基础语法

### 1. 基础类型

```typescript
// ===== 基本类型 =====

let isDone: boolean = false;
let age: number = 25;
let username: string = 'Alice';

// ===== 数组 =====

let list: number[] = [1, 2, 3];
let list2: Array<number> = [1, 2, 3];  // 等价写法

// ===== 元组 Tuple =====

let tuple: [string, number] = ['Alice', 25];
tuple[0].toLowerCase();  // ✅ string 方法
tuple[1].toFixed(2);     // ✅ number 方法

// ===== 枚举 Enum =====

enum Color {
  Red,
  Green,
  Blue
}

let color: Color = Color.Green;

// 更推荐：const enum（编译后移除）
const enum Status {
  Pending = 'PENDING',
  Success = 'SUCCESS',
  Failed = 'FAILED'
}

// ===== Any / Unknown =====

let notSure: any = 4;
notSure = 'string';  // ✅ any 可以赋任何值
notSure.toFixed();   // ✅ any 可以调用任何方法（危险）

let value: unknown = 4;
value = 'string';         // ✅ unknown 可以赋任何值
// value.toFixed();       // ❌ 不能直接调用方法
if (typeof value === 'number') {
  value.toFixed();        // ✅ 类型守卫后可以
}

// ===== Void / Never =====

function log(message: string): void {
  console.log(message);
  // 不返回值
}

function error(message: string): never {
  throw new Error(message);
  // 永远不会返回
}
```

### 2. 接口 Interface

```typescript
// ===== 定义接口 =====

interface User {
  id: number;
  name: string;
  email: string;

  readonly createdAt: Date;  // 只读属性
}

const user: User = {
  id: 1,
  name: 'Alice',
  email: 'alice@example.com',
  createdAt: new Date()
};

// user.id = 2;  // ❌ 只读属性不能修改

// ===== 函数接口 =====

interface SearchFunc {
  (query: string, page: number): Promise<string[]>;
}

const search: SearchFunc = async (query, page) => {
  // 实现搜索逻辑
  return [];
};

// ===== 索引签名 =====

interface Dictionary {
  [key: string]: string;
}

const dict: Dictionary = {
  name: 'Alice',
  email: 'alice@example.com'

// ===== 接口继承 =====
```

### 3. 类型别名 Type

```typescript
// ===== 基础类型别名 =====

type ID = string | number;
type Status = 'pending' | 'success' | 'failed';

let userId: ID = 123;
userId = 'abc';  // ✅

let status: Status = 'pending';
// status = 'unknown';  // ❌ 类型错误

// ===== 联合类型 =====

type Result = string | number;

function processResult(result: Result) {
  if (typeof result === 'string') {
    console.log(result.toUpperCase());
  } else {
    console.log(result.toFixed(2));
  }
}

// ===== 交叉类型 =====

type WithTimestamp = {
  createdAt: Date;
  updatedAt: Date;
};

type User = {
  id: number;
  name: string;
};

type UserWithTimestamp = User & WithTimestamp;

const user: UserWithTimestamp = {
  id: 1,
  name: 'Alice',
  createdAt: new Date(),
  updatedAt: new Date()
};
```

### 4. 函数类型

```typescript
// ===== 函数声明 =====

function add(a: number, b: number): number {
  return a + b;
}

// ===== 函数表达式 =====

const subtract: (a: number, b: number) => number = (a, b) => {
  return a - b;
};

// ===== 可选参数 =====

function greet(name: string, greeting?: string): string {
  return `${greeting || 'Hello'}, ${name}`;
}

// ===== 默认参数 =====

function greet2(name: string, greeting: string = 'Hello'): string {
  return `${greeting}, ${name}`;
}

// ===== 剩余参数 =====

function sum(...numbers: number[]): number {
  return numbers.reduce((a, b) => a + b, 0);
}

// ===== 函数重载 =====

function processValue(value: string): string;
function processValue(value: number): number;
function processValue(value: string | number): string | number {
  if (typeof value === 'string') {
    return value.toUpperCase();
  } else {
    return value * 2;
  }
}
```

### 5. 泛型 Generics

```typescript
// ===== 基础泛型 =====

function identity<T>(arg: T): T {
  return arg;
}

identity<number>(123);      // 显式指定
identity('hello');          // 类型推断

// ===== 泛型接口 =====

interface Response<T> {
  code: number;
  message: string;
  data: T;
}

const userResponse: Response<User> = {
  code: 200,
  message: 'Success',
  data: {
    id: 1,
    name: 'Alice',
    email: 'alice@example.com',
    createdAt: new Date()
  }
};

// ===== 泛型类 =====

class Queue<T> {
  private items: T[] = [];
  
  enqueue(item: T): void {
    this.items.push(item);
  }
  
  dequeue(): T | undefined {
    return this.items.shift();
  }
}

const numberQueue = new Queue<number>();
numberQueue.enqueue(1);
numberQueue.enqueue(2);

// ===== 泛型约束 =====

interface HasLength {
  length: number;
}

function logLength<T extends HasLength>(arg: T): void {
  console.log(arg.length);
}

logLength('hello');       // ✅ string 有 length
logLength([1, 2, 3]);     // ✅ array 有 length
// logLength(123);        // ❌ number 没有 length
```

## 🎯 Express + TypeScript 实战

### 完整项目示例

```bash
# 项目结构
src/
├── types/
│   ├── user.ts
│   └── express.d.ts
├── routes/
│   └── user.routes.ts
├── controllers/
│   └── user.controller.ts
├── middleware/
│   └── validate.ts
└── index.ts
```

```typescript
// src/types/user.ts
export interface User {
  id: number;
  name: string;
  email: string;
  age: number;
}

export interface CreateUserDto {
  name: string;
  email: string;
  age: number;
}
```

```typescript
// src/types/express.d.ts（扩展 Express 类型）
import { User } from './user';

declare global {
  namespace Express {
    interface Request {
      user?: User;  // 添加自定义属性
    }
  }
}
```

```typescript
// src/controllers/user.controller.ts
import { Request, Response } from 'express';
import { CreateUserDto, User } from '../types/user';

// 模拟数据库
const users: User[] = [];
let nextId = 1;

export const createUser = (
  req: Request<{}, {}, CreateUserDto>,
  res: Response
) => {
  const user: User = {
    id: nextId++,
    ...req.body
  };
  
  users.push(user);
  res.status(201).json(user);
};

export const getUsers = (req: Request, res: Response) => {
  res.json(users);
};

export const getUserById = (
  req: Request<{ id: string }>,
  res: Response
) => {
  const id = parseInt(req.params.id, 10);
  const user = users.find(u => u.id === id);
  
  if (!user) {
    res.status(404).json({ message: 'User not found' });
    return;
  }
  
  res.json(user);
};
```

```typescript
// src/routes/user.routes.ts
import { Router } from 'express';
import * as userController from '../controllers/user.controller';

const router = Router();

router.post('/users', userController.createUser);
router.get('/users', userController.getUsers);
router.get('/users/:id', userController.getUserById);

export default router;
```

## 💡 最佳实践

### ✅ 推荐做法

```typescript
// 1. 启用 strict 模式
{
  "compilerOptions": {
    "strict": true
  }
}

// 2. 使用接口定义数据结构
interface User {
  id: number;
  name: string;
}

// 3. 避免使用 any（如必须用，注释说明原因）
// ❌
let data: any = fetchData();

// ✅
let data: unknown = fetchData();
if (isValidData(data)) {
  // 类型守卫
}

// 4. 使用枚举或联合类型替代魔法字符串
// ❌
const status = 'pending';

// ✅
type Status = 'pending' | 'success' | 'failed';
const status: Status = 'pending';

// 5. 善用类型推断
// ❌ 冗余
const name: string = 'Alice';

// ✅ 简洁
const name = 'Alice';  // 自动推断为 string
```

### ❌ 避免的做法

```typescript
// ❌ 滥用 any
function process(data: any) {
  return data.value;
}

// ❌ 类型断言绕过检查
const user = {} as User;  // 危险！

// ❌ 忽略 TypeScript 错误
// @ts-ignore
user.name.toUpperCase();

// ❌ 不使用类型声明
const users = [];  // any[]
```
