# 8.5 CLI 文件操作

> 读取、写入、复制、删除文件和目录

---

## 🎯 为什么CLI需要文件操作？

### 核心概念

```
CLI文件操作 = 自动化文件管理任务

常见场景：
✅ 生成配置文件（package.json、.eslintrc等）
✅ 创建项目模板（复制模板文件）
✅ 读取配置文件
✅ 批量处理文件
✅ 文件查找和过滤

Node.js 文件API：
• fs - 标准文件系统模块
• fs/promises - Promise版本（推荐）
• fs-extra - 增强版（推荐）
```

## 📦 fs-extra 入门

### 1. 为什么用 fs-extra？

```javascript
// ===== 原生 fs vs fs-extra =====

// 原生 fs（麻烦）
import { mkdir, writeFile } from 'fs/promises';

// 需要先判断目录是否存在
try {
  await mkdir('path/to/dir');
} catch (error) {
  if (error.code !== 'EEXIST') throw error;
}
await writeFile('path/to/dir/file.txt', 'content');

// fs-extra（简单）
import fs from 'fs-extra';

// 自动创建父目录
await fs.outputFile('path/to/dir/file.txt', 'content');

// 💡 fs-extra 的优势：
// ✅ 支持 Promise
// ✅ 自动创建目录
// ✅ 更多实用方法
// ✅ 向后兼容 fs
```

### 2. 安装

```bash
npm install fs-extra
```

## 📝 读写文件

### 1. 读取文件

```javascript
// ===== 读取文本文件 =====

import fs from 'fs-extra';

// 方式1：读取整个文件
const content = await fs.readFile('file.txt', 'utf8');
console.log(content);

// 方式2：读取 JSON 文件
const data = await fs.readJson('config.json');
console.log(data);  // 自动解析

// 方式3：检查文件是否存在
if (await fs.pathExists('file.txt')) {
  const content = await fs.readFile('file.txt', 'utf8');
}

// ===== 实战：读取 package.json =====

async function readPackageJson() {
  try {
    const pkg = await fs.readJson('./package.json');
    console.log('Project:', pkg.name);
    console.log('Version:', pkg.version);
    return pkg;
  } catch (error) {
    console.error('Failed to read package.json:', error.message);
    return null;
  }
}

// ===== 读取多个文件 =====

async function readMultipleFiles(files) {
  const contents = await Promise.all(
    files.map(file => fs.readFile(file, 'utf8'))
  );
  return contents;
}

const files = ['file1.txt', 'file2.txt', 'file3.txt'];
const contents = await readMultipleFiles(files);
```

### 2. 写入文件

```javascript
// ===== 写入文本文件 =====

import fs from 'fs-extra';

// 方式1：写入文本
await fs.writeFile('output.txt', 'Hello World');

// 方式2：写入JSON（自动格式化）
await fs.writeJson('config.json', { port: 3000, host: 'localhost' }, {
  spaces: 2  // 缩进2个空格
});

// 方式3：自动创建父目录
await fs.outputFile('path/to/nested/file.txt', 'content');
// 会自动创建 path/to/nested/ 目录

// 方式4：追加内容
await fs.appendFile('log.txt', 'New log entry\n');

// ===== 实战：生成配置文件 =====

async function generateConfig(options) {
  const config = {
    name: options.name,
    version: '1.0.0',
    port: options.port || 3000,
    env: options.env || 'development'
  };
  
  // 生成 JSON 文件
  await fs.outputJson('config/app.json', config, { spaces: 2 });
  
  // 生成 .env 文件
  const envContent = Object.entries(config)
    .map(([key, value]) => `${key.toUpperCase()}=${value}`)
    .join('\n');
  
  await fs.outputFile('.env', envContent);
  
  console.log('✔ Config files generated');
}

await generateConfig({
  name: 'my-app',
  port: 8080,
  env: 'production'
});

// 生成的文件：
// config/app.json:
// {
//   "name": "my-app",
//   "version": "1.0.0",
//   "port": 8080,
//   "env": "production"
// }
//
// .env:
// NAME=my-app
// VERSION=1.0.0
// PORT=8080
// ENV=production
```

## 📁 目录操作

### 1. 创建目录

```javascript
// ===== 创建目录 =====

import fs from 'fs-extra';

// 创建单个目录
await fs.mkdir('new-folder');

// 创建嵌套目录（自动创建父目录）
await fs.ensureDir('path/to/nested/folder');
// 等同于 mkdir -p

// 创建多个目录
await Promise.all([
  fs.ensureDir('src'),
  fs.ensureDir('tests'),
  fs.ensureDir('dist')
]);

// ===== 实战：初始化项目结构 =====

async function initProjectStructure(projectName) {
  const dirs = [
    `${projectName}/src`,
    `${projectName}/src/components`,
    `${projectName}/src/utils`,
    `${projectName}/tests`,
    `${projectName}/public`,
    `${projectName}/dist`
  ];
  
  for (const dir of dirs) {
    await fs.ensureDir(dir);
    console.log(`✔ Created ${dir}`);
  }
}

await initProjectStructure('my-app');
// ✔ Created my-app/src
// ✔ Created my-app/src/components
// ✔ Created my-app/src/utils
// ✔ Created my-app/tests
// ✔ Created my-app/public
// ✔ Created my-app/dist
```

### 2. 读取目录

```javascript
// ===== 读取目录内容 =====

import fs from 'fs-extra';
import path from 'path';

// 方式1：读取文件名
const files = await fs.readdir('src');
console.log(files);  // ['index.js', 'utils.js', 'config.js']

// 方式2：读取详细信息
const entries = await fs.readdir('src', { withFileTypes: true });

for (const entry of entries) {
  if (entry.isFile()) {
    console.log('File:', entry.name);
  } else if (entry.isDirectory()) {
    console.log('Directory:', entry.name);
  }
}

// ===== 递归读取目录树 =====

async function readDirRecursive(dir, fileList = []) {
  const files = await fs.readdir(dir);
  
  for (const file of files) {
    const filePath = path.join(dir, file);
    const stat = await fs.stat(filePath);
    
    if (stat.isDirectory()) {
      await readDirRecursive(filePath, fileList);
    } else {
      fileList.push(filePath);
    }
  }
  
  return fileList;
}

const allFiles = await readDirRecursive('src');
console.log('All files:', allFiles);

// ===== 过滤文件 =====

async function findFiles(dir, pattern) {
  const allFiles = await readDirRecursive(dir);
  return allFiles.filter(file => file.match(pattern));
}

// 查找所有 .js 文件
const jsFiles = await findFiles('src', /\.js$/);
console.log('JavaScript files:', jsFiles);
```

### 3. 删除目录

```javascript
// ===== 删除目录 =====

import fs from 'fs-extra';

// 删除空目录
await fs.rmdir('empty-folder');

// 删除目录及其内容
await fs.remove('folder-with-contents');

// 清空目录（保留目录本身）
await fs.emptyDir('dist');

// ===== 实战：清理构建产物 =====

async function clean() {
  console.log('Cleaning build artifacts...');
  
  await fs.emptyDir('dist');
  await fs.emptyDir('.cache');
  await fs.remove('node_modules/.cache');
  
  console.log('✔ Clean completed');
}

await clean();
```

## 📋 复制和移动

### 1. 复制文件和目录

```javascript
// ===== 复制操作 =====

import fs from 'fs-extra';

// 复制文件
await fs.copy('source.txt', 'destination.txt');

// 复制目录（递归）
await fs.copy('src', 'backup/src');

// 复制时过滤
await fs.copy('src', 'dist', {
  filter: (src) => {
    // 排除 .test.js 文件
    return !src.endsWith('.test.js');
  }
});

// ===== 实战：复制项目模板 =====

async function copyTemplate(templateName, projectName) {
  const templatePath = `templates/${templateName}`;
  const targetPath = projectName;
  
  console.log(`Copying template: ${templateName}...`);
  
  await fs.copy(templatePath, targetPath, {
    filter: (src) => {
      // 排除 node_modules 和 .git
      return !src.includes('node_modules') && !src.includes('.git');
    }
  });
  
  console.log(`✔ Template copied to ${projectName}`);
}

await copyTemplate('react-ts', 'my-new-app');
```

### 2. 移动文件

```javascript
// ===== 移动操作 =====

import fs from 'fs-extra';

// 移动文件
await fs.move('old-location.txt', 'new-location.txt');

// 移动目录
await fs.move('old-folder', 'new-folder');

// 覆盖已存在的文件
await fs.move('source.txt', 'dest.txt', { overwrite: true });

// ===== 实战：整理文件 =====

async function organizeFiles(sourceDir) {
  const files = await fs.readdir(sourceDir);
  
  for (const file of files) {
    const ext = path.extname(file);
    const targetDir = path.join(sourceDir, ext.slice(1) || 'other');
    
    await fs.ensureDir(targetDir);
    await fs.move(
      path.join(sourceDir, file),
      path.join(targetDir, file)
    );
    
    console.log(`✔ Moved ${file} to ${ext.slice(1) || 'other'}/`);
  }
}

await organizeFiles('downloads');
// ✔ Moved photo.jpg to jpg/
// ✔ Moved document.pdf to pdf/
// ✔ Moved script.js to js/
```

## 🎯 实战案例

### 案例1：项目脚手架

```javascript
// ===== 完整的项目生成器 =====

import fs from 'fs-extra';
import path from 'path';
import inquirer from 'inquirer';
import ora from 'ora';
import chalk from 'chalk';

async function createProject() {
  // 1. 询问用户
  const answers = await inquirer.prompt([
    {
      type: 'input',
      name: 'projectName',
      message: 'Project name:',
      validate: (input) => {
        if (!input) return 'Project name is required';
        if (fs.existsSync(input)) return 'Directory already exists';
        return true;
      }
    },
    {
      type: 'list',
      name: 'template',
      message: 'Select template:',
      choices: ['react', 'vue', 'node']
    }
  ]);
  
  const { projectName, template } = answers;
  
  // 2. 创建项目目录
  const spinner = ora('Creating project...').start();
  
  try {
    await fs.ensureDir(projectName);
    
    // 3. 复制模板
    spinner.text = 'Copying template files...';
    await fs.copy(
      path.join(__dirname, `../templates/${template}`),
      projectName
    );
    
    // 4. 生成 package.json
    spinner.text = 'Generating package.json...';
    const packageJson = {
      name: projectName,
      version: '1.0.0',
      description: `${projectName} - Created with CLI`,
      scripts: {
        dev: 'vite',
        build: 'vite build',
        preview: 'vite preview'
      },
      dependencies: {
        [template]: '^18.0.0'
      }
    };
    
    await fs.writeJson(
      path.join(projectName, 'package.json'),
      packageJson,
      { spaces: 2 }
    );
    
    // 5. 生成 README
    spinner.text = 'Generating README...';
    const readme = `# ${projectName}

## Getting Started

\`\`\`bash
cd ${projectName}
npm install
npm run dev
\`\`\`

## Build

\`\`\`bash
npm run build
\`\`\`
`;
    
    await fs.writeFile(
      path.join(projectName, 'README.md'),
      readme
    );
    
    // 6. 生成 .gitignore
    spinner.text = 'Generating .gitignore...';
    const gitignore = `node_modules
dist
.env
.DS_Store
`;
    
    await fs.writeFile(
      path.join(projectName, '.gitignore'),
      gitignore
    );
    
    spinner.succeed('Project created successfully!');
    
    // 7. 显示后续步骤
    console.log(chalk.green('\n✨ Done! Next steps:\n'));
    console.log(chalk.white(`  cd ${projectName}`));
    console.log(chalk.white('  npm install'));
    console.log(chalk.white('  npm run dev\n'));
    
  } catch (error) {
    spinner.fail('Failed to create project');
    console.error(chalk.red(error.message));
    
    // 清理失败的目录
    await fs.remove(projectName);
  }
}

createProject();
```

### 案例2：批量处理文件

```javascript
// ===== 批量重命名文件 =====

import fs from 'fs-extra';
import path from 'path';

async function batchRename(dir, pattern, replacement) {
  const files = await fs.readdir(dir);
  
  for (const file of files) {
    if (file.match(pattern)) {
      const oldPath = path.join(dir, file);
      const newName = file.replace(pattern, replacement);
      const newPath = path.join(dir, newName);
      
      await fs.move(oldPath, newPath);
      console.log(`✔ Renamed ${file} to ${newName}`);
    }
  }
}

// 将所有 .txt 文件重命名为 .md
await batchRename('docs', /\.txt$/, '.md');

// ===== 批量转换文件内容 =====

async function batchConvert(dir, from, to) {
  const files = await fs.readdir(dir);
  
  for (const file of files) {
    const filePath = path.join(dir, file);
    const content = await fs.readFile(filePath, 'utf8');
    const converted = content.replaceAll(from, to);
    
    await fs.writeFile(filePath, converted);
    console.log(`✔ Converted ${file}`);
  }
}

// 将所有文件中的 "foo" 替换为 "bar"
await batchConvert('src', 'foo', 'bar');
```

## 💡 最佳实践

```javascript
// ✅ 1. 检查文件是否存在
if (await fs.pathExists('file.txt')) {
  // 文件存在
}

// ✅ 2. 使用 try/catch 处理错误
try {
  await fs.readFile('file.txt');
} catch (error) {
  if (error.code === 'ENOENT') {
    console.error('File not found');
  } else {
    throw error;
  }
}

// ✅ 3. 使用 outputFile 代替 writeFile
// 自动创建父目录
await fs.outputFile('path/to/file.txt', 'content');

// ✅ 4. 使用 readJson/writeJson 处理JSON
const data = await fs.readJson('config.json');
await fs.writeJson('config.json', data, { spaces: 2 });

// ✅ 5. 删除前确认
const { confirm } = await inquirer.prompt([{
  type: 'confirm',
  name: 'confirm',
  message: 'Delete all files?'
}]);

if (confirm) {
  await fs.remove('folder');
}

// ✅ 6. 复制时过滤不需要的文件
await fs.copy('src', 'dist', {
  filter: (src) => {
    return !src.includes('node_modules') &&
           !src.includes('.git') &&
           !src.includes('.test.');
  }
});

// ✅ 7. 使用 ensureDir 而不是 mkdir
await fs.ensureDir('path/to/dir');  // 不会报错

// ✅ 8. 批量操作使用 Promise.all
await Promise.all(
  files.map(file => fs.remove(file))
);
```
