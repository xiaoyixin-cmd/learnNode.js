# 8.1 CLI 项目初始化

> 从零开始创建一个专业的命令行工具项目

---

## 🎯 什么是 CLI 工具？

### 核心概念

```
CLI = Command Line Interface（命令行界面）

生活中的比喻：
GUI应用 = 用鼠标点击的应用（如Word、Chrome）
CLI应用 = 在终端输入命令的工具（如git、npm）

常见的CLI工具：
• git - 版本控制
• npm - 包管理
• docker - 容器管理
• vue-cli - 项目脚手架
• nodemon - 自动重启

为什么要学CLI开发？
✅ 自动化重复任务
✅ 提高工作效率
✅ 打造个人工具
✅ 增加简历亮点
```

## 🚀 项目初始化

### 1. 创建项目目录

```bash
# ===== 创建项目 =====

mkdir my-cli-tool
cd my-cli-tool

# ===== 初始化 package.json =====

npm init -y

# 输出：
# Wrote to /path/to/my-cli-tool/package.json:
# {
#   "name": "my-cli-tool",
#   "version": "1.0.0",
#   ...
# }
```

### 2. 配置 package.json

```json
{
  "name": "my-cli-tool",
  "version": "1.0.0",
  "description": "A powerful CLI tool",
  
  // ===== 重要：CLI 工具配置 =====
  "type": "module",  // 使用 ES Modules
  
  "bin": {
    // 命令名称: 入口文件
    "my-cli": "./bin/cli.js"
    // 💡 用户安装后可以直接运行: my-cli
  },
  
  "files": [
    // 发布到 npm 时包含的文件
    "bin",
    "lib",
    "README.md"
  ],
  
  "scripts": {
    "dev": "node --watch bin/cli.js",  // 开发模式（Node.js 18+）
    "test": "node --test",              // 运行测试
    "lint": "eslint .",                 // 代码检查
    "format": "prettier --write ."      // 代码格式化
  },
  
  "keywords": ["cli", "tool", "command-line"],
  "author": "Your Name <your.email@example.com>",
  "license": "MIT",
  
  "engines": {
    "node": ">=18.0.0"  // 要求 Node.js 版本
  }
}
```

### 3. 创建入口文件

```javascript
// ===== bin/cli.js =====

#!/usr/bin/env node
// ☝️ Shebang：告诉系统用 node 运行这个文件

// 💡 为什么需要 Shebang？
// 在 Unix/Linux/macOS 上，可以直接运行：./cli.js
// 而不需要：node cli.js

console.log('Hello from my CLI tool!');
console.log('Node version:', process.version);
console.log('Arguments:', process.argv);

// process.argv 是命令行参数数组
// 例如：my-cli create app
// process.argv = [
//   '/path/to/node',       // Node.js 可执行文件路径
//   '/path/to/cli.js',     // 当前脚本路径
//   'create',              // 用户输入的参数
//   'app'
// ]
```

### 4. 设置执行权限（Unix/Linux/macOS）

```bash
# ===== 赋予执行权限 =====

chmod +x bin/cli.js

# 💡 这样就可以直接运行
./bin/cli.js

# 输出：
# Hello from my CLI tool!
# Node version: v20.10.0
# Arguments: [ '/usr/local/bin/node', '/path/to/bin/cli.js' ]
```

## 🎯 第一个完整示例

### 创建基础 CLI

```javascript
// ===== bin/cli.js（完整版）=====

#!/usr/bin/env node

import { program } from 'commander';
import chalk from 'chalk';
import boxen from 'boxen';

// 读取 package.json
import { readFileSync } from 'fs';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

// 获取当前文件目录
const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

// 读取版本号
const pkgPath = join(__dirname, '../package.json');
const pkg = JSON.parse(readFileSync(pkgPath, 'utf8'));

// ===== 配置 commander =====

program
  .name('my-cli')
  .description('A powerful CLI tool')
  .version(pkg.version);

// ===== 定义命令：greet =====

program
  .command('greet <name>')
  .description('Greet someone')
  .option('-u, --upper', 'Uppercase the greeting')
  .action((name, options) => {
    let greeting = `Hello, ${name}!`;
    
    // 如果有 --upper 选项，转为大写
    if (options.upper) {
      greeting = greeting.toUpperCase();
    }
    
    // 彩色输出
    console.log(chalk.green(greeting));
    
    // 添加边框
    console.log(boxen(greeting, {
      padding: 1,
      margin: 1,
      borderColor: 'green',
      borderStyle: 'round'
    }));
  });

// ===== 定义命令：info =====

program
  .command('info')
  .description('Show CLI information')
  .action(() => {
    console.log(chalk.cyan('CLI Information:'));
    console.log(chalk.yellow(`Name: ${pkg.name}`));
    console.log(chalk.yellow(`Version: ${pkg.version}`));
    console.log(chalk.yellow(`Description: ${pkg.description}`));
  });

// ===== 解析命令行参数 =====

program.parse();

// ===== 如果没有参数，显示帮助 =====

if (!process.argv.slice(2).length) {
  program.outputHelp();
}
```

## 🎯 实战案例：项目生成器

```javascript
// ===== 完整的项目生成器 CLI =====

// bin/cli.js
#!/usr/bin/env node

import { program } from 'commander';
import { createCommand } from '../lib/commands/create.js';
import { readFileSync } from 'fs';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const pkg = JSON.parse(readFileSync(join(__dirname, '../package.json'), 'utf8'));

program
  .name('project-gen')
  .description('Project generator CLI')
  .version(pkg.version);

program
  .command('create <project-name>')
  .description('Create a new project')
  .option('-t, --template <type>', 'Project template (react/vue/node)', 'node')
  .option('--skip-install', 'Skip npm install')
  .action(createCommand);

program.parse();

// ===== lib/commands/create.js =====

import inquirer from 'inquirer';
import ora from 'ora';
import chalk from 'chalk';
import { mkdirSync, writeFileSync, cpSync } from 'fs';
import { join } from 'path';
import { execSync } from 'child_process';

export async function createCommand(projectName, options) {
  console.log(chalk.cyan(`\n🚀 Creating project: ${projectName}\n`));
  
  // 1. 确认模板
  if (!['react', 'vue', 'node'].includes(options.template)) {
    const { template } = await inquirer.prompt([
      {
        type: 'list',
        name: 'template',
        message: 'Select a template:',
        choices: ['react', 'vue', 'node']
      }
    ]);
    options.template = template;
  }
  
  // 2. 创建项目目录
  const spinner = ora('Creating project directory...').start();
  try {
    mkdirSync(projectName, { recursive: true });
    spinner.succeed('Project directory created');
  } catch (error) {
    spinner.fail('Failed to create directory');
    console.error(chalk.red(error.message));
    process.exit(1);
  }
  
  // 3. 复制模板文件
  spinner.start('Copying template files...');
  try {
    const templatePath = new URL(`../../templates/${options.template}`, import.meta.url);
    cpSync(templatePath, projectName, { recursive: true });
    spinner.succeed('Template files copied');
  } catch (error) {
    spinner.fail('Failed to copy template');
    console.error(chalk.red(error.message));
    process.exit(1);
  }
  
  // 4. 创建 package.json
  spinner.start('Creating package.json...');
  const packageJson = {
    name: projectName,
    version: '1.0.0',
    description: `${projectName} - Created by project-gen`,
    main: 'index.js',
    scripts: {
      start: 'node index.js',
      dev: 'nodemon index.js'
    }
  };
  writeFileSync(
    join(projectName, 'package.json'),
    JSON.stringify(packageJson, null, 2)
  );
  spinner.succeed('package.json created');
  
  // 5. 安装依赖
  if (!options.skipInstall) {
    spinner.start('Installing dependencies...');
    try {
      execSync('npm install', {
        cwd: projectName,
        stdio: 'ignore'
      });
      spinner.succeed('Dependencies installed');
    } catch (error) {
      spinner.fail('Failed to install dependencies');
      console.log(chalk.yellow('You can run "npm install" manually'));
    }
  }
  
  // 6. 完成提示
  console.log(chalk.green('\n✨ Project created successfully!\n'));
  console.log(chalk.cyan('Next steps:'));
  console.log(chalk.white(`  cd ${projectName}`));
  if (options.skipInstall) {
    console.log(chalk.white('  npm install'));
  }
  console.log(chalk.white('  npm start\n'));
}
```
