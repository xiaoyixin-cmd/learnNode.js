# 8.8 CLI 错误处理

> 优雅地处理错误,提供友好的错误提示

## 🎯 为什么需要好的错误处理？

### 核心概念

```
CLI错误处理 = 让用户知道发生了什么,如何解决

糟糕的错误提示（❌）：
Error: ENOENT
Exit code: 1

良好的错误提示（✅）：
❌ Error: File not found

  Cannot find config.json

  Suggestions:
    • Make sure the file exists
    • Check the file path
    • Run 'my-cli init' to create a default config

好的错误处理应该：
✅ 清晰的错误信息
✅ 提供上下文
✅ 给出解决建议
✅ 记录错误日志
✅ 正确的退出码
```

## 🎨 友好的错误提示

### 1. 基础错误处理

```javascript
// ===== 捕获并美化错误 =====

import chalk from 'chalk';

function showError(message, suggestions = []) {
  console.error(chalk.red('\n❌ Error: ' + message + '\n'));
  
  if (suggestions.length > 0) {
    console.error(chalk.yellow('Suggestions:'));
    suggestions.forEach(suggestion => {
      console.error(chalk.white('  • ' + suggestion));
    });
    console.error();
  }
}

// ===== 使用 =====

try {
  const config = await readConfig();
} catch (error) {
  showError('Cannot read config file', [
    'Make sure config.json exists',
    'Run "my-cli init" to create a default config',
    'Check file permissions'
  ]);
  process.exit(1);
}

// ===== 输出：=====
// ❌ Error: Cannot read config file
//
// Suggestions:
//   • Make sure config.json exists
//   • Run "my-cli init" to create a default config
//   • Check file permissions
```

### 2. 错误类型分类

```javascript
// ===== 自定义错误类 =====

class CLIError extends Error {
  constructor(message, suggestions = []) {
    super(message);
    this.name = 'CLIError';
    this.suggestions = suggestions;
  }
}

class ValidationError extends CLIError {
  constructor(message, suggestions = []) {
    super(message, suggestions);
    this.name = 'ValidationError';
  }
}

class NetworkError extends CLIError {
  constructor(message, suggestions = []) {
    super(message, suggestions);
    this.name = 'NetworkError';
  }
}

class FileError extends CLIError {
  constructor(message, suggestions = []) {
    super(message, suggestions);
    this.name = 'FileError';
  }
}

// ===== 使用 =====

function validateEmail(email) {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  
  if (!emailRegex.test(email)) {
    throw new ValidationError('Invalid email address', [
      'Email must contain @',
      'Example: user@example.com'
    ]);
  }
}

async function downloadFile(url) {
  try {
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }
    return response;
  } catch (error) {
    throw new NetworkError('Failed to download file', [
      'Check your internet connection',
      'Verify the URL is correct',
      'Try again later'
    ]);
  }
}

// ===== 统一错误处理 =====

function handleError(error) {
  if (error instanceof CLIError) {
    console.error(chalk.red(`\n❌ ${error.name}: ${error.message}\n`));
    
    if (error.suggestions.length > 0) {
      console.error(chalk.yellow('Suggestions:'));
      error.suggestions.forEach(s => {
        console.error(chalk.white(`  • ${s}`));
      });
      console.error();
    }
  } else {
    // 未知错误
    console.error(chalk.red(`\n❌ Unexpected error: ${error.message}\n`));
    console.error(chalk.gray(error.stack));
  }
  
  process.exit(1);
}

// ===== 主函数包装 =====

async function main() {
  try {
    // CLI 逻辑
    validateEmail('invalid-email');
  } catch (error) {
    handleError(error);
  }
}

main();
```

## 🔍 常见错误处理

### 1. 文件不存在

```javascript
// ===== 处理 ENOENT 错误 =====

import fs from 'fs-extra';
import chalk from 'chalk';

async function readConfig() {
  try {
    return await fs.readJson('config.json');
  } catch (error) {
    if (error.code === 'ENOENT') {
      throw new FileError('Config file not found', [
        `Looking for: ${process.cwd()}/config.json`,
        'Run "my-cli init" to create a config file'
      ]);
    }
    throw error;
  }
}
```

### 2. 网络错误

```javascript
// ===== 处理网络请求错误 =====

async function fetchData(url) {
  try {
    const response = await fetch(url, {
      signal: AbortSignal.timeout(5000)  // 5秒超时
    });
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }
    
    return await response.json();
    
  } catch (error) {
    let suggestions = [];
    
    if (error.name === 'AbortError') {
      suggestions = [
        'Request timed out (>5s)',
        'Check your internet connection',
        'Try again later'
      ];
    } else if (error.message.includes('HTTP')) {
      suggestions = [
        `Server returned ${error.message}`,
        'Check the API endpoint',
        'Verify your credentials'
      ];
    } else {
      suggestions = [
        'Network error occurred',
        'Check your internet connection'
      ];
    }
    
    throw new NetworkError('Failed to fetch data', suggestions);
  }
}
```

### 3. 参数验证

```javascript
// ===== 验证命令行参数 =====

function validateProjectName(name) {
  if (!name || name.length === 0) {
    throw new ValidationError('Project name is required', [
      'Usage: my-cli create <project-name>',
      'Example: my-cli create my-app'
    ]);
  }
  
  if (!/^[a-z0-9-_]+$/.test(name)) {
    throw new ValidationError('Invalid project name', [
      'Only lowercase letters, numbers, hyphens and underscores are allowed',
      'Examples: my-app, my_app, myapp123'
    ]);
  }
  
  if (fs.existsSync(name)) {
    throw new ValidationError(`Directory "${name}" already exists`, [
      'Choose a different name',
      'Delete the existing directory',
      'Use --force to overwrite'
    ]);
  }
}
```

### 4. 权限错误

```javascript
// ===== 处理 EACCES 错误 =====

async function writeFile(path, content) {
  try {
    await fs.writeFile(path, content);
  } catch (error) {
    if (error.code === 'EACCES') {
      throw new FileError('Permission denied', [
        `Cannot write to: ${path}`,
        'Check file permissions',
        'Try running with sudo (not recommended)',
        'Choose a different location'
      ]);
    }
    throw error;
  }
}
```

## 🎯 实战案例

### 案例1：完整的错误处理系统

```javascript
// ===== 完整的CLI错误处理 =====

import chalk from 'chalk';
import boxen from 'boxen';
import fs from 'fs-extra';

// 错误类
class CLIError extends Error {
  constructor(message, code, suggestions = [], context = {}) {
    super(message);
    this.name = 'CLIError';
    this.code = code;
    this.suggestions = suggestions;
    this.context = context;
  }
}

// 错误处理器
class ErrorHandler {
  static handle(error) {
    // 记录错误日志
    this.logError(error);
    
    // 显示友好的错误信息
    this.displayError(error);
    
    // 退出程序
    process.exit(this.getExitCode(error));
  }
  
  static displayError(error) {
    console.error();
    
    if (error instanceof CLIError) {
      // 自定义错误
      console.error(chalk.red.bold(`❌ ${error.message}`));
      console.error();
      
      // 显示上下文
      if (Object.keys(error.context).length > 0) {
        console.error(chalk.gray('Context:'));
        for (const [key, value] of Object.entries(error.context)) {
          console.error(chalk.gray(`  ${key}: ${value}`));
        }
        console.error();
      }
      
      // 显示建议
      if (error.suggestions.length > 0) {
        console.error(chalk.yellow('💡 Suggestions:'));
        error.suggestions.forEach((s, i) => {
          console.error(chalk.white(`  ${i + 1}. ${s}`));
        });
        console.error();
      }
      
      // 显示错误代码
      console.error(chalk.gray(`Error code: ${error.code}`));
      
    } else {
      // 未知错误
      console.error(boxen(
        chalk.red.bold('Unexpected Error\n\n') +
        chalk.white(error.message) +
        '\n\n' +
        chalk.gray('This might be a bug. Please report it:\n') +
        chalk.cyan('https://github.com/user/cli/issues'),
        {
          padding: 1,
          borderColor: 'red',
          borderStyle: 'round'
        }
      ));
      
      // 显示堆栈（开发模式）
      if (process.env.DEBUG) {
        console.error();
        console.error(chalk.gray('Stack trace:'));
        console.error(chalk.gray(error.stack));
      }
    }
    
    console.error();
  }
  
  static logError(error) {
    const logDir = 'logs';
    const logFile = `${logDir}/error.log`;
    
    try {
      fs.ensureDirSync(logDir);
      
      const logEntry = {
        timestamp: new Date().toISOString(),
        error: {
          name: error.name,
          message: error.message,
          code: error.code || 'UNKNOWN',
          stack: error.stack
        },
        environment: {
          node: process.version,
          platform: process.platform,
          cwd: process.cwd()
        }
      };
      
      fs.appendFileSync(
        logFile,
        JSON.stringify(logEntry, null, 2) + '\n\n'
      );
    } catch (err) {
      // 静默失败（不要因为日志错误而崩溃）
    }
  }
  
  static getExitCode(error) {
    if (error instanceof CLIError) {
      // 根据错误代码返回不同的退出码
      const exitCodes = {
        'FILE_NOT_FOUND': 2,
        'PERMISSION_DENIED': 3,
        'NETWORK_ERROR': 4,
        'VALIDATION_ERROR': 5
      };
      return exitCodes[error.code] || 1;
    }
    return 1;  // 默认错误退出码
  }
}

// ===== 使用 =====

async function createProject(name, options) {
  try {
    // 验证项目名
    if (!/^[a-z0-9-_]+$/.test(name)) {
      throw new CLIError(
        'Invalid project name',
        'VALIDATION_ERROR',
        [
          'Use only lowercase letters, numbers, hyphens and underscores',
          'Examples: my-app, my_app, myapp123'
        ],
        { input: name }
      );
    }
    
    // 检查目录是否存在
    if (fs.existsSync(name)) {
      throw new CLIError(
        `Directory "${name}" already exists`,
        'FILE_EXISTS',
        [
          'Choose a different name',
          'Delete the existing directory',
          'Use --force to overwrite'
        ],
        { directory: name, path: process.cwd() }
      );
    }
    
    // 创建项目...
    
  } catch (error) {
    ErrorHandler.handle(error);
  }
}
```

### 案例2：优雅的退出

// ===== 捕获意外终止 =====

import chalk from 'chalk';

class GracefulShutdown {
  static init() {
    // Ctrl+C
    process.on('SIGINT', () => {
      console.log(chalk.yellow('\n\n⚠️  Interrupted by user'));
      console.log(chalk.gray('Cleaning up...'));
      this.cleanup();
      process.exit(130);  // 130 = 128 + SIGINT(2)
    });
    
    // 终止信号
    process.on('SIGTERM', () => {
      console.log(chalk.yellow('\n\n⚠️  Termination signal received'));
      console.log(chalk.gray('Cleaning up...'));
      this.cleanup();
      process.exit(143);  // 143 = 128 + SIGTERM(15)
    });
    
    // 未捕获的异常
    process.on('uncaughtException', (error) => {
      console.error(chalk.red('\n\n❌ Uncaught Exception'));
      console.error(error);
      this.cleanup();
      process.exit(1);
    });
    
    // 未处理的 Promise 拒绝
    process.on('unhandledRejection', (reason, promise) => {
      console.error(chalk.red('\n\n❌ Unhandled Promise Rejection'));
      console.error('Reason:', reason);
      this.cleanup();
      process.exit(1);
    });
  }
  
  static cleanup() {
    // 清理临时文件
    // 关闭数据库连接
    // 保存状态
    // etc.
  }
}

GracefulShutdown.init();
