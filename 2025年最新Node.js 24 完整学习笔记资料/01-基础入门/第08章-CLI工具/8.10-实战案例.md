### 2. CLI 入口 (bin/cli.js)

```javascript
#!/usr/bin/env node

import { program } from 'commander';
import chalk from 'chalk';
import boxen from 'boxen';
import { readFileSync } from 'fs';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

// 导入命令
import { createCommand } from '../lib/commands/create.js';
import { configCommand } from '../lib/commands/config.js';
import { listCommand } from '../lib/commands/list.js';
import { ErrorHandler } from '../lib/errors.js';

// 获取 package.json
const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const pkg = JSON.parse(readFileSync(join(__dirname, '../package.json'), 'utf8'));

// 显示横幅
console.log(
  boxen(
    chalk.cyan.bold('Create Modern App') + '\n' +
    chalk.gray(`v${pkg.version}`) + '\n\n' +
    chalk.white('Interactive project scaffolding tool'),
    {
      padding: 1,
      margin: 1,
      borderStyle: 'round',
      borderColor: 'cyan'
    }
  )
);

// 配置 CLI
program
  .name('create-modern-app')
  .description('Modern project scaffolding CLI tool')
  .version(pkg.version);

// create 命令
program
  .command('create [project-name]')
  .description('Create a new project')
  .option('-t, --template <type>', 'Project template (react/vue/node)')
  .option('--skip-install', 'Skip npm install')
  .option('--skip-git', 'Skip git init')
  .option('-f, --force', 'Force overwrite existing directory')
  .action(async (projectName, options) => {
    try {
      await createCommand(projectName, options);
    } catch (error) {
      ErrorHandler.handle(error);
    }
  });

// config 命令
program
  .command('config')
  .description('Manage configuration')
  .action(async () => {
    try {
      await configCommand();
    } catch (error) {
      ErrorHandler.handle(error);
    }
  });

// list 命令
program
  .command('list')
  .alias('ls')
  .description('List available templates')
  .action(async () => {
    try {
      await listCommand();
    } catch (error) {
      ErrorHandler.handle(error);
    }
  });

// 解析命令
program.parse();

// 如果没有参数，显示帮助
if (!process.argv.slice(2).length) {
  program.outputHelp();
}
```

### 3. Create 命令 (lib/commands/create.js)

```javascript
import inquirer from 'inquirer';
import ora from 'ora';
import chalk from 'chalk';
import fs from 'fs-extra';
import path from 'path';
import { fileURLToPath } from 'url';
import { dirname } from 'path';

import { Logger } from '../utils/logger.js';
import { validateProjectName } from '../utils/file.js';
import { initGit } from '../utils/git.js';
import { installDependencies } from '../utils/npm.js';
import { ConfigManager } from '../config-manager.js';
import { CLIError } from '../errors.js';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

export async function createCommand(projectName, options) {
  const config = new ConfigManager();
  
  // 步骤1：询问项目名称（如果未提供）
  if (!projectName) {
    const answers = await inquirer.prompt([
      {
        type: 'input',
        name: 'projectName',
        message: 'Project name:',
        default: 'my-app',
        validate: (input) => {
          try {
            validateProjectName(input);
            return true;
          } catch (error) {
            return error.message;
          }
        }
      }
    ]);
    projectName = answers.projectName;
  } else {
    // 验证项目名
    try {
      validateProjectName(projectName);
    } catch (error) {
      throw new CLIError(error.message, 'VALIDATION_ERROR', [
        'Use only lowercase letters, numbers, hyphens and underscores',
        'Examples: my-app, my_app, myapp123'
      ]);
    }
  }
  
  // 检查目录是否存在
  if (fs.existsSync(projectName) && !options.force) {
    const { overwrite } = await inquirer.prompt([
      {
        type: 'confirm',
        name: 'overwrite',
        message: `Directory "${projectName}" already exists. Overwrite?`,
        default: false
      }
    ]);
    
    if (!overwrite) {
      Logger.info('Operation cancelled');
      return;
    }
    
    // 删除现有目录
    await fs.remove(projectName);
  }
  
  // 步骤2：选择模板
  let template = options.template;
  
  if (!template) {
    const answers = await inquirer.prompt([
      {
        type: 'list',
        name: 'template',
        message: 'Select a template:',
        choices: [
          { name: 'React', value: 'react' },
          { name: 'Vue 3', value: 'vue' },
          { name: 'Node.js', value: 'node' }
        ],
        default: config.get('defaultTemplate') || 'react'
      }
    ]);
    template = answers.template;
  }
  
  // 步骤3：其他配置
  const projectConfig = await inquirer.prompt([
    {
      type: 'confirm',
      name: 'typescript',
      message: 'Use TypeScript?',
      default: true
    },
    {
      type: 'checkbox',
      name: 'features',
      message: 'Select features:',
      choices: [
        { name: 'ESLint', value: 'eslint', checked: true },
        { name: 'Prettier', value: 'prettier', checked: true },
        { name: 'Husky (Git hooks)', value: 'husky' },
        { name: 'Jest (Testing)', value: 'jest' }
      ]
    }
  ]);
  
  // 步骤4：创建项目
  Logger.section(`Creating project: ${projectName}`);
  
  // 创建目录
  const spinner = ora('Creating project directory...').start();
  try {
    await fs.ensureDir(projectName);
    spinner.succeed('Project directory created');
  } catch (error) {
    spinner.fail('Failed to create directory');
    throw error;
  }
  
  // 复制模板
  spinner.start('Copying template files...');
  try {
    const templatePath = path.join(__dirname, '../../templates', template);
    await fs.copy(templatePath, projectName, {
      filter: (src) => !src.includes('node_modules')
    });
    spinner.succeed('Template files copied');
  } catch (error) {
    spinner.fail('Failed to copy template');
    throw error;
  }
  
  // 生成 package.json
  spinner.start('Generating package.json...');
  const packageJson = {
    name: projectName,
    version: '1.0.0',
    description: `${projectName} - Created by create-modern-app`,
    scripts: {
      dev: 'vite',
      build: 'vite build',
      preview: 'vite preview',
      ...(projectConfig.features.includes('eslint') && { lint: 'eslint .' }),
      ...(projectConfig.features.includes('jest') && { test: 'jest' })
    },
    dependencies: {},
    devDependencies: {}
  };
  
  await fs.writeJson(
    path.join(projectName, 'package.json'),
    packageJson,
    { spaces: 2 }
  );
  spinner.succeed('package.json generated');
  
  // 生成 README
  spinner.start('Generating README.md...');
  const readme = `# ${projectName}

Created with create-modern-app

## Getting Started

\`\`\`bash
cd ${projectName}
npm install
npm run dev
\`\`\`

## Build

\`\`\`bash
npm run build
\`\`\`
`;
  
  await fs.writeFile(path.join(projectName, 'README.md'), readme);
  spinner.succeed('README.md generated');
  
  // Git 初始化
  if (!options.skipGit) {
    spinner.start('Initializing git repository...');
    try {
      await initGit(projectName);
      spinner.succeed('Git repository initialized');
    } catch (error) {
      spinner.warn('Failed to initialize git (skipped)');
    }
  }
  
  // 安装依赖
  if (!options.skipInstall) {
    const packageManager = config.get('packageManager') || 'npm';
    
    spinner.start(`Installing dependencies with ${packageManager}...`);
    try {
      await installDependencies(projectName, packageManager);
      spinner.succeed('Dependencies installed');
    } catch (error) {
      spinner.warn('Failed to install dependencies');
      Logger.warning('You can install them manually later');
    }
  }
  
  // 完成
  Logger.success('\n✨ Project created successfully!\n');
  
  // 显示后续步骤
  console.log(chalk.cyan('Next steps:\n'));
  console.log(chalk.white(`  cd ${projectName}`));
  if (options.skipInstall) {
    console.log(chalk.white('  npm install'));
  }
  console.log(chalk.white('  npm run dev\n'));
  
  // 保存到最近项目
  config.addRecentProject(path.resolve(projectName));
}
```

### 4. 工具函数 (lib/utils/logger.js)

```javascript
import chalk from 'chalk';

export class Logger {
  static success(message) {
    console.log(chalk.green('✔'), message);
  }
  
  static error(message) {
    console.error(chalk.red('✖'), message);
  }
  
  static warning(message) {
    console.warn(chalk.yellow('⚠'), message);
  }
  
  static info(message) {
    console.log(chalk.blue('ℹ'), message);
  }
  
  static section(title) {
    console.log('\n' + chalk.bold.cyan(title));
    console.log(chalk.gray('─'.repeat(50)));
  }
  
  static step(number, message) {
    console.log(chalk.cyan(`[${number}]`), message);
  }
}
```

### 5. 错误处理 (lib/errors.js)

```javascript
import chalk from 'chalk';
import boxen from 'boxen';

export class CLIError extends Error {
  constructor(message, code, suggestions = []) {
    super(message);
    this.name = 'CLIError';
    this.code = code;
    this.suggestions = suggestions;
  }
}

export class ErrorHandler {
  static handle(error) {
    console.error();
    
    if (error instanceof CLIError) {
      console.error(chalk.red.bold(`❌ ${error.message}`));
      console.error();
      
      if (error.suggestions.length > 0) {
        console.error(chalk.yellow('💡 Suggestions:'));
        error.suggestions.forEach((s, i) => {
          console.error(chalk.white(`  ${i + 1}. ${s}`));
        });
        console.error();
      }
      
      console.error(chalk.gray(`Error code: ${error.code}`));
    } else {
      console.error(boxen(
        chalk.red.bold('Unexpected Error\n\n') +
        chalk.white(error.message),
        {
          padding: 1,
          borderColor: 'red',
          borderStyle: 'round'
        }
      ));
    }
    
    console.error();
    process.exit(1);
  }
}
```

### 6. 配置管理 (lib/config-manager.js)

```javascript
import Conf from 'conf';

export class ConfigManager {
  constructor() {
    this.config = new Conf({
      projectName: 'create-modern-app',
      defaults: {
        defaultTemplate: 'react',
        packageManager: 'npm',
        recentProjects: []
      }
    });
  }
  
  get(key) {
    return this.config.get(key);
  }
  
  set(key, value) {
    this.config.set(key, value);
  }
  
  addRecentProject(projectPath) {
    const recent = this.config.get('recentProjects');
    const filtered = recent.filter(p => p !== projectPath);
    filtered.unshift(projectPath);
    const limited = filtered.slice(0, 10);
    this.config.set('recentProjects', limited);
  }
}
```

import { describe, it, expect, beforeEach, afterEach } from 'vitest';

import fs from 'fs-extra';
import path from 'path';

const CLI_PATH = path.join(__dirname, '../bin/cli.js');

      '--skip-git'
