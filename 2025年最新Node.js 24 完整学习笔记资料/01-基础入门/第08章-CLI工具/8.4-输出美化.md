# 8.4 CLI 输出美化

> 使用 Chalk、Ora、Boxen 等工具美化命令行输出

---

## 🎯 为什么需要美化输出？

### 核心概念

```
美化输出 = 让CLI更专业、易用、美观

对比：

普通输出（❌）:
Success: File created
Error: File not found
Processing...

美化输出（✅）:
✔ Success: File created (绿色)
✖ Error: File not found (红色)
⠋ Processing... (旋转动画)

好处：
✅ 视觉吸引力
✅ 信息层次清晰
✅ 更好的用户体验
✅ 专业的外观
```

## 🎨 Chalk - 彩色输出

### 1. 安装与基础使用

```bash
# ===== 安装 =====
npm install chalk

# ⚠️ 注意：Chalk v5 只支持 ESM
# 如果使用 CommonJS，安装 v4
npm install chalk@4
```

```javascript
// ===== 基础用法（ESM）=====

import chalk from 'chalk';

// 基础颜色
console.log(chalk.red('Error: Something went wrong'));
console.log(chalk.green('✔ Success!'));
console.log(chalk.yellow('⚠ Warning'));
console.log(chalk.blue('ℹ Information'));
console.log(chalk.cyan('→ Processing'));
console.log(chalk.magenta('Debug info'));
console.log(chalk.gray('Verbose output'));

// 背景颜色
console.log(chalk.bgRed('Error'));
console.log(chalk.bgGreen('Success'));

// 样式
console.log(chalk.bold('Bold text'));
console.log(chalk.italic('Italic text'));
console.log(chalk.underline('Underlined text'));
console.log(chalk.dim('Dimmed text'));

// 组合使用
console.log(chalk.bold.green('✔ Success!'));
console.log(chalk.bgRed.white(' ERROR '));
console.log(chalk.red.bold.underline('Critical error'));
```

### 2. 高级用法

```javascript
// ===== 模板字符串 =====

import chalk from 'chalk';

const name = 'Alice';
const age = 25;

// 方式1：普通拼接
console.log(chalk.blue('User:') + ' ' + chalk.yellow(name));

// 方式2：模板字符串
console.log(`${chalk.blue('User:')} ${chalk.yellow(name)}`);

// 方式3：模板标签（推荐）
console.log(chalk`
  {blue User:} {yellow ${name}}
  {blue Age:} {yellow ${age}}
`);

// ===== RGB 颜色 =====

// RGB
console.log(chalk.rgb(255, 136, 0)('Orange text'));

// Hex
console.log(chalk.hex('#FF8800')('Orange text'));

// ===== 真彩色（TrueColor）=====

for (let i = 0; i < 10; i++) {
  console.log(chalk.rgb(255, i * 25, 0)('█'));
}

// ===== 条件颜色 =====

const level = 'error';
const colorMap = {
  error: chalk.red,
  warning: chalk.yellow,
  info: chalk.blue,
  success: chalk.green
};

console.log(colorMap[level]('Message'));
```

### 3. 实战案例

```javascript
// ===== 日志工具类 =====

import chalk from 'chalk';

class Logger {
  static success(message) {
    console.log(chalk.green('✔'), message);
  }
  
  static error(message) {
    console.error(chalk.red('✖'), message);
  }
  
  static warning(message) {
    console.warn(chalk.yellow('⚠'), message);
  }
  
  static info(message) {
    console.log(chalk.blue('ℹ'), message);
  }
  
  static debug(message) {
    console.log(chalk.gray('🐛'), message);
  }
  
  static section(title) {
    console.log('\n' + chalk.bold.cyan(title));
    console.log(chalk.gray('─'.repeat(50)));
  }
  
  static step(number, message) {
    console.log(chalk.cyan(`[${number}]`), message);
  }
}

// 使用
Logger.success('File created');
Logger.error('File not found');
Logger.warning('Deprecated API');
Logger.info('Loading...');
Logger.debug('Debug information');

Logger.section('Build Process');
Logger.step(1, 'Cleaning dist folder');
Logger.step(2, 'Compiling TypeScript');
Logger.step(3, 'Bundling assets');
Logger.success('Build completed!');
```

## ⏳ Ora - 加载动画

### 1. 基础使用

```bash
# ===== 安装 =====
npm install ora
```

```javascript
// ===== 基础用法 =====

import ora from 'ora';

// 方式1：最简单
const spinner = ora('Loading...').start();

setTimeout(() => {
  spinner.succeed('Loaded!');
}, 2000);

// 方式2：更多配置
const spinner = ora({
  text: 'Loading...',
  spinner: 'dots',     // 动画样式
  color: 'cyan',       // 颜色
  indent: 2            // 缩进
}).start();

// 方式3：Promise
async function loadData() {
  const spinner = ora('Loading data...').start();
  
  try {
    const data = await fetchData();
    spinner.succeed('Data loaded');
    return data;
  } catch (error) {
    spinner.fail('Failed to load data');
    throw error;
  }
}
```

### 2. 状态控制

```javascript
// ===== 各种状态 =====

import ora from 'ora';

const spinner = ora('Processing...').start();

// 成功
spinner.succeed('Success!');
// ✔ Success!

// 失败
spinner.fail('Failed!');
// ✖ Failed!

// 警告
spinner.warn('Warning!');
// ⚠ Warning!

// 信息
spinner.info('Information');
// ℹ Information

// 停止（不显示图标）
spinner.stop();

// 清除（删除整行）
spinner.clear();

// ===== 修改文本 =====

const spinner = ora('Step 1...').start();

setTimeout(() => {
  spinner.text = 'Step 2...';
}, 1000);

setTimeout(() => {
  spinner.text = 'Step 3...';
}, 2000);

setTimeout(() => {
  spinner.succeed('All steps completed!');
}, 3000);
```

### 3. 自定义样式

```javascript
// ===== 内置动画样式 =====

import ora from 'ora';

// 点动画
ora({ spinner: 'dots' }).start();

// 线动画
ora({ spinner: 'line' }).start();

// 弧线动画
ora({ spinner: 'arc' }).start();

// 星星动画
ora({ spinner: 'star' }).start();

// 箭头动画
ora({ spinner: 'arrow' }).start();

// 💡 更多样式：https://github.com/sindresorhus/cli-spinners

// ===== 自定义动画 =====

const spinner = ora({
  text: 'Processing...',
  spinner: {
    interval: 80,
    frames: ['◰', '◳', '◲', '◱']
  }
}).start();
```

### 4. 实战案例

```javascript
// ===== 多步骤进度 =====

import ora from 'ora';
import chalk from 'chalk';

async function buildProject() {
  console.log(chalk.cyan('\n🚀 Building project...\n'));
  
  // 步骤1：清理
  const spinner1 = ora('Cleaning dist folder...').start();
  await sleep(1000);
  spinner1.succeed('Dist folder cleaned');
  
  // 步骤2：编译
  const spinner2 = ora('Compiling TypeScript...').start();
  await sleep(2000);
  spinner2.succeed('TypeScript compiled');
  
  // 步骤3：打包
  const spinner3 = ora('Bundling assets...').start();
  await sleep(1500);
  spinner3.succeed('Assets bundled');
  
  // 步骤4：优化
  const spinner4 = ora('Optimizing output...').start();
  await sleep(1000);
  spinner4.succeed('Output optimized');
  
  console.log(chalk.green('\n✨ Build completed!\n'));
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

buildProject();

// 输出：
// 🚀 Building project...
//
// ✔ Dist folder cleaned
// ✔ TypeScript compiled
// ✔ Assets bundled
// ✔ Output optimized
//
// ✨ Build completed!
```

## 📦 Boxen - 边框装饰

### 1. 基础使用

```bash
# ===== 安装 =====
npm install boxen
```

```javascript
// ===== 基础用法 =====

import boxen from 'boxen';

// 简单边框
console.log(boxen('Hello World!'));
// ┌─────────────┐
// │Hello World! │
// └─────────────┘

// 带标题
console.log(boxen('Message content', {
  title: 'Success',
  titleAlignment: 'center'
}));
// ┌─── Success ───┐
// │Message content│
// └───────────────┘

// 内边距
console.log(boxen('Padded text', {
  padding: 1
}));
// ┌─────────────┐
// │             │
// │ Padded text │
// │             │
// └─────────────┘

// 外边距
console.log(boxen('Margin text', {
  margin: 1
}));
//
//    ┌──────────────┐
//    │ Margin text  │
//    └──────────────┘
//
```

### 2. 样式配置

```javascript
// ===== 边框样式 =====

import boxen from 'boxen';
import chalk from 'chalk';

// 圆角
console.log(boxen('Round corners', {
  borderStyle: 'round'
}));
// ╭─────────────────╮
// │ Round corners   │
// ╰─────────────────╯

// 双线
console.log(boxen('Double border', {
  borderStyle: 'double'
}));
// ╔═══════════════╗
// ║ Double border ║
// ╚═══════════════╝

// 粗线
console.log(boxen('Bold border', {
  borderStyle: 'bold'
}));
// ┏━━━━━━━━━━━━━┓
// ┃ Bold border ┃
// ┗━━━━━━━━━━━━━┛

// ===== 颜色 =====

console.log(boxen('Green border', {
  borderColor: 'green',
  borderStyle: 'round'
}));

console.log(boxen(chalk.green('Success message'), {
  borderColor: 'green',
  borderStyle: 'round',
  padding: 1
}));

// ===== 自定义边框字符 =====

console.log(boxen('Custom border', {
  borderStyle: {
    topLeft: '+',
    topRight: '+',
    bottomLeft: '+',
    bottomRight: '+',
    horizontal: '-',
    vertical: '|'
  }
}));
// +----------------+
// | Custom border  |
// +----------------+
```

### 3. 实战案例

```javascript
// ===== 欢迎信息 =====

import boxen from 'boxen';
import chalk from 'chalk';

function showWelcome() {
  const message = `
${chalk.bold('My CLI Tool')}
${chalk.gray('Version 1.0.0')}

${chalk.cyan('Get started:')}
  ${chalk.white('$ my-cli create <project>')}
  ${chalk.white('$ my-cli build')}
  ${chalk.white('$ my-cli deploy')}

${chalk.yellow('Documentation:')}
  ${chalk.white('https://example.com/docs')}
  `.trim();
  
  console.log(boxen(message, {
    padding: 1,
    margin: 1,
    borderStyle: 'round',
    borderColor: 'cyan'
  }));
}

showWelcome();

// ===== 错误提示 =====

function showError(message) {
  console.log(boxen(chalk.red(message), {
    padding: 1,
    borderColor: 'red',
    borderStyle: 'round',
    title: 'ERROR',
    titleAlignment: 'center'
  }));
}

showError('File not found: config.json');

// ===== 成功提示 =====

function showSuccess(message) {
  console.log(boxen(chalk.green(message), {
    padding: 1,
    borderColor: 'green',
    borderStyle: 'round',
    title: 'SUCCESS',
    titleAlignment: 'center'
  }));
}

showSuccess('Project created successfully!');
```

## 📊 CLI-Table3 - 表格输出

### 1. 基础使用

```bash
# ===== 安装 =====
npm install cli-table3
```

```javascript
// ===== 基础表格 =====

import Table from 'cli-table3';

const table = new Table({
  head: ['Name', 'Age', 'City'],
  colWidths: [20, 10, 15]
});

table.push(
  ['Alice', 25, 'Beijing'],
  ['Bob', 30, 'Shanghai'],
  ['Charlie', 35, 'Shenzhen']
);

console.log(table.toString());

// 输出：
// ┌────────────────────┬──────────┬───────────────┐
// │ Name               │ Age      │ City          │
// ├────────────────────┼──────────┼───────────────┤
// │ Alice              │ 25       │ Beijing       │
// ├────────────────────┼──────────┼───────────────┤
// │ Bob                │ 30       │ Shanghai      │
// ├────────────────────┼──────────┼───────────────┤
// │ Charlie            │ 35       │ Shenzhen      │
// └────────────────────┴──────────┴───────────────┘

// ===== 带颜色的表格 =====

import chalk from 'chalk';

const table = new Table({
  head: [
    chalk.cyan('Name'),
    chalk.cyan('Status'),
    chalk.cyan('Progress')
  ]
});

table.push(
  ['Build', chalk.green('✔ Success'), '100%'],
  ['Test', chalk.yellow('⚠ Warning'), '80%'],
  ['Deploy', chalk.red('✖ Failed'), '0%']
);

console.log(table.toString());
```

## 🎨 Figlet - ASCII 艺术字

### 1. 基础使用

```bash
# ===== 安装 =====
npm install figlet
```

```javascript
// ===== 生成ASCII艺术字 =====

import figlet from 'figlet';
import { promisify } from 'util';

const figletAsync = promisify(figlet);

// 异步方式
const art = await figletAsync('My CLI');
console.log(art);

// 输出：
//  __  __          ____ _     ___
// |  \/  |_   _   / ___| |   |_ _|
// | |\/| | | | | | |   | |    | |
// | |  | | |_| | | |___| |___ | |
// |_|  |_|\__, |  \____|_____|___|
//         |___/

// ===== 不同字体 =====

const fonts = ['Standard', 'Big', 'Small', 'Banner'];

for (const font of fonts) {
  console.log(`\n=== ${font} ===\n`);
  const art = await figletAsync('CLI', { font });
  console.log(art);
}

// ===== 实战：CLI启动横幅 =====

import chalk from 'chalk';

async function showBanner() {
  const art = await figletAsync('My CLI Tool', {
    font: 'Standard'
  });
  
  console.log(chalk.cyan(art));
  console.log(chalk.gray('Version 1.0.0'));
  console.log(chalk.gray('─'.repeat(50)));
  console.log();
}

showBanner();
```
