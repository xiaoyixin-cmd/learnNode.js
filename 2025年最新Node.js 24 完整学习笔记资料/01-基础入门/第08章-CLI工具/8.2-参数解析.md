# 8.2 命令行参数解析

> 使用 Commander.js 处理复杂的命令行参数

---

## 🎯 为什么需要参数解析？

### 核心概念

```
命令行参数 = 用户输入的命令、选项、参数

示例：git commit -m "Initial commit"
      │   │      │  │
      │   │      │  └─ 选项值
      │   │      └─ 选项（-m 或 --message）
      │   └─ 子命令
      └─ 主命令

复杂度递增：
1. 简单：my-cli
2. 带参数：my-cli create app
3. 带选项：my-cli create app --template react
4. 多选项：my-cli create app -t react -i --verbose

手动解析 vs Commander：
手动：process.argv（麻烦、易出错）
Commander：优雅、强大、易维护
```

## 📦 process.argv 基础

### 1. 理解 process.argv

```javascript
// ===== 命令：node cli.js create app --force =====

console.log(process.argv);
// [
//   '/usr/local/bin/node',      // [0] Node.js 可执行文件路径
//   '/path/to/cli.js',           // [1] 当前脚本路径
//   'create',                     // [2] 第一个参数
//   'app',                        // [3] 第二个参数
//   '--force'                     // [4] 选项
// ]

// ===== 获取用户输入的参数 =====

const args = process.argv.slice(2);  // 跳过前两个
console.log(args);  // ['create', 'app', '--force']

// ===== 手动解析（不推荐）=====

const command = args[0];              // 'create'
const name = args[1];                 // 'app'
const isForce = args.includes('--force');  // true

// ❌ 问题：
// - 代码冗长
// - 难以处理复杂选项
// - 没有帮助信息
// - 没有类型验证
```

## 🚀 Commander.js 入门

### 1. 基础使用

```bash
# ===== 安装 =====
npm install commander
```

```javascript
// ===== 最简单的示例 =====

import { program } from 'commander';

program
  .name('my-cli')
  .description('My awesome CLI tool')
  .version('1.0.0');

program.parse();

// ===== 运行 =====

// $ my-cli --version
// 1.0.0

// $ my-cli --help
// Usage: my-cli [options]
// My awesome CLI tool
//
// Options:
//   -V, --version  output the version number
//   -h, --help     display help for command
```

### 2. 添加子命令

```javascript
// ===== 定义子命令 =====

import { program } from 'commander';

program
  .command('create <project-name>')  // <> 表示必需参数
  .description('Create a new project')
  .action((projectName) => {
    console.log(`Creating project: ${projectName}`);
  });

program
  .command('delete [project-name]')  // [] 表示可选参数
  .description('Delete a project')
  .action((projectName) => {
    if (!projectName) {
      console.log('Please specify a project name');
      return;
    }
    console.log(`Deleting project: ${projectName}`);
  });

program.parse();

// ===== 使用 =====

// $ my-cli create my-app
// Creating project: my-app

// $ my-cli delete
// Please specify a project name

// $ my-cli delete my-app
// Deleting project: my-app
```

### 3. 添加选项

```javascript
// ===== 各种类型的选项 =====

program
  .command('create <name>')
  .description('Create a new project')
  
  // 布尔选项（有或无）
  .option('-f, --force', 'Force overwrite')
  
  // 带值的选项
  .option('-t, --template <type>', 'Project template')
  
  // 带默认值的选项
  .option('-p, --port <number>', 'Server port', '3000')
  
  // 可选值的选项
  .option('-e, --env [environment]', 'Environment')
  
  // 可重复的选项
  .option('-i, --include <path...>', 'Include paths')
  
  .action((name, options) => {
    console.log('Project name:', name);
    console.log('Options:', options);
  });

program.parse();

// ===== 使用示例 =====

// $ my-cli create app
// Project name: app
// Options: {}

// $ my-cli create app --force
// Options: { force: true }

// $ my-cli create app --template react
// Options: { template: 'react' }

// $ my-cli create app --port 8080
// Options: { port: '8080' }

// $ my-cli create app --env production
// Options: { env: 'production' }

// $ my-cli create app --include src tests
// Options: { include: ['src', 'tests'] }
```

## 🎯 高级用法

### 1. 选项类型转换

```javascript
// ===== 自定义解析函数 =====

import { program } from 'commander';

function parseNumber(value) {
  const num = parseInt(value, 10);
  if (isNaN(num)) {
    throw new Error('Not a number');
  }
  return num;
}

function parseList(value) {
  return value.split(',').map(item => item.trim());
}

program
  .command('server')
  .option('-p, --port <number>', 'Server port', parseNumber, 3000)
  .option('-h, --host <host>', 'Server host', 'localhost')
  .option('--allowed <list>', 'Allowed IPs', parseList)
  .action((options) => {
    console.log('Port (number):', options.port, typeof options.port);
    console.log('Host:', options.host);
    console.log('Allowed IPs:', options.allowed);
  });

program.parse();

// ===== 使用 =====

// $ my-cli server --port 8080
// Port (number): 8080 number

// $ my-cli server --allowed 127.0.0.1,192.168.1.1
// Allowed IPs: ['127.0.0.1', '192.168.1.1']
```

### 2. 选项验证

```javascript
// ===== 验证选项值 =====

program
  .command('deploy')
  .option('-e, --env <environment>', 'Deployment environment')
  .action((options) => {
    // 验证环境
    const validEnvs = ['development', 'staging', 'production'];
    if (!validEnvs.includes(options.env)) {
      console.error(`Error: Invalid environment "${options.env}"`);
      console.error(`Valid environments: ${validEnvs.join(', ')}`);
      process.exit(1);
    }
    
    console.log(`Deploying to ${options.env}...`);
  });

// ===== 更好的方式：使用 choices（Commander v9+）=====

program
  .command('deploy')
  .addOption(
    new Option('-e, --env <environment>', 'Deployment environment')
      .choices(['development', 'staging', 'production'])
      .makeOptionMandatory()  // 必需选项
  )
  .action((options) => {
    console.log(`Deploying to ${options.env}...`);
  });

// $ my-cli deploy
// error: required option '-e, --env <environment>' not specified

// $ my-cli deploy --env test
// error: option '-e, --env <environment>' argument 'test' is invalid.
// Allowed choices are development, staging, production.
```

### 3. 命令别名

```javascript
// ===== 为命令设置别名 =====

program
  .command('install')
  .alias('i')  // 别名
  .description('Install dependencies')
  .action(() => {
    console.log('Installing...');
  });

program
  .command('remove')
  .aliases(['rm', 'uninstall', 'delete'])  // 多个别名
  .description('Remove a package')
  .action(() => {
    console.log('Removing...');
  });

// ===== 使用 =====

// $ my-cli install
// $ my-cli i          # 等同于 install

// $ my-cli remove
// $ my-cli rm         # 等同于 remove
// $ my-cli uninstall  # 等同于 remove
// $ my-cli delete     # 等同于 remove
```

### 4. 嵌套子命令

```javascript
// ===== 多层命令结构 =====

import { program, Command } from 'commander';

// 主命令
program
  .name('myapp')
  .description('My application CLI');

// 一级子命令：config
const configCmd = new Command('config')
  .description('Manage configuration');

// 二级子命令：config get
configCmd
  .command('get <key>')
  .description('Get a config value')
  .action((key) => {
    console.log(`Getting config: ${key}`);
  });

// 二级子命令：config set
configCmd
  .command('set <key> <value>')
  .description('Set a config value')
  .action((key, value) => {
    console.log(`Setting config: ${key} = ${value}`);
  });

// 添加到主命令
program.addCommand(configCmd);

program.parse();

// ===== 使用 =====

// $ myapp config get apiKey
// Getting config: apiKey

// $ myapp config set apiKey abc123
// Setting config: apiKey = abc123
```

## 🎯 实战案例

### 案例1：项目脚手架

```javascript
// ===== 完整的项目脚手架 CLI =====

import { program } from 'commander';
import chalk from 'chalk';

program
  .name('create-app')
  .description('Project scaffolding tool')
  .version('1.0.0');

// ===== create 命令 =====

program
  .command('create <project-name>')
  .description('Create a new project')
  .option('-t, --template <type>', 'Project template', 'default')
  .option('--skip-install', 'Skip npm install')
  .option('--skip-git', 'Skip git init')
  .option('-v, --verbose', 'Verbose output')
  .action(async (projectName, options) => {
    console.log(chalk.cyan('\n🚀 Creating project...\n'));
    
    console.log(chalk.white('Project name:'), chalk.yellow(projectName));
    console.log(chalk.white('Template:'), chalk.yellow(options.template));
    console.log(chalk.white('Skip install:'), chalk.yellow(options.skipInstall || false));
    console.log(chalk.white('Skip git:'), chalk.yellow(options.skipGit || false));
    
    if (options.verbose) {
      console.log(chalk.gray('\nDetailed options:'), options);
    }
    
    // TODO: 实际的项目创建逻辑
  });

// ===== list 命令 =====

program
  .command('list')
  .alias('ls')
  .description('List available templates')
  .option('-f, --format <type>', 'Output format', 'table')
  .action((options) => {
    const templates = ['react', 'vue', 'node', 'express'];
    
    if (options.format === 'json') {
      console.log(JSON.stringify(templates, null, 2));
    } else {
      console.log(chalk.cyan('\nAvailable templates:\n'));
      templates.forEach(t => console.log(chalk.yellow(`  • ${t}`)));
    }
  });

// ===== init 命令 =====

program
  .command('init')
  .description('Initialize a project in current directory')
  .option('-f, --force', 'Force initialization')
  .action((options) => {
    if (!options.force) {
      console.log(chalk.yellow('⚠️  Current directory is not empty'));
      console.log(chalk.white('Use --force to initialize anyway'));
      return;
    }
    
    console.log(chalk.green('✔ Initializing project...'));
  });

program.parse();
```

### 案例2：Git 风格的命令

```javascript
// ===== 模仿 Git 的命令结构 =====

import { program } from 'commander';

program
  .name('mygit')
  .usage('<command> [options]');

// ===== mygit init =====

program
  .command('init')
  .description('Initialize a repository')
  .option('--bare', 'Create a bare repository')
  .action((options) => {
    console.log('Initializing repository...');
    if (options.bare) {
      console.log('Creating bare repository');
    }
  });

// ===== mygit add =====

program
  .command('add <files...>')
  .description('Add files to staging area')
  .option('-A, --all', 'Add all files')
  .action((files, options) => {
    if (options.all) {
      console.log('Adding all files');
    } else {
      console.log('Adding files:', files.join(', '));
    }
  });

// ===== mygit commit =====

program
  .command('commit')
  .description('Commit changes')
  .option('-m, --message <message>', 'Commit message')
  .option('-a, --all', 'Commit all changes')
  .option('--amend', 'Amend previous commit')
  .action((options) => {
    if (!options.message) {
      console.error('Error: Commit message is required');
      process.exit(1);
    }
    
    console.log('Committing:', options.message);
    if (options.all) console.log('Including all changes');
    if (options.amend) console.log('Amending previous commit');
  });

// ===== mygit log =====

program
  .command('log')
  .description('Show commit logs')
  .option('-n, --number <count>', 'Number of commits', parseInt)
  .option('--oneline', 'Show one line per commit')
  .action((options) => {
    console.log('Showing commit logs');
    if (options.number) console.log(`Limit: ${options.number}`);
    if (options.oneline) console.log('Format: one line');
  });

program.parse();

// ===== 使用示例 =====

// $ mygit init
// $ mygit add file1.js file2.js
// $ mygit commit -m "Initial commit"
// $ mygit log --oneline -n 10
```

### 案例3：Docker 风格的命令

```javascript
// ===== 模仿 Docker 的命令结构 =====

import { program } from 'commander';

program
  .name('mydocker')
  .description('Container management tool');

// ===== mydocker run =====

program
  .command('run <image>')
  .description('Run a container')
  .option('-d, --detach', 'Run in background')
  .option('-p, --publish <ports>', 'Publish ports')
  .option('--name <name>', 'Container name')
  .option('-e, --env <vars...>', 'Environment variables')
  .action((image, options) => {
    console.log(`Running container from image: ${image}`);
    if (options.name) console.log(`Name: ${options.name}`);
    if (options.detach) console.log('Mode: Detached');
    if (options.publish) console.log(`Ports: ${options.publish}`);
    if (options.env) console.log(`Environment: ${options.env.join(', ')}`);
  });

// ===== mydocker ps =====

program
  .command('ps')
  .description('List containers')
  .option('-a, --all', 'Show all containers')
  .option('-q, --quiet', 'Only display IDs')
  .action((options) => {
    console.log('Listing containers...');
    if (options.all) console.log('Including stopped containers');
    if (options.quiet) console.log('Quiet mode');
  });

// ===== mydocker stop =====

program
  .command('stop <container>')
  .description('Stop a container')
  .option('-t, --time <seconds>', 'Seconds to wait', parseInt, 10)
  .action((container, options) => {
    console.log(`Stopping container: ${container}`);
    console.log(`Timeout: ${options.time}s`);
  });

program.parse();

// ===== 使用示例 =====

// $ mydocker run nginx -d --name web --publish 80:80
// $ mydocker ps --all
// $ mydocker stop web --time 30
```

## 💡 最佳实践

```javascript
// ✅ 1. 提供清晰的描述
program
  .command('deploy')
  .description('Deploy application to production server')  // 清晰明确

// ✅ 2. 使用长短选项
.option('-e, --env <type>', 'Environment')  // 短：-e，长：--env

// ✅ 3. 提供默认值
.option('-p, --port <number>', 'Server port', '3000')

// ✅ 4. 验证必需参数
program
  .command('deploy <target>')  // <> 表示必需
  .action((target) => {
    if (!target) {
      console.error('Target is required');
      process.exit(1);
    }
  });

// ✅ 5. 使用有意义的参数名
.command('create <project-name>')  // ✅ 清晰
// 而不是 .command('create <name>')  // ❌ 模糊

// ✅ 6. 提供示例
program
  .command('deploy')
  .description('Deploy application\nExample: my-cli deploy production')

// ✅ 7. 处理错误
.action((options) => {
  try {
    // ...
  } catch (error) {
    console.error('Error:', error.message);
    process.exit(1);
  }
});

// ✅ 8. 使用 --help 触发完整帮助
program.parse();
if (!process.argv.slice(2).length) {
  program.outputHelp();
}
```
