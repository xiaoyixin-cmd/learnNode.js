### 2. 编写单元测试

```javascript
// ===== lib/utils.js =====

export function validateProjectName(name) {
  if (!name || name.length === 0) {
    throw new Error('Project name is required');
  }
  
  if (!/^[a-z0-9-_]+$/.test(name)) {
    throw new Error('Invalid project name format');
  }
  
  return true;
}

export function formatBytes(bytes) {
  if (bytes === 0) return '0 B';
  
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  
  return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i];
}

// ===== tests/utils.test.js =====

import { describe, it, expect } from 'vitest';
import { validateProjectName, formatBytes } from '../lib/utils.js';

describe('validateProjectName', () => {
  it('should accept valid project names', () => {
    expect(validateProjectName('my-app')).toBe(true);
    expect(validateProjectName('my_app')).toBe(true);
    expect(validateProjectName('myapp123')).toBe(true);
  });
  
  it('should reject empty names', () => {
    expect(() => validateProjectName('')).toThrow('required');
    expect(() => validateProjectName(null)).toThrow('required');
  });
  
  it('should reject invalid characters', () => {
    expect(() => validateProjectName('My App')).toThrow('Invalid');
    expect(() => validateProjectName('MyApp')).toThrow('Invalid');
    expect(() => validateProjectName('my@app')).toThrow('Invalid');
  });
});

describe('formatBytes', () => {
  it('should format bytes correctly', () => {
    expect(formatBytes(0)).toBe('0 B');
    expect(formatBytes(1024)).toBe('1 KB');
    expect(formatBytes(1048576)).toBe('1 MB');
    expect(formatBytes(1073741824)).toBe('1 GB');
  });
  
  it('should handle decimals', () => {
    expect(formatBytes(1536)).toBe('1.5 KB');
    expect(formatBytes(2621440)).toBe('2.5 MB');
  });
});
```

### 3. 测试CLI命令

```javascript
// ===== tests/cli.test.js =====

import { describe, it, expect, vi } from 'vitest';
import { execa } from 'execa';
import fs from 'fs-extra';
import path from 'path';

const CLI_PATH = path.join(__dirname, '../bin/cli.js');

describe('CLI Integration Tests', () => {
  it('should show version', async () => {
    const { stdout } = await execa('node', [CLI_PATH, '--version']);
    expect(stdout).toMatch(/\d+\.\d+\.\d+/);
  });
  
  it('should show help', async () => {
    const { stdout } = await execa('node', [CLI_PATH, '--help']);
    expect(stdout).toContain('Usage:');
    expect(stdout).toContain('Commands:');
  });
  
  it('should create a project', async () => {
    const projectName = 'test-project-' + Date.now();
    
    try {
      await execa('node', [CLI_PATH, 'create', projectName]);
      
      // 验证项目目录已创建
      expect(fs.existsSync(projectName)).toBe(true);
      
      // 验证 package.json 已创建
      const pkgPath = path.join(projectName, 'package.json');
      expect(fs.existsSync(pkgPath)).toBe(true);
      
      // 验证 package.json 内容
      const pkg = await fs.readJson(pkgPath);
      expect(pkg.name).toBe(projectName);
      
    } finally {
      // 清理测试项目
      await fs.remove(projectName);
    }
  });
  
  it('should reject invalid project name', async () => {
    try {
      await execa('node', [CLI_PATH, 'create', 'Invalid Name']);
      // 不应该执行到这里
      expect(true).toBe(false);
    } catch (error) {
      expect(error.exitCode).toBe(1);
      expect(error.stderr).toContain('Invalid');
    }
  });
});
```

### 4. Mock 和 Spy

```javascript
// ===== 测试用户输入 =====

import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import inquirer from 'inquirer';

describe('Interactive Input', () => {
  beforeEach(() => {
    // Mock inquirer
    vi.mock('inquirer');
  });
  
  afterEach(() => {
    vi.restoreAllMocks();
  });
  
  it('should collect user input', async () => {
    // Mock 用户输入
    inquirer.prompt = vi.fn().mockResolvedValue({
      projectName: 'my-app',
      template: 'react'
    });
    
    const { createProject } = await import('../lib/create.js');
    await createProject();
    
    expect(inquirer.prompt).toHaveBeenCalled();
  });
});

// ===== 测试文件操作 =====

import { vi } from 'vitest';
import fs from 'fs-extra';

vi.mock('fs-extra');

describe('File Operations', () => {
  it('should read config file', async () => {
    // Mock fs.readJson
    fs.readJson = vi.fn().mockResolvedValue({
      name: 'my-app',
      version: '1.0.0'
    });
    
    const { readConfig } = await import('../lib/config.js');
    const config = await readConfig();
    
    expect(config.name).toBe('my-app');
    expect(fs.readJson).toHaveBeenCalledWith('config.json');
  });
});
```

### 2. 发布流程

```bash
# ===== 1. 测试本地安装 =====

# 创建全局链接
npm link

# 测试命令
my-cli --version
my-cli --help
my-cli create test-app

# 取消链接
npm unlink -g my-cli-tool

# ===== 2. 构建（如果需要）=====

npm run build

# ===== 3. 检查发布文件 =====

npm pack --dry-run

# 输出：
# package: my-cli-tool@1.0.0
# Tarball Contents
#   156B package.json
#   1.2kB bin/cli.js
#   2.5kB lib/create.js
#   ...

# ===== 4. 发布到 npm =====

npm publish

# 如果是作用域包（@username/package）
npm publish --access public

# ===== 5. 验证发布 =====

# 查看包信息
npm view my-cli-tool

# 安装测试
npm install -g my-cli-tool
my-cli --version
```

### 3. 更新版本

```bash
# ===== 语义化版本 =====

# 修复 bug: 1.0.0 → 1.0.1
npm version patch

# 新增功能: 1.0.1 → 1.1.0
npm version minor

# 重大变更: 1.1.0 → 2.0.0
npm version major

# ===== 手动指定版本 =====

npm version 1.2.3

# ===== 发布新版本 =====

# 1. 更新版本号
npm version patch

# 2. 提交到 Git
git push --tags

# 3. 发布到 npm
npm publish
```

## 🎯 CI/CD 自动发布

### GitHub Actions

```yaml
# ===== .github/workflows/publish.yml =====

name: Publish to npm

on:
  push:
    tags:
      - 'v*'  # 当推送 v* 标签时触发

jobs:
  publish:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          registry-url: 'https://registry.npmjs.org'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run tests
        run: npm test
      
      - name: Build
        run: npm run build
      
      - name: Publish to npm
        run: npm publish
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}
```
