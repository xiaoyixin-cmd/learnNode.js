# 8.7 CLI 配置管理

> 使用 Conf 和 dotenv 管理用户配置

---

## 🎯 为什么需要配置管理？

### 核心概念

```
配置管理 = 持久化保存CLI工具的设置

配置类型：
1. 全局配置 - 所有项目共享（如用户信息、API密钥）
2. 项目配置 - 特定项目的设置（如构建选项）
3. 环境变量 - 运行时配置（如端口、数据库地址）

配置存储位置：
• Linux/Mac: ~/.config/app-name/
• Windows: %APPDATA%/app-name/
• 项目根目录: .config.json, .env

常用场景：
✅ 用户身份信息（token、用户名）
✅ API密钥和密码
✅ 默认选项（模板、包管理器）
✅ 历史记录（最近使用的项目）
```

## 📦 Conf - 配置存储

### 1. 安装与基础使用

```bash
# ===== 安装 =====
npm install conf
```

```javascript
// ===== 基础用法 =====

import Conf from 'conf';

// 创建配置实例
const config = new Conf({
  projectName: 'my-cli',  // 配置文件名
  defaults: {             // 默认值
    username: '',
    theme: 'dark',
    recentProjects: []
  }
});

// ===== 读取配置 =====

// 获取单个值
const username = config.get('username');
console.log('Username:', username);

// 获取多个值
const theme = config.get('theme');
const projects = config.get('recentProjects');

// 获取全部配置
const allConfig = config.store;
console.log('All config:', allConfig);

// 获取值（带默认值）
const port = config.get('port', 3000);

// ===== 写入配置 =====

// 设置单个值
config.set('username', 'Alice');

// 设置多个值
config.set({
  username: 'Alice',
  email: 'alice@example.com',
  theme: 'light'
});

// 删除值
config.delete('theme');

// 清空所有配置
config.clear();

// ===== 检查键是否存在 =====

if (config.has('username')) {
  console.log('Username is set');
}

// ===== 获取配置文件路径 =====

console.log('Config file:', config.path);
// 输出：/Users/alice/.config/my-cli/config.json
```

### 2. 高级用法

```javascript
// ===== 加密敏感信息 =====

import Conf from 'conf';

const config = new Conf({
  projectName: 'my-cli',
  encryptionKey: 'my-secret-key',  // 加密密钥
  defaults: {
    apiKey: '',
    token: ''
  }
});

// 设置加密的值
config.set('apiKey', 'sk-abc123...');  // 自动加密
config.set('token', 'token-xyz...');

// 读取时自动解密
const apiKey = config.get('apiKey');

// ===== 监听配置变化 =====

config.onDidChange('theme', (newValue, oldValue) => {
  console.log(`Theme changed from ${oldValue} to ${newValue}`);
});

config.set('theme', 'dark');
// 输出：Theme changed from light to dark

// ===== 使用点号访问嵌套属性 =====

config.set('user.profile.name', 'Alice');
config.set('user.profile.age', 25);

console.log(config.get('user.profile.name'));  // Alice
console.log(config.get('user'));
// {
//   profile: {
//     name: 'Alice',
//     age: 25
//   }
// }

// ===== Schema验证 =====

const config = new Conf({
  projectName: 'my-cli',
  schema: {
    username: {
      type: 'string',
      minLength: 3
    },
    port: {
      type: 'number',
      minimum: 1,
      maximum: 65535
    },
    theme: {
      type: 'string',
      enum: ['light', 'dark']
    }
  }
});

// 有效值
config.set('username', 'Alice');  // ✅

// 无效值（会抛出错误）
config.set('username', 'Al');  // ❌ 太短
config.set('port', 99999);     // ❌ 超出范围
config.set('theme', 'blue');   // ❌ 不在枚举中
```

## 🔐 dotenv - 环境变量

### 1. 基础使用

```bash
# ===== 安装 =====
npm install dotenv
```

```javascript
// ===== .env 文件 =====

// .env
NODE_ENV=development
PORT=3000
DATABASE_URL=mongodb://localhost:27017/myapp
API_KEY=sk-abc123xyz
SECRET=super-secret-key

// ===== 加载环境变量 =====

import dotenv from 'dotenv';

// 加载 .env 文件
dotenv.config();

// 使用环境变量
console.log('Environment:', process.env.NODE_ENV);
console.log('Port:', process.env.PORT);
console.log('Database:', process.env.DATABASE_URL);
console.log('API Key:', process.env.API_KEY);

// ===== 指定 .env 文件路径 =====

dotenv.config({ path: './config/.env' });

// ===== 多环境配置 =====

// .env.development
NODE_ENV=development
API_URL=http://localhost:3000

// .env.production
NODE_ENV=production
API_URL=https://api.example.com

// 加载对应环境的配置
const envFile = `.env.${process.env.NODE_ENV || 'development'}`;
dotenv.config({ path: envFile });
```

### 2. 最佳实践

```javascript
// ===== 不要将 .env 提交到 Git =====

// .gitignore
.env
.env.local
.env.*.local

// ===== 提供 .env.example 模板 =====

// .env.example
NODE_ENV=development
PORT=3000
DATABASE_URL=mongodb://localhost:27017/myapp
API_KEY=your-api-key-here
SECRET=your-secret-here

// ===== 验证必需的环境变量 =====

import dotenv from 'dotenv';

dotenv.config();

const requiredEnvVars = ['DATABASE_URL', 'API_KEY', 'SECRET'];

for (const envVar of requiredEnvVars) {
  if (!process.env[envVar]) {
    console.error(`Error: ${envVar} is not set in .env file`);
    process.exit(1);
  }
}

// ===== 类型转换助手 =====

function getEnv(key, defaultValue) {
  return process.env[key] || defaultValue;
}

function getEnvNumber(key, defaultValue) {
  const value = process.env[key];
  return value ? parseInt(value, 10) : defaultValue;
}

function getEnvBool(key, defaultValue) {
  const value = process.env[key];
  if (value === 'true') return true;
  if (value === 'false') return false;
  return defaultValue;
}

// 使用
const port = getEnvNumber('PORT', 3000);
const debug = getEnvBool('DEBUG', false);
const host = getEnv('HOST', 'localhost');
```

## 🎯 实战案例

### 案例1：用户配置管理

```javascript
// ===== 完整的用户配置系统 =====

import Conf from 'conf';
import inquirer from 'inquirer';
import chalk from 'chalk';

class ConfigManager {
  constructor() {
    this.config = new Conf({
      projectName: 'my-cli',
      defaults: {
        username: '',
        email: '',
        defaultTemplate: 'react',
        packageManager: 'npm',
        recentProjects: []
      }
    });
  }
  
  // 获取配置
  get(key) {
    return this.config.get(key);
  }
  
  // 设置配置
  set(key, value) {
    this.config.set(key, value);
  }
  
  // 显示所有配置
  show() {
    console.log(chalk.cyan('\n📋 Current Configuration:\n'));
    
    const config = this.config.store;
    for (const [key, value] of Object.entries(config)) {
      console.log(chalk.white(`${key}:`), chalk.yellow(JSON.stringify(value)));
    }
    
    console.log(chalk.gray(`\nConfig file: ${this.config.path}\n`));
  }
  
  // 交互式设置
  async setup() {
    console.log(chalk.cyan('\n⚙️  Configuration Setup\n'));
    
    const answers = await inquirer.prompt([
      {
        type: 'input',
        name: 'username',
        message: 'Your name:',
        default: this.config.get('username')
      },
      {
        type: 'input',
        name: 'email',
        message: 'Your email:',
        default: this.config.get('email'),
        validate: (input) => {
          const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
          return emailRegex.test(input) || 'Please enter a valid email';
        }
      },
      {
        type: 'list',
        name: 'defaultTemplate',
        message: 'Default template:',
        choices: ['react', 'vue', 'node', 'express'],
        default: this.config.get('defaultTemplate')
      },
      {
        type: 'list',
        name: 'packageManager',
        message: 'Package manager:',
        choices: ['npm', 'yarn', 'pnpm'],
        default: this.config.get('packageManager')
      }
    ]);
    
    // 保存配置
    this.config.set(answers);
    
    console.log(chalk.green('\n✔ Configuration saved!\n'));
  }
  
  // 重置配置
  async reset() {
    const { confirm } = await inquirer.prompt([
      {
        type: 'confirm',
        name: 'confirm',
        message: 'Reset all configuration to defaults?',
        default: false
      }
    ]);
    
    if (confirm) {
      this.config.clear();
      console.log(chalk.green('✔ Configuration reset'));
    }
  }
  
  // 添加最近项目
  addRecentProject(projectPath) {
    const recent = this.config.get('recentProjects');
    
    // 移除重复项
    const filtered = recent.filter(p => p !== projectPath);
    
    // 添加到开头
    filtered.unshift(projectPath);
    
    // 只保留最近10个
    const limited = filtered.slice(0, 10);
    
    this.config.set('recentProjects', limited);
  }
  
  // 获取最近项目
  getRecentProjects() {
    return this.config.get('recentProjects');
  }
}

// ===== 使用 =====

const configManager = new ConfigManager();

// CLI 命令
program
  .command('config')
  .description('Manage configuration')
  .action(async () => {
    const { action } = await inquirer.prompt([
      {
        type: 'list',
        name: 'action',
        message: 'What do you want to do?',
        choices: ['Show config', 'Setup config', 'Reset config']
      }
    ]);
    
    if (action === 'Show config') {
      configManager.show();
    } else if (action === 'Setup config') {
      await configManager.setup();
    } else if (action === 'Reset config') {
      await configManager.reset();
    }
  });
```

### 案例2：项目配置

```javascript
// ===== 项目级配置 =====

import fs from 'fs-extra';
import path from 'path';

class ProjectConfig {
  constructor() {
    this.configFile = '.myrc.json';
    this.config = this.load();
  }
  
  // 加载配置
  load() {
    try {
      if (fs.existsSync(this.configFile)) {
        return fs.readJsonSync(this.configFile);
      }
      return this.getDefaults();
    } catch (error) {
      console.error('Failed to load config:', error.message);
      return this.getDefaults();
    }
  }
  
  // 保存配置
  save() {
    try {
      fs.writeJsonSync(this.configFile, this.config, { spaces: 2 });
      return true;
    } catch (error) {
      console.error('Failed to save config:', error.message);
      return false;
    }
  }
  
  // 默认配置
  getDefaults() {
    return {
      build: {
        outDir: 'dist',
        sourceMap: true,
        minify: true
      },
      dev: {
        port: 3000,
        host: 'localhost',
        open: true
      },
      plugins: []
    };
  }
  
  // 获取配置值
  get(key) {
    const keys = key.split('.');
    let value = this.config;
    
    for (const k of keys) {
      value = value?.[k];
      if (value === undefined) break;
    }
    
    return value;
  }
  
  // 设置配置值
  set(key, value) {
    const keys = key.split('.');
    const lastKey = keys.pop();
    
    let obj = this.config;
    for (const k of keys) {
      if (!(k in obj)) obj[k] = {};
      obj = obj[k];
    }
    
    obj[lastKey] = value;
    this.save();
  }
  
  // 初始化项目配置
  async init() {
    if (fs.existsSync(this.configFile)) {
      const { overwrite } = await inquirer.prompt([
        {
          type: 'confirm',
          name: 'overwrite',
          message: 'Config file exists. Overwrite?',
          default: false
        }
      ]);
      
      if (!overwrite) return;
    }
    
    const answers = await inquirer.prompt([
      {
        type: 'input',
        name: 'outDir',
        message: 'Output directory:',
        default: 'dist'
      },
      {
        type: 'number',
        name: 'port',
        message: 'Dev server port:',
        default: 3000
      },
      {
        type: 'confirm',
        name: 'sourceMap',
        message: 'Generate source maps?',
        default: true
      }
    ]);
    
    this.config = {
      ...this.getDefaults(),
      build: {
        ...this.getDefaults().build,
        outDir: answers.outDir,
        sourceMap: answers.sourceMap
      },
      dev: {
        ...this.getDefaults().dev,
        port: answers.port
      }
    };
    
    this.save();
    console.log(chalk.green(`✔ Config saved to ${this.configFile}`));
  }
}

// ===== 使用 =====

const projectConfig = new ProjectConfig();

// 读取配置
const outDir = projectConfig.get('build.outDir');
const port = projectConfig.get('dev.port');

// 修改配置
projectConfig.set('build.minify', false);
projectConfig.set('dev.open', false);
```
