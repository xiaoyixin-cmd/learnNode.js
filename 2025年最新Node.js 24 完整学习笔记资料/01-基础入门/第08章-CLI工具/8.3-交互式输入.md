# 8.3 交互式输入

> 使用 Inquirer.js 创建友好的命令行交互

---

## 🎯 为什么需要交互式输入？

### 核心概念

```
交互式输入 = 在CLI中与用户问答式交互

传统方式 vs 交互式：

传统方式（参数）：
$ my-cli create app --template react --install --git

❌ 问题：
- 参数多了记不住
- 容易输错
- 不友好

交互式方式：
$ my-cli create
? Project name: app
? Select template: (Use arrow keys)
  ❯ React
    Vue
    Node.js
? Install dependencies? (Y/n) Yes
? Initialize git? (Y/n) Yes

✅ 优势：
- 引导式输入
- 实时验证
- 用户友好
- 减少错误
```

## 📦 Inquirer.js 入门

### 1. 安装

```bash
# ===== 安装 Inquirer =====

npm install inquirer

# ⚠️ 注意版本：
# Inquirer v9+只支持 ESM
# 如果用 CommonJS，安装 v8
npm install inquirer@^8
```

### 2. 基础使用

```javascript
// ===== 最简单的示例 =====

import inquirer from 'inquirer';

const answers = await inquirer.prompt([
  {
    type: 'input',        // 问题类型
    name: 'username',     // 答案的键名
    message: 'What is your name?',  // 提示信息
    default: 'Anonymous'  // 默认值
  }
]);

console.log('Hello,', answers.username);

// ===== 运行效果 =====

// ? What is your name? (Anonymous) Alice
// Hello, Alice
```

## 🎨 问题类型

### 1. input - 文本输入

```javascript
// ===== 基础文本输入 =====

const answers = await inquirer.prompt([
  {
    type: 'input',
    name: 'projectName',
    message: 'Project name:',
    default: 'my-app'
  },
  {
    type: 'input',
    name: 'author',
    message: 'Author:',
    default: () => {
      // 动态默认值
      return process.env.USER || 'Unknown';
    }
  }
]);

console.log(answers);
// { projectName: 'my-app', author: 'alice' }

// ===== 带验证的输入 =====

const answers = await inquirer.prompt([
  {
    type: 'input',
    name: 'email',
    message: 'Email address:',
    validate: (input) => {
      // 验证邮箱格式
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (emailRegex.test(input)) {
        return true;  // 验证通过
      }
      return 'Please enter a valid email address';  // 验证失败
    }
  },
  {
    type: 'input',
    name: 'port',
    message: 'Server port:',
    default: '3000',
    validate: (input) => {
      const port = parseInt(input);
      if (isNaN(port) || port < 1 || port > 65535) {
        return 'Please enter a valid port number (1-65535)';
      }
      return true;
    },
    filter: (input) => {
      // 过滤/转换输入
      return parseInt(input);
    }
  }
]);

// ===== 带转换的输入 =====

const answers = await inquirer.prompt([
  {
    type: 'input',
    name: 'tags',
    message: 'Tags (comma-separated):',
    filter: (input) => {
      // 转换为数组
      return input.split(',').map(tag => tag.trim());
    }
  }
]);

console.log(answers.tags);  // ['nodejs', 'cli', 'tool']
```

### 5. password - 密码输入

```javascript
// ===== 隐藏输入内容 =====

const answers = await inquirer.prompt([
  {
    type: 'password',
    name: 'password',
    message: 'Enter password:',
    mask: '*',  // 显示 * 代替字符
    validate: (input) => {
      if (input.length < 6) {
        return 'Password must be at least 6 characters';
      }
      return true;
    }
  },
  {
    type: 'password',
    name: 'confirmPassword',
    message: 'Confirm password:',
    mask: '*',
    validate: (input, answers) => {
      // 验证两次密码是否一致
      if (input !== answers.password) {
        return 'Passwords do not match';
      }
      return true;
    }
  }
]);

// ===== 使用效果 =====

// ? Enter password: ******
// ? Confirm password: ******
```

## 🎯 高级技巧

### 1. 条件问题（when）

```javascript
// ===== 根据前面的答案决定是否显示问题 =====

const answers = await inquirer.prompt([
  {
    type: 'confirm',
    name: 'useTypeScript',
    message: 'Use TypeScript?',
    default: false
  },
  {
    type: 'list',
    name: 'tsConfig',
    message: 'TypeScript configuration:',
    choices: ['Strict', 'Recommended', 'Loose'],
    when: (answers) => {
      // 只有选择了 TypeScript 才显示此问题
      return answers.useTypeScript;
    }
  },
  {
    type: 'confirm',
    name: 'installDeps',
    message: 'Install dependencies?',
    default: true
  },
  {
    type: 'list',
    name: 'packageManager',
    message: 'Package manager:',
    choices: ['npm', 'yarn', 'pnpm'],
    when: (answers) => answers.installDeps
  }
]);
```

### 2. 动态选项（choices 函数）

```javascript
// ===== 根据前面的答案动态生成选项 =====

const answers = await inquirer.prompt([
  {
    type: 'list',
    name: 'framework',
    message: 'Select framework:',
    choices: ['React', 'Vue', 'Svelte']
  },
  {
    type: 'list',
    name: 'buildTool',
    message: 'Select build tool:',
    choices: (answers) => {
      // 根据框架选择不同的构建工具
      if (answers.framework === 'React') {
        return ['Create React App', 'Vite', 'Next.js'];
      } else if (answers.framework === 'Vue') {
        return ['Vue CLI', 'Vite', 'Nuxt.js'];
      } else {
        return ['Vite', 'SvelteKit'];
      }
    }
  }
]);
```

### 3. 循环问题

```javascript
// ===== 重复添加多个项目 =====

async function addDependencies() {
  const dependencies = [];
  let addMore = true;
  
  while (addMore) {
    const answers = await inquirer.prompt([
      {
        type: 'input',
        name: 'package',
        message: 'Package name:',
        validate: (input) => {
          if (!input) return 'Package name is required';
          return true;
        }
      },
      {
        type: 'input',
        name: 'version',
        message: 'Version:',
        default: 'latest'
      },
      {
        type: 'confirm',
        name: 'addAnother',
        message: 'Add another package?',
        default: false
      }
    ]);
    
    dependencies.push({
      package: answers.package,
      version: answers.version
    });
    
    addMore = answers.addAnother;
  }
  
  return dependencies;
}

const deps = await addDependencies();
console.log('Dependencies:', deps);
```

## 🎯 实战案例

### 案例1：完整的项目生成器

```javascript
// ===== 交互式项目生成器 =====

import inquirer from 'inquirer';
import chalk from 'chalk';
import ora from 'ora';
import { execSync } from 'child_process';
import { mkdirSync, writeFileSync } from 'fs';

async function createProject() {
  console.log(chalk.cyan('\n🚀 Project Generator\n'));
  
  // 第一步：项目基本信息
  const basicInfo = await inquirer.prompt([
    {
      type: 'input',
      name: 'projectName',
      message: 'Project name:',
      default: 'my-app',
      validate: (input) => {
        if (!/^[a-z0-9-_]+$/.test(input)) {
          return 'Project name can only contain lowercase letters, numbers, hyphens and underscores';
        }
        return true;
      }
    },
    {
      type: 'input',
      name: 'description',
      message: 'Project description:',
      default: 'A new project'
    },
    {
      type: 'input',
      name: 'author',
      message: 'Author:',
      default: process.env.USER || ''
    }
  ]);
  
  // 第二步：技术栈选择
  const techStack = await inquirer.prompt([
    {
      type: 'list',
      name: 'framework',
      message: 'Select framework:',
      choices: [
        { name: 'React', value: 'react' },
        { name: 'Vue 3', value: 'vue3' },
        { name: 'Svelte', value: 'svelte' },
        { name: 'Vanilla JS', value: 'vanilla' }
      ]
    },
    {
      type: 'confirm',
      name: 'useTypeScript',
      message: 'Use TypeScript?',
      default: true
    },
    {
      type: 'list',
      name: 'cssPreprocessor',
      message: 'CSS preprocessor:',
      choices: ['None', 'Sass', 'Less', 'Stylus']
    }
  ]);
  
  // 第三步：工具链配置
  const tooling = await inquirer.prompt([
    {
      type: 'checkbox',
      name: 'features',
      message: 'Select features:',
      choices: [
        { name: 'ESLint', value: 'eslint', checked: true },
        { name: 'Prettier', value: 'prettier', checked: true },
        { name: 'Husky (Git hooks)', value: 'husky' },
        { name: 'Jest (Testing)', value: 'jest' },
        { name: 'Cypress (E2E Testing)', value: 'cypress' }
      ]
    },
    {
      type: 'list',
      name: 'packageManager',
      message: 'Package manager:',
      choices: ['npm', 'yarn', 'pnpm'],
      default: 'npm'
    },
    {
      type: 'confirm',
      name: 'initGit',
      message: 'Initialize git repository?',
      default: true
    }
  ]);
  
  // 第四步：确认信息
  console.log(chalk.cyan('\n📋 Project Summary:\n'));
  console.log(chalk.white('Name:'), chalk.yellow(basicInfo.projectName));
  console.log(chalk.white('Framework:'), chalk.yellow(techStack.framework));
  console.log(chalk.white('TypeScript:'), chalk.yellow(techStack.useTypeScript ? 'Yes' : 'No'));
  console.log(chalk.white('Features:'), chalk.yellow(tooling.features.join(', ')));
  console.log();
  
  const { confirm } = await inquirer.prompt([
    {
      type: 'confirm',
      name: 'confirm',
      message: 'Create project with these settings?',
      default: true
    }
  ]);
  
  if (!confirm) {
    console.log(chalk.yellow('❌ Project creation cancelled'));
    return;
  }
  
  // 第五步：创建项目
  const spinner = ora('Creating project...').start();
  
  try {
    // 创建目录
    mkdirSync(basicInfo.projectName, { recursive: true });
    
    // 创建 package.json
    const packageJson = {
      name: basicInfo.projectName,
      version: '1.0.0',
      description: basicInfo.description,
      author: basicInfo.author,
      scripts: {
        dev: 'vite',
        build: 'vite build',
        preview: 'vite preview'
      }
    };
    
    writeFileSync(
      `${basicInfo.projectName}/package.json`,
      JSON.stringify(packageJson, null, 2)
    );
    
    // 初始化 git
    if (tooling.initGit) {
      execSync('git init', { cwd: basicInfo.projectName });
    }
    
    spinner.succeed('Project created successfully!');
    
    console.log(chalk.green('\n✨ Done! Next steps:\n'));
    console.log(chalk.white(`  cd ${basicInfo.projectName}`));
    console.log(chalk.white(`  ${tooling.packageManager} install`));
    console.log(chalk.white(`  ${tooling.packageManager} run dev\n`));
    
  } catch (error) {
    spinner.fail('Failed to create project');
    console.error(chalk.red(error.message));
  }
}

createProject();
```

### 案例2：配置文件生成器

```javascript
// ===== 生成配置文件的交互式CLI =====

import inquirer from 'inquirer';
import { writeFileSync } from 'fs';

async function generateConfig() {
  const answers = await inquirer.prompt([
    {
      type: 'list',
      name: 'configType',
      message: 'What config file do you want to create?',
      choices: [
        'ESLint (.eslintrc.json)',
        'Prettier (.prettierrc.json)',
        'TypeScript (tsconfig.json)',
        'Jest (jest.config.js)'
      ]
    }
  ]);
  
  let config;
  
  if (answers.configType.includes('ESLint')) {
    const eslintAnswers = await inquirer.prompt([
      {
        type: 'list',
        name: 'env',
        message: 'Environment:',
        choices: ['Browser', 'Node.js', 'Both']
      },
      {
        type: 'checkbox',
        name: 'extends',
        message: 'Extends:',
        choices: [
          'eslint:recommended',
          'plugin:react/recommended',
          'plugin:vue/recommended',
          'airbnb',
          'standard'
        ]
      }
    ]);
    
    config = {
      env: {
        browser: eslintAnswers.env !== 'Node.js',
        node: eslintAnswers.env !== 'Browser',
        es2021: true
      },
      extends: eslintAnswers.extends,
      rules: {}
    };
    
    writeFileSync('.eslintrc.json', JSON.stringify(config, null, 2));
    console.log('✔ Created .eslintrc.json');
  }
  
  // ... 其他配置文件类似
}

generateConfig();
```

## 💡 最佳实践

```javascript
// ✅ 1. 提供合理的默认值
{
  default: 'my-app'  // 或者动态生成
  default: () => process.env.USER
}

// ✅ 2. 添加输入验证
{
  validate: (input) => {
    if (!input) return 'This field is required';
    return true;
  }
}

// ✅ 3. 使用 filter 转换输入
{
  filter: (input) => input.trim().toLowerCase()
}

// ✅ 4. 使用 when 避免无关问题
{
  when: (answers) => answers.someCondition
}

// ✅ 5. 分步骤询问
// 不要一次性问太多问题,分成多个 prompt

// ✅ 6. 提供清晰的提示
{
  message: 'Project name (lowercase, no spaces):'  // 说明格式要求
}

// ✅ 7. 使用 Separator 分组
choices: [
  new inquirer.Separator('=== Basic ==='),
  'Option 1',
  'Option 2',
  new inquirer.Separator('=== Advanced ==='),
  'Option 3'
]

// ✅ 8. 显示最终确认
const { confirm } = await inquirer.prompt([
  {
    type: 'confirm',
    message: 'Proceed with these settings?'
  }
]);
```
