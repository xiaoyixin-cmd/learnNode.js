# 10.4 HTTPS与TLS加密

> 配置安全的HTTPS服务器

---

## 🎯 为什么需要HTTPS？

### 核心概念

```
HTTPS = HTTP + TLS/SSL加密

HTTP的问题：
❌ 明文传输
❌ 容易被窃听
❌ 容易被篡改
❌ 无法验证身份

HTTPS的优势：
✅ 加密传输
✅ 防止窃听
✅ 防止篡改
✅ 身份验证
✅ SEO优势

TLS/SSL版本：
• SSL 2.0/3.0 - 已废弃 ❌
• TLS 1.0/1.1 - 不推荐 ⚠️
• TLS 1.2 - 可用 ✅
• TLS 1.3 - 推荐 ✅✅
```

## 🔐 生成SSL证书

### 1. 开发环境（自签名证书）

```bash
# ===== 使用 OpenSSL 生成自签名证书 =====

# 生成私钥
openssl genrsa -out server.key 2048

# 生成证书签名请求（CSR）
openssl req -new -key server.key -out server.csr

# 生成自签名证书（有效期365天）
openssl x509 -req -days 365 -in server.csr -signkey server.key -out server.crt

# 一步生成（开发用）
openssl req -x509 -newkey rsa:2048 -nodes -keyout server.key -out server.crt -days 365

# ===== 使用 mkcert（推荐，更简单）=====

# 安装 mkcert
# macOS: brew install mkcert
# Windows: choco install mkcert

# 安装本地CA
mkcert -install

# 生成证书
mkcert localhost 127.0.0.1 ::1

# 生成两个文件：
# localhost+2.pem（证书）
# localhost+2-key.pem（私钥）
```

### 2. 生产环境（Let's Encrypt）

```bash
# ===== 使用 Certbot 获取免费证书 =====

# 安装 Certbot
# Ubuntu/Debian
sudo apt install certbot

# CentOS/RHEL
sudo yum install certbot

# macOS
brew install certbot

# 获取证书（需要域名）
sudo certbot certonly --standalone -d yourdomain.com -d www.yourdomain.com

# 证书存放在：
# /etc/letsencrypt/live/yourdomain.com/fullchain.pem（证书）
# /etc/letsencrypt/live/yourdomain.com/privkey.pem（私钥）

# 自动续期（证书90天有效期）
sudo certbot renew --dry-run  # 测试
sudo certbot renew            # 实际续期

# 添加到 cron 自动续期
0 0 * * * certbot renew --quiet
```

## 🚀 Node.js HTTPS服务器

### 1. 基础HTTPS服务器

```javascript
// ===== https-server.js =====

import https from 'https';
import fs from 'fs';

// 读取证书
const options = {
  key: fs.readFileSync('server.key'),
  cert: fs.readFileSync('server.crt')
};

// 创建HTTPS服务器
const server = https.createServer(options, (req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello HTTPS!\n');
});

server.listen(443, () => {
  console.log('HTTPS server running on https://localhost:443');
});

// ⚠️ 注意：443端口需要root权限
// 开发时可以使用其他端口，如3000
```

### 2. Express HTTPS服务器

```javascript
// ===== express-https.js =====

import express from 'express';
import https from 'https';
import http from 'http';
import fs from 'fs';

const app = express();

// 路由
app.get('/', (req, res) => {
  res.send('Hello HTTPS!');
});

// HTTPS配置
const httpsOptions = {
  key: fs.readFileSync('server.key'),
  cert: fs.readFileSync('server.crt')
};

// 创建HTTPS服务器
const httpsServer = https.createServer(httpsOptions, app);

httpsServer.listen(443, () => {
  console.log('HTTPS server on https://localhost:443');
});

// 同时创建HTTP服务器（重定向到HTTPS）
const httpServer = http.createServer((req, res) => {
  // 重定向到HTTPS
  res.writeHead(301, {
    Location: `https://${req.headers.host}${req.url}`
  });
  res.end();
});

httpServer.listen(80, () => {
  console.log('HTTP server on http://localhost:80 (redirects to HTTPS)');
});
```

### 3. 强制HTTPS中间件

```javascript
// ===== middleware/https-redirect.js =====

export function httpsRedirect(req, res, next) {
  // 检查是否是HTTPS
  if (!req.secure && req.get('x-forwarded-proto') !== 'https') {
    // 重定向到HTTPS
    return res.redirect(301, `https://${req.headers.host}${req.url}`);
  }
  next();
}

// 使用
import express from 'express';
import { httpsRedirect } from './middleware/https-redirect.js';

const app = express();

// 只在生产环境强制HTTPS
if (process.env.NODE_ENV === 'production') {
  app.use(httpsRedirect);
}
```

## 🛡️ 安全配置

### 1. TLS配置

```javascript
// ===== 安全的TLS配置 =====

import https from 'https';
import fs from 'fs';

const options = {
  key: fs.readFileSync('server.key'),
  cert: fs.readFileSync('server.crt'),
  
  // TLS版本
  minVersion: 'TLSv1.2',  // 最低TLS 1.2
  maxVersion: 'TLSv1.3',  // 最高TLS 1.3
  
  // 密码套件（推荐的强加密）
  ciphers: [
    'ECDHE-ECDSA-AES128-GCM-SHA256',
    'ECDHE-RSA-AES128-GCM-SHA256',
    'ECDHE-ECDSA-AES256-GCM-SHA384',
    'ECDHE-RSA-AES256-GCM-SHA384',
    'ECDHE-ECDSA-CHACHA20-POLY1305',
    'ECDHE-RSA-CHACHA20-POLY1305'
  ].join(':'),
  
  // 优先使用服务器密码套件
  honorCipherOrder: true,
  
  // 禁用不安全的协议
  secureProtocol: 'TLS_method',
  secureOptions: 
    require('constants').SSL_OP_NO_SSLv2 |
    require('constants').SSL_OP_NO_SSLv3 |
    require('constants').SSL_OP_NO_TLSv1 |
    require('constants').SSL_OP_NO_TLSv1_1
};

const server = https.createServer(options, app);
```

### 2. 安全响应头

```javascript
// ===== 使用 Helmet 设置安全头 =====

import helmet from 'helmet';
import express from 'express';

const app = express();

// 使用所有 Helmet 中间件
app.use(helmet());

// 或者单独配置
app.use(helmet({
  // 强制HTTPS
  hsts: {
    maxAge: 31536000,  // 1年
    includeSubDomains: true,
    preload: true
  },
  
  // 内容安全策略
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: ["'self'"],
      styleSrc: ["'self'"],
      imgSrc: ["'self'", 'data:', 'https:']
    }
  },
  
  // 禁止在iframe中加载
  frameguard: {
    action: 'deny'
  },
  
  // 禁用浏览器MIME类型嗅探
  noSniff: true,
  
  // XSS过滤器
  xssFilter: true,
  
  // 隐藏X-Powered-By
  hidePoweredBy: true
}));
```

## 🎯 实战案例

### 案例1：完整的生产HTTPS服务器

```javascript
// ===== server.js =====

import express from 'express';
import https from 'https';
import http from 'http';
import fs from 'fs';
import helmet from 'helmet';
import rateLimit from 'express-rate-limit';
import compression from 'compression';

const app = express();

// ===== 安全中间件 =====

// Helmet安全头
app.use(helmet({
  hsts: {
    maxAge: 31536000,
    includeSubDomains: true,
    preload: true
  }
}));

// 压缩响应
app.use(compression());

// 限流
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000,  // 15分钟
  max: 100,  // 最多100个请求
  message: 'Too many requests from this IP'
});
app.use('/api/', limiter);

// JSON解析
app.use(express.json({ limit: '1mb' }));

// ===== 路由 =====

app.get('/', (req, res) => {
  res.json({
    message: 'Secure HTTPS Server',
    secure: req.secure,
    protocol: req.protocol
  });
});

app.get('/api/health', (req, res) => {
  res.json({ status: 'ok', timestamp: Date.now() });
});

// ===== HTTPS配置 =====

const httpsOptions = {
  key: fs.readFileSync('/path/to/privkey.pem'),
  cert: fs.readFileSync('/path/to/fullchain.pem'),
  minVersion: 'TLSv1.2',
  ciphers: [
    'ECDHE-ECDSA-AES128-GCM-SHA256',
    'ECDHE-RSA-AES128-GCM-SHA256',
    'ECDHE-ECDSA-AES256-GCM-SHA384',
    'ECDHE-RSA-AES256-GCM-SHA384'
  ].join(':'),
  honorCipherOrder: true
};

// ===== 启动服务器 =====

const PORT_HTTPS = process.env.PORT_HTTPS || 443;
const PORT_HTTP = process.env.PORT_HTTP || 80;

// HTTPS服务器
const httpsServer = https.createServer(httpsOptions, app);

httpsServer.listen(PORT_HTTPS, () => {
  console.log(`✅ HTTPS server running on port ${PORT_HTTPS}`);
});

// HTTP服务器（重定向）
const httpServer = http.createServer((req, res) => {
  res.writeHead(301, {
    Location: `https://${req.headers.host}${req.url}`
  });
  res.end();
});

httpServer.listen(PORT_HTTP, () => {
  console.log(`✅ HTTP server running on port ${PORT_HTTP} (redirects to HTTPS)`);
});

// ===== 优雅关闭 =====

process.on('SIGTERM', () => {
  console.log('SIGTERM signal received: closing servers');
  
  httpsServer.close(() => {
    console.log('HTTPS server closed');
  });
  
  httpServer.close(() => {
    console.log('HTTP server closed');
    process.exit(0);
  });
});
```

### 案例2：Nginx反向代理

```nginx
# ===== /etc/nginx/sites-available/myapp =====

# HTTP -> HTTPS 重定向
server {
    listen 80;
    server_name yourdomain.com www.yourdomain.com;
    
    # Let's Encrypt验证
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }
    
    # 重定向到HTTPS
    location / {
        return 301 https://$server_name$request_uri;
    }
}

# HTTPS服务器
server {
    listen 443 ssl http2;
    server_name yourdomain.com www.yourdomain.com;
    
    # SSL证书
    ssl_certificate /etc/letsencrypt/live/yourdomain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/yourdomain.com/privkey.pem;
    
    # SSL配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers 'ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256';
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    
    # HSTS
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    
    # 其他安全头
    add_header X-Frame-Options "DENY" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    
    # 反向代理到Node.js
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
```
