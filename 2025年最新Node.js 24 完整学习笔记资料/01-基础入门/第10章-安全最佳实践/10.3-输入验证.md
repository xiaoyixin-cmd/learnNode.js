# 10.3 输入验证与注入防护

> 验证用户输入，防止注入攻击

---

## 🎯 为什么输入验证重要？

### 核心概念

```
输入验证 = 确保用户输入符合预期格式

常见攻击类型：
• SQL注入（SQL Injection）
• NoSQL注入（NoSQL Injection）
• XSS（跨站脚本）
• 命令注入（Command Injection）
• 路径遍历（Path Traversal）

真实案例：
• 2017年 Equifax数据泄露 - SQL注入
• 2019年 Capital One数据泄露 - SSRF

防护原则：
✅ 永远不信任用户输入
✅ 验证所有输入
✅ 使用白名单而非黑名单
✅ 参数化查询
✅ 转义输出
```

```javascript
// ===== 基础验证 =====

import Joi from 'joi';

// 定义Schema
const userSchema = Joi.object({
  username: Joi.string()
    .alphanum()
    .min(3)
    .max(30)
    .required(),
  
  email: Joi.string()
    .email()
    .required(),
  
  password: Joi.string()
    .min(8)
    .pattern(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/)
    .required(),
  
  age: Joi.number()
    .integer()
    .min(0)
    .max(150),
  
  website: Joi.string()
    .uri(),
  
  tags: Joi.array()
    .items(Joi.string())
    .max(10)
});

// 验证数据
const userData = {
  username: 'alice123',
  email: 'alice@example.com',
  password: 'SecurePass123',
  age: 25
};

const { error, value } = userSchema.validate(userData);

if (error) {
  console.error('验证失败:', error.details);
  // [
  //   {
  //     message: '"password" must be at least 8 characters',
  //     path: ['password'],
  //     type: 'string.min'
  //   }
  // ]
} else {
  console.log('验证通过:', value);
}

// ===== Express中间件 =====

import express from 'express';

const app = express();
app.use(express.json());

function validate(schema) {
  return (req, res, next) => {
    const { error, value } = schema.validate(req.body);
    
    if (error) {
      return res.status(400).json({
        error: 'Validation failed',
        details: error.details.map(d => ({
          field: d.path.join('.'),
          message: d.message
        }))
      });
    }
    
    // 使用验证后的值
    req.body = value;
    next();
  };
}

// 使用中间件
app.post('/api/users', validate(userSchema), (req, res) => {
  // req.body 已经验证过了
  const user = req.body;
  // 创建用户...
  res.json({ success: true, user });
});
```

// ===== Zod验证 =====

import { z } from 'zod';

// 定义Schema
const UserSchema = z.object({
  username: z.string()
    .min(3, '用户名至少3个字符')
    .max(30, '用户名最多30个字符')
    .regex(/^[a-zA-Z0-9_]+$/, '只能包含字母、数字和下划线'),
  
  email: z.string()
    .email('无效的邮箱地址'),
  
  password: z.string()
    .min(8, '密码至少8个字符')
    .regex(
      /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/,
      '密码必须包含大小写字母和数字'
    ),
  
  age: z.number()
    .int('年龄必须是整数')
    .min(0, '年龄不能为负数')
    .max(150, '年龄不能超过150')
    .optional(),
  
  role: z.enum(['user', 'admin', 'moderator'])
    .default('user')
});

// TypeScript类型推导
type User = z.infer<typeof UserSchema>;

// 验证
try {
  const user: User = UserSchema.parse({
    username: 'alice123',
    email: 'alice@example.com',
    password: 'SecurePass123',
    role: 'user'
  });
  
  console.log('验证通过:', user);
} catch (error) {
  if (error instanceof z.ZodError) {
    console.error('验证失败:');
    error.issues.forEach(issue => {
      console.error(`  ${issue.path.join('.')}: ${issue.message}`);
    });
  }
}

// ===== Express中间件 =====

function validateZod(schema) {
  return (req, res, next) => {
    try {
      req.body = schema.parse(req.body);
      next();
    } catch (error) {
      if (error instanceof z.ZodError) {
        return res.status(400).json({
          error: 'Validation failed',
          details: error.issues.map(issue => ({
            field: issue.path.join('.'),
            message: issue.message
          }))
        });
      }
      next(error);
    }
  };
}

## 🛡️ SQL注入防护

### 1. 使用参数化查询

```javascript
// ===== MySQL2 - 正确的方式 =====

import mysql from 'mysql2/promise';

const pool = mysql.createPool({
  host: 'localhost',
  user: 'root',
  database: 'myapp'
});

// ❌ 错误：字符串拼接（容易SQL注入）
async function getUserBad(username) {
  const sql = `SELECT * FROM users WHERE username = '${username}'`;
  const [rows] = await pool.query(sql);
  return rows;
}

// 如果 username = "admin' OR '1'='1"
// 最终SQL: SELECT * FROM users WHERE username = 'admin' OR '1'='1'
// 返回所有用户！

// ✅ 正确：参数化查询
async function getUserGood(username) {
  const sql = 'SELECT * FROM users WHERE username = ?';
  const [rows] = await pool.query(sql, [username]);
  return rows;
}

// 即使 username = "admin' OR '1'='1"
// 也会被当作字符串处理，不会执行注入

// ===== PostgreSQL - 正确的方式 =====

import pg from 'pg';

const client = new pg.Client({
  host: 'localhost',
  database: 'myapp'
});

await client.connect();

// ✅ 使用占位符 $1, $2, ...
const result = await client.query(
  'SELECT * FROM users WHERE username = $1 AND age > $2',
  ['alice', 18]
);
```

## 🔒 NoSQL注入防护

### MongoDB注入

```javascript
// ===== Mongoose - 防止NoSQL注入 =====

import mongoose from 'mongoose';

// ❌ 错误：直接使用用户输入
async function loginBad(req, res) {
  const { username, password } = req.body;
  
  // 如果用户发送：{ "username": {"$ne": null}, "password": {"$ne": null} }
  // 会匹配任何用户！
  const user = await User.findOne({
    username,
    password
  });
}

// ✅ 正确：验证输入类型
async function loginGood(req, res) {
  const { username, password } = req.body;
  
  // 1. 验证类型
  if (typeof username !== 'string' || typeof password !== 'string') {
    return res.status(400).json({ error: 'Invalid input' });
  }
  
  // 2. 使用 Schema 验证
  const userSchema = Joi.object({
    username: Joi.string().required(),
    password: Joi.string().required()
  });
  
  const { error } = userSchema.validate({ username, password });
  if (error) {
    return res.status(400).json({ error: 'Validation failed' });
  }
  
  // 3. 安全查询
  const user = await User.findOne({
    username: username.toString(),
    password: password.toString()
  });
}

// ===== 使用 mongo-sanitize =====

import mongoSanitize from 'express-mongo-sanitize';
import express from 'express';

const app = express();

// 全局中间件：移除 $ 和 . 字符
app.use(mongoSanitize());

// 或者手动清理
import sanitize from 'mongo-sanitize';

const clean = sanitize(req.body);
// { "username": {"$ne": null} } → { "username": {} }
```

## 🎯 XSS防护

### 1. 输出转义

```javascript
// ===== HTML转义 =====

function escapeHtml(text) {
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  
  return text.replace(/[&<>"']/g, m => map[m]);
}

// 使用
const userInput = '<script>alert("XSS")</script>';
const safe = escapeHtml(userInput);
// &lt;script&gt;alert(&quot;XSS&quot;)&lt;/script&gt;

// ===== 使用库 =====

import he from 'he';  // npm install he

const encoded = he.encode('<script>alert("XSS")</script>');
const decoded = he.decode('&lt;script&gt;');

// ===== DOMPurify（浏览器端）=====

import DOMPurify from 'isomorphic-dompurify';

const dirty = '<img src=x onerror=alert("XSS")>';
const clean = DOMPurify.sanitize(dirty);
// <img src="x">
```

## 🎯 实战案例

### 案例1：完整的API验证

```javascript
// ===== api/routes/users.js =====

import express from 'express';
import { z } from 'zod';
import { User } from '../models/User.js';
import bcrypt from 'bcrypt';

const router = express.Router();

// ===== Schema定义 =====

const RegisterSchema = z.object({
  username: z.string()
    .min(3)
    .max(30)
    .regex(/^[a-zA-Z0-9_]+$/),
  
  email: z.string()
    .email()
    .toLowerCase()
    .transform(email => email.trim()),
  
  password: z.string()
    .min(8)
    .regex(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])/),
  
  age: z.number()
    .int()
    .min(13)
    .max(150)
    .optional(),
  
  bio: z.string()
    .max(500)
    .optional()
});

const LoginSchema = z.object({
  email: z.string().email(),
  password: z.string()
});

// ===== 验证中间件 =====

function validate(schema) {
  return async (req, res, next) => {
    try {
      req.body = await schema.parseAsync(req.body);
      next();
    } catch (error) {
      if (error instanceof z.ZodError) {
        return res.status(400).json({
          error: 'Validation failed',
          details: error.issues.map(issue => ({
            field: issue.path.join('.'),
            message: issue.message
          }))
        });
      }
      next(error);
    }
  };
}

// ===== 注册API =====

router.post('/register', validate(RegisterSchema), async (req, res) => {
  const { username, email, password, age, bio } = req.body;
  
  try {
    // 检查用户是否已存在
    const exists = await User.findOne({ $or: [{ username }, { email }] });
    if (exists) {
      return res.status(409).json({
        error: '用户名或邮箱已存在'
      });
    }
    
    // 哈希密码
    const hashedPassword = await bcrypt.hash(password, 10);
    
    // 创建用户
    const user = new User({
      username,
      email,
      password: hashedPassword,
      age,
      bio: bio ? escapeHtml(bio) : undefined  // 转义bio
    });
    
    await user.save();
    
    res.status(201).json({
      success: true,
      user: {
        id: user._id,
        username: user.username,
        email: user.email
      }
    });
  } catch (error) {
    console.error('注册失败:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// ===== 登录API =====

router.post('/login', validate(LoginSchema), async (req, res) => {
  const { email, password } = req.body;
  
  try {
    // 查找用户
    const user = await User.findOne({ email });
    if (!user) {
      return res.status(401).json({ error: '邮箱或密码错误' });
    }
    
    // 验证密码
    const valid = await bcrypt.compare(password, user.password);
    if (!valid) {
      return res.status(401).json({ error: '邮箱或密码错误' });
    }
    
    // 生成token...
    res.json({ success: true, token: '...' });
  } catch (error) {
    console.error('登录失败:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```
