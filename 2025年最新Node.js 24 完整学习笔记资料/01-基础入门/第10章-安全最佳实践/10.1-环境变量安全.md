# 10.1 环境变量安全管理

> 安全地管理敏感配置信息

---

## 🎯 为什么环境变量安全重要？

### 核心概念

```
环境变量 = 存储配置信息的键值对

常见的敏感信息：
• API密钥（API Keys）
• 数据库密码（Database Passwords）
• JWT密钥（Secret Keys）
• 第三方服务凭证（Service Credentials）
• 加密密钥（Encryption Keys）

安全隐患：
❌ 硬编码在代码中
❌ 提交到Git仓库
❌ 明文存储
❌ 日志中泄露
❌ 错误信息暴露

正确做法：
✅ 使用环境变量
✅ .env文件不提交
✅ 生产环境加密
✅ 最小权限原则
✅ 定期轮换密钥
```

```javascript
  }
}
```

### 2. 类型转换与验证

```javascript
// ===== config.js - 环境变量配置管理 =====

import dotenv from 'dotenv';

dotenv.config();

// 类型转换辅助函数
function getEnvString(key, defaultValue = '') {
  const value = process.env[key];
  if (value === undefined && defaultValue === '') {
    throw new Error(`环境变量 ${key} 未设置`);
  }
  return value || defaultValue;
}

function getEnvNumber(key, defaultValue) {
  const value = process.env[key];
  if (!value) {
    if (defaultValue === undefined) {
      throw new Error(`环境变量 ${key} 未设置`);
    }
    return defaultValue;
  }
  
  const num = parseInt(value, 10);
  if (isNaN(num)) {
    throw new Error(`环境变量 ${key} 必须是数字，当前值: ${value}`);
  }
  return num;
}

function getEnvBoolean(key, defaultValue = false) {
  const value = process.env[key];
  if (!value) return defaultValue;
  
  return value.toLowerCase() === 'true';
}

function getEnvArray(key, separator = ',') {
  const value = process.env[key];
  if (!value) return [];
  
  return value.split(separator).map(item => item.trim());
}

// ===== 导出配置对象 =====

export const config = {
  // 应用配置
  env: getEnvString('NODE_ENV', 'development'),
  port: getEnvNumber('PORT', 3000),
  host: getEnvString('HOST', 'localhost'),
  debug: getEnvBoolean('DEBUG', false),
  
  // 数据库配置
  database: {
    url: getEnvString('DATABASE_URL'),
    user: getEnvString('DATABASE_USER'),
    password: getEnvString('DATABASE_PASSWORD'),
    poolSize: getEnvNumber('DATABASE_POOL_SIZE', 10)
  },
  
  // JWT配置
  jwt: {
    secret: getEnvString('JWT_SECRET'),
    expiresIn: getEnvString('JWT_EXPIRES_IN', '1d')
  },
  
  // API配置
  api: {
    key: getEnvString('API_KEY'),
    secret: getEnvString('API_SECRET'),
    baseUrl: getEnvString('API_BASE_URL', 'https://api.example.com')
  },
  
  // CORS配置
  cors: {
    origins: getEnvArray('CORS_ORIGINS', ','),
    credentials: getEnvBoolean('CORS_CREDENTIALS', true)
  }
};

// ===== 验证配置 =====

function validateConfig() {
  // 验证JWT密钥长度
  if (config.jwt.secret.length < 32) {
    throw new Error('JWT_SECRET 长度必须至少32个字符');
  }
  
  // 验证数据库URL格式
  if (!config.database.url.startsWith('mongodb://')) {
    throw new Error('DATABASE_URL 必须是有效的MongoDB连接字符串');
  }
  
  // 验证端口范围
  if (config.port < 1 || config.port > 65535) {
    throw new Error('PORT 必须在 1-65535 范围内');
  }
  
  console.log('✅ 配置验证通过');
}

validateConfig();

// ===== 使用 =====

import { config } from './config.js';

console.log('应用运行在:', `${config.host}:${config.port}`);
console.log('数据库:', config.database.url);
console.log('调试模式:', config.debug ? '开启' : '关闭');
```

### 1. 使用密钥管理服务

```javascript
// ===== AWS Secrets Manager =====

import {
  SecretsManagerClient,
  GetSecretValueCommand
} from '@aws-sdk/client-secrets-manager';

async function getSecret(secretName) {
  const client = new SecretsManagerClient({
    region: process.env.AWS_REGION
  });
  
  try {
    const response = await client.send(
      new GetSecretValueCommand({
        SecretId: secretName
      })
    );
    
    return JSON.parse(response.SecretString);
  } catch (error) {
    console.error('获取密钥失败:', error);
    throw error;
  }
}

// 使用
const dbCredentials = await getSecret('prod/database');
const apiKeys = await getSecret('prod/api-keys');

// ===== Hashicorp Vault =====

import vault from 'node-vault';

const vaultClient = vault({
  apiVersion: 'v1',
  endpoint: process.env.VAULT_ADDR,
  token: process.env.VAULT_TOKEN
});

async function getVaultSecret(path) {
  try {
    const result = await vaultClient.read(path);
    return result.data;
  } catch (error) {
    console.error('读取Vault密钥失败:', error);
    throw error;
  }
}

// 使用
const secrets = await getVaultSecret('secret/data/myapp');
```

### 2. 加密存储

```javascript
// ===== 加密环境变量 =====

import crypto from 'crypto';

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY; // 32字节
const IV_LENGTH = 16;

function encrypt(text) {
  const iv = crypto.randomBytes(IV_LENGTH);
  const cipher = crypto.createCipheriv(
    'aes-256-cbc',
    Buffer.from(ENCRYPTION_KEY, 'hex'),
    iv
  );
  
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  return iv.toString('hex') + ':' + encrypted;
}

function decrypt(text) {
  const parts = text.split(':');
  const iv = Buffer.from(parts[0], 'hex');
  const encryptedText = parts[1];
  
  const decipher = crypto.createDecipheriv(
    'aes-256-cbc',
    Buffer.from(ENCRYPTION_KEY, 'hex'),
    iv
  );
  
  let decrypted = decipher.update(encryptedText, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  
  return decrypted;
}

// ===== .env.encrypted =====
DATABASE_PASSWORD=IV:ENCRYPTED_DATA
API_SECRET=IV:ENCRYPTED_DATA

// ===== 加载并解密 =====
const encryptedPassword = process.env.DATABASE_PASSWORD;
const password = decrypt(encryptedPassword);
```

```javascript
// ===== config/index.js =====

import dotenv from 'dotenv';
import Joi from 'joi';  // npm install joi

// 加载环境变量
dotenv.config();

// ===== 配置Schema验证 =====

const envSchema = Joi.object({
  // 应用配置
  NODE_ENV: Joi.string()
    .valid('development', 'production', 'test')
    .default('development'),
  
  PORT: Joi.number()
    .port()
    .default(3000),
  
  HOST: Joi.string()
    .default('localhost'),
  
  // 数据库配置
  DATABASE_URL: Joi.string()
    .uri()
    .required(),
  
  DATABASE_POOL_SIZE: Joi.number()
    .integer()
    .min(1)
    .max(100)
    .default(10),
  
  // JWT配置
  JWT_SECRET: Joi.string()
    .min(32)
    .required(),
  
  JWT_EXPIRES_IN: Joi.string()
    .default('1d'),
  
  // API配置
  API_KEY: Joi.string()
    .required(),
  
  API_RATE_LIMIT: Joi.number()
    .integer()
    .min(1)
    .default(100),
  
  // CORS配置
  CORS_ORIGINS: Joi.string()
    .default('*'),
  
  // 日志配置
  LOG_LEVEL: Joi.string()
    .valid('error', 'warn', 'info', 'debug')
    .default('info')
}).unknown();  // 允许其他环境变量

// ===== 验证环境变量 =====

const { error, value: validatedEnv } = envSchema.validate(process.env);

if (error) {
  throw new Error(`配置验证失败: ${error.message}`);
}

// ===== 导出配置 =====

export const config = {
  env: validatedEnv.NODE_ENV,
  isDevelopment: validatedEnv.NODE_ENV === 'development',
  isProduction: validatedEnv.NODE_ENV === 'production',
  isTest: validatedEnv.NODE_ENV === 'test',
  
  server: {
    host: validatedEnv.HOST,
    port: validatedEnv.PORT
  },
  
  database: {
    url: validatedEnv.DATABASE_URL,
    poolSize: validatedEnv.DATABASE_POOL_SIZE
  },
  
  jwt: {
    secret: validatedEnv.JWT_SECRET,
    expiresIn: validatedEnv.JWT_EXPIRES_IN
  },

    key: validatedEnv.API_KEY,
```

### 案例2：密钥轮换

```javascript
// ===== 密钥轮换系统 =====

import crypto from 'crypto';
import fs from 'fs-extra';

class KeyRotation {
  constructor(keyFile = '.keys.json') {
    this.keyFile = keyFile;
    this.keys = this.loadKeys();
  }
  
  loadKeys() {
    if (!fs.existsSync(this.keyFile)) {
      return {
        current: this.generateKey(),
        previous: null,
        rotatedAt: Date.now()
      };
    }
    
    return fs.readJsonSync(this.keyFile);
  }
  
  generateKey() {
    return crypto.randomBytes(32).toString('hex');
  }
  
  rotateKey() {
    const newKeys = {
      current: this.generateKey(),
      previous: this.keys.current,
      rotatedAt: Date.now()
    };
    
    fs.writeJsonSync(this.keyFile, newKeys, { spaces: 2 });
    this.keys = newKeys;
    
    console.log('✅ 密钥已轮换');
    return newKeys;
  }
  
  getCurrentKey() {
    return this.keys.current;
  }
  
  verifyKey(key) {
    return key === this.keys.current || key === this.keys.previous;
  }
  
  shouldRotate(maxAge = 30 * 24 * 60 * 60 * 1000) {  // 30天
    const age = Date.now() - this.keys.rotatedAt;
    return age > maxAge;
  }
}

// 使用
const keyRotation = new KeyRotation();

// 检查是否需要轮换
if (keyRotation.shouldRotate()) {
  keyRotation.rotateKey();
}

// 获取当前密钥
const currentKey = keyRotation.getCurrentKey();
```
