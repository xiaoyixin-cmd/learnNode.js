# 5.4 并发控制实战

> 如何优雅地同时处理多个异步任务，既快又稳

---

## 🎯 什么是并发控制？

### 核心概念

```
并发控制 = 同时处理多个异步任务的策略

生活中的比喻：
餐厅点餐：
❌ 串行：一个顾客点完，下一个才能点（太慢）
✅ 并发：多个窗口同时接单（快）
✅ 限流：只开3个窗口，避免厨房忙不过来（稳）

关键概念：
• 串行：一个接一个，慢但稳定
• 并发：同时多个，快但可能压力大
• 限流：控制并发数量，平衡速度和压力
```

## 🔀 Promise 并发方法全解

### 1. Promise.all() - 全部成功才成功

```javascript
// ===== 使用场景：所有任务都必须成功 =====

async function loadUserData(userId) {
  try {
    // 同时获取用户的个人资料、帖子、好友
    const [profile, posts, friends] = await Promise.all([
      fetch(`/api/users/${userId}/profile`).then(r => r.json()),
      fetch(`/api/users/${userId}/posts`).then(r => r.json()),
      fetch(`/api/users/${userId}/friends`).then(r => r.json())
    ]);
    
    console.log('用户资料：', profile);
    console.log('帖子数量：', posts.length);
    console.log('好友数量：', friends.length);
    
    return { profile, posts, friends };
    
  } catch (error) {
    // ⚠️ 只要有一个失败，就会进入这里
    console.error('加载用户数据失败：', error);
    throw error;
  }
}

// ===== 特点 =====

const promises = [
  Promise.resolve(1),
  Promise.resolve(2),
  Promise.resolve(3)
];

Promise.all(promises).then(results => {
  console.log(results);  // [1, 2, 3]
  // ✅ 返回数组，顺序与输入一致
});

// ⚠️ 失败案例：一个失败，全部失败
const mixedPromises = [
  Promise.resolve('成功1'),
  Promise.reject('失败'),
  Promise.resolve('成功2')
];

Promise.all(mixedPromises)
  .then(results => {
    // 不会执行
  })
  .catch(error => {
    console.error(error);  // "失败"
    // ⚠️ 只捕获第一个失败，其他结果丢失
  });

// 💡 使用时机：
// ✅ 多个请求互不依赖，需要同时加载
// ✅ 所有数据都必需，缺一不可
// ❌ 部分数据可以失败时不合适
```

### 2. Promise.allSettled() - 等待所有结果

```javascript
// ===== 使用场景：需要所有结果，无论成功失败 =====

async function loadDashboardData() {
  const results = await Promise.allSettled([
    fetch('/api/user').then(r => r.json()),
    fetch('/api/orders').then(r => r.json()),
    fetch('/api/notifications').then(r => r.json())
  ]);
  
  // ✅ 总是成功，返回每个 Promise 的状态
  results.forEach((result, index) => {
    if (result.status === 'fulfilled') {
      // 成功的结果
      console.log(`任务${index}成功：`, result.value);
    } else {
      // 失败的原因
      console.error(`任务${index}失败：`, result.reason);
    }
  });
  
  return results;
}

// ===== 实战案例：批量删除文件 =====

async function deleteFiles(filePaths) {
  const results = await Promise.allSettled(
    filePaths.map(path => fs.promises.unlink(path))
  );
  
  // 统计成功和失败
  const succeeded = results.filter(r => r.status === 'fulfilled').length;
  const failed = results.filter(r => r.status === 'rejected').length;
  
  console.log(`删除完成：${succeeded}个成功，${failed}个失败`);
  
  // 显示失败的文件
  results.forEach((result, index) => {
    if (result.status === 'rejected') {
      console.error(`删除失败：${filePaths[index]} - ${result.reason}`);
    }
  });
  
  return { succeeded, failed };
}

// 使用
await deleteFiles([
  '/path/file1.txt',
  '/path/file2.txt',
  '/path/not-exist.txt'  // 这个不存在
]);
// 输出：
// 删除完成：2个成功，1个失败
// 删除失败：/path/not-exist.txt - ENOENT: no such file or directory

// 💡 使用时机：
// ✅ 批量操作，需要知道每个结果
// ✅ 部分失败可以接受
// ✅ 需要收集所有错误信息
```

### 3. Promise.race() - 最快的赢

```javascript
// ===== 使用场景：只需要最快的一个结果 =====

async function fetchWithTimeout(url, timeout = 5000) {
  // 竞速：fetch 和 timeout
  return Promise.race([
    fetch(url).then(r => r.json()),
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error('请求超时')), timeout)
    )
  ]);
}

// 使用
try {
  const data = await fetchWithTimeout('/api/slow', 3000);
  console.log('数据：', data);
} catch (error) {
```

```javascript
const promises = [
```

## 🎮 并发控制实战

### 案例1：限制并发数量

```javascript
// ===== 问题：同时发送100个请求会压垮服务器 =====

// ❌ 错误方式：一次性全部并发
async function downloadAllBad(urls) {
  // 如果有1000个URL，会同时发送1000个请求！
  return Promise.all(urls.map(url => fetch(url)));
}

// ✅ 解决方案：限制并发数量

// 方案1：手动分批处理
async function downloadWithLimit(urls, limit = 5) {
  const results = [];
  
  // 每次处理 limit 个
  for (let i = 0; i < urls.length; i += limit) {
    const batch = urls.slice(i, i + limit);
    
    console.log(`处理批次 ${Math.floor(i / limit) + 1}，共${batch.length}个`);
    
    // 等待这一批完成
    const batchResults = await Promise.all(
      batch.map(url => fetch(url).then(r => r.json()))
    );
    
    results.push(...batchResults);
  }
  
  return results;
}

// 使用
const urls = [
  '/api/data1', '/api/data2', '/api/data3',
  '/api/data4', '/api/data5', '/api/data6',
  '/api/data7', '/api/data8'
];

await downloadWithLimit(urls, 3);
// 批次1：处理 data1, data2, data3
// 批次2：处理 data4, data5, data6
// 批次3：处理 data7, data8

// 方案2：滑动窗口（更高效）
class ConcurrencyController {
  constructor(limit) {
    this.limit = limit;  // 最大并发数
    this.running = 0;    // 当前运行数
    this.queue = [];     // 等待队列
  }
  
  async add(fn) {
    // 如果达到限制，等待
    while (this.running >= this.limit) {
      await new Promise(resolve => this.queue.push(resolve));
    }
    
    this.running++;
    
    try {
      // 执行任务
      return await fn();
    } finally {
      this.running--;
      
      // 通知下一个任务可以开始
      const resolve = this.queue.shift();
      if (resolve) resolve();
    }
  }
}

// 使用滑动窗口
async function downloadWithController(urls, limit = 5) {
  const controller = new ConcurrencyController(limit);
  
  // 所有任务同时提交，但自动控制并发
  return Promise.all(
    urls.map(url =>
      controller.add(async () => {
        console.log(`开始下载：${url}`);
        const response = await fetch(url);
        console.log(`完成下载：${url}`);
        return response.json();
      })
    )
  );
}

// 方案3：使用成熟的库（推荐）
// npm install p-limit
import pLimit from 'p-limit';

async function downloadWithLibrary(urls, limit = 5) {
  const limiter = pLimit(limit);
  
  return Promise.all(
    urls.map(url =>
      limiter(() => fetch(url).then(r => r.json()))
    )
  );
}
```

### 案例2：带重试的并发请求

```javascript
// ===== 实战：下载文件，失败自动重试 =====

async function fetchWithRetry(url, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      console.log(`尝试 ${attempt}/${maxRetries}: ${url}`);
      
      const response = await fetch(url);
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
      
      console.log(`✅ 成功: ${url}`);
      return await response.json();
      
    } catch (error) {
      console.error(`❌ 失败 ${attempt}/${maxRetries}: ${error.message}`);
      
      // 最后一次尝试失败，抛出错误
      if (attempt === maxRetries) {
        throw new Error(`${url} 失败：${error.message}`);
      }
      
      // 等待一段时间再重试（指数退避）
      const delay = Math.pow(2, attempt - 1) * 1000;  // 1s, 2s, 4s
      console.log(`等待 ${delay}ms 后重试...`);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
}

// 结合并发控制
async function downloadMultipleWithRetry(urls, concurrency = 5) {
  const controller = new ConcurrencyController(concurrency);
  
  const results = await Promise.allSettled(
    urls.map(url =>
      controller.add(() => fetchWithRetry(url))
    )
  );
  
  // 统计结果
  const succeeded = results.filter(r => r.status === 'fulfilled').length;
  const failed = results.filter(r => r.status === 'rejected').length;
  
  console.log(`\n下载完成：${succeeded}个成功，${failed}个失败`);
  
  return results;
}
```

## 🎮 高级并发模式

### 1. 队列模式（Queue）

```javascript
// ===== 先进先出的任务队列 =====

class TaskQueue {
  constructor(concurrency = 1) {
    this.concurrency = concurrency;  // 并发数
    this.running = 0;                // 当前运行任务数
    this.queue = [];                 // 等待队列
  }
  
  async add(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this.process();
    });
  }
  
  async process() {
    if (this.running >= this.concurrency || this.queue.length === 0) {
      return;
    }
    
    this.running++;
    const { task, resolve, reject } = this.queue.shift();
    
    try {
      const result = await task();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      this.running--;
      this.process();  // 处理下一个任务
    }
  }
  
  get pending() {
    return this.queue.length;
  }
  
  get active() {
    return this.running;
  }
}

// 使用示例
const queue = new TaskQueue(3);  // 最多3个并发

const urls = [
  '/api/1', '/api/2', '/api/3', 
  '/api/4', '/api/5', '/api/6'
];

const results = await Promise.all(
  urls.map(url => 
    queue.add(() => fetch(url).then(r => r.json()))
  )
);

console.log('所有任务完成：', results);
```

### 2. 优先队列（Priority Queue）

```javascript
// ===== 高优先级任务先执行 =====

class PriorityQueue {
  constructor(concurrency = 1) {
    this.concurrency = concurrency;
    this.running = 0;
    this.queue = [];  // { task, priority, resolve, reject }
  }
  
  async add(task, priority = 0) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, priority, resolve, reject });
      // 按优先级排序（高优先级在前）
      this.queue.sort((a, b) => b.priority - a.priority);
      this.process();
    });
  }
  
  async process() {
    if (this.running >= this.concurrency || this.queue.length === 0) {
      return;
    }
    
    this.running++;
    const { task, resolve, reject } = this.queue.shift();
    
    try {
      const result = await task();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      this.running--;
      this.process();
    }
  }
}

// 使用示例
const pQueue = new PriorityQueue(2);

// 添加不同优先级的任务
pQueue.add(() => fetchData('普通任务1'), 1);
pQueue.add(() => fetchData('高优先级'), 10);  // 先执行
pQueue.add(() => fetchData('普通任务2'), 1);
pQueue.add(() => fetchData('最高优先级'), 100);  // 最先执行
```

### 3. 断点续传模式

```javascript
// ===== 下载失败后可以从断点继续 =====

class ResumeDownloader {
  constructor() {
    this.progress = {};  // 保存进度
  }
  
  async download(url, filename) {
    // 检查是否有未完成的下载
    const savedProgress = this.progress[filename];
    let startByte = savedProgress?.downloadedBytes || 0;
    
    try {
      console.log(`开始下载：${filename}`);
      if (startByte > 0) {
        console.log(`从 ${startByte} 字节继续下载`);
      }
      
      // 发送带范围的请求
      const response = await fetch(url, {
        headers: {
          'Range': `bytes=${startByte}-`
        }
      });
      
      const totalSize = parseInt(
        response.headers.get('Content-Length')
      ) + startByte;
      
      const reader = response.body.getReader();
      const chunks = [];
      let downloadedBytes = startByte;
      
      while (true) {
        const { done, value } = await reader.read();
        
        if (done) break;
        
        chunks.push(value);
        downloadedBytes += value.length;
        
        // 保存进度
        this.progress[filename] = {
          downloadedBytes,
          totalSize
        };
        
        // 显示进度
        const percent = ((downloadedBytes / totalSize) * 100).toFixed(1);
        console.log(`下载进度：${percent}%`);
      }
      
      // 合并数据
      const blob = new Blob(chunks);
      
      // 清除进度
      delete this.progress[filename];
      
      console.log(`✅ 下载完成：${filename}`);
      return blob;
      
    } catch (error) {
      console.error(`下载失败：${error.message}`);
      console.log('进度已保存，可以稍后继续');
      throw error;
    }
  }
}

// 使用示例
const downloader = new ResumeDownloader();

try {
  await downloader.download('/big-file.zip', 'file.zip');
} catch (error) {
  console.log('下载中断，稍后继续...');
  
  // 模拟稍后继续
  setTimeout(async () => {
    await downloader.download('/big-file.zip', 'file.zip');
  }, 5000);
}
```

### 4. 流式处理（Stream Processing）

```javascript
// ===== 边接收边处理，不需要等全部完成 =====

async function* fetchPaginated(baseUrl, totalPages) {
  for (let page = 1; page <= totalPages; page++) {
    const response = await fetch(`${baseUrl}?page=${page}`);
    const data = await response.json();
    yield data;  // 立即返回当前页数据
  }
}

// 使用流式处理
async function processAllPages() {
  const processor = fetchPaginated('/api/items', 10);
  
  let totalItems = 0;
  
  for await (const pageData of processor) {
    // 立即处理当前页，不等待其他页
    console.log(`处理第${pageData.page}页，${pageData.items.length}条数据`);
    
    await processItems(pageData.items);
    totalItems += pageData.items.length;
  }
  
  console.log(`全部完成，共处理 ${totalItems} 条数据`);
}

// ===== 实战：大文件分块上传 =====

async function* chunkFile(file, chunkSize = 1024 * 1024) {
  let offset = 0;
  
  while (offset < file.size) {
    const chunk = file.slice(offset, offset + chunkSize);
    yield {
      chunk,
      offset,
      total: file.size
    };
    offset += chunkSize;
  }
}

async function uploadLargeFile(file) {
  const chunker = chunkFile(file);
  
  let uploadedBytes = 0;
  
  for await (const { chunk, offset, total } of chunker) {
    // 上传当前块
    await fetch('/upload', {
      method: 'POST',
      headers: {
        'Content-Range': `bytes ${offset}-${offset + chunk.size}/${total}`
      },
      body: chunk
    });
    
    uploadedBytes += chunk.size;
    const percent = ((uploadedBytes / total) * 100).toFixed(1);
    console.log(`上传进度：${percent}%`);
  }
  
  console.log('✅ 上传完成');
}
```

## 🎓 常见面试题

### 1. Promise.all vs Promise.allSettled 的区别？

```javascript
// 问题：这两个方法有什么区别？

const promises = [
  Promise.resolve('成功1'),
  Promise.reject('失败'),
  Promise.resolve('成功2')
];

// Promise.all：一个失败，全部失败
Promise.all(promises)
  .then(results => {
    // 不会执行
  })
  .catch(error => {
    console.log(error);  // "失败"
    // ❌ 无法获取成功的结果
  });

// Promise.allSettled：等待所有完成
Promise.allSettled(promises)
  .then(results => {
    console.log(results);
    // ✅ 可以获取所有结果
    // [
    //   { status: 'fulfilled', value: '成功1' },
    //   { status: 'rejected', reason: '失败' },
    //   { status: 'fulfilled', value: '成功2' }
    // ]
  });

// 答案：
// Promise.all：只要一个失败就立即失败
// Promise.allSettled：等待所有完成，返回每个结果
```

### 2. 如何限制并发数量？

```javascript
// 问题：100个请求，如何控制最多5个并发？

// 答案：使用并发控制器或分批处理

// 方法1：分批处理（简单）
async function batchProcess(urls, batchSize = 5) {
  const results = [];
  
  for (let i = 0; i < urls.length; i += batchSize) {
    const batch = urls.slice(i, i + batchSize);
    const batchResults = await Promise.all(
      batch.map(url => fetch(url))
    );
    results.push(...batchResults);
  }
  
  return results;
}

// 方法2：滑动窗口（高效）
class ConcurrencyController {
  constructor(limit) {
    this.limit = limit;
    this.running = 0;
    this.queue = [];
  }
  
  async add(fn) {
    while (this.running >= this.limit) {
      await new Promise(resolve => this.queue.push(resolve));
    }
    
    this.running++;
    
    try {
      return await fn();
    } finally {
      this.running--;
      const resolve = this.queue.shift();
      if (resolve) resolve();
    }
  }
}
```

### 3. Promise.race 会取消其他 Promise 吗？

```javascript
// 问题：Promise.race 返回最快的结果后，其他 Promise 会被取消吗？

const promises = [
  new Promise(resolve => {
    setTimeout(() => {
      console.log('Promise 1 完成');
      resolve('结果1');
    }, 1000);
  }),
  new Promise(resolve => {
    setTimeout(() => {
      console.log('Promise 2 完成');
      resolve('结果2');
    }, 2000);
  })
];

const result = await Promise.race(promises);
console.log('获胜者：', result);

// 输出：
// 获胜者： 结果1
// Promise 1 完成  （1秒后）
// Promise 2 完成  （2秒后）← 仍然会执行！

// 答案：
// ❌ 不会取消！其他 Promise 会继续执行
// ✅ 只是忽略了它们的结果
// 💡 如需取消，使用 AbortController
```

## 📚 总结

### 并发控制核心要点

| 方法 | 适用场景 | 特点 |
|------|----------|------|
| **Promise.all()** | 所有任务必须成功 | 一个失败全部失败 |
| **Promise.allSettled()** | 需要所有结果 | 总是成功，返回每个状态 |
| **Promise.race()** | 只需最快结果 | 返回第一个完成的 |
| **Promise.any()** | 任意成功即可 | 返回第一个成功的 |
| **并发控制器** | 大量任务限流 | 控制同时运行数量 |

### 性能优化策略

1. **独立任务并发**：能并行就并行
2. **大量任务限流**：避免资源耗尽
3. **任务分批**：平衡速度和稳定性
4. **添加重试**：提高成功率
5. **追踪进度**：提升用户体验
