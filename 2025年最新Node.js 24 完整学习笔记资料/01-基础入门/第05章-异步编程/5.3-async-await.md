# 5.3 async/await 实战

> async/await 是 Promise 的语法糖，让异步代码看起来像同步代码

---

## 🎯 async/await 是什么？

### 基础概念

```
async/await = Promise 的更优雅写法

生活中的比喻：
Promise 像是：
"我给你一个取餐号，你拿着号等着，好了我叫你"

async/await 像是：
"你在窗口排队等着，轮到你了直接给你"

关键特点：
✅ 代码看起来像同步的，但实际是异步的
✅ 更易读易写
✅ 更好的错误处理（用 try/catch）
✅ 基于 Promise，可以混用
```

### 第一个 async/await 示例

```javascript
// ===== Promise 版本 vs async/await 版本 =====

// 方式1：使用 Promise（传统方式）
function getUserData() {
  return fetch('/api/user')
    .then(response => response.json())
    .then(user => {
      console.log('用户：', user);
      return user;
    })
    .catch(error => {
      console.error('错误：', error);
    });
}

// 方式2：使用 async/await（现代方式，推荐）
async function getUserDataAsync() {
  try {
    // await 会等待 Promise 完成，然后返回结果
    const response = await fetch('/api/user');
    const user = await response.json();
    
    console.log('用户：', user);
    return user;
    
  } catch (error) {
    // 捕获任何步骤的错误
    console.error('错误：', error);
  }
}

// 💡 对比：
// Promise：需要 .then().then().catch()
// async/await：就像写同步代码一样，更直观
```

## 📝 async 函数基础

### 定义 async 函数

```javascript
// ===== 多种定义方式 =====

// 方式1：普通函数声明
async function func1() {
  return 'Hello';
}

// 方式2：函数表达式
const func2 = async function() {
  return 'Hello';
};

// 方式3：箭头函数
const func3 = async () => {
  return 'Hello';
};

// 方式4：对象方法
const obj = {
  async method() {
    return 'Hello';
  }
};

// 方式5：类方法
class MyClass {
  async method() {
    return 'Hello';
  }
}

// ===== async 函数的特点 =====

// 特点1：async 函数总是返回 Promise
async function greet() {
  return 'Hello';  // 即使返回普通值
}

const result = greet();
console.log(result);  // Promise { 'Hello' }

result.then(value => {
  console.log(value);  // Hello
});

// 特点2：等价于返回 Promise.resolve()
async function func1() {
  return 'Hello';
}

function func2() {
  return Promise.resolve('Hello');
}

// func1 和 func2 是等价的

// 特点3：抛出错误等价于 Promise.reject()
async function func3() {
  throw new Error('出错了');
}

function func4() {
  return Promise.reject(new Error('出错了'));
}

// func3 和 func4 是等价的
```

## ⏰ await 关键字详解

### await 的基本用法

```javascript
// ===== await 只能在 async 函数中使用 =====

// ❌ 错误：在普通函数中使用 await
function normalFunc() {
  const result = await somePromise();  // SyntaxError!
}

// ✅ 正确：在 async 函数中使用 await
async function asyncFunc() {
  const result = await somePromise();  // OK
}

// ===== await 会暂停执行，等待 Promise 完成 =====

async function example() {
  console.log('1. 开始');
  
  // await 会等待 Promise 完成（暂停这里）
  const result = await new Promise(resolve => {
    setTimeout(() => {
      resolve('结果');
    }, 1000);
  });
  
  console.log('3. 继续执行');
  console.log('结果：', result);
}

example();
console.log('2. async 函数外部继续执行');

// 输出顺序：
// 1. 开始
// 2. async 函数外部继续执行
// （等待1秒）
// 3. 继续执行
// 结果： 结果

// 💡 理解：
// - await 只暂停 async 函数内部的执行
// - 不会阻塞外部代码
// - 类似于在这里"等待"，但不阻塞主线程

// ===== await 可以用于任何 Promise =====

async function demo() {
  // 等待 fetch 请求
  const response = await fetch('/api/data');
  
  // 等待 JSON 解析
  const data = await response.json();
  
  // 等待定时器
  await new Promise(resolve => setTimeout(resolve, 1000));
  
  // 等待自定义 Promise
  const result = await myPromiseFunction();
  
  return result;
}
```

### await 的值

```javascript
// ===== await 返回 Promise 的结果 =====

async function example1() {
  // Promise.resolve() 的结果是 42
  const num = await Promise.resolve(42);
  console.log(num);  // 42（不是 Promise）
}

// ===== await 非 Promise 值 =====

async function example2() {
  // 如果 await 后面不是 Promise，会自动包装
  const num = await 42;  // 等价于 await Promise.resolve(42)
  console.log(num);  // 42
}

// ===== Promise 失败时，await 会抛出错误 =====

async function example3() {
  try {
    const result = await Promise.reject(new Error('失败'));
  } catch (error) {
    console.error('捕获错误：', error.message);  // 失败
  }
}
```

## 🎯 async/await 实战案例

### 案例1：串行请求（按顺序执行）

```javascript
// ===== 场景：需要按顺序获取数据 =====

// 需求：
// 1. 先获取用户信息
// 2. 根据用户ID获取订单
// 3. 根据订单ID获取订单详情

// Promise 版本
function getOrderDetails() {
  return fetch('/api/user')
    .then(res => res.json())
    .then(user => {
      return fetch(`/api/orders/${user.id}`);
    })
    .then(res => res.json())
    .then(orders => {
      const firstOrder = orders[0];
      return fetch(`/api/orders/${firstOrder.id}/details`);
    })
    .then(res => res.json())
    .then(details => {
      console.log('订单详情：', details);
      return details;
    })
    .catch(error => {
      console.error('获取失败：', error);
    });
}

// async/await 版本（更清晰）
async function getOrderDetailsAsync() {
  try {
    // 步骤1：获取用户
    const userResponse = await fetch('/api/user');
    const user = await userResponse.json();
    console.log('用户：', user.name);
    
    // 步骤2：获取订单列表
    const ordersResponse = await fetch(`/api/orders/${user.id}`);
    const orders = await ordersResponse.json();
    console.log('订单数量：', orders.length);
    
    // 步骤3：获取第一个订单的详情
    const firstOrder = orders[0];
    const detailsResponse = await fetch(`/api/orders/${firstOrder.id}/details`);
    const details = await detailsResponse.json();
    
    console.log('订单详情：', details);
    return details;
    
  } catch (error) {
    console.error('获取失败：', error);
    throw error;  // 继续抛出错误
  }
}

// 💡 优势：
// ✅ 代码像同步的，容易理解
// ✅ 每一步都很清晰
// ✅ 统一的错误处理
```

### 案例2：并行请求（同时执行）

```javascript
// ===== 场景：多个独立的请求，不需要等待 =====

// ❌ 错误方式：串行执行（慢）
async function getBadUserData(userId) {
  // 这三个请求互不依赖，但在等待
  const profile = await fetch(`/api/users/${userId}/profile`);
  const posts = await fetch(`/api/users/${userId}/posts`);      // 等上一个完成
  const friends = await fetch(`/api/users/${userId}/friends`);  // 等上一个完成
  
  // 总耗时 = 请求1时间 + 请求2时间 + 请求3时间
  // 假设每个请求1秒，总共需要3秒
  
  return {
    profile: await profile.json(),
    posts: await posts.json(),
    friends: await friends.json()
  };
}

// ✅ 正确方式：并行执行（快）
async function getGoodUserData(userId) {
  // 同时发起三个请求（不等待）
  const profilePromise = fetch(`/api/users/${userId}/profile`);
  const postsPromise = fetch(`/api/users/${userId}/posts`);
  const friendsPromise = fetch(`/api/users/${userId}/friends`);
  
  // 等待所有请求完成
  const [profile, posts, friends] = await Promise.all([
    profilePromise,
    postsPromise,
    friendsPromise
  ]);
  
  // 总耗时 = max(请求1时间, 请求2时间, 请求3时间)
  // 假设每个请求1秒，总共只需要1秒！
  
  return {
    profile: await profile.json(),
    posts: await posts.json(),
    friends: await friends.json()
  };
}

// ✅ 更简洁的写法
async function getBestUserData(userId) {
  const [profile, posts, friends] = await Promise.all([
    fetch(`/api/users/${userId}/profile`).then(r => r.json()),
    fetch(`/api/users/${userId}/posts`).then(r => r.json()),
    fetch(`/api/users/${userId}/friends`).then(r => r.json())
  ]);
  
  return { profile, posts, friends };
}

// 💡 性能对比：
// 串行：3秒（1+1+1）
// 并行：1秒（同时进行）
// 提升：3倍！
```

### 案例4：循环中使用 await

```javascript
// ===== 场景：批量处理数据 =====

// ❌ 错误方式：forEach 中使用 await
async function processItemsBad(items) {
  items.forEach(async (item) => {
    // ⚠️ forEach 不会等待 async 函数
    await processItem(item);
  });
  
  console.log('完成');  // 会立即执行，不等待处理完成
}

// ✅ 方式1：串行处理（for...of）
async function processItemsSerial(items) {
  for (const item of items) {
    // 依次处理，每个都等待完成
    await processItem(item);
    console.log(`处理完成：${item}`);
  }
  
  console.log('全部完成');
}

// 使用示例
await processItemsSerial([1, 2, 3, 4, 5]);
// 输出：
// 处理完成：1
// 处理完成：2
// 处理完成：3
// 处理完成：4
// 处理完成：5
// 全部完成

// ✅ 方式2：并行处理（Promise.all + map）
async function processItemsParallel(items) {
  // 同时处理所有项目
  await Promise.all(
    items.map(item => processItem(item))
  );
  
  console.log('全部完成');
}

// 使用示例
await processItemsParallel([1, 2, 3, 4, 5]);
// 所有项目同时处理，速度更快

// ✅ 方式3：限制并发数量
async function processItemsWithLimit(items, limit = 3) {
  const results = [];
  
  for (let i = 0; i < items.length; i += limit) {
    // 每次处理 limit 个
    const batch = items.slice(i, i + limit);
    const batchResults = await Promise.all(
      batch.map(item => processItem(item))
    );
    results.push(...batchResults);
    
    console.log(`批次 ${Math.floor(i / limit) + 1} 完成`);
  }
  
  return results;
}

// 使用示例：每次并发3个
await processItemsWithLimit([1,2,3,4,5,6,7,8,9], 3);
// 批次 1 完成（处理了1,2,3）
// 批次 2 完成（处理了4,5,6）
// 批次 3 完成（处理了7,8,9）
```

### 案例5：超时控制

```javascript
// ===== 为 async 函数添加超时限制 =====

function timeout(ms) {
  return new Promise((_, reject) => {
    setTimeout(() => {
      reject(new Error(`操作超时（${ms}ms）`));
    }, ms);
  });
}

async function fetchWithTimeout(url, ms = 5000) {
  try {
    // 竞速：fetch 和 timeout
    const response = await Promise.race([
      fetch(url),
      timeout(ms)
    ]);
    
    return await response.json();
    
  } catch (error) {
    if (error.message.includes('超时')) {
      console.error('请求超时');
    } else {
      console.error('请求失败：', error);
    }
    throw error;
  }
}

// 使用
try {
  const data = await fetchWithTimeout('/api/slow-data', 3000);
  console.log('数据：', data);
} catch (error) {
  console.error('最终错误：', error.message);
}

// ===== 通用的超时包装函数 =====

function withTimeout(promise, ms) {
  return Promise.race([
    promise,
    timeout(ms)
  ]);
}

// 使用
async function example() {
  // 为任何异步操作添加超时
  const data1 = await withTimeout(fetch('/api/data1'), 3000);
  const data2 = await withTimeout(longOperation(), 10000);
  const data3 = await withTimeout(
    Promise.all([
      fetch('/api/a'),
      fetch('/api/b')
    ]),
    5000
  );
}
```

## 📊 async/await vs Promise vs 回调对比

```javascript
// ===== 同一个功能的三种写法 =====

// 需求：读取文件，处理内容，保存结果

// 方式1：回调（传统）
fs.readFile('input.txt', 'utf8', (err, data) => {
  if (err) return console.error(err);
  
  const processed = processData(data);
  
  fs.writeFile('output.txt', processed, (err) => {
    if (err) return console.error(err);
    console.log('完成');
  });
});

// 方式2：Promise（改进）
fs.promises.readFile('input.txt', 'utf8')
  .then(data => processData(data))
  .then(processed => fs.promises.writeFile('output.txt', processed))
  .then(() => console.log('完成'))
  .catch(error => console.error(error));

// 方式3：async/await（最佳）
async function processFile() {
  try {
    const data = await fs.promises.readFile('input.txt', 'utf8');
    const processed = processData(data);
    await fs.promises.writeFile('output.txt', processed);
    console.log('完成');
  } catch (error) {
    console.error(error);
  }
}

processFile();

// 💡 结论：
// async/await 代码最清晰，最易维护！
```

## 🚀 实战项目：用户数据管理系统

```javascript
// ===== 完整案例：用户注册登录系统 =====

import fs from 'fs/promises';
import crypto from 'crypto';

// 工具函数：延迟
function delay(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// 工具函数：哈希密码
function hashPassword(password) {
  return crypto.createHash('sha256').update(password).digest('hex');
}

// ===== 用户管理类 =====
class UserManager {
  constructor(dbPath = './users.json') {
    this.dbPath = dbPath;
  }
  
  // 初始化数据库
  async init() {
    try {
      await fs.access(this.dbPath);
    } catch {
      // 文件不存在，创建空数据库
      await fs.writeFile(this.dbPath, JSON.stringify([]));
    }
  }
  
  // 读取所有用户
  async getAllUsers() {
    try {
      const content = await fs.readFile(this.dbPath, 'utf8');
      return JSON.parse(content);
    } catch (error) {
      console.error('读取用户数据失败：', error);
      return [];
    }
  }
  
  // 保存用户数据
  async saveUsers(users) {
    try {
      await fs.writeFile(
        this.dbPath,
        JSON.stringify(users, null, 2)
      );
      return true;
    } catch (error) {
      console.error('保存用户数据失败：', error);
      return false;
    }
  }
  
  // 注册新用户
  async register(username, password, email) {
    try {
      // 1. 验证输入
      if (!username || !password || !email) {
        throw new Error('用户名、密码和邮箱都是必填项');
      }
      
      // 2. 获取现有用户
      const users = await this.getAllUsers();
      
      // 3. 检查用户名是否已存在
      const existingUser = users.find(u => u.username === username);
      if (existingUser) {
        throw new Error('用户名已被注册');
      }
      
      // 4. 检查邮箱是否已存在
      const existingEmail = users.find(u => u.email === email);
      if (existingEmail) {
        throw new Error('邮箱已被注册');
      }
      
      // 5. 模拟发送验证邮件（异步操作）
      console.log(`正在发送验证邮件到 ${email}...`);
      await delay(1000);
      console.log('验证邮件已发送');
      
      // 6. 创建新用户
      const newUser = {
        id: Date.now().toString(),
        username,
        password: hashPassword(password),
        email,
        createdAt: new Date().toISOString(),
        isVerified: false
      };
      
      // 7. 保存用户
      users.push(newUser);
      await this.saveUsers(users);
      
      console.log(`✅ 用户 ${username} 注册成功！`);
      return { success: true, userId: newUser.id };
      
    } catch (error) {
      console.error(`❌ 注册失败：${error.message}`);
      return { success: false, error: error.message };
    }
  }
  
  // 用户登录
  async login(username, password) {
    try {
      // 1. 获取所有用户
      const users = await this.getAllUsers();
      
      // 2. 查找用户
      const user = users.find(u => u.username === username);
      if (!user) {
        throw new Error('用户名或密码错误');
      }
      
      // 3. 验证密码
      const hashedPassword = hashPassword(password);
      if (user.password !== hashedPassword) {
        throw new Error('用户名或密码错误');
      }
      
      // 4. 模拟生成会话令牌
      await delay(500);
      const token = crypto.randomBytes(32).toString('hex');
      
      console.log(`✅ 用户 ${username} 登录成功！`);
      return {
        success: true,
        token,
        user: {
          id: user.id,
          username: user.username,
          email: user.email
        }
      };
      
    } catch (error) {
      console.error(`❌ 登录失败：${error.message}`);
      return { success: false, error: error.message };
    }
  }
  
  // 获取用户信息
  async getUserInfo(userId) {
    try {
      const users = await this.getAllUsers();
      const user = users.find(u => u.id === userId);
      
      if (!user) {
        throw new Error('用户不存在');
      }
      
      // 不返回密码
      const { password, ...userInfo } = user;
      return { success: true, user: userInfo };
      
    } catch (error) {
      return { success: false, error: error.message };
    }
  }
  
  // 更新用户信息
  async updateUser(userId, updates) {
    try {
      const users = await this.getAllUsers();
      const userIndex = users.findIndex(u => u.id === userId);
      
      if (userIndex === -1) {
        throw new Error('用户不存在');
      }
      
      // 不允许更新某些字段
      const { id, password, ...allowedUpdates } = updates;
      
      users[userIndex] = {
        ...users[userIndex],
        ...allowedUpdates,
        updatedAt: new Date().toISOString()
      };
      
      await this.saveUsers(users);
      
      console.log(`✅ 用户 ${userId} 信息已更新`);
      return { success: true };
      
    } catch (error) {
      console.error(`❌ 更新失败：${error.message}`);
      return { success: false, error: error.message };
    }
  }
  
  // 删除用户
  async deleteUser(userId) {
    try {
      const users = await this.getAllUsers();
      const newUsers = users.filter(u => u.id !== userId);
      
      if (newUsers.length === users.length) {
        throw new Error('用户不存在');
      }
      
      await this.saveUsers(newUsers);
      
      console.log(`✅ 用户 ${userId} 已删除`);
      return { success: true };
      
    } catch (error) {
      console.error(`❌ 删除失败：${error.message}`);
      return { success: false, error: error.message };
    }
  }
  
  // 批量导入用户
  async batchImport(users) {
    const results = {
      success: [],
      failed: []
    };
    
    for (const user of users) {
      try {
        const result = await this.register(
          user.username,
          user.password,
          user.email
        );
        
        if (result.success) {
          results.success.push(user.username);
        } else {
          results.failed.push({
            username: user.username,
            error: result.error
          });
        }
        
        // 避免过快请求
        await delay(100);
        
      } catch (error) {
        results.failed.push({
          username: user.username,
          error: error.message
        });
      }
    }
    
    console.log(`\n批量导入完成：`);
    console.log(`✅ 成功：${results.success.length} 个`);
    console.log(`❌ 失败：${results.failed.length} 个`);
    
    return results;
  }
}
```

## 🎓 常见面试题

### 1. async 函数返回什么？

```javascript
// ===== 问题：async 函数返回什么类型？=====

async function test() {
  return 42;
}

const result = test();
console.log(result);  // 输出什么？

// 答案：Promise { 42 }
// 解释：async 函数总是返回 Promise

// ===== 如何获取值？=====

// 方式1：使用 .then()
test().then(value => console.log(value));  // 42

// 方式2：使用 await
const value = await test();
console.log(value);  // 42
```

### 2. await 可以在哪里使用？

```javascript
// ===== 问题：await 可以在哪些地方使用？=====

// ❌ 1. 不能在普通函数中
function normalFunc() {
  const result = await somePromise();  // SyntaxError
}

// ✅ 2. 可以在 async 函数中
async function asyncFunc() {
  const result = await somePromise();  // OK
}

// ✅ 3. 可以在模块顶层（ESM + Node.js 14.8+）
// main.js
const data = await loadData();  // OK（需要 "type": "module"）

// ❌ 4. 不能在回调函数中（除非回调是 async）
[1, 2, 3].forEach(async (item) => {
  await process(item);  // 可以写，但不会等待！
});

// ✅ 5. 正确的循环方式
for (const item of [1, 2, 3]) {
  await process(item);  // 会正确等待
}
```

### 3. 如何并行执行多个异步操作？

```javascript
// ===== 问题：如何让多个异步操作同时执行？=====

// ❌ 错误：串行执行（慢）
async function slow() {
  const a = await fetch('/api/a');  // 等待1秒
  const b = await fetch('/api/b');  // 等待1秒
  const c = await fetch('/api/c');  // 等待1秒
  // 总共：3秒
}

// ✅ 正确：并行执行（快）
async function fast() {
  const [a, b, c] = await Promise.all([
    fetch('/api/a'),
    fetch('/api/b'),
    fetch('/api/c')
  ]);
  // 总共：1秒（最慢的那个）
}

// ✅ 另一种写法
async function fast2() {
  const aPromise = fetch('/api/a');
  const bPromise = fetch('/api/b');
  const cPromise = fetch('/api/c');
  
  const a = await aPromise;
  const b = await bPromise;
  const c = await cPromise;
}
```

### 4. try/catch 能捕获所有错误吗？

```javascript
// ===== 问题：try/catch 能捕获异步错误吗？=====

// ❌ 不能捕获未 await 的 Promise 错误
async function test1() {
  try {
    // 忘记 await，错误不会被捕获！
    fetch('/api/data');  // 如果失败，会是未捕获的 Promise rejection
  } catch (error) {
    console.log('捕获：', error);  // 不会执行
  }
}

// ✅ 正确：await Promise
async function test2() {
  try {
    await fetch('/api/data');  // 错误会被捕获
  } catch (error) {
    console.log('捕获：', error);  // 会执行
  }
}

// ❌ 不能捕获回调中的错误
async function test3() {
  try {
    setTimeout(() => {
      throw new Error('回调错误');  // 不会被捕获
    }, 1000);
  } catch (error) {
    console.log('捕获：', error);  // 不会执行
  }
}

// ✅ 正确：使用 Promise 包装
async function test4() {
  try {
    await new Promise((resolve, reject) => {
      setTimeout(() => {
        reject(new Error('回调错误'));  // 会被捕获
      }, 1000);
    });
  } catch (error) {
    console.log('捕获：', error);  // 会执行
  }
}
```

### 5. async/await 的执行顺序

```javascript
// ===== 问题：以下代码的执行顺序是什么？=====

console.log('1');

async function test() {
  console.log('2');
  await Promise.resolve();
  console.log('4');
}

test();
console.log('3');

// 答案：1 → 2 → 3 → 4

// 解释：
// 1. 同步代码：console.log('1')
// 2. 调用 test()，执行到 await 之前：console.log('2')
// 3. await 之后的代码进入微任务队列
// 4. 继续执行外部代码：console.log('3')
// 5. 微任务执行：console.log('4')
```

```javascript
// ===== 1. 使用命名函数，便于调试 =====

// ===== 2. 添加详细的错误信息 =====

async function betterFetch(url) {
  try {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(
        `请求失败：${url}\n` +
        `状态码：${response.status}\n` +
        `状态文本：${response.statusText}`
      );
    }
    
    return await response.json();
    
  } catch (error) {
    // 添加上下文信息
    error.url = url;
    error.timestamp = new Date().toISOString();
    throw error;
  }
}

async function fetchWithLog(url) {
  console.log(`[${new Date().toISOString()}] 开始请求：${url}`);
  
  const startTime = Date.now();
  
  try {
    const response = await fetch(url);
    const data = await response.json();
    
    const duration = Date.now() - startTime;

    const duration = Date.now() - startTime;
    console.error(`[${new Date().toISOString()}] 请求失败：${url}（${duration}ms）`, error);
    throw error;
  }
}
```

## 📚 总结

### async/await 核心要点

| 特性 | 说明 |
|------|------|
| **本质** | Promise 的语法糖 |
| **async 函数** | 总是返回 Promise |
| **await 关键字** | 等待 Promise 完成，返回结果 |
| **错误处理** | 使用 try/catch |
| **并行执行** | 使用 Promise.all() |
| **兼容性** | Node.js 7.6+ |

### 使用场景

✅ **适合使用 async/await：**
- 需要按顺序执行的异步操作
- 需要清晰的错误处理
- 代码可读性要求高
- 复杂的异步流程控制

⚠️ **需要注意：**
- 并行操作使用 Promise.all()
- 循环中注意串行/并行
- 不要忘记 await
- 错误处理不要遗漏

### 性能优化建议

1. **并行 > 串行**：能并行就并行
2. **适当超时**：避免无限等待
3. **错误恢复**：提供降级方案
4. **资源清理**：使用 finally
