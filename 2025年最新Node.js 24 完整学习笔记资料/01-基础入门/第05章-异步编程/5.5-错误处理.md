# 5.5 异步错误处理完全指南

> 正确处理异步错误，让程序健壮不崩溃

---

## 🎯 为什么异步错误处理很重要？

### 核心概念

```
异步错误 = 未来某个时刻才发生的错误

同步错误：立即发生，try/catch 能捕获
异步错误：延迟发生，普通 try/catch 捕获不到

危险：
❌ 未捕获的异步错误会导致程序崩溃
❌ 错误信息丢失，难以调试
❌ 用户体验差

正确处理：
✅ 所有异步操作都有错误处理
✅ 错误信息清晰可追踪
✅ 优雅降级，不影响其他功能
```

## 🔍 三种异步模式的错误处理

### 1. 回调函数的错误处理

```javascript
// ===== Node.js 约定：错误优先回调 =====

// ✅ 正确方式：第一个参数是错误
fs.readFile('file.txt', 'utf8', (err, data) => {
  // 先检查错误
  if (err) {
    console.error('读取文件失败：', err.message);
    return;  // 不要继续执行
  }
  
  // 错误检查通过，处理数据
  console.log('文件内容：', data);
});

// ===== 回调地狱的错误处理 =====

// ❌ 问题：每一步都要检查错误，代码很乱
fs.readFile('input.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('读取失败：', err);
    return;
  }
  
  processData(data, (err, result) => {
    if (err) {
      console.error('处理失败：', err);
      return;
    }
    
    fs.writeFile('output.txt', result, (err) => {
      if (err) {
        console.error('写入失败：', err);
        return;
      }
      
      console.log('完成');
    });
  });
});
```

### 2. Promise 的错误处理

```javascript
// ===== 方式1：.catch() 方法 =====

fetch('/api/data')
  .then(response => response.json())
  .then(data => {
    console.log('数据：', data);
  })
  .catch(error => {
    // 捕获任何步骤的错误
    console.error('获取数据失败：', error);
  });

// ===== 方式2：链式 catch（精细控制）=====

fetch('/api/data')
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP错误：${response.status}`);
    }
    return response.json();
  })
  .catch(error => {
    // 只捕获上面的错误
    console.error('请求或解析失败：', error);
    return { default: true };  // 返回默认值，继续执行
  })
  .then(data => {
    // 这里一定会执行
    console.log('数据（可能是默认值）：', data);
  });

// ===== 方式3：finally（清理资源）=====

let connection;

connect()
  .then(conn => {
    connection = conn;
    return connection.query('SELECT * FROM users');
  })
  .then(results => {
    console.log('查询结果：', results);
  })
  .catch(error => {
    console.error('操作失败：', error);
  })
  .finally(() => {
    // 无论成功失败，都执行清理
    if (connection) {
      connection.close();
      console.log('连接已关闭');
    }
  });

// ===== 错误传播（Error Propagation）=====

function step1() {
  return Promise.reject(new Error('步骤1失败'));
}

function step2() {
  return Promise.resolve('步骤2成功');
}

step1()
  .then(() => step2())  // 不会执行
  .then(() => {          // 不会执行
    console.log('完成');
  })
  .catch(error => {
    // 错误会一直传播到这里
    console.error('捕获错误：', error.message);  // 步骤1失败
  });

// 💡 理解：
// Promise 链中的错误会"跳过"后续的 .then()
// 直到遇到 .catch() 才被捕获
```

### 3. async/await 的错误处理

```javascript
// ===== 方式1：try/catch（最常用）=====

async function fetchData() {
  try {
    const response = await fetch('/api/data');
    
    if (!response.ok) {
      throw new Error(`HTTP错误：${response.status}`);
    }
    
    const data = await response.json();
    console.log('数据：', data);
    return data;
    
  } catch (error) {
    // 捕获任何步骤的错误
    console.error('获取数据失败：', error.message);
    
    // 可以返回默认值
    return null;
    
    // 或者继续抛出
    // throw error;
  }
}

// ===== 方式2：多个 try/catch（精细控制）=====

async function complexOperation() {
  let user;
  let orders;
  
  // 第一步：获取用户
  try {
    user = await fetchUser();
  } catch (error) {
    console.error('获取用户失败，使用默认用户');
    user = { id: 0, name: 'Guest' };
  }
  
  // 第二步：获取订单（即使用户获取失败也继续）
  try {
    orders = await fetchOrders(user.id);
  } catch (error) {
    console.error('获取订单失败，返回空列表');
    orders = [];
  }
  
  return { user, orders };
}

// ===== 方式3：try/catch/finally =====

async function uploadFile(file) {
  let connection;
  
  try {
    // 建立连接
    connection = await createConnection();
    console.log('连接建立');
    
    // 上传文件
    await connection.upload(file);
    console.log('上传成功');
    
    return { success: true };
    
  } catch (error) {
    console.error('上传失败：', error);
    return { success: false, error: error.message };
    
  } finally {
    // 无论成功失败，都关闭连接
    if (connection) {
      await connection.close();
      console.log('连接已关闭');
    }
  }
}

// 执行流程示例：
// 1. 建立连接
// 2. 上传文件（假设成功）
// 3. 输出：上传成功
// 4. finally：连接已关闭
// 5. 返回：{ success: true }

// 💡 理解 finally：
// ✅ 总是执行
// ✅ 在 return 之前执行
// ✅ 适合清理资源（关闭连接、删除临时文件等）
```

```javascript
// ===== 2. 未处理的 Promise 拒绝 =====

process.on('unhandledRejection', (reason, promise) => {
  console.error('未处理的Promise拒绝：', reason);
  console.error('Promise：', promise);
  
  // 记录错误
  logError(reason);
  
  // ⚠️ Node.js 15+ 会自动退出进程
  // 早期版本不会，需要手动处理

// ===== 实战示例：完整的错误处理系统 =====

class ErrorHandler {
  constructor() {
    this.setupHandlers();
  }
  
  setupHandlers() {
    // 未捕获的异常
    process.on('uncaughtException', (error) => {
      this.handleFatalError('uncaughtException', error);
    });
    
    // 未处理的 Promise 拒绝
    process.on('unhandledRejection', (reason, promise) => {
      this.handleFatalError('unhandledRejection', reason);
    });
    
    // 警告（如废弃的 API）
    process.on('warning', (warning) => {
      console.warn('⚠️ 警告：', warning.name);
      console.warn('消息：', warning.message);
      console.warn('堆栈：', warning.stack);
    });
  }
  
  handleFatalError(type, error) {
    console.error(`\n❌ 致命错误 [${type}]：`);
    console.error('错误：', error);
    console.error('堆栈：', error.stack);
    console.error('时间：', new Date().toISOString());
    
    // 记录到日志文件
    this.logToFile(type, error);
    
    // 发送告警（邮件、钉钉、短信等）
    this.sendAlert(type, error);
    
    // 优雅退出
    this.gracefulShutdown();
  }
  
  logToFile(type, error) {
    const log = {
      type,
      message: error.message,
      stack: error.stack,
      timestamp: new Date().toISOString()
    };
    
    fs.appendFileSync(
      'error.log',
      JSON.stringify(log, null, 2) + '\n'
    );
  }
  
  sendAlert(type, error) {
    // 实现告警逻辑
    console.log('📧 发送告警邮件...');
  }
  
  gracefulShutdown() {
    console.log('正在优雅退出...');
    
    // 停止接受新请求
    if (global.server) {
      global.server.close(() => {
        console.log('服务器已关闭');
        process.exit(1);
      });
    }
    
    // 10秒后强制退出
    setTimeout(() => {
      console.error('强制退出');
      process.exit(1);
    }, 10000);
  }
}

// 启动错误处理
const errorHandler = new ErrorHandler();
```

### 案例1：HTTP 请求的完整错误处理

```javascript
// ===== 封装一个健壮的 fetch 函数 =====

async function robustFetch(url, options = {}) {
  const {
    timeout = 5000,
    retries = 3,
    retryDelay = 1000
  } = options;
  
  // 尝试多次
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      console.log(`请求 ${url}，尝试 ${attempt}/${retries}`);
      
      // 添加超时
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), timeout);
      
      try {
        const response = await fetch(url, {
          ...options,
          signal: controller.signal
        });
        
        clearTimeout(timeoutId);
        
        // 检查 HTTP 状态
        if (!response.ok) {
          throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        // 尝试解析 JSON
        const data = await response.json().catch(() => {
          throw new Error('响应不是有效的 JSON');
        });
        
        console.log(`✅ 请求成功`);
        return { success: true, data };
        
      } finally {
        clearTimeout(timeoutId);
      }
      
    } catch (error) {
      console.error(`❌ 尝试 ${attempt} 失败：${error.message}`);
      
      // 最后一次尝试失败
      if (attempt === retries) {
        return {
          success: false,
          error: {
            message: error.message,
            attempts: retries
          }
        };
      }
      
      // 等待后重试
      console.log(`等待 ${retryDelay}ms 后重试...`);
      await new Promise(resolve => setTimeout(resolve, retryDelay));
    }
  }
}

// 使用
const result = await robustFetch('/api/data', {
  timeout: 3000,
  retries: 3,
  retryDelay: 1000
});

if (result.success) {
  console.log('数据：', result.data);
} else {
  console.error('请求失败：', result.error);
}
```

```javascript
async function queryDatabase(sql, params) {

    // 从连接池获取连接
```

```javascript
class AppError extends Error {

    super(message);
    this.name = this.constructor.name;

class NotFoundError extends AppError {

    super(message, 400);
  }
}

class AuthenticationError extends AppError {
  constructor(message = '未认证') {
    super(message, 401);
  }

  }
```

## 🎓 常见面试题

### 1. try/catch 能捕获所有错误吗？

```javascript
// 问题：以下代码能捕获错误吗？

// 情况1：同步错误
try {
  throw new Error('同步错误');
} catch (error) {
  console.log('捕获：', error.message);  // ✅ 能捕获
}

// 情况2：异步回调中的错误
try {
  setTimeout(() => {
    throw new Error('回调错误');  // ❌ 不能捕获！
  }, 1000);
} catch (error) {
  console.log('捕获：', error.message);  // 不会执行
}

// 情况3：Promise 错误（未 await）
try {
  Promise.reject(new Error('Promise错误'));  // ❌ 不能捕获！
} catch (error) {
  console.log('捕获：', error.message);  // 不会执行
}

// 情况4：await Promise 错误
async function test() {
  try {
    await Promise.reject(new Error('Promise错误'));  // ✅ 能捕获
  } catch (error) {
    console.log('捕获：', error.message);
  }
}

// 答案：
// try/catch 只能捕获：
// ✅ 同步代码的错误
// ✅ await 的 Promise 错误
// ❌ 异步回调的错误
// ❌ 未 await 的 Promise 错误
```

## 🎯 实战：构建完整的错误处理系统

```javascript
// ===== 完整的企业级错误处理 =====

import fs from 'fs/promises';
import { EventEmitter } from 'events';

// 1. 错误类型定义
class AppError extends Error {
  constructor(message, code, statusCode = 500, meta = {}) {
    super(message);
    this.name = this.constructor.name;
    this.code = code;
    this.statusCode = statusCode;
    this.meta = meta;
    this.timestamp = new Date().toISOString();
    Error.captureStackTrace(this, this.constructor);
  }
}

class ValidationError extends AppError {
  constructor(message, meta) {
    super(message, 'VALIDATION_ERROR', 400, meta);
  }
}

class NotFoundError extends AppError {
  constructor(resource, meta) {
    super(`${resource} 不存在`, 'NOT_FOUND', 404, meta);
  }
}

// 2. 错误处理器
class ErrorHandler extends EventEmitter {
  constructor(options = {}) {
    super();
    this.logFile = options.logFile || 'error.log';
    this.alertThreshold = options.alertThreshold || 10;
    this.errorCount = new Map();  // 错误计数
    this.setupGlobalHandlers();
  }
  
  setupGlobalHandlers() {
    process.on('uncaughtException', (error) => {
      this.handleFatalError('uncaughtException', error);
    });
    
    process.on('unhandledRejection', (reason) => {
      this.handleFatalError('unhandledRejection', reason);
    });
  }
  
  async handle(error) {
    // 1. 记录错误
    await this.log(error);
    
    // 2. 统计错误
    this.countError(error);
    
    // 3. 判断是否需要告警
    if (this.shouldAlert(error)) {
      await this.alert(error);
    }
    
    // 4. 触发事件
    this.emit('error', error);
    
    return this.formatError(error);
  }
  
  async log(error) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      name: error.name,
      code: error.code,
      message: error.message,
      stack: error.stack,
      meta: error.meta
    };
    
    // 写入文件
    await fs.appendFile(
      this.logFile,
      JSON.stringify(logEntry) + '\n'
    );
    
    // 控制台输出
    console.error(`[${logEntry.timestamp}] ${error.name}: ${error.message}`);
  }
  
  countError(error) {
    const key = error.code || error.name;
    const count = this.errorCount.get(key) || 0;
    this.errorCount.set(key, count + 1);
  }
  
  shouldAlert(error) {
    const key = error.code || error.name;
    const count = this.errorCount.get(key) || 0;
    
    // 同类错误超过阈值，发送告警
    return count >= this.alertThreshold;
  }
  
  async alert(error) {
    // 实现告警逻辑（邮件、短信、钉钉等）
    console.log(`🚨 告警：${error.name} 错误频繁发生`);
    
    // 重置计数
    const key = error.code || error.name;
    this.errorCount.set(key, 0);
  }
  
  formatError(error) {
    if (error instanceof AppError) {
      return {
        code: error.code,
        message: error.message,
        statusCode: error.statusCode,
        meta: error.meta
      };
    }
    
    return {
      code: 'INTERNAL_ERROR',
      message: '服务器内部错误',
      statusCode: 500
    };
  }
  
  handleFatalError(type, error) {
    console.error(`\n💀 致命错误 [${type}]：`);
    console.error(error);
    
    this.log(error).then(() => {
      process.exit(1);
    });
  }
}

// 3. 使用错误处理器
const errorHandler = new ErrorHandler({
  logFile: 'logs/error.log',
  alertThreshold: 5
});

// 监听错误事件
errorHandler.on('error', (error) => {
  // 可以做额外的处理
  console.log('错误已记录');
});

// 在业务代码中使用
async function createUser(userData) {
  try {
    // 验证数据
    if (!userData.email || !userData.email.includes('@')) {
      throw new ValidationError('邮箱格式不正确', { 
        field: 'email',
        value: userData.email 
      });
    }
    
    // 检查用户是否存在
    const existingUser = await db.findUser(userData.email);
    if (existingUser) {
      throw new ValidationError('用户已存在', {
        field: 'email',
        value: userData.email
      });
    }
    
    // 创建用户
    const user = await db.createUser(userData);
    return user;
    
  } catch (error) {
    // 统一错误处理
    const formattedError = await errorHandler.handle(error);
    throw formattedError;
  }
}
```

## 🎓 总结

| 场景 | 推荐方法 | 示例 |
|------|----------|------|
| **async/await** | try/catch | `try { await fn() } catch (e) {}` |
| **Promise** | .catch() | `promise.catch(e => {})` |
| **批量操作** | Promise.allSettled | `await Promise.allSettled(tasks)` |
| **清理资源** | finally | `try {} catch {} finally {}` |
| **全局捕获** | process.on() | `process.on('unhandledRejection')` |
| **自定义错误** | 继承 Error | `class MyError extends Error {}` |

### 核心要点

1. **总是处理错误**：未处理的错误会导致程序崩溃
2. **提供上下文**：错误信息要能帮助定位问题
3. **分层处理**：业务层、服务层、数据层分别处理
4. **优雅降级**：部分功能失败不影响整体
5. **记录日志**：所有错误都要留下痕迹
6. **监控告警**：频繁错误需要及时发现
7. **用户友好**：向用户展示友好的错误信息

### 错误处理金字塔

```
           🔴 致命错误
          (uncaughtException)
               ↓ 记录并退出
               
        🟠 业务错误
       (ValidationError)
          ↓ 返回错误信息
          
     🟡 可恢复错误
    (NetworkTimeout)
       ↓ 重试或降级
       
   🟢 预期错误
  (NotFound)
    ↓ 正常处理
```
