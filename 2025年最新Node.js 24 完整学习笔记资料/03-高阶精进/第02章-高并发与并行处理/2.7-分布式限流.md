## 📖 一、理论知识

### 1.1 为什么需要分布式限流？

**单机限流的局限**：

```
场景：部署 10 个 Node.js 实例

单机限流（100 req/s/实例）:
┌────────┐ 100 req/s
│ 实例 1 │ ───────→ 总 QPS = 1000 req/s ❌ 超出预期
│ 实例 2 │ 100 req/s
│ 实例 3 │ 100 req/s
│  ...   │
│实例 10 │ 100 req/s
└────────┘

分布式限流（总共 100 req/s）:
┌────────┐
│ Redis  │ ← 全局计数器
└────┬───┘
     │ 所有实例共享配额
     ↓
10 个实例平均每个 10 req/s ✅
```

### 1.2 四种限流算法

| 算法 | 原理 | 优点 | 缺点 | 适用场景 |
|-----|------|------|------|---------|
| **固定窗口** | 每 N 秒重置计数 | 简单高效 | 边界突刺流量 | 粗粒度限流 |
| **滑动窗口** | 滑动时间窗口 | 精确平滑 | 内存占用高 | 精确限流 |
| **令牌桶** | 固定速率生成令牌 | 允许突发 | 实现复杂 | 流量整形 |
| **漏桶** | 固定速率处理 | 流量平滑 | 无法应对突发 | 流量整形 |

### 1.3 滑动窗口算法详解

```
滑动窗口示例（限制 10 req/10s）:

时间轴:  0s    2s    4s    6s    8s   10s   12s
         ├─────┼─────┼─────┼─────┼─────┼─────┤
请求:    3     2     1     4     2     3     1

当前时间 12s，窗口 [2s, 12s]:
- 2s: 2 个请求
- 4s: 1 个请求
- 6s: 4 个请求
- 8s: 2 个请求
- 10s: 3 个请求
- 12s: 1 个请求
总计: 13 个请求 > 10 → 拒绝 ❌

当前时间 13s，窗口 [3s, 13s]:
- 4s: 1 个请求
- 6s: 4 个请求
- 8s: 2 个请求
- 10s: 3 个请求
- 12s: 1 个请求
总计: 11 个请求 > 10 → 拒绝 ❌

当前时间 14s，窗口 [4s, 14s]:
- 6s: 4 个请求
- 8s: 2 个请求
- 10s: 3 个请求
- 12s: 1 个请求
总计: 10 个请求 = 10 → 允许 ✅
```

const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');

const redisClient = new Redis({
  host: 'localhost',
  port: 6379,
  enableOfflineQueue: false
});

// 创建限流器
const rateLimiter = new RateLimiterRedis({
  storeClient: redisClient,
  keyPrefix: 'rlim',
  points: 100,      // 100 个请求
  duration: 60,     // 60 秒内
  blockDuration: 0  // 不阻塞（立即返回错误）
});

// Express 中间件
async function rateLimitMiddleware(req, res, next) {
  try {
    // 使用 IP 作为限流 key
    const key = req.ip;
    
    await rateLimiter.consume(key, 1);  // 消耗 1 个配额
    
    next();
  } catch (error) {
    if (error instanceof Error) {
      // 其他错误
      return res.status(500).json({ error: 'Rate limiter error' });
    }
    
    // 限流错误
    res.status(429).json({
      error: 'Too Many Requests',
      retryAfter: Math.ceil(error.msBeforeNext / 1000)
    });
  }
}

module.exports = rateLimitMiddleware;

### 2.2 方法2：Redis + Lua 脚本（滑动窗口）

#### 实现原理

```lua
-- 滑动窗口限流 Lua 脚本
-- KEYS[1]: 限流 key
-- ARGV[1]: 窗口大小（秒）
-- ARGV[2]: 限制数量
-- ARGV[3]: 当前时间戳（毫秒）

local key = KEYS[1]
local window = tonumber(ARGV[1])
local limit = tonumber(ARGV[2])
local now = tonumber(ARGV[3])

-- 窗口起始时间
local windowStart = now - (window * 1000)

-- 删除窗口外的数据
redis.call('ZREMRANGEBYSCORE', key, 0, windowStart)

-- 统计窗口内的请求数
local current = redis.call('ZCARD', key)

if current < limit then
  -- 未超限，添加当前请求
  redis.call('ZADD', key, now, now)
  redis.call('EXPIRE', key, window)
  return {1, limit - current - 1}  -- [允许, 剩余配额]
else
  -- 超限
  return {0, 0}
end
```

#### Node.js 实现

```javascript
// limiter/slidingWindow.js
const Redis = require('ioredis');
const fs = require('fs');
const path = require('path');

class SlidingWindowLimiter {
  constructor(options = {}) {
    this.redis = new Redis(options.redis);
    this.window = options.window || 60;  // 60 秒
    this.limit = options.limit || 100;   // 100 次
    
    // 加载 Lua 脚本
    this.luaScript = fs.readFileSync(
      path.join(__dirname, 'slidingWindow.lua'),
      'utf8'
    );
  }
  
  async checkLimit(key) {
    const now = Date.now();
    
    const result = await this.redis.eval(
      this.luaScript,
      1,  // KEYS 数量
      `limiter:${key}`,  // KEYS[1]
      this.window,       // ARGV[1]
      this.limit,        // ARGV[2]
      now               // ARGV[3]
    );
    
    return {
      allowed: result[0] === 1,
      remaining: result[1]
    };
  }
  
  async middleware(req, res, next) {
    const key = this.getKey(req);
    const result = await this.checkLimit(key);
    
    // 设置响应头
    res.setHeader('X-RateLimit-Limit', this.limit);
    res.setHeader('X-RateLimit-Remaining', result.remaining);
    res.setHeader('X-RateLimit-Window', this.window);
    
    if (result.allowed) {
      next();
    } else {
      res.status(429).json({
        error: 'Too Many Requests',
        message: `Rate limit exceeded. Try again in ${this.window} seconds.`
      });
    }
  }
  
  getKey(req) {
    // 可以根据不同维度限流
    return req.ip;  // 按 IP
    // return req.user?.id;  // 按用户
    // return req.headers['api-key'];  // 按 API Key
  }
}

module.exports = SlidingWindowLimiter;
```

### 2.3 不同维度的限流

#### 按 API 接口限流

```javascript
// limiter/apiLimiter.js
const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');

const redis = new Redis();

class APILimiter {
  constructor() {
    this.limiters = new Map();
  }
  
  createLimiter(route, config) {
    const limiter = new RateLimiterRedis({
      storeClient: redis,
      keyPrefix: `api:${route}`,
      points: config.limit,
      duration: config.window
    });
    
    this.limiters.set(route, limiter);
    return limiter;
  }
  
  async middleware(req, res, next) {
    const route = req.route?.path || req.path;
    let limiter = this.limiters.get(route);
    
    if (!limiter) {
      // 默认限流配置
      limiter = this.createLimiter(route, { limit: 100, window: 60 });
    }
    
    try {
      const key = `${req.ip}:${route}`;
      const result = await limiter.consume(key);
      
      res.setHeader('X-RateLimit-Remaining', result.remainingPoints);
      next();
    } catch (error) {
      res.status(429).json({ error: 'Rate limit exceeded' });
    }
  }
}

// 使用
const apiLimiter = new APILimiter();

// 不同接口不同限制
apiLimiter.createLimiter('/api/search', { limit: 10, window: 60 });    // 10 req/min
apiLimiter.createLimiter('/api/data', { limit: 100, window: 60 });      // 100 req/min
apiLimiter.createLimiter('/api/upload', { limit: 5, window: 3600 });    // 5 req/hour

app.use((req, res, next) => apiLimiter.middleware(req, res, next));
```

### 2.4 实战示例：秒杀系统限流

```javascript
// limiter/seckill.js
const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');

class SeckillLimiter {
  constructor() {
    this.redis = new Redis();
    
    // 全局限流（保护系统）
    this.globalLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'seckill:global',
      points: 10000,     // 系统最多 10000 req/s
      duration: 1
    });
    
    // 用户限流（防止刷单）
    this.userLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'seckill:user',
      points: 10,        // 用户最多 10 次/分钟
      duration: 60,
      blockDuration: 300 // 超限后封禁 5 分钟
    });
    
    // IP 限流（防止攻击）
    this.ipLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'seckill:ip',
      points: 100,       // IP 最多 100 req/min
      duration: 60,
      blockDuration: 600 // 超限后封禁 10 分钟
    });
  }
  
  async checkLimits(req) {
    try {
      // 1. 全局限流
      await this.globalLimiter.consume('global', 1);
      
      // 2. 用户限流
      if (req.user) {
        await this.userLimiter.consume(req.user.id, 1);
      }
      
      // 3. IP 限流
      await this.ipLimiter.consume(req.ip, 1);
      
      return { allowed: true };
    } catch (error) {
      if (error.msBeforeNext) {
        return {
          allowed: false,
          type: 'rate_limit',
          retryAfter: Math.ceil(error.msBeforeNext / 1000)
        };
      }
      
      return {
        allowed: false,
        type: 'blocked',
        message: 'You have been temporarily blocked'
      };
    }
  }
  
  async middleware(req, res, next) {
    const result = await this.checkLimits(req);
    
    if (result.allowed) {
      next();
    } else {
      if (result.type === 'blocked') {
        res.status(403).json({ error: result.message });
      } else {
        res.status(429).json({
          error: 'Too Many Requests',
          retryAfter: result.retryAfter
        });
      }
    }
  }
}

module.exports = new SeckillLimiter();
```

  try {

    next();
  } catch (error) {
    if (error.name === 'ReplyError' || error.message.includes('ECONNREFUSED')) {

## 🎯 四、2个应用场景

### 场景1：API 网关限流

#### 需求背景

- API 网关对接 100+ 后端服务
- 不同 API 不同限流策略
- 支持用户级别限流

#### 完整实现

```javascript
// gateway/rateLimiter.js
const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');
const yaml = require('js-yaml');
const fs = require('fs');

class GatewayRateLimiter {
  constructor(configFile) {
    this.redis = new Redis({ enableOfflineQueue: false });
    this.config = yaml.load(fs.readFileSync(configFile, 'utf8'));
    this.limiters = new Map();
    
    this.initLimiters();
  }
  
  initLimiters() {
    // 全局限流
    this.globalLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'global',
      points: this.config.global.points,
      duration: this.config.global.duration
    });
    
    // API 级别限流
    this.config.apis.forEach(api => {
      const limiter = new RateLimiterRedis({
        storeClient: this.redis,
        keyPrefix: `api:${api.path}`,
        points: api.limit,
        duration: api.window
      });
      
      this.limiters.set(api.path, {
        limiter,
        config: api
      });
    });
  }
  
  async checkLimits(req) {
    const errors = [];
    
    // 1. 全局限流
    try {
      await this.globalLimiter.consume('global');
    } catch (error) {
      errors.push({ type: 'global', error });
    }
    
    // 2. API 限流
    const route = req.route?.path || req.path;
    const apiLimiter = this.limiters.get(route);
    
    if (apiLimiter) {
      try {
        const key = this.getKey(req, apiLimiter.config);
        await apiLimiter.limiter.consume(key);
      } catch (error) {
        errors.push({ type: 'api', error });
      }
    }
    
    return errors;
  }
  
  getKey(req, config) {
    switch (config.key) {
      case 'ip':
        return req.ip;
      case 'user':
        return req.user?.id || req.ip;
      case 'apikey':
        return req.headers['x-api-key'] || req.ip;
      default:
        return req.ip;
    }
  }
  
  async middleware(req, res, next) {
    const errors = await this.checkLimits(req);
    
    if (errors.length > 0) {
      const firstError = errors[0];
      
      res.status(429).json({
        error: 'Rate limit exceeded',
        type: firstError.type,
        retryAfter: Math.ceil(firstError.error.msBeforeNext / 1000)
      });
    } else {
      next();
    }
  }
}

module.exports = GatewayRateLimiter;
```

### 场景2：短信验证码防刷

#### 需求背景

- 防止恶意刷短信验证码
- 多层限流：IP + 手机号 + 全局
- 成本控制：每天最多 10000 条

#### 完整实现

```javascript
// limiter/smsLimiter.js
const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');

class SMSLimiter {
  constructor() {
    this.redis = new Redis();
    
    // IP 限流（防止单 IP 批量发送）
    this.ipLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'sms:ip',
      points: 10,        // 10 次/小时
      duration: 3600,
      blockDuration: 7200  // 超限后封禁 2 小时
    });
    
    // 手机号限流（防止单号频繁发送）
    this.phoneLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'sms:phone',
      points: 5,         // 5 次/天
      duration: 86400
    });
    
    // 全局限流（成本控制）
    this.globalLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'sms:global',
      points: 10000,     // 10000 条/天
      duration: 86400
    });
    
    // 短时间重复限流（60 秒内只能发 1 次）
    this.shortTermLimiter = new RateLimiterRedis({
      storeClient: this.redis,
      keyPrefix: 'sms:short',
      points: 1,
      duration: 60
    });
  }
  
  async checkLimits(phone, ip) {
    const results = [];
    
    try {
      // 1. 检查全局限额
      await this.globalLimiter.consume('global');
      results.push({ type: 'global', passed: true });
    } catch (error) {
      return {
        allowed: false,
        reason: '系统每日发送量已达上限，请明天再试',
        type: 'global'
      };
    }
    
    try {
      // 2. 检查 IP 限流
      await this.ipLimiter.consume(ip);
      results.push({ type: 'ip', passed: true });
    } catch (error) {
      return {
        allowed: false,
        reason: '您的 IP 发送过于频繁，请稍后再试',
        type: 'ip',
        retryAfter: Math.ceil(error.msBeforeNext / 1000)
      };
    }
    
    try {
      // 3. 检查手机号限流
      await this.phoneLimiter.consume(phone);
      results.push({ type: 'phone', passed: true });
    } catch (error) {
      return {
        allowed: false,
        reason: '该手机号今日发送次数已达上限',
        type: 'phone'
      };
    }
    
    try {
      // 4. 检查短时间重复
      await this.shortTermLimiter.consume(phone);
      results.push({ type: 'short', passed: true });
    } catch (error) {
      return {
        allowed: false,
        reason: '发送过于频繁，请 60 秒后再试',
        type: 'short',
        retryAfter: Math.ceil(error.msBeforeNext / 1000)
      };
    }
    
    return { allowed: true };
  }
  
  async sendSMS(phone, ip) {
    const result = await this.checkLimits(phone, ip);
    
    if (!result.allowed) {
      throw new Error(result.reason);
    }
    
    // 发送短信
    const code = Math.floor(100000 + Math.random() * 900000);
    await sendSMSService(phone, code);
    
    // 保存验证码到 Redis（5 分钟有效）
    await this.redis.setex(`sms:code:${phone}`, 300, code);
    
    return { success: true };
  }
}

module.exports = new SMSLimiter();
```

## ❓ 六、常见问题

### Q1: 固定窗口 vs 滑动窗口？

**A**: 滑动窗口更精确，但开销略大。

```
固定窗口（100 req/分钟）:
时间: 00:00-00:59 | 01:00-01:59
      100 次      |  100 次

问题：00:59 发送 100 次，01:00 再发送 100 次
→ 2 秒内 200 次 ❌ 突刺流量

滑动窗口（100 req/分钟）:
时间: 任意 60 秒窗口
始终保证 60 秒内不超过 100 次 ✅
```

### Q2: Redis 单点故障怎么办？

**A**: 使用 Redis Cluster 或降级策略。

```javascript
// 方案1：Redis Cluster（高可用）
const redis = new Redis.Cluster([
  { host: 'redis-1', port: 6379 },
  { host: 'redis-2', port: 6379 },
  { host: 'redis-3', port: 6379 }
]);

// 方案2：降级到内存限流
let memoryLimiter = new Map();

try {
  await redisLimiter.consume(key);
} catch (error) {
  if (error.name === 'ReplyError') {
    // Redis 故障，降级到内存限流
    console.warn('Redis 故障，降级到内存限流');
    const count = (memoryLimiter.get(key) || 0) + 1;
    memoryLimiter.set(key, count);
    
    if (count > 100) {
      throw new Error('Rate limit exceeded');
    }
  }
}
```

### Q3: 如何实现动态限流？

**A**: 根据系统负载动态调整。

```javascript
const baseLimit = 1000;

async function getDynamicLimit() {
  const cpuUsage = os.loadavg()[0] / os.cpus().length;
  
  if (cpuUsage > 0.8) {
    return baseLimit * 0.5;  // CPU 高负载，减少 50%
  } else if (cpuUsage < 0.3) {
    return baseLimit * 1.5;  // CPU 低负载，增加 50%
  }
  
  return baseLimit;
}

app.use(async (req, res, next) => {
  const limit = await getDynamicLimit();
  
  const limiter = new RateLimiterRedis({
    storeClient: redis,
    points: limit,
    duration: 60
  });
  
  await limiter.consume(req.ip);
  next();
});
```
