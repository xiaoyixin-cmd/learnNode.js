# 2.2 线程间通信

> **本节目标**：掌握 Worker Threads 线程间通信的多种方式，理解 MessagePort、SharedArrayBuffer 和 Atomics 的使用场景，构建高效的多线程协作系统。

## 📖 一、理论知识

### 1.1 线程间通信方式

Node.js Worker Threads 提供了三种主要的线程间通信方式：

```
┌─────────────────────────────────────────────┐
│         线程间通信方式对比                  │
├─────────────────────────────────────────────┤
│ 1. parentPort (基础通信)                    │
│    - 主线程 ↔ Worker 单向消息               │
│    - 结构化克隆算法（序列化）               │
│    - 适合：一次性数据传递                   │
│                                             │
│ 2. MessageChannel (双向通信)                │
│    - Worker ↔ Worker 直接通信               │
│    - 支持多个通信通道                       │
│    - 适合：复杂协作场景                     │
│                                             │
│ 3. SharedArrayBuffer (共享内存)             │
│    - 零拷贝，直接内存访问                   │
│    - 需要 Atomics 原子操作                  │
│    - 适合：高频数据同步                     │
└─────────────────────────────────────────────┘
```

### 1.2 结构化克隆算法

**支持的类型**：
```javascript
// ✅ 可以传递
- 原始类型: string, number, boolean, null, undefined, BigInt
- 对象和数组
- Date, RegExp
- Map, Set
- ArrayBuffer, TypedArray, DataView
- Error 对象

// ❌ 无法传递
- 函数
- Symbol
- WeakMap, WeakSet
- DOM 节点
- 包含循环引用的对象（部分情况）
```

### 1.3 性能对比

| 通信方式 | 延迟 | 吞吐量 | 适用数据大小 |
|---------|------|--------|-------------|
| **parentPort** | ~0.1ms | 中等 | < 10MB |
| **MessageChannel** | ~0.1ms | 中等 | < 10MB |
| **SharedArrayBuffer** | ~0.001ms | 极高 | 任意 |

### 2.1 方法1：parentPort（基础通信）

#### 主线程 → Worker（单向）

```javascript
// main.js
const { Worker } = require('worker_threads');

const worker = new Worker('./worker.js', {
  workerData: { initialValue: 42 }  // 启动时传递数据
});

// 发送消息给 Worker
worker.postMessage({ command: 'process', data: [1, 2, 3] });

// 接收 Worker 的消息
worker.on('message', (result) => {
  console.log('收到 Worker 结果:', result);
});
```

```javascript
// worker.js
const { parentPort, workerData } = require('worker_threads');

console.log('初始值:', workerData.initialValue);

// 接收主线程的消息
parentPort.on('message', (message) => {
  if (message.command === 'process') {
    const result = message.data.reduce((a, b) => a + b, 0);
    
    // 发送结果回主线程
    parentPort.postMessage({ result, processed: true });
  }
});
```

### 2.2 方法2：MessageChannel（Worker间通信）

#### 基础示例

```javascript
// main.js - 创建两个 Worker 直接通信
const { Worker, MessageChannel } = require('worker_threads');

// 创建两个 Worker
const worker1 = new Worker('./worker1.js');
const worker2 = new Worker('./worker2.js');

// 创建通信通道
const { port1, port2 } = new MessageChannel();

// 将 port1 传给 worker1，port2 传给 worker2
worker1.postMessage({ port: port1 }, [port1]);  // [port1] 表示转移所有权
worker2.postMessage({ port: port2 }, [port2]);

console.log('两个 Worker 已建立直接通信通道');
```

```javascript
// worker1.js
const { parentPort } = require('worker_threads');

let communicationPort = null;

parentPort.on('message', (message) => {
  if (message.port) {
    // 接收通信端口
    communicationPort = message.port;
    
    // 监听来自 worker2 的消息
    communicationPort.on('message', (data) => {
      console.log('[Worker1] 收到 Worker2 消息:', data);
    });
    
    // 发送消息给 worker2
    communicationPort.postMessage({
      from: 'worker1',
      message: 'Hello from Worker 1!'
    });
  }
});
```

```javascript
// worker2.js
const { parentPort } = require('worker_threads');

let communicationPort = null;

parentPort.on('message', (message) => {
  if (message.port) {
    communicationPort = message.port;
    
    communicationPort.on('message', (data) => {
      console.log('[Worker2] 收到 Worker1 消息:', data);
      
      // 回复 worker1
      communicationPort.postMessage({
        from: 'worker2',
        message: 'Hello from Worker 2!',
        echo: data
      });
    });
  }
});
```

#### 实战：分布式计算协作

```javascript
// main.js - 主节点协调多个计算节点
const { Worker, MessageChannel } = require('worker_threads');

class DistributedCompute {
  constructor(numWorkers = 4) {
    this.workers = [];
    this.channels = [];
    
    // 创建 Worker 网格
    for (let i = 0; i < numWorkers; i++) {
      const worker = new Worker('./compute-worker.js', {
        workerData: { workerId: i }
      });
      this.workers.push(worker);
    }
    
    // 建立 Worker 间通信（网格拓扑）
    this.setupMesh();
  }
  
  setupMesh() {
    for (let i = 0; i < this.workers.length; i++) {
      for (let j = i + 1; j < this.workers.length; j++) {
        const { port1, port2 } = new MessageChannel();
        
        // Worker i 和 Worker j 建立通信
        this.workers[i].postMessage({
          type: 'connect',
          peerId: j,
          port: port1
        }, [port1]);
        
        this.workers[j].postMessage({
          type: 'connect',
          peerId: i,
          port: port2
        }, [port2]);
      }
    }
  }
  
  async compute(taskData) {
    // 将任务分配给所有 Worker
    const promises = this.workers.map((worker, i) => {
      return new Promise((resolve) => {
        worker.once('message', (result) => {
          if (result.type === 'result') {
            resolve(result.data);
          }
        });
        
        worker.postMessage({
          type: 'task',
          data: taskData[i]
        });
      });
    });
    
    return Promise.all(promises);
  }
}

// 使用示例
const cluster = new DistributedCompute(4);

const results = await cluster.compute([
  { range: [0, 250] },
  { range: [250, 500] },
  { range: [500, 750] },
  { range: [750, 1000] }
]);

console.log('计算结果:', results);
```

#### Atomics 原子操作

```javascript
// main.js - 使用 Atomics 避免竞态条件
const { Worker } = require('worker_threads');
const { Atomics } = require('worker_threads');

const sharedBuffer = new SharedArrayBuffer(4);
const sharedArray = new Int32Array(sharedBuffer);

// 创建多个 Worker 同时修改
const workers = Array.from({ length: 10 }, () => 
  new Worker('./atomic-worker.js', { workerData: { sharedBuffer } })
);

Promise.all(workers.map(w => new Promise(resolve => {
  w.on('exit', resolve);
}))).then(() => {
  console.log('最终计数:', sharedArray[0]);
  // 使用 Atomics: 10000 ✅
  // 不使用 Atomics: ~8000-9500 ❌ (竞态条件导致丢失更新)
});
```

```javascript
// atomic-worker.js
const { workerData } = require('worker_threads');
const { Atomics } = require('worker_threads');

const sharedArray = new Int32Array(workerData.sharedBuffer);

// 每个 Worker 累加 1000 次
for (let i = 0; i < 1000; i++) {
  // ✅ 原子操作：add(array, index, value)
  Atomics.add(sharedArray, 0, 1);
  
  // ❌ 非原子操作（会导致竞态条件）:
  // sharedArray[0] = sharedArray[0] + 1;
}
```

#### Atomics 常用方法

| 方法 | 说明 | 示例 |
|-----|------|------|
| `Atomics.add()` | 原子加法 | `Atomics.add(arr, 0, 5)` |
| `Atomics.sub()` | 原子减法 | `Atomics.sub(arr, 0, 3)` |
| `Atomics.and()` | 原子按位与 | `Atomics.and(arr, 0, 0b1010)` |
| `Atomics.or()` | 原子按位或 | `Atomics.or(arr, 0, 0b1010)` |
| `Atomics.xor()` | 原子按位异或 | `Atomics.xor(arr, 0, 0b1010)` |
| `Atomics.load()` | 原子读取 | `Atomics.load(arr, 0)` |
| `Atomics.store()` | 原子写入 | `Atomics.store(arr, 0, 42)` |
| `Atomics.wait()` | 阻塞等待 | `Atomics.wait(arr, 0, 42)` |
| `Atomics.notify()` | 唤醒等待线程 | `Atomics.notify(arr, 0, 1)` |

### 2.4 实战：生产者-消费者队列

```javascript
// shared-queue.js - 使用 SharedArrayBuffer 实现队列
class SharedQueue {
  constructor(size = 1024) {
    // 队列结构：
    // [0]: 读指针
    // [1]: 写指针
    // [2]: 队列大小
    // [3-...]: 数据区
    this.buffer = new SharedArrayBuffer((size + 3) * 4);
    this.queue = new Int32Array(this.buffer);
    
    Atomics.store(this.queue, 0, 0);  // 读指针
    Atomics.store(this.queue, 1, 0);  // 写指针
    Atomics.store(this.queue, 2, size);  // 队列大小
  }
  
  enqueue(value) {
    const size = Atomics.load(this.queue, 2);
    const writeIdx = Atomics.load(this.queue, 1);
    const readIdx = Atomics.load(this.queue, 0);
    
    // 检查队列是否已满
    if ((writeIdx + 1) % size === readIdx) {
      return false;  // 队列已满
    }
    
    // 写入数据
    Atomics.store(this.queue, 3 + writeIdx, value);
    
    // 更新写指针
    Atomics.store(this.queue, 1, (writeIdx + 1) % size);
    
    // 唤醒等待的消费者
    Atomics.notify(this.queue, 1, 1);
    
    return true;
  }
  
  dequeue(timeout = Infinity) {
    const size = Atomics.load(this.queue, 2);
    let readIdx = Atomics.load(this.queue, 0);
    let writeIdx = Atomics.load(this.queue, 1);
    
    // 检查队列是否为空
    if (readIdx === writeIdx) {
      // 等待生产者写入数据
      const result = Atomics.wait(this.queue, 1, writeIdx, timeout);
      
      if (result === 'timed-out') {
        return null;  // 超时
      }
      
      writeIdx = Atomics.load(this.queue, 1);
    }
    
    // 读取数据
    const value = Atomics.load(this.queue, 3 + readIdx);
    
    // 更新读指针
    Atomics.store(this.queue, 0, (readIdx + 1) % size);
    
    return value;
  }
  
  getBuffer() {
    return this.buffer;
  }
}

module.exports = SharedQueue;
```

**生产者**：

```javascript

}

console.log('生产完成');
```

## 🐛 三、3个高频错误

### 错误1：忘记转移 MessagePort 所有权

#### 问题代码

```javascript
// ❌ 错误：未转移所有权
const { port1, port2 } = new MessageChannel();

worker.postMessage({ port: port1 });  // 忘记第二个参数

// 在 Worker 中：
// port.postMessage()  → Error: port is not a MessagePort
```

#### 正确写法

```javascript
// ✅ 正确：转移所有权
worker.postMessage({ port: port1 }, [port1]);

// [port1] 表示将 port1 的所有权转移给 Worker
// 转移后，主线程无法再使用 port1
```

### 错误2：SharedArrayBuffer 未使用 Atomics

#### 问题代码

```javascript
// main.js + worker.js 同时修改 sharedArray[0]

// ❌ 错误：非原子操作
sharedArray[0] = sharedArray[0] + 1;

// 竞态条件：
// 线程A读取: 0
// 线程B读取: 0
// 线程A写入: 1
// 线程B写入: 1  ← 期望是 2，实际是 1
```

#### 正确写法

```javascript
// ✅ 使用 Atomics
Atomics.add(sharedArray, 0, 1);

// 或使用锁
const LOCK_INDEX = 0;
const VALUE_INDEX = 1;

// 获取锁
while (Atomics.compareExchange(sharedArray, LOCK_INDEX, 0, 1) !== 0) {
  // 自旋等待
}

// 临界区
sharedArray[VALUE_INDEX]++;

// 释放锁
Atomics.store(sharedArray, LOCK_INDEX, 0);
```

### 错误3：Atomics.wait() 在主线程使用

#### 问题代码

```javascript
// main.js
const sharedArray = new Int32Array(sharedBuffer);

// ❌ 错误：主线程不允许使用 Atomics.wait()
Atomics.wait(sharedArray, 0, 0);

// Error: Atomics.wait cannot be called in this context
```

**错误原因**：`Atomics.wait()` 会阻塞线程，主线程阻塞会导致整个应用无响应。

#### 正确写法

```javascript
// ✅ 方法1：只在 Worker 中使用 Atomics.wait()
// worker.js
Atomics.wait(sharedArray, 0, 0);  // ✅ Worker 中可以阻塞

// ✅ 方法2：主线程使用轮询
// main.js
const interval = setInterval(() => {
  const value = Atomics.load(sharedArray, 0);
  if (value !== 0) {
    console.log('检测到变化:', value);
    clearInterval(interval);
  }
}, 100);

// ✅ 方法3：使用 Node.js 18+ 的 Atomics.waitAsync()
const result = Atomics.waitAsync(sharedArray, 0, 0);
if (result.async) {
  result.value.then(() => {
    console.log('值已改变');
  });
}
```

## 🎯 四、2个应用场景

### 场景1：实时数据同步（股票行情）

#### 需求背景

- 实时股票行情数据
- 多个 Worker 同时处理不同股票
- 需要共享最新价格数据

#### 实现方案

```javascript
// shared-price-board.js
class SharedPriceBoard {
  constructor(numStocks = 1000) {
    // 每个股票占用 2 个位置：[价格, 时间戳]
    this.buffer = new SharedArrayBuffer(numStocks * 2 * 4);
    this.prices = new Int32Array(this.buffer);
  }
  
  updatePrice(stockId, price) {
    const timestamp = Date.now();
    
    // 使用 Atomics 保证原子更新
    Atomics.store(this.prices, stockId * 2, Math.floor(price * 100));  // 价格（分）
    Atomics.store(this.prices, stockId * 2 + 1, timestamp);
    
    // 通知其他 Worker
    Atomics.notify(this.prices, stockId * 2);
  }
  
  getPrice(stockId) {
    const price = Atomics.load(this.prices, stockId * 2) / 100;
    const timestamp = Atomics.load(this.prices, stockId * 2 + 1);
    return { price, timestamp };
  }
  
  waitForUpdate(stockId, timeout = 1000) {
    const currentValue = Atomics.load(this.prices, stockId * 2);
    Atomics.wait(this.prices, stockId * 2, currentValue, timeout);
  }
  
  getBuffer() {
    return this.buffer;
  }
}

module.exports = SharedPriceBoard;
```

### 场景2：并行图像处理管线

#### 需求背景

- 图像处理管线：下载 → 解码 → 滤镜 → 编码 → 上传
- 每个阶段由专门的 Worker 处理
- Worker 间通过 MessageChannel 传递数据

#### 实现方案

```javascript
// main.js - 管线协调器
const { Worker, MessageChannel } = require('worker_threads');

class ImagePipeline {
  constructor() {
    // 创建 5 个阶段的 Worker
    this.downloader = new Worker('./stages/downloader.js');
    this.decoder = new Worker('./stages/decoder.js');
    this.filter = new Worker('./stages/filter.js');
    this.encoder = new Worker('./stages/encoder.js');
    this.uploader = new Worker('./stages/uploader.js');
    
    // 建立管线通道
    this.setupPipeline();
  }
  
  setupPipeline() {
    // 下载器 → 解码器
    const { port1: d2dec_1, port2: d2dec_2 } = new MessageChannel();
    this.downloader.postMessage({ nextStage: d2dec_1 }, [d2dec_1]);
    this.decoder.postMessage({ prevStage: d2dec_2, nextStage: null }, [d2dec_2]);
    
    // 解码器 → 滤镜
    const { port1: dec2fil_1, port2: dec2fil_2 } = new MessageChannel();
    this.decoder.postMessage({ nextStage: dec2fil_1 }, [dec2fil_1]);
    this.filter.postMessage({ prevStage: dec2fil_2 }, [dec2fil_2]);
    
    // 滤镜 → 编码器
    const { port1: fil2enc_1, port2: fil2enc_2 } = new MessageChannel();
    this.filter.postMessage({ nextStage: fil2enc_1 }, [fil2enc_1]);
    this.encoder.postMessage({ prevStage: fil2enc_2 }, [fil2enc_2]);
    
    // 编码器 → 上传器
    const { port1: enc2up_1, port2: enc2up_2 } = new MessageChannel();
    this.encoder.postMessage({ nextStage: enc2up_1 }, [enc2up_1]);
    this.uploader.postMessage({ prevStage: enc2up_2 }, [enc2up_2]);
  }
  
  async processImage(url) {
    return new Promise((resolve) => {
      this.uploader.once('message', (result) => {
        if (result.type === 'complete') {
          resolve(result.uploadUrl);
        }
      });
      
      // 启动管线
      this.downloader.postMessage({ type: 'start', url });
    });
  }
}

// 使用
const pipeline = new ImagePipeline();
const result = await pipeline.processImage('https://example.com/image.jpg');
console.log('处理完成:', result);
```

// stages/filter.js
const { parentPort } = require('worker_threads');
const sharp = require('sharp');

let prevStagePort = null;

parentPort.on('message', (message) => {
  if (message.prevStage) {
    prevStagePort = message.prevStage;

      console.log('[滤镜] 收到图片数据');

      const filtered = await sharp(data.buffer)
        .blur(5)
        .modulate({ brightness: 1.2 })
        .toBuffer();
      
      // 传递给下一阶段
      if (nextStagePort) {
        nextStagePort.postMessage({
          ...data,
          buffer: filtered,
          stage: 'filter'
        });
      }
    });
  }
  
  if (message.nextStage) {
    nextStagePort = message.nextStage;
  }
});

## 📝 五、总结

### 核心要点

✅ **parentPort** 适合主线程与 Worker 的基础通信  
✅ **MessageChannel** 实现 Worker 间直接通信，避免主线程中转  
✅ **SharedArrayBuffer** 提供零拷贝的共享内存，性能最高  
✅ **Atomics** 必须配合 SharedArrayBuffer 使用，避免竞态条件  
✅ **转移所有权** 使用 `postMessage(data, [transferList])`  
✅ **Atomics.wait()** 只能在 Worker 中使用，主线程禁用  
✅ **选择指南**：小数据用 parentPort，大数据/高频用 SharedArrayBuffer

### 最佳实践

1. **消息传递**：优先使用 parentPort，简单高效
2. **Worker 协作**：使用 MessageChannel 建立点对点通信
3. **共享内存**：仅在高性能场景（如音视频处理）使用
4. **原子操作**：SharedArrayBuffer 必须配合 Atomics
5. **错误处理**：所有通信端口都要监听 `error` 事件

## ❓ 六、常见问题

### Q1: 如何选择通信方式？

**A**: 根据数据大小和频率选择。

| 数据大小 | 通信频率 | 推荐方案 |
|---------|---------|---------|
| < 1MB | 低（< 10次/秒） | parentPort |
| < 1MB | 高（> 100次/秒） | SharedArrayBuffer |
| > 10MB | 任意 | SharedArrayBuffer |

### Q2: MessageChannel 和 parentPort 的区别？

**A**: 

```javascript
// parentPort: 主线程 ↔ Worker（固定通道）
worker.postMessage();         // 主线程发送
parentPort.on('message');     // Worker 接收

// MessageChannel: 任意两点通信（自定义通道）
port1.postMessage();          // 一端发送
port2.on('message');          // 另一端接收
```

### Q3: Atomics.wait() 超时后如何重试？

**A**:

```javascript
function waitWithRetry(array, index, value, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    const result = Atomics.wait(array, index, value, 1000);
    
    if (result === 'ok') {
      return true;  // 成功
    }
    
    if (result === 'not-equal') {
      return true;  // 值已改变
    }
    
    console.log(`重试 ${i + 1}/${maxRetries}`);
  }
  
  return false;  // 超时失败
}
```
