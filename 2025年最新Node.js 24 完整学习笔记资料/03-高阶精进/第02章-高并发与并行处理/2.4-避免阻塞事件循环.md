# 2.4 避免阻塞事件循环

> **本节目标**：识别和消除阻塞事件循环的代码，掌握异步改造技巧，提升应用响应性能。

## 📖 一、理论知识

### 1.1 什么是事件循环阻塞？

**事件循环阻塞** 是指 JavaScript 主线程被同步代码占用，无法处理其他任务（如 HTTP 请求、定时器、I/O 回调）。

```javascript
// ❌ 阻塞示例
app.get('/block', (req, res) => {
  const start = Date.now();
  // 同步循环阻塞 1 秒
  while (Date.now() - start < 1000) {}
  res.send('Done');
});

// 在这 1 秒内：
// - 其他所有请求都被阻塞
// - 定时器无法触发
// - 用户感知明显延迟
```

### 1.2 阻塞的危害

```
正常应用（无阻塞）:
Time →  [Req1] [Req2] [Req3] [Req4] [Req5]
        ✅ 并发处理

阻塞应用:
Time →  [Req1阻塞5秒................] [Req2] [Req3] [Req4]
        ❌ 串行处理，吞吐量下降 80%
```

| 指标 | 无阻塞 | 有阻塞（1秒） | 影响 |
|-----|--------|-------------|------|
| **QPS** | 1000 req/s | 1 req/s | ↓ 99.9% |
| **P99延迟** | 50ms | 5000ms | ↑ 100倍 |
| **用户体验** | 流畅 | 卡顿 | 极差 |

### 1.3 常见阻塞操作

| 操作类型 | 风险等级 | 典型场景 |
|---------|---------|---------|
| **同步 I/O** | ⚠️⚠️⚠️ 严重 | `fs.readFileSync()`, `execSync()` |
| **大 JSON 解析** | ⚠️⚠️⚠️ 严重 | `JSON.parse(largeString)` |
| **正则回溯** | ⚠️⚠️⚠️ 严重 | `/^(a+)+b$/.test('aaa...')` |
| **复杂计算** | ⚠️⚠️ 中等 | 大数组排序、加密解密 |
| **同步加密** | ⚠️⚠️ 中等 | `bcrypt.hashSync()` |

### 2.1 方法1：事件循环延迟监控

```javascript
// monitor/lagDetector.js
const { monitorEventLoopDelay } = require('perf_hooks');

class EventLoopLagDetector {
  constructor(threshold = 10) {
    this.threshold = threshold;  // 阈值（毫秒）
    this.monitor = monitorEventLoopDelay({ resolution: 1 });
  }
  
  start() {
    this.monitor.enable();
    
    this.interval = setInterval(() => {
      const lag = this.monitor.mean / 1e6;  // 转换为毫秒
      
      if (lag > this.threshold) {
        console.warn(`⚠️ 事件循环延迟: ${lag.toFixed(2)}ms`);
        console.trace('可能的阻塞代码');
      }
      
      this.monitor.reset();
    }, 1000);
  }
  
  stop() {
    clearInterval(this.interval);
    this.monitor.disable();
  }
}

// 使用
const detector = new EventLoopLagDetector(10);
detector.start();
```

### 3.3 方案3：CPU 密集任务分片

#### 问题代码

```javascript
// ❌ 一次性处理 100 万条数据
app.get('/process', (req, res) => {
  const data = Array.from({ length: 1000000 }, (_, i) => i);

```

#### 优化方案1：setImmediate 分片

```javascript
// ✅ 分片处理，插入事件循环
app.get('/process', async (req, res) => {
  const data = Array.from({ length: 1000000 }, (_, i) => i);
  const chunkSize = 10000;
  const results = [];
  
  for (let i = 0; i < data.length; i += chunkSize) {
    const chunk = data.slice(i, i + chunkSize);
    
    // 处理一批数据
    results.push(...chunk.map(x => expensiveOperation(x)));
    
    // 让出控制权，处理其他请求
    await new Promise(resolve => setImmediate(resolve));
  }
  
  res.json({ count: results.length });
});

// 效果：
// - 每处理 1 万条，让出一次控制权
// - 其他请求可以被处理
// - 总耗时略有增加（~5.5秒），但不阻塞其他请求
```

#### 优化方案2：Worker Threads 并行

```javascript

const Piscina = require('piscina');

const pool = new Piscina({
  filename: './workers/processor.js',
  maxThreads: 8

app.get('/process', async (req, res) => {
```

### 3.4 方案4：正则表达式优化

```javascript
// ✅ 安全的正则表达式
app.post('/validate', (req, res) => {
  const email = req.body.email;
  
  // 避免嵌套量词和回溯
  const regex = /^[a-zA-Z0-9]+@domain\.com$/;
  
  // 或使用更简单的字符串方法
  const isValid = email.endsWith('@domain.com') && 
                  /^[a-zA-Z0-9]+@/.test(email);
  
  res.json({ valid: isValid });
});

// ✅ 使用 safe-regex 检测危险正则
const safeRegex = require('safe-regex');

const regex = /^([a-zA-Z0-9]+)*@domain\.com$/;

}
```

## 🐛 四、3个高频错误

### 错误1：误用 setImmediate/process.nextTick

#### 问题代码

```javascript
// ❌ 错误：仍然是同步执行
app.get('/wrong', (req, res) => {
  process.nextTick(() => {
    // 虽然在 nextTick 中，但仍然是同步阻塞
    const start = Date.now();
    while (Date.now() - start < 1000) {}
  });
  
  res.send('Done');  // 必须等 nextTick 完成
});
```

**正确理解**：
- `setImmediate` 和 `process.nextTick` 只是延迟执行
- 不会让同步代码变成异步非阻塞

#### 正确写法

```javascript
// ✅ 正确：真正的异步处理
app.get('/correct', (req, res) => {
  // 立即响应
  res.send('Processing...');
  
  // 后台异步处理（不阻塞响应）
  setImmediate(() => {
    heavyTask();
  });
});

// ✅ 或使用 Worker Threads
app.get('/correct2', async (req, res) => {
  const worker = new Worker('./worker.js');
  worker.postMessage({ task: 'heavy' });
  
  res.send('Processing in worker...');
});
```

#### 正确写法

```javascript
// ✅ 方法1：分片排序
async function sortInChunks(array, chunkSize = 10000) {
  const sorted = [];
  
  for (let i = 0; i < array.length; i += chunkSize) {
    const chunk = array.slice(i, i + chunkSize);
    chunk.sort((a, b) => a - b);
    sorted.push(...chunk);
    
    // 让出控制权
    await new Promise(resolve => setImmediate(resolve));
  }
  
  // 最后合并排序
  return sorted.sort((a, b) => a - b);
}

app.get('/sort', async (req, res) => {
  const data = Array.from({ length: 1000000 }, () => Math.random());
  const sorted = await sortInChunks(data);
  
  res.json({ first: sorted[0], last: sorted[sorted.length - 1] });
```

```javascript
const { createReadStream } = require('fs');
const { createInterface } = require('readline');
const Piscina = require('piscina');

const pool = new Piscina({
  filename: './workers/logAnalyzer.js',
  maxThreads: 8
});

app.get('/analyze', async (req, res) => {
  const taskId = generateTaskId();
  res.json({ taskId, status: 'processing' });
  
  analyzeLogsParallel(taskId, './access.log');

async function analyzeLogsParallel(taskId, logFile) {
  const batches = [];
  let currentBatch = [];
  const batchSize = 10000;
  
  // 流式读取日志
  const rl = createInterface({
    input: createReadStream(logFile),
    crlfDelay: Infinity
  });
  
  for await (const line of rl) {
    currentBatch.push(line);
    
    if (currentBatch.length >= batchSize) {
      batches.push([...currentBatch]);
      currentBatch = [];
    }
  }
  
  if (currentBatch.length > 0) {
    batches.push(currentBatch);
  }
  
  // 并行分析各批次
  const results = await Promise.all(
    batches.map(batch => pool.run({ lines: batch }))
  );
  
  // 合并结果
  const finalStats = mergeStats(results);
  completeTask(taskId, finalStats);
}

function mergeStats(results) {
  const merged = { ips: {}, errors: 0, total: 0 };
  
  results.forEach(result => {
    Object.entries(result.ips).forEach(([ip, count]) => {
      merged.ips[ip] = (merged.ips[ip] || 0) + count;
    });
    merged.errors += result.errors;
    merged.total += result.total;
  });
  
  return merged;
}
```

## 📝 六、总结

### 核心要点

✅ **识别阻塞**：使用事件循环监控和性能分析工具  
✅ **同步改异步**：优先使用异步 API（fs.promises, bcrypt.hash）  
✅ **大任务分片**：使用 setImmediate 或 Worker Threads  
✅ **流式处理**：处理大文件时使用 Stream API  
✅ **正则优化**：避免嵌套量词和回溯攻击  
✅ **Worker 并行**：CPU 密集任务使用 Worker Threads  
✅ **监控告警**：实时监控事件循环延迟

### 最佳实践

1. **代码审查**：检查所有 `Sync` 后缀的方法
2. **压测验证**：上线前进行压力测试
3. **监控部署**：生产环境监控事件循环延迟
4. **分片处理**：大数组操作每 1 万条分一片
5. **异步优先**：永远优先使用异步 API

## ❓ 七、常见问题

### Q1: 如何判断代码是否阻塞？

**A**: 三个方法：

```javascript
// 1. 事件循环监控
const { monitorEventLoopDelay } = require('perf_hooks');
const h = monitorEventLoopDelay({ resolution: 10 });
h.enable();

// 2. 压测时观察 QPS 下降
autocannon -c 100 -d 10 http://localhost:3000

// 3. 使用 0x 生成火焰图
0x app.js
```

### Q2: setImmediate vs setTimeout(fn, 0)？

**A**: setImmediate 更适合让出控制权。

```javascript
// setImmediate: 在当前事件循环结束后立即执行
setImmediate(() => console.log('immediate'));

// setTimeout(fn, 0): 在定时器阶段执行（可能延迟）
setTimeout(() => console.log('timeout'), 0);

// 输出顺序:
// immediate
// timeout

// 推荐：使用 setImmediate 分片处理
```

### Q3: 如何处理无法避免的同步操作？

**A**: 使用 Worker Threads。

```javascript
// 某些第三方库只提供同步 API
const someLib = require('some-sync-lib');

// ✅ 在 Worker 中调用
const pool = new Piscina({
  filename: './workers/syncWrapper.js'
});

app.get('/api', async (req, res) => {
  const result = await pool.run(req.body);
  res.json(result);
});

// workers/syncWrapper.js
const someLib = require('some-sync-lib');
module.exports = (data) => someLib.processSync(data);
```
