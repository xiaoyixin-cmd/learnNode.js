# 2.5 背压处理

> **本节目标**：理解流式编程中的背压机制，掌握 Node.js Stream API 的背压处理，构建高吞吐量的数据处理系统。

---

## 📖 一、理论知识

### 1.1 什么是背压（Backpressure）？

**背压** 是指数据生产速度超过消费速度时，系统自动暂停生产者，避免内存溢出的机制。

```
生产者快 → 消费者慢 = 背压

┌─────────────┐    Fast     ┌─────────────┐    Slow     ┌─────────────┐
│  数据源     │ ─────────→  │   缓冲区    │ ─────────→  │  目标       │
│ (1000MB/s)  │             │  (10MB)     │             │ (10MB/s)    │
└─────────────┘             └─────────────┘             └─────────────┘
                                   ↓
                            缓冲区填满 → 暂停生产者 → 背压生效
```

### 1.2 背压的重要性

**无背压控制的问题**：

```javascript
// ❌ 无背压控制
const fs = require('fs');

const readStream = fs.createReadStream('./large-file.txt');
const writeStream = fs.createWriteStream('./output.txt');

// 快速读取（1000MB/s）→ 慢速写入（10MB/s）
readStream.on('data', (chunk) => {
  writeStream.write(chunk);  // 数据积压在内存中
});

// 结果：内存占用持续增长 → OOM崩溃
```

**有背压控制**：

```javascript
// ✅ 有背压控制
readStream.on('data', (chunk) => {
  const canContinue = writeStream.write(chunk);
  
  if (!canContinue) {
    // 写入缓冲区已满，暂停读取
    readStream.pause();
  }
});

writeStream.on('drain', () => {
  // 缓冲区已清空，恢复读取
  readStream.resume();
});

// 结果：内存占用稳定在 16-64KB
```

### 1.3 Stream 背压机制

```
┌───────────────────────────────────────────────┐
│         Node.js Stream 背压机制              │
├───────────────────────────────────────────────┤
│                                               │
│  Readable Stream                              │
│  ├─ highWaterMark: 16KB（默认缓冲区大小）     │
│  ├─ flowing mode（流动模式）                  │
│  └─ paused mode（暂停模式）                   │
│         ↓                                     │
│  Writable Stream                              │
│  ├─ highWaterMark: 16KB                       │
│  ├─ write() 返回 false → 缓冲区满            │
│  └─ emit('drain') → 缓冲区清空               │
│                                               │
└───────────────────────────────────────────────┘
```

## 💻 二、使用指南

### 2.1 基础示例：文件复制

#### 错误方式（无背压控制）

```javascript
// ❌ 错误：直接 pipe（旧版本可能没有自动背压）
const fs = require('fs');

const readStream = fs.createReadStream('./input.txt');
const writeStream = fs.createWriteStream('./output.txt');

// 简单 pipe（依赖 Node.js 自动处理背压）
readStream.pipe(writeStream);

// 问题：如果 writeStream 故障，可能导致内存泄漏
```

#### 正确方式（显式背压处理）

```javascript
// ✅ 正确：显式处理背压
const fs = require('fs');

const readStream = fs.createReadStream('./input.txt');
const writeStream = fs.createWriteStream('./output.txt');

readStream.on('data', (chunk) => {
  const canContinue = writeStream.write(chunk);
  
  if (!canContinue) {
    console.log('⚠️ 写入缓冲区满，暂停读取');
    readStream.pause();
  }
});

writeStream.on('drain', () => {
  console.log('✅ 缓冲区清空，恢复读取');
  readStream.resume();
});

readStream.on('end', () => {
  writeStream.end();
  console.log('文件复制完成');
});

writeStream.on('error', (error) => {
  console.error('写入错误:', error);
  readStream.destroy();
});
```

#### 有背压控制（安全）

```javascript
// ✅ 安全：使用 pipeline + 限流
const { pipeline } = require('stream/promises');
const { Transform } = require('stream');

// 限速 Transform Stream（模拟慢速写入）
class ThrottleStream extends Transform {
  constructor(bytesPerSecond) {
    super();
    this.bytesPerSecond = bytesPerSecond;
    this.lastTime = Date.now();
  }
  
  _transform(chunk, encoding, callback) {
    const now = Date.now();
    const elapsed = now - this.lastTime;
    const delay = (chunk.length / this.bytesPerSecond * 1000) - elapsed;
    
    if (delay > 0) {
      setTimeout(() => {
        this.push(chunk);
        this.lastTime = Date.now();
        callback();
      }, delay);
    } else {
      this.push(chunk);
      this.lastTime = now;
      callback();
    }
  }
}

app.post('/upload', async (req, res) => {
  const filePath = `./uploads/${Date.now()}.bin`;
  
  try {
    await pipeline(
      req,
      new ThrottleStream(1024 * 1024),  // 限速 1MB/s
      fs.createWriteStream(filePath)
    );
    
    res.json({ success: true, file: filePath });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
```

### 2.4 实战示例：数据库批量导入

#### 无背压控制（内存溢出）

```javascript
// ❌ 错误：一次性读取所有数据
app.post('/import', async (req, res) => {
  const data = await fs.promises.readFile('./data.csv', 'utf8');  // 100MB 全部加载到内存
  const lines = data.split('\n');  // 再次复制
  
  // 批量插入
  await db.insertMany(lines.map(parseLine));
  
  res.send('导入完成');
});

// 内存占用：200MB+（原始数据 + 解析后数据）
```

#### 有背压控制（流式处理）

```javascript
// ✅ 正确：流式处理 + 背压控制
const { pipeline } = require('stream/promises');
const { createReadStream } = require('fs');
const { createInterface } = require('readline');
const { Writable } = require('stream');

class DatabaseWriter extends Writable {
  constructor(batchSize = 1000) {
    super({ objectMode: true });
    this.batchSize = batchSize;
    this.batch = [];
  }
  
  async _write(line, encoding, callback) {
    this.batch.push(parseLine(line));
    
    if (this.batch.length >= this.batchSize) {
      try {
        await db.insertMany(this.batch);
        this.batch = [];
        callback();
      } catch (error) {
        callback(error);
      }
    } else {
      callback();
    }
  }
  
  async _final(callback) {
    // 处理剩余数据
    if (this.batch.length > 0) {
      try {
        await db.insertMany(this.batch);
        callback();
      } catch (error) {
        callback(error);
      }
    } else {
      callback();
    }
  }
}

app.post('/import', async (req, res) => {
  try {
    const rl = createInterface({
      input: createReadStream('./data.csv'),
      crlfDelay: Infinity
    });
    
    await pipeline(
      rl,
      new DatabaseWriter(1000)
    );
    
    res.send('导入完成');
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 内存占用：< 10MB（稳定）
```

### 2.5 自定义 Transform Stream

```javascript
// 自定义 CSV 解析器（带背压）
const { Transform } = require('stream');

class CSVParser extends Transform {
  constructor(options = {}) {
    super({ objectMode: true, ...options });
    this.buffer = '';
  }
  
  _transform(chunk, encoding, callback) {
    this.buffer += chunk.toString();
    const lines = this.buffer.split('\n');
    
    // 保留最后一个不完整的行
    this.buffer = lines.pop();
    
    lines.forEach(line => {
      if (line.trim()) {
        const parsed = this.parseLine(line);
        this.push(parsed);  // 推送到下游
      }
    });
    
    callback();
  }
  
  _flush(callback) {
    // 处理最后的缓冲数据
    if (this.buffer.trim()) {
      this.push(this.parseLine(this.buffer));
    }
    callback();
  }
  
  parseLine(line) {
    const values = line.split(',');
    return {
      name: values[0],
      age: parseInt(values[1]),
      email: values[2]
    };
  }
}

// 使用示例
await pipeline(
  fs.createReadStream('./data.csv'),
  new CSVParser(),
  new DatabaseWriter()
);
```

## 🎯 四、2个应用场景

### 场景1：视频流服务

#### 需求背景

- 用户请求观看 4K 视频（50GB）
- 服务器带宽有限（10MB/s）
- 需要边读边传，不能一次性加载到内存

#### 实现方案

```javascript
// video-streamer.js
const { pipeline } = require('stream/promises');
const { Transform } = require('stream');
const fs = require('fs');

// 限速 Transform（控制发送速度）
class RateLimiter extends Transform {
  constructor(bytesPerSecond) {
    super();
    this.bytesPerSecond = bytesPerSecond;
    this.lastTime = Date.now();
    this.bucket = bytesPerSecond;  // 令牌桶
  }
  
  async _transform(chunk, encoding, callback) {
    const now = Date.now();
    const elapsed = (now - this.lastTime) / 1000;
    
    // 补充令牌
    this.bucket = Math.min(
      this.bucket + elapsed * this.bytesPerSecond,
      this.bytesPerSecond
    );
    
    if (this.bucket >= chunk.length) {
      // 有足够令牌，立即发送
      this.bucket -= chunk.length;
      this.lastTime = now;
      callback(null, chunk);
    } else {
      // 令牌不足，等待
      const waitTime = ((chunk.length - this.bucket) / this.bytesPerSecond) * 1000;
      
      setTimeout(() => {
        this.bucket = 0;
        this.lastTime = Date.now();
        callback(null, chunk);
      }, waitTime);
    }
  }
}

app.get('/video/:id', async (req, res) => {
  const videoPath = `./videos/${req.params.id}.mp4`;
  const stat = await fs.promises.stat(videoPath);
  
  // 支持 Range 请求（断点续传）
  const range = req.headers.range;
  if (range) {
    const parts = range.replace(/bytes=/, '').split('-');
    const start = parseInt(parts[0], 10);
    const end = parts[1] ? parseInt(parts[1], 10) : stat.size - 1;
    const chunksize = (end - start) + 1;
    
    res.writeHead(206, {
      'Content-Range': `bytes ${start}-${end}/${stat.size}`,
      'Accept-Ranges': 'bytes',
      'Content-Length': chunksize,
      'Content-Type': 'video/mp4'
    });
    
    const stream = fs.createReadStream(videoPath, { start, end });
    
    try {
      await pipeline(
        stream,
        new RateLimiter(10 * 1024 * 1024),  // 限速 10MB/s
        res
      );
    } catch (error) {
      console.error('视频流传输失败:', error);
    }
  } else {
    // 完整视频
    res.writeHead(200, {
      'Content-Length': stat.size,
      'Content-Type': 'video/mp4'
    });
    
    await pipeline(
      fs.createReadStream(videoPath),
      new RateLimiter(10 * 1024 * 1024),
      res
    );
  }
});
```

### 场景2：日志收集系统

#### 需求背景

- 收集 1000 台服务器的日志
- 每台服务器 10MB/s 日志量
- 总流量 10GB/s
- 需要实时写入 Elasticsearch

#### 实现方案

```javascript
// log-collector.js
const { pipeline } = require('stream/promises');
const { Transform, Writable } = require('stream');
const net = require('net');
const { Client } = require('@elastic/elasticsearch');

const esClient = new Client({ node: 'http://localhost:9200' });

// Elasticsearch 批量写入器（带背压）
class ElasticsearchWriter extends Writable {
  constructor(options = {}) {
    super({ objectMode: true, ...options });
    this.batchSize = options.batchSize || 1000;
    this.batch = [];
    this.index = options.index || 'logs';
  }
  
  async _write(log, encoding, callback) {
    this.batch.push({
      index: { _index: this.index }
    }, log);
    
    if (this.batch.length >= this.batchSize * 2) {  // *2 因为有 index 行
      try {
        await this.flush();
        callback();
      } catch (error) {
        callback(error);
      }
    } else {
      callback();
    }
  }
  
  async _final(callback) {
    try {
      await this.flush();
      callback();
    } catch (error) {
      callback(error);
    }
  }
  
  async flush() {
    if (this.batch.length === 0) return;
    
    const { body } = await esClient.bulk({
      refresh: false,
      body: this.batch
    });
    
    if (body.errors) {
      console.error('Elasticsearch 写入错误');
    }
    
    this.batch = [];
  }
}

// 日志解析器
class LogParser extends Transform {
  constructor() {
    super({ objectMode: true });
  }
  
  _transform(chunk, encoding, callback) {
    const lines = chunk.toString().split('\n');
    
    lines.forEach(line => {
      if (line.trim()) {
        try {
          const log = JSON.parse(line);
          this.push(log);
        } catch (error) {
          console.error('日志解析错误:', line);
        }
      }
    });
    
    callback();
  }
}

// TCP 服务器接收日志
const server = net.createServer((socket) => {
  console.log('客户端连接:', socket.remoteAddress);
  
  pipeline(
    socket,
    new LogParser(),
    new ElasticsearchWriter({ batchSize: 1000, index: 'logs' })
  ).catch(error => {
    console.error('日志处理失败:', error);
  });
  
  socket.on('end', () => {
    console.log('客户端断开');
  });
});

server.listen(9999, () => {
  console.log('日志收集服务器运行在 9999 端口');
});
```

## 📝 五、总结

### 核心要点

✅ **背压** 是流式处理的核心机制，防止内存溢出  
✅ **write() 返回 false** 表示缓冲区满，需暂停生产  
✅ **drain 事件** 表示缓冲区清空，可恢复生产  
✅ **pipeline** 自动处理背压、错误、资源清理（推荐）  
✅ **highWaterMark** 控制缓冲区大小（默认 16KB）  
✅ **Transform Stream** 用于数据转换（解析、压缩、加密）  
✅ **objectMode** 允许流传输对象而非 Buffer

### 最佳实践

1. **优先使用 pipeline**：自动处理背压和错误
2. **检查 write() 返回值**：false 时暂停生产
3. **监听 drain 事件**：恢复生产
4. **错误处理**：所有 stream 都要监听 error 事件
5. **资源清理**：使用 pipeline 或 finished 确保清理

## ❓ 六、常见问题

### Q1: pipe() vs pipeline()？

**A**: 优先使用 pipeline()。

```javascript
// pipe(): 旧 API，需手动处理错误
readStream
  .pipe(transform)
  .pipe(writeStream)
  .on('error', handleError);  // 只捕获最后一个 stream 的错误

// pipeline(): 新 API，自动处理
await pipeline(
  readStream,
  transform,
  writeStream
);  // 自动传播错误 + 清理资源
```

### Q2: 如何调整缓冲区大小？

**A**: 使用 highWaterMark 选项。

```javascript
// 增大缓冲区（减少背压触发频率）
const stream = fs.createReadStream('./file.txt', {
  highWaterMark: 64 * 1024  // 64KB（默认 16KB）
});

// 适用场景：
// - 网络带宽高
// - 磁盘 I/O 快
// - 内存充足

// 减小缓冲区（更细粒度控制）
const stream = fs.createReadStream('./file.txt', {
  highWaterMark: 4 * 1024  // 4KB
});

// 适用场景：
// - 内存受限
// - 需要实时处理
```

### Q3: 如何监控背压状态？

**A**: 监听关键事件。

```javascript
const monitor = {
  pauseCount: 0,
  resumeCount: 0,
  bytesWritten: 0
};

readStream.on('pause', () => {
  monitor.pauseCount++;
  console.log('⏸ 读取暂停（背压生效）');
});

readStream.on('resume', () => {
  monitor.resumeCount++;
  console.log('▶️ 读取恢复');
});

writeStream.on('drain', () => {
  console.log('💧 缓冲区清空');
});

setInterval(() => {
  console.log('背压统计:', monitor);
}, 5000);
```
