## 📖 一、理论知识

### 1.1 为什么需要线程池？

**问题**：频繁创建和销毁 Worker 的开销

```javascript
// ❌ 性能问题：每次请求都创建新 Worker
app.get('/process', async (req, res) => {
  const worker = new Worker('./worker.js');  // 创建开销 ~30ms
  
  worker.on('message', result => {
    res.json(result);
    worker.terminate();  // 销毁
  });
  
  worker.postMessage(req.body);
});

// 100 并发 = 创建 100 个 Worker = 3000ms 开销！
```

**解决方案**：线程池复用 Worker

```javascript
// ✅ 线程池：复用 Worker
const pool = new Piscina({
  filename: './worker.js',
  minThreads: 2,
  maxThreads: 8
});

app.get('/process', async (req, res) => {
  const result = await pool.run(req.body);  // 复用已有 Worker
  res.json(result);
});

// 100 并发 = 复用 8 个 Worker = 接近 0ms 额外开销 ✅
```

### 1.2 线程池架构

```
┌─────────────────────────────────────────────┐
│            线程池架构                       │
├─────────────────────────────────────────────┤
│                                             │
│  主线程                                     │
│  ├─ 任务队列 (FIFO)                         │
│  │   [Task1, Task2, Task3, ...]            │
│  │                                          │
│  ├─ 空闲 Worker 池                          │
│  │   [Worker1, Worker2, Worker3, Worker4]  │
│  │                                          │
│  └─ 调度器                                  │
│      ├─ 分配任务到空闲 Worker               │
│      ├─ 监控 Worker 状态                    │
│      └─ 动态调整池大小                      │
│                                             │
│  Worker1  Worker2  Worker3  Worker4        │
│  [执行中] [空闲]   [执行中] [空闲]          │
│                                             │
└─────────────────────────────────────────────┘
```

### 1.3 线程池参数设置

| 参数 | 推荐值 | 说明 |
|-----|--------|------|
| **minThreads** | 2-4 | 最小保持的 Worker 数量 |
| **maxThreads** | CPU核心数 | 最大 Worker 数量 |
| **idleTimeout** | 60000ms | 空闲 Worker 超时时间 |
| **maxQueue** | Infinity | 任务队列最大长度 |

const Piscina = require('piscina');
const path = require('path');
const glob = require('glob');

class BatchFileProcessor {
  constructor(options = {}) {
    this.pool = new Piscina({
      filename: path.resolve(__dirname, 'workers/fileProcessor.js'),
      maxThreads: options.maxThreads || 8,

    });

        this.pool.run({ filePath, operation })

      duration: `${duration}ms`,
      results

    return {
      queueSize: this.pool.queueSize,
      threads: this.pool.threads.length,
      completed: this.pool.completed,
      runTime: this.pool.runTime
    };
  }
  
  async destroy() {
    await this.pool.destroy();
  }
}

### 2.5 监控与统计

```javascript
// monitor/poolMonitor.js
const Piscina = require('piscina');

class PoolMonitor {
  constructor(pool, name = 'default') {
    this.pool = pool;
    this.name = name;
    this.metrics = {
      totalTasks: 0,
      successTasks: 0,
      failedTasks: 0,
      totalDuration: 0
    };
    
    this.startMonitoring();
  }
  
  startMonitoring() {
    // 定期输出统计
    this.interval = setInterval(() => {
      this.logMetrics();
    }, 10000);
    
    // 监听任务完成事件
    this.pool.on('drain', () => {
      console.log(`[${this.name}] 队列已清空`);
    });
    
    this.pool.on('error', (error) => {
      console.error(`[${this.name}] 池错误:`, error);
    });
  }
  
  async runTask(data) {
    const start = Date.now();
    this.metrics.totalTasks++;
    
    try {
      const result = await this.pool.run(data);
      
      this.metrics.successTasks++;
      this.metrics.totalDuration += Date.now() - start;
      
      return result;
    } catch (error) {
      this.metrics.failedTasks++;
      throw error;
    }
  }
  
  logMetrics() {
    const avgDuration = this.metrics.totalTasks > 0
      ? (this.metrics.totalDuration / this.metrics.totalTasks).toFixed(2)
      : 0;
    
    console.log(`
[${this.name}] 线程池指标:
  - 队列大小: ${this.pool.queueSize}
  - 活跃线程: ${this.pool.threads.length}
  - 总任务数: ${this.metrics.totalTasks}
  - 成功: ${this.metrics.successTasks}
  - 失败: ${this.metrics.failedTasks}
  - 平均耗时: ${avgDuration}ms
  - 完成任务: ${this.pool.completed}
    `);
  }
  
  stop() {
    clearInterval(this.interval);
  }
}

module.exports = PoolMonitor;
```

#### 问题代码

module.exports = (data) => {

module.exports = async (data) => {

#### 问题代码

```javascript
// ❌ 错误：无限队列 + 大量任务
const pool = new Piscina({

// 瞬间添加 100 万个任务

// ✅ 方法1：限制队列大小
const pool = new Piscina({
  filename: './worker.js',
  maxThreads: 4,
  maxQueue: 1000  // 限制队列 1000 个任务
});

// ✅ 方法2：流式处理（分批）
async function processBatch(tasks, batchSize = 100) {
  for (let i = 0; i < tasks.length; i += batchSize) {
    const batch = tasks.slice(i, i + batchSize);
    await Promise.all(batch.map(task => pool.run(task)));
    console.log(`完成批次 ${i / batchSize + 1}`);
  }
}

await processBatch(Array.from({ length: 1000000 }, (_, i) => ({ id: i })));

// ✅ 方法3：队列满时等待
async function runWithBackpressure(data) {
  while (pool.queueSize >= 1000) {
    await new Promise(resolve => setTimeout(resolve, 100));
  }
  return pool.run(data);
}

// workers/videoTranscoder.js
const ffmpeg = require('fluent-ffmpeg');
const fs = require('fs').promises;

module.exports = async ({ inputPath, outputPath, format, quality = 'medium' }) => {
  return new Promise((resolve, reject) => {
    const startTime = Date.now();
    
    const qualitySettings = {
      low: { videoBitrate: '500k', audioBitrate: '64k' },
      medium: { videoBitrate: '1000k', audioBitrate: '128k' },
      high: { videoBitrate: '2500k', audioBitrate: '192k' }
    };
    
    const settings = qualitySettings[quality];
    
    ffmpeg(inputPath)
      .output(outputPath)
      .videoBitrate(settings.videoBitrate)
      .audioBitrate(settings.audioBitrate)
      .format(format)
      .on('progress', (progress) => {
        // 进度通过文件记录（Piscina 不支持实时消息）
        const progressFile = `${outputPath}.progress`;
        fs.writeFile(progressFile, JSON.stringify({
          percent: progress.percent,
          currentTime: progress.timemark
        })).catch(() => {});
      })
      .on('end', async () => {
        const duration = Date.now() - startTime;
        const fileSize = (await fs.stat(outputPath)).size;
        
        resolve({
          success: true,
          outputPath,
          format,
          duration: `${(duration / 1000).toFixed(2)}s`,
          fileSize: `${(fileSize / 1024 / 1024).toFixed(2)}MB`
        });
      })
      .on('error', (error) => {
        reject(new Error(`转码失败: ${error.message}`));
      })
      .run();
  });
};

// services/videoService.js
const Piscina = require('piscina');
const path = require('path');
const fs = require('fs').promises;

class VideoTranscodeService {
  constructor() {
    // 根据 CPU 核心数设置线程池
    const cpuCount = require('os').cpus().length;
    
    this.pool = new Piscina({
      filename: path.resolve(__dirname, '../workers/videoTranscoder.js'),
      minThreads: 1,
      maxThreads: Math.max(2, cpuCount - 2),  // 保留 2 个核心给系统
      maxQueue: 100,  // 限制队列长度
      idleTimeout: 120000,  // 2 分钟空闲超时
      resourceLimits: {
        maxOldGenerationSizeMb: 1024  // 限制内存使用
      }
    });
    
    this.jobs = new Map();
  }
  
  async transcode(jobId, inputPath, formats = ['mp4', 'webm']) {
    const tasks = formats.map(format => ({
      jobId,
      inputPath,
      outputPath: inputPath.replace(/\.\w+$/, `.${format}`),
      format,
      quality: 'medium'
    }));
    
    this.jobs.set(jobId, {
      status: 'processing',
      total: tasks.length,
      completed: 0,
      results: []
    });
    
    try {
      const results = await Promise.all(
        tasks.map(task => this.processTask(jobId, task))
      );
      
      this.jobs.get(jobId).status = 'completed';
      return results;
    } catch (error) {
      this.jobs.get(jobId).status = 'failed';
      this.jobs.get(jobId).error = error.message;
      throw error;
    }
  }
  
  async processTask(jobId, task) {
    try {
      const result = await this.pool.run(task, {
        signal: AbortSignal.timeout(300000)  // 5 分钟超时
      });
      
      const job = this.jobs.get(jobId);
      job.completed++;
      job.results.push(result);
      
      return result;
    } catch (error) {
      console.error(`任务失败 [${jobId}]:`, error);
      throw error;
    }
  }
  
  getJobStatus(jobId) {
    return this.jobs.get(jobId) || null;
  }
  
  getPoolStats() {
    return {
      queueSize: this.pool.queueSize,
      threads: this.pool.threads.length,
      completed: this.pool.completed,
      runTime: this.pool.runTime
    };
  }
  
  async destroy() {
    await this.pool.destroy();
  }
}

module.exports = VideoTranscodeService;
```

```javascript
// analyzer.js
const Piscina = require('piscina');
const path = require('path');

class LogAnalyzer {
  constructor() {
    this.pool = new Piscina({
      filename: path.resolve(__dirname, 'workers/logAnalyzer.js'),
      maxThreads: 8,
      maxQueue: 50
    });
  }
  
  async analyze(logs, analysisTypes) {
    // 分片处理（每片 10 万条）
    const chunkSize = 100000;
    const chunks = [];
    
    for (let i = 0; i < logs.length; i += chunkSize) {
      chunks.push(logs.slice(i, i + chunkSize));
    }
    
    console.log(`将 ${logs.length} 条日志分为 ${chunks.length} 片`);
    
    // 并行分析每个分片
    const results = await Promise.all(
      analysisTypes.map(async (analysis) => {
        const chunkResults = await Promise.all(
          chunks.map(chunk => 
            this.pool.run({ logs: chunk, analysis })
          )
        );
        
        // 合并结果
        return this.mergeResults(analysis, chunkResults);
      })
    );
    
    return results;
  }
  
  mergeResults(analysis, chunkResults) {
    const merged = {};
    
    chunkResults.forEach(result => {
      Object.entries(result.results).forEach(([key, value]) => {
        merged[key] = (merged[key] || 0) + value;
      });
    });
    
    return {
      analysis,
      totalRecords: chunkResults.reduce((sum, r) => sum + r.count, 0),
      results: merged
    };
  }
  
  async destroy() {
    await this.pool.destroy();
  }
}

// 使用示例
(async () => {
  const analyzer = new LogAnalyzer();
  
  // 模拟 1000 万条日志
  const logs = Array.from({ length: 10000000 }, (_, i) => ({
    ip: `192.168.${Math.floor(i / 65536)}.${i % 256}`,
    timestamp: Date.now() - Math.random() * 86400000,
    statusCode: [200, 404, 500][Math.floor(Math.random() * 3)]
  }));
  
  console.time('分析');
  const results = await analyzer.analyze(logs, [
    'ip-distribution',
    'hourly-traffic',
    'status-codes'
  ]);
  console.timeEnd('分析');
  
  console.log('分析结果:', results);
  
  await analyzer.destroy();
})();

// 输出示例：
// 将 10000000 条日志分为 100 片
// 分析: 95s  ← 比单线程快 6 倍
```

## 📝 五、总结

### 核心要点

✅ **线程池复用** Worker 避免频繁创建销毁开销  
✅ **Piscina** 是生产级线程池库，API 简单高效  
✅ **maxThreads** 推荐设置为 CPU 核心数  
✅ **maxQueue** 必须限制，避免内存溢出  
✅ **任务超时** 使用 AbortSignal 防止长时间阻塞  
✅ **监控统计** 跟踪队列大小、线程数、完成任务数  
✅ **优雅关闭** 使用 `pool.destroy()` 释放资源

### 最佳实践

1. **池大小**：`maxThreads = os.cpus().length` 或 `cpus().length - 1`
2. **队列限制**：设置合理的 `maxQueue`，避免内存问题
3. **超时控制**：为长任务设置超时，防止资源耗尽
4. **错误处理**：捕获 Worker 错误，避免影响整个池
5. **资源限制**：使用 `resourceLimits` 限制单个 Worker 内存

## ❓ 六、常见问题

### Q1: Piscina vs 自己实现线程池？

**A**: 优先使用 Piscina。

| 特性 | Piscina | 自己实现 |
|-----|---------|---------|
| **开发成本** | 低 | 高 |
| **稳定性** | 高（经过大量生产验证） | 取决于实现质量 |
| **功能** | 丰富（优先级、超时、限流） | 需自己实现 |
| **推荐** | ✅ **推荐** | ⚠️ 特殊需求 |

### Q2: 如何动态调整线程池大小？

**A**: Piscina 5.x 支持动态调整。

```javascript
const pool = new Piscina({
  filename: './worker.js',
  minThreads: 2,
  maxThreads: 8
});

// 根据负载动态调整
setInterval(() => {
  const queueSize = pool.queueSize;
  
  if (queueSize > 100 && pool.threads.length < 8) {
    console.log('队列积压，增加线程');
    // Piscina 会自动增加到 maxThreads
  }
}, 10000);
```

### Q3: 如何取消正在执行的任务？

**A**: 使用 AbortController。

```javascript
const controller = new AbortController();

const promise = pool.run(data, {
  signal: controller.signal
});

// 5 秒后取消任务
setTimeout(() => {
  controller.abort();
}, 5000);

try {
  await promise;
} catch (error) {
  if (error.name === 'AbortError') {
    console.log('任务已取消');
  }
}
```
