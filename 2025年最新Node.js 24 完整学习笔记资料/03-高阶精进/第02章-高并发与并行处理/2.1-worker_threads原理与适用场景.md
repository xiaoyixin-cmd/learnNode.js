# 2.1 Worker Threads 原理与适用场景

> **本节目标**：深入理解 Worker Threads 多线程机制，掌握 CPU 密集型任务的并行处理，提升 Node.js 应用性能。

## 📖 一、理论知识

### 1.1 Node.js 单线程模型的局限性

**单线程优势**：
- ✅ 简单的编程模型（无需考虑锁、竞态条件）
- ✅ 高效的 I/O 处理（事件驱动 + 非阻塞）
- ✅ 低内存占用

**单线程局限**：
```javascript
// ❌ CPU 密集任务会阻塞整个进程
app.get('/image/process', (req, res) => {
  // 图片处理耗时 2 秒
  const processed = processImage(req.file);
  res.send(processed);
});

// 在这 2 秒内：
// - 所有其他请求都被阻塞
// - 服务器无法响应任何请求
```

### 1.2 Worker Threads vs Cluster vs Child Process

| 特性 | Worker Threads | Cluster | Child Process |
|-----|----------------|---------|---------------|
| **使用场景** | CPU 密集任务 | 利用多核 CPU | 独立进程隔离 |
| **内存** | 共享内存 | 独立内存 | 独立内存 |
| **通信方式** | MessagePort | IPC | IPC / stdio |
| **启动开销** | 低（~30ms） | 高（~200ms） | 高（~200ms） |
| **隔离级别** | 低 | 高 | 高 |
| **适用场景** | 并行计算 | 负载均衡 | 执行外部命令 |

**选择指南**：

```
需求：处理大量并发请求
└─→ 使用 Cluster（多进程负载均衡）

需求：执行 CPU 密集计算（图片处理、加密）
└─→ 使用 Worker Threads（共享内存，快速通信）

需求：执行外部程序（FFmpeg、ImageMagick）
└─→ 使用 Child Process
```

### 1.3 Worker Threads 架构

```
┌─────────────────────────────────────────┐
│          主线程 (Main Thread)           │
│                                         │
│  ┌─────────────────────────────────┐   │
│  │  Event Loop                     │   │
│  │  ├─ HTTP 请求处理               │   │
│  │  └─ 分发任务到 Worker           │   │
│  └─────────────────────────────────┘   │
│              ↓ MessagePort              │
├─────────────────────────────────────────┤
│       Worker Thread 1                   │
│  ┌─────────────────────────────────┐   │
│  │  独立的 Event Loop               │   │
│  │  执行 CPU 密集任务               │   │
│  └─────────────────────────────────┘   │
├─────────────────────────────────────────┤
│       Worker Thread 2                   │
│  ┌─────────────────────────────────┐   │
│  │  独立的 Event Loop               │   │
│  │  执行 CPU 密集任务               │   │
│  └─────────────────────────────────┘   │
└─────────────────────────────────────────┘

共享内存区域：SharedArrayBuffer
```

### 2.2 实战示例：图片并行处理

#### 场景需求

- 用户上传 100 张图片
- 需要压缩 + 添加水印 + 生成缩略图
- 单线程处理需要 200 秒，要求降低到 20 秒以内

#### 实现方案

**Step 1：Worker 文件（图片处理逻辑）**

```javascript
// workers/imageProcessor.js
const { parentPort, workerData } = require('worker_threads');
const sharp = require('sharp');
const fs = require('fs');

async function processImage(inputPath, outputPath) {
  try {
    // 1. 压缩图片
    await sharp(inputPath)
      .resize(800, 600, { fit: 'inside' })
      .jpeg({ quality: 80 })
      .toFile(outputPath);

    // 2. 添加水印
    await sharp(outputPath)
      .composite([{
        input: './watermark.png',
        gravity: 'southeast'
      }])
      .toFile(outputPath + '.watermarked.jpg');

    // 3. 生成缩略图
    await sharp(inputPath)
      .resize(200, 150)
      .toFile(outputPath + '.thumb.jpg');

    return {
      success: true,
      outputPath,
      size: fs.statSync(outputPath).size
    };
  } catch (error) {
    return {
      success: false,
      error: error.message
    };
  }
}

// 执行任务
processImage(workerData.inputPath, workerData.outputPath)
  .then(result => {
    parentPort.postMessage(result);
  })
  .catch(error => {
    parentPort.postMessage({ success: false, error: error.message });
  });
```

**Step 2：主线程（任务分发）**

```javascript
// main.js
const { Worker } = require('worker_threads');
const path = require('path');

/**
 * 并行处理多个图片
 * @param {Array<string>} imagePaths - 图片路径数组
 * @param {number} maxWorkers - 最大并发 Worker 数量
 */
async function processImagesParallel(imagePaths, maxWorkers = 4) {
  const results = [];
  const queue = [...imagePaths];
  const activeWorkers = new Set();

  return new Promise((resolve, reject) => {
    // 启动 Worker
    function startWorker() {
      if (queue.length === 0 && activeWorkers.size === 0) {
        // 所有任务完成
        resolve(results);
        return;
      }

      if (queue.length === 0 || activeWorkers.size >= maxWorkers) {
        return;
      }

      const inputPath = queue.shift();
      const outputPath = path.join('./output', path.basename(inputPath));

      const worker = new Worker('./workers/imageProcessor.js', {
        workerData: { inputPath, outputPath }
      });

      activeWorkers.add(worker);

      worker.on('message', (result) => {
        results.push({ inputPath, ...result });
        console.log(`✅ 处理完成: ${inputPath}`);
        
        // 清理 Worker
        worker.terminate();
        activeWorkers.delete(worker);
        
        // 启动下一个任务
        startWorker();
      });

      worker.on('error', (error) => {
        results.push({ inputPath, success: false, error: error.message });
        console.error(`❌ 处理失败: ${inputPath}`, error);
        
        worker.terminate();
        activeWorkers.delete(worker);
        startWorker();
      });

      // 递归启动下一个 Worker
      startWorker();
    }

    // 启动初始 Workers
    for (let i = 0; i < maxWorkers; i++) {
      startWorker();
    }
  });
}

// 使用示例
const imagePaths = Array.from({ length: 100 }, (_, i) => `./images/img${i}.jpg`);

console.time('图片处理');
processImagesParallel(imagePaths, 8)  // 8 个并发 Worker
  .then(results => {
    console.timeEnd('图片处理');
    
    const success = results.filter(r => r.success).length;
    const failed = results.filter(r => !r.success).length;
    
    console.log(`✅ 成功: ${success}, ❌ 失败: ${failed}`);
  });
```

**性能对比**：

```
单线程处理 100 张图片: 200 秒
8 个 Worker 并行处理:   25 秒  ← 提升 8 倍性能 ✅
```

```javascript
// main.js
const { Worker } = require('worker_threads');
const fs = require('fs');

/**
 * 在 Worker 中解析大 JSON
 * @param {string} filePath - JSON 文件路径
 */
function parseJSONInWorker(filePath) {
  return new Promise((resolve, reject) => {
    // 读取文件（异步，不阻塞）
    fs.readFile(filePath, 'utf8', (err, jsonString) => {
      if (err) return reject(err);

      // 在 Worker 中解析
      const worker = new Worker('./workers/jsonParser.js', {
        workerData: { jsonString }
      });

      worker.on('message', (result) => {
        if (result.success) {
          console.log(`✅ 解析成功，处理了 ${result.count} 条记录`);
          resolve(result.data);
        } else {
          reject(new Error(result.error));
        }
        worker.terminate();
      });

      worker.on('error', reject);
    });
  });
}

// Express 集成示例
const express = require('express');
const app = express();

app.get('/api/data/parse', async (req, res) => {
  try {
    // 异步解析，不阻塞主线程
    const data = await parseJSONInWorker('./data/large.json');
    res.json({ success: true, data });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

### 2.4 性能监控

```javascript
// monitor/workerMetrics.js
const { Worker } = require('worker_threads');
const { performance } = require('perf_hooks');

class WorkerMonitor {
  constructor() {
    this.metrics = {
      totalCreated: 0,
      active: 0,
      completed: 0,
      failed: 0,
      avgDuration: 0
    };
    this.durations = [];
  }

  createWorker(script, options = {}) {
    const start = performance.now();
    this.metrics.totalCreated++;
    this.metrics.active++;

    const worker = new Worker(script, options);

    worker.on('exit', (code) => {
      const duration = performance.now() - start;
      this.durations.push(duration);
      
      this.metrics.active--;
      if (code === 0) {
        this.metrics.completed++;
      } else {
        this.metrics.failed++;
      }

      // 计算平均耗时
      this.metrics.avgDuration = 
        this.durations.reduce((a, b) => a + b, 0) / this.durations.length;
    });

    return worker;
  }

  getMetrics() {
    return {
      ...this.metrics,
      successRate: (this.metrics.completed / this.metrics.totalCreated * 100).toFixed(2) + '%'
    };
  }
}

module.exports = WorkerMonitor;
```

```javascript
// 定期输出指标

  console.log('Worker 指标:', monitor.getMetrics());
```

#### 问题代码

```javascript
// main.js
const globalConfig = { apiKey: 'secret123' };
```

```javascript
// services/videoService.js
const { Worker } = require('worker_threads');
const path = require('path');

class VideoTranscoder {
  constructor(maxWorkers = 4) {
    this.maxWorkers = maxWorkers;
    this.queue = [];
    this.activeWorkers = new Set();
  }

  async transcode(inputPath, formats = ['mp4', 'webm']) {
    const tasks = formats.map(format => ({
      inputPath,
      outputPath: inputPath.replace(/\.\w+$/, `.${format}`),
      format
    }));

    return Promise.all(tasks.map(task => this.addTask(task)));
  }

  addTask(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this.processQueue();
    });
  }

  processQueue() {
    if (this.queue.length === 0 || this.activeWorkers.size >= this.maxWorkers) {
      return;
    }

    const { task, resolve, reject } = this.queue.shift();

    const worker = new Worker('./workers/videoTranscoder.js', {
      workerData: task
    });

    this.activeWorkers.add(worker);

    worker.on('message', (result) => {
      if (result.success) {
        resolve(result);
      } else {
        reject(new Error(result.error));
      }
      
      worker.terminate();
      this.activeWorkers.delete(worker);
      this.processQueue();
    });

    worker.on('error', (error) => {
      reject(error);
      worker.terminate();
      this.activeWorkers.delete(worker);
      this.processQueue();
    });
  }
}

module.exports = VideoTranscoder;
```

### 场景2：加密货币挖矿（教学用途）

#### 需求背景

- 模拟 Proof of Work 算法
- 需要暴力破解满足条件的哈希值
- 单线程需要 60 秒，要求降低到 10 秒以内

#### 实现方案

```javascript
// workers/miner.js
const { parentPort, workerData } = require('worker_threads');
const crypto = require('crypto');

function mine(data, difficulty, startNonce, endNonce) {
  const target = '0'.repeat(difficulty);
  
  for (let nonce = startNonce; nonce < endNonce; nonce++) {
    const hash = crypto
      .createHash('sha256')
      .update(data + nonce)
      .digest('hex');
    
    if (hash.startsWith(target)) {
      return { found: true, nonce, hash };
    }
  }
  
  return { found: false };
}

const result = mine(
  workerData.data,
  workerData.difficulty,
  workerData.startNonce,
  workerData.endNonce
);

parentPort.postMessage(result);
```

```javascript
// main.js
const { Worker } = require('worker_threads');
const os = require('os');

async function parallelMine(data, difficulty) {
  const numWorkers = os.cpus().length;  // 使用所有 CPU 核心
  const rangePerWorker = Math.floor(1e9 / numWorkers);

  const workers = Array.from({ length: numWorkers }, (_, i) => {
    return new Promise((resolve) => {
      const worker = new Worker('./workers/miner.js', {
        workerData: {
          data,
          difficulty,
          startNonce: i * rangePerWorker,
          endNonce: (i + 1) * rangePerWorker
        }
      });

      worker.on('message', (result) => {
        if (result.found) {
          console.log(`✅ Worker ${i} 找到哈希！`);
          resolve(result);
          
          // 终止其他 Worker
          workers.forEach(w => w !== worker && w.terminate());
        }
        worker.terminate();
      });
    });
  });

  return Promise.race(workers);  // 返回第一个找到的结果
}

// 使用示例
console.time('挖矿');
parallelMine('Hello Blockchain', 5)
  .then(result => {
    console.timeEnd('挖矿');
    console.log('结果:', result);
  });

// 输出示例：
// ✅ Worker 3 找到哈希！
// 挖矿: 8.5s
// 结果: { found: true, nonce: 3456789, hash: '00000abc...' }
```

```javascript
const { Worker } = require('worker_threads');

function getMemoryUsage() {
  return (process.memoryUsage().heapUsed / 1024 / 1024).toFixed(2) + ' MB';
}

console.log('初始内存:', getMemoryUsage());

const workers = [];
for (let i = 0; i < 10; i++) {
  workers.push(new Worker('./worker.js'));
  console.log(`${i + 1} 个 Worker:`, getMemoryUsage());
}

// 输出示例：
// 初始内存: 15.23 MB
// 1 个 Worker: 18.45 MB  (+3.22 MB)
// 2 个 Worker: 21.67 MB  (+3.22 MB)
// ...
// 10 个 Worker: 47.43 MB (+32.2 MB)

// 结论：每个 Worker 约占用 3-5 MB 内存
```

## 📝 六、总结

### 核心要点

✅ **Worker Threads** 适用于 CPU 密集型任务，不适合 I/O 密集任务  
✅ **创建开销** 约 30ms，适合长任务（> 50ms）  
✅ **内存隔离** Worker 无法访问主线程变量，需通过消息传递  
✅ **数据传递** 仅支持可序列化对象（无法传递函数）  
✅ **错误处理** 必须监听 `error` 和 `exit` 事件  
✅ **并发控制** 建议 Worker 数量 = CPU 核心数  
✅ **性能提升** 可达 4-8 倍（取决于 CPU 核心数）

### 最佳实践

1. **任务选择**：只对耗时 > 50ms 的 CPU 任务使用 Worker
2. **Worker 数量**：`os.cpus().length` 或 `os.cpus().length - 1`
3. **错误处理**：始终监听 `error` 和 `exit` 事件
4. **资源清理**：任务完成后调用 `worker.terminate()`
5. **监控指标**：跟踪 Worker 创建数、成功率、平均耗时

## ❓ 七、常见问题

### Q1: Worker Threads 和 Cluster 如何选择？

**A**: 根据场景选择。

| 场景 | 推荐方案 |
|-----|---------|
| **并发 HTTP 请求** | Cluster（多进程负载均衡） |
| **CPU 密集计算** | Worker Threads（共享内存） |
| **两者结合** | Cluster + Worker Threads |

**组合示例**：

```javascript
// Cluster 负责负载均衡
const cluster = require('cluster');
const os = require('os');

if (cluster.isMaster) {
  for (let i = 0; i < os.cpus().length; i++) {
    cluster.fork();
  }
} else {
  // 每个 Worker 进程内部使用 Worker Threads
  const express = require('express');
  const { Worker } = require('worker_threads');
  
  const app = express();
  
  app.get('/compute', (req, res) => {
    const worker = new Worker('./compute.js');
    worker.on('message', result => res.json(result));
  });
  
  app.listen(3000);
}
```

### Q2: SharedArrayBuffer 和 MessagePort 如何选择？

**A**: 根据数据大小和通信频率选择。

| 特性 | MessagePort | SharedArrayBuffer |
|-----|------------|-------------------|
| **使用场景** | 一次性传递数据 | 频繁读写共享数据 |
| **性能** | 中等（需要序列化） | 高（直接内存访问） |
| **复杂度** | 低 | 高（需要 Atomics） |
| **数据大小** | < 10MB | > 10MB |

**示例**：

```javascript
// SharedArrayBuffer 示例
const sharedBuffer = new SharedArrayBuffer(1024);
const sharedArray = new Int32Array(sharedBuffer);

// 主线程写入
sharedArray[0] = 42;

const worker = new Worker('./worker.js', {
  workerData: { sharedBuffer }
});

// worker.js 读取
const { workerData } = require('worker_threads');
const sharedArray = new Int32Array(workerData.sharedBuffer);
console.log(sharedArray[0]);  // 42（无需消息传递）
```

### Q3: 如何限制 Worker 数量避免系统过载？

**A**: 使用队列 + 信号量。

```javascript
class WorkerPool {
  constructor(maxWorkers = 4) {
    this.maxWorkers = maxWorkers;
    this.activeWorkers = 0;
    this.queue = [];
  }

  async run(script, workerData) {
    // 等待可用 Worker
    while (this.activeWorkers >= this.maxWorkers) {
      await new Promise(resolve => this.queue.push(resolve));
    }

    this.activeWorkers++;

    try {
      return await this.executeWorker(script, workerData);
    } finally {
      this.activeWorkers--;
      
      // 唤醒队列中的任务
      const resolve = this.queue.shift();
      if (resolve) resolve();
    }
  }

  executeWorker(script, workerData) {
    return new Promise((resolve, reject) => {
      const worker = new Worker(script, { workerData });
      worker.on('message', resolve);
      worker.on('error', reject);
    });
  }
}

// 使用
const pool = new WorkerPool(4);

// 即使启动 100 个任务，同时最多只有 4 个 Worker
const tasks = Array.from({ length: 100 }, (_, i) =>
  pool.run('./worker.js', { taskId: i })
);

await Promise.all(tasks);
```
