## 🎯 RBAC（基于角色的访问控制）

### 1. RBAC 核心概念

**RBAC** = Role-Based Access Control

```
用户（User） → 角色（Role） → 权限（Permission） → 资源（Resource）

示例：
Alice → Admin → [create, read, update, delete] → Articles
Bob   → Editor → [create, read, update]        → Articles
Carol → Viewer → [read]                        → Articles
```

---

### 2. 使用 accesscontrol 实现 RBAC

```bash
npm install accesscontrol
```

```javascript
// rbac.js
import { AccessControl } from 'accesscontrol';

const ac = new AccessControl();

// 定义角色和权限
ac.grant('guest')
  .readAny('article')           // 可读任何文章
  .readOwn('profile');          // 可读自己的资料

ac.grant('user')
  .extend('guest')              // 继承 guest 权限
  .createOwn('article')         // 可创建自己的文章
  .updateOwn('article')
  .deleteOwn('article')
  .updateOwn('profile');

ac.grant('editor')
  .extend('user')
  .updateAny('article')         // 可编辑任何文章
  .deleteAny('article');

ac.grant('admin')
  .extend('editor')
  .createAny('user')            // 可创建用户
  .readAny('user')
  .updateAny('user')
  .deleteAny('user')
  .createAny('role')
  .updateAny('role')
  .deleteAny('role');

export default ac;
```

### 3. RBAC 中间件

```javascript
// middleware/checkPermission.js
import ac from '../rbac.js';

// 检查权限
export function checkPermission(action, resource) {
  return (req, res, next) => {
    const userRole = req.user.role;
    const permission = ac.can(userRole)[action](resource);

    if (!permission.granted) {
      return res.status(403).json({
        error: 'Access denied',
        required: `${action}:${resource}`,
        yourRole: userRole,
      });
    }

    // 将允许的属性存储到 req
    req.permission = permission;
    next();
  };
}

// 检查资源所有权
export async function checkOwnership(Model) {
  return async (req, res, next) => {
    const resourceId = req.params.id;
    const userId = req.user.id;

    const resource = await Model.findById(resourceId);

    if (!resource) {
      return res.status(404).json({ error: 'Resource not found' });
    }

    // 检查所有权
    if (resource.userId.toString() !== userId && req.user.role !== 'admin') {
      return res.status(403).json({ error: 'Not your resource' });
    }

    req.resource = resource;
    next();
  };
}
```

### 4. 使用示例

```javascript
import express from 'express';
import { checkPermission, checkOwnership } from './middleware/checkPermission.js';
import { authenticateToken } from './middleware/auth.js';
import Article from './models/Article.js';

const app = express();

// 读取文章（任何人）
app.get('/api/articles/:id',
  authenticateToken,
  checkPermission('readAny', 'article'),
  async (req, res) => {
    const article = await Article.findById(req.params.id);

    // 使用 permission 过滤属性
    const filteredArticle = req.permission.filter(article);
    res.json(filteredArticle);
  }
);

// 更新文章（作者或管理员）
app.put('/api/articles/:id',
  authenticateToken,
  checkPermission('updateOwn', 'article'),
  checkOwnership(Article),
  async (req, res) => {
    const article = req.resource;

    // 只更新允许的字段
    const allowedFields = req.permission.attributes;
    const updates = {};

    for (const field of allowedFields) {
      if (req.body[field] !== undefined) {
        updates[field] = req.body[field];
      }
    }

    Object.assign(article, updates);
    await article.save();

    res.json(article);
  }
);

// 删除文章（作者或管理员）
app.delete('/api/articles/:id',
  authenticateToken,
  checkPermission('deleteOwn', 'article'),
  checkOwnership(Article),
  async (req, res) => {
    await req.resource.delete();
    res.json({ message: 'Article deleted' });
  }
);

// 创建用户（仅管理员）
app.post('/api/users',
  authenticateToken,
  checkPermission('createAny', 'user'),
  async (req, res) => {
    const user = await User.create(req.body);
    res.json(user);
  }
);
```

### 5. 动态角色管理

```javascript
// 数据库存储角色和权限
const RoleSchema = new mongoose.Schema({
  name: String,
  permissions: [{
    resource: String,
    actions: [String],  // ['create', 'read', 'update', 'delete']
    attributes: [String],  // ['*'] or ['title', 'content']
  }],
});

const Role = mongoose.model('Role', RoleSchema);

// 从数据库加载 RBAC 配置
async function loadRBACFromDB() {
  const roles = await Role.find();

  const ac = new AccessControl();

  for (const role of roles) {
    for (const perm of role.permissions) {
      const grant = ac.grant(role.name);

      for (const action of perm.actions) {
        grant[action + 'Any'](perm.resource, perm.attributes);
      }
    }
  }

  return ac;
}

// 启动时加载
const ac = await loadRBACFromDB();
```

## 🎯 ABAC（基于属性的访问控制）

### 1. ABAC 核心概念

**ABAC** = Attribute-Based Access Control

```
决策基于：
- 用户属性：角色、部门、级别
- 资源属性：所有者、状态、分类
- 环境属性：时间、IP、设备
- 操作属性：读、写、删除
```

---

### 2. 使用 casbin 实现 ABAC

```bash
npm install casbin
```

**模型定义（model.conf）：**

```ini
[request_definition]
r = sub, obj, act

[policy_definition]
p = sub, obj, act

[policy_effect]
e = some(where (p.eft == allow))

[matchers]
m = r.sub.role == p.sub && r.obj.owner == r.sub.id && r.act == p.act || r.sub.role == "admin"
```

**策略定义（policy.csv）：**

```csv
p, user, article, read
p, user, article, write
p, editor, article, publish
p, admin, *, *
```

---

### 3. Casbin 实现

```javascript
import { newEnforcer } from 'casbin';
import path from 'path';

// 初始化 Casbin
const enforcer = await newEnforcer(
  path.join(__dirname, 'model.conf'),
  path.join(__dirname, 'policy.csv')
);

// ABAC 中间件
export async function checkABAC(req, res, next) {
  const user = req.user;
  const resource = req.resource;
  const action = req.method.toLowerCase();

  // 构建请求上下文
  const subject = {
    id: user.id,
    role: user.role,
    department: user.department,
  };

  const object = {
    id: resource.id,
    owner: resource.userId,
    status: resource.status,
    department: resource.department,
  };

  // 执行策略判断
  const allowed = await enforcer.enforce(subject, object, action);

  if (!allowed) {
    return res.status(403).json({ error: 'Access denied' });
  }

  next();
}
```

## 🎯 JWT 黑名单与刷新令牌

### 1. JWT 的安全问题

**问题：**
- JWT 无法主动撤销（在过期前一直有效）
- 用户登出后，JWT 仍然有效
- 如果 JWT 泄露，攻击者可以一直使用

**解决方案：**
1. **JWT 黑名单**（Redis）
2. **刷新令牌轮换**（Refresh Token Rotation）
3. **短期访问令牌 + 长期刷新令牌**

---

### 2. JWT 黑名单实现

```javascript
import jwt from 'jsonwebtoken';
import { createClient } from 'redis';

const redis = createClient({ url: process.env.REDIS_URL });
await redis.connect();

// 生成 JWT
export function generateTokens(user) {
  // 访问令牌（15 分钟）
  const accessToken = jwt.sign(
    { id: user.id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '15m' }
  );

  // 刷新令牌（7 天）
  const refreshToken = jwt.sign(
    { id: user.id, type: 'refresh' },
    process.env.REFRESH_SECRET,
    { expiresIn: '7d' }
  );

  return { accessToken, refreshToken };
}

// 登出（加入黑名单）
export async function logout(req, res) {
  const token = req.headers.authorization?.split(' ')[1];

  if (!token) {
    return res.status(400).json({ error: 'No token provided' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);

    // 计算剩余有效时间
    const expiresIn = decoded.exp - Math.floor(Date.now() / 1000);

    // 加入黑名单（Redis）
    await redis.setEx(`blacklist:${token}`, expiresIn, '1');

    res.json({ message: 'Logged out successfully' });
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
}

// 验证 JWT（检查黑名单）
export async function authenticateToken(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'No token provided' });
  }

  // 检查黑名单
  const isBlacklisted = await redis.exists(`blacklist:${token}`);
  if (isBlacklisted) {
    return res.status(401).json({ error: 'Token has been revoked' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
}
```

### 3. 刷新令牌轮换

```javascript
// 刷新访问令牌
export async function refreshAccessToken(req, res) {
  const { refreshToken } = req.body;

  if (!refreshToken) {
    return res.status(400).json({ error: 'No refresh token' });
  }

  try {
    const decoded = jwt.verify(refreshToken, process.env.REFRESH_SECRET);

    if (decoded.type !== 'refresh') {
      return res.status(401).json({ error: 'Invalid token type' });
    }

    // 检查刷新令牌黑名单
    const isBlacklisted = await redis.exists(`refresh_blacklist:${refreshToken}`);
    if (isBlacklisted) {
      return res.status(401).json({ error: 'Refresh token revoked' });
    }

    // 查询用户
    const user = await User.findById(decoded.id);
    if (!user) {
      return res.status(401).json({ error: 'User not found' });
    }

    // 撤销旧刷新令牌（轮换）
    const oldRefreshExpiry = decoded.exp - Math.floor(Date.now() / 1000);
    await redis.setEx(`refresh_blacklist:${refreshToken}`, oldRefreshExpiry, '1');

    // 生成新的访问令牌和刷新令牌
    const newTokens = generateTokens(user);

    res.json(newTokens);
  } catch (error) {
    res.status(401).json({ error: 'Invalid refresh token' });
  }
}
```

### 4. 防止 XSS 窃取 Token

```javascript
// ✅ 存储在 httpOnly Cookie（无法被 JavaScript 访问）
app.post('/login', async (req, res) => {
  const { username, password } = req.body;

  const user = await User.findOne({ username });
  if (!user || !(await bcrypt.compare(password, user.password))) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const { accessToken, refreshToken } = generateTokens(user);

  // 设置 httpOnly Cookie
  res.cookie('accessToken', accessToken, {
    httpOnly: true,
    secure: true,      // HTTPS only
    sameSite: 'strict', // 防止 CSRF
    maxAge: 15 * 60 * 1000, // 15 分钟
  });

  res.cookie('refreshToken', refreshToken, {
    httpOnly: true,
    secure: true,
    sameSite: 'strict',
    maxAge: 7 * 24 * 60 * 60 * 1000, // 7 天
  });

  res.json({ message: 'Login successful', user: { id: user.id, role: user.role } });
});

// 从 Cookie 读取 Token
export async function authenticateTokenFromCookie(req, res, next) {

  if (!token) {

  }

  // 检查黑名单
  const isBlacklisted = await redis.exists(`blacklist:${token}`);
  if (isBlacklisted) {
    return res.status(401).json({ error: 'Token revoked' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
}
```

## 🎯 双因素认证（2FA / TOTP）

### 1. TOTP 原理

**TOTP** = Time-based One-Time Password

```
原理：
1. 服务器生成密钥（Secret）
2. 用户扫描二维码添加到 Google Authenticator
3. App 基于密钥 + 当前时间生成 6 位数字
4. 用户输入验证码
5. 服务器验证（时间窗口：±30秒）
```

### 3. 启用 2FA

```javascript
import speakeasy from 'speakeasy';
import QRCode from 'qrcode';

// 生成 2FA 密钥
app.post('/api/2fa/enable', authenticateToken, async (req, res) => {
  const user = await User.findById(req.user.id);

  // 生成密钥
  const secret = speakeasy.generateSecret({
    name: `MyApp (${user.email})`,
    issuer: 'MyApp',
  });

  // 保存密钥（临时，待验证后正式启用）
  user.twoFactorTempSecret = secret.base32;
  await user.save();

  // 生成二维码
  const qrCodeUrl = await QRCode.toDataURL(secret.otpauth_url);

  res.json({
    secret: secret.base32,
    qrCode: qrCodeUrl,
  });
});

// 验证并启用 2FA
app.post('/api/2fa/verify', authenticateToken, async (req, res) => {
  const { token } = req.body;
  const user = await User.findById(req.user.id);

  if (!user.twoFactorTempSecret) {
    return res.status(400).json({ error: '2FA not initiated' });
  }

  // 验证 TOTP 令牌
  const verified = speakeasy.totp.verify({
    secret: user.twoFactorTempSecret,
    encoding: 'base32',
    token: token,
    window: 2,  // ±2 个时间窗口（30秒×2 = ±60秒）
  });

  if (!verified) {
    return res.status(401).json({ error: 'Invalid token' });
  }

  // 正式启用 2FA
  user.twoFactorSecret = user.twoFactorTempSecret;
  user.twoFactorEnabled = true;
  user.twoFactorTempSecret = undefined;

  // 生成备用恢复码
  const recoveryCodes = [];
  for (let i = 0; i < 10; i++) {
    const code = crypto.randomBytes(4).toString('hex').toUpperCase();
    recoveryCodes.push(code);
  }

  user.recoveryCodes = recoveryCodes.map((code) =>
    bcrypt.hashSync(code, 10)
  );

  await user.save();

  res.json({
    message: '2FA enabled successfully',
    recoveryCodes,  // 仅此一次显示，让用户保存
  });
});
```

### 4. 登录时验证 2FA

```javascript
// 第一步：用户名 + 密码
app.post('/api/login', async (req, res) => {
  const { username, password } = req.body;

  const user = await User.findOne({ username });
  if (!user || !(await bcrypt.compare(password, user.password))) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // 如果启用了 2FA
  if (user.twoFactorEnabled) {
    // 生成临时令牌（5 分钟有效）
    const tempToken = jwt.sign(
      { id: user.id, requiresTwoFactor: true },
      process.env.JWT_SECRET,
      { expiresIn: '5m' }
    );

    return res.json({
      requires2FA: true,
      tempToken,
    });
  }

  // 未启用 2FA，直接登录
  const { accessToken, refreshToken } = generateTokens(user);
  res.json({ accessToken, refreshToken });
});

// 第二步：验证 2FA 令牌
app.post('/api/login/2fa', async (req, res) => {
  const { tempToken, token, useRecoveryCode } = req.body;

  try {
    const decoded = jwt.verify(tempToken, process.env.JWT_SECRET);

    if (!decoded.requiresTwoFactor) {
      return res.status(400).json({ error: 'Invalid temp token' });
    }

    const user = await User.findById(decoded.id);

    // 使用备用恢复码
    if (useRecoveryCode) {
      const isValidRecoveryCode = user.recoveryCodes.some((hashedCode) =>
        bcrypt.compareSync(token, hashedCode)
      );

      if (!isValidRecoveryCode) {
        return res.status(401).json({ error: 'Invalid recovery code' });
      }

      // 移除已使用的恢复码
      user.recoveryCodes = user.recoveryCodes.filter((hashedCode) =>
        !bcrypt.compareSync(token, hashedCode)
      );
      await user.save();

    } else {
      // 验证 TOTP 令牌
      const verified = speakeasy.totp.verify({
        secret: user.twoFactorSecret,
        encoding: 'base32',
        token: token,
        window: 2,
      });

      if (!verified) {
        return res.status(401).json({ error: 'Invalid 2FA token' });
      }
    }

    // 生成正式令牌
    const { accessToken, refreshToken } = generateTokens(user);
    res.json({ accessToken, refreshToken });

  } catch (error) {
    res.status(401).json({ error: 'Temp token expired' });
  }
});
```

## 📚 FAQ

### 1. RBAC vs ABAC 如何选择？

| 场景 | 推荐方案 |
|------|---------|
| 简单权限模型 | RBAC |
| 复杂业务规则 | ABAC |
| 需要动态权限 | ABAC |
| 多租户系统 | ABAC |

### 2. JWT 黑名单会影响性能吗？

**是的**。每次请求都要查询 Redis。

**优化方案：**
1. 使用短期 JWT（15 分钟）
2. Redis 集群
3. 只对敏感操作启用黑名单

### 3. 2FA 如何处理时间同步问题？

```javascript
// 使用更大的时间窗口
speakeasy.totp.verify({
  secret: user.twoFactorSecret,
  encoding: 'base32',
  token: token,
  window: 2,  // ±60 秒
});
```
