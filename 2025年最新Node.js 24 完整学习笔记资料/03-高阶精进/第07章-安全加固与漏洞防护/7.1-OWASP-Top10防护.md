## 🎯 A01: 权限控制失效（Broken Access Control）

### 漏洞原理

**访问控制失效**指用户可以访问超出其权限范围的资源。

**常见场景：**
```javascript
// ❌ 错误：直接从 URL 参数获取用户 ID
app.get('/api/users/:userId/profile', async (req, res) => {
  const profile = await User.findById(req.params.userId);
  res.json(profile);
});

// 攻击：用户可以修改 userId 参数访问其他用户资料
// GET /api/users/12345/profile  ← 可以访问任意用户
```

---

### 防护措施

**1. 强制访问控制检查**

```javascript
// ✅ 正确：验证用户身份
app.get('/api/users/:userId/profile', authenticateToken, async (req, res) => {
  const requestedUserId = req.params.userId;
  const currentUserId = req.user.id;  // 从 JWT 解析

  // 检查权限
  if (requestedUserId !== currentUserId && !req.user.isAdmin) {
    return res.status(403).json({ error: 'Access denied' });
  }

  const profile = await User.findById(requestedUserId);
  res.json(profile);
});
```

---

**2. 基于角色的访问控制（RBAC）**

```javascript
import { AccessControl } from 'accesscontrol';

const ac = new AccessControl();

// 定义角色和权限
ac.grant('guest')
  .readOwn('profile')
  .readAny('article');

ac.grant('user')
  .extend('guest')
  .createOwn('article')
  .updateOwn('article')
  .deleteOwn('article');

ac.grant('admin')
  .extend('user')
  .readAny('profile')
  .updateAny('article')
  .deleteAny('article');

// 中间件：检查权限
function checkPermission(resource, action) {
  return (req, res, next) => {
    const role = req.user.role;
    const permission = ac.can(role)[action](resource);

    if (!permission.granted) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }

    next();
  };
}

// 使用
app.delete('/api/articles/:id', 
  authenticateToken, 
  checkPermission('article', 'deleteOwn'),
  async (req, res) => {
    // 检查资源所有权
    const article = await Article.findById(req.params.id);
    
    if (article.authorId !== req.user.id && req.user.role !== 'admin') {
      return res.status(403).json({ error: 'Not your article' });
    }

    await article.delete();
    res.json({ message: 'Deleted successfully' });
  }
);
```

**3. 基于属性的访问控制（ABAC）**

```javascript
import { newEnforcer } from 'casbin';

// model.conf
// [request_definition]
// r = sub, obj, act
//
// [policy_definition]
// p = sub, obj, act
//
// [policy_effect]
// e = some(where (p.eft == allow))
//

  if (!allowed) {
    return res.status(403).json({ error: 'Access denied' });
  }
  
  next();
}
```

## 🎯 A02: 加密失败（Cryptographic Failures）

### 漏洞原理

**加密失败**指敏感数据未加密或使用弱加密算法。

**常见错误：**
```javascript
// ❌ 错误1：明文存储密码
await User.create({
  username: 'alice',
  password: 'password123',  // 明文！
});

// ❌ 错误2：使用 MD5（已破解）
const crypto = require('crypto');
const hash = crypto.createHash('md5').update(password).digest('hex');

// ❌ 错误3：HTTP 传输敏感数据
app.post('/login', (req, res) => {
  // 密码通过 HTTP 传输，可被中间人窃取
});
```

---

### 防护措施

**1. 强密码哈希（bcrypt / Argon2）**

```javascript
import bcrypt from 'bcrypt';
import argon2 from 'argon2';

// ✅ 方法1：bcrypt（工业标准）
async function hashPasswordBcrypt(password) {
  const saltRounds = 12;  // 2^12 次迭代
  return await bcrypt.hash(password, saltRounds);
}

async function verifyPasswordBcrypt(password, hash) {
  return await bcrypt.compare(password, hash);
}

// ✅ 方法2：Argon2（2015 密码哈希竞赛冠军）
async function hashPasswordArgon2(password) {
  return await argon2.hash(password, {
    type: argon2.argon2id,  // 推荐：混合模式
    memoryCost: 65536,      // 64 MB
    timeCost: 3,            // 迭代次数
    parallelism: 4,         // 并行度
  });
}

async function verifyPasswordArgon2(password, hash) {
  return await argon2.verify(hash, password);
}

// 注册用户
app.post('/register', async (req, res) => {
  const { username, password } = req.body;

  const hashedPassword = await hashPasswordArgon2(password);

  await User.create({
    username,
    password: hashedPassword,  // ✅ 存储哈希值
  });

  res.json({ message: 'User created' });
});

// 登录验证
app.post('/login', async (req, res) => {
  const { username, password } = req.body;

  const user = await User.findOne({ username });
  if (!user) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const isValid = await verifyPasswordArgon2(password, user.password);
  if (!isValid) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const token = generateJWT(user);
  res.json({ token });
});
```

**2. 敏感数据加密存储**

```javascript
import crypto from 'crypto';

// AES-256-GCM 加密（推荐）
function encrypt(text, secretKey) {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-gcm', Buffer.from(secretKey, 'hex'), iv);
  
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  const authTag = cipher.getAuthTag();
  
  return {
    iv: iv.toString('hex'),
    encryptedData: encrypted,
    authTag: authTag.toString('hex'),
  };
}

function decrypt(encryptedData, secretKey, iv, authTag) {
  const decipher = crypto.createDecipheriv(
    'aes-256-gcm',
    Buffer.from(secretKey, 'hex'),
    Buffer.from(iv, 'hex')
  );
  
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));
  
  let decrypted = decipher.update(encryptedData, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  
  return decrypted;
}

// 使用示例
const SECRET_KEY = process.env.ENCRYPTION_KEY;  // 32 字节（64 字符 hex）

// 存储信用卡信息
app.post('/api/payment-methods', async (req, res) => {
  const { cardNumber } = req.body;

  const { iv, encryptedData, authTag } = encrypt(cardNumber, SECRET_KEY);

  await PaymentMethod.create({
    userId: req.user.id,
    cardNumberEncrypted: encryptedData,
    iv,
    authTag,
  });

  res.json({ message: 'Payment method added' });
});

// 读取信用卡信息
app.get('/api/payment-methods/:id', async (req, res) => {
  const method = await PaymentMethod.findById(req.params.id);

  const cardNumber = decrypt(
    method.cardNumberEncrypted,
    SECRET_KEY,
    method.iv,
    method.authTag
  );

  res.json({ cardNumber: `****-****-****-${cardNumber.slice(-4)}` });
});
```

## 🎯 A03: 注入攻击（Injection）

### 漏洞原理

**注入攻击**指将恶意代码注入到应用中执行。

**常见类型：**
- SQL 注入
- NoSQL 注入
- 命令注入
- XSS（跨站脚本）
- LDAP 注入

---

### 防护措施

**1. SQL 注入防护**

```javascript
import mysql from 'mysql2/promise';

// ❌ 错误：字符串拼接
app.get('/users', async (req, res) => {
  const { username } = req.query;
  const sql = `SELECT * FROM users WHERE username = '${username}'`;
  const [rows] = await db.query(sql);
  // 攻击：?username=' OR '1'='1
  // 生成：SELECT * FROM users WHERE username = '' OR '1'='1'
  // 结果：返回所有用户
});

// ✅ 正确：参数化查询
app.get('/users', async (req, res) => {
  const { username } = req.query;
  const sql = 'SELECT * FROM users WHERE username = ?';
  const [rows] = await db.query(sql, [username]);
  // 攻击无效：参数会被正确转义
  res.json(rows);
});
```

**2. NoSQL 注入防护**

```javascript
import express from 'express';
import mongoSanitize from 'express-mongo-sanitize';

const app = express();
app.use(express.json());

// ✅ 全局清理 MongoDB 操作符
app.use(mongoSanitize({
  replaceWith: '_',  // 将 $ 替换为 _
}));

// ❌ 错误：直接使用用户输入
app.post('/login', async (req, res) => {
  const { username, password } = req.body;
  const user = await User.findOne({ username, password });
  // 攻击：{"username": {"$ne": null}, "password": {"$ne": null}}
  // 绕过密码验证
});

// ✅ 正确：验证输入类型
app.post('/login', async (req, res) => {
  const { username, password } = req.body;

  // 验证输入是字符串
  if (typeof username !== 'string' || typeof password !== 'string') {
    return res.status(400).json({ error: 'Invalid input' });
  }

  const user = await User.findOne({ username });
  if (!user) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const isValid = await bcrypt.compare(password, user.password);
  if (!isValid) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  res.json({ token: generateJWT(user) });
});
```

**3. 命令注入防护**

```javascript
import { exec, execFile, spawn } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

// ❌ 错误：使用 exec 并拼接用户输入
app.post('/convert-image', async (req, res) => {
  const { filename } = req.body;
  await execAsync(`convert ${filename} output.jpg`);
  // 攻击：filename = "image.png; rm -rf /"
  // 执行：convert image.png; rm -rf / output.jpg
});

// ✅ 正确1：使用 execFile（不启动 shell）
app.post('/convert-image', async (req, res) => {
  const { filename } = req.body;

  // 白名单验证
  if (!/^[a-zA-Z0-9._-]+$/.test(filename)) {
    return res.status(400).json({ error: 'Invalid filename' });
  }

  const execFileAsync = promisify(execFile);
  await execFileAsync('convert', [filename, 'output.jpg']);
  res.json({ message: 'Converted' });
});

// ✅ 正确2：使用 spawn（更安全）
app.post('/convert-image', async (req, res) => {
  const { filename } = req.body;

  const child = spawn('convert', [filename, 'output.jpg'], {
    shell: false,  // 不使用 shell
  });

  child.on('close', (code) => {
    if (code === 0) {
      res.json({ message: 'Converted' });
    } else {
      res.status(500).json({ error: 'Conversion failed' });
    }
  });
});
```

### A10: 服务端请求伪造（SSRF）

```javascript
// ❌ 错误：允许任意 URL
app.post('/fetch-url', async (req, res) => {
  const { url } = req.body;
  const response = await fetch(url);
  // 攻击：url = "http://localhost:6379"（访问内部 Redis）
});

// ✅ 正确：URL 白名单
const ALLOWED_DOMAINS = ['api.example.com', 'cdn.example.com'];

app.post('/fetch-url', async (req, res) => {
  const { url } = req.body;

  const parsedUrl = new URL(url);

  // 检查域名白名单
  if (!ALLOWED_DOMAINS.includes(parsedUrl.hostname)) {
    return res.status(400).json({ error: 'Invalid domain' });
  }

  // 阻止内网 IP
  if (parsedUrl.hostname === 'localhost' || parsedUrl.hostname.startsWith('192.168.')) {
    return res.status(400).json({ error: 'Internal IP not allowed' });
  }

  const response = await fetch(url);
  res.json(await response.json());
});
```

## 📚 FAQ

### 1. 如何选择密码哈希算法？

| 算法 | 推荐度 | 特点 |
|------|--------|------|
| **Argon2** | ⭐⭐⭐⭐⭐ | 最新、抗 GPU/ASIC 攻击 |
| **bcrypt** | ⭐⭐⭐⭐ | 工业标准、成熟稳定 |
| **scrypt** | ⭐⭐⭐ | 内存密集、较少使用 |
| **PBKDF2** | ⭐⭐ | 较老、需高迭代次数 |
| **MD5/SHA1** | ❌ | 已破解，禁止使用 |

**推荐：** Argon2id（混合模式）

### 2. JWT 如何防止 XSS 窃取？

1. **存储在 httpOnly Cookie**（无法被 JS 访问）
2. **短期有效期**（15 分钟）
3. **刷新令牌轮换**
4. **CSP 防护**

### 3. 如何防止 DDoS？

1. **速率限制**：express-rate-limit
2. **反向代理**：Nginx、Cloudflare
3. **验证码**：reCAPTCHA
4. **熔断器**：opossum
