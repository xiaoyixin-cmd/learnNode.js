# 1.4 异步钩子实战

> **本节目标**：掌握 `async_hooks` 和 `AsyncLocalStorage` 的使用，实现分布式追踪和请求上下文传递，为生产环境提供完整的链路监控能力。

---

## 📖 一、理论知识

### 1.1 什么是异步钩子（Async Hooks）？

**异步钩子** 是 Node.js 提供的追踪异步资源生命周期的 API，可以监听异步操作的创建、执行和销毁。

#### 核心概念

```
HTTP 请求的异步链路：
┌─────────────────────────────────────┐
│ HTTP Request (asyncId: 1)          │
│  ├─ Database Query (asyncId: 2)    │
│  ├─ Redis GET (asyncId: 3)         │
│  │  └─ Promise (asyncId: 4)        │
│  └─ Response (asyncId: 5)          │
└─────────────────────────────────────┘

使用 async_hooks 可以追踪：
- asyncId: 每个异步操作的唯一 ID
- triggerAsyncId: 父异步操作的 ID
- type: 异步资源类型（TCPWRAP, PROMISE, Timeout...）
```

### 1.2 async_hooks vs AsyncLocalStorage

| 特性 | async_hooks | AsyncLocalStorage |
|-----|-------------|-------------------|
| **复杂度** | 高（需手动管理上下文） | 低（自动管理） |
| **性能开销** | ~10-20% | ~5-10% |
| **适用场景** | 深度定制追踪 | 请求上下文传递 |
| **推荐度** | ⚠️ 高级用户 | ✅ **推荐** |

**结论**：优先使用 `AsyncLocalStorage`，除非需要深度定制。

## 💻 二、使用指南

### 2.1 AsyncLocalStorage - 请求上下文传递

最常用的场景：为每个 HTTP 请求分配唯一 ID，在日志中关联所有操作。

#### 基础示例

```javascript
// context/request.js
const { AsyncLocalStorage } = require('async_hooks');

// 创建全局存储实例
const requestContext = new AsyncLocalStorage();

/**
 * Express 中间件：为每个请求创建上下文
 */
function contextMiddleware(req, res, next) {
  const requestId = req.headers['x-request-id'] || generateRequestId();
  const context = {
    requestId,
    userId: req.user?.id,
    ip: req.ip,
    startTime: Date.now(),
  };

  // 在异步上下文中存储数据
  requestContext.run(context, () => {
    next();
  });
}

/**
 * 获取当前请求上下文
 */
function getContext() {
  return requestContext.getStore();
}

/**
 * 生成请求 ID
 */
function generateRequestId() {
  return `req_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
}

module.exports = { contextMiddleware, getContext };
```

#### 完整应用示例

```javascript
// app.js

const { contextMiddleware, getContext } = require('./context/request');
const logger = require('./utils/logger');

const app = express();

// 1. 应用上下文中间件（必须在最前面）

// 2. 在日志中自动包含请求 ID
app.use((req, res, next) => {
  const context = getContext();
  logger.info('收到请求', {
    requestId: context.requestId,
    method: req.method,
    url: req.url,
  });
  next();
});

// 3. 业务路由
app.get('/api/users/:id', async (req, res) => {
  const context = getContext();
  logger.info('开始查询用户', { requestId: context.requestId });

  // 即使在深层嵌套的异步函数中，也能获取上下文
  const user = await getUserById(req.params.id);
  
  res.json(user);
});

// 4. 深层嵌套函数
async function getUserById(userId) {
  const context = getContext();
  logger.info('查询数据库', { requestId: context.requestId, userId });

  // 查询 Redis 缓存

app.listen(3000);
```

### 2.2 async_hooks - 深度追踪

适用于需要完整追踪异步调用链的场景。

#### 基础示例

```javascript
// tracer/async.js
const async_hooks = require('async_hooks');
const fs = require('fs');

// 存储异步资源信息
const asyncMap = new Map();

// 创建钩子
const hook = async_hooks.createHook({
  /**
   * init: 异步资源创建时调用
   * @param {number} asyncId - 当前资源 ID
   * @param {string} type - 资源类型
   * @param {number} triggerAsyncId - 触发者 ID
   */
  init(asyncId, type, triggerAsyncId) {
    const info = {
      type,
      triggerAsyncId,
      stack: new Error().stack,
      timestamp: Date.now(),
    };
    asyncMap.set(asyncId, info);
  },

  /**
   * before: 异步回调执行前调用
   */
  before(asyncId) {
    const info = asyncMap.get(asyncId);
    if (info) {
      info.executionStart = Date.now();
    }
  },

  /**
   * after: 异步回调执行后调用
   */
  after(asyncId) {
    const info = asyncMap.get(asyncId);
    if (info && info.executionStart) {
      info.executionTime = Date.now() - info.executionStart;
    }
  },

  /**
   * destroy: 异步资源销毁时调用
   */
  destroy(asyncId) {
    asyncMap.delete(asyncId);
  },
});

// 启用钩子
hook.enable();

/**
 * 获取当前异步调用链
 */
function getAsyncCallChain() {
  const asyncId = async_hooks.executionAsyncId();
  const chain = [];
  let currentId = asyncId;

  while (currentId !== 0) {
    const info = asyncMap.get(currentId);
    if (!info) break;

    chain.push({
      asyncId: currentId,
      type: info.type,
      executionTime: info.executionTime,
    });

    currentId = info.triggerAsyncId;
  }

  return chain.reverse();
}

module.exports = { getAsyncCallChain };
```

#### 使用示例

```javascript
// app.js
const { getAsyncCallChain } = require('./tracer/async');

async function handleRequest() {
  console.log('开始处理请求');
  
  // 模拟异步操作
  await new Promise(resolve => setTimeout(resolve, 100));
  
  // 获取调用链
  const chain = getAsyncCallChain();
  console.log('异步调用链:', JSON.stringify(chain, null, 2));
}

handleRequest();

// 输出示例：
// [
//   { "asyncId": 1, "type": "TIMERWRAP", "executionTime": 102 },
//   { "asyncId": 2, "type": "PROMISE", "executionTime": 1 },
//   { "asyncId": 3, "type": "PROMISE", "executionTime": 0 }
// ]
```

### 2.3 实战：分布式追踪系统

结合 AsyncLocalStorage 实现完整的分布式追踪。

#### Step 1：Trace Context 管理

```javascript
// tracer/context.js
const { AsyncLocalStorage } = require('async_hooks');
const { v4: uuidv4 } = require('uuid');

const traceContext = new AsyncLocalStorage();

class TraceContext {
  constructor(options = {}) {
    this.traceId = options.traceId || uuidv4();
    this.spanId = options.spanId || uuidv4();
    this.parentSpanId = options.parentSpanId || null;
    this.serviceName = options.serviceName || 'unknown';
    this.spans = [];
  }

  /**
   * 创建子 Span
   */
  createChildSpan(name, metadata = {}) {
    const span = {
      spanId: uuidv4(),
      parentSpanId: this.spanId,
      name,
      startTime: Date.now(),
      metadata,
    };

    this.spans.push(span);
    return span;
  }

  /**
   * 结束 Span
   */
  endSpan(span) {
    span.endTime = Date.now();
    span.duration = span.endTime - span.startTime;
  }

  /**
   * 获取当前上下文
   */
  static getCurrent() {
    return traceContext.getStore();
  }

  /**
   * 设置上下文
   */
  static run(context, fn) {
    return traceContext.run(context, fn);
  }
}

module.exports = TraceContext;
```

#### Step 2：Express 中间件

```javascript
// middleware/tracing.js
const TraceContext = require('../tracer/context');

function tracingMiddleware(req, res, next) {
  // 从请求头提取 Trace ID（分布式追踪）
  const traceId = req.headers['x-trace-id'];
  const parentSpanId = req.headers['x-parent-span-id'];

  const context = new TraceContext({
    traceId,
    parentSpanId,
    serviceName: 'api-server',
  });

  // 创建 HTTP 请求 Span
  const httpSpan = context.createChildSpan('http.request', {
    method: req.method,
    url: req.url,
    userAgent: req.headers['user-agent'],
  });

  // 响应完成时结束 Span
  res.on('finish', () => {
    context.endSpan(httpSpan);
    
    // 上报到 APM 平台
    reportTrace(context);
  });

  // 将上下文传递给后续中间件
  TraceContext.run(context, () => {
    // 设置响应头（用于下游服务）
    res.setHeader('X-Trace-Id', context.traceId);
    res.setHeader('X-Span-Id', context.spanId);
    
    next();
  });
}

/**
 * 上报追踪数据到 APM 平台
 */
function reportTrace(context) {
  console.log('Trace Report:', {
    traceId: context.traceId,
    serviceName: context.serviceName,
    spans: context.spans,
  });
  
  // 实际项目中上报到 Jaeger, Zipkin, Datadog 等
  // jaeger.report(context);
}

module.exports = tracingMiddleware;
```

#### Step 3：数据库查询追踪

```javascript
// db/traced-query.js
const TraceContext = require('../tracer/context');
const mongoose = require('mongoose');

/**
 * 包装 Mongoose 查询，添加追踪
 */
function tracedQuery(model, method, ...args) {
  const context = TraceContext.getCurrent();
  
  if (!context) {
    // 无追踪上下文，直接执行
    return model[method](...args);
  }

  // 创建数据库 Span
  const dbSpan = context.createChildSpan('db.query', {
    collection: model.collection.name,
    method,
    query: JSON.stringify(args[0]),
  });

  const promise = model[method](...args);

  // 查询完成后结束 Span
  promise
    .then((result) => {
      dbSpan.metadata.resultCount = Array.isArray(result) ? result.length : 1;
      context.endSpan(dbSpan);
      return result;
    })
    .catch((error) => {
      dbSpan.metadata.error = error.message;
      context.endSpan(dbSpan);
      throw error;
    });

  return promise;
}

// 使用示例
const User = require('./models/User');

// 普通查询（无追踪）
const users = await User.find({ status: 'active' });

// 追踪查询
const tracedUsers = await tracedQuery(User, 'find', { status: 'active' });
```

#### Step 4：HTTP 客户端追踪

```javascript
// http/traced-client.js
const TraceContext = require('../tracer/context');
const axios = require('axios');

/**
 * 包装 Axios 请求，添加追踪
 */
function createTracedClient() {
  const client = axios.create();

  // 请求拦截器：注入 Trace 头
  client.interceptors.request.use((config) => {
    const context = TraceContext.getCurrent();
    
    if (context) {
      config.headers['X-Trace-Id'] = context.traceId;
      config.headers['X-Parent-Span-Id'] = context.spanId;
      
      // 创建 HTTP 客户端 Span
      config.metadata = {
        span: context.createChildSpan('http.client', {
          method: config.method,
          url: config.url,
        }),
      };
    }
    
    return config;
  });

  // 响应拦截器：结束 Span
  client.interceptors.response.use(
    (response) => {
      const context = TraceContext.getCurrent();
      if (context && response.config.metadata?.span) {
        const span = response.config.metadata.span;
        span.metadata.statusCode = response.status;
        context.endSpan(span);
      }
      return response;
    },
    (error) => {
      const context = TraceContext.getCurrent();
      if (context && error.config?.metadata?.span) {
        const span = error.config.metadata.span;
        span.metadata.error = error.message;
        context.endSpan(span);
      }
      throw error;
    }
  );

  return client;
}

module.exports = createTracedClient;
```

## 🐛 三、3个高频错误

### 错误1：在 async_hooks 回调中使用异步操作

#### 问题代码

```javascript
const async_hooks = require('async_hooks');
const fs = require('fs');

const hook = async_hooks.createHook({
  init(asyncId, type) {
    // ❌ 错误：在钩子中使用异步 I/O
    fs.appendFile('log.txt', `${asyncId} ${type}\n`, (err) => {
      if (err) console.error(err);
    });
  },
});

hook.enable();
```

**错误原因**：在钩子回调中使用异步操作会创建新的异步资源，导致无限递归。

**正确写法**：

```javascript
// ✅ 使用同步 I/O
const hook = async_hooks.createHook({
  init(asyncId, type) {
    fs.appendFileSync('log.txt', `${asyncId} ${type}\n`);
  },
});

// ✅ 或使用 console（Node.js 内部优化，不会触发钩子）
const hook2 = async_hooks.createHook({
  init(asyncId, type) {
    console.log(`${asyncId} ${type}`);
  },
});
```

### 错误2：忘记清理 AsyncLocalStorage

#### 问题代码

```javascript
const { AsyncLocalStorage } = require('async_hooks');
const storage = new AsyncLocalStorage();

app.use((req, res, next) => {
  const context = { requestId: generateId() };

  res.setHeader('Content-Type', 'text/event-stream');
```

**正确写法**：

```javascript
// ✅ 在连接关闭时清理
app.get('/events', (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  
  const timer = setInterval(() => {
    res.write(`data: ${Date.now()}\n\n`);
  }, 1000);
  
  // 监听连接关闭
  req.on('close', () => {
    clearInterval(timer);
    console.log('客户端断开连接，上下文已释放');
  });
```

```javascript
app.get('/api/users/:id', async (req, res) => {
```

## 🎯 四、2个应用场景

### 场景1：微服务链路追踪

#### 需求背景

- 电商系统包含 5 个微服务：API Gateway、User、Order、Payment、Notification
- 一个用户下单请求需要调用所有服务
- 需要追踪完整的调用链路，定位慢查询

#### 解决方案

**架构图**：

```
Client → API Gateway → User Service → Order Service → Payment Service
                                    ↓
                              Notification Service

所有服务共享同一个 Trace ID
```

**API Gateway 实现**：

```javascript
// api-gateway/app.js
const express = require('express');
const TraceContext = require('./tracer/context');
const tracedAxios = require('./http/traced-client');

const app = express();

app.use((req, res, next) => {
  // 创建 Trace 上下文
  const context = new TraceContext({
    serviceName: 'api-gateway',
  });

  TraceContext.run(context, () => {
    res.setHeader('X-Trace-Id', context.traceId);
    next();
  });
});

app.post('/api/orders', async (req, res) => {
  try {
    // 1. 验证用户（调用 User Service）
    const user = await tracedAxios.post('http://user-service/verify', {
      token: req.headers.authorization,
    });

    // 2. 创建订单（调用 Order Service）
    const order = await tracedAxios.post('http://order-service/orders', {
      userId: user.data.id,
      items: req.body.items,
    });

    // 3. 处理支付（调用 Payment Service）
    const payment = await tracedAxios.post('http://payment-service/charge', {
      orderId: order.data.id,
      amount: order.data.total,
    });

    res.json({ success: true, order: order.data });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

#### 需求背景

#### 解决方案

**Step 1：创建日志中间件**

## 🔬 五、深度解析

### 5.1 AsyncLocalStorage 实现原理

```javascript
// Node.js 内部简化实现
class AsyncLocalStorage {
  constructor() {
    this.kResourceStore = Symbol('kResourceStore');
  }

  run(store, callback) {
    // 获取当前异步资源
    const resource = asyncResource();
    
    // 将 store 绑定到资源
    resource[this.kResourceStore] = store;
    
    // 执行回调
    return callback();
  }

  getStore() {
    const resource = asyncResource();
    return resource[this.kResourceStore];
  }
}

// 异步资源会自动传递给子异步操作
async function example() {
  const storage = new AsyncLocalStorage();
  
  storage.run({ id: 1 }, async () => {
    console.log(storage.getStore());  // { id: 1 }
    
    await new Promise(resolve => setTimeout(resolve, 100));
    
    console.log(storage.getStore());  // 仍然是 { id: 1 } ✅
  });
}
```

## 📝 六、总结

### 核心要点

✅ **AsyncLocalStorage** 是请求上下文传递的最佳方案，性能开销约 10%  
✅ **async_hooks** 提供深度追踪能力，但复杂度高，开销约 20%  
✅ **分布式追踪** 需要在所有服务间传递 Trace ID 和 Span ID  
✅ **请求日志关联** 通过 Request ID 实现，极大提升问题排查效率  
✅ **避免在钩子回调中使用异步操作**，会导致无限递归  
✅ **及时清理长连接的上下文**，避免内存泄漏

### 最佳实践

1. **优先使用 AsyncLocalStorage**：简单、高效、安全
2. **统一 Trace ID 格式**：UUID 或 Snowflake ID
3. **在所有 HTTP 客户端注入 Trace 头**：确保跨服务追踪
4. **日志中自动包含 Request ID**：方便问题排查
5. **集成 APM 平台**：Jaeger, Zipkin, Datadog
6. **性能监控**：定期评估 async_hooks 的开销

### Q3: 如何与 Prometheus 集成？

**A**: 使用 Histogram 记录 Span 延迟。

```javascript
const client = require('prom-client');

const spanDuration = new client.Histogram({
  name: 'trace_span_duration_seconds',
  help: 'Span duration in seconds',
  labelNames: ['service', 'span_name'],
  buckets: [0.001, 0.01, 0.1, 0.5, 1, 5],
});

// 在 endSpan 时记录
function endSpan(span) {
  span.endTime = Date.now();
  span.duration = span.endTime - span.startTime;
  
  spanDuration.observe(
    { service: context.serviceName, span_name: span.name },
    span.duration / 1000
  );
}
```

### Q4: 如何处理跨进程追踪（PM2 Cluster）？

**A**: 使用 Redis 存储 Trace 数据。

```javascript
const Redis = require('ioredis');
const redis = new Redis();

class DistributedTraceContext extends TraceContext {
  async saveToRedis() {
    await redis.setex(
      `trace:${this.traceId}`,
      3600,  // 1 小时过期
      JSON.stringify(this.spans)
    );
  }

  static async loadFromRedis(traceId) {
    const data = await redis.get(`trace:${traceId}`);
    return data ? JSON.parse(data) : null;
  }
}
```
