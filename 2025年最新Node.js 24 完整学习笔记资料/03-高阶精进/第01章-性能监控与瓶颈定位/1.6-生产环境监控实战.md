## 📖 一、理论知识

### 1.1 黄金指标（Golden Signals）

Google SRE 提出的四大核心监控指标：

```
┌────────────────────────────────────────┐
│          黄金指标体系                  │
├────────────────────────────────────────┤
│ 1. 延迟 (Latency)                      │
│    → P50/P95/P99 响应时间              │
│                                        │
│ 2. 流量 (Traffic)                      │
│    → QPS, 请求数, 带宽                 │
│                                        │
│ 3. 错误 (Errors)                       │
│    → 错误率, 5xx 响应                  │
│                                        │
│ 4. 饱和度 (Saturation)                 │
│    → CPU, 内存, 磁盘, 网络使用率        │
└────────────────────────────────────────┘
```

#### 指标优先级

| 优先级 | 指标 | 告警阈值 | 影响 |
|-------|------|---------|------|
| **P0** | 错误率 | > 1% | 用户无法使用 |
| **P0** | P99 延迟 | > 1s | 用户体验极差 |
| **P1** | QPS 下降 | < 50% 正常值 | 流量异常 |
| **P1** | CPU/内存 | > 80% | 服务不稳定 |
| **P2** | P95 延迟 | > 500ms | 部分用户慢 |

### 2.4 事件循环监控

```javascript
// src/monitor/eventLoop.js
const { monitorEventLoopDelay } = require('perf_hooks');
const { eventLoopLag } = require('./metrics');

class EventLoopMonitor {
  constructor(options = {}) {
    this.interval = options.interval || 5000;
    this.warnThreshold = options.warnThreshold || 100;  // 100ms
    this.monitor = monitorEventLoopDelay({ resolution: 10 });
  }

  start() {
    this.monitor.enable();

    this.timer = setInterval(() => {
      const p99 = this.monitor.percentile(99) / 1e6;  // 转换为毫秒
      const mean = this.monitor.mean / 1e6;

      // 更新 Prometheus 指标
      eventLoopLag.set(mean / 1000);  // 转换为秒

      // 日志记录
      console.log({
        timestamp: new Date().toISOString(),
        eventLoopLag: {
          mean: mean.toFixed(2) + 'ms',
          p99: p99.toFixed(2) + 'ms',
        },
      });

      // 告警
      if (p99 > this.warnThreshold) {
        console.warn(`⚠️ 事件循环延迟超过阈值！P99=${p99.toFixed(2)}ms`);
      }

      // 重置统计
      this.monitor.reset();
    }, this.interval);
  }

  stop() {
    clearInterval(this.timer);
    this.monitor.disable();
  }
}

module.exports = EventLoopMonitor;
```

### 2.5 内存监控

```javascript
// src/monitor/memory.js
const { heapUsageRatio } = require('./metrics');

class MemoryMonitor {
  constructor(options = {}) {
    this.interval = options.interval || 10000;
    this.warnThreshold = options.warnThreshold || 0.8;  // 80%
  }

  start() {
    this.timer = setInterval(() => {
      const mem = process.memoryUsage();
      const heapRatio = mem.heapUsed / mem.heapTotal;

      // 更新 Prometheus 指标
      heapUsageRatio.set(heapRatio);

      // 日志记录
      console.log({
        timestamp: new Date().toISOString(),
        memory: {
          heapUsed: (mem.heapUsed / 1024 / 1024).toFixed(2) + ' MB',
          heapTotal: (mem.heapTotal / 1024 / 1024).toFixed(2) + ' MB',
          heapRatio: (heapRatio * 100).toFixed(2) + '%',
          rss: (mem.rss / 1024 / 1024).toFixed(2) + ' MB',
        },
      });

      // 告警
      if (heapRatio > this.warnThreshold) {
        console.warn(`⚠️ 堆内存使用率超过 ${this.warnThreshold * 100}%`);
      }
    }, this.interval);
  }

  stop() {
    clearInterval(this.timer);
  }
}

module.exports = MemoryMonitor;
```

### 2.6 分布式追踪

```javascript
// src/monitor/tracing.js
const { AsyncLocalStorage } = require('async_hooks');
const { v4: uuidv4 } = require('uuid');

const traceContext = new AsyncLocalStorage();

class TraceContext {
  constructor(options = {}) {
    this.traceId = options.traceId || uuidv4();
    this.spanId = options.spanId || uuidv4();
    this.parentSpanId = options.parentSpanId || null;
    this.startTime = Date.now();
    this.spans = [];
  }

  createSpan(name, metadata = {}) {
    const span = {
      spanId: uuidv4(),
      parentSpanId: this.spanId,
      name,
      startTime: Date.now(),
      metadata,
    };
    this.spans.push(span);
    return span;
  }

  endSpan(span) {
    span.endTime = Date.now();
    span.duration = span.endTime - span.startTime;
  }

  static getCurrent() {
    return traceContext.getStore();
  }

  static run(context, fn) {
    return traceContext.run(context, fn);
  }

  toJSON() {
    return {
      traceId: this.traceId,
      spanId: this.spanId,
      duration: Date.now() - this.startTime,
      spans: this.spans,
    };
  }
}

module.exports = TraceContext;
```

### 2.7 中间件集成

```javascript
// src/middleware/prometheus.js
const {
  httpRequestsTotal,
  httpRequestDuration,
  httpErrorsTotal,
  activeConnections,
} = require('../monitor/metrics');

function prometheusMiddleware(req, res, next) {
  const start = Date.now();

  // 连接计数
  activeConnections.inc();

  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    const route = req.route?.path || req.path;
    const labels = {
      method: req.method,
      route,
      status_code: res.statusCode,
    };

    // 记录请求
    httpRequestsTotal.inc(labels);
    httpRequestDuration.observe(labels, duration);

    // 记录错误
    if (res.statusCode >= 500) {
      httpErrorsTotal.inc({
        ...labels,
        error_type: res.locals.errorType || 'unknown',
      });
    }

    // 减少连接计数
    activeConnections.dec();
  });

  next();
}

module.exports = prometheusMiddleware;
```

```javascript
// src/middleware/tracing.js
const TraceContext = require('../monitor/tracing');

function tracingMiddleware(req, res, next) {
  // 从请求头提取 Trace ID
  const traceId = req.headers['x-trace-id'];
  const parentSpanId = req.headers['x-parent-span-id'];

  const context = new TraceContext({ traceId, parentSpanId });

  // 创建 HTTP Span
  const httpSpan = context.createSpan('http.request', {
    method: req.method,
    url: req.url,
    userAgent: req.headers['user-agent'],
  });

  res.on('finish', () => {
    context.endSpan(httpSpan);
    
    // 设置响应头
    res.setHeader('X-Trace-Id', context.traceId);
    
    // 记录追踪数据
    if (process.env.DEBUG_TRACING) {
      console.log('Trace:', context.toJSON());
    }
  });

  TraceContext.run(context, () => next());
}

module.exports = tracingMiddleware;
```

```javascript
  res.status(healthy ? 200 : 503).json({
    healthy,
    uptime: process.uptime(),
    memory: {
      heapUsed: (mem.heapUsed / 1024 / 1024).toFixed(2) + ' MB',
      heapTotal: (mem.heapTotal / 1024 / 1024).toFixed(2) + ' MB',
      heapRatio: (heapRatio * 100).toFixed(2) + '%',
    },
  });

  // 模拟业务逻辑
  res.json({ users: [] });

  console.log(`指标端点: http://localhost:${PORT}/metrics`);
  console.log(`健康检查: http://localhost:${PORT}/health`);
```

### 2.10 Prometheus 告警规则

```yaml
# prometheus/alerts.yml
groups:
  - name: production_alerts
    interval: 30s
    rules:
      # 告警1：P99 延迟超过 1 秒
      - alert: HighLatency
        expr: |
          histogram_quantile(0.99, 
            rate(http_request_duration_seconds_bucket[5m])
          ) > 1
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "P99 延迟超过 1 秒"
          description: "当前 P99: {{ $value }}s"

      # 告警2：错误率超过 1%
      - alert: HighErrorRate
        expr: |
          sum(rate(http_errors_total[1m])) 
          / 
          sum(rate(http_requests_total[1m])) 
          > 0.01
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "错误率超过 1%"
          description: "当前错误率: {{ $value | humanizePercentage }}"

      # 告警3：QPS 下降超过 50%
      - alert: QPSDrop
        expr: |
          sum(rate(http_requests_total[1m])) 
          < 
          0.5 * sum(rate(http_requests_total[5m] offset 10m))
        for: 3m
        labels:
          severity: warning
        annotations:
          summary: "QPS 下降超过 50%"

      # 告警4：内存使用率超过 85%
      - alert: HighMemoryUsage
        expr: heap_usage_ratio > 0.85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "堆内存使用率超过 85%"

      # 告警5：事件循环延迟超过 100ms
      - alert: HighEventLoopLag
        expr: event_loop_lag_seconds > 0.1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "事件循环延迟超过 100ms"

      # 告警6：数据库查询慢
      - alert: SlowDatabaseQuery
        expr: |
          histogram_quantile(0.95, 
            rate(db_query_duration_seconds_bucket[5m])
          ) > 0.5
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "数据库 P95 延迟超过 500ms"
```

## 📝 四、总结

### 核心要点

✅ **黄金指标** 覆盖延迟、流量、错误、饱和度四大维度  
✅ **事件循环监控** 实时发现性能瓶颈  
✅ **内存监控** 防止 OOM 崩溃  
✅ **分布式追踪** 定位跨服务调用问题  
✅ **自动告警** P99 延迟、错误率、QPS 异常  
✅ **Grafana 可视化** 实时监控和历史趋势分析

### 最佳实践

1. **监控全覆盖**：应用层 + 系统层 + 业务层
2. **告警分级**：P0/P1/P2，区分严重程度
3. **可观测性三支柱**：Metrics + Logs + Traces
4. **定期复盘**：每周/月分析监控数据，持续优化
5. **压测验证**：上线前进行压测，确保监控有效

## ❓ 五、常见问题

### Q1: 如何降低监控开销？

**A**: 控制采集频率和指标数量。

```yaml
# Prometheus 配置
scrape_interval: 30s  # 从 15s 调整为 30s

# 减少标签基数
labelNames: ['method', 'route']  # 不要加 user_id, ip
```

### Q2: 告警风暴如何处理？

**A**: 使用 AlertManager 分组和抑制规则。

```yaml
# alertmanager.yml
route:
  group_by: ['alertname', 'instance']
  group_wait: 10s
  group_interval: 10s
  repeat_interval: 1h  # 1 小时内同一告警只发送一次

inhibit_rules:
  - source_match:
      severity: 'critical'
    target_match:
      severity: 'warning'
    equal: ['instance']  # critical 告警会抑制 warning
```

### Q3: 如何集成到 K8s？

**A**: 使用 ServiceMonitor 和 PodMonitor。

```yaml
# servicemonitor.yaml
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: nodejs-app
spec:
  selector:
    matchLabels:
      app: nodejs-app
  endpoints:
  - port: metrics
    interval: 30s
    path: /metrics
```
