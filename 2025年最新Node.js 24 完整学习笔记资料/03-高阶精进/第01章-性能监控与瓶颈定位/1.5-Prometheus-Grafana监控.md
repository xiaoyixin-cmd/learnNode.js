## 📖 一、理论知识

### 1.1 Prometheus 架构

```
┌─────────────────────────────────────────────┐
│         Prometheus 监控架构                 │
├─────────────────────────────────────────────┤
│                                             │
│  Node.js App (prom-client)                  │
│       ↓ /metrics 端点                       │
│  Prometheus Server (采集 + 存储)            │
│       ↓ PromQL 查询                         │
│  Grafana (可视化 + 告警)                     │
│       ↓ AlertManager (告警通知)              │
│  钉钉/邮件/Slack                             │
│                                             │
└─────────────────────────────────────────────┘
```

### 1.2 四种指标类型

| 类型 | 说明 | 适用场景 | 示例 |
|-----|------|---------|------|
| **Counter** | 只增不减的计数器 | 请求总数、错误数 | `http_requests_total` |
| **Gauge** | 可增可减的仪表盘 | 当前内存、并发数 | `memory_usage_bytes` |
| **Histogram** | 分桶统计 | 响应时间分布 | `http_request_duration_seconds` |
| **Summary** | 分位数统计 | P50/P95/P99 延迟 | `http_request_duration_summary` |

#### Counter vs Gauge

```javascript
// Counter（累加）
const requestsTotal = new Counter({
  name: 'http_requests_total',
  help: 'Total HTTP requests',
});

requestsTotal.inc();  // 每次请求 +1
// 值: 1 → 2 → 3 → 4 → ...

// Gauge（当前值）
const currentUsers = new Gauge({
  name: 'current_users',
  help: 'Current online users',
});

currentUsers.inc();   // 用户上线 +1
currentUsers.dec();   // 用户下线 -1
// 值: 0 → 1 → 2 → 1 → 0
```

#### Histogram vs Summary

| 特性 | Histogram | Summary |
|-----|-----------|---------|
| **计算位置** | Prometheus 服务端 | 客户端（Node.js） |
| **精度** | 分桶近似 | 精确 |
| **性能** | 低开销 | 高开销 |
| **聚合** | ✅ 支持 | ❌ 不支持 |
| **推荐** | ✅ **优先使用** | ⚠️ 特殊场景 |

### 2.2 Node.js 集成 `prom-client`

#### 安装

```bash
npm install prom-client
```

#### 基础示例

```javascript
// monitor/metrics.js
const client = require('prom-client');

// 1. 创建注册表
const register = new client.Registry();

// 2. 启用默认指标（CPU, 内存, GC 等）
client.collectDefaultMetrics({ register });

// 3. 创建自定义指标
const httpRequestsTotal = new client.Counter({
  name: 'http_requests_total',
  help: 'Total number of HTTP requests',
  labelNames: ['method', 'route', 'status_code'],
  registers: [register],
});

const httpRequestDuration = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.01, 0.05, 0.1, 0.5, 1, 5],  // 10ms ~ 5s
  registers: [register],
});

const activeConnections = new client.Gauge({
  name: 'active_connections',
  help: 'Number of active connections',
  registers: [register],
});

module.exports = {
  register,
  httpRequestsTotal,
  httpRequestDuration,
  activeConnections,
};
```

#### Express 中间件

```javascript
// middleware/prometheus.js
const { httpRequestsTotal, httpRequestDuration } = require('../monitor/metrics');

function prometheusMiddleware(req, res, next) {
  const start = Date.now();

  // 响应完成时记录指标
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;  // 转换为秒
    const route = req.route?.path || req.path;
    const labels = {
      method: req.method,
      route,
      status_code: res.statusCode,
    };

    // 记录请求总数
    httpRequestsTotal.inc(labels);

    // 记录请求延迟
    httpRequestDuration.observe(labels, duration);
  });

  next();
}

module.exports = prometheusMiddleware;
```

#### 暴露 /metrics 端点

```javascript
// app.js
const express = require('express');
const prometheusMiddleware = require('./middleware/prometheus');
const { register } = require('./monitor/metrics');

const app = express();

// 应用 Prometheus 中间件
app.use(prometheusMiddleware);

// 暴露指标端点（供 Prometheus 采集）
app.get('/metrics', async (req, res) => {
  res.setHeader('Content-Type', register.contentType);
  res.send(await register.metrics());
});

// 业务接口
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
  console.log('Metrics available at http://localhost:3000/metrics');
});
```

### 2.3 自定义业务指标

#### 示例1：数据库查询监控

```javascript
// monitor/database.js
const client = require('prom-client');
const { register } = require('./metrics');

const dbQueryDuration = new client.Histogram({
  name: 'db_query_duration_seconds',
  help: 'Database query duration in seconds',
  labelNames: ['collection', 'operation'],
  buckets: [0.001, 0.01, 0.1, 0.5, 1],
  registers: [register],
});

const dbQueryErrors = new client.Counter({
  name: 'db_query_errors_total',
  help: 'Total database query errors',
  labelNames: ['collection', 'operation', 'error_type'],
  registers: [register],
});

/**
 * 包装 Mongoose 查询，添加监控
 */
function tracedQuery(model, operation, ...args) {
  const start = Date.now();
  const collection = model.collection.name;

  return model[operation](...args)
    .then((result) => {
      const duration = (Date.now() - start) / 1000;
      dbQueryDuration.observe({ collection, operation }, duration);
      return result;
    })
    .catch((error) => {
      const duration = (Date.now() - start) / 1000;
      dbQueryDuration.observe({ collection, operation }, duration);
      dbQueryErrors.inc({ collection, operation, error_type: error.name });
      throw error;
    });
}

module.exports = { tracedQuery };
```

#### 示例2：Redis 监控

```javascript
// monitor/redis.js
const client = require('prom-client');
const { register } = require('./metrics');

const redisCacheHits = new client.Counter({
  name: 'redis_cache_hits_total',
  help: 'Total Redis cache hits',
  labelNames: ['key_prefix'],
  registers: [register],
});

const redisCacheMisses = new client.Counter({
  name: 'redis_cache_misses_total',
  help: 'Total Redis cache misses',
  labelNames: ['key_prefix'],
  registers: [register],
});

const redisCommandDuration = new client.Histogram({
  name: 'redis_command_duration_seconds',
  help: 'Redis command duration in seconds',
  labelNames: ['command'],
  buckets: [0.001, 0.01, 0.1, 0.5],
  registers: [register],
});

/**
 * 包装 Redis 客户端
 */
function createTracedRedis(redis) {
  const handler = {
    get(target, prop) {
      const original = target[prop];
      if (typeof original !== 'function') return original;

      return async function (...args) {
        const start = Date.now();
        try {
          const result = await original.apply(target, args);
          const duration = (Date.now() - start) / 1000;
          
          redisCommandDuration.observe({ command: prop }, duration);
          
          // 记录缓存命中/未命中
          if (prop === 'get') {
            const keyPrefix = args[0].split(':')[0];
            if (result) {
              redisCacheHits.inc({ key_prefix: keyPrefix });
            } else {
              redisCacheMisses.inc({ key_prefix: keyPrefix });
            }
          }
          
          return result;
        } catch (error) {
          const duration = (Date.now() - start) / 1000;
          redisCommandDuration.observe({ command: prop }, duration);
          throw error;
        }
      };
    },
  };

  return new Proxy(redis, handler);
}

module.exports = { createTracedRedis };
```

## 🐛 三、3个高频错误

### 错误1：指标名称不符合规范

#### 问题代码

```javascript
// ❌ 错误：驼峰命名
const httpRequests = new client.Counter({
  name: 'httpRequestsTotal',  // 错误！
  help: 'Total HTTP requests',
});

// ❌ 错误：包含特殊字符
const dbQueries = new client.Counter({
  name: 'db-queries-total',  // 错误！
  help: 'Total database queries',
});
```

**Prometheus 命名规范**：

```
✅ 正确格式：[namespace]_[subsystem]_[name]_[unit]
- 小写字母
- 下划线分隔
- 以 _total / _seconds / _bytes 结尾（推荐）
```

#### 正确写法

```javascript
// ✅ 正确
const httpRequestsTotal = new client.Counter({
  name: 'http_requests_total',  // ✅
  help: 'Total number of HTTP requests',
});

const dbQueryDurationSeconds = new client.Histogram({
  name: 'db_query_duration_seconds',  // ✅
  help: 'Database query duration in seconds',
  buckets: [0.001, 0.01, 0.1, 1],
});

const memoryUsageBytes = new client.Gauge({
  name: 'memory_usage_bytes',  // ✅
  help: 'Memory usage in bytes',
});
```

### 错误2：Histogram 分桶设置不合理

#### 问题代码

```javascript
// ❌ 错误：分桶间隔太大
const httpDuration = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'HTTP request duration',
  buckets: [1, 10, 100],  // 1s, 10s, 100s
});

// 实际大部分请求在 50ms ~ 200ms
// → 所有请求都落在第一个桶 (< 1s)
// → 无法区分快慢请求
```

#### 正确写法

```javascript
// ✅ 根据实际情况设置分桶
const httpDuration = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'HTTP request duration',
  buckets: [0.01, 0.05, 0.1, 0.5, 1, 5],  // 10ms ~ 5s
});

// 或使用对数分桶
const buckets = client.exponentialBuckets(0.001, 2, 10);
// [0.001, 0.002, 0.004, 0.008, 0.016, ...]

// 或使用线性分桶
const linearBuckets = client.linearBuckets(0, 0.1, 10);
// [0, 0.1, 0.2, 0.3, 0.4, ...]
```

**设置指南**：

```bash
# 1. 先运行应用，观察实际延迟范围
# 2. 使用 autocannon 压测
autocannon -c 100 -d 10 http://localhost:3000/api/users

# 3. 查看延迟分布
# Latency (ms):
#   Avg: 50
#   P50: 45
#   P95: 120
#   P99: 200

# 4. 设置分桶覆盖主要范围
buckets: [0.01, 0.05, 0.1, 0.2, 0.5, 1]
#          ↑    ↑    ↑    ↑    ↑    ↑
#         10ms 50ms 100ms 200ms 500ms 1s
```

### 错误3：标签（Labels）滥用导致基数爆炸

#### 问题代码

```javascript
// ❌ 错误：使用高基数标签
const httpRequests = new client.Counter({
  name: 'http_requests_total',
  labelNames: ['method', 'route', 'user_id', 'ip'],  // ❌ user_id 和 ip 有无限可能！
});

httpRequests.inc({
  method: 'GET',
  route: '/api/users',
  user_id: 'user_12345',  // ❌ 100 万用户 = 100 万个时间序列
  ip: '192.168.1.100',    // ❌ 更多！
});

// 结果：Prometheus 内存爆炸，查询超时
```

**高基数标签危害**：

```
时间序列数量 = 标签组合数

例如：
- method: 5 种 (GET, POST, PUT, DELETE, PATCH)
- route: 100 个接口
- user_id: 100 万用户

→ 5 × 100 × 1,000,000 = 5 亿个时间序列！
→ Prometheus 内存占用: ~50GB+
→ 查询速度: 超时
```

#### 正确写法

```javascript
// ✅ 只使用低基数标签
const httpRequests = new client.Counter({
  name: 'http_requests_total',
  labelNames: ['method', 'route', 'status_code'],  // ✅ 总共约 5 × 100 × 5 = 2500 个序列
});

httpRequests.inc({
  method: 'GET',
  route: '/api/users',
  status_code: 200,
});

// 如果需要按用户统计，使用聚合查询
// 而不是为每个用户创建指标
```

**标签使用指南**：

| 标签类型 | 基数 | 推荐 | 示例 |
|---------|------|------|------|
| **方法** | < 10 | ✅ | GET, POST |
| **路由** | < 1000 | ✅ | /api/users |
| **状态码** | < 100 | ✅ | 200, 404, 500 |
| **用户 ID** | 无限 | ❌ | user_12345 |
| **IP 地址** | 无限 | ❌ | 192.168.1.1 |
| **请求 ID** | 无限 | ❌ | req_abc123 |

const client = require('prom-client');

**Step 2：业务代码集成**

```javascript
// controllers/order.js
const { ordersCreated, orderValue, paymentDuration } = require('../monitor/business');

async function createOrder(req, res) {
  const start = Date.now();

  try {
    const order = await Order.create({
      userId: req.user.id,
      items: req.body.items,
      total: req.body.total,
    });

    // 记录订单创建
    ordersCreated.inc({ payment_method: req.body.paymentMethod });
    orderValue.observe(req.body.total);

    // 处理支付
    await processPayment(order);

    const duration = (Date.now() - start) / 1000;
    paymentDuration.observe(duration);

    res.json({ success: true, order });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
}
```

**Step 4：告警规则**

```yaml
# prometheus/alerts.yml
groups:
  - name: promotion_alerts
    interval: 30s
    rules:
      # 告警1：QPS 下降 50%
      - alert: QPSDrop
        expr: |
          rate(http_requests_total[1m]) < 0.5 * rate(http_requests_total[5m] offset 10m)
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "QPS 下降超过 50%"
          description: "当前 QPS: {{ $value }}"

      # 告警2：P95 延迟超过 1 秒
      - alert: HighLatency
        expr: |
          histogram_quantile(0.95, rate(http_request_duration_seconds_bucket[5m])) > 1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "P95 延迟超过 1 秒"

      # 告警3：错误率超过 5%
      - alert: HighErrorRate
        expr: |
          sum(rate(http_requests_total{status_code=~"5.."}[1m])) 
          / 
          sum(rate(http_requests_total[1m])) 
          > 0.05
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "错误率超过 5%"

histogram_quantile(0.95,

)

# Panel 3: 内存使用对比
process_resident_memory_bytes by (instance) / 1024 / 1024

# Panel 4: CPU 使用率对比
rate(process_cpu_seconds_total[1m]) by (instance) * 100
```

## 🔬 五、深度解析

### 5.1 PromQL 查询语言

#### 基础查询

```promql
# 1. 即时向量（当前值）
http_requests_total

# 2. 范围向量（时间段）
http_requests_total[5m]

# 3. 标签过滤
http_requests_total{method="GET", status_code="200"}

# 4. 正则匹配
http_requests_total{status_code=~"5.."}  # 5xx 错误
```

#### 聚合函数

```promql
# 求和
sum(http_requests_total)

# 按标签分组
sum by (method) (http_requests_total)

# 平均值
avg(http_request_duration_seconds)

# 计数
count(http_requests_total)

# 最大/最小
max(memory_usage_bytes)
min(memory_usage_bytes)
```

#### 速率计算

```promql
# rate: 每秒增长率（适用于 Counter）
rate(http_requests_total[1m])

# irate: 瞬时增长率（更敏感）
irate(http_requests_total[1m])

# increase: 总增长量
increase(http_requests_total[1h])
```

#### 分位数计算

```promql
# P50 延迟
histogram_quantile(0.5, rate(http_request_duration_seconds_bucket[5m]))

# P95 延迟
histogram_quantile(0.95, rate(http_request_duration_seconds_bucket[5m]))

# P99 延迟
histogram_quantile(0.99, rate(http_request_duration_seconds_bucket[5m]))
```

## 📝 六、总结

### 核心要点

✅ **Prometheus + Grafana** 是最流行的开源监控方案  
✅ **四种指标类型**：Counter（累计）、Gauge（当前值）、Histogram（分布）、Summary（分位数）  
✅ **优先使用 Histogram** 而非 Summary（支持聚合）  
✅ **指标命名规范**：小写字母 + 下划线，以单位结尾  
✅ **避免高基数标签**：不使用用户 ID、IP 等作为标签  
✅ **合理设置分桶**：根据实际延迟范围调整 Histogram buckets  
✅ **告警规则**：监控 QPS、延迟、错误率，及时发现问题

### 最佳实践

1. **默认指标 + 自定义指标**：`collectDefaultMetrics()` + 业务指标
2. **标签使用原则**：低基数、有意义、可聚合
3. **分桶设置**：覆盖大部分请求的延迟范围
4. **告警设置**：P95 延迟、错误率、QPS 异常
5. **仪表板分层**：总览 → 服务详情 → 实例详情

## ❓ 七、常见问题

### Q1: Histogram 和 Summary 如何选择？

**A**: 优先使用 Histogram。

| 场景 | 推荐 | 原因 |
|-----|------|------|
| **HTTP 延迟** | Histogram | 支持聚合，服务端计算 |
| **数据库延迟** | Histogram | 可跨实例聚合 |
| **自定义分位数** | Summary | 客户端精确计算 |

### Q2: 如何减少 Prometheus 内存占用？

**A**: 控制时间序列数量。

```yaml
# 1. 限制标签基数
labelNames: ['method', 'route']  # ✅ 低基数

# 2. 缩短数据保留时间
prometheus --storage.tsdb.retention.time=15d  # 默认 15 天

# 3. 使用 Federation（联邦集群）
# 将历史数据迁移到长期存储
```

### Q3: 如何实现告警通知（钉钉/邮件）？

**A**: 使用 AlertManager。

```yaml
# alertmanager.yml
route:
  receiver: 'dingtalk'
  group_by: ['alertname']
  group_wait: 10s
  group_interval: 10s
  repeat_interval: 1h

receivers:
  - name: 'dingtalk'
    webhook_configs:
      - url: 'https://oapi.dingtalk.com/robot/send?access_token=YOUR_TOKEN'
```

### Q4: 如何监控异步任务（定时任务）？

**A**: 记录任务执行次数和延迟。

```javascript
const taskExecutions = new client.Counter({
  name: 'task_executions_total',
  labelNames: ['task_name', 'status'],
});

const taskDuration = new client.Histogram({
  name: 'task_duration_seconds',
  labelNames: ['task_name'],
  buckets: [1, 10, 30, 60, 300],
});

cron.schedule('*/5 * * * *', async () => {
  const start = Date.now();
  try {
    await syncData();
    taskExecutions.inc({ task_name: 'sync_data', status: 'success' });
  } catch (error) {
    taskExecutions.inc({ task_name: 'sync_data', status: 'failure' });
  } finally {
    taskDuration.observe({ task_name: 'sync_data' }, (Date.now() - start) / 1000);
  }
});
```

### Q5: 如何集成到 K8s？

**A**: 使用 ServiceMonitor（Prometheus Operator）。

```yaml
# servicemonitor.yaml
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: nodejs-app
spec:
  selector:
    matchLabels:
      app: nodejs-app
  endpoints:
  - port: metrics
    interval: 30s
```
