### 2. 查询区块链数据

```javascript
// 获取区块高度
const blockNumber = await provider.getBlockNumber();
console.log('当前区块:', blockNumber);  // 18500000

// 获取区块详情
const block = await provider.getBlock(blockNumber);
console.log('区块哈希:', block.hash);
console.log('矿工:', block.miner);

// 获取账户余额
const balance = await provider.getBalance('0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb');
console.log('余额:', ethers.formatEther(balance), 'ETH');
// "1234.567" ETH

// 获取交易详情
const tx = await provider.getTransaction(
  '0x1234...abcd'
);
console.log('发送方:', tx.from);
console.log('接收方:', tx.to);
console.log('金额:', ethers.formatEther(tx.value), 'ETH');

// 获取 Gas 价格
const gasPrice = await provider.getFeeData();

```

### 4. 发送交易

```javascript
// 发送 ETH
async function sendETH(toAddress, amount) {
  const tx = await wallet.sendTransaction({
    to: toAddress,
    value: ethers.parseEther(amount),  // "1.5" → 1500000000000000000 wei
  });
  
  console.log('交易哈希:', tx.hash);
  
  // 等待确认（1个区块）
  const receipt = await tx.wait();
  console.log('交易成功，区块:', receipt.blockNumber);
  console.log('Gas 消耗:', receipt.gasUsed.toString());
  
  return receipt;
}

await sendETH('0x742d35Cc6634C0532925a3b844Bc9e7595f0bEb', '0.1');
```

## 🎯 智能合约交互

### 1. 读取合约（无需 Gas）

```javascript
// ERC20 合约 ABI
const ERC20_ABI = [
  'function name() view returns (string)',
  'function symbol() view returns (string)',
  'function totalSupply() view returns (uint256)',
  'function balanceOf(address) view returns (uint256)',
];

// 连接合约（USDT）
const usdtAddress = '0xdac17f958d2ee523a2206206994597c13d831ec7';
const usdtContract = new ethers.Contract(usdtAddress, ERC20_ABI, provider);

// 读取数据
const name = await usdtContract.name();
console.log('代币名称:', name);  // "Tether USD"

const symbol = await usdtContract.symbol();
console.log('代币符号:', symbol);  // "USDT"

const totalSupply = await usdtContract.totalSupply();
console.log('总供应量:', ethers.formatUnits(totalSupply, 6));  // USDT 6位小数

const balance = await usdtContract.balanceOf('0x742d35Cc...');
console.log('余额:', ethers.formatUnits(balance, 6), 'USDT');
```

---

### 2. 写入合约（需要 Gas）

```javascript
// 完整 ERC20 ABI
const ERC20_ABI_FULL = [
  ...ERC20_ABI,
  'function transfer(address to, uint256 amount) returns (bool)',
  'function approve(address spender, uint256 amount) returns (bool)',
];

// 连接钱包到合约
const usdtWithSigner = new ethers.Contract(usdtAddress, ERC20_ABI_FULL, wallet);

// 转账 USDT
async function transferUSDT(toAddress, amount) {
  const tx = await usdtWithSigner.transfer(
    toAddress,
    ethers.parseUnits(amount, 6)  // 6位小数
  );
  
  console.log('交易哈希:', tx.hash);
  const receipt = await tx.wait();
  console.log('转账成功！');
  
  return receipt;
}

await transferUSDT('0x742d35Cc...', '100');  // 转100 USDT
```

### 3. 监听合约事件

```javascript
// Transfer 事件
const transferEvent = 'event Transfer(address indexed from, address indexed to, uint256 value)';

const usdtContract = new ethers.Contract(
  usdtAddress,
  [transferEvent, ...ERC20_ABI],
  provider
);

// 实时监听
usdtContract.on('Transfer', (from, to, value, event) => {
  console.log(`转账事件: ${from} → ${to}`);
  console.log(`金额: ${ethers.formatUnits(value, 6)} USDT`);
  console.log(`区块: ${event.log.blockNumber}`);
});

// 监听特定地址的转账
const myAddress = '0x742d35Cc...';
const filter = usdtContract.filters.Transfer(null, myAddress);  // 接收方是我

usdtContract.on(filter, (from, to, value) => {
  console.log(`收到 ${ethers.formatUnits(value, 6)} USDT，来自 ${from}`);
});

// 查询历史事件
const events = await usdtContract.queryFilter(
  filter,
  18000000,  // 起始区块
  18100000   // 结束区块
);

events.forEach((event) => {
  console.log('历史转账:', event.args);
});
```

## 🎯 NFT 操作（ERC721）

```javascript
const ERC721_ABI = [

  'function balanceOf(address owner) view returns (uint256)',

// CryptoPunks 合约
const cryptoPunksAddress = '0xb47e3cd837dDF8e4c57F05d70Ab865de6e193BBB';

// 查询元数据 URI

const metadataResponse = await fetch(tokenURI.replace('ipfs://', 'https://ipfs.io/ipfs/'));
```

### 2. 铸造 NFT

```javascript
const ERC721_MINTABLE_ABI = [
  ...ERC721_ABI,
  'function mint(address to, uint256 tokenId) returns (bool)',
  'function safeMint(address to, string memory uri) returns (uint256)',
];

const myNFTContract = new ethers.Contract(
  '0xYOUR_NFT_CONTRACT',
  ERC721_MINTABLE_ABI,
  wallet
);

// 铸造 NFT
async function mintNFT(toAddress, tokenId, metadataURI) {
  const tx = await myNFTContract.safeMint(toAddress, metadataURI);
  const receipt = await tx.wait();
  
  console.log('NFT 铸造成功！');
  console.log('Token ID:', tokenId);
  console.log('元数据:', metadataURI);
  
  return receipt;
}

await mintNFT(
  wallet.address,
  1,
  'ipfs://QmYwAPJzv5CZsnA625s3Xf2nemtYgPpHdWEz79ojWnPbdG/1'
);
```

## 🎯 DeFi 交互（Uniswap）

### 1. 查询交易对价格

```javascript
import { ChainId, Token, Fetcher, Route } from '@uniswap/sdk';

// 定义代币
const DAI = new Token(ChainId.MAINNET, '0x6B175474E89094C44Da98b954EedeAC495271d0F', 18);
const USDC = new Token(ChainId.MAINNET, '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48', 6);

// 获取交易对
const pair = await Fetcher.fetchPairData(DAI, USDC, provider);

// 计算价格
const route = new Route([pair], DAI);
console.log('1 DAI =', route.midPrice.toSignificant(6), 'USDC');
// "0.999850 USDC"

console.log('1 USDC =', route.midPrice.invert().toSignificant(6), 'DAI');
// "1.000150 DAI"
```

### 2. Swap 交易

```javascript
import { Trade, TokenAmount, TradeType } from '@uniswap/sdk';
import { ethers } from 'ethers';

// Uniswap Router 合约
const UNISWAP_ROUTER = '0x7a250d5630B4cF539739dF2C5dAcb4c659F2488D';
const ROUTER_ABI = [
  'function swapExactTokensForTokens(uint amountIn, uint amountOutMin, address[] calldata path, address to, uint deadline) returns (uint[] memory amounts)',
];

const routerContract = new ethers.Contract(UNISWAP_ROUTER, ROUTER_ABI, wallet);

// 执行 Swap（DAI → USDC）
async function swapTokens(amountIn) {
  // 1. 授权 Router 使用 DAI
  const daiContract = new ethers.Contract(DAI.address, ERC20_ABI_FULL, wallet);
  const approveTx = await daiContract.approve(
    UNISWAP_ROUTER,
    ethers.parseUnits(amountIn, 18)
  );
  await approveTx.wait();
  
  // 2. 执行交易
  const path = [DAI.address, USDC.address];
  const deadline = Math.floor(Date.now() / 1000) + 60 * 20;  // 20分钟
  
  const tx = await routerContract.swapExactTokensForTokens(
    ethers.parseUnits(amountIn, 18),  // 输入金额
    0,  // 最小输出（实际应计算滑点）
    path,
    wallet.address,
    deadline
  );
  
  const receipt = await tx.wait();
  console.log('Swap 成功！');
  
  return receipt;
}

await swapTokens('100');  // Swap 100 DAI
```

## ⚠️ 三个高频错误

### 错误 1：私钥泄露

```javascript
// ❌ 错误：硬编码私钥
const wallet = new ethers.Wallet('0x1234abcd...');

// ✅ 正确：环境变量
const wallet = new ethers.Wallet(process.env.PRIVATE_KEY);

// ✅ 更安全：加密存储
const encryptedWallet = readFileSync('./wallet.json', 'utf8');
const wallet = await ethers.Wallet.fromEncryptedJson(
  encryptedWallet,
  process.env.WALLET_PASSWORD
);
```

---

### 错误 2：忘记处理 Gas 估算

```javascript
// ❌ 错误：没有 Gas Limit
const tx = await contract.transfer(to, amount);  // 可能失败

// ✅ 正确：估算 Gas
const gasLimit = await contract.transfer.estimateGas(to, amount);
const tx = await contract.transfer(to, amount, { gasLimit });

// ✅ 动态 Gas Price
const feeData = await provider.getFeeData();
const tx = await contract.transfer(to, amount, {
  gasLimit,
  maxFeePerGas: feeData.maxFeePerGas,
  maxPriorityFeePerGas: feeData.maxPriorityFeePerGas,
});
```

---

### 错误 3：单位转换错误

```javascript
// ❌ 错误：忘记转换单位
const balance = await provider.getBalance(address);
console.log(balance);  // 1500000000000000000（wei）

// ✅ 正确
```

## 📚 FAQ

### 1. ethers.js vs web3.js？

**推荐 ethers.js：**
- 更小的体积
- 更好的 TypeScript 支持
- 更现代的 API 设计
- 更活跃的维护

**使用 web3.js：**
- 老项目迁移成本高

**优化技巧：**
1. 批量操作（multicall）
2. 使用 Layer 2（Arbitrum/Optimism）
3. Gas Token（CHI/GST2）
4. 非高峰期交易
