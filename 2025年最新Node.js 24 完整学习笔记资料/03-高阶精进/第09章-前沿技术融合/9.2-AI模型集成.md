## 📚 知识导读

### 为什么在 Node.js 中运行 AI 模型？

**优势：**
- **隐私保护**：敏感数据不离开服务器
- **低延迟**：本地推理，无需网络请求
- **成本优化**：减少云端 API 调用费用
- **离线可用**：无需互联网连接

**场景：**
- 图像分类、目标检测
- 文本分类、情感分析
- 推荐系统、异常检测

### 2. 图像分类（MobileNet）

```javascript
import * as tf from '@tensorflow/tfjs-node';
import mobilenet from '@tensorflow-models/mobilenet';
import { readFileSync } from 'fs';
import Jimp from 'jimp';

// 加载模型
const model = await mobilenet.load({
  version: 2,
  alpha: 1.0,  // 模型大小（0.25, 0.5, 0.75, 1.0）
});

// 图像分类
async function classifyImage(imagePath) {
  // 读取图像
  const image = await Jimp.read(imagePath);
  
  // 转换为 Tensor
  const tensor = tf.node.decodeImage(image.bitmap.data, 3);
  
  // 预测（Top 3）
  const predictions = await model.classify(tensor);
  
  console.log('Predictions:', predictions);
  // [
  //   { className: 'Persian cat', probability: 0.95 },
  //   { className: 'Egyptian cat', probability: 0.03 },
  //   { className: 'tiger cat', probability: 0.01 }
  // ]
  
  // 释放内存
  tensor.dispose();
  
  return predictions;
}

// 使用
await classifyImage('./cat.jpg');
```

### 4. 加载自定义模型

```javascript
// 从 TensorFlow SavedModel 转换
// Python: model.save('my_model', save_format='tf')
// CLI: tensorflowjs_converter --input_format=tf_saved_model my_model/ web_model/

// 加载模型
const model = await tf.loadLayersModel('file://./web_model/model.json');

// 预测
async function predict(inputData) {
  const inputTensor = tf.tensor2d([inputData]);
  const output = model.predict(inputTensor);
  const result = await output.data();
  
  inputTensor.dispose();
  output.dispose();
  
  return Array.from(result);
}

const result = await predict([1, 2, 3, 4]);
console.log('Prediction:', result);
```

### 5. 批量推理优化

```javascript
// ❌ 低效：逐个推理
for (const image of images) {
  const prediction = await model.classify(image);
}

// ✅ 高效：批量推理
const tensors = images.map(img => tf.node.decodeImage(img, 3));
const batchTensor = tf.stack(tensors);

const predictions = await model.classify(batchTensor);

batchTensor.dispose();
tensors.forEach(t => t.dispose());
```

### 3. PyTorch → ONNX 转换

**Python 代码：**
```python
import torch
import torch.onnx

# 定义模型
class SimpleModel(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.linear = torch.nn.Linear(4, 2)
    
    def forward(self, x):
        return self.linear(x)

model = SimpleModel()
model.eval()

# 导出 ONNX
dummy_input = torch.randn(1, 4)
torch.onnx.export(
    model,
    dummy_input,
    "model.onnx",
    input_names=['input'],
    output_names=['output'],
    dynamic_axes={
        'input': {0: 'batch_size'},
        'output': {0: 'batch_size'}
    }
)
```

**Node.js 使用：**
```javascript
const session = await ort.InferenceSession.create('./model.onnx');

const input = new ort.Tensor('float32', new Float32Array([1, 2, 3, 4]), [1, 4]);
const results = await session.run({ input });

console.log('Prediction:', results.output.data);
```

### 4. 图像分类（ResNet-50）

```javascript
import Jimp from 'jimp';

// 预处理图像
async function preprocessImage(imagePath) {
  const image = await Jimp.read(imagePath);
  
  // 调整大小到 224x224
  image.resize(224, 224);
  
  // 转换为 Float32Array
  const data = new Float32Array(3 * 224 * 224);
  
  const mean = [0.485, 0.456, 0.406];
  const std = [0.229, 0.224, 0.225];
  
  let offset = 0;
  for (let c = 0; c < 3; c++) {
    for (let h = 0; h < 224; h++) {
      for (let w = 0; w < 224; w++) {
        const pixel = Jimp.intToRGBA(image.getPixelColor(w, h));
        const value = (pixel[['r', 'g', 'b'][c]] / 255.0 - mean[c]) / std[c];
        data[offset++] = value;
      }
    }
  }
  
  return new ort.Tensor('float32', data, [1, 3, 224, 224]);
}

// 加载 ResNet-50
const session = await ort.InferenceSession.create('./resnet50.onnx');

// 分类
const inputTensor = await preprocessImage('./cat.jpg');
const results = await session.run({ input: inputTensor });

const output = results.output.data;
const maxIndex = output.indexOf(Math.max(...output));

console.log('Predicted class:', maxIndex);
// 加载 ImageNet 标签
const labels = JSON.parse(readFileSync('./imagenet_labels.json'));
console.log('Label:', labels[maxIndex]);
```

## 🎯 模型优化

### 1. 量化（Quantization）

**INT8 量化：**
```python
# Python: TensorFlow → TFLite → ONNX
import tensorflow as tf

converter = tf.lite.TFLiteConverter.from_saved_model('my_model')
converter.optimizations = [tf.lite.Optimize.DEFAULT]
converter.target_spec.supported_types = [tf.int8]

tflite_model = converter.convert()
with open('model_int8.tflite', 'wb') as f:
    f.write(tflite_model)
```

**性能对比：**
- FP32：100ms
- FP16：60ms（GPU）
- INT8：30ms ✅（精度损失 < 1%）

### 3. 知识蒸馏（Knowledge Distillation）

```python
# 教师模型（大模型）
teacher_model = load_model('large_model.h5')

# 学生模型（小模型）
student_model = create_small_model()

# 蒸馏训练
def distillation_loss(y_true, y_pred, teacher_pred, temperature=3):
    student_loss = tf.keras.losses.sparse_categorical_crossentropy(y_true, y_pred)
    distillation_loss = tf.keras.losses.kl_divergence(
        tf.nn.softmax(teacher_pred / temperature),
        tf.nn.softmax(y_pred / temperature)
    ) * temperature ** 2
    return student_loss + 0.5 * distillation_loss

# 训练学生模型
for x, y in dataset:
    teacher_pred = teacher_model.predict(x)
    student_model.train_on_batch(x, y, teacher_pred)
```

## 🎯 实时推理服务

### 1. Express API

```javascript
import express from 'express';
import multer from 'multer';
import * as tf from '@tensorflow/tfjs-node';
import mobilenet from '@tensorflow-models/mobilenet';

const app = express();
const upload = multer({ dest: 'uploads/' });

// 加载模型（启动时）
let model;
(async () => {
  model = await mobilenet.load();
  console.log('Model loaded');
})();

// 分类 API
app.post('/classify', upload.single('image'), async (req, res) => {
  try {
    const imageBuffer = readFileSync(req.file.path);
    const tensor = tf.node.decodeImage(imageBuffer, 3);
    
    const predictions = await model.classify(tensor);
    
    tensor.dispose();
    unlinkSync(req.file.path);  // 删除临时文件
    
    res.json({ predictions });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000, () => {
  console.log('AI API running on http://localhost:3000');
});
```

### 2. 批量推理队列

```javascript
import Bull from 'bull';

const inferenceQueue = new Bull('image-inference', process.env.REDIS_URL);

// 生产者
app.post('/classify-async', upload.single('image'), async (req, res) => {
  const job = await inferenceQueue.add({
    imagePath: req.file.path,
    userId: req.user.id,
  });
  
  res.json({ jobId: job.id });
});

// 消费者（批量处理）
inferenceQueue.process(10, async (job) => {  // 并发10个
  const { imagePath } = job.data;
  
  const imageBuffer = readFileSync(imagePath);
  const tensor = tf.node.decodeImage(imageBuffer, 3);
  
  const predictions = await model.classify(tensor);
  
  tensor.dispose();
  unlinkSync(imagePath);
  
  return predictions;
});

// 查询结果
app.get('/classify-result/:jobId', async (req, res) => {
  const job = await inferenceQueue.getJob(req.params.jobId);
  
  if (!job) {
    return res.status(404).json({ error: 'Job not found' });
  }
  
  const state = await job.getState();
  
  if (state === 'completed') {
    res.json({ status: 'completed', result: job.returnvalue });
  } else if (state === 'failed') {
    res.json({ status: 'failed', error: job.failedReason });
  } else {
    res.json({ status: state });
  }
});
```

## ⚠️ 三个高频错误

### 错误 1：忘记释放 Tensor

```javascript
// ❌ 错误：内存泄漏
async function predict(imageData) {
  const tensor = tf.tensor(imageData);
  const result = model.predict(tensor);
  return result.data();
  // 忘记 dispose()
}

// ✅ 正确
async function predict(imageData) {
  return tf.tidy(() => {
    const tensor = tf.tensor(imageData);
    const result = model.predict(tensor);
    return result.data();  // tf.tidy 自动清理
  });
}
```

## 📚 FAQ

### 1. TensorFlow.js vs ONNX Runtime？

| 特性 | TensorFlow.js | ONNX Runtime |
|------|---------------|--------------|
| **生态** | TensorFlow 官方 | 微软开源 |
| **模型** | TF/Keras | 多框架（PyTorch/TF/Scikit-learn） |
| **性能** | 中等 | 高（优化更好） |
| **易用性** | 高（预训练模型多） | 中（需要转换） |

**推荐：**
- 简单场景：TensorFlow.js
- 高性能/多框架：ONNX Runtime

### 2. 如何选择模型大小？

**权衡：**
- **大模型**（ResNet-152）：精度高、速度慢
- **小模型**（MobileNetV2）：精度中、速度快

**建议：**
1. 测试多个模型
2. 找到精度/速度平衡点
3. 使用量化/剪枝优化

### 3. GPU 内存不足怎么办？

**方案：**
1. 减小 batch size
2. 使用模型量化（INT8）
3. 启用混合精度（FP16）
4. 使用梯度累积
