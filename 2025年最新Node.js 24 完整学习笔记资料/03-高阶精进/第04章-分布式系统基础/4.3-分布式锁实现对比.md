# 4.3 分布式锁实现对比

> Redis / Redlock / ZooKeeper / etcd 四种方案的深度对比

## 📚 核心知识

### 1. 为什么需要分布式锁？

**单机锁 vs 分布式锁：**

```javascript
// ❌ 单机锁（无法跨进程/跨服务器）
const lock = new Map();

if (!lock.get('resource')) {
  lock.set('resource', true);
  // 执行业务逻辑
  lock.delete('resource');
}

// 问题：多个Node.js进程无法共享内存

// ✅ 分布式锁（所有进程共享）
const locked = await redis.setnx('lock:resource', 'token');

if (locked) {
  // 执行业务逻辑
  await redis.del('lock:resource');
}
```

**应用场景：**

1. **秒杀系统**：防止超卖
2. **定时任务**：避免重复执行
3. **资源分配**：确保互斥访问
4. **分布式爬虫**：任务去重

### 实战1：Redis SETNX 实现（基础版）

```javascript
// redis-lock-basic.js
import Redis from 'ioredis';

class RedisLock {
  constructor() {
    this.redis = new Redis();
  }
  
  // 获取锁
  async acquire(lockKey, ttl = 10000) {
    const locked = await this.redis.set(
      lockKey,
      'locked',
      'PX', ttl,  // 毫秒级过期时间
      'NX'        // 仅当 key 不存在时设置
    );
    
    return locked === 'OK';
  }
  
  // 释放锁
  async release(lockKey) {
    await this.redis.del(lockKey);
  }
}

// 测试
async function testRedisLock() {
  const lock = new RedisLock();
  
  const acquired = await lock.acquire('lock:resource', 5000);
  
  if (acquired) {
    console.log('✅ Lock acquired');
    
    // 执行业务逻辑
    await doWork();
    
    await lock.release('lock:resource');
    console.log('✅ Lock released');
  } else {
    console.log('❌ Failed to acquire lock');
  }
}

async function doWork() {
  console.log('Working...');
  await new Promise(r => setTimeout(r, 1000));
}

testRedisLock();
```

**问题：**
```
❌ 误删别人的锁：进程A的锁过期后，进程B获取锁，进程A又删除了B的锁
❌ 锁过期问题：业务执行时间超过TTL，锁自动释放
```

### 实战2：Redis Lua 脚本实现（生产级）

```javascript
// redis-lock-lua.js
import Redis from 'ioredis';
import { nanoid } from 'nanoid';

class RedisLockWithLua {
  constructor() {
    this.redis = new Redis();
    
    // Lua 脚本：原子性释放锁（检查token + 删除）
    this.unlockScript = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("del", KEYS[1])
      else
        return 0
      end
    `;
  }
  
  // 获取锁
  async acquire(lockKey, ttl = 10000) {
    const token = nanoid();  // 唯一标识
    
    const locked = await this.redis.set(
      lockKey,
      token,
      'PX', ttl,
      'NX'
    );
    
    if (locked === 'OK') {
      return { success: true, token };
    } else {
      return { success: false, token: null };
    }
  }
  
  // 释放锁（使用 Lua 脚本）
  async release(lockKey, token) {
    const result = await this.redis.eval(
      this.unlockScript,
      1,
      lockKey,
      token
    );
    
    return result === 1;
  }
  
  // 自动续期（Watch Dog）
  startWatchDog(lockKey, token, ttl, renewInterval) {
    const intervalId = setInterval(async () => {
      try {
        // 检查锁是否还属于自己
        const currentToken = await this.redis.get(lockKey);
        
        if (currentToken === token) {
          // 续期
          await this.redis.pexpire(lockKey, ttl);
          console.log(`🔄 Lock renewed: ${lockKey}`);
        } else {
          // 锁已被其他进程获取，停止续期
          clearInterval(intervalId);
          console.log('⏹️ Lock lost, stopping watch dog');
        }
      } catch (error) {
        console.error('❌ Watch dog error:', error);
        clearInterval(intervalId);
      }
    }, renewInterval);
    
    return intervalId;
  }
}

// 测试
async function testRedisLockWithLua() {
  const lock = new RedisLockWithLua();
  
  const { success, token } = await lock.acquire('lock:resource', 10000);
  
  if (success) {
    console.log('✅ Lock acquired with token:', token);
    
    // 启动 Watch Dog（每隔 3 秒续期）
    const watchDog = lock.startWatchDog('lock:resource', token, 10000, 3000);
    
    // 执行长时间业务逻辑
    await new Promise(r => setTimeout(r, 15000));  // 15秒
    
    // 停止 Watch Dog
    clearInterval(watchDog);
    
    // 释放锁
    const released = await lock.release('lock:resource', token);
    
    if (released) {
      console.log('✅ Lock released');
    } else {
      console.log('⚠️ Lock was already released or expired');
    }
  } else {
    console.log('❌ Failed to acquire lock');
  }
}

testRedisLockWithLua();
```

### 实战3：Redlock 算法（多节点Redis）

**原理：**

```
Redlock = 多个独立的 Redis 实例

步骤：
1. 获取当前时间戳 T1
2. 依次尝试从 N 个 Redis 实例获取锁
3. 如果在 TTL 时间内，从大多数（N/2+1）实例获取到锁，则成功
4. 计算锁的有效时间 = TTL - (T2 - T1) - 时钟漂移
5. 如果有效时间 > 0，则锁有效
6. 否则，释放所有已获取的锁
```

**代码实现：**

```javascript
// redlock.js
import Redlock from 'redlock';
import Redis from 'ioredis';

class RedlockManager {
  constructor() {
    // 3 个独立的 Redis 实例
    const client1 = new Redis({ host: '192.168.1.10', port: 6379 });
    const client2 = new Redis({ host: '192.168.1.11', port: 6379 });
    const client3 = new Redis({ host: '192.168.1.12', port: 6379 });
    
    this.redlock = new Redlock(
      [client1, client2, client3],
      {
        driftFactor: 0.01,  // 时钟漂移因子
        retryCount: 3,      // 重试次数
        retryDelay: 200,    // 重试延迟（毫秒）
        retryJitter: 200,   // 重试随机延迟
      }
    );
  }
  
  // 获取锁
  async acquire(resource, ttl = 10000) {
    try {
      const lock = await this.redlock.acquire([`lock:${resource}`], ttl);
      
      console.log('✅ Redlock acquired:', lock.value);
      
      return lock;
    } catch (error) {
      console.error('❌ Failed to acquire Redlock:', error.message);
      return null;
    }
  }
  
  // 释放锁
  async release(lock) {
    try {
      await lock.release();
      console.log('✅ Redlock released');
    } catch (error) {
      console.error('❌ Failed to release Redlock:', error.message);
    }
  }
}

// 测试
async function testRedlock() {
  const manager = new RedlockManager();
  
  const lock = await manager.acquire('resource', 10000);
  
  if (lock) {
    // 执行业务逻辑
    await new Promise(r => setTimeout(r, 5000));
    
    // 释放锁
    await manager.release(lock);
  }
}

testRedlock();
```

### 实战4：ZooKeeper 分布式锁

**原理：**

```
ZooKeeper 临时顺序节点实现锁

步骤：
1. 创建临时顺序节点 /lock/0000000001
2. 获取 /lock 下所有子节点
3. 如果当前节点是最小的，则获取锁
4. 否则，监听前一个节点的删除事件
5. 前一个节点删除后，重新判断是否最小
```

**代码实现：**

```javascript
// zookeeper-lock.js
import ZooKeeper from 'node-zookeeper-client';

class ZooKeeperLock {
  constructor() {
    this.client = ZooKeeper.createClient('localhost:2181');
    this.lockPath = '/locks';
    this.nodePath = null;
  }
  
  async connect() {
    return new Promise((resolve, reject) => {
      this.client.once('connected', resolve);
      this.client.once('error', reject);
      this.client.connect();
    });
  }
  
  // 获取锁
  async acquire(resource) {
    const fullLockPath = `${this.lockPath}/${resource}`;
    
    // 确保父节点存在
    await this.createPath(fullLockPath);
    
    // 创建临时顺序节点
    this.nodePath = await this.createNode(`${fullLockPath}/lock-`, true, true);
    
    console.log(`Created node: ${this.nodePath}`);
    
    // 尝试获取锁
    while (true) {
      const children = await this.getChildren(fullLockPath);
      const sortedChildren = children.sort();
      
      const currentNode = this.nodePath.split('/').pop();
      const index = sortedChildren.indexOf(currentNode);
      
      if (index === 0) {
        // 当前节点是最小的，获取锁成功
        console.log('✅ Lock acquired');
        return true;
      } else {
        // 监听前一个节点
        const prevNode = sortedChildren[index - 1];
        const prevPath = `${fullLockPath}/${prevNode}`;
        
        console.log(`Waiting for ${prevPath} to be deleted...`);
        
        await this.watchNode(prevPath);
      }
    }
  }
  
  // 释放锁
  async release() {
    if (this.nodePath) {
      await this.deleteNode(this.nodePath);
      console.log('✅ Lock released');
    }
  }
  
  // 辅助方法
  createPath(path) {
    return new Promise((resolve, reject) => {
      this.client.mkdirp(path, (error) => {
        if (error && error.code !== ZooKeeper.Exception.NODE_EXISTS) {
          reject(error);
        } else {
          resolve();
        }
      });
    });
  }
  
  createNode(path, ephemeral, sequential) {
    return new Promise((resolve, reject) => {
      const mode = (ephemeral ? ZooKeeper.CreateMode.EPHEMERAL : 0) |
                   (sequential ? ZooKeeper.CreateMode.SEQUENTIAL : 0);
      
      this.client.create(path, Buffer.from(''), mode, (error, nodePath) => {
        if (error) reject(error);
        else resolve(nodePath);
      });
    });
  }
  
  getChildren(path) {
    return new Promise((resolve, reject) => {
      this.client.getChildren(path, (error, children) => {
        if (error) reject(error);
        else resolve(children);
      });
    });
  }
  
  watchNode(path) {
    return new Promise((resolve) => {
      this.client.exists(path, (event) => {
        if (event.type === ZooKeeper.Event.NODE_DELETED) {
          resolve();
        }
      }, (error, stat) => {
        if (!stat) {
          // 节点已删除
          resolve();
        }
      });
    });
  }
  
  deleteNode(path) {
    return new Promise((resolve, reject) => {
      this.client.remove(path, (error) => {
        if (error) reject(error);
        else resolve();
      });
    });
  }
}

// 测试
async function testZooKeeperLock() {
  const lock = new ZooKeeperLock();
  
  await lock.connect();
  
  await lock.acquire('resource');
  
  // 执行业务逻辑
  await new Promise(r => setTimeout(r, 5000));
  
  await lock.release();
}

testZooKeeperLock();
```

### 实战5：etcd 分布式锁

**原理：**

```
etcd Lease + Revision 实现锁

步骤：
1. 创建租约（Lease，10秒）
2. 使用租约创建 key（/lock/resource）
3. 检查 key 的 Revision 是否最小
4. 如果是最小，则获取锁
5. 否则，监听前一个 key 的删除事件
```

**代码实现：**

```javascript
// etcd-lock.js
import { Etcd3 } from 'etcd3';

class EtcdLock {
  constructor() {
    this.client = new Etcd3({ hosts: 'http://localhost:2379' });
    this.lease = null;
    this.lockKey = null;
  }
  
  // 获取锁
  async acquire(resource, ttl = 10) {
    const lockPrefix = `/locks/${resource}/`;
    
    // 创建租约
    this.lease = this.client.lease(ttl);
    await this.lease.grant();
    
    // 创建 key（绑定租约）
    this.lockKey = `${lockPrefix}${Date.now()}`;
    
    await this.client.put(this.lockKey)
      .value('locked')
      .lease(this.lease)
      .exec();
    
    console.log(`Created key: ${this.lockKey}`);
    
    // 获取所有以 lockPrefix 开头的 key
    const keys = await this.client.getAll()
      .prefix(lockPrefix)
      .keys();
    
    // 按字典序排序
    const sortedKeys = keys.sort();
    
    if (sortedKeys[0] === this.lockKey) {
      // 当前 key 是最小的，获取锁成功
      console.log('✅ Lock acquired');
      
      // 保持租约
      await this.lease.keepalive();
      
      return true;
    } else {
      // 监听前一个 key
      const index = sortedKeys.indexOf(this.lockKey);
      const prevKey = sortedKeys[index - 1];
      
      console.log(`Waiting for ${prevKey} to be deleted...`);
      
      await this.watchKey(prevKey);
      
      // 递归尝试获取锁
      return this.acquire(resource, ttl);
    }
  }
  
  // 释放锁
  async release() {
    if (this.lockKey) {
      await this.client.delete().key(this.lockKey).exec();
      console.log('✅ Lock released');
    }
    
    if (this.lease) {
      await this.lease.revoke();
    }
  }
  
  // 监听 key 删除事件
  watchKey(key) {
    return new Promise((resolve) => {
      const watcher = this.client.watch()
        .key(key)
        .create();
      
      watcher.on('delete', () => {
        watcher.cancel();
        resolve();
      });
    });
  }
}

// 测试
async function testEtcdLock() {
  const lock = new EtcdLock();
  
  await lock.acquire('resource', 10);
  
  // 执行业务逻辑
  await new Promise(r => setTimeout(r, 5000));
  
  await lock.release();
}

testEtcdLock();
```

async function seckill(productId, userId) {
  const lock = new RedisLockWithLua();
  const { success, token } = await lock.acquire(`lock:seckill:${productId}`, 5000);
  
  if (!success) {
    return { success: false, error: 'Too many requests' };

    // 检查库存

    if (parseInt(stock) <= 0) {

    // 扣减库存
    await redis.decr(`stock:${productId}`);
    
    // 创建订单
    const order = await createOrder(productId, userId);
    
    return { success: true, order };
  } finally {
    await lock.release(`lock:seckill:${productId}`, token);
  }
}

## ❌ 高频错误

### 错误1：忘记设置锁的过期时间

**错误：**
```javascript
// ❌ 没有设置过期时间
await redis.setnx('lock:resource', 'locked');

// 问题：进程崩溃后，锁永远不会释放（死锁）
```

**正确：**
```javascript
// ✅ 设置过期时间（原子操作）
await redis.set('lock:resource', 'locked', 'PX', 10000, 'NX');
```

### Q1：如何选择分布式锁方案？

| 方案 | 可靠性 | 性能 | 复杂度 | 适用场景 |
