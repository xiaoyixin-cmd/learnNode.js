# 4.5 幂等性设计：Token / 去重表 / 状态机

> 保证接口多次调用结果一致

---

## 📚 核心知识

### 1. 什么是幂等性？

**定义：**同一个操作执行多次，结果与执行一次相同。

**数学定义：**
```
f(f(x)) = f(x)

示例：
- 幂等操作：SET x = 5（无论执行多少次，x都是5）
- 非幂等操作：x = x + 1（每次执行，x都会增加）
```

**HTTP 方法的幂等性：**

| 方法 | 幂等性 | 说明 |
|------|--------|------|
| GET | ✅ 是 | 只读操作 |
| PUT | ✅ 是 | 全量更新 |
| DELETE | ✅ 是 | 删除相同资源 |
| POST | ❌ 否 | 创建新资源 |
| PATCH | ❌ 否 | 部分更新 |

**为什么需要幂等性？**

```
客户端发送请求 → 服务端处理 → 返回响应
                              ↑
                            超时/丢包
                              ↓
客户端重试 → 服务端再次处理 → 重复操作！

问题：
❌ 订单重复创建
❌ 支付重复扣款
❌ 消息重复消费
```

### 实战1：Token 机制（防重复提交）

**原理：**

```
1. 前端请求 Token
2. 后端生成 Token 并存储到 Redis
3. 前端提交表单时携带 Token
4. 后端验证 Token：
   - Token 存在 → 删除 Token，执行业务
   - Token 不存在 → 拒绝请求（重复提交）
```

**代码实现：**

```javascript
// token-idempotent.js
import express from 'express';
import Redis from 'ioredis';
import { nanoid } from 'nanoid';

const app = express();
app.use(express.json());

const redis = new Redis();

// 获取 Token
app.get('/api/token', async (req, res) => {
  const token = nanoid();
  
  // 存储到 Redis（5 分钟过期）
  await redis.setex(`token:${token}`, 300, '1');
  
  res.json({ token });
});

// 创建订单（幂等）
app.post('/api/orders', async (req, res) => {
  const { token, userId, productId, amount } = req.body;
  
  if (!token) {
    return res.status(400).json({ error: 'Missing token' });
  }
  
  // 验证 Token（原子操作：检查 + 删除）
  const deleted = await redis.del(`token:${token}`);
  
  if (deleted === 0) {
    // Token 不存在或已被使用
    return res.status(409).json({ error: 'Duplicate request' });
  }
  
  try {
    // 执行业务逻辑
    const order = {
      id: `ORD${Date.now()}`,
      userId,
      productId,
      amount,
      status: 'pending',
      createdAt: new Date(),
    };
    
    // 保存订单到数据库
    await saveOrder(order);
    
    res.json(order);
  } catch (error) {
    // 业务失败，恢复 Token（可选）
    await redis.setex(`token:${token}`, 300, '1');
    
    res.status(500).json({ error: 'Internal server error' });
  }
});

async function saveOrder(order) {
  // 模拟保存到数据库
  console.log('Order saved:', order);
}

app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});
```

### 实战2：去重表（唯一索引）

**原理：**

```
使用数据库唯一索引防止重复插入

CREATE TABLE orders (
  id VARCHAR(64) PRIMARY KEY,
  idempotent_key VARCHAR(64) UNIQUE NOT NULL,  -- 幂等键
  user_id VARCHAR(64),
  amount INT
);

INSERT INTO orders (id, idempotent_key, user_id, amount)
VALUES ('ORD123', 'key123', 'user123', 100);

-- 重复插入会失败（UNIQUE 约束）
INSERT INTO orders (id, idempotent_key, user_id, amount)
VALUES ('ORD456', 'key123', 'user456', 200);
-- Error: Duplicate entry 'key123' for key 'idempotent_key'
```

**代码实现：**

```javascript
// deduplication-table.js
import express from 'express';
import { Client as PgClient } from 'pg';
import { nanoid } from 'nanoid';

const app = express();
app.use(express.json());

const db = new PgClient({
  host: 'localhost',
  port: 5432,
  user: 'postgres',
  password: 'secret123',
  database: 'distributed_demo',
});

await db.connect();

// 创建订单表（带幂等键）
await db.query(`
  CREATE TABLE IF NOT EXISTS orders (
    id VARCHAR(64) PRIMARY KEY,
    idempotent_key VARCHAR(64) UNIQUE NOT NULL,
    user_id VARCHAR(64),
    product_id VARCHAR(64),
    amount INT,
    status VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  )
`);

// 创建订单（幂等）
app.post('/api/orders', async (req, res) => {
  const { idempotentKey, userId, productId, amount } = req.body;
  
  if (!idempotentKey) {
    return res.status(400).json({ error: 'Missing idempotent key' });
  }
  
  try {
    // 插入订单（唯一索引保证幂等）
    const result = await db.query(
      `INSERT INTO orders (id, idempotent_key, user_id, product_id, amount, status)
       VALUES ($1, $2, $3, $4, $5, $6)
       RETURNING *`,
      [`ORD${Date.now()}`, idempotentKey, userId, productId, amount, 'pending']
    );
    
    res.json(result.rows[0]);
  } catch (error) {
    if (error.code === '23505') {
      // 唯一索引冲突（重复请求）
      // 查询已存在的订单并返回
      const existing = await db.query(
        'SELECT * FROM orders WHERE idempotent_key = $1',
        [idempotentKey]
      );
      
      res.json(existing.rows[0]);
    } else {
      res.status(500).json({ error: 'Internal server error' });
    }
  }
});

app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});
```

### 实战3：状态机（状态流转）

**原理：**

```
订单状态流转：

pending → paid → shipped → completed
   ↓       ↓        ↓
cancelled cancelled cancelled

规则：
- 只能从 pending → paid
- paid 状态不能重复支付
- completed 状态不能再修改
```

**代码实现：**

```javascript
// state-machine-idempotent.js
import express from 'express';
import { Client as PgClient } from 'pg';

const app = express();
app.use(express.json());

const db = new PgClient({
  host: 'localhost',
  port: 5432,
  user: 'postgres',
  password: 'secret123',
  database: 'distributed_demo',
});

await db.connect();

// 状态流转规则
const STATE_TRANSITIONS = {
  pending: ['paid', 'cancelled'],
  paid: ['shipped', 'cancelled'],
  shipped: ['completed', 'cancelled'],
  completed: [],
  cancelled: [],
};

// 支付订单（幂等）
app.post('/api/orders/:orderId/pay', async (req, res) => {
  const { orderId } = req.params;
  
  try {
    await db.query('BEGIN');
    
    // 查询订单当前状态（加锁）
    const result = await db.query(
      'SELECT * FROM orders WHERE id = $1 FOR UPDATE',
      [orderId]
    );
    
    if (result.rows.length === 0) {
      await db.query('ROLLBACK');
      return res.status(404).json({ error: 'Order not found' });
    }
    
    const order = result.rows[0];
    const currentStatus = order.status;
    
    // 检查状态流转是否合法
    if (currentStatus === 'paid') {
      // 已支付，返回成功（幂等）
      await db.query('COMMIT');
      return res.json({ message: 'Already paid', order });
    }
    
    if (!STATE_TRANSITIONS[currentStatus]?.includes('paid')) {
      await db.query('ROLLBACK');
      return res.status(400).json({
        error: `Cannot transition from ${currentStatus} to paid`,
      });
    }
    
    // 更新状态
    await db.query(
      'UPDATE orders SET status = $1, updated_at = NOW() WHERE id = $2',
      ['paid', orderId]
    );
    
    await db.query('COMMIT');
    
    res.json({ message: 'Payment successful' });
  } catch (error) {
    await db.query('ROLLBACK');
    res.status(500).json({ error: 'Internal server error' });
  }
});

app.listen(3000);
```

### 实战4：分布式锁 + 版本号

**原理：**

```
使用乐观锁（版本号）保证幂等性

UPDATE orders
SET amount = 100, version = version + 1
WHERE id = 'ORD123' AND version = 1;

-- 如果 version 不匹配，更新失败（说明已被修改）
```

**代码实现：**

```javascript
// optimistic-lock-idempotent.js
app.post('/api/orders/:orderId/update', async (req, res) => {
  const { orderId } = req.params;
  const { amount, version } = req.body;
  
  // 更新订单（乐观锁）
  const result = await db.query(
    `UPDATE orders
     SET amount = $1, version = version + 1, updated_at = NOW()
     WHERE id = $2 AND version = $3
     RETURNING *`,
    [amount, orderId, version]
  );
  
  if (result.rows.length === 0) {
    // 版本号不匹配（已被其他请求修改）
    return res.status(409).json({
      error: 'Conflict: Order has been modified',
    });
  }
  
  res.json(result.rows[0]);
});
```

### 场景1：支付接口幂等性

```javascript
// 支付接口（使用 Token + 去重表）
app.post('/api/payments', async (req, res) => {
  const { orderId, amount, idempotentKey } = req.body;
  
  try {
    await db.query('BEGIN');
    
    // 1. 检查支付记录是否已存在（去重表）
    const existing = await db.query(
      'SELECT * FROM payments WHERE idempotent_key = $1',
      [idempotentKey]
    );
    
    if (existing.rows.length > 0) {
      // 已支付，返回已有记录
      await db.query('COMMIT');
      return res.json(existing.rows[0]);
    }
    
    // 2. 检查订单状态
    const order = await db.query(
      'SELECT * FROM orders WHERE id = $1 FOR UPDATE',
      [orderId]
    );
    
    if (order.rows[0].status !== 'pending') {
      await db.query('ROLLBACK');
      return res.status(400).json({ error: 'Invalid order status' });
    }
    
    // 3. 创建支付记录
    const payment = await db.query(
      `INSERT INTO payments (id, order_id, amount, idempotent_key, status)
       VALUES ($1, $2, $3, $4, $5)
       RETURNING *`,
      [`PAY${Date.now()}`, orderId, amount, idempotentKey, 'success']
    );
    
    // 4. 更新订单状态
    await db.query(
      'UPDATE orders SET status = $1 WHERE id = $2',
      ['paid', orderId]
    );
    
    await db.query('COMMIT');
    
    res.json(payment.rows[0]);
  } catch (error) {
    await db.query('ROLLBACK');
    res.status(500).json({ error: 'Payment failed' });
  }
});
```

### 场景2：消息队列消费幂等性

// 消息消费者（使用去重表）
channel.consume('order.created', async (msg) => {
  const order = JSON.parse(msg.content.toString());
  const messageId = msg.properties.messageId;
  
  try {
    // 检查消息是否已处理
    const processed = await redis.get(`processed:${messageId}`);
    
    if (processed) {
      console.log(`Message ${messageId} already processed, skipping`);
      channel.ack(msg);
      return;
    }
    
    // 处理消息
    await processOrder(order);
    
    // 标记为已处理（24 小时过期）
    await redis.setex(`processed:${messageId}`, 86400, '1');
    
    channel.ack(msg);
  } catch (error) {
    console.error('Failed to process message:', error);
    channel.nack(msg);
  }
});

## ❌ 高频错误

### 错误1：未处理 Token 删除失败

**错误：**
```javascript
// ❌ Token 删除后，业务处理失败，导致 Token 无法恢复
const deleted = await redis.del(`token:${token}`);

// 业务处理（可能失败）
await processOrder(order);  // 抛出异常

// Token 已删除，用户无法重试
```

**正确：**
```javascript
// ✅ 使用 Lua 脚本原子性删除 + 处理
const script = `
  local token = redis.call("get", KEYS[1])
  if token then
    redis.call("del", KEYS[1])
    return token
  else
    return nil
  end
`;

const token = await redis.eval(script, 1, `token:${tokenId}`);

if (!token) {
  return res.status(409).json({ error: 'Duplicate request' });
}

try {
  await processOrder(order);
} catch (error) {
  // 恢复 Token
  await redis.setex(`token:${tokenId}`, 300, token);
  throw error;
}
```

## 🤔 FAQ

### Q1：如何选择幂等性方案？

| 方案 | 复杂度 | 性能 | 适用场景 |
|------|--------|------|---------|
| Token | 低 | 高 | 表单提交 |
| 去重表 | 低 | 中 | 订单创建 |
| 状态机 | 中 | 高 | 复杂业务 |
| 乐观锁 | 低 | 高 | 高并发更新 |

### Q2：Token 过期时间如何设置？

**建议：**
- 表单提交：5-10 分钟
- 支付接口：30 分钟
- API 调用：1-5 分钟

**权衡：**
- 过短：用户操作慢导致 Token 过期
- 过长：Redis 内存占用高
