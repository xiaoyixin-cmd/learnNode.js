## 📖 知识点概述

### 核心概念

**Mongoose**是Node.js中最流行的MongoDB对象数据建模（ODM）库，提供Schema定义、数据验证、查询构建、中间件等功能。Mongoose 8.x（2024年发布）带来了更好的TypeScript支持和性能优化。

### 解决的问题

MongoDB原生驱动的不足：
- **缺少Schema**：需要手动验证数据结构
- **重复代码**：每次都要写连接、查询代码
- **类型安全**：TypeScript支持不完善
- **复杂查询**：没有构建器，容易出错

Mongoose的解决方案：
- **Schema定义**：强类型数据结构
- **内置验证**：自动验证字段类型、必填、唯一等
- **查询构建器**：链式调用，易读易写
- **中间件**：pre/post钩子处理业务逻辑
- **关联查询**：populate实现类似SQL的JOIN

### 主要特点

1. **Schema驱动**：定义数据结构和验证规则
2. **中间件支持**：保存前/后执行自定义逻辑
3. **查询构建器**：链式API，支持复杂查询
4. **虚拟属性**：计算字段，不存储到数据库
5. **插件系统**：可扩展功能

### 典型应用场景

- **内容管理系统**：文章、评论、用户
- **电商平台**：商品、订单、库存
- **社交网络**：用户、好友关系、动态
- **实时应用**：聊天记录、通知

#### 第一步：定义Schema和Model

```javascript
// models/User.js
import mongoose from 'mongoose';

// 💡 定义Schema
const userSchema = new mongoose.Schema({
  username: {
    type: String,
    required: [true, '用户名不能为空'], // 自定义错误消息
    unique: true,
    trim: true, // 去除首尾空格
    minlength: [3, '用户名至少3个字符'],
    maxlength: [20, '用户名最多20个字符'],
    lowercase: true // 自动转小写
  },
  email: {
    type: String,
    required: true,
    unique: true,
    lowercase: true,
    match: [/^\S+@\S+\.\S+$/, '邮箱格式不正确']
  },
  password: {
    type: String,
    required: true,
    minlength: 6,
    select: false // 💡 查询时默认不返回此字段
  },
  age: {
    type: Number,
    min: [0, '年龄不能为负数'],
    max: [150, '年龄不能超过150']
  },
  avatar: {
    type: String,
    default: 'default-avatar.png'
  },
  role: {
    type: String,
    enum: ['user', 'admin', 'moderator'], // 枚举值
    default: 'user'
  },
  isActive: {
    type: Boolean,
    default: true
  },
  tags: [{
    type: String
  }],
  profile: {
    bio: String,
    location: String,
    website: String
  },
  createdAt: {
    type: Date,
    default: Date.now
  },
  updatedAt: {
    type: Date,
    default: Date.now
  }
}, {
  // 💡 Schema选项
  timestamps: true, // 自动添加createdAt和updatedAt
  versionKey: false // 禁用__v字段
});

// 💡 索引：提升查询性能
userSchema.index({ username: 1 }); // 单字段索引
userSchema.index({ email: 1 });
userSchema.index({ createdAt: -1 }); // 降序索引
userSchema.index({ username: 1, email: 1 }); // 复合索引

// 💡 虚拟属性：不存储到数据库
userSchema.virtual('fullProfile').get(function() {
  return {
    username: this.username,
    email: this.email,
    bio: this.profile?.bio || '这个人很懒，什么都没写'
  };
});

// 💡 实例方法：操作单个文档
userSchema.methods.comparePassword = async function(password) {
  // 需要显式查询password字段
  const user = await mongoose.model('User').findById(this._id).select('+password');
  return user.password === password; // 实际应该用bcrypt
};

userSchema.methods.toPublicJSON = function() {
  const obj = this.toObject();
  delete obj.password;
  return obj;
};

// 💡 静态方法：操作Model
userSchema.statics.findByEmail = function(email) {
  return this.findOne({ email: email.toLowerCase() });
};

userSchema.statics.findActiveUsers = function() {
  return this.find({ isActive: true });
};

// 💡 查询助手：扩展查询链
userSchema.query.byRole = function(role) {
  return this.where({ role });
};

userSchema.query.recent = function() {
  return this.sort({ createdAt: -1 });
};

// 💡 中间件：保存前执行
userSchema.pre('save', function(next) {
  // this指向当前文档
  if (this.isModified('username')) {
    console.log('用户名已修改:', this.username);
  }
  
  this.updatedAt = Date.now();
  next();
});

// 💡 中间件：保存后执行
userSchema.post('save', function(doc, next) {
  console.log('用户已保存:', doc.username);
  next();
});

// 💡 中间件：删除前执行
userSchema.pre('deleteOne', { document: true }, function(next) {
  console.log('用户将被删除:', this.username);
  next();
});

// 💡 创建Model
const User = mongoose.model('User', userSchema);

export default User;
```

#### 第四步：聚合管道（Aggregation）

```javascript
// aggregation-examples.js
import User from './models/User.js';
import Post from './models/Post.js';

async function aggregationExamples() {
  // 💡 基础聚合：统计用户数
  const count = await User.aggregate([
    { $match: { isActive: true } }, // 筛选
    { $count: 'totalActiveUsers' }  // 计数
  ]);
  console.log('活跃用户数:', count[0]?.totalActiveUsers);
  
  // 💡 分组统计
  const roleStats = await User.aggregate([
    {
      $group: {
        _id: '$role', // 按role分组
        count: { $sum: 1 }, // 计数
        avgAge: { $avg: '$age' }, // 平均年龄
        minAge: { $min: '$age' },
        maxAge: { $max: '$age' }
      }
    },
    { $sort: { count: -1 } } // 按数量降序
  ]);
  console.log('角色统计:', roleStats);
  
  // 💡 复杂聚合：文章统计
  const postStats = await Post.aggregate([
    // 阶段1：关联作者
    {
      $lookup: {
        from: 'users', // 集合名（小写复数）
        localField: 'author',
        foreignField: '_id',
        as: 'authorInfo'
      }
    },
    // 阶段2：展开数组
    { $unwind: '$authorInfo' },
    // 阶段3：添加计算字段
    {
      $addFields: {
        commentCount: { $size: '$comments' },
        likeCount: { $size: '$likes' }
      }
    },
    // 阶段4：筛选
    {
      $match: {
        commentCount: { $gte: 10 } // 评论数>=10
      }
    },
    // 阶段5：投影（选择字段）
    {
      $project: {
        title: 1,
        'authorInfo.username': 1,
        commentCount: 1,
        likeCount: 1,
        score: {
          $add: [
            { $multiply: ['$commentCount', 2] },
            '$likeCount'
          ]
        }
      }
    },
    // 阶段6：排序
    { $sort: { score: -1 } },
    // 阶段7：分页
    { $skip: 0 },
    { $limit: 10 }
  ]);
  
  console.log('热门文章:', postStats);
  
  // 💡 时间范围统计
  const last7Days = new Date();
  last7Days.setDate(last7Days.getDate() - 7);
  
  const recentStats = await Post.aggregate([
    {
      $match: {
        createdAt: { $gte: last7Days }
      }
    },
    {
      $group: {
        _id: {
          $dateToString: {
            format: '%Y-%m-%d',
            date: '$createdAt'
          }
        },
        count: { $sum: 1 }
      }
    },
    { $sort: { _id: 1 } }
  ]);
  
  console.log('最近7天发帖量:', recentStats);
}
```

#### 第五步：事务处理

```javascript
// transaction-example.js
import mongoose from 'mongoose';
import User from './models/User.js';
import Post from './models/Post.js';

async function transferPoints() {
  // 💡 启动会话
  const session = await mongoose.startSession();
  
  try {
    // 💡 开始事务
    session.startTransaction();
    
    // 操作1：扣除用户A的积分
    const userA = await User.findOneAndUpdate(
      { username: 'alice', points: { $gte: 100 } },
      { $inc: { points: -100 } },
      { session, new: true } // 💡 传入session
    );
    
    if (!userA) {
      throw new Error('用户A积分不足');
    }
    
    // 操作2：增加用户B的积分
    const userB = await User.findOneAndUpdate(
      { username: 'bob' },
      { $inc: { points: 100 } },
      { session, new: true }
    );
    
    if (!userB) {
      throw new Error('用户B不存在');
    }
    
    // 💡 提交事务
    await session.commitTransaction();
    console.log('转账成功');
    console.log('用户A剩余积分:', userA.points);
    console.log('用户B积分:', userB.points);
    
  } catch (error) {
    // 💡 回滚事务
    await session.abortTransaction();
    console.error('转账失败，已回滚:', error.message);
  } finally {
    // 💡 结束会话
    session.endSession();
  }
}

// ⚠️ 注意：MongoDB事务需要副本集（Replica Set）
// 本地开发时，可以使用单节点副本集：
// mongod --replSet rs0
// 进入mongo shell执行：rs.initiate()
```

### 错误处理：3个高频问题

#### 问题1：ValidationError（验证失败）

**错误表现**：
```
ValidationError: User validation failed: email: 邮箱格式不正确
```

**原因**：数据不符合Schema定义的验证规则

**解决方案**：
```javascript
// ✅ 捕获验证错误
try {
  const user = await User.create({
    username: 'ab', // 少于3个字符
    email: 'invalid-email' // 格式错误
  });
} catch (error) {
  if (error.name === 'ValidationError') {
    // 获取所有验证错误
    const errors = Object.values(error.errors).map(err => ({
      field: err.path,
      message: err.message
    }));
    
    console.log('验证失败:', errors);
    // [
    //   { field: 'username', message: '用户名至少3个字符' },
    //   { field: 'email', message: '邮箱格式不正确' }
    // ]
  }
}

// ✅ 前端验证 + 后端验证
function validateUserInput(data) {
  const errors = {};
  
  if (!data.username || data.username.length < 3) {
    errors.username = '用户名至少3个字符';
  }
  
  if (!/^\S+@\S+\.\S+$/.test(data.email)) {
    errors.email = '邮箱格式不正确';
  }
  
  return { isValid: Object.keys(errors).length === 0, errors };
}
```

#### 问题2：E11000 duplicate key error（唯一索引冲突）

**错误表现**：
```
MongoServerError: E11000 duplicate key error collection: myapp.users index: email_1 dup key: { email: "test@example.com" }
```

**原因**：插入的数据违反唯一索引

**解决方案**：
```javascript
// ✅ 捕获重复键错误
try {
  const user = await User.create({
    username: 'john',
    email: 'john@example.com', // 已存在
    password: '123456'
  });
} catch (error) {
  if (error.code === 11000) {
    const field = Object.keys(error.keyPattern)[0];
    console.error(`${field} 已存在`);
    
    // 返回友好的错误消息
    return { error: `该${field === 'email' ? '邮箱' : field}已被注册` };
  }
}

// ✅ 先检查再创建
const existingUser = await User.findOne({ email: 'john@example.com' });
if (existingUser) {
  console.log('邮箱已存在');
} else {
  const newUser = await User.create({ /* ... */ });
}
```

```javascript
async function login(username, password) {
  try {
    // 查找用户（包含password字段）

      return { error: '用户名或密码错误' };
```

### 场景2：电商商品管理

```javascript
// product-system.js
import mongoose from 'mongoose';

const productSchema = new mongoose.Schema({
  name: { type: String, required: true, index: true },
  slug: { type: String, unique: true },
  price: { type: Number, required: true, min: 0 },
  stock: { type: Number, required: true, min: 0 },
  category: { type: mongoose.Schema.Types.ObjectId, ref: 'Category' },
  images: [String],
  status: { type: String, enum: ['draft', 'active', 'sold_out'], default: 'draft' }
}, { timestamps: true });

const Product = mongoose.model('Product', productSchema);

// 减少库存（事务）
async function purchaseProduct(productId, quantity) {
  const session = await mongoose.startSession();
  session.startTransaction();
  
  try {
    // 检查并减少库存
    const product = await Product.findOneAndUpdate(
      { _id: productId, stock: { $gte: quantity } },
      { $inc: { stock: -quantity } },
      { session, new: true }
    );
    
    if (!product) {
      throw new Error('库存不足');
    }
    
    // 更新商品状态
    if (product.stock === 0) {
      product.status = 'sold_out';
      await product.save({ session });
    }
    
    await session.commitTransaction();
    return { success: true, product };
  } catch (error) {
    await session.abortTransaction();
    return { error: error.message };
  } finally {
    session.endSession();
  }
}

// 热门商品统计
async function getHotProducts() {
  return await Product.aggregate([
    { $match: { status: 'active' } },
    {
      $lookup: {
        from: 'orders',
        localField: '_id',
        foreignField: 'items.product',
        as: 'orders'
      }
    },
    {
      $addFields: {
        salesCount: { $size: '$orders' }
      }
    },
    { $sort: { salesCount: -1 } },
    { $limit: 10 },
    { $project: { name: 1, price: 1, salesCount: 1 } }
  ]);
}
```

**可量化效果**：
- **并发安全**：事务保证库存准确
- **查询性能**：聚合查询<100ms
- **数据一致性**：自动更新状态

### 性能优化案例

#### 优化前：N+1查询问题

```javascript
// ❌ 每个post都查询一次author
const posts = await Post.find();
for (const post of posts) {
  post.author = await User.findById(post.author);
}
// 1000个post = 1001次查询
```

#### 优化后：使用Populate

```javascript
// ✅ 一次查询所有posts，一次查询所有authors
const posts = await Post.find().populate('author');
// 2次查询
```

#### 优化指标

- **查询次数**：1001次 → 2次（减少99.8%）
- **响应时间**：5000ms → 50ms（快100倍）

## 📋 总结

### 核心要点

1. **Mongoose**：MongoDB的ODM库，提供Schema和验证
2. **CRUD**：create/find/update/delete四种操作
3. **Populate**：实现类似SQL的JOIN
4. **聚合**：复杂统计和数据处理

### 实现关键

1. **定义Schema**：`new mongoose.Schema({})`
2. **创建Model**：`mongoose.model('Name', schema)`
3. **CRUD操作**：`Model.find/create/update/delete`
4. **关联查询**：`.populate('field')`
5. **事务**：`startSession() + startTransaction()`

### 应用价值

- **开发效率**：Schema定义、自动验证
- **代码质量**：类型安全、易维护
- **性能**：索引优化、连接池
- **安全**：防注入、数据验证

### 面试重点

1. **Mongoose的优势是什么？**
   - Schema定义、自动验证、中间件、Populate

2. **如何优化Mongoose查询性能？**
   - 添加索引、使用lean()、select指定字段、避免N+1

3. **MongoDB事务的限制？**
   - 需要副本集、性能略低、不适合高频操作

4. **如何防止并发修改冲突？**
   - 使用事务、乐观锁（版本号）、原子操作（$inc）

```javascript
const users = await User.find().lean();
```
