## 📖 知识点概述

### 核心概念

**pg**是Node.js中PostgreSQL的官方客户端库（GitHub 12K+ stars）。PostgreSQL是功能最强大的开源关系型数据库，支持JSON、全文搜索、地理信息等高级特性。

### PostgreSQL vs MySQL

| 特性 | PostgreSQL | MySQL |
|------|------------|-------|
| **SQL标准** | 高度兼容 | 部分兼容 |
| **JSON支持** | 原生JSONB类型 | JSON类型较弱 |
| **全文搜索** | 内置 | 需插件 |
| **窗口函数** | 完整支持 | 8.0+支持 |
| **事务隔离** | 更严格 | 一般 |
| **扩展性** | 丰富插件 | 较少 |

### 主要特点

1. **JSONB类型**：高效存储和查询JSON
2. **LISTEN/NOTIFY**：实时消息推送
3. **全文搜索**：内置搜索引擎
4. **数组类型**：原生支持数组
5. **窗口函数**：复杂分析查询

### 典型应用场景

- **复杂查询**：金融、数据分析
- **JSON存储**：灵活数据结构
- **全文搜索**：内容管理系统
- **实时通知**：聊天、监控系统

### 实践：从入门到精通

#### 第一步：基础连接

```javascript
// db.js - 连接池
import pg from 'pg';
const { Pool } = pg;

// 💡 创建连接池
export const pool = new Pool({
  host: process.env.DB_HOST || 'localhost',
  port: process.env.DB_PORT || 5432,
  user: process.env.DB_USER || 'postgres',
  password: process.env.DB_PASSWORD || '',
  database: process.env.DB_NAME || 'myapp',
  
  // 💡 连接池配置
  max: 20, // 最大连接数
  idleTimeoutMillis: 30000, // 空闲超时
  connectionTimeoutMillis: 2000, // 连接超时
  
  // 💡 SSL配置（生产环境）
  ssl: process.env.NODE_ENV === 'production' ? {
    rejectUnauthorized: false
  } : false
});

// 测试连接
pool.query('SELECT NOW()', (error, result) => {
  if (error) {
    console.error('❌ PostgreSQL连接失败:', error);
  } else {
    console.log('✅ PostgreSQL已连接:', result.rows[0].now);
  }
});

// 监听错误
pool.on('error', (error) => {
  console.error('PostgreSQL Pool Error:', error);
});

// 优雅关闭
process.on('SIGINT', async () => {
  await pool.end();
  console.log('PostgreSQL连接池已关闭');
  process.exit(0);
});
```

```javascript
// crud.js
import { pool } from './db.js';

// ========== 创建（Create） ==========

async function createUser(username, email, password) {
  // 💡 PostgreSQL使用$1, $2占位符
  const sql = `
    INSERT INTO users (username, email, password, profile)
    VALUES ($1, $2, $3, $4)
    RETURNING *
  `;
  
  const values = [
    username,
    email,
    password,
    JSON.stringify({ bio: 'Hello' }) // JSONB
  ];
  
  try {
    const result = await pool.query(sql, values);
    console.log('用户已创建:', result.rows[0]);
    return result.rows[0];
  } catch (error) {
    if (error.code === '23505') { // unique_violation
      console.error('用户名或邮箱已存在');
    }
    throw error;
  }
}

// 💡 批量插入
async function createMultipleUsers(users) {
  const client = await pool.connect();
  
  try {
    await client.query('BEGIN');
    
    for (const user of users) {
      await client.query(
        'INSERT INTO users (username, email, password) VALUES ($1, $2, $3)',
        [user.username, user.email, user.password]
      );
    }
    
    await client.query('COMMIT');
    console.log('批量创建成功');
  } catch (error) {
    await client.query('ROLLBACK');
    console.error('批量创建失败:', error);
    throw error;
  } finally {
    client.release();
  }
}

// ========== 读取（Read） ==========

async function readUsers() {
  // 💡 基础查询
  const result = await pool.query('SELECT * FROM users');
  console.log('所有用户:', result.rows.length);
  
  // 💡 条件查询
  const activeUsers = await pool.query(
    'SELECT * FROM users WHERE is_active = $1',
    [true]

  console.log('活跃用户:', activeUsers.rows.length);
  
  // 💡 分页
  const page = 1;
  const limit = 10;
  const offset = (page - 1) * limit;
  
  const paginatedUsers = await pool.query(
    'SELECT * FROM users ORDER BY created_at DESC LIMIT $1 OFFSET $2',
    [limit, offset]
  );
  console.log('分页结果:', paginatedUsers.rows.length);
  
  // 💡 数组查询
  const taggedUsers = await pool.query(
    "SELECT * FROM users WHERE 'nodejs' = ANY(tags)"
  );
  console.log('标签包含nodejs的用户:', taggedUsers.rows.length);
  
  // 💡 JSONB查询
  const jsonbUsers = await pool.query(

    ['Beijing']
  );
  console.log('位于北京的用户:', jsonbUsers.rows.length);
}

async function updateUser(id, updates) {

}
```

#### 第三步：LISTEN/NOTIFY实现实时通知

```javascript
// listen-notify.js
import pg from 'pg';
const { Client } = pg;

// 💡 监听器（独立连接）
async function setupListener() {
  const client = new Client({
    host: 'localhost',
    port: 5432,
    user: 'postgres',
    password: '',
    database: 'myapp'
  });
  
  await client.connect();
  
  // 💡 监听channel
  await client.query('LISTEN new_user');
  
  // 💡 接收通知
  client.on('notification', (msg) => {
    console.log('收到通知:', msg.channel, msg.payload);
    
    const data = JSON.parse(msg.payload);
    console.log('新用户:', data.username, data.email);
  });
  
  console.log('✅ 已开始监听 new_user 频道');
}

// 💡 发送通知
async function sendNotification(pool) {
  const sql = `
    INSERT INTO users (username, email, password)
    VALUES ($1, $2, $3)
    RETURNING *
  `;
  
  const result = await pool.query(sql, ['newuser', 'new@example.com', 'pass']);
  const user = result.rows[0];
  
  // 💡 发送NOTIFY
  await pool.query('NOTIFY new_user, $1', [
    JSON.stringify({
      username: user.username,
      email: user.email,
      id: user.id
    })
  ]);
  
  console.log('已发送通知');
}

// 💡 使用触发器自动发送通知
const createTriggerSQL = `
  CREATE OR REPLACE FUNCTION notify_new_user()
  RETURNS TRIGGER AS $$
  BEGIN
    PERFORM pg_notify(
      'new_user',
      json_build_object(
        'id', NEW.id,
        'username', NEW.username,
        'email', NEW.email
      )::text
    );
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  DROP TRIGGER IF EXISTS user_insert_trigger ON users;
  
  CREATE TRIGGER user_insert_trigger
  AFTER INSERT ON users
  FOR EACH ROW
  EXECUTE FUNCTION notify_new_user();
`;

// 应用场景：
// 1. 实时聊天系统（新消息通知）
// 2. 缓存失效通知
// 3. 数据变更监控
// 4. 微服务间通信
```

```javascript
  // 💡 查询嵌套JSON
  const result2 = await pool.query(`
    SELECT username, profile->'social'->>'github' as github
    FROM users
    WHERE profile->'social'->>'github' IS NOT NULL
  `);
  console.log('有GitHub的用户:', result2.rows);
  
  // 💡 JSONB数组包含查询
  const result3 = await pool.query(`
    SELECT username
    FROM users
    WHERE profile->'skills' ? 'Node.js'
  `);
  console.log('会Node.js的用户:', result3.rows);
  
  // 💡 JSONB数组包含多个值
  const result4 = await pool.query(`
    SELECT username
    FROM users
    WHERE profile->'skills' ?& ARRAY['JavaScript', 'Node.js']
  `);
  console.log('同时会JS和Node.js的用户:', result4.rows);
  
  // 💡 更新JSONB字段
  await pool.query(`
    UPDATE users
    SET profile = jsonb_set(profile, '{bio}', '"Senior Developer"')
    WHERE username = 'john'
  `);
  
  // 💡 添加JSONB字段
  await pool.query(`
    UPDATE users
    SET profile = profile || '{"website": "https://example.com"}'::jsonb
    WHERE username = 'john'
  `);
  
  // 💡 删除JSONB字段
  await pool.query(`
    UPDATE users
    SET profile = profile - 'website'
    WHERE username = 'john'
  `);
}
```

#### 第五步：全文搜索

```javascript
// full-text-search.js
import { pool } from './db.js';

// 💡 创建全文搜索表
async function setupFullTextSearch() {
  await pool.query(`
    CREATE TABLE articles (
      id SERIAL PRIMARY KEY,
      title TEXT NOT NULL,
      content TEXT NOT NULL,
      search_vector tsvector
    );
    
    -- 创建GIN索引
    CREATE INDEX idx_search_vector ON articles USING GIN (search_vector);
    
    -- 自动更新search_vector的触发器
    CREATE OR REPLACE FUNCTION update_search_vector()
    RETURNS TRIGGER AS $$
    BEGIN
      NEW.search_vector :=
        setweight(to_tsvector('english', COALESCE(NEW.title, '')), 'A') ||
        setweight(to_tsvector('english', COALESCE(NEW.content, '')), 'B');
      RETURN NEW;
    END;
    $$ LANGUAGE plpgsql;
    
    CREATE TRIGGER article_search_update
    BEFORE INSERT OR UPDATE ON articles
    FOR EACH ROW
    EXECUTE FUNCTION update_search_vector();
  `);
}

// 💡 插入文章
async function createArticle(title, content) {
  const sql = `
    INSERT INTO articles (title, content)
    VALUES ($1, $2)
    RETURNING *
  `;
  
  const result = await pool.query(sql, [title, content]);
  return result.rows[0];
}

// 💡 全文搜索
async function searchArticles(query) {
  const sql = `
    SELECT
      id,
      title,
      content,
      ts_rank(search_vector, plainto_tsquery('english', $1)) as rank
    FROM articles
    WHERE search_vector @@ plainto_tsquery('english', $1)
    ORDER BY rank DESC
    LIMIT 10
  `;
  
  const result = await pool.query(sql, [query]);
  console.log('搜索结果:', result.rows);
  return result.rows;
}

// 💡 高亮搜索结果
async function searchWithHighlight(query) {
  const sql = `
    SELECT
      id,
      ts_headline('english', title, plainto_tsquery('english', $1)) as title,
      ts_headline('english', content, plainto_tsquery('english', $1)) as content
    FROM articles
    WHERE search_vector @@ plainto_tsquery('english', $1)
  `;
  
  const result = await pool.query(sql, [query]);
  console.log('高亮结果:', result.rows);
  return result.rows;
}
```

### 错误处理：3个高频问题

#### 问题1：连接池耗尽

**错误表现**：
```
Error: timeout while waiting for slot from the pool
```

**解决方案**：
```javascript
// ✅ 确保释放client
const client = await pool.connect();
try {
  await client.query('...');
} finally {
  client.release(); // 必须
}

// ✅ 增加连接池大小
const pool = new Pool({ max: 20 });

// ✅ 监控连接池状态
console.log('总连接数:', pool.totalCount);
console.log('空闲连接:', pool.idleCount);
console.log('等待连接:', pool.waitingCount);
```

#### 问题2：JSONB查询性能慢

**错误表现**：JSONB查询耗时长

**解决方案**：
```javascript
// ✅ 创建GIN索引
await pool.query('CREATE INDEX idx_profile ON users USING GIN (profile)');

// ✅ 创建表达式索引
await pool.query(`
  CREATE INDEX idx_profile_location
  ON users ((profile->>'location'))
`);

// ✅ 使用EXPLAIN分析
const explain = await pool.query(`
  EXPLAIN ANALYZE
  SELECT * FROM users WHERE profile->>'location' = 'Beijing'
`);
console.log(explain.rows);
```

#### 问题3：锁冲突导致死锁

**错误表现**：
```
Error: deadlock detected
```

**解决方案**：
```javascript
// ✅ 按相同顺序访问资源
async function transferMoney(fromId, toId, amount) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    
    // 💡 按ID顺序锁定，避免死锁
    const ids = [fromId, toId].sort();
    
    for (const id of ids) {
      await client.query(
        'SELECT * FROM accounts WHERE id = $1 FOR UPDATE',
        [id]
      );
    }
    
    // 执行转账...
    
    await client.query('COMMIT');
  } catch (error) {
    await client.query('ROLLBACK');
    throw error;
  } finally {
    client.release();
  }
}
```

## 📋 总结

### 核心要点

1. **PostgreSQL**：功能最强大的开源关系型数据库
2. **JSONB**：高效JSON存储和查询
3. **LISTEN/NOTIFY**：实时消息推送
4. **全文搜索**：内置搜索引擎

### 实现关键

1. **创建连接池**：`new Pool(config)`
2. **执行查询**：`pool.query(sql, params)`
3. **占位符**：使用`$1, $2`而非`?`
4. **LISTEN/NOTIFY**：`LISTEN channel` + `NOTIFY channel, 'data'`
5. **JSONB查询**：`->`和`->>`操作符

### 应用价值

- **功能强大**：JSON、全文搜索、数组等
- **性能优异**：JSONB、GIN索引
- **实时通知**：LISTEN/NOTIFY
- **SQL标准**：高度兼容

### 面试重点

1. **PostgreSQL vs MySQL？**
   - PostgreSQL功能更强（JSON、全文搜索、窗口函数）

2. **LISTEN/NOTIFY的应用场景？**
   - 实时通知、缓存失效、微服务通信

3. **JSONB如何优化查询？**
   - 创建GIN索引、表达式索引

4. **如何避免死锁？**
   - 按相同顺序访问资源、缩短事务时间

### 注意事项

⚠️ **占位符使用$1, $2**：
```javascript
// PostgreSQL使用$1, $2，而不是?
await pool.query('SELECT * FROM users WHERE id = $1', [userId]);
```

⚠️ **JSONB性能依赖索引**：
```javascript
// 创建GIN索引
CREATE INDEX idx_profile ON users USING GIN (profile);
```
