## 📖 知识点概述

### 核心概念

**mysql2**是Node.js中最流行的MySQL客户端库（GitHub 4K+ stars），相比mysql库提供更好的性能和Promise支持。mysql2 3.x版本（2024+）带来了ES Module支持和性能改进。

### 解决的问题

- **异步操作**：提供Promise和async/await支持
- **连接池管理**：自动管理数据库连接
- **防SQL注入**：参数化查询
- **性能优化**：预编译语句、连接复用
- **类型转换**：自动处理MySQL数据类型

### 主要特点

1. **Promise支持**：原生支持async/await
2. **连接池**：自动管理连接生命周期
3. **预编译**：提升重复查询性能
4. **流式查询**：处理大量数据
5. **完全兼容**：兼容mysql库API

### 典型应用场景

- **关系型数据**：用户、订单、商品
- **事务处理**：转账、库存扣减
- **复杂查询**：JOIN、子查询、聚合
- **数据分析**：统计报表

#### 第二步：连接池（推荐）

```javascript
// db.js - 连接池
import mysql from 'mysql2/promise';

// 💡 创建连接池
export const pool = mysql.createPool({
  // 基础配置
  host: process.env.DB_HOST || 'localhost',
  port: process.env.DB_PORT || 3306,
  user: process.env.DB_USER || 'root',
  password: process.env.DB_PASSWORD || '',
  database: process.env.DB_NAME || 'myapp',
  
  // 💡 连接池配置
  connectionLimit: 10, // 最大连接数
  queueLimit: 0, // 队列限制（0=无限制）
  waitForConnections: true, // 等待可用连接
  
  // 💡 性能优化
  enableKeepAlive: true, // 保持连接活跃
  keepAliveInitialDelay: 0, // 初始延迟
  
  // 💡 字符集
  charset: 'utf8mb4',
  
  // 💡 时区
  timezone: '+08:00',
  
  // 💡 日期处理
  dateStrings: false, // 返回Date对象而非字符串
  
  // 💡 调试
  debug: false
});

// 💡 测试连接
pool.getConnection()
  .then(connection => {
    console.log('✅ MySQL连接池已创建');
    connection.release();
  })
  .catch(error => {
    console.error('❌ MySQL连接失败:', error);
  });

// 💡 监听错误
pool.on('error', (error) => {
  console.error('MySQL Pool Error:', error);
});

// 💡 优雅关闭
process.on('SIGINT', async () => {
  await pool.end();
  console.log('MySQL连接池已关闭');
  process.exit(0);
});
```

```javascript
import { pool } from './db.js';
```

```javascript
async function unsafeQuery(username) {

// mysql2会自动转义参数：
```

#### 第五步：事务处理

```javascript
// transaction.js
import { pool } from './db.js';

// 💡 基础事务
async function basicTransaction() {
  const connection = await pool.getConnection();
  
  try {
    // 开始事务
    await connection.beginTransaction();
    
    // 操作1：扣除用户A余额
    await connection.query(
      'UPDATE users SET balance = balance - ? WHERE id = ?',
      [100, 1]
    );
    
    // 操作2：增加用户B余额
    await connection.query(
      'UPDATE users SET balance = balance + ? WHERE id = ?',
      [100, 2]
    );
    
    // 提交事务
    await connection.commit();
    console.log('转账成功');
    
  } catch (error) {
    // 回滚事务
    await connection.rollback();
    console.error('转账失败，已回滚:', error);
    throw error;
  } finally {
    // 释放连接
    connection.release();
  }
}

// 💡 复杂事务：订单创建
async function createOrder(userId, items) {
  const connection = await pool.getConnection();
  
  try {
    await connection.beginTransaction();
    
    // 1. 创建订单
    const [orderResult] = await connection.query(
      'INSERT INTO orders (user_id, total_amount, status) VALUES (?, ?, ?)',
      [userId, calculateTotal(items), 'pending']
    );
    
    const orderId = orderResult.insertId;
    
    // 2. 插入订单明细
    for (const item of items) {
      await connection.query(
        'INSERT INTO order_items (order_id, product_id, quantity, price) VALUES (?, ?, ?, ?)',
        [orderId, item.productId, item.quantity, item.price]
      );
      
      // 3. 减少库存
      const [updateResult] = await connection.query(
        'UPDATE products SET stock = stock - ? WHERE id = ? AND stock >= ?',
        [item.quantity, item.productId, item.quantity]
      );
      
      // 检查库存是否充足
      if (updateResult.affectedRows === 0) {
        throw new Error(`商品 ${item.productId} 库存不足`);
      }
    }
    
    // 4. 提交事务
    await connection.commit();
    console.log('订单创建成功，ID:', orderId);
    
    return { orderId };
    
  } catch (error) {
    await connection.rollback();
    console.error('订单创建失败:', error.message);
    throw error;
  } finally {
    connection.release();
  }
}

function calculateTotal(items) {
  return items.reduce((sum, item) => sum + item.price * item.quantity, 0);
}

// 💡 事务隔离级别
async function setIsolationLevel() {
  const connection = await pool.getConnection();
  
  try {
    // 设置隔离级别
    await connection.query('SET TRANSACTION ISOLATION LEVEL READ COMMITTED');
    
    await connection.beginTransaction();
    
    // 执行事务...
    
    await connection.commit();
  } finally {
    connection.release();
  }
}

// 💡 可用的隔离级别：
// - READ UNCOMMITTED（读未提交，最低级别，可能脏读）
// - READ COMMITTED（读已提交，防止脏读）
// - REPEATABLE READ（可重复读，MySQL默认，防止不可重复读）
// - SERIALIZABLE（串行化，最高级别，防止幻读）
```

### 错误处理：3个高频问题

#### 问题1：ER_TOO_MANY_CONNECTIONS（连接数过多）

**错误表现**：
```
Error: ER_TOO_MANY_CONNECTIONS: Too many connections
```

**原因**：连接数超过MySQL限制或连接未正确释放

**解决方案**：
```javascript
// ✅ 调整连接池大小
const pool = mysql.createPool({
  connectionLimit: 10, // 根据实际情况调整
  queueLimit: 0
});

// ✅ 确保连接释放
async function query() {
  const connection = await pool.getConnection();
  try {
    const [rows] = await connection.query('SELECT * FROM users');
    return rows;
  } finally {
    connection.release(); // 必须释放
  }
}

// ✅ 增加MySQL最大连接数（my.cnf）
// max_connections = 200
```

#### 问题2：ER_LOCK_WAIT_TIMEOUT（锁等待超时）

**错误表现**：
```
Error: ER_LOCK_WAIT_TIMEOUT: Lock wait timeout exceeded
```

**原因**：事务持有锁时间过长，其他事务等待超时

**解决方案**：
```javascript
// ✅ 缩短事务时间
async function optimizedTransaction() {
  const connection = await pool.getConnection();
  try {
    await connection.beginTransaction();
    
    // 只在事务中执行必要的更新操作
    await connection.query('UPDATE users SET balance = balance - 100 WHERE id = 1');
    
    await connection.commit();
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}

// ✅ 调整锁等待超时时间（my.cnf）
// innodb_lock_wait_timeout = 120

// ✅ 避免死锁：按相同顺序访问资源
```

## 📋 总结

### 核心要点

1. **mysql2**：高性能MySQL客户端库
2. **连接池**：复用连接，提升性能
3. **防注入**：使用占位符？而非拼接SQL
4. **事务**：保证数据一致性

### 实现关键

1. **创建连接池**：`mysql.createPool(config)`
2. **执行查询**：`pool.query(sql, params)`
3. **防注入**：使用`?`占位符
4. **事务处理**：`beginTransaction() + commit/rollback`
5. **释放连接**：`connection.release()`

### 应用价值

- **性能**：连接池、预编译语句
- **安全**：防SQL注入
- **可靠**：事务保证一致性
- **易用**：Promise/async-await

### 面试重点

1. **如何防止SQL注入？**
   - 使用参数化查询（占位符？）

2. **连接池的作用？**
   - 复用连接、减少创建开销、限制并发数

3. **事务的ACID特性？**
   - 原子性、一致性、隔离性、持久性

4. **如何优化MySQL查询性能？**
   - 添加索引、避免全表扫描、使用EXPLAIN分析

### 注意事项

⚠️ **必须释放连接**：
```javascript
const connection = await pool.getConnection();
try {
  // 查询...
} finally {
  connection.release(); // 必须
}
```

⚠️ **不要拼接SQL**：
```javascript
// ❌ 危险
const sql = `SELECT * FROM users WHERE id = ${id}`;

// ✅ 安全
const sql = 'SELECT * FROM users WHERE id = ?';
await pool.query(sql, [id]);
```
