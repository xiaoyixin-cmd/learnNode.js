- **错误优先**：所有异步操作都要捕获错误
- **安全优先**：密码加密、参数验证、SQL 注入防护

**`src/middlewares/errorHandler.js`**

```javascript
const logger = require('../utils/logger');
const { AppError } = require('../utils/errors');

/**
 * 全局错误处理中间件
 */
const errorHandler = (err, req, res, next) => {
  // 💡 如果是操作错误（预期错误），直接返回
  if (err.isOperational) {
    logger.warn({
      message: err.message,
      statusCode: err.statusCode,
      path: req.path,
      method: req.method
    });

    return res.status(err.statusCode).json({
      success: false,
      message: err.message,
      errors: err.errors || null,
      timestamp: new Date().toISOString()
    });
  }

  // 💡 如果是 Mongoose 验证错误
  if (err.name === 'ValidationError') {
    const errors = Object.values(err.errors).map(e => ({
      field: e.path,
      message: e.message
    }));

    return res.status(400).json({
      success: false,
      message: 'Validation failed',
      errors,
      timestamp: new Date().toISOString()
    });
  }

  // 💡 如果是 Mongoose 重复键错误
  if (err.code === 11000) {
    const field = Object.keys(err.keyPattern)[0];
    return res.status(409).json({
      success: false,
      message: `${field} already exists`,
      timestamp: new Date().toISOString()
    });
  }

  // 💡 如果是 JWT 错误
  if (err.name === 'JsonWebTokenError') {
    return res.status(401).json({
      success: false,
      message: 'Invalid token',
      timestamp: new Date().toISOString()
    });
  }

  if (err.name === 'TokenExpiredError') {
    return res.status(401).json({
      success: false,
      message: 'Token expired',
      timestamp: new Date().toISOString()
    });
  }

  // 💡 未知错误（程序错误）- 记录详细日志
  logger.error({
    message: err.message,
    stack: err.stack,
    path: req.path,
    method: req.method,
    body: req.body,
    params: req.params,
    query: req.query
  });

  // 💡 生产环境不泄露错误详情
  const message = process.env.NODE_ENV === 'production'
    ? 'Internal server error'
    : err.message;

  return res.status(500).json({
    success: false,
    message,
    timestamp: new Date().toISOString()
  });
};

module.exports = errorHandler;
```

##### 3️⃣ 认证服务（Service）

**`src/services/authService.js`**

```javascript
const jwt = require('jsonwebtoken');
const User = require('../models/User');
const redis = require('../config/redis');
const { 
  AuthenticationError, 
  ConflictError,
  NotFoundError 
} = require('../utils/errors');

/**
 * 用户注册
 */
const register = async ({ email, password, username }) => {
  // 检查邮箱是否已存在
  const existingUser = await User.findOne({ email });
  if (existingUser) {
    throw new ConflictError('Email already exists');
  }

  // 检查用户名是否已存在
  const existingUsername = await User.findOne({ username });
  if (existingUsername) {
    throw new ConflictError('Username already exists');
  }

  // 创建用户
  const user = await User.create({
    email,
    password,
    username
  });

  return user;
};

/**
 * 用户登录
 */
const login = async ({ email, password }) => {
  // 查找用户（包含密码字段）
  const user = await User.findOne({ email }).select('+password');
  
  if (!user) {
    throw new AuthenticationError('Invalid email or password');
  }

  // 检查账号状态
  if (!user.isActive) {
    throw new AuthenticationError('Account is disabled');
  }

  // 验证密码
  const isMatch = await user.comparePassword(password);
  if (!isMatch) {
    throw new AuthenticationError('Invalid email or password');
  }

  // 更新最后登录时间
  user.lastLoginAt = new Date();
  await user.save();

  // 生成 Token
  const accessToken = generateAccessToken(user);
  const refreshToken = generateRefreshToken(user);

  // 将 Refresh Token 存储到 Redis（用于后续刷新）
  await redis.setex(
    `refresh_token:${user._id}`,
    7 * 24 * 60 * 60,  // 7 天
    refreshToken
  );

  return {
    user,
    accessToken,
    refreshToken
  };
};

/**
 * 刷新 Access Token
 */
const refreshAccessToken = async (refreshToken) => {
  try {
    // 验证 Refresh Token
    const payload = jwt.verify(refreshToken, process.env.JWT_REFRESH_SECRET);
    
    // 检查 Redis 中是否存在
    const storedToken = await redis.get(`refresh_token:${payload.id}`);
    if (storedToken !== refreshToken) {
      throw new AuthenticationError('Invalid refresh token');
    }

    // 查找用户
    const user = await User.findById(payload.id);
    if (!user || !user.isActive) {
      throw new AuthenticationError('User not found or inactive');
    }

    // 生成新的 Access Token
    const accessToken = generateAccessToken(user);

    return { accessToken };
    
  } catch (error) {
    throw new AuthenticationError('Invalid refresh token');
  }
};

/**
 * 登出（将 Refresh Token 加入黑名单）
 */
const logout = async (userId, refreshToken) => {
  // 删除 Redis 中的 Refresh Token
  await redis.del(`refresh_token:${userId}`);
  
  // 将 Refresh Token 加入黑名单（可选）
  const payload = jwt.decode(refreshToken);
  const ttl = payload.exp - Math.floor(Date.now() / 1000);
  
  if (ttl > 0) {
    await redis.setex(`blacklist:${refreshToken}`, ttl, '1');
  }
};

/**
 * 生成 Access Token（短期，15分钟）
 */
function generateAccessToken(user) {
  return jwt.sign(
    user.toJWT(),
    process.env.JWT_SECRET,
    { expiresIn: '15m' }
  );
}

/**
 * 生成 Refresh Token（长期，7天）
 */
function generateRefreshToken(user) {
  return jwt.sign(
    { id: user._id },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: '7d' }
  );
}

module.exports = {
  register,
  login,
  refreshAccessToken,
  logout
};
```

##### 5️⃣ 认证中间件

**`src/middlewares/auth.js`**

```javascript
const jwt = require('jsonwebtoken');
const redis = require('../config/redis');
const { AuthenticationError, AuthorizationError } = require('../utils/errors');

/**
 * JWT 认证中间件
 */
const authenticate = async (req, res, next) => {
  try {
    // 1️⃣ 从 Header 提取 Token
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      throw new AuthenticationError('No token provided');
    }

    const token = authHeader.split(' ')[1];

    // 2️⃣ 检查 Token 是否在黑名单中
    const isBlacklisted = await redis.exists(`blacklist:${token}`);
    if (isBlacklisted) {
      throw new AuthenticationError('Token has been revoked');
    }

    // 3️⃣ 验证 Token
    const payload = jwt.verify(token, process.env.JWT_SECRET);

    // 4️⃣ 将用户信息挂载到 req.user
    req.user = payload;

    next();
    
  } catch (error) {
    if (error.name === 'JsonWebTokenError') {
      return next(new AuthenticationError('Invalid token'));
    }
    if (error.name === 'TokenExpiredError') {
      return next(new AuthenticationError('Token expired'));
    }
    next(error);
  }
};

/**
 * 角色权限中间件（工厂函数）
 */
const authorize = (...allowedRoles) => {
  return (req, res, next) => {
    if (!req.user) {
      return next(new AuthenticationError('Authentication required'));
    }

    if (!allowedRoles.includes(req.user.role)) {
      return next(new AuthorizationError(`Access denied. Required roles: ${allowedRoles.join(', ')}`));
    }

    next();
  };
};

/**
 * 资源所有权验证中间件（工厂函数）
 */
const authorizeOwner = (Model, paramKey = 'id') => {
  return async (req, res, next) => {
    try {
      const resourceId = req.params[paramKey];
      const resource = await Model.findById(resourceId);

      if (!resource) {
        throw new NotFoundError(Model.modelName);
      }

      // 检查是否是资源所有者或管理员
      const isOwner = resource.author?.toString() === req.user.id;
      const isAdmin = req.user.role === 'admin';

      if (!isOwner && !isAdmin) {
        throw new AuthorizationError('You do not have permission to modify this resource');
      }

      // 将资源挂载到 req 对象上（避免重复查询）
      req.resource = resource;

      next();
      
    } catch (error) {
      next(error);
    }
  };
};

module.exports = {
  authenticate,
  authorize,
  authorizeOwner
};
```

```javascript
    enum: ['技术', '生活', '随笔', '教程'],
    default: '技术'
  },
  tags: [{
    type: String,
    trim: true
  }],
  coverImage: {
    type: String,
    default: ''
  },
  status: {
    type: String,
    enum: ['draft', 'published', 'archived'],
    default: 'draft'
  },
  stats: {
    viewsCount: { type: Number, default: 0 },
    likesCount: { type: Number, default: 0 },
    commentsCount: { type: Number, default: 0 }
  },
  publishedAt: Date
}, {
  timestamps: true
});

// 💡 索引
PostSchema.index({ author: 1, createdAt: -1 });
PostSchema.index({ status: 1, publishedAt: -1 });
PostSchema.index({ tags: 1 });
PostSchema.index({ title: 'text', content: 'text' });  // 全文索引

// 💡 保存前自动生成 slug
PostSchema.pre('save', async function(next) {
  if (this.isModified('title')) {
    this.slug = slugify(this.title, { lower: true, strict: true });
    
    // 处理重复 slug（添加数字后缀）
    const existingPost = await this.constructor.findOne({ slug: this.slug });
    if (existingPost && existingPost._id.toString() !== this._id.toString()) {
      this.slug = `${this.slug}-${Date.now()}`;
    }
  }
  
  next();
});

// 💡 发布时自动设置发布时间
PostSchema.pre('save', function(next) {
  if (this.isModified('status') && this.status === 'published' && !this.publishedAt) {
    this.publishedAt = new Date();
  }
  next();
});

module.exports = mongoose.model('Post', PostSchema);
```

```javascript
const Post = require('../models/Post');
const User = require('../models/User');
const redis = require('../config/redis');
const { NotFoundError } = require('../utils/errors');

/**
 * 获取文章列表（分页、排序、搜索、过滤）
 */
const getPosts = async (query) => {
  const {
    page = 1,
    limit = 10,
    sortBy = 'createdAt',
    order = 'desc',
    status = 'published',
    category,
    tags,
    search,
    author
  } = query;

  // 构建查询条件
  const filter = {};
  
  if (status) filter.status = status;
  if (category) filter.category = category;
  if (tags) filter.tags = { $in: tags.split(',') };
  if (author) filter.author = author;
  
  // 全文搜索
  if (search) {
    filter.$text = { $search: search };
  }

  // 计算分页
  const skip = (page - 1) * limit;

  // 构建排序
  const sort = { [sortBy]: order === 'desc' ? -1 : 1 };

  // 查询数据
  const [posts, total] = await Promise.all([
    Post.find(filter)
      .populate('author', 'username avatar')
      .sort(sort)
      .skip(skip)
      .limit(parseInt(limit))
      .lean(),  // 💡 返回普通 JS 对象（性能更好）
    Post.countDocuments(filter)
  ]);

  return {
    posts,
    pagination: {
      page: parseInt(page),
      limit: parseInt(limit),
      total
    }
  };
};

/**
 * 获取文章详情（带缓存）
 */
const getPostById = async (id) => {
  // 1️⃣ 尝试从 Redis 获取缓存
  const cacheKey = `post:${id}`;
  const cached = await redis.get(cacheKey);
  
  if (cached) {
    return JSON.parse(cached);
  }

  // 2️⃣ 查询数据库
  const post = await Post.findById(id)
    .populate('author', 'username avatar bio')
    .lean();

  if (!post) {
    throw new NotFoundError('Post');
  }

  // 3️⃣ 存入 Redis（1小时过期）
  await redis.setex(cacheKey, 3600, JSON.stringify(post));

  // 4️⃣ 异步增加浏览量（不阻塞响应）
  Post.findByIdAndUpdate(id, { $inc: { 'stats.viewsCount': 1 } }).exec();

  return post;
};

/**
 * 创建文章
 */
const createPost = async (userId, data) => {
  const post = await Post.create({
    ...data,
    author: userId
  });

  // 更新用户文章数
  await User.findByIdAndUpdate(userId, {
    $inc: { 'stats.postsCount': 1 }
  });

  return post;
};

/**
 * 更新文章
 */
const updatePost = async (postId, data) => {
  const post = await Post.findByIdAndUpdate(
    postId,
    { $set: data },
    { new: true, runValidators: true }
  ).populate('author', 'username avatar');

  if (!post) {
    throw new NotFoundError('Post');
  }

  // 清除缓存
  await redis.del(`post:${postId}`);

  return post;
};

/**
 * 删除文章
 */
const deletePost = async (postId, userId) => {
  const post = await Post.findByIdAndDelete(postId);

  if (!post) {
    throw new NotFoundError('Post');
  }

  // 更新用户文章数
  await User.findByIdAndUpdate(userId, {
    $inc: { 'stats.postsCount': -1 }
  });

  // 清除缓存
  await redis.del(`post:${postId}`);

  return post;
};

module.exports = {
  getPosts,
  getPostById,
  createPost,
  updatePost,
  deletePost
};
```

```javascript
const multer = require('multer');
const path = require('path');
const { ValidationError } = require('../utils/errors');

// 💡 配置存储
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, process.env.UPLOAD_DIR || 'uploads/');
  },
  filename: (req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
  }
});

// 💡 文件过滤器（仅允许图片）
const fileFilter = (req, file, cb) => {
  const allowedTypes = /jpeg|jpg|png|gif|webp/;
  const extname = allowedTypes.test(path.extname(file.originalname).toLowerCase());
  const mimetype = allowedTypes.test(file.mimetype);

  if (mimetype && extname) {
    return cb(null, true);
  } else {
    cb(new ValidationError('Only image files are allowed (jpeg, jpg, png, gif, webp)'));
  }
};

// 💡 创建 Multer 实例
const upload = multer({
  storage,
  fileFilter,
  limits: {
    fileSize: parseInt(process.env.MAX_FILE_SIZE) || 5 * 1024 * 1024  // 默认 5MB
  }
});
```

## 3个高频错误

### 错误1：密码未加密就存入数据库

**现象**：数据库中密码是明文

**原因**：忘记添加 `pre('save')` Hook

**解决方案**：

```javascript
// ✅ 确保 UserSchema 包含加密 Hook
UserSchema.pre('save', async function(next) {
  if (!this.isModified('password')) return next();
  
  const salt = await bcrypt.genSalt(10);
  this.password = await bcrypt.hash(this.password, salt);
  next();
});
```

---

### 错误2：忘记在路由中添加认证中间件

**现象**：未登录用户也能创建文章

**原因**：路由缺少 `authenticate` 中间件

**解决方案**：

```javascript
// ❌ 错误
router.post('/posts', postController.createPost);

// ✅ 正确
router.post('/posts', authenticate, postController.createPost);
```

---

### 错误3：异步错误未捕获导致服务器崩溃

**现象**：Promise rejection 未处理，服务器退出

**解决方案**：

```javascript
// ❌ 错误：未捕获异步错误
router.post('/posts', async (req, res) => {
  const post = await Post.create(req.body);  // 如果失败，会导致 UnhandledPromiseRejection
  res.json({ post });
});

// ✅ 正确：使用 try-catch 或传递给 next
router.post('/posts', async (req, res, next) => {
  try {
    const post = await Post.create(req.body);
    res.json({ post });
  } catch (error) {
    next(error);  // 传递给错误处理中间件
  }
});
```

## 🤔 常见问题（FAQ）

### Q1：为什么要分 Controller 和 Service 层？

**A**：职责分离，提升代码可维护性：

```javascript
// ❌ 不分层：所有逻辑都在 Controller（难测试、难复用）
router.post('/posts', async (req, res) => {
  const { title, content } = req.body;
  if (!title) return res.status(400).json({ error: '标题必填' });
  const post = await Post.create({ title, content, authorId: req.user.id });
  res.json({ post });
});

// ✅ 分层：Controller 处理 HTTP，Service 处理业务逻辑（易测试、易复用）
// Controller
router.post('/posts', async (req, res) => {
  const post = await postService.createPost(req.body, req.user.id);
  res.json({ post });
});

// Service
async createPost(data, authorId) {
  if (!data.title) throw new Error('标题必填');
  return await Post.create({ ...data, authorId });
}
```

**优势**：Service 可被 Controller、定时任务、Worker 复用。

### Q2：JWT 认证中间件如何工作？

**A**：提取 Token → 验证签名 → 解析用户信息：

```javascript
// 认证中间件执行流程
1. 从 Header 中提取 Token
   Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...

2. jwt.verify() 验证签名
   - 签名正确 → 解析 payload { userId: '123' }
   - 签名错误/过期 → 抛出异常

3. 根据 userId 查询用户
   const user = await User.findById(userId);

4. 将 user 挂载到 req.user
   req.user = user;

5. 调用 next() 进入下一个中间件/Controller
```

### Q3：为什么密码比对用 `bcrypt.compare` 而不是直接对比？

**A**：bcrypt 每次生成的哈希都不同（含随机盐）：

```javascript
const password = '123456';

// 哈希 2 次，结果不同
const hash1 = await bcrypt.hash(password, 10);
const hash2 = await bcrypt.hash(password, 10);

console.log(hash1 === hash2); // false

// 但 compare 都返回 true
console.log(await bcrypt.compare(password, hash1)); // true
console.log(await bcrypt.compare(password, hash2)); // true
```

**原因**：bcrypt 在哈希中包含了盐（salt），`compare` 会自动提取盐进行比对。

### Q4：如何防止 N+1 查询问题？

**A**：使用 `populate()` 或提前加载：

```javascript
// ❌ N+1 问题（查询 100 篇文章 = 1 + 100 次数据库查询）
const posts = await Post.find();
for (const post of posts) {
  post.author = await User.findById(post.authorId);  // 每个 post 都查一次
}

// ✅ 使用 populate（只需 2 次查询）
const posts = await Post.find().populate('author');  // Mongoose 自动 JOIN
```

**性能对比**：
- N+1 查询：101 次数据库请求（100ms × 101 = 10秒）
- populate：2 次数据库请求（100ms × 2 = 200ms）
