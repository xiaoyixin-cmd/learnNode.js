#### 优化2：N+1 查询问题

**问题**：文章列表加载作者信息时产生 N+1 查询

```javascript
// ❌ N+1 查询
const posts = await Post.find({ status: 'published' }).limit(10);

// 🐌 循环查询作者（10 篇文章 = 10 次查询）
for (const post of posts) {
  post.author = await User.findById(post.author);
}

// 总查询数：1（文章列表）+ 10（作者）= 11 次
```

**解决方案**：使用 `.populate()` 或手动 JOIN

```javascript
// ✅ 方案1：使用 populate（自动关联查询）
const posts = await Post.find({ status: 'published' })
  .limit(10)
  .populate('author', 'username avatar');

// 总查询数：2 次（1次文章 + 1次作者批量查询）

// ✅ 方案2：手动批量查询（性能更好）
const posts = await Post.find({ status: 'published' }).limit(10).lean();
const authorIds = posts.map(p => p.author);
const authors = await User.find({ _id: { $in: authorIds } }).lean();

const authorMap = new Map(authors.map(a => [a._id.toString(), a]));
posts.forEach(p => {
  p.author = authorMap.get(p.author.toString());
});
```

**性能对比**：

| 方案             | 查询次数 | 响应时间 |
|------------------|----------|----------|
| N+1 查询         | 11 次    | 450ms    |
| populate         | 2 次     | 80ms     |
| 手动批量查询     | 2 次     | 50ms     |

```javascript
const redis = require('../config/redis');

/**
 * 获取文章详情（带缓存）
 */
async function getPostById(id) {
  const cacheKey = `post:${id}`;

  // 1️⃣ 尝试从 Redis 获取
  const cached = await redis.get(cacheKey);
  if (cached) {
    console.log('✅ Cache HIT:', cacheKey);
    return JSON.parse(cached);
  }

  console.log('❌ Cache MISS:', cacheKey);

  // 2️⃣ 从数据库查询
  const post = await Post.findById(id)
    .populate('author', 'username avatar')
    .lean();

  if (!post) {
    throw new NotFoundError('Post');
  }

  // 3️⃣ 写入 Redis（1小时过期）
  await redis.setex(cacheKey, 3600, JSON.stringify(post));

  // 4️⃣ 异步更新浏览量（不阻塞响应）
  Post.findByIdAndUpdate(id, { $inc: { 'stats.viewsCount': 1 } }).exec();

  return post;
}

/**
 * 更新文章时清除缓存
 */
async function updatePost(postId, data) {
  const post = await Post.findByIdAndUpdate(
    postId,
    { $set: data },
    { new: true, runValidators: true }
  ).populate('author', 'username avatar');

  // 💡 清除缓存
  await redis.del(`post:${postId}`);

  return post;
}
```

#### 优化4：文章列表分页缓存

**问题**：文章列表每次都查询数据库

**解决方案**：缓存分页结果

```javascript
async function getPosts(query) {
  const {
    page = 1,
    limit = 10,
    sortBy = 'createdAt',
    order = 'desc',
    category,
    tags
  } = query;

  // 1️⃣ 生成缓存键（包含查询参数）
  const cacheKey = `posts:${page}:${limit}:${sortBy}:${order}:${category || ''}:${tags || ''}`;

  // 2️⃣ 尝试从 Redis 获取
  const cached = await redis.get(cacheKey);
  if (cached) {
    return JSON.parse(cached);
  }

  // 3️⃣ 查询数据库
  const filter = { status: 'published' };
  if (category) filter.category = category;
  if (tags) filter.tags = { $in: tags.split(',') };

  const skip = (page - 1) * limit;
  const sort = { [sortBy]: order === 'desc' ? -1 : 1 };

  const [posts, total] = await Promise.all([
    Post.find(filter)
      .populate('author', 'username avatar')
      .sort(sort)
      .skip(skip)
      .limit(parseInt(limit))
      .lean(),
    Post.countDocuments(filter)
  ]);

  const result = {
    posts,
    pagination: {
      page: parseInt(page),
      limit: parseInt(limit),
      total
    }
  };

  // 4️⃣ 缓存结果（5分钟过期）
  await redis.setex(cacheKey, 300, JSON.stringify(result));

  return result;
}
```

**缓存失效策略**：

```javascript
/**
 * 发布新文章时清除列表缓存
 */
async function createPost(userId, data) {
  const post = await Post.create({
    ...data,
    author: userId
  });

  // 💡 清除所有列表缓存（使用模式匹配）

  if (keys.length > 0) {
    await redis.del(...keys);
  }

  return post;
}
```

## 3个高频错误

### 错误1：缓存击穿（热点 Key 过期瞬间压垮数据库）

**现象**：某热门文章缓存过期，瞬间 1000+ 请求同时查询数据库

**解决方案**：使用分布式锁

```javascript
const redis = require('../config/redis');

async function getPostById(id) {
  const cacheKey = `post:${id}`;
  const lockKey = `lock:${cacheKey}`;

  // 1️⃣ 尝试从缓存获取
  let post = await redis.get(cacheKey);
  if (post) {
    return JSON.parse(post);
  }

  // 2️⃣ 尝试获取分布式锁（只有一个请求能获取到）
  const acquired = await redis.set(lockKey, '1', 'EX', 10, 'NX');

  if (acquired === 'OK') {
    try {
      // 3️⃣ 获取锁成功，查询数据库
      post = await Post.findById(id).populate('author').lean();
      
      if (post) {
        await redis.setex(cacheKey, 3600, JSON.stringify(post));
      }
      
      return post;
      
    } finally {
      // 4️⃣ 释放锁
      await redis.del(lockKey);
    }
    
  } else {
    // 5️⃣ 获取锁失败，等待100ms后重试
    await new Promise(resolve => setTimeout(resolve, 100));
    return getPostById(id);  // 递归重试
  }
}
```

### 错误2：缓存雪崩（大量 Key 同时过期）

**现象**：设置了相同的过期时间（如1小时），大量缓存同时失效

**解决方案**：添加随机过期时间

```javascript
// ❌ 错误：所有缓存1小时过期
await redis.setex(cacheKey, 3600, JSON.stringify(data));

// ✅ 正确：添加随机过期时间（3600 ± 300秒）
const ttl = 3600 + Math.floor(Math.random() * 600) - 300;
await redis.setex(cacheKey, ttl, JSON.stringify(data));
```

### 错误3：索引未生效导致全表扫描

**现象**：明明添加了索引，但查询仍然很慢

**原因**：查询条件与索引不匹配

```javascript
// 索引定义

// ❌ 错误：查询条件未包含索引首字段

// ❌ 错误：排序方向与索引不一致
db.posts.find({ status: 'published' }).sort({ createdAt: 1 })  // 部分使用索引

// ✅ 正确：完全匹配索引
db.posts.find({ status: 'published' }).sort({ createdAt: -1 })  // 完全使用索引
```

### Q2：为什么添加缓存后 QPS 反而下降了？

**A**：可能是缓存命中率低或 Redis 网络延迟：

**原因1：缓存命中率低**
```javascript
// 问题：缓存 Key 设置错误，每次都 Miss
const key = `post:${Math.random()}`;  // 每次都不同 ❌

// 解决：使用固定 Key
const key = `posts:latest`;  // 缓存命中 ✅
```

**原因2：Redis 网络延迟**
```javascript
// 问题：每个请求都多次访问 Redis
for (const post of posts) {
  await redis.get(`post:${post.id}`);  // N 次网络往返
}

// 解决：批量获取（Pipeline）
const keys = posts.map(p => `post:${p.id}`);
const values = await redis.mget(keys);  // 1 次网络往返
```

**性能对比**：
- 无缓存：QPS = 500
- 缓存命中率 10%：QPS = 450（反而下降）
- 缓存命中率 80%：QPS = 2000（提升 4 倍）

### Q3：PM2 Cluster 模式下如何共享缓存？

**A**：使用 Redis 而非内存缓存：

**问题**：
```javascript
// ❌ 内存缓存在 Cluster 模式下不共享
const cache = new NodeCache();  // 每个 Worker 独立缓存

// Worker 1 缓存了数据
cache.set('posts', data);

// Worker 2 读取 → 缓存未命中（不同进程）
cache.get('posts');  // undefined
```

**解决方案**：
```javascript
// ✅ 使用 Redis（所有 Worker 共享）
const redis = require('ioredis');
const client = new Redis();

// Worker 1 缓存
await client.set('posts', JSON.stringify(data));

// Worker 2 读取 → 缓存命中
const cached = await client.get('posts');  // 能读到数据
```

```bash
# 结果：QPS = 3000, P99 = 50ms

# 步骤3：测试 100 并发
autocannon -c 100 -d 30 http://localhost:3000
# 结果：QPS = 4000, P99 = 200ms  ← 开始变慢

# 步骤4：测试 200 并发
autocannon -c 200 -d 30 http://localhost:3000
# 结果：QPS = 4100, P99 = 500ms  ← QPS 不再增长，延迟上升

# 结论：最佳并发数 = 100（拐点）
```

**拐点判断**：
- QPS 不再增长
- P99 延迟开始急剧上升
- 错误率 > 1%
