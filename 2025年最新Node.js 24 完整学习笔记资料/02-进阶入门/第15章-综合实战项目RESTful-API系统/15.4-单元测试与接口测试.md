# 15.4 单元测试与接口测试

## 知识点概述

### 核心概念

测试是保障代码质量的最后一道防线。完善的测试能让你自信地重构代码、快速定位 Bug、避免回归问题。本节将实现：

- **单元测试**：测试单个函数/模块的功能（Service 层）
- **集成测试**：测试 API 接口的完整流程（Controller + Service + Model）
- **测试覆盖率**：衡量测试质量的指标（目标 80%+）
- **测试数据管理**：Fixture 工厂、数据库清理、Mock 策略

### 测试金字塔

```
        /\
       /  \  E2E 测试（10%）- 端到端测试
      /    \
     /------\ 集成测试（20%）- API 接口测试
    /        \
   /----------\ 单元测试（70%）- 函数/模块测试
  /--------------\
```

- **单元测试**：快速、隔离、大量
- **集成测试**：验证模块协作、数据库交互
- **E2E 测试**：模拟真实用户操作（本章不涉及）

## 使用指南

### 第一步：理解 - 为什么需要测试？

#### 没有测试的痛点

```javascript
// 修改了 userService.register() 逻辑
// ❌ 不知道是否破坏了其他功能
// ❌ 需要手动测试所有相关接口
// ❌ 上线后才发现问题
```

#### 有测试的优势

```javascript
// 修改代码后，运行测试
npm test

// ✅ 30 秒内测试所有功能
// ✅ 自动发现破坏性改动
// ✅ 重构更有信心
```

```javascript
afterAll(async () => {
```

```javascript
const authService = require('../../../src/services/authService');
const User = require('../../../src/models/User');
const { ConflictError } = require('../../../src/utils/errors');

describe('AuthService', () => {
  describe('register', () => {
    const validUserData = {
      email: 'test@example.com',
      password: 'Test123456',
      username: 'testuser'
    };

    test('应该成功注册新用户', async () => {
      // 执行
      const user = await authService.register(validUserData);

      // 断言
      expect(user).toBeDefined();
      expect(user.email).toBe(validUserData.email);
      expect(user.username).toBe(validUserData.username);
      expect(user.password).toBeUndefined();  // 密码不应返回
      expect(user.role).toBe('user');  // 默认角色
    });

    test('应该自动加密密码', async () => {
      const user = await authService.register(validUserData);
      
      // 从数据库查询（包含密码字段）
      const userWithPassword = await User.findById(user._id).select('+password');
      
      // 密码应该被加密（不等于原始密码）
      expect(userWithPassword.password).not.toBe(validUserData.password);
      
      // 验证密码是否正确
      const isMatch = await userWithPassword.comparePassword(validUserData.password);
      expect(isMatch).toBe(true);
    });

    test('邮箱已存在时应抛出 ConflictError', async () => {
      // 先创建一个用户
      await authService.register(validUserData);

      // 尝试用相同邮箱注册
      await expect(
        authService.register(validUserData)
      ).rejects.toThrow(ConflictError);

      await expect(
        authService.register(validUserData)
      ).rejects.toThrow('Email already exists');
    });

    test('用户名已存在时应抛出 ConflictError', async () => {
      await authService.register(validUserData);

      // 不同邮箱，但用户名相同
      await expect(
        authService.register({
          ...validUserData,
          email: 'another@example.com'
        })
      ).rejects.toThrow('Username already exists');
    });
  });

  describe('login', () => {
    const userData = {
      email: 'test@example.com',
      password: 'Test123456',
      username: 'testuser'
    };

    beforeEach(async () => {
      // 每个测试前创建用户
      await authService.register(userData);
    });

    test('正确的邮箱密码应登录成功', async () => {
      const result = await authService.login({
        email: userData.email,
        password: userData.password
      });

      expect(result).toHaveProperty('user');
      expect(result).toHaveProperty('accessToken');
      expect(result).toHaveProperty('refreshToken');
      expect(result.user.email).toBe(userData.email);
    });

    test('错误的密码应抛出 AuthenticationError', async () => {
      await expect(
        authService.login({
          email: userData.email,
          password: 'WrongPassword'
        })
      ).rejects.toThrow('Invalid email or password');
    });

    test('不存在的邮箱应抛出 AuthenticationError', async () => {
      await expect(
        authService.login({
          email: 'nonexistent@example.com',
          password: userData.password
        })
      ).rejects.toThrow('Invalid email or password');
    });

    test('登录应更新 lastLoginAt', async () => {
      const beforeLogin = await User.findOne({ email: userData.email });
      expect(beforeLogin.lastLoginAt).toBeUndefined();

      await authService.login({
        email: userData.email,
        password: userData.password
      });

      const afterLogin = await User.findOne({ email: userData.email });
      expect(afterLogin.lastLoginAt).toBeDefined();
    });
  });
});
```

```javascript
const postService = require('../../../src/services/postService');

describe('PostService', () => {

      await postService.createPost(testUser._id, validPostData);

      expect(updatedUser.stats.postsCount).toBe(beforeCount + 1);
    });
  });

  describe('getPostById', () => {
    let testPost;

    beforeEach(async () => {
      testPost = await postService.createPost(testUser._id, {
        title: 'Test Post',
        content: 'Test content',
        status: 'published'
      });
    });

    test('应该返回文章详情', async () => {
      const post = await postService.getPostById(testPost._id);

      expect(post).toBeDefined();
      expect(post.title).toBe(testPost.title);

      expect(cached).toBeDefined();
      expect(JSON.parse(cached).title).toBe(testPost.title);

      // Mock Post.findById 来验证是否被调用
      const findByIdSpy = jest.spyOn(Post, 'findById');

      // 第二次查询（应使用缓存）
      await postService.getPostById(testPost._id);

      // findById 不应被调用
      expect(findByIdSpy).not.toHaveBeenCalled();

      findByIdSpy.mockRestore();

      const fakeId = '507f1f77bcf86cd799439011';

      });

      expect(result.posts[0].title).toBe('Post 1');
```

```javascript
const request = require('supertest');
const app = require('../../src/app');
const User = require('../../src/models/User');

describe('Auth API', () => {
  describe('POST /api/auth/register', () => {
    const validUserData = {
      email: 'test@example.com',
      password: 'Test123456',
      username: 'testuser'
    };

    test('有效数据应注册成功', async () => {
      const res = await request(app)
        .post('/api/auth/register')
        .send(validUserData)
        .expect(201);

      expect(res.body.success).toBe(true);
      expect(res.body.data.user.email).toBe(validUserData.email);
      expect(res.body.data.user.password).toBeUndefined();
    });

    test('缺少必填字段应返回 400', async () => {
      const res = await request(app)
        .post('/api/auth/register')
        .send({
          email: 'test@example.com'
          // 缺少 password 和 username
        })
        .expect(400);

      expect(res.body.success).toBe(false);
      expect(res.body.errors).toBeDefined();
    });

    test('邮箱格式错误应返回 400', async () => {
      const res = await request(app)
        .post('/api/auth/register')
        .send({
          ...validUserData,
          email: 'invalid-email'
        })
        .expect(400);

      expect(res.body.message).toContain('valid email');
    });

    test('密码强度不足应返回 400', async () => {
      const res = await request(app)
        .post('/api/auth/register')
        .send({
          ...validUserData,
          password: '123'  // 太短
        })
        .expect(400);

      expect(res.body.errors).toBeDefined();
    });

    test('重复邮箱应返回 409', async () => {
      // 先注册一次
      await request(app)
        .post('/api/auth/register')
        .send(validUserData);

      // 再次注册
      const res = await request(app)
        .post('/api/auth/register')
        .send(validUserData)
        .expect(409);

      expect(res.body.message).toContain('Email already exists');
    });
  });

  describe('POST /api/auth/login', () => {
    const userData = {
      email: 'test@example.com',
      password: 'Test123456',
      username: 'testuser'
    };

    beforeEach(async () => {
      // 先注册用户
      await request(app)
        .post('/api/auth/register')
        .send(userData);
    });

    test('正确的邮箱密码应登录成功', async () => {
      const res = await request(app)
        .post('/api/auth/login')
        .send({
          email: userData.email,
          password: userData.password
        })
        .expect(200);

      expect(res.body.success).toBe(true);
      expect(res.body.data.accessToken).toBeDefined();
      expect(res.body.data.refreshToken).toBeDefined();
      expect(res.body.data.user.email).toBe(userData.email);
    });

    test('错误的密码应返回 401', async () => {
      const res = await request(app)
        .post('/api/auth/login')
        .send({
          email: userData.email,
          password: 'WrongPassword'
        })
        .expect(401);

      expect(res.body.success).toBe(false);
      expect(res.body.message).toContain('Invalid');
    });
  });

  describe('GET /api/auth/me', () => {
    let token;

    beforeEach(async () => {
      // 注册并登录获取 token
      await request(app)
        .post('/api/auth/register')
        .send({
          email: 'test@example.com',
          password: 'Test123456',
          username: 'testuser'
        });

      const loginRes = await request(app)
        .post('/api/auth/login')
        .send({
          email: 'test@example.com',
          password: 'Test123456'
        });

      token = loginRes.body.data.accessToken;
    });

    test('携带有效 Token 应返回用户信息', async () => {
      const res = await request(app)
        .get('/api/auth/me')
        .set('Authorization', `Bearer ${token}`)
        .expect(200);

      expect(res.body.data.user.email).toBe('test@example.com');
    });

    test('未携带 Token 应返回 401', async () => {
      const res = await request(app)
        .get('/api/auth/me')
        .expect(401);

      expect(res.body.message).toContain('No token provided');
    });

    test('无效 Token 应返回 401', async () => {
      const res = await request(app)
        .get('/api/auth/me')
        .set('Authorization', 'Bearer invalid-token')
        .expect(401);

      expect(res.body.message).toContain('Invalid token');
    });
  });
});
```

```javascript
    });
  });
```

## 总结

### ✅ 核心要点

1. **测试金字塔**：单元测试（70%）+ 集成测试（20%）+ E2E测试（10%）
2. **测试环境**：使用 mongodb-memory-server 避免污染真实数据库
3. **覆盖率目标**：至少 80% 行覆盖率
4. **测试数据**：使用工厂函数创建测试数据

### 🔑 实现关键

1. **Jest 配置**：设置 testEnvironment、覆盖率阈值
2. **Setup 文件**：beforeAll 启动数据库、afterEach 清理数据
3. **Supertest**：测试 HTTP 接口
4. **工厂模式**：复用测试数据创建逻辑

### 💎 应用价值

- **快速反馈**：30秒测试所有功能
- **重构信心**：修改代码后立即发现破坏性改动
- **文档作用**：测试即文档，展示如何使用 API

### 🎯 面试重点

- **Q1**：如何测试异步代码？  
  **A**：使用 `async/await` 或 `rejects.toThrow()`。

- **Q2**：如何避免测试污染真实数据库？  
  **A**：使用 `mongodb-memory-server` 内存数据库。

- **Q3**：测试覆盖率多少算合格？  
  **A**：至少 80% 行覆盖率，核心业务逻辑 90%+。

## 🤔 常见问题（FAQ）

### Q1：单元测试和集成测试有什么区别？

**A**：测试范围和依赖不同：

| 类型       | 测试范围           | 外部依赖     | 执行速度 | 占比   |
|------------|-------------------|--------------|----------|--------|
| 单元测试   | 单个函数/模块      | Mock（模拟） | 快（ms级）| 70%    |
| 集成测试   | 多个模块+数据库    | 真实数据库   | 慢（s级） | 20%    |
| E2E测试    | 完整用户流程       | 真实环境     | 很慢      | 10%    |

**示例**：
```javascript
// 单元测试：只测 hashPassword 函数
test('应该正确哈希密码', async () => {
  const hashed = await hashPassword('123456');
  expect(hashed).toMatch(/^\$2[ayb]\$.{56}$/);  // bcrypt 格式
});

// 集成测试：测试完整注册流程（包含数据库）
test('应该成功注册用户', async () => {
  const res = await request(app)
    .post('/api/auth/register')
    .send({ username: 'test', password: '123456' });
  
  expect(res.status).toBe(201);
  const user = await User.findOne({ username: 'test' });  // 真实数据库查询
  expect(user).toBeDefined();
});
```

### Q2：如何避免测试污染真实数据库？

**A**：使用内存数据库或测试数据库：

**方案1：mongodb-memory-server（推荐）**
```javascript
const { MongoMemoryServer } = require('mongodb-memory-server');

beforeAll(async () => {
  mongoServer = await MongoMemoryServer.create();
  const uri = mongoServer.getUri();
  await mongoose.connect(uri);  // 连接内存数据库
});

afterAll(async () => {
  await mongoose.disconnect();
  await mongoServer.stop();
});
```

**方案2：独立测试数据库**
```javascript
// .env.test
MONGO_URI=mongodb://localhost:27017/blog_test  // 专门的测试库

afterEach(async () => {
  await User.deleteMany({});  // 每次测试后清空数据
  await Post.deleteMany({});
});
```

### Q3：如何测试需要登录的接口？

**A**：先登录获取 Token，再携带 Token 测试：

```javascript
describe('POST /api/posts', () => {
  let token;
  
  beforeEach(async () => {
    // 1. 创建测试用户
    const user = await User.create({
      username: 'testuser',

    token = jwt.sign({ userId: user._id }, process.env.JWT_SECRET);
  });
  
  it('应该创建文章', async () => {
    const res = await request(app)
      .post('/api/posts')
      .set('Authorization', `Bearer ${token}`)  // 携带 Token
      .send({ title: 'Test Post', content: 'Content' });
    
    expect(res.status).toBe(201);
  });
});
```

### Q4：如何提高测试覆盖率？

**A**：补充边界条件和异常场景测试：

```javascript
// ❌ 只测试正常情况（覆盖率低）
test('应该创建文章', async () => {
```

### Q5：测试运行很慢怎么办？

**A**：采用以下优化策略：

**优化1：并行执行测试**
```bash
# Jest 默认并行，如果禁用了，重新启用
jest --maxWorkers=4  # 使用 4 个 Worker
```

**优化2：只测试修改的文件**
```bash
jest --onlyChanged  # 只测试 Git 中修改的文件
jest --watchAll     # 监听模式，自动重新测试
```

**优化3：跳过慢测试**
```javascript
// 标记慢测试
test.skip('耗时的 E2E 测试', async () => {
  // 本地开发时跳过
});

// CI 环境才运行
const describeIf = process.env.CI ? describe : describe.skip;
describeIf('E2E tests', () => {
  // E2E 测试
});
```

**优化4：减少数据库操作**
```javascript
// ❌ 每个测试都创建用户（慢）
beforeEach(async () => {
  user = await User.create({ username: 'test' });
});

// ✅ 所有测试共享一个用户（快）
beforeAll(async () => {
  user = await User.create({ username: 'test' });
});
```
