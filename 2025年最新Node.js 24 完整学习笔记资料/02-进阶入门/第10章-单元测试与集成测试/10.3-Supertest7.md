## 📖 知识点概述

### 核心概念

**Supertest**是专门用于HTTP API集成测试的库，提供优雅的链式断言API，支持所有主流Node.js框架，无需启动真实服务器即可测试。

### 测试类型对比

```
┌──────────────────────────────────────────────────────────┐
│              测试类型金字塔                                │
├──────────────────────────────────────────────────────────┤
│                                                            │
│           E2E测试（Playwright）                            │
│              ├─ 真实浏览器                                 │
│              ├─ 完整用户流程                               │

│                                                            │

```

### Supertest核心特性

| 特性 | 说明 | 示例 |
|------|------|------|
| **链式断言** | 优雅的API | `.expect(200).expect('Content-Type', /json/)` |
| **无需服务器** | 自动启动 | `supertest(app)` |
| **多框架支持** | Express/Koa/Fastify | 统一API |
| **异步支持** | Promise/async | `await request(app).get('/')` |
| **文件上传** | multipart | `.attach('file', buffer)` |
| **Cookie支持** | Session测试 | `.set('Cookie', cookies)` |

```javascript
// app.js
import express from 'express';

const app = express();
app.use(express.json());

// 简单路由
app.get('/api/hello', (req, res) => {
  res.json({ message: 'Hello, World!' });
});

// 带参数路由
app.get('/api/users/:id', (req, res) => {
  res.json({ id: req.params.id, name: 'John' });
});

// POST路由
app.post('/api/users', (req, res) => {
  const { name, email } = req.body;
  
  if (!name || !email) {
    return res.status(400).json({ error: 'Name and email required' });
  }
  
  res.status(201).json({ id: 1, name, email });
});

export default app;

// app.test.js
import request from 'supertest';
import app from './app.js';

describe('GET /api/hello', () => {
  it('返回问候消息', async () => {
    // 💡 基础GET请求测试
    const response = await request(app)
      .get('/api/hello')
      .expect(200)                              // 断言状态码
      .expect('Content-Type', /json/);          // 断言Content-Type
    
    expect(response.body).toEqual({
      message: 'Hello, World!'
    });
  });
});

describe('GET /api/users/:id', () => {
  it('返回用户信息', async () => {
    const response = await request(app)
      .get('/api/users/123')
      .expect(200);
    
    expect(response.body.id).toBe('123');
    expect(response.body.name).toBe('John');
  });
});

describe('POST /api/users', () => {
  it('创建新用户', async () => {
    const newUser = {
      name: 'Alice',
      email: 'alice@example.com'
    };
    
    // 💡 POST请求测试
    const response = await request(app)
      .post('/api/users')
      .send(newUser)                            // 发送请求体
      .expect(201)
      .expect('Content-Type', /json/);
    
    expect(response.body).toMatchObject({
      id: expect.any(Number),
      name: 'Alice',
      email: 'alice@example.com'
    });
  });
  
  it('验证必填字段', async () => {
    const response = await request(app)
      .post('/api/users')
      .send({ name: 'Bob' })                    // 缺少email
      .expect(400);
    
    expect(response.body.error).toBe('Name and email required');
  });
});
```

### 实践2：认证测试（JWT）

```javascript
// auth.middleware.js
import jwt from 'jsonwebtoken';

const SECRET = 'your-secret-key';

export function authMiddleware(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  
  if (!token) {
    return res.status(401).json({ error: 'Token required' });
  }
  
  try {
    const decoded = jwt.verify(token, SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
}

// app.js
app.post('/api/login', (req, res) => {
  const { username, password } = req.body;
  
  if (username === 'admin' && password === 'password') {
    const token = jwt.sign({ id: 1, username }, SECRET, { expiresIn: '1h' });
    res.json({ token });
  } else {
    res.status(401).json({ error: 'Invalid credentials' });
  }
});

app.get('/api/profile', authMiddleware, (req, res) => {
  res.json({ user: req.user });
});

// auth.test.js
describe('Authentication', () => {
  let authToken;
  
  // 💡 登录获取token
  beforeAll(async () => {
    const response = await request(app)
      .post('/api/login')
      .send({ username: 'admin', password: 'password' })
      .expect(200);
    
    authToken = response.body.token;
  });
  
  it('登录成功返回token', async () => {
    const response = await request(app)
      .post('/api/login')
      .send({ username: 'admin', password: 'password' })
      .expect(200);
    
    expect(response.body.token).toBeDefined();
    expect(typeof response.body.token).toBe('string');
  });
  
  it('登录失败返回401', async () => {
    await request(app)
      .post('/api/login')
      .send({ username: 'admin', password: 'wrong' })
      .expect(401)
      .expect({ error: 'Invalid credentials' });
  });
  
  it('访问受保护路由需要token', async () => {
    await request(app)
      .get('/api/profile')
      .expect(401)
      .expect({ error: 'Token required' });
  });
  
  it('使用有效token访问受保护路由', async () => {
    // 💡 设置Authorization header
    const response = await request(app)
      .get('/api/profile')
      .set('Authorization', `Bearer ${authToken}`)
      .expect(200);
    
    expect(response.body.user.username).toBe('admin');
  });
  
  it('无效token返回401', async () => {
    await request(app)
      .get('/api/profile')
      .set('Authorization', 'Bearer invalid-token')
      .expect(401)
      .expect({ error: 'Invalid token' });
  });
});
```

```javascript
const upload = multer({ dest: 'uploads/' });

app.post('/api/upload', upload.single('file'), (req, res) => {

    files: req.files.map(f => ({

      size: f.size
    }))
  });
});

// upload.test.js
import fs from 'fs';
import path from 'path';

describe('File Upload', () => {
  const testFilePath = path.join(__dirname, 'test-file.txt');
  
  // 创建测试文件
  beforeAll(() => {
    fs.writeFileSync(testFilePath, 'Test content');
  });
  
  // 清理测试文件

    if (fs.existsSync(testFilePath)) {
      fs.unlinkSync(testFilePath);

    // 💡 文件上传测试
    const response = await request(app)
      .post('/api/upload')
      .attach('file', testFilePath)             // 附加文件
      .expect(200);

    expect(response.body.originalname).toBe('test-file.txt');
    expect(response.body.mimetype).toBe('text/plain');
    expect(response.body.size).toBeGreaterThan(0);

  it('上传多个文件', async () => {
    const response = await request(app)
      .post('/api/upload-multiple')
      .attach('files', testFilePath)
      .attach('files', testFilePath)
      .expect(200);
    
    expect(response.body.count).toBe(2);
    expect(response.body.files).toHaveLength(2);
  });
  
  it('未提供文件返回400', async () => {
    await request(app)
      .post('/api/upload')
      .expect(400)
      .expect({ error: 'File required' });
  });
  
  // 💡 使用Buffer上传
```

## 🔧 错误处理

### 高频错误1：异步测试未等待

**现象：**
```javascript
it('test', () => {
  // ❌ 忘记await
  request(app).get('/api/users').expect(200);
  // 测试总是通过
});
```

**解决方案：**
```javascript
// ✅ 方案1：async/await
it('test', async () => {
  await request(app).get('/api/users').expect(200);
});

// ✅ 方案2：返回Promise
it('test', () => {
  return request(app).get('/api/users').expect(200);
});

// ✅ 方案3：done回调（旧方式）
it('test', (done) => {
  request(app)
    .get('/api/users')
    .expect(200)
    .end(done);
});
```

### 高频错误2：端口冲突

**现象：**
```
Error: listen EADDRINUSE: address already in use :::3000
```

**原因：**
在测试中启动了服务器但未关闭

**解决方案：**
```javascript
// ❌ 错误：启动服务器
const server = app.listen(3000);

// ✅ 正确：直接传app
const response = await request(app).get('/');

// 💡 如果必须启动服务器
let server;

beforeAll((done) => {
  server = app.listen(0, done);  // 端口0=随机端口
});

afterAll((done) => {
  server.close(done);
});

it('test', async () => {
  const port = server.address().port;
  // 使用动态端口
});
```

### 高频错误3：Cookie/Session未保持

**现象：**
```javascript
// ❌ 每个请求都是独立的
await request(app).post('/login').send({ username: 'john' });
await request(app).get('/profile');  // 未携带cookie，失败
```

**解决方案：**
```javascript
// ✅ 使用agent保持cookie
const agent = request.agent(app);

await agent.post('/login').send({ username: 'john' });
await agent.get('/profile');  // 自动携带cookie

// 💡 或手动管理cookie
const loginRes = await request(app).post('/login');
const cookies = loginRes.headers['set-cookie'];

await request(app)
  .get('/profile')
  .set('Cookie', cookies);
```

```javascript
      .set('Authorization', `Bearer ${authToken}`)

  });
```

## 📋 总结

### 核心要点

1. **HTTP测试**：专注API集成测试
2. **链式断言**：优雅的测试API
3. **多框架支持**：Express/Koa/Fastify统一
4. **无需服务器**：自动管理生命周期

### 实现关键

1. **异步处理**：必须await或return
2. **Cookie管理**：使用agent保持状态
3. **文件上传**：attach方法
4. **自定义断言**：expect函数

### 应用价值

- **API质量**：确保接口正确性
- **重构信心**：安全重构代码
- **文档作用**：测试即文档
- **快速反馈**：比E2E测试快10倍+

### 面试重点

**Q1：Supertest与Postman的区别？**
答：Supertest是代码化测试，可自动化、CI集成、版本控制。Postman是手动测试工具，适合探索性测试和API文档。

**Q2：为什么Supertest不需要启动服务器？**
答：Supertest直接调用app的请求处理函数，绕过网络层，因此更快且无需端口。

**Q3：如何测试需要数据库的API？**
答：1) 使用测试数据库 2) 使用内存数据库（SQLite） 3) Mock数据库层 4) 使用docker-compose快速启动数据库。

### 注意事项

⚠️ **异步测试**：必须await或return Promise
⚠️ **Cookie管理**：使用agent保持会话
⚠️ **端口冲突**：不要在测试中启动服务器
