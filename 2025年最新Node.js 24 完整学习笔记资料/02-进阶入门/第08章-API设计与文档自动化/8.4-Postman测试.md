# 8.4 Postman 与自动化测试

> 使用Postman进行API测试和自动化

---

## 📖 知识点概述

### 核心概念

**Postman**是最流行的API测试工具，提供友好的GUI界面，支持环境变量、测试脚本、自动化测试、团队协作等功能。

### 解决的问题

- **手动测试**：重复测试费时费力
- **环境切换**：开发/测试/生产环境切换麻烦
- **团队协作**：API信息难以共享
- **回归测试**：代码改动后需要重新测试所有接口

### 主要特点

1. **集合管理**：组织和管理API请求
2. **环境变量**：管理不同环境配置
3. **自动化测试**：编写测试脚本
4. **CI/CD集成**：Newman命令行运行

## 🚀 使用指南

### 准备：安装Postman

```bash
# 下载Postman Desktop
# https://www.postman.com/downloads/

# 安装Newman（命令行工具）
npm install -g newman

# Newman HTML Reporter
npm install -g newman-reporter-html
```

### 实践：创建集合

#### 第一步：基础请求

```javascript
// 在Postman中创建请求

// GET请求
GET {{baseUrl}}/api/users?page=1&limit=10

// POST请求（Body → raw → JSON）
POST {{baseUrl}}/api/users
{
  "username": "john_doe",
  "email": "john@example.com",
  "password": "secret123"
}

// PUT请求
PUT {{baseUrl}}/api/users/{{userId}}
{
  "username": "john_updated",
  "email": "john.new@example.com"
}

// DELETE请求
DELETE {{baseUrl}}/api/users/{{userId}}
```

#### 第二步：环境变量

```json
// 开发环境（Development.json）
{
  "id": "dev-env",
  "name": "Development",
  "values": [
    {
      "key": "baseUrl",
      "value": "http://localhost:3000",
      "type": "default"
    },
    {
      "key": "token",
      "value": "",
      "type": "secret"
    },
    {
      "key": "userId",
      "value": "",
      "type": "default"
    }
  ]
}

// 生产环境（Production.json）
{
  "id": "prod-env",
  "name": "Production",
  "values": [
    {
      "key": "baseUrl",
      "value": "https://api.example.com",
      "type": "default"
    },
    {
      "key": "token",
      "value": "",
      "type": "secret"
    }
  ]
}
```

#### 第三步：Pre-request脚本

```javascript
// 💡 在请求前执行（Pre-request Script标签）

// 设置时间戳
pm.environment.set("timestamp", Date.now());

// 生成随机数据
pm.environment.set("randomEmail", `user${Math.random().toString(36).substring(7)}@example.com`);

// 从环境变量获取token
const token = pm.environment.get("token");
if (token) {
  pm.request.headers.add({
    key: 'Authorization',
    value: `Bearer ${token}`
  });
}

// 签名请求（如果需要）
const crypto = require('crypto-js');
const secretKey = pm.environment.get("secretKey");
const timestamp = Date.now();
const signature = crypto.HmacSHA256(`${timestamp}`, secretKey).toString();

pm.environment.set("signature", signature);
pm.environment.set("timestamp", timestamp);
```

#### 第四步：测试脚本（Tests）

```javascript
// 💡 在响应后执行（Tests标签）

// 1. 测试状态码
pm.test("Status code is 200", () => {
  pm.response.to.have.status(200);
});

// 2. 测试响应时间
pm.test("Response time is less than 200ms", () => {
  pm.expect(pm.response.responseTime).to.be.below(200);
});

// 3. 测试响应体
pm.test("Response has data property", () => {
  const jsonData = pm.response.json();
  pm.expect(jsonData).to.have.property('data');
  pm.expect(jsonData.success).to.be.true;
});

// 4. 测试数据类型
pm.test("Data is array", () => {
  const jsonData = pm.response.json();
  pm.expect(jsonData.data).to.be.an('array');
});

// 5. 测试数组长度
pm.test("Data array has items", () => {
  const jsonData = pm.response.json();
  pm.expect(jsonData.data.length).to.be.above(0);
});

// 6. 测试对象属性
pm.test("User has required fields", () => {
  const jsonData = pm.response.json();
  const user = jsonData.data[0];
  
  pm.expect(user).to.have.property('id');
  pm.expect(user).to.have.property('username');
  pm.expect(user).to.have.property('email');
  pm.expect(user.email).to.match(/^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$/);
});

// 7. 保存响应数据到环境变量
pm.test("Save user ID", () => {
  const jsonData = pm.response.json();
  if (jsonData.data && jsonData.data.id) {
    pm.environment.set("userId", jsonData.data.id);
    console.log("Saved userId:", jsonData.data.id);
  }
});

// 8. 保存Token
pm.test("Save auth token", () => {
  const jsonData = pm.response.json();
  if (jsonData.token) {
    pm.environment.set("token", jsonData.token);
    console.log("Token saved");
  }
});

// 9. 测试响应头
pm.test("Content-Type is JSON", () => {
  pm.response.to.have.header("Content-Type");
  pm.expect(pm.response.headers.get("Content-Type")).to.include("application/json");
});

// 10. 测试分页信息
pm.test("Pagination info is correct", () => {
  const jsonData = pm.response.json();
  const { pagination } = jsonData;
  
  pm.expect(pagination).to.have.property('page');
  pm.expect(pagination).to.have.property('limit');
  pm.expect(pagination).to.have.property('total');
  pm.expect(pagination.page).to.be.a('number');
  pm.expect(pagination.total).to.be.at.least(pagination.data?.length || 0);
});
```

#### 第五步：集合级别脚本

```javascript
// 💡 Collection → Edit → Pre-request Scripts（所有请求前执行）

// 全局设置
pm.globals.set("apiVersion", "v1");
pm.globals.set("contentType", "application/json");

// 检查环境
const env = pm.environment.name;
console.log(`Current environment: ${env}`);

// 💡 Collection → Edit → Tests（所有请求后执行）

// 全局测试
pm.test("Status code is not 500", () => {
  pm.expect(pm.response.code).to.not.equal(500);
});

pm.test("Response is JSON", () => {
  pm.response.to.be.json;
});
```

## 💡 高级技巧

### 技巧1：动态请求

```javascript
// Pre-request Script

// 根据环境设置不同的请求体
const env = pm.environment.name;

if (env === "Development") {
  pm.request.body.raw = JSON.stringify({
    username: "test_user",
    email: "test@example.com",
    password: "test123"
  });
} else {
  pm.request.body.raw = JSON.stringify({
    username: "prod_user",
    email: "prod@example.com",
    password: "securePassword123!"
  });
}
```

### 技巧2：链式请求

```javascript
// 💡 请求1：登录 → 保存Token

// POST /api/auth/login
// Tests:
pm.test("Login successful", () => {
  const jsonData = pm.response.json();
  pm.environment.set("token", jsonData.token);
});

// 💡 请求2：创建用户 → 使用Token + 保存用户ID

// POST /api/users
// Headers: Authorization: Bearer {{token}}
// Tests:
pm.test("User created", () => {
  const jsonData = pm.response.json();
  pm.environment.set("userId", jsonData.data.id);
});

// 💡 请求3：获取用户详情 → 使用Token和userId

// GET /api/users/{{userId}}
// Headers: Authorization: Bearer {{token}}
```

### 技巧3：数据驱动测试

```javascript
// 创建CSV文件（users.csv）

  "email": "{{email}}",
  "password": "{{password}}"

// Tests中验证：
pm.test("User created with data from CSV", () => {
  const jsonData = pm.response.json();
  pm.expect(jsonData.data.username).to.equal(pm.iterationData.get("username"));
});
```

## 🔧 Newman命令行

### 基础运行

```bash
# 运行集合
newman run collection.json

# 指定环境
newman run collection.json -e environment.json

# 生成HTML报告
newman run collection.json -r html --reporter-html-export report.html

# 并发运行
newman run collection.json --iteration-count 10 --delay-request 1000

# 使用全局变量
newman run collection.json --global-var "apiKey=abc123"
```

### CI/CD集成

```yaml
# .github/workflows/api-test.yml
name: API Tests

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v2
      
      - name: Setup Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '18'
      
      - name: Install Newman
        run: npm install -g newman newman-reporter-html
      
      - name: Run API Tests
        run: |
          newman run postman/collection.json \
            -e postman/environment.json \
            -r cli,html \
            --reporter-html-export results/report.html
      
      - name: Upload Test Results
        if: always()
        uses: actions/upload-artifact@v2
        with:
          name: test-results
          path: results/
```

### Package.json脚本

```json
{
```

## 🎯 完整测试套件示例

```javascript
// 💡 集合：User Management API Tests

// ==================== 1. 登录 ====================
// POST {{baseUrl}}/api/auth/login
{
  "email": "admin@example.com",
  "password": "admin123"
}

// Tests:
pm.test("Login successful", () => {
  pm.response.to.have.status(200);
  const jsonData = pm.response.json();
  pm.expect(jsonData).to.have.property('token');
  pm.environment.set("token", jsonData.token);
});

// ==================== 2. 创建用户 ====================
// POST {{baseUrl}}/api/users
// Headers: Authorization: Bearer {{token}}
{
  "username": "{{$randomUserName}}",
  "email": "{{$randomEmail}}",
  "password": "test123"
}

// Tests:
pm.test("User created successfully", () => {
  pm.response.to.have.status(201);
  const jsonData = pm.response.json();
  
  pm.expect(jsonData.success).to.be.true;
  pm.expect(jsonData.data).to.have.property('id');
  
  // 保存用户ID供后续使用
  pm.environment.set("createdUserId", jsonData.data.id);
  pm.environment.set("createdUsername", jsonData.data.username);
});

// ==================== 3. 获取用户列表 ====================
// GET {{baseUrl}}/api/users?page=1&limit=10

// Tests:
pm.test("Get users list", () => {
  pm.response.to.have.status(200);
  const jsonData = pm.response.json();
  
  pm.expect(jsonData.data).to.be.an('array');
  pm.expect(jsonData.pagination).to.exist;
  pm.expect(jsonData.pagination.page).to.equal(1);
});

// ==================== 4. 获取单个用户 ====================
// GET {{baseUrl}}/api/users/{{createdUserId}}

// Tests:
pm.test("Get user by ID", () => {
  pm.response.to.have.status(200);
  const jsonData = pm.response.json();
  
  pm.expect(jsonData.data.id).to.equal(pm.environment.get("createdUserId"));
  pm.expect(jsonData.data.username).to.equal(pm.environment.get("createdUsername"));
});

// ==================== 5. 更新用户 ====================
// PATCH {{baseUrl}}/api/users/{{createdUserId}}
{
  "username": "{{createdUsername}}_updated"
}

// Tests:
pm.test("Update user", () => {
  pm.response.to.have.status(200);
  const jsonData = pm.response.json();
  
  pm.expect(jsonData.data.username).to.include("_updated");
});

// ==================== 6. 删除用户 ====================
// DELETE {{baseUrl}}/api/users/{{createdUserId}}

// Tests:
pm.test("Delete user", () => {
  pm.response.to.have.status(204);
});

// ==================== 7. 验证删除（应该404） ====================
// GET {{baseUrl}}/api/users/{{createdUserId}}

// Tests:
pm.test("User not found after deletion", () => {
  pm.response.to.have.status(404);
});
```

## 📋 总结

### 核心要点

1. **Postman**：最流行的API测试工具
2. **环境变量**：管理不同环境配置
3. **测试脚本**：自动化验证响应
4. **Newman**：命令行运行，CI/CD集成

### 实现关键

1. **集合组织**：按功能模块组织请求
2. **变量管理**：合理使用环境变量和集合变量
3. **链式请求**：通过环境变量传递数据
4. **完整测试**：覆盖正常流程和异常情况

### 应用价值

- **提高效率**：一键运行所有测试
- **质量保证**：自动化回归测试
- **团队协作**：共享集合和环境
- **CI/CD集成**：自动化测试流程

### 面试重点

**Q1：Postman有哪些测试功能？**
答：状态码测试、响应时间测试、JSON Schema验证、数据驱动测试、链式请求

**Q2：如何在CI/CD中使用Postman？**
答：使用Newman命令行工具，结合GitHub Actions/Jenkins运行集合

**Q3：如何实现API测试的数据驱动？**
答：使用CSV/JSON文件作为数据源，在Collection Runner中导入
