# 8.3 Swagger UI 集成

> 自动生成交互式API文档

---

## 📖 知识点概述

### 核心概念

**Swagger UI**是一个可视化的API文档工具，可以将OpenAPI规范渲染为美观的交互式文档，支持在线测试API。

### 解决的问题

- **文档难看**：纯文本文档不直观
- **无法测试**：需要另外的工具测试API
- **更新麻烦**：手动维护文档费时费力
- **学习成本**：新人需要时间理解API

### 主要特点

1. **交互式**：在线测试API，无需Postman
2. **自动生成**：从代码注释生成文档
3. **实时预览**：代码改变，文档自动更新
4. **美观易用**：清晰的UI，易于阅读

---

### 实践：Express集成

#### 方式1：使用swagger-jsdoc（推荐）

```javascript
// swagger-express.js
import express from 'express';
import swaggerJsdoc from 'swagger-jsdoc';
import swaggerUi from 'swagger-ui-express';

const app = express();
app.use(express.json());

// 💡 Swagger配置
const swaggerOptions = {
  definition: {
    openapi: '3.1.0',
    info: {
      title: 'User Management API',
      version: '1.0.0',
      description: '用户管理系统API文档',
      contact: {
        name: 'API Support',
        email: 'support@example.com'
      }
    },
    servers: [
      {
        url: 'http://localhost:3000',
        description: '开发服务器'
      },
      {
        url: 'https://api.example.com',
        description: '生产服务器'
      }
    ],
    components: {
      securitySchemes: {
        BearerAuth: {
          type: 'http',
          scheme: 'bearer',
          bearerFormat: 'JWT'
        }
      }
    },
    security: [{
      BearerAuth: []
    }]
  },
  // 💡 扫描哪些文件中的注释
  apis: ['./routes/*.js', './models/*.js']
};

// 生成Swagger规范
const swaggerSpec = swaggerJsdoc(swaggerOptions);

// 💡 Swagger UI路由
app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerSpec, {
  customCss: '.swagger-ui .topbar { display: none }',
  customSiteTitle: 'API Documentation'
}));

// 💡 提供原始JSON
app.get('/api-docs.json', (req, res) => {
  res.setHeader('Content-Type', 'application/json');
  res.send(swaggerSpec);
});

// 💡 路由注释（JSDoc格式）
/**
 * @openapi
 * /api/users:
 *   get:
 *     summary: 获取用户列表
 *     description: 返回所有用户的分页列表
 *     tags:
 *       - Users
 *     parameters:
 *       - in: query
 *         name: page
 *         schema:
 *           type: integer
 *           minimum: 1
 *           default: 1
 *         description: 页码
 *       - in: query
 *         name: limit
 *         schema:
 *           type: integer
 *           minimum: 1
 *           maximum: 100
 *           default: 10
 *         description: 每页数量
 *     responses:
 *       200:
 *         description: 成功返回用户列表
 *         content:
 *           application/json:
 *             schema:
 *               type: object
 *               properties:
 *                 success:
 *                   type: boolean
 *                   example: true
 *                 data:
 *                   type: array
 *                   items:
 *                     $ref: '#/components/schemas/User'
 *                 pagination:
 *                   type: object
 *                   properties:
 *                     page:
 *                       type: integer
 *                     limit:
 *                       type: integer
 *                     total:
 *                       type: integer
 *       401:
 *         description: 未授权
 *     security:
 *       - BearerAuth: []
 */
app.get('/api/users', async (req, res) => {
  const { page = 1, limit = 10 } = req.query;
  
  const users = await User.find()
    .limit(limit * 1)
    .skip((page - 1) * limit);
  
  const total = await User.countDocuments();
  
  res.json({
    success: true,
    data: users,
    pagination: {
      page: parseInt(page),
      limit: parseInt(limit),
      total
    }
  });
});

/**
 * @openapi
 * /api/users/{id}:
 *   get:
 *     summary: 获取单个用户
 *     tags:
 *       - Users
 *     parameters:
 *       - in: path
 *         name: id
 *         required: true
 *         schema:
 *           type: string
 *         description: 用户ID
 *     responses:
 *       200:
 *         description: 成功
 *         content:
 *           application/json:
 *             schema:
 *               type: object
 *               properties:
 *                 success:
 *                   type: boolean
 *                 data:
 *                   $ref: '#/components/schemas/User'
 *       404:
 *         description: 用户不存在
 */
app.get('/api/users/:id', async (req, res) => {
  const user = await User.findById(req.params.id);
  
  if (!user) {
    return res.status(404).json({
      success: false,
      error: '用户不存在'
    });
  }
  
  res.json({
    success: true,
    data: user
  });
});

/**
 * @openapi
 * /api/users:
 *   post:
 *     summary: 创建用户
 *     tags:
 *       - Users
 *     requestBody:
 *       required: true
 *       content:
 *         application/json:
 *           schema:
 *             type: object
 *             required:
 *               - username
 *               - email
 *               - password
 *             properties:
 *               username:
 *                 type: string
 *                 minLength: 3
 *                 maxLength: 20
 *                 example: john_doe
 *               email:
 *                 type: string
 *                 format: email
 *                 example: john@example.com
 *               password:
 *                 type: string
 *                 minLength: 6
 *                 format: password
 *                 example: secret123
 *     responses:
 *       201:
 *         description: 创建成功
 *       400:
 *         description: 请求参数错误
 *       409:
 *         description: 邮箱已存在
 */
app.post('/api/users', async (req, res) => {
  try {
    const user = await User.create(req.body);
    res.status(201).json({
      success: true,
      data: user
    });
  } catch (error) {
    if (error.code === 11000) {
      return res.status(409).json({
        success: false,
        error: '邮箱已存在'
      });
    }
    res.status(400).json({
      success: false,
      error: error.message
    });
  }
});

/**
 * @openapi
 * components:
 *   schemas:
 *     User:
 *       type: object
 *       properties:
 *         id:
 *           type: string
 *           description: 用户ID
 *         username:
 *           type: string
 *           description: 用户名
 *         email:
 *           type: string
 *           format: email
 *           description: 邮箱
 *         role:
 *           type: string
 *           enum: [admin, user]
 *           description: 角色
 *         createdAt:
 *           type: string
 *           format: date-time
 *           description: 创建时间
 */

app.listen(3000, () => {
  console.log('Server: http://localhost:3000');
  console.log('API Docs: http://localhost:3000/api-docs');
});
```

#### 方式2：使用独立YAML文件

```javascript
// swagger-yaml.js
import express from 'express';
import swaggerUi from 'swagger-ui-express';
import YAML from 'yamljs';

const app = express();

// 💡 加载YAML文件
const swaggerDocument = YAML.load('./openapi.yaml');

// 集成Swagger UI
app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerDocument, {
  explorer: true,
  customCss: '.swagger-ui .topbar { display: none }',
  customSiteTitle: 'My API Docs'
}));

app.listen(3000);
```

### 实践：Fastify集成

```javascript
// swagger-fastify.js
import Fastify from 'fastify';
import swagger from '@fastify/swagger';
import swaggerUi from '@fastify/swagger-ui';

const fastify = Fastify({ logger: true });

// 💡 注册Swagger插件
await fastify.register(swagger, {
  openapi: {
    openapi: '3.1.0',
    info: {
      title: 'User Management API',
      description: 'Fastify API Documentation',
      version: '1.0.0'
    },
    servers: [
      {
        url: 'http://localhost:3000',
        description: 'Development server'
      }
    ],
    components: {
      securitySchemes: {
        BearerAuth: {
          type: 'http',
          scheme: 'bearer',
          bearerFormat: 'JWT'
        }
      }
    }
  }
});

// 💡 注册Swagger UI
await fastify.register(swaggerUi, {
  routePrefix: '/docs',
  uiConfig: {
    docExpansion: 'list',
    deepLinking: false
  },
  staticCSP: true,
  transformStaticCSP: (header) => header
});

// 💡 带Schema的路由（自动生成文档）
fastify.get('/api/users', {
  schema: {
    description: '获取用户列表',
    tags: ['users'],
    querystring: {
      type: 'object',
      properties: {
        page: {
          type: 'integer',
          minimum: 1,
          default: 1,
          description: '页码'
        },
        limit: {
          type: 'integer',
          minimum: 1,
          maximum: 100,
          default: 10,
          description: '每页数量'
        }
      }
    },
    response: {
      200: {
        description: 'Successful response',
        type: 'object',
        properties: {
          success: { type: 'boolean' },
          data: {
            type: 'array',
            items: {
              type: 'object',
              properties: {
                id: { type: 'string' },
                username: { type: 'string' },
                email: { type: 'string', format: 'email' }
              }
            }
          },
          pagination: {
            type: 'object',
            properties: {
              page: { type: 'integer' },
              limit: { type: 'integer' },
              total: { type: 'integer' }
            }
          }
        }
      }
    }
  }
}, async (request, reply) => {
  const { page, limit } = request.query;
  
  // 业务逻辑
  return {
    success: true,
    data: [],
    pagination: { page, limit, total: 0 }
  };
});

// 💡 POST路由
fastify.post('/api/users', {
  schema: {
    description: '创建新用户',
    tags: ['users'],
    body: {
      type: 'object',
      required: ['username', 'email', 'password'],
      properties: {
        username: {
          type: 'string',
          minLength: 3,
          maxLength: 20
        },
        email: {
          type: 'string',
          format: 'email'
        },
        password: {
          type: 'string',
          minLength: 6
        }
      }
    },
    response: {
      201: {
        description: 'User created',
        type: 'object',
        properties: {
          success: { type: 'boolean' },
          data: {
            type: 'object',
            properties: {
              id: { type: 'string' },
              username: { type: 'string' },
              email: { type: 'string' }
            }
          }
        }
      }
    }
  }
}, async (request, reply) => {
  // 创建用户
  reply.code(201);
  return {
    success: true,
    data: request.body
  };
});

await fastify.listen({ port: 3000 });
console.log('API Docs: http://localhost:3000/docs');
```

### 自定义Swagger UI样式

```javascript
// custom-swagger.js
import swaggerUi from 'swagger-ui-express';

const customCss = `
  .swagger-ui .topbar { 
    display: none; 
  }
  .swagger-ui .info {
    margin: 50px 0;
  }
  .swagger-ui .scheme-container {
    background: #fafafa;
    box-shadow: 0 1px 2px 0 rgba(0,0,0,.15);
  }
`;

const customSiteTitle = "My API Documentation";

const customfavIcon = "/favicon.ico";

app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerSpec, {
  customCss,
  customSiteTitle,
  customfavIcon,
  customCssUrl: '/custom.css',
  swaggerOptions: {
    persistAuthorization: true, // 保持登录状态
    displayRequestDuration: true, // 显示请求时间
    filter: true, // 启用搜索
    syntaxHighlight: {
      activate: true,
      theme: 'monokai'
    }
  }
}));
```

## 📋 总结

### 核心要点

1. **swagger-jsdoc**：从JSDoc注释生成OpenAPI规范
2. **swagger-ui-express**：渲染为Swagger UI
3. **Fastify内置**：原生支持JSON Schema
4. **交互测试**：在文档中直接测试API

### 实现关键

1. **JSDoc注释**：使用`@openapi`标记
2. **Schema定义**：详细的请求/响应schema
3. **分组标签**：使用tags组织API
4. **安全配置**：配置JWT等认证方式

### 应用价值

- **开发效率**：代码即文档，无需手写
- **团队协作**：统一的API规范
- **测试便捷**：无需Postman，直接测试
- **易于维护**：代码改变，文档自动更新
