## 📖 知识点概述

### 核心概念

**云存储（Cloud Storage）**是由云服务商提供的对象存储服务，具有高可靠性、高可用性、无限扩展、按量付费等特点。

### 解决的问题

- **存储容量限制**：本地磁盘空间有限
- **高可用性要求**：需要99.99%以上可用性
- **全球访问加速**：通过CDN加速全球访问
- **运维成本高**：无需自己维护存储集群
- **数据备份困难**：自动多副本、跨区域备份

### 主要特点

1. **无限扩展**：按需使用，无需提前规划容量
2. **高可靠性**：99.9999999%（11个9）数据持久性
3. **低成本**：按实际使用量付费，0.12元/GB/月起
4. **全球CDN**：内置CDN加速，全球访问快
5. **丰富功能**：图片处理、视频转码、防盗链等

### 典型应用场景

- **用户上传内容**：头像、图片、视频
- **静态网站托管**：前端项目部署
- **大数据存储**：日志、备份、归档
- **内容分发**：下载站、视频点播

| 维度 | 本地存储 | 云存储 |
|------|---------|--------|
| **容量** | 有限（需扩容） | 无限 |

| **访问速度** | 单机带宽限制 | CDN全球加速 |
| **运维** | 需专人维护 | 零运维 |
| **扩展性** | 手动扩容 | 自动扩展 |

```javascript
// ali-oss-integration.js
import OSS from 'ali-oss';
import express from 'express';
import multer from 'multer';
import sharp from 'sharp';
import crypto from 'crypto';

const app = express();

// 💡 创建OSS客户端
const ossClient = new OSS({
  region: 'oss-cn-hangzhou',
  accessKeyId: process.env.ALI_ACCESS_KEY_ID,
  accessKeySecret: process.env.ALI_ACCESS_KEY_SECRET,
  bucket: 'my-bucket',
  // 自定义域名（CDN加速域名）
  cname: true,
  endpoint: 'cdn.example.com'
});

const upload = multer({ storage: multer.memoryStorage() });

// 💡 基础上传
app.post('/api/upload/oss/basic', upload.single('file'), async (req, res) => {
  try {
    const file = req.file;
    
    // 生成文件名
    const hash = crypto.randomUUID();
    const filename = `uploads/${Date.now()}-${hash}.jpg`;
    
    // 上传到OSS
    const result = await ossClient.put(filename, file.buffer, {
      // 设置访问权限
      headers: {
        'x-oss-object-acl': 'public-read',
        'Cache-Control': 'public, max-age=31536000',
        'Content-Type': file.mimetype
      }
    });
    
    res.json({
      message: '上传成功',
      url: result.url,
      name: result.name
    });
  } catch (error) {
    console.error('OSS上传失败:', error);
    res.status(500).json({ error: error.message });
  }
});

// 💡 流式上传（大文件）
app.post('/api/upload/oss/stream', upload.single('file'), async (req, res) => {
  try {
    const file = req.file;
    const filename = `uploads/${Date.now()}.mp4`;
    
    // 使用putStream上传
    const stream = require('stream');
    const bufferStream = new stream.PassThrough();
    bufferStream.end(file.buffer);
    
    const result = await ossClient.putStream(filename, bufferStream);
    
    res.json({
      message: '上传成功',
      url: result.url
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 图片处理（实时裁剪、缩放）
app.get('/api/image/process', async (req, res) => {
  try {
    const { key, width = 200, height = 200, format = 'jpg' } = req.query;
    
    // OSS图片处理参数
    // https://help.aliyun.com/document_detail/44688.html
    const process = `image/resize,w_${width},h_${height}/format,${format}`;
    
    // 生成处理后的URL
    const signedUrl = ossClient.signatureUrl(key, {
      expires: 3600,
      process
    });
    
    res.json({ url: signedUrl });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 批量删除
app.post('/api/upload/oss/delete-batch', async (req, res) => {
  try {
    const { keys } = req.body; // ['file1.jpg', 'file2.jpg']
    
    const result = await ossClient.deleteMulti(keys, {
      quiet: true // 静默模式，只返回失败的
    });
    
    res.json({
      message: '删除成功',
      deleted: result.deleted?.length || keys.length
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 生成临时下载链接（带签名）
app.get('/api/upload/oss/signed-url', async (req, res) => {
  try {
    const { key } = req.query;
    
    // 生成1小时有效的签名URL
    const url = ossClient.signatureUrl(key, {
      expires: 3600,
      method: 'GET'
    });
    
    res.json({
      url,
      expiresAt: new Date(Date.now() + 3600 * 1000).toISOString()
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 列出所有文件
app.get('/api/upload/oss/list', async (req, res) => {
  try {
    const { prefix = 'uploads/', maxKeys = 100 } = req.query;
    
    const result = await ossClient.list({
      prefix,
      'max-keys': maxKeys
    });
    
    const files = result.objects.map(obj => ({
      name: obj.name,
      url: obj.url,
      size: obj.size,
      lastModified: obj.lastModified
    }));
    
    res.json({
      files,
      total: files.length,
      nextMarker: result.nextMarker
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

```javascript
// aws-s3-integration.js
import { S3Client, PutObjectCommand, GetObjectCommand, DeleteObjectCommand, ListObjectsV2Command } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import express from 'express';
import multer from 'multer';
import crypto from 'crypto';

const app = express();

// 💡 创建S3客户端（AWS SDK v3）
const s3Client = new S3Client({
  region: 'us-east-1',
  credentials: {
    accessKeyId: process.env.AWS_ACCESS_KEY_ID,
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY
  }
});

const BUCKET_NAME = 'my-s3-bucket';
const upload = multer({ storage: multer.memoryStorage() });

// 💡 基础上传
app.post('/api/upload/s3/basic', upload.single('file'), async (req, res) => {
  try {
    const file = req.file;
    const key = `uploads/${Date.now()}-${crypto.randomUUID()}.jpg`;
    
    const command = new PutObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key,
      Body: file.buffer,
      ContentType: file.mimetype,
      // ACL（访问控制）
      ACL: 'public-read',
      // 缓存控制
      CacheControl: 'public, max-age=31536000',
      // 元数据
      Metadata: {
        originalname: file.originalname,
        uploadedBy: 'user123'
      }
    });
    
    await s3Client.send(command);
    
    // 生成公开URL
    const url = `https://${BUCKET_NAME}.s3.amazonaws.com/${key}`;
    
    res.json({
      message: '上传成功',
      url,
      key
    });
  } catch (error) {
    console.error('S3上传失败:', error);
    res.status(500).json({ error: error.message });
  }
});

// 💡 生成预签名URL（上传）
app.post('/api/upload/s3/presigned-upload', async (req, res) => {
  try {
    const { filename, contentType } = req.body;
    const key = `uploads/${Date.now()}-${filename}`;
    
    const command = new PutObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key,
      ContentType: contentType
    });
    
    // 生成15分钟有效的上传URL
    const signedUrl = await getSignedUrl(s3Client, command, {
      expiresIn: 900
    });
    
    res.json({
      uploadUrl: signedUrl,
      key,
      expiresAt: new Date(Date.now() + 900 * 1000).toISOString()
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 生成预签名URL（下载）
app.get('/api/upload/s3/presigned-download', async (req, res) => {
  try {
    const { key } = req.query;
    
    const command = new GetObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key
    });
    
    // 生成1小时有效的下载URL
    const signedUrl = await getSignedUrl(s3Client, command, {
      expiresIn: 3600
    });
    
    res.json({
      downloadUrl: signedUrl,
      expiresAt: new Date(Date.now() + 3600 * 1000).toISOString()
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 列出文件
app.get('/api/upload/s3/list', async (req, res) => {
  try {
    const { prefix = 'uploads/', maxKeys = 100 } = req.query;
    
    const command = new ListObjectsV2Command({
      Bucket: BUCKET_NAME,
      Prefix: prefix,
      MaxKeys: parseInt(maxKeys)
    });
    
    const response = await s3Client.send(command);
    
    const files = response.Contents?.map(obj => ({
      key: obj.Key,
      size: obj.Size,
      lastModified: obj.LastModified,
      url: `https://${BUCKET_NAME}.s3.amazonaws.com/${obj.Key}`
    })) || [];
    
    res.json({
      files,
      total: files.length
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 删除文件
app.delete('/api/upload/s3/delete', async (req, res) => {
  try {
    const { key } = req.query;
    
    const command = new DeleteObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key
    });
    
    await s3Client.send(command);
    
    res.json({ message: '删除成功' });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

## 🔧 错误处理

### 高频错误1：AccessDenied（权限不足）

**现象：**
```
AccessDenied: Access Denied
```

**原因：**
1. AccessKey权限不足
2. Bucket策略限制
3. 跨域配置错误

**解决方案：**

```javascript
// 1. 检查AccessKey权限
// 阿里云：RAM控制台 → 用户 → 添加权限 → AliyunOSSFullAccess
// AWS：IAM → 用户 → 附加策略 → AmazonS3FullAccess

// 2. 配置Bucket跨域（CORS）
// 阿里云OSS示例
await ossClient.putBucketCORS('my-bucket', [{
  allowedOrigin: ['*'],
  allowedMethod: ['GET', 'POST', 'PUT', 'DELETE'],
  allowedHeader: ['*'],
  exposeHeader: ['ETag'],
  maxAgeSeconds: 3600
}]);

// AWS S3 CORS配置
{
  "CORSRules": [{
    "AllowedOrigins": ["*"],
    "AllowedMethods": ["GET", "POST", "PUT", "DELETE"],
    "AllowedHeaders": ["*"],
    "ExposeHeaders": ["ETag"],
    "MaxAgeSeconds": 3600
  }]
}
```

### 场景1：视频网站完整上传方案

**问题背景：**
在线教育平台需要支持用户上传大视频文件（最大5GB），要求支持断点续传、自动转码、CDN加速。

**技术选型理由：**
1. **腾讯云COS**：专注视频场景，有媒体处理服务
2. **分片上传**：支持断点续传
3. **CDN加速**：全球快速访问
4. **媒体处理**：自动转码、截图、水印

### 场景2：图片社交平台（海量图片存储）

**问题背景：**
图片社交App，日均上传100万张图片，需要实时裁剪、滤镜、水印，支持全球访问。

**完整实现：**

```javascript
// image-social-platform.js
import express from 'express';
import multer from 'multer';
import OSS from 'ali-oss';
import sharp from 'sharp';
import crypto from 'crypto';

const app = express();

const ossClient = new OSS({
  region: 'oss-cn-shanghai',
  accessKeyId: process.env.ALI_ACCESS_KEY_ID,
  accessKeySecret: process.env.ALI_ACCESS_KEY_SECRET,
  bucket: 'social-images'
});

const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 20 * 1024 * 1024 } // 20MB
});

// 💡 智能图片上传
app.post('/api/image/upload', upload.single('image'), async (req, res) => {
  try {
    const file = req.file;
    const userId = req.user?.id || 'anonymous';
    
    // 1. 图片验证和优化
    const metadata = await sharp(file.buffer).metadata();
    
    // 2. 生成多尺寸
    const sizes = {
      original: { width: null, quality: 95 },
      large: { width: 1920, quality: 90 },
      medium: { width: 1024, quality: 85 },
      small: { width: 480, quality: 80 },
      thumbnail: { width: 200, quality: 75 }
    };
    
    const imageId = crypto.randomUUID();
    const uploadPromises = [];
    
    for (const [sizeName, config] of Object.entries(sizes)) {
      let pipeline = sharp(file.buffer);
      
      // 调整大小
      if (config.width) {
        pipeline = pipeline.resize(config.width, null, {
          fit: 'inside',
          withoutEnlargement: true
        });
      }
      
      // 转WebP并压缩
      const buffer = await pipeline
        .webp({ quality: config.quality })
        .toBuffer();
      
      // 上传到OSS
      const key = `images/${userId}/${imageId}/${sizeName}.webp`;
      uploadPromises.push(
        ossClient.put(key, buffer, {
          headers: {
            'Cache-Control': 'public, max-age=31536000, immutable',
            'Content-Type': 'image/webp'
          }
        })
      );
    }
    
    await Promise.all(uploadPromises);
    
    // 3. 生成CDN URLs
    const baseUrl = `https://cdn.example.com/images/${userId}/${imageId}`;
    
    res.json({
      message: '上传成功',
      imageId,
      urls: {
        original: `${baseUrl}/original.webp`,
        large: `${baseUrl}/large.webp`,
        medium: `${baseUrl}/medium.webp`,
        small: `${baseUrl}/small.webp`,
        thumbnail: `${baseUrl}/thumbnail.webp`
      },
      originalSize: file.buffer.length,
      width: metadata.width,
      height: metadata.height
    });
  } catch (error) {
    console.error('上传失败:', error);
    res.status(500).json({ error: error.message });
  }
});

// 💡 实时图片处理（OSS图片处理）
app.get('/api/image/transform', (req, res) => {
  const {
    imageId,
    width = 800,
    height = 600,
    mode = 'fit',
    quality = 90,
    format = 'webp',
    watermark = false
  } = req.query;
  
  const key = `images/${imageId}/original.webp`;
  
  // 构建OSS图片处理参数
  let process = `image/resize,w_${width}`;
  
  if (height) {
    process += `,h_${height}`;
  }
  
  if (mode === 'fill') {
    process += ',m_fill';
  }
  
  process += `/quality,q_${quality}/format,${format}`;
  
  // 添加水印
  if (watermark) {
    process += '/watermark,text_QGV4YW1wbGU'; // Base64编码的文本
  }
  
  // 生成处理后的URL
  const url = ossClient.signatureUrl(key, {
    expires: 3600,
    process
  });
  
  res.redirect(url);
});

app.listen(3000);
```

**可量化效果：**
- ✅ 日处理100万张图片
- ✅ 图片大小减少75%（WebP格式）
- ✅ 多尺寸生成，节省客户端流量
- ✅ 实时裁剪，灵活适配
- ✅ CDN加速，全球访问<100ms

## 📋 总结

### 核心要点

1. **三大云平台**：阿里云OSS、AWS S3、腾讯云COS
2. **核心优势**：无限扩展、高可靠、低成本、CDN加速
3. **关键功能**：分片上传、图片处理、预签名URL
4. **选型原则**：国内首选阿里云/腾讯云，全球化选AWS

### 实现关键

1. **分片上传**：大文件必须使用分片，支持断点续传
2. **预签名URL**：前端直传，不占用服务器带宽
3. **CDN加速**：启用自定义域名+CDN
4. **图片优化**：WebP格式 + 多尺寸

### 应用价值

- **降低成本**：相比自建，节省80%+成本
- **提升性能**：CDN加速，全球访问快10倍
- **提高可靠性**：99.9999999%数据持久性
- **减少运维**：零运维，自动扩展

### 面试重点

**Q1：为什么要用云存储？**
答：无限扩展、高可靠（11个9）、低成本（按量付费）、CDN加速、零运维

**Q2：如何实现前端直传？**
答：后端生成预签名URL，前端用该URL直接PUT到云存储，不经过后端服务器

**Q3：大文件如何上传？**
答：使用分片上传，将文件切成多个小块并发上传，支持断点续传

**Q4：三大云平台如何选择？**
答：国内为主选阿里云/腾讯云（速度快），全球化选AWS（覆盖广），视频场景优先腾讯云

### 注意事项

⚠️ **AccessKey安全**：不要硬编码，使用环境变量
⚠️ **费用控制**：设置Bucket防盗链，避免流量被盗用
⚠️ **CDN配置**：开启HTTPS，设置合理缓存策略
⚠️ **备份策略**：重要数据开启跨区域复制
