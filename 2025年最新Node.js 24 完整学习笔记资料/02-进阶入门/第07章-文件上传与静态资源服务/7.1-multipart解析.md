# 7.1 multipart/form-data 解析

> 深入理解multipart协议和原生解析实现

---

## 📖 知识点概述

### 核心概念

**multipart/form-data**是一种HTTP内容类型，用于在表单中上传文件。它将表单数据分成多个部分（part），每个部分可以是文本或文件。

### 为什么需要multipart？

普通表单（application/x-www-form-urlencoded）只能传输文本，无法传输二进制文件。multipart解决了这个问题。

---

## 🔍 multipart协议解析

### HTTP请求示例

```http
POST /upload HTTP/1.1
Host: localhost:3000
Content-Type: multipart/form-data; boundary=----WebKitFormBoundary7MA4YWxkTrZu0gW
Content-Length: 12345

------WebKitFormBoundary7MA4YWxkTrZu0gW
Content-Disposition: form-data; name="username"

John Doe
------WebKitFormBoundary7MA4YWxkTrZu0gW
Content-Disposition: form-data; name="avatar"; filename="photo.jpg"
Content-Type: image/jpeg

<二进制数据>
------WebKitFormBoundary7MA4YWxkTrZu0gW--
```

### 协议结构

```
1. boundary（边界）：分隔不同的part
2. Content-Disposition：描述字段信息
   - name：字段名
   - filename：文件名（如果是文件）
3. Content-Type：文件类型（如果是文件）
4. 空行 + 数据内容
5. 结束boundary：--boundary--
```

## 💡 原生解析实现

### 基础解析器

```javascript
// multipart-parser.js
import { IncomingMessage } from 'http';

class MultipartParser {
  constructor(req) {
    this.req = req;
    this.boundary = this.getBoundary();
  }
  
  // 💡 从Content-Type提取boundary
  getBoundary() {
    const contentType = this.req.headers['content-type'];
    const match = contentType.match(/boundary=(.+)/);
    
    if (!match) {
      throw new Error('No boundary found');
    }
    
    return '--' + match[1];
  }
  
  // 💡 解析multipart数据
  async parse() {
    return new Promise((resolve, reject) => {
      const chunks = [];
      
      this.req.on('data', chunk => {
        chunks.push(chunk);
      });
      
      this.req.on('end', () => {
        try {
          const buffer = Buffer.concat(chunks);
          const parts = this.parseParts(buffer);
          resolve(parts);
        } catch (error) {
          reject(error);
        }
      });
      
      this.req.on('error', reject);
    });
  }
  
  // 💡 分割成多个part
  parseParts(buffer) {
    const boundaryBuffer = Buffer.from(this.boundary);
    const parts = [];
    let start = 0;
    
    while (true) {
      // 查找下一个boundary
      const boundaryIndex = buffer.indexOf(boundaryBuffer, start);
      
      if (boundaryIndex === -1) {
        break;
      }
      
      // 提取part内容
      if (start > 0) {
        const partBuffer = buffer.slice(start, boundaryIndex);
        const part = this.parsePart(partBuffer);
        
        if (part) {
          parts.push(part);
        }
      }
      
      start = boundaryIndex + boundaryBuffer.length + 2; // +2跳过\r\n
      
      // 检查是否是结束boundary
      if (buffer[start - 2] === 45 && buffer[start - 1] === 45) { // --
        break;
      }
    }
    
    return parts;
  }
  
  // 💡 解析单个part
  parsePart(buffer) {
    // 查找空行（\r\n\r\n），分隔header和body
    const headerEndIndex = buffer.indexOf(Buffer.from('\r\n\r\n'));
    
    if (headerEndIndex === -1) {
      return null;
    }
    
    const headerBuffer = buffer.slice(0, headerEndIndex);
    const bodyBuffer = buffer.slice(headerEndIndex + 4, buffer.length - 2); // -2去除结尾\r\n
    
    // 解析header
    const headers = this.parseHeaders(headerBuffer.toString());
    
    // 提取字段信息
    const disposition = headers['content-disposition'];
    const nameMatch = disposition.match(/name="([^"]+)"/);
    const filenameMatch = disposition.match(/filename="([^"]+)"/);
    
    const part = {
      name: nameMatch ? nameMatch[1] : null,
      filename: filenameMatch ? filenameMatch[1] : null,
      contentType: headers['content-type'] || 'text/plain',
      data: bodyBuffer
    };
    
    // 如果不是文件，转换为字符串
    if (!part.filename) {
      part.value = bodyBuffer.toString('utf8');
      delete part.data;
    }
    
    return part;
  }
  
  // 解析headers
  parseHeaders(headerString) {
    const headers = {};
    const lines = headerString.split('\r\n');
    
    lines.forEach(line => {
      const colonIndex = line.indexOf(':');
      if (colonIndex > 0) {
        const key = line.slice(0, colonIndex).trim().toLowerCase();
        const value = line.slice(colonIndex + 1).trim();
        headers[key] = value;
      }
    });
    
    return headers;
  }
}

export default MultipartParser;
```

### 使用原生解析器

```javascript
// server.js
import http from 'http';
import fs from 'fs';
import path from 'path';
import MultipartParser from './multipart-parser.js';

const server = http.createServer(async (req, res) => {
  if (req.method === 'POST' && req.url === '/upload') {
    try {
      // 检查Content-Type
      const contentType = req.headers['content-type'];
      
      if (!contentType || !contentType.includes('multipart/form-data')) {
        res.writeHead(400);
        res.end('Invalid Content-Type');
        return;
      }
      
      // 💡 解析multipart数据
      const parser = new MultipartParser(req);
      const parts = await parser.parse();
      
      console.log('解析结果:', parts.length, '个字段');
      
      // 处理字段
      const fields = {};
      const files = [];
      
      parts.forEach(part => {
        if (part.filename) {
          // 文件
          files.push({
            fieldname: part.name,
            filename: part.filename,
            mimetype: part.contentType,
            size: part.data.length,
            buffer: part.data
          });
          
          // 💡 保存文件
          const uploadPath = path.join('uploads', part.filename);
          fs.writeFileSync(uploadPath, part.data);
          
          console.log('文件已保存:', uploadPath);
        } else {
          // 普通字段
          fields[part.name] = part.value;
        }
      });
      
      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({
        message: '上传成功',
        fields,
        files: files.map(f => ({
          fieldname: f.fieldname,
          filename: f.filename,
          size: f.size
        }))
      }));
    } catch (error) {
      console.error('上传错误:', error);
      res.writeHead(500);
      res.end('Upload failed');
    }
  } else {
    res.writeHead(404);
    res.end('Not Found');
  }
});

server.listen(3000, () => {
  console.log('服务器运行在 http://localhost:3000');
});
```

## 🎯 流式解析（处理大文件）

```javascript
// stream-parser.js
import { EventEmitter } from 'events';

class StreamingMultipartParser extends EventEmitter {
  constructor(req) {
    super();
    this.req = req;
    this.boundary = this.getBoundary();
    this.buffer = Buffer.alloc(0);
    this.state = 'PREAMBLE';
    this.currentPart = null;
  }
  
  getBoundary() {
    const contentType = this.req.headers['content-type'];
    const match = contentType.match(/boundary=(.+)/);
    return '--' + match[1];
  }
  
  parse() {
    this.req.on('data', chunk => {
      this.buffer = Buffer.concat([this.buffer, chunk]);
      this.processBuffer();
    });
    
    this.req.on('end', () => {
      this.emit('finish');
    });
    
    this.req.on('error', error => {
      this.emit('error', error);
    });
  }
  
  // 💡 逐步处理buffer
  processBuffer() {
    while (this.buffer.length > 0) {
      if (this.state === 'PREAMBLE' || this.state === 'BOUNDARY') {
        if (!this.findBoundary()) break;
      } else if (this.state === 'HEADERS') {
        if (!this.parsePartHeaders()) break;
      } else if (this.state === 'DATA') {
        if (!this.extractPartData()) break;
      }
    }
  }
  
  findBoundary() {
    const boundaryIndex = this.buffer.indexOf(this.boundary);
    
    if (boundaryIndex === -1) {
      return false;
    }
    
    this.buffer = this.buffer.slice(boundaryIndex + this.boundary.length + 2);
    this.state = 'HEADERS';
    
    return true;
  }
  
  parsePartHeaders() {
    const headerEndIndex = this.buffer.indexOf('\r\n\r\n');
    
    if (headerEndIndex === -1) {
      return false;
    }
    
    const headerBuffer = this.buffer.slice(0, headerEndIndex);
    this.buffer = this.buffer.slice(headerEndIndex + 4);
    
    // 解析headers
    const headers = {};
    headerBuffer.toString().split('\r\n').forEach(line => {
      const colonIndex = line.indexOf(':');
      if (colonIndex > 0) {
        const key = line.slice(0, colonIndex).trim().toLowerCase();
        const value = line.slice(colonIndex + 1).trim();
        headers[key] = value;
      }
    });
    
    // 提取信息
    const disposition = headers['content-disposition'];
    const nameMatch = disposition.match(/name="([^"]+)"/);
    const filenameMatch = disposition.match(/filename="([^"]+)"/);
    
    this.currentPart = {
      name: nameMatch ? nameMatch[1] : null,
      filename: filenameMatch ? filenameMatch[1] : null,
      contentType: headers['content-type'] || 'text/plain'
    };
    
    // 💡 触发part开始事件
    this.emit('part', this.currentPart);
    
    this.state = 'DATA';
    return true;
  }
  
  extractPartData() {
    const nextBoundaryIndex = this.buffer.indexOf(this.boundary);
    
    if (nextBoundaryIndex === -1) {
      // 还没到下一个boundary，先发送当前数据
      if (this.buffer.length > this.boundary.length + 4) {
        const safeLength = this.buffer.length - this.boundary.length - 4;
        const chunk = this.buffer.slice(0, safeLength);
        
        // 💡 触发数据事件（流式）
        this.emit('data', chunk);
        
        this.buffer = this.buffer.slice(safeLength);
      }
      
      return false;
    }
    
    // 找到下一个boundary
    const data = this.buffer.slice(0, nextBoundaryIndex - 2); // -2去除\r\n
    
    // 💡 发送最后的数据
    if (data.length > 0) {
      this.emit('data', data);
    }
    
    // 💡 part结束
    this.emit('partEnd');
    
    this.buffer = this.buffer.slice(nextBoundaryIndex);
    this.state = 'BOUNDARY';
    
    return true;
  }
}

// 使用流式解析器
const parser = new StreamingMultipartParser(req);

parser.on('part', (part) => {
  console.log('开始接收:', part.filename || part.name);
  
  if (part.filename) {
    // 创建写入流
    const writeStream = fs.createWriteStream(`uploads/${part.filename}`);
    
    parser.on('data', (chunk) => {
      writeStream.write(chunk);
    });
    
    parser.on('partEnd', () => {
      writeStream.end();
      console.log('文件保存完成:', part.filename);
    });
  }
});

parser.parse();
```

## 📋 总结

### 核心要点

1. **multipart协议**：用boundary分隔多个part
2. **原生解析**：理解协议，手动解析
3. **流式解析**：处理大文件，边接收边保存
4. **实际开发**：使用multer等成熟库

### 协议关键

- **boundary**：分隔符，每个请求都不同
- **Content-Disposition**：描述字段信息
- **Content-Type**：文件MIME类型
- **二进制数据**：使用Buffer处理

### 注意事项

⚠️ **生产环境使用成熟库**：手写解析器主要用于学习
⚠️ **大文件用流式解析**：避免内存溢出
⚠️ **验证文件类型和大小**：安全第一
