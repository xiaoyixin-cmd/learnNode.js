# 7.2 multer 2.x 中间件

> 使用multer简化文件上传处理

---

## 📖 知识点概述

**multer**是Node.js中最流行的文件上传中间件，基于busboy构建，专为Express设计。

### 核心功能

- 单文件/多文件上传
- 文件过滤和验证
- 自定义存储引擎
- 内存/磁盘存储

---

## 🚀 基础使用

### 安装

```bash
npm install multer
```

### 单文件上传

```javascript
// upload-single.js
import express from 'express';
import multer from 'multer';
import path from 'path';

const app = express();

// 💡 配置存储
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/'); // 存储目录
  },
  filename: (req, file, cb) => {
    // 💡 生成唯一文件名
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
  }
});

// 💡 创建upload实例
const upload = multer({
  storage: storage,
  limits: {
    fileSize: 5 * 1024 * 1024 // 5MB限制
  },
  fileFilter: (req, file, cb) => {
    // 💡 文件类型过滤
    const allowedTypes = /jpeg|jpg|png|gif/;
    const extname = allowedTypes.test(path.extname(file.originalname).toLowerCase());
    const mimetype = allowedTypes.test(file.mimetype);
    
    if (extname && mimetype) {
      cb(null, true);
    } else {
      cb(new Error('只允许上传图片文件'));
    }
  }
});

// 💡 单文件上传路由
app.post('/upload', upload.single('avatar'), (req, res) => {
  // req.file包含文件信息
  console.log('上传文件:', req.file);
  console.log('表单字段:', req.body);
  
  res.json({
    message: '上传成功',
    file: {
      filename: req.file.filename,
      originalname: req.file.originalname,
      size: req.file.size,
      path: req.file.path
    }
  });
});

// 💡 错误处理
app.use((err, req, res, next) => {
  if (err instanceof multer.MulterError) {
    if (err.code === 'LIMIT_FILE_SIZE') {
      return res.status(400).json({ error: '文件太大，最大5MB' });
    }
    if (err.code === 'LIMIT_UNEXPECTED_FILE') {
      return res.status(400).json({ error: '意外的文件字段' });
    }
  }
  
  res.status(500).json({ error: err.message });
});

app.listen(3000);
```

### 多文件上传

```javascript
// upload-multiple.js

// 💡 多个文件（同一字段名）
app.post('/upload/photos', upload.array('photos', 5), (req, res) => {
  // req.files是数组
  console.log('上传文件数:', req.files.length);
  
  res.json({
    message: '上传成功',
    files: req.files.map(f => ({
      filename: f.filename,
      size: f.size
    }))
  });
});

// 💡 多个字段
const uploadFields = upload.fields([
  { name: 'avatar', maxCount: 1 },
  { name: 'photos', maxCount: 5 }
]);

app.post('/upload/mixed', uploadFields, (req, res) => {
  // req.files是对象
  console.log('avatar:', req.files.avatar);
  console.log('photos:', req.files.photos);
  
  res.json({
    message: '上传成功',
    files: {
      avatar: req.files.avatar?.[0]?.filename,
      photos: req.files.photos?.map(f => f.filename)
    }
  });
});

// 💡 任意文件
app.post('/upload/any', upload.any(), (req, res) => {
  // req.files是数组，包含所有文件
  res.json({
    files: req.files.map(f => ({
      fieldname: f.fieldname,
      filename: f.filename
    }))
  });
});
```

### 自定义存储引擎

```javascript
// custom-storage.js
import multer from 'multer';
import crypto from 'crypto';
import path from 'path';
import fs from 'fs/promises';

// 💡 自定义存储引擎
class CustomStorage {
  constructor(options) {
    this.destination = options.destination || 'uploads';
  }
  
  async _handleFile(req, file, cb) {
    try {
      // 生成hash文件名
      const hash = crypto.createHash('md5');
      const writeStream = fs.createWriteStream(path.join(this.destination, 'temp'));
      
      file.stream.pipe(writeStream);
      file.stream.on('data', chunk => hash.update(chunk));
      
      file.stream.on('end', async () => {
        const filename = hash.digest('hex') + path.extname(file.originalname);
        const finalPath = path.join(this.destination, filename);
        
        // 重命名临时文件
        await fs.rename(
          path.join(this.destination, 'temp'),
          finalPath
        );
        
        cb(null, {
          path: finalPath,
          size: writeStream.bytesWritten,
          filename: filename
        });
      });
      
      file.stream.on('error', cb);
    } catch (error) {
      cb(error);
    }
  }
  
  async _removeFile(req, file, cb) {
    try {
      await fs.unlink(file.path);
      cb(null);
    } catch (error) {
      cb(error);
    }
  }
}

// 使用自定义存储
const customUpload = multer({
  storage: new CustomStorage({ destination: 'uploads' })
});

app.post('/upload/custom', customUpload.single('file'), (req, res) => {
  res.json({ file: req.file });
});
```

## 🎯 实战：完整上传系统

```javascript
// complete-upload.js
import express from 'express';
import multer from 'multer';
import path from 'path';
import fs from 'fs/promises';
import sharp from 'sharp'; // 图片处理

const app = express();

// 💡 确保目录存在
await fs.mkdir('uploads', { recursive: true });
await fs.mkdir('uploads/images', { recursive: true });
await fs.mkdir('uploads/documents', { recursive: true });

// 💡 配置存储
const storage = multer.diskStorage({
  destination: async (req, file, cb) => {
    // 根据文件类型存储到不同目录
    const isImage = file.mimetype.startsWith('image/');
    const dir = isImage ? 'uploads/images' : 'uploads/documents';
    cb(null, dir);
  },
  filename: (req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
  }
});

// 💡 文件过滤
const fileFilter = (req, file, cb) => {
  // 允许的文件类型
  const allowedMimes = [
    'image/jpeg',
    'image/png',
    'image/gif',
    'application/pdf',
    'application/msword',
    'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
  ];
  
  if (allowedMimes.includes(file.mimetype)) {
    cb(null, true);
  } else {
    cb(new Error('不支持的文件类型'));
  }
};

const upload = multer({
  storage: storage,
  fileFilter: fileFilter,
  limits: {
    fileSize: 10 * 1024 * 1024, // 10MB
    files: 5 // 最多5个文件
  }
});

// 💡 上传并处理图片
app.post('/upload/image', upload.single('image'), async (req, res) => {
  try {
    const file = req.file;
    
    // 生成缩略图
    const thumbnailPath = file.path.replace(
      path.extname(file.path),
      '-thumb' + path.extname(file.path)
    );
    
    await sharp(file.path)
      .resize(200, 200, {
        fit: 'cover',
        position: 'center'
      })
      .toFile(thumbnailPath);
    
    res.json({
      message: '上传成功',
      file: {
        url: `/uploads/images/${file.filename}`,
        thumbnail: `/uploads/images/${path.basename(thumbnailPath)}`,
        size: file.size
      }
    });
  } catch (error) {
    console.error('图片处理失败:', error);
    res.status(500).json({ error: '图片处理失败' });
  }
});

// 💡 批量上传
app.post('/upload/batch', upload.array('files', 10), async (req, res) => {
  const results = [];
  
  for (const file of req.files) {
    results.push({
      originalname: file.originalname,
      filename: file.filename,
      size: file.size,
      url: `/uploads/${path.basename(path.dirname(file.path))}/${file.filename}`
    });
  }
  
  res.json({
    message: `成功上传${req.files.length}个文件`,
    files: results
  });
});

// 静态文件服务
app.use('/uploads', express.static('uploads'));

app.listen(3000);
```

## 📋 总结

### 核心要点

1. **multer**：Express文件上传中间件
2. **storage**：磁盘存储 or 内存存储
3. **fileFilter**：文件类型过滤
4. **limits**：文件大小和数量限制

### 使用方法

- **单文件**：`upload.single('fieldname')`
- **多文件（同字段）**：`upload.array('fieldname', maxCount)`
- **多字段**：`upload.fields([{name, maxCount}])`
- **任意文件**：`upload.any()`

### 注意事项

⚠️ **设置文件大小限制**
⚠️ **验证文件类型**
⚠️ **使用唯一文件名**
⚠️ **处理上传错误**
