# 7.3 大文件分片上传

> 实现断点续传和秒传功能

---

## 📖 知识点概述

**分片上传**：将大文件切分成多个小块，分别上传后再合并。

### 优势

- 支持断点续传
- 提升上传成功率
- 支持并发上传
- 可实现秒传

---

## 🚀 完整实现

### 前端：文件分片

```javascript
// client.js
class ChunkUploader {
  constructor(file, options = {}) {
    this.file = file;
    this.chunkSize = options.chunkSize || 2 * 1024 * 1024; // 2MB
    this.chunks = Math.ceil(file.size / this.chunkSize);
    this.uploadedChunks = new Set();
  }
  
  // 💡 计算文件hash（用于秒传）
  async calculateHash() {
    return new Promise((resolve) => {
      const spark = new SparkMD5.ArrayBuffer();
      const fileReader = new FileReader();
      let currentChunk = 0;
      
      const loadNext = () => {
        const start = currentChunk * this.chunkSize;
        const end = Math.min(start + this.chunkSize, this.file.size);
        
        fileReader.readAsArrayBuffer(this.file.slice(start, end));
      };
      
      fileReader.onload = (e) => {
        spark.append(e.target.result);
        currentChunk++;
        
        if (currentChunk < this.chunks) {
          loadNext();
        } else {
          resolve(spark.end());
        }
      };
      
      loadNext();
    });
  }
  
  // 💡 上传单个分片
  async uploadChunk(index) {
    const start = index * this.chunkSize;
    const end = Math.min(start + this.chunkSize, this.file.size);
    const chunk = this.file.slice(start, end);
    
    const formData = new FormData();
    formData.append('chunk', chunk);
    formData.append('index', index);
    formData.append('hash', this.fileHash);
    formData.append('filename', this.file.name);
    formData.append('total', this.chunks);
    
    const response = await fetch('/upload/chunk', {
      method: 'POST',
      body: formData
    });
    
    if (response.ok) {
      this.uploadedChunks.add(index);
      this.onProgress?.(this.uploadedChunks.size, this.chunks);
      return true;
    }
    
    return false;
  }
  
  // 💡 开始上传
  async upload() {
    // 1. 计算文件hash
    this.fileHash = await this.calculateHash();
    
    // 2. 检查是否已上传（秒传）
    const checkResponse = await fetch('/upload/check', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        hash: this.fileHash,
        filename: this.file.name
      })
    });
    
    const checkData = await checkResponse.json();
    
    if (checkData.exists) {
      this.onComplete?.(checkData.url);
      return checkData.url;
    }
    
    // 3. 获取已上传的分片
    this.uploadedChunks = new Set(checkData.uploadedChunks || []);
    
    // 4. 上传未完成的分片
    const tasks = [];
    for (let i = 0; i < this.chunks; i++) {
      if (!this.uploadedChunks.has(i)) {
        tasks.push(this.uploadChunk(i));
      }
    }
    
    await Promise.all(tasks);
    
    // 5. 通知服务器合并
    const mergeResponse = await fetch('/upload/merge', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        hash: this.fileHash,
        filename: this.file.name,
        total: this.chunks
      })
    });
    
    const mergeData = await mergeResponse.json();
    this.onComplete?.(mergeData.url);
    
    return mergeData.url;
  }
}

// 使用
const uploader = new ChunkUploader(file);

uploader.onProgress = (uploaded, total) => {
  const percent = (uploaded / total * 100).toFixed(2);
  console.log(`上传进度: ${percent}%`);
};

uploader.onComplete = (url) => {
  console.log('上传完成:', url);
};

await uploader.upload();
```

### 后端：接收和合并

```javascript
// chunk-server.js
import express from 'express';
import multer from 'multer';
import fs from 'fs/promises';
import path from 'path';

const app = express();
app.use(express.json());

const UPLOAD_DIR = 'uploads';
const CHUNK_DIR = 'uploads/chunks';

// 确保目录存在
await fs.mkdir(UPLOAD_DIR, { recursive: true });
await fs.mkdir(CHUNK_DIR, { recursive: true });

const upload = multer({ dest: CHUNK_DIR });

// 💡 检查文件是否存在（秒传）
app.post('/upload/check', async (req, res) => {
  try {
    const { hash, filename } = req.body;
    const filePath = path.join(UPLOAD_DIR, `${hash}${path.extname(filename)}`);
    
    // 检查文件是否存在
    try {
      await fs.access(filePath);
      return res.json({
        exists: true,
        url: `/uploads/${path.basename(filePath)}`
      });
    } catch {
      // 文件不存在，检查已上传的分片
      const chunkDir = path.join(CHUNK_DIR, hash);
      let uploadedChunks = [];
      
      try {
        const files = await fs.readdir(chunkDir);
        uploadedChunks = files.map(f => parseInt(f));
      } catch {
        // 目录不存在
      }
      
      res.json({
        exists: false,
        uploadedChunks
      });
    }
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 接收分片
app.post('/upload/chunk', upload.single('chunk'), async (req, res) => {
  try {
    const { index, hash } = req.body;
    const chunkDir = path.join(CHUNK_DIR, hash);
    
    // 创建分片目录
    await fs.mkdir(chunkDir, { recursive: true });
    
    // 移动分片到指定位置
    const chunkPath = path.join(chunkDir, index.toString());
    await fs.rename(req.file.path, chunkPath);
    
    res.json({ message: '分片上传成功' });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 💡 合并分片
app.post('/upload/merge', async (req, res) => {
  try {
    const { hash, filename, total } = req.body;
    const chunkDir = path.join(CHUNK_DIR, hash);
    const ext = path.extname(filename);
    const finalPath = path.join(UPLOAD_DIR, `${hash}${ext}`);
    
    // 创建写入流
    const writeStream = fs.createWriteStream(finalPath);
    
    // 💡 按顺序合并分片
    for (let i = 0; i < total; i++) {
      const chunkPath = path.join(chunkDir, i.toString());
      const chunkBuffer = await fs.readFile(chunkPath);
      writeStream.write(chunkBuffer);
    }
    
    writeStream.end();
    
    await new Promise((resolve, reject) => {
      writeStream.on('finish', resolve);
      writeStream.on('error', reject);
    });
    
    // 💡 删除分片目录
    await fs.rm(chunkDir, { recursive: true, force: true });
    
    res.json({
      message: '合并成功',
      url: `/uploads/${path.basename(finalPath)}`
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.use('/uploads', express.static(UPLOAD_DIR));
app.listen(3000);
```
