# 7.4 静态资源服务与CDN集成

> 构建高性能静态资源服务系统

## 📖 知识点概述

### 核心概念

**静态资源**指不需要服务器动态生成的文件，如HTML、CSS、JavaScript、图片、字体等。与动态内容相比，静态资源的特点是内容固定、可缓存、可通过CDN分发。

### 解决的问题

- **加载速度慢**：通过缓存和CDN加速
- **服务器压力大**：静态资源消耗大量带宽和CPU
- **用户体验差**：首屏加载时间过长
- **成本高**：带宽费用、服务器费用

### 主要特点

1. **高并发**：静态资源请求量通常占总请求的80%
2. **可缓存**：内容变化频率低，适合长期缓存
3. **可分发**：适合通过CDN全球分发
4. **低成本**：相比动态内容，处理成本低

### 典型应用场景

- **网站资源**：CSS、JS、图片、字体
- **文件下载**：软件安装包、文档
- **媒体文件**：音频、视频
- **用户上传**：头像、附件

#### 第二步：缓存策略配置

```javascript
// static-cache.js
import express from 'express';
import path from 'path';

const app = express();

// 💡 高级缓存配置
app.use('/static', express.static('public', {
  // 缓存时间
  maxAge: '1y', // 1年 = 31536000000ms
  
  // 是否发送ETag（用于协商缓存）
  etag: true,
  
  // 是否发送Last-Modified
  lastModified: true,
  
  // 是否启用immutable（告诉浏览器文件永不改变）
  immutable: true,
  
  // 索引文件
  index: ['index.html', 'index.htm'],
  
  // 是否重定向（/dir 自动跳转到 /dir/）
  redirect: true,
  
  // 自定义响应头
  setHeaders: (res, filePath, stat) => {
    // 💡 根据文件类型设置不同缓存策略
    
    // HTML文件：不缓存（always fetch）
    if (filePath.endsWith('.html')) {
      res.setHeader('Cache-Control', 'no-cache, no-store, must-revalidate');
      res.setHeader('Pragma', 'no-cache');
      res.setHeader('Expires', '0');
    }
    
    // JS和CSS：带版本号的可以长期缓存
    else if (filePath.match(/\.(js|css)$/)) {
      // 检查是否有hash（如app.abc123.js）
      const hasHash = /\.[a-f0-9]{8,}\.(js|css)$/.test(filePath);
      
      if (hasHash) {
        // 有hash版本号，强缓存1年
        res.setHeader('Cache-Control', 'public, max-age=31536000, immutable');
      } else {
        // 无版本号，协商缓存
        res.setHeader('Cache-Control', 'public, max-age=0, must-revalidate');
      }
    }
    
    // 图片和字体：长期缓存
    else if (filePath.match(/\.(png|jpg|jpeg|gif|svg|ico|woff|woff2|ttf|eot)$/)) {
      res.setHeader('Cache-Control', 'public, max-age=31536000, immutable');
    }
    
    // 其他文件：短期缓存
    else {
      res.setHeader('Cache-Control', 'public, max-age=3600'); // 1小时
    }
    
    // 💡 安全头
    res.setHeader('X-Content-Type-Options', 'nosniff');
  }
}));

app.listen(3000);
```

**缓存策略解释：**

```javascript
// 💡 Cache-Control指令详解

// no-cache：必须先与服务器确认资源是否变化（协商缓存）
'Cache-Control: no-cache'

// no-store：完全不缓存
'Cache-Control: no-store'

// public：可被任何缓存存储（CDN、代理、浏览器）
'Cache-Control: public'

// private：只能被浏览器缓存，不能被CDN缓存
'Cache-Control: private'

// max-age=3600：缓存3600秒（1小时）
'Cache-Control: public, max-age=3600'

// immutable：告诉浏览器文件永不改变，不要发起条件请求
'Cache-Control: public, max-age=31536000, immutable'

// must-revalidate：过期后必须验证
'Cache-Control: public, max-age=3600, must-revalidate'
```

#### 第三步：压缩优化

```javascript
// static-compression.js
import express from 'express';
import compression from 'compression';

const app = express();

// 💡 启用Gzip压缩
app.use(compression({
  // 压缩级别（0-9，9最高）
  level: 6,
  
  // 只压缩大于1KB的文件
  threshold: 1024,
  
  // 过滤函数：决定是否压缩
  filter: (req, res) => {
    // 如果请求头明确说不压缩，则不压缩
    if (req.headers['x-no-compression']) {
      return false;
    }
    
    // 否则使用compression的默认过滤器
    return compression.filter(req, res);
  }
}));

// 💡 静态文件服务
app.use('/static', express.static('public', {
  maxAge: '1y',
  setHeaders: (res, filePath) => {
    // 告诉CDN这个文件适合压缩
    if (filePath.match(/\.(html|css|js|json|xml|svg)$/)) {
      res.setHeader('Vary', 'Accept-Encoding');
    }
  }
}));

app.listen(3000);
```

#### 第四步：高级功能（范围请求、防盗链）

```javascript
// static-advanced.js
import express from 'express';
import fs from 'fs';
import path from 'path';

const app = express();

// 💡 范围请求支持（用于视频、音频、大文件下载）
app.get('/download/:filename', (req, res) => {
  const filename = req.params.filename;
  const filePath = path.join(process.cwd(), 'files', filename);
  
  // 检查文件是否存在
  if (!fs.existsSync(filePath)) {
    return res.status(404).send('File not found');
  }
  
  const stat = fs.statSync(filePath);
  const fileSize = stat.size;
  const range = req.headers.range;
  
  if (range) {
    // 💡 解析Range头
    const parts = range.replace(/bytes=/, '').split('-');
    const start = parseInt(parts[0], 10);
    const end = parts[1] ? parseInt(parts[1], 10) : fileSize - 1;
    const chunkSize = (end - start) + 1;
    
    // 创建读取流
    const fileStream = fs.createReadStream(filePath, { start, end });
    
    // 设置206 Partial Content响应
    res.status(206);
    res.setHeader('Content-Range', `bytes ${start}-${end}/${fileSize}`);
    res.setHeader('Accept-Ranges', 'bytes');
    res.setHeader('Content-Length', chunkSize);
    res.setHeader('Content-Type', 'application/octet-stream');
    
    fileStream.pipe(res);
  } else {
    // 不支持范围请求，返回整个文件
    res.setHeader('Content-Length', fileSize);
    res.setHeader('Content-Type', 'application/octet-stream');
    
    const fileStream = fs.createReadStream(filePath);
    fileStream.pipe(res);
  }
});

// 💡 防盗链
app.use('/protected', (req, res, next) => {
  const referer = req.headers.referer || req.headers.referrer;
  
  // 允许的来源域名
  const allowedDomains = [
    'localhost',
    'yourdomain.com',
    'www.yourdomain.com'
  ];
  
  if (!referer) {
    // 没有referer（直接访问），可以选择允许或拒绝
    return res.status(403).send('Access denied: No referer');
  }
  
  const refererUrl = new URL(referer);
  const refererHost = refererUrl.hostname;
  
  if (allowedDomains.includes(refererHost)) {
    next();
  } else {
    res.status(403).send('Access denied: Invalid referer');
  }
});

app.use('/protected', express.static('protected'));

app.listen(3000);
```

```javascript
// enterprise-static.js
import express from 'express';
import compression from 'compression';
import OSS from 'ali-oss';
import multer from 'multer';
import sharp from 'sharp';
import path from 'path';

const app = express();

// 1. 💡 Gzip压缩
app.use(compression({
  level: 6,
  threshold: 1024
}));

// 2. 💡 本地静态资源（带缓存策略）
app.use('/static', express.static('public', {
  maxAge: '1y',
  immutable: true,
  etag: true,
  setHeaders: (res, filePath) => {
    if (filePath.match(/\.(js|css)$/)) {
      // JS/CSS：检查是否有hash
      const hasHash = /\.[a-f0-9]{8,}\.(js|css)$/.test(filePath);
      res.setHeader(
        'Cache-Control',
        hasHash ? 'public, max-age=31536000, immutable' : 'public, max-age=3600'
      );
    } else if (filePath.match(/\.(png|jpg|jpeg|gif|svg)$/)) {
      // 图片：长期缓存
      res.setHeader('Cache-Control', 'public, max-age=31536000, immutable');
    } else if (filePath.endsWith('.html')) {
      // HTML：不缓存
      res.setHeader('Cache-Control', 'no-cache');
    }
    
    // CORS
    res.setHeader('Access-Control-Allow-Origin', '*');
  }
}));

// 3. 💡 OSS客户端配置
const ossClient = new OSS({
  region: 'oss-cn-hangzhou',
  accessKeyId: process.env.OSS_ACCESS_KEY_ID,
  accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
  bucket: 'my-static-assets',
  // 自定义域名（CDN加速域名）
  cname: true,
  endpoint: 'cdn.example.com'
});

// 4. 💡 图片上传并优化
const upload = multer({ storage: multer.memoryStorage() });

app.post('/api/upload/image', upload.single('image'), async (req, res) => {
  try {
    const file = req.file;
    
    // 图片优化（压缩、转webp）
    const optimized = await sharp(file.buffer)
      .resize(1920, 1080, { fit: 'inside', withoutEnlargement: true })
      .webp({ quality: 85 })
      .toBuffer();
    
    // 生成文件名（带hash）
    const crypto = await import('crypto');
    const hash = crypto.createHash('md5').update(optimized).digest('hex').slice(0, 8);
    const filename = `images/${Date.now()}-${hash}.webp`;
    
    // 上传到OSS
    const result = await ossClient.put(filename, optimized, {
      headers: {
        'Cache-Control': 'public, max-age=31536000, immutable',
        'Content-Type': 'image/webp'
      }
    });
    
    // CDN URL
    const cdnUrl = `https://cdn.example.com/${filename}`;
    
    res.json({
      message: '上传成功',
      url: cdnUrl,
      originalSize: file.buffer.length,
      optimizedSize: optimized.length,
      savings: `${((1 - optimized.length / file.buffer.length) * 100).toFixed(2)}%`
    });
  } catch (error) {
    console.error('上传失败:', error);
    res.status(500).json({ error: error.message });
  }
});

// 5. 💡 批量迁移到OSS
app.post('/api/migrate-to-oss', async (req, res) => {
  try {
    const fs = await import('fs/promises');
    const files = await fs.readdir('public/images');
    
    const results = [];
    
    for (const file of files) {
      const filePath = path.join('public/images', file);
      const fileBuffer = await fs.readFile(filePath);
      
      const ossPath = `images/${file}`;
      await ossClient.put(ossPath, fileBuffer);
      
      results.push({
        filename: file,
        ossUrl: `https://cdn.example.com/${ossPath}`
      });
    }
    
    res.json({
      message: `成功迁移${results.length}个文件`,
      files: results
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);
```

### 场景2：视频网站点播系统

**问题背景：**
在线教育平台有10000+视频文件（总大小5TB），用户观看时需要支持拖动进度条、倍速播放、清晰度切换。

**技术选型理由：**
1. **范围请求（Range）**：支持断点续传和进度条拖动
2. **HLS/DASH**：自适应码率流媒体
3. **腾讯云COS**：视频存储和CDN加速
4. **防盗链**：保护视频资源

**完整实现：**

```javascript
// video-streaming.js
import express from 'express';
import fs from 'fs';
import path from 'path';
import { createHmac } from 'crypto';

const app = express();

// 1. 💡 视频范围请求（支持进度条拖动）
app.get('/video/:filename', (req, res) => {
  const filename = req.params.filename;
  const videoPath = path.join(process.cwd(), 'videos', filename);
  
  // 检查文件
  if (!fs.existsSync(videoPath)) {
    return res.status(404).send('Video not found');
  }
  
  const stat = fs.statSync(videoPath);
  const fileSize = stat.size;
  const range = req.headers.range;
  
  if (range) {
    // 💡 解析Range请求
    const parts = range.replace(/bytes=/, '').split('-');
    const start = parseInt(parts[0], 10);
    const end = parts[1] ? parseInt(parts[1], 10) : fileSize - 1;
    const chunkSize = (end - start) + 1;
    
    // 防止读取超出文件大小
    if (start >= fileSize || end >= fileSize) {
      res.status(416).setHeader('Content-Range', `bytes */${fileSize}`);
      return res.end();
    }
    
    // 创建流
    const fileStream = fs.createReadStream(videoPath, { start, end });
    
    // 设置响应头
    res.status(206); // Partial Content
    res.setHeader('Content-Range', `bytes ${start}-${end}/${fileSize}`);
    res.setHeader('Accept-Ranges', 'bytes');
    res.setHeader('Content-Length', chunkSize);
    res.setHeader('Content-Type', 'video/mp4');
    
    // 缓存策略
    res.setHeader('Cache-Control', 'public, max-age=86400'); // 1天
    
    fileStream.pipe(res);
  } else {
    // 无Range请求，返回整个文件
    res.setHeader('Content-Length', fileSize);
    res.setHeader('Content-Type', 'video/mp4');
    res.setHeader('Accept-Ranges', 'bytes');
    
    fs.createReadStream(videoPath).pipe(res);
  }
});

// 2. 💡 防盗链（时间戳+签名）
function generateSecureUrl(resourcePath, expiresIn = 3600) {
  const secret = process.env.SECRET_KEY || 'your-secret-key';
  const timestamp = Math.floor(Date.now() / 1000) + expiresIn;
  
  // 生成签名：HMAC-SHA256(资源路径 + 过期时间)
  const data = `${resourcePath}${timestamp}`;
  const signature = createHmac('sha256', secret).update(data).digest('hex');
  
  return {
    url: `/secure${resourcePath}?t=${timestamp}&s=${signature}`,
    expiresAt: new Date(timestamp * 1000).toISOString()
  };
}

// 验证签名中间件
function verifySignature(req, res, next) {
  const { t, s } = req.query;
  const resourcePath = req.path.replace('/secure', '');
  
  if (!t || !s) {
    return res.status(403).send('Missing signature');
  }
  
  // 检查是否过期
  const timestamp = parseInt(t);
  const now = Math.floor(Date.now() / 1000);
  
  if (timestamp < now) {
    return res.status(403).send('Link expired');
  }
  
  // 验证签名
  const secret = process.env.SECRET_KEY || 'your-secret-key';
  const data = `${resourcePath}${timestamp}`;
  const expectedSignature = createHmac('sha256', secret).update(data).digest('hex');
  
  if (s !== expectedSignature) {
    return res.status(403).send('Invalid signature');
  }
  
  next();
}

// 安全视频路由
app.get('/secure/video/:filename', verifySignature, (req, res) => {
  const filename = req.params.filename;
  const videoPath = path.join(process.cwd(), 'videos', filename);
  
  // ... 同上面的范围请求处理
  
  // 简化版：直接返回文件
  if (fs.existsSync(videoPath)) {
    res.sendFile(videoPath);
  } else {
    res.status(404).send('Video not found');
  }
});

// 3. 💡 生成安全URL API
app.post('/api/video/get-url', (req, res) => {
  const { filename } = req.body;
  
  // 验证用户权限（省略）
  // ...
  
  // 生成安全URL（1小时有效）
  const secureUrl = generateSecureUrl(`/video/${filename}`, 3600);
  
  res.json(secureUrl);
});

// 4. 💡 HLS播放列表（可选）
app.get('/hls/:videoId/:file', (req, res) => {
  const { videoId, file } = req.params;
  const hlsPath = path.join(process.cwd(), 'hls', videoId, file);
  
  if (!fs.existsSync(hlsPath)) {
    return res.status(404).send('Not found');
  }
  
  // 根据文件类型设置Content-Type
  if (file.endsWith('.m3u8')) {
    res.setHeader('Content-Type', 'application/vnd.apple.mpegurl');
    res.setHeader('Cache-Control', 'no-cache');
  } else if (file.endsWith('.ts')) {
    res.setHeader('Content-Type', 'video/MP2T');
    res.setHeader('Cache-Control', 'public, max-age=31536000');
  }
  
  res.sendFile(hlsPath);
});

app.listen(3000);
```

### 性能优化实战

#### 优化前：未优化的静态服务

```javascript
// ❌ 性能差的写法
import express from 'express';

const app = express();

// 没有压缩
// 没有缓存
// 没有CDN
app.use(express.static('public'));

app.listen(3000);
```

**问题分析：**
- ❌ 无Gzip压缩，传输大小是原始大小
- ❌ 无缓存策略，每次都重新下载
- ❌ 单服务器，带宽瓶颈
- ❌ 无CDN，全球访问慢

**性能指标（Before）：**
- 首屏加载时间：5.2s
- 总传输大小：3.5MB
- 请求数：45
- LCP（Largest Contentful Paint）：4.8s
- 带宽成本：1200元/月（10Mbps）

#### 优化后：完整优化方案

```javascript
// ✅ 高性能静态服务
import express from 'express';
import compression from 'compression';
import helmet from 'helmet';

const app = express();

// 1. 安全头
app.use(helmet());

// 2. Gzip压缩
app.use(compression({
  level: 6,
  threshold: 1024
}));

// 3. 智能缓存策略
app.use('/static', express.static('public', {
  maxAge: '1y',
  immutable: true,
  etag: true,
  setHeaders: (res, filePath) => {
    // HTML不缓存
    if (filePath.endsWith('.html')) {
      res.setHeader('Cache-Control', 'no-cache');
    }
    // 带hash的资源强缓存
    else if (/\.[a-f0-9]{8,}\.(js|css|png|jpg)$/.test(filePath)) {
      res.setHeader('Cache-Control', 'public, max-age=31536000, immutable');
    }
    // 其他资源短期缓存
    else {
      res.setHeader('Cache-Control', 'public, max-age=3600');
    }
    
    // CORS
    res.setHeader('Access-Control-Allow-Origin', '*');
    
    // 安全头
    res.setHeader('X-Content-Type-Options', 'nosniff');
  }
}));

// 4. CDN集成（在Nginx或CDN服务商配置）
// - 静态资源域名：cdn.example.com
// - 回源地址：origin.example.com
// - 缓存规则：遵循Cache-Control

app.listen(3000);
```

**Nginx配置（CDN回源）：**

```nginx
# /etc/nginx/sites-available/static

server {
    listen 80;
    server_name origin.example.com;

    # Gzip压缩
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/xml+rss application/json;

    # 静态文件缓存
    location /static/ {
        root /var/www;
        expires 1y;
        add_header Cache-Control "public, immutable";
        
        # CDN回源优化
        add_header X-Cache-Status $upstream_cache_status;
    }

    # 代理到Node.js
    location / {
        proxy_pass http://localhost:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
```

**优化关键：**
1. **Gzip压缩**：文本文件减少70%

## 📋 总结

### 核心要点

1. **express.static**：Express内置静态文件中间件
2. **缓存策略**：根据文件类型设置不同的Cache-Control
3. **Gzip压缩**：减少70%传输大小
4. **范围请求**：支持视频拖动和断点续传

### 实现关键

1. **路径配置**：使用绝对路径避免错误
2. **缓存分类**：HTML不缓存，带hash资源强缓存
3. **压缩优化**：启用compression中间件
4. **安全加固**：防盗链、CORS、安全头

### 应用价值

- **提升性能**：首屏加载时间减少70%+
- **降低成本**：带宽成本减少80%+
- **改善体验**：全球访问速度<100ms
- **保护资源**：防盗链有效率99%+

### 面试重点

**Q1：如何优化静态资源加载速度？**
答：Gzip压缩、长期缓存、CDN加速、资源合并、图片优化（WebP、懒加载）

**Q2：Cache-Control的max-age和Expires有什么区别？**
答：max-age是HTTP/1.1标准，优先级高；Expires是HTTP/1.0标准，使用绝对时间

**Q3：如何实现视频拖动进度条？**
答：使用HTTP Range请求，服务器返回206 Partial Content

**Q4：如何防止静态资源被盗链？**
答：Referer验证、时间戳+签名、Token鉴权

### 注意事项

⚠️ **HTML文件不要长期缓存**：用户无法获取最新内容
⚠️ **带版本号的资源才能强缓存**：否则更新困难
⚠️ **CDN配置注意回源策略**：避免频繁回源
⚠️ **大文件使用流式传输**：避免内存溢出
