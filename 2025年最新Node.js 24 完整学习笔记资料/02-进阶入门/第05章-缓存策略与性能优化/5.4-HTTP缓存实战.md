# 5.4 HTTP 缓存实战

> 掌握Cache-Control、ETag、Last-Modified等HTTP缓存机制

---

## 📖 知识点概述

### 核心概念

**HTTP缓存**：通过HTTP响应头控制浏览器和CDN的缓存行为，减少网络请求，提升加载速度。

### 缓存类型

- **强缓存**：直接使用缓存，不请求服务器（Cache-Control、Expires）
- **协商缓存**：询问服务器缓存是否有效（ETag、Last-Modified）

### 缓存流程

```
用户请求
    ↓
浏览器检查强缓存
    ↓
[未过期] → 直接使用缓存（200 from cache）
    ↓
[已过期] → 发送请求（带If-None-Match或If-Modified-Since）
    ↓
服务器验证
    ↓
[未变化] → 304 Not Modified（使用缓存）
    ↓
[已变化] → 200 OK（返回新数据）
```

## 🎯 HTTP缓存头详解

### 1. Cache-Control（现代推荐）

```javascript
// cache-control.js
import express from 'express';

const app = express();

// ===== 强缓存示例 =====

// 💡 静态资源：长期缓存
app.get('/assets/logo.png', (req, res) => {
  res.set('Cache-Control', 'public, max-age=31536000, immutable');
  res.sendFile('./logo.png');
});

// 💡 HTML页面：不缓存或短期缓存
app.get('/', (req, res) => {
  res.set('Cache-Control', 'no-cache, no-store, must-revalidate');
  res.send('<html>...</html>');
});

// 💡 API数据：短期缓存
app.get('/api/products', (req, res) => {
  res.set('Cache-Control', 'public, max-age=300'); // 5分钟
  res.json({ products: [] });
});

// 💡 用户数据：私有缓存
app.get('/api/user/profile', (req, res) => {
  res.set('Cache-Control', 'private, max-age=600'); // 10分钟，仅浏览器缓存
  res.json({ user: {} });
});
```

**Cache-Control 指令**：

| 指令 | 说明 | 适用场景 |
|------|------|---------|
| `public` | 可被任何缓存（浏览器、CDN）缓存 | 公开资源 |
| `private` | 只能被浏览器缓存，不能被CDN缓存 | 用户数据 |
| `no-cache` | 可以缓存，但每次都需要验证 | 经常变化的数据 |
| `no-store` | 完全不缓存 | 敏感数据 |
| `max-age=N` | 缓存N秒 | 所有资源 |
| `s-maxage=N` | CDN缓存N秒（覆盖max-age） | CDN资源 |
| `immutable` | 内容永不改变 | 带hash的静态资源 |
| `must-revalidate` | 过期后必须验证 | 重要数据 |

### 2. ETag（协商缓存）

```javascript
// etag.js
import express from 'express';
import crypto from 'crypto';

const app = express();

// 💡 生成ETag
function generateETag(content) {
  return crypto
    .createHash('md5')
    .update(content)
    .digest('hex');
}

// 手动实现ETag
app.get('/api/data', (req, res) => {
  const data = { timestamp: Date.now(), message: 'Hello' };
  const content = JSON.stringify(data);
  const etag = generateETag(content);
  
  // 💡 检查If-None-Match
  if (req.headers['if-none-match'] === etag) {
    // 内容未变化，返回304
    res.status(304).end();
    return;
  }
  
  // 内容已变化，返回200
  res.set('ETag', etag);
  res.set('Cache-Control', 'no-cache'); // 允许缓存但需验证
  res.json(data);
});

// 💡 Express自动生成ETag
app.set('etag', 'strong'); // 'weak'或'strong'

app.get('/auto-etag', (req, res) => {
  // Express自动生成ETag并处理304
  res.json({ data: 'content' });
});
```

**ETag类型**：

```
强ETag: "686897696a7c876b7e"
弱ETag: W/"686897696a7c876b7e"  // 内容大致相同

强ETag：字节级完全相同
弱ETag：语义相同即可（如空格、换行差异）
```

### 3. Last-Modified（协商缓存）

```javascript
// last-modified.js
import express from 'express';
import fs from 'fs';

const app = express();

app.get('/file.txt', (req, res) => {
  const filePath = './file.txt';
  const stats = fs.statSync(filePath);
  const lastModified = stats.mtime.toUTCString();
  
  // 💡 检查If-Modified-Since
  const ifModifiedSince = req.headers['if-modified-since'];
  if (ifModifiedSince === lastModified) {
    // 文件未修改，返回304
    res.status(304).end();
    return;
  }
  
  // 文件已修改，返回200
  res.set('Last-Modified', lastModified);
  res.set('Cache-Control', 'no-cache');
  res.sendFile(filePath);
});
```

## 💡 实战：完整缓存策略

### 场景1：静态资源服务器

```javascript
// static-server.js
import express from 'express';
import path from 'path';

const app = express();

// 💡 静态资源中间件
app.use('/static', express.static('public', {
  // 强缓存1年
  maxAge: '1y',
  
  // 允许immutable（带hash的文件）
  immutable: true,
  
  // 自动生成ETag
  etag: true,
  
  // 自动处理Last-Modified
  lastModified: true,
  
  // 设置Cache-Control
  setHeaders: (res, filePath) => {
    // 根据文件类型设置不同策略
    if (filePath.endsWith('.html')) {
      res.set('Cache-Control', 'no-cache');
    } else if (filePath.match(/\.(js|css|png|jpg|gif)$/)) {
      res.set('Cache-Control', 'public, max-age=31536000, immutable');
    }
  }
}));

app.listen(3000);
```

### 场景2：API缓存策略

```javascript
// api-cache.js
import express from 'express';

const app = express();

// 💡 缓存中间件
function cacheControl(options = {}) {
  return (req, res, next) => {
    const {
      maxAge = 0,
      isPrivate = false,
      mustRevalidate = false
    } = options;
    
    const directives = [];
    
    if (isPrivate) {
      directives.push('private');
    } else {
      directives.push('public');
    }
    
    directives.push(`max-age=${maxAge}`);
    
    if (mustRevalidate) {
      directives.push('must-revalidate');
    }
    
    res.set('Cache-Control', directives.join(', '));
    next();
  };
}

// 公开数据：5分钟缓存
app.get('/api/products',
  cacheControl({ maxAge: 300 }),
  (req, res) => {
    res.json({ products: [] });
  }
);

// 用户数据：私有缓存10分钟
app.get('/api/user/profile',
  cacheControl({ maxAge: 600, isPrivate: true }),
  (req, res) => {
    res.json({ user: {} });
  }
);

// 实时数据：不缓存
app.get('/api/realtime/stock',
  cacheControl({ maxAge: 0 }),
  (req, res) => {
    res.json({ price: 100.5 });
  }
);
```

### 场景3：条件请求完整实现

```javascript
// conditional-request.js
import express from 'express';
import crypto from 'crypto';

const app = express();

// 模拟数据
let articles = [
  { id: 1, title: 'Article 1', content: 'Content 1', updatedAt: new Date() }
];

// 生成ETag
function generateETag(data) {
  return crypto.createHash('md5').update(JSON.stringify(data)).digest('hex');
}

// 💡 完整的条件请求处理
app.get('/api/articles/:id', (req, res) => {
  const article = articles.find(a => a.id === parseInt(req.params.id));
  
  if (!article) {
    return res.status(404).json({ error: 'Not found' });
  }
  
  // 生成ETag和Last-Modified
  const etag = generateETag(article);
  const lastModified = article.updatedAt.toUTCString();
  
  // 💡 检查If-None-Match（优先级更高）
  if (req.headers['if-none-match'] === etag) {
    return res.status(304).end();
  }
  
  // 💡 检查If-Modified-Since
  if (req.headers['if-modified-since'] === lastModified) {
    return res.status(304).end();
  }
  
  // 返回完整数据
  res.set('ETag', etag);
  res.set('Last-Modified', lastModified);
  res.set('Cache-Control', 'no-cache'); // 允许缓存但需验证
  res.json(article);
});

// 更新文章时，updatedAt会变化，ETag也会变化
app.put('/api/articles/:id', express.json(), (req, res) => {
  const article = articles.find(a => a.id === parseInt(req.params.id));
  
  if (!article) {
    return res.status(404).json({ error: 'Not found' });
  }
  
  Object.assign(article, req.body);
  article.updatedAt = new Date(); // 更新时间
  
  res.json(article);
});
```

## 🌐 CDN缓存策略

### CDN缓存配置

```javascript
// cdn-cache.js
import express from 'express';

const app = express();

// 💡 针对CDN的缓存策略
app.get('/api/public/news', (req, res) => {
  res.set('Cache-Control', [
    'public',           // CDN可以缓存
    'max-age=60',       // 浏览器缓存1分钟
    's-maxage=3600'     // CDN缓存1小时
  ].join(', '));
  
  res.json({ news: [] });
});

// 💡 Vary头：根据不同条件缓存不同版本
app.get('/api/content', (req, res) => {
  // 根据Accept-Language返回不同内容
  res.set('Vary', 'Accept-Language');
  res.set('Cache-Control', 'public, max-age=3600');
  
  const lang = req.headers['accept-language']?.split(',')[0] || 'en';
  res.json({ content: `Content in ${lang}` });
});

// 💡 CDN刷新：使用带版本号的URL
app.get('/assets/style.css', (req, res) => {
  // URL: /assets/style.css?v=1.2.3
  // 版本变化时，URL变化，CDN自动刷新
  
  res.set('Cache-Control', 'public, max-age=31536000');
  res.sendFile('./style.css');
});
```

### 缓存清除策略

```javascript
// cache-purge.js

// 方式1：改变URL（版本号或hash）
// 旧：/assets/app.js
// 新：/assets/app.js?v=2.0.0
// 或：/assets/app.abc123.js

// 方式2：Cache-Control: no-cache
app.get('/api/data', (req, res) => {
  res.set('Cache-Control', 'no-cache');
  // 浏览器每次都会验证，但可能返回304
});

// 方式3：立即过期
app.get('/api/realtime', (req, res) => {
  res.set('Cache-Control', 'no-store');
  // 完全不缓存
});

// 方式4：CDN API清除（如Cloudflare）
import axios from 'axios';

async function purgeCDNCache(urls) {
  await axios.post('https://api.cloudflare.com/client/v4/zones/{zone_id}/purge_cache', {
    files: urls
  }, {
    headers: {
      'Authorization': 'Bearer YOUR_API_TOKEN'
    }
  });
}
```

## 📋 总结

### 核心要点

1. **强缓存**：Cache-Control、Expires，不请求服务器
2. **协商缓存**：ETag、Last-Modified，请求服务器验证
3. **CDN缓存**：s-maxage、Vary
4. **缓存清除**：版本号、no-cache、no-store

### 实现关键

1. **静态资源**：`Cache-Control: public, max-age=31536000, immutable`
2. **HTML页面**：`Cache-Control: no-cache`
3. **API数据**：`Cache-Control: public, max-age=300`
4. **用户数据**：`Cache-Control: private, max-age=600`

### 应用价值

- **性能提升**：减少网络请求，加快加载速度
- **节省带宽**：减少重复传输
- **降低负载**：减少服务器压力

### 面试重点

1. **强缓存vs协商缓存？**
   - 强缓存：直接使用，不请求服务器
   - 协商缓存：请求服务器验证，可能返回304

2. **ETag vs Last-Modified？**
   - ETag：内容hash，更准确
   - Last-Modified：文件修改时间，精度秒级

3. **如何让静态资源永久缓存？**
   - 文件名加hash，URL变化即刷新缓存

4. **Cache-Control有哪些指令？**
   - public、private、no-cache、no-store、max-age等

### 注意事项

⚠️ **不要缓存敏感数据**：
```javascript
// ❌ 敏感数据
res.set('Cache-Control', 'private, max-age=3600');

// ✅ 敏感数据
res.set('Cache-Control', 'no-store');
```

⚠️ **带hash的资源设置immutable**：
```javascript
// /assets/app.abc123.js
res.set('Cache-Control', 'public, max-age=31536000, immutable');
```
