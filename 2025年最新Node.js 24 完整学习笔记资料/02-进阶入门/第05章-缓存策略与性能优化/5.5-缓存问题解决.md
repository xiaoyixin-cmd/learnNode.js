# 5.5 缓存问题与解决方案

> 解决缓存雪崩、击穿、穿透三大经典问题

---

## 📖 知识点概述

### 三大缓存问题

**缓存雪崩（Cache Avalanche）**：大量缓存同时过期，请求全部打到数据库。

**缓存击穿（Cache Breakdown）**：热点数据过期，大量请求同时查询数据库。

**缓存穿透（Cache Penetration）**：查询不存在的数据，缓存和数据库都没有。

### 问题对比

| 问题 | 原因 | 影响 | 解决难度 |
|------|------|------|---------|
| **雪崩** | 大量key同时过期 | 数据库瞬间负载暴增 | 中 |
| **击穿** | 热点key过期 | 短时间大量查询 | 中 |
| **穿透** | 查询不存在的数据 | 每次都查数据库 | 高 |

---

## ❄️ 缓存雪崩

### 问题场景

```
时间点0: 100万个key同时设置，TTL=3600秒
时间点3600: 100万个key同时过期
    ↓
100万个请求同时打到数据库
    ↓
数据库崩溃 💥
```

### 原因分析

1. **批量导入数据**：设置相同TTL
2. **Redis重启**：所有缓存丢失
3. **缓存服务器宕机**：缓存全部失效

### 解决方案

#### 方案1：随机TTL

```javascript
// cache-avalanche-solution1.js
import Redis from 'ioredis';

const redis = new Redis();

// ❌ 错误：所有key使用相同TTL
async function setCacheBad(key, value) {
  await redis.setex(key, 3600, value); // 所有key 1小时后同时过期
}

// ✅ 正确：添加随机偏移
async function setCacheGood(key, value) {
  const baseTTL = 3600; // 基础TTL 1小时
  const randomOffset = Math.floor(Math.random() * 300); // 0-300秒随机偏移
  const ttl = baseTTL + randomOffset;
  
  await redis.setex(key, ttl, value);
}

// 💡 封装缓存类
class Cache {
  constructor(redis, baseTTL = 3600, maxOffset = 300) {
    this.redis = redis;
    this.baseTTL = baseTTL;
    this.maxOffset = maxOffset;
  }
  
  async set(key, value) {
    const ttl = this.baseTTL + Math.floor(Math.random() * this.maxOffset);
    await this.redis.setex(key, ttl, JSON.stringify(value));
  }
  
  async get(key) {
    const value = await this.redis.get(key);
    return value ? JSON.parse(value) : null;
  }
}
```

#### 方案2：互斥锁（分布式锁）

```javascript
// cache-avalanche-solution2.js

class CacheWithLock {
  constructor(redis) {
    this.redis = redis;
  }
  
  async get(key, fetchData) {
    // 1. 查询缓存
    let value = await this.redis.get(key);
    if (value) {
      return JSON.parse(value);
    }
    
    // 2. 缓存未命中，尝试获取锁
    const lockKey = `lock:${key}`;
    const lockValue = Date.now().toString();
    
    const acquired = await this.redis.set(
      lockKey,
      lockValue,
      'NX',
      'PX',
      5000 // 5秒超时
    );
    
    if (acquired === 'OK') {
      // 💡 获取锁成功，查询数据库
      try {
        const data = await fetchData();
        
        // 写入缓存
        const ttl = 3600 + Math.floor(Math.random() * 300);
        await this.redis.setex(key, ttl, JSON.stringify(data));
        
        return data;
      } finally {
        // 释放锁
        const script = `
          if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
          end
        `;
        await this.redis.eval(script, 1, lockKey, lockValue);
      }
    } else {
      // 💡 未获取锁，等待并重试
      await new Promise(resolve => setTimeout(resolve, 50));
      return await this.get(key, fetchData); // 递归重试
    }
  }
}

// 使用
const cache = new CacheWithLock(redis);

const user = await cache.get('user:123', async () => {
  // 查询数据库
  const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [123]);
  return rows[0];
});
```

#### 方案3：永不过期

```javascript
// cache-avalanche-solution3.js

// 💡 方案：逻辑过期
class CacheNeverExpire {
  constructor(redis) {
    this.redis = redis;
  }
  
  async set(key, value, ttl) {
    const data = {
      value: value,
      expireAt: Date.now() + ttl * 1000
    };
    
    // 💡 不设置Redis TTL，数据永不过期
    await this.redis.set(key, JSON.stringify(data));
  }
  
  async get(key, fetchData) {
    const cached = await this.redis.get(key);
    
    if (!cached) {
      // 缓存不存在，查询数据库
      const data = await fetchData();
      await this.set(key, data, 3600);
      return data;
    }
    
    const { value, expireAt } = JSON.parse(cached);
    
    if (Date.now() < expireAt) {
      // 💡 未过期，直接返回
      return value;
    }
    
    // 💡 已过期，异步更新（返回旧数据）
    setImmediate(async () => {
      const data = await fetchData();
      await this.set(key, data, 3600);
    });
    
    return value; // 返回旧数据，不影响用户体验
  }
}
```

#### 方案4：多级缓存

```javascript
// cache-avalanche-solution4.js
import { LRUCache } from 'lru-cache';

// 💡 内存缓存 + Redis
class TwoLevelCache {
  constructor(redis) {
    this.redis = redis;
    this.localCache = new LRUCache({
      max: 1000,
      ttl: 1000 * 60 // 本地缓存1分钟
    });
  }
  
  async get(key, fetchData) {
    // 1. 查询本地缓存
    let value = this.localCache.get(key);
    if (value) {
      console.log('本地缓存命中');
      return value;
    }
    
    // 2. 查询Redis
    const cached = await this.redis.get(key);
    if (cached) {
      console.log('Redis缓存命中');
      value = JSON.parse(cached);
      
      // 写入本地缓存
      this.localCache.set(key, value);
      return value;
    }
    
    // 3. 查询数据库
    console.log('缓存未命中，查询数据库');
    value = await fetchData();
    
    // 写入Redis和本地缓存
    await this.redis.setex(key, 3600, JSON.stringify(value));
    this.localCache.set(key, value);
    
    return value;
  }
}
```

## 💥 缓存击穿

### 问题场景

```
热点商品key过期
    ↓
1000个并发请求同时到达
    ↓
1000个请求同时查询数据库
    ↓
数据库压力暴增 💥
```

### 解决方案

#### 方案1：互斥锁（推荐）

```javascript
// cache-breakdown-solution1.js

class CacheBreakdownSolution {
  constructor(redis) {
    this.redis = redis;
  }
  
  async get(key, fetchData) {
    // 1. 查询缓存
    let value = await this.redis.get(key);
    if (value) {
      return JSON.parse(value);
    }
    
    // 2. 💡 获取分布式锁
    const lockKey = `lock:${key}`;
    const lockValue = Date.now().toString();
    
    const acquired = await this.redis.set(lockKey, lockValue, 'NX', 'PX', 3000);
    
    if (acquired === 'OK') {
      // 获取锁成功
      try {
        // 再次查询缓存（双重检查）
        value = await this.redis.get(key);
        if (value) {
          return JSON.parse(value);
        }
        
        // 查询数据库
        const data = await fetchData();
        
        // 写入缓存
        await this.redis.setex(key, 3600, JSON.stringify(data));
        
        return data;
      } finally {
        // 释放锁
        const script = `
          if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
          end
        `;
        await this.redis.eval(script, 1, lockKey, lockValue);
      }
    } else {
      // 未获取锁，等待后重试
      await new Promise(resolve => setTimeout(resolve, 100));
      return await this.get(key, fetchData);
    }
  }
}
```

#### 方案2：热点数据永不过期

```javascript
// cache-breakdown-solution2.js

// 💡 热点数据检测
class HotDataCache {
  constructor(redis) {
    this.redis = redis;
    this.hotKeys = new Set(); // 热点key集合
  }
  
  // 记录访问
  async recordAccess(key) {
    const countKey = `access:${key}`;
    const count = await this.redis.incr(countKey);
    
    // 首次访问，设置1分钟过期
    if (count === 1) {
      await this.redis.expire(countKey, 60);
    }
    
    // 💡 1分钟内访问超过100次，标记为热点
    if (count > 100) {
      this.hotKeys.add(key);
    }
  }
  
  async set(key, value) {
    const ttl = this.hotKeys.has(key)
      ? 0 // 热点数据永不过期
      : 3600; // 普通数据1小时
    
    if (ttl === 0) {
      // 永不过期
      await this.redis.set(key, JSON.stringify(value));
    } else {
      await this.redis.setex(key, ttl, JSON.stringify(value));
    }
  }
  
  async get(key, fetchData) {
    // 记录访问
    await this.recordAccess(key);
    
    let value = await this.redis.get(key);
    
    if (!value) {
      value = await fetchData();
      await this.set(key, value);
    } else {
      value = JSON.parse(value);
    }
    
    return value;
  }
}
```

## 🕳️ 缓存穿透

### 问题场景

```
攻击者查询 user:99999999（不存在）
    ↓
缓存没有
    ↓
查询数据库，也没有
    ↓
不写入缓存（因为不存在）
    ↓
下次请求重复上述过程
    ↓
每次都查询数据库 💥
```

### 解决方案

#### 方案1：缓存空值

```javascript
// cache-penetration-solution1.js

class CachePenetrationSolution {
  constructor(redis) {
    this.redis = redis;
  }
  
  async get(key, fetchData) {
    // 1. 查询缓存
    let value = await this.redis.get(key);
    
    if (value !== null) {
      // 💡 空字符串表示数据不存在
      if (value === '') {
        return null;
      }
      
      return JSON.parse(value);
    }
    
    // 2. 查询数据库
    const data = await fetchData();
    
    if (data === null) {
      // 💡 数据不存在，缓存空值（短TTL）
      await this.redis.setex(key, 60, ''); // 1分钟
      return null;
    }
    
    // 数据存在，正常缓存
    await this.redis.setex(key, 3600, JSON.stringify(data));
    return data;
  }
}

// 使用
const cache = new CachePenetrationSolution(redis);

const user = await cache.get('user:99999999', async () => {
  const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [99999999]);
  return rows[0] || null;
});

console.log(user); // null（第二次查询会命中缓存）
```

#### 方案2：布隆过滤器（推荐）

```javascript
// cache-penetration-solution2.js
import { BloomFilter } from 'bloomfilter';

// 💡 布隆过滤器原理：
// - 用bit数组存储数据
// - 多个hash函数计算位置
// - 查询快，但有误判率（说存在可能不存在，说不存在一定不存在）

class BloomFilterCache {
  constructor(redis) {
    this.redis = redis;
    
    // 创建布隆过滤器（1000000个元素，误判率0.01）
    this.bloomFilter = new BloomFilter(
      1000000 * 10, // bit数组大小
      10 // hash函数个数
    );
    
    // 初始化：将所有存在的key加入布隆过滤器
    this.init();
  }
  
  async init() {
    // 查询所有用户ID
    const [rows] = await pool.query('SELECT id FROM users');
    
    rows.forEach(row => {
      this.bloomFilter.add(`user:${row.id}`);
    });
    
    console.log('布隆过滤器初始化完成');
  }
  
  async get(key, fetchData) {
    // 1. 💡 先查询布隆过滤器
    if (!this.bloomFilter.test(key)) {
      console.log('布隆过滤器判断：数据不存在');
      return null; // 一定不存在，直接返回
    }
    
    // 2. 可能存在，查询缓存
    let value = await this.redis.get(key);
    if (value) {
      return JSON.parse(value);
    }
    
    // 3. 查询数据库
    const data = await fetchData();
    
    if (data) {
      await this.redis.setex(key, 3600, JSON.stringify(data));
    }
    
    return data;
  }
  
  // 新增数据时，更新布隆过滤器
  async add(key, value) {
    // 写入数据库...
    
    // 💡 更新布隆过滤器
    this.bloomFilter.add(key);
    
    // 写入缓存
    await this.redis.setex(key, 3600, JSON.stringify(value));
  }
}
```

## 📋 总结

### 核心要点

1. **雪崩**：大量key同时过期 → 随机TTL + 多级缓存
2. **击穿**：热点key过期 → 互斥锁 + 永不过期
3. **穿透**：查询不存在的数据 → 布隆过滤器 + 缓存空值

### 实现关键

1. **雪崩**：TTL加随机偏移`baseTTL + random()`
2. **击穿**：分布式锁`SET NX PX`
3. **穿透**：布隆过滤器预判存在性

### 应用价值

- **保护数据库**：避免突发流量打垮数据库
- **提升稳定性**：系统更加健壮
- **节省成本**：减少资源浪费

### 面试重点

1. **什么是缓存雪崩？如何解决？**
   - 大量key同时过期，用随机TTL和多级缓存

2. **什么是缓存击穿？如何解决？**
   - 热点key过期，用互斥锁或永不过期

3. **什么是缓存穿透？如何解决？**
   - 查询不存在的数据，用布隆过滤器

4. **布隆过滤器的原理？**
   - bit数组+多个hash函数，空间效率高但有误判

### 注意事项

⚠️ **空值缓存要设置短TTL**：
```javascript
await redis.setex(key, 60, ''); // 1分钟，避免浪费内存
```

⚠️ **布隆过滤器需要初始化**：
```javascript
// 应用启动时，将所有存在的key加入布隆过滤器
await initBloomFilter();
```

⚠️ **互斥锁要设置超时**：
```javascript
await redis.set(lockKey, value, 'NX', 'PX', 5000); // 5秒超时
```
