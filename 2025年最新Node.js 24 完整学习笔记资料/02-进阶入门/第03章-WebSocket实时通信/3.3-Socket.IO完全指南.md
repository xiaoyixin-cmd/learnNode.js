# 3.3 Socket.IO 完全指南

> 掌握Socket.IO 4.7+的自动降级、房间管理和Redis集群

---

## 📖 知识点概述

### 核心概念

**Socket.IO**是基于WebSocket的实时通信框架（GitHub 61K+ stars），提供自动降级、房间管理、二进制支持等高级特性。相比ws库，Socket.IO更易用但文件更大。

### 解决的问题

- **自动降级**：WebSocket不可用时降级到长轮询
- **自动重连**：断线后自动重连
- **房间/命名空间**：内置消息分组
- **事件驱动**：emit/on模式，类似EventEmitter
- **集群支持**：Redis适配器实现多服务器

### 主要特点

1. **自动降级**：WebSocket → 长轮询 → JSONP轮询
2. **房间管理**：内置join/leave API
3. **命名空间**：逻辑隔离不同功能
4. **二进制支持**：自动处理Buffer/Blob
5. **中间件**：类似Express的use()

### 典型应用场景

- **跨浏览器实时通信**：兼容所有浏览器
- **复杂房间管理**：多房间、私聊、群聊
- **集群部署**：多服务器负载均衡
- **企业级应用**：需要稳定性和兼容性

#### 第一步：基础服务器

```javascript
// server.js
import { Server } from 'socket.io';
import http from 'http';

const httpServer = http.createServer();
const io = new Server(httpServer, {
  cors: {
    origin: '*',
    methods: ['GET', 'POST']
  }
});

io.on('connection', (socket) => {
  console.log('客户端连接:', socket.id);
  console.log('传输协议:', socket.conn.transport.name); // websocket 或 polling
  
  // 💡 监听自定义事件
  socket.on('chat message', (msg) => {
    console.log('收到消息:', msg);
    
    // 广播给所有客户端
    io.emit('chat message', msg);
  });
  
  socket.on('disconnect', (reason) => {
    console.log('客户端断开:', socket.id, reason);
  });
});

httpServer.listen(3000, () => {
  console.log('Socket.IO服务器运行在 http://localhost:3000');
});
```

#### 第二步：房间管理

```javascript
// server-rooms.js
import { Server } from 'socket.io';

const io = new Server(3000);

io.on('connection', (socket) => {
  console.log('用户连接:', socket.id);
  
  // 💡 加入房间
  socket.on('join room', (room) => {
    socket.join(room);
    console.log(`${socket.id} 加入房间 ${room}`);
    
    // 通知房间内其他人
    socket.to(room).emit('user joined', {
      userId: socket.id,
      room: room
    });
    
    // 发送房间内人数
    const roomSize = io.sockets.adapter.rooms.get(room)?.size || 0;
    io.to(room).emit('room size', roomSize);
  });
  
  // 💡 离开房间
  socket.on('leave room', (room) => {
    socket.leave(room);
    console.log(`${socket.id} 离开房间 ${room}`);
    
    socket.to(room).emit('user left', {
      userId: socket.id,
      room: room
    });
  });
  
  // 💡 向房间发送消息
  socket.on('room message', (room, msg) => {
    // 发送给房间内所有人（包括自己）
    io.to(room).emit('room message', {
      user: socket.id,
      message: msg
    });
    
    // 或：发送给房间内除自己外的人
    // socket.to(room).emit('room message', { user: socket.id, message: msg });
  });
  
  // 💡 向多个房间发送
  socket.on('multi room message', (rooms, msg) => {
    io.to(rooms).emit('message', msg);
  });
  
  // 💡 获取socket所在的房间
  socket.on('my rooms', () => {
    const rooms = Array.from(socket.rooms); // Set<string>
    socket.emit('my rooms', rooms.filter(r => r !== socket.id));
  });
});

console.log('Socket.IO房间服务器运行在 http://localhost:3000');
```

#### 第四步：Redis适配器（集群）

```javascript
// server-cluster.js
import { Server } from 'socket.io';
import { createAdapter } from '@socket.io/redis-adapter';
import { createClient } from 'redis';

const io = new Server(3000);

// 💡 创建Redis客户端
const pubClient = createClient({ host: 'localhost', port: 6379 });
const subClient = pubClient.duplicate();

Promise.all([pubClient.connect(), subClient.connect()]).then(() => {
  // 💡 使用Redis适配器
  io.adapter(createAdapter(pubClient, subClient));
  
  console.log('Socket.IO集群服务器运行在 http://localhost:3000');
  console.log('Redis适配器已启用，支持多服务器');
});

io.on('connection', (socket) => {
  console.log('用户连接:', socket.id);
  
  socket.on('message', (msg) => {
    // 💡 会广播到所有服务器的所有客户端
    io.emit('message', msg);
  });
});

// 💡 跨服务器获取所有socket
io.of('/').adapter.on('create-room', (room) => {
  console.log(`房间 ${room} 已创建`);
});

io.of('/').adapter.on('delete-room', (room) => {
  console.log(`房间 ${room} 已删除`);
});

io.of('/').adapter.on('join-room', (room, id) => {
  console.log(`socket ${id} 加入房间 ${room}`);
});

io.of('/').adapter.on('leave-room', (room, id) => {
  console.log(`socket ${id} 离开房间 ${room}`);
});
```

#### 第五步：中间件和认证

```javascript
// server-auth.js
import { Server } from 'socket.io';
import jwt from 'jsonwebtoken';

const io = new Server(3000);
const JWT_SECRET = 'your-secret-key';

// 💡 全局中间件
io.use((socket, next) => {
  const token = socket.handshake.auth.token;
  
  if (!token) {
    return next(new Error('Authentication error'));
  }
  
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    socket.userId = decoded.userId;
    socket.username = decoded.username;
    next();
  } catch (error) {
    next(new Error('Invalid token'));
  }
});

// 💡 命名空间中间件
const adminNsp = io.of('/admin');

adminNsp.use((socket, next) => {
  if (socket.username === 'admin') {
    next();
  } else {
    next(new Error('Admin only'));
  }
});

io.on('connection', (socket) => {
  console.log(`用户 ${socket.username} 连接`);
  
  // 💡 socket级中间件
  socket.use(([event, ...args], next) => {
    console.log(`事件: ${event}`);
    next();
  });
  
  socket.on('message', (msg) => {
    io.emit('message', {
      user: socket.username,
      text: msg,
      timestamp: Date.now()
    });
  });
});

// 生成token的API（Express示例）
import express from 'express';
const app = express();
app.use(express.json());

app.post('/login', (req, res) => {
  const { username, password } = req.body;
  
  // 简化：实际应该查询数据库
  if (username && password) {
    const token = jwt.sign({ userId: 1, username }, JWT_SECRET, {
      expiresIn: '7d'
    });
    
    res.json({ token });
  } else {
    res.status(401).json({ error: 'Invalid credentials' });
  }
});

app.listen(4000, () => {
  console.log('HTTP API运行在 http://localhost:4000');
});
```

### 验证：完整聊天应用

```javascript
// complete-chat.js
import { Server } from 'socket.io';
import { createAdapter } from '@socket.io/redis-adapter';
import { createClient } from 'redis';

const io = new Server(3000, {
  cors: { origin: '*' }
});

// Redis适配器
const pubClient = createClient();
const subClient = pubClient.duplicate();

await Promise.all([pubClient.connect(), subClient.connect()]);
io.adapter(createAdapter(pubClient, subClient));

// 在线用户
const onlineUsers = new Map();

io.on('connection', (socket) => {
  // 💡 用户上线
  socket.on('user online', (username) => {
    socket.username = username;
    onlineUsers.set(socket.id, username);
    
    // 广播在线用户列表
    io.emit('online users', Array.from(onlineUsers.values()));
  });
  
  // 💡 加入房间
  socket.on('join room', (room) => {
    socket.join(room);
    socket.currentRoom = room;
    
    io.to(room).emit('system message', {
      text: `${socket.username} 加入了房间`,
      room: room
    });
  });
  
  // 💡 发送消息
  socket.on('chat message', (msg) => {
    if (socket.currentRoom) {
      io.to(socket.currentRoom).emit('chat message', {
        user: socket.username,
        text: msg,
        room: socket.currentRoom,
        timestamp: Date.now()
      });
    }
  });
  
  // 💡 私聊
  socket.on('private message', ({ to, msg }) => {
    const targetSocketId = Array.from(onlineUsers.entries())
      .find(([id, name]) => name === to)?.[0];
    
    if (targetSocketId) {
      io.to(targetSocketId).emit('private message', {
        from: socket.username,
        text: msg,
        timestamp: Date.now()
      });
      
      socket.emit('private message', {
        to: to,
        text: msg,
        timestamp: Date.now()
      });
    }
  });
  
  // 💡 正在输入...
  socket.on('typing', () => {
    if (socket.currentRoom) {
      socket.to(socket.currentRoom).emit('typing', {
        user: socket.username
      });
    }
  });
  
  // 💡 断开连接
  socket.on('disconnect', () => {
    onlineUsers.delete(socket.id);
    
    if (socket.currentRoom) {
      io.to(socket.currentRoom).emit('system message', {
        text: `${socket.username} 离开了房间`,
        room: socket.currentRoom
      });
    }
    
    io.emit('online users', Array.from(onlineUsers.values()));
  });
});

console.log('完整聊天服务器运行在 http://localhost:3000');
```

## 🔍 深度解析

### 对比分析：Socket.IO vs ws

| 特性 | Socket.IO | ws | 说明 |
|------|-----------|----|----|
| **自动降级** | ✅ | ❌ | Socket.IO兼容所有浏览器 |
| **房间管理** | 内置 | 手动 | Socket.IO开箱即用 |
| **事件系统** | emit/on | message | Socket.IO更灵活 |
| **文件大小** | 200KB | 50KB | ws更轻量 |
| **性能** | 中 | 高 | ws性能更好 |
| **学习曲线** | 低 | 低 | 都很简单 |

socket.emit('message', 'Hello'); // ~100字节

## 📋 总结

### 核心要点

1. **Socket.IO**：功能丰富的实时通信框架
2. **自动降级**：WebSocket → 长轮询
3. **房间管理**：join/leave API
4. **命名空间**：逻辑隔离

### 实现关键

1. **创建服务器**：`new Server(port)`
2. **监听连接**：`io.on('connection')`
3. **发送消息**：`socket.emit()` / `io.emit()`
4. **房间操作**：`socket.join()` / `socket.leave()`
5. **Redis集群**：`io.adapter(createAdapter())`

### 应用价值

- **兼容性好**：支持所有浏览器
- **易于使用**：房间、事件等开箱即用
- **集群支持**：Redis适配器实现多服务器
- **功能丰富**：中间件、认证、二进制等

### 面试重点

1. **Socket.IO如何实现自动降级？**
   - WebSocket → HTTP长轮询 → JSONP轮询

2. **房间和命名空间的区别？**
   - 房间：同一命名空间内的消息分组
   - 命名空间：完全独立的连接通道

3. **如何实现Socket.IO集群？**
   - 使用Redis适配器，共享连接状态

4. **emit和broadcast的区别？**
   - emit：发送给指定对象
   - broadcast：发送给除自己外的所有人

### 注意事项

⚠️ **不要滥用命名空间**：
```javascript
// 命名空间会创建独立的连接，开销大
// 应该使用房间而不是命名空间
```

⚠️ **集群必须使用Redis适配器**：
```javascript
// 否则消息只在单个服务器内广播
io.adapter(createAdapter(pubClient, subClient));
```
