# 3.5 实战：完整聊天系统

> 综合运用WebSocket技术，构建生产级聊天系统

---

## 📖 项目概述

### 功能需求

一个功能完善的实时聊天系统，包括：
- **用户认证**：JWT登录、Token刷新
- **多房间**：创建/加入/离开房间
- **实时消息**：文本、表情、图片
- **在线状态**：用户列表、正在输入
- **私聊**：一对一消息
- **历史记录**：消息持久化
- **文件传输**：图片、文件上传

### 技术栈

- **后端**：Node.js + Fastify + Socket.IO + Redis + MongoDB
- **前端**：React + Socket.IO Client
- **认证**：JWT
- **存储**：MongoDB（消息） + Redis（在线状态）

const userSchema = new mongoose.Schema({

    unique: true,
    trim: true,
    minlength: 3,
    maxlength: 20
  },

    type: String,
    required: true,

    lowercase: true
  },

    type: String,
    required: true,
    minlength: 6
  },
  avatar: {
    type: String,
    default: ''
  },
  createdAt: {
    type: Date,
    default: Date.now
  }
});

// 💡 保存前加密密码
userSchema.pre('save', async function(next) {
  if (!this.isModified('password')) return next();
  
  this.password = await bcrypt.hash(this.password, 10);
  next();
});

// 💡 验证密码
userSchema.methods.comparePassword = async function(password) {
  return await bcrypt.compare(password, this.password);
};

export default mongoose.model('User', userSchema);

### 第四步：JWT认证

**server/auth/jwt.js**：

```javascript
import jwt from 'jsonwebtoken';
import dotenv from 'dotenv';

dotenv.config();

const JWT_SECRET = process.env.JWT_SECRET;
const JWT_EXPIRES_IN = '7d';

export function generateToken(userId) {
  return jwt.sign({ userId }, JWT_SECRET, {
    expiresIn: JWT_EXPIRES_IN
  });
}

export function verifyToken(token) {
  try {
    return jwt.verify(token, JWT_SECRET);
  } catch (error) {
    return null;
  }
}
```

**server/auth/middleware.js**：

```javascript
import { verifyToken } from './jwt.js';
import User from '../models/User.js';

export async function authenticate(request, reply) {
  try {
    const token = request.headers.authorization?.replace('Bearer ', '');
    
    if (!token) {
      return reply.code(401).send({ error: 'No token provided' });
    }
    
    const decoded = verifyToken(token);
    if (!decoded) {
      return reply.code(401).send({ error: 'Invalid token' });
    }
    
    const user = await User.findById(decoded.userId).select('-password');
    if (!user) {
      return reply.code(401).send({ error: 'User not found' });
    }
    
    request.user = user;
  } catch (error) {
    return reply.code(401).send({ error: 'Authentication failed' });
  }
}
```

### 第五步：HTTP API

**server/routes/auth.js**：

```javascript
import User from '../models/User.js';
import { generateToken } from '../auth/jwt.js';

export default async function authRoutes(fastify, options) {
  // 注册
  fastify.post('/register', async (request, reply) => {
    try {
      const { username, email, password } = request.body;
      
      // 检查用户是否存在
      const existingUser = await User.findOne({ 
        $or: [{ username }, { email }]
      });
      
      if (existingUser) {
        return reply.code(400).send({ 
          error: '用户名或邮箱已存在' 
        });
      }
      
      // 创建用户
      const user = new User({ username, email, password });
      await user.save();
      
      // 生成token
      const token = generateToken(user._id);
      
      reply.code(201).send({
        user: {
          id: user._id,
          username: user.username,
          email: user.email,
          avatar: user.avatar
        },
        token
      });
    } catch (error) {
      reply.code(500).send({ error: error.message });
    }
  });
  
  // 登录
  fastify.post('/login', async (request, reply) => {
    try {
      const { username, password } = request.body;
      
      // 查找用户
      const user = await User.findOne({ username });
      if (!user) {
        return reply.code(401).send({ error: '用户名或密码错误' });
      }
      
      // 验证密码
      const isValid = await user.comparePassword(password);
      if (!isValid) {
        return reply.code(401).send({ error: '用户名或密码错误' });
      }
      
      // 生成token
      const token = generateToken(user._id);
      
      reply.send({
        user: {
          id: user._id,
          username: user.username,
          email: user.email,
          avatar: user.avatar
        },
        token
      });
    } catch (error) {
      reply.code(500).send({ error: error.message });
    }
  });
}
```

**server/routes/rooms.js**：

```javascript
import Room from '../models/Room.js';
import Message from '../models/Message.js';
import { authenticate } from '../auth/middleware.js';

export default async function roomRoutes(fastify, options) {
  // 所有路由都需要认证
  fastify.addHook('preHandler', authenticate);
  
  // 获取所有房间
  fastify.get('/rooms', async (request, reply) => {
    try {
      const rooms = await Room.find()
        .populate('creator', 'username avatar')
        .sort({ createdAt: -1 });
      
      reply.send({ rooms });
    } catch (error) {
      reply.code(500).send({ error: error.message });
    }
  });
  
  // 创建房间
  fastify.post('/rooms', async (request, reply) => {
    try {
      const { name, description } = request.body;
      const userId = request.user._id;
      
      const room = new Room({
        name,
        description,
        creator: userId,
        members: [userId]
      });
      
      await room.save();
      await room.populate('creator', 'username avatar');
      
      reply.code(201).send({ room });
    } catch (error) {
      reply.code(500).send({ error: error.message });
    }
  });
  
  // 获取房间消息历史
  fastify.get('/rooms/:roomId/messages', async (request, reply) => {
    try {
      const { roomId } = request.params;
      const { limit = 50, before } = request.query;
      
      const query = { room: roomId };
      if (before) {
        query.createdAt = { $lt: new Date(before) };
      }
      
      const messages = await Message.find(query)
        .populate('sender', 'username avatar')
        .sort({ createdAt: -1 })
        .limit(parseInt(limit));
      
      reply.send({ messages: messages.reverse() });
    } catch (error) {
      reply.code(500).send({ error: error.message });
    }
  });
}
```

### 第六步：Socket.IO实现

**server/socket.js**：

```javascript
import { Server } from 'socket.io';
import { verifyToken } from './auth/jwt.js';
import User from './models/User.js';
import Room from './models/Room.js';
import Message from './models/Message.js';
import { redis } from './utils/redis.js';

export function setupSocketIO(httpServer) {
  const io = new Server(httpServer, {
    cors: {
      origin: '*',
      methods: ['GET', 'POST']
    }
  });
  
  // 💡 认证中间件
  io.use(async (socket, next) => {
    try {
      const token = socket.handshake.auth.token;
      
      if (!token) {
        return next(new Error('No token'));
      }
      
      const decoded = verifyToken(token);
      if (!decoded) {
        return next(new Error('Invalid token'));
      }
      
      const user = await User.findById(decoded.userId).select('-password');
      if (!user) {
        return next(new Error('User not found'));
      }
      
      socket.userId = user._id.toString();
      socket.username = user.username;
      socket.avatar = user.avatar;
      
      next();
    } catch (error) {
      next(new Error('Authentication failed'));
    }
  });
  
  io.on('connection', async (socket) => {
    console.log(`用户 ${socket.username} 连接`);
    
    // 💡 用户上线
    await redis.sadd('online_users', socket.userId);
    await redis.hset(`user:${socket.userId}`, {
      username: socket.username,
      avatar: socket.avatar,
      socketId: socket.id
    });
    
    // 广播在线用户列表
    const onlineUserIds = await redis.smembers('online_users');
    const onlineUsers = await Promise.all(
      onlineUserIds.map(id => redis.hgetall(`user:${id}`))
    );
    
    io.emit('online users', onlineUsers);
    
    // 💡 加入房间
    socket.on('join room', async (roomId) => {
      try {
        const room = await Room.findById(roomId);
        if (!room) {
          socket.emit('error', { message: '房间不存在' });
          return;
        }
        
        socket.join(roomId);
        socket.currentRoom = roomId;
        
        // 通知房间其他人
        socket.to(roomId).emit('user joined', {
          userId: socket.userId,
          username: socket.username,
          avatar: socket.avatar
        });
        
        // 发送房间信息
        const roomSize = (await io.in(roomId).fetchSockets()).length;
        socket.emit('room joined', {
          roomId,
          roomName: room.name,
          userCount: roomSize
        });
        
        console.log(`${socket.username} 加入房间 ${room.name}`);
      } catch (error) {
        socket.emit('error', { message: error.message });
      }
    });
    
    // 💡 发送消息
    socket.on('chat message', async (data) => {
      try {
        const { roomId, content, type = 'text' } = data;
        
        // 保存到数据库
        const message = new Message({
          room: roomId,
          sender: socket.userId,
          type,
          content
        });
        
        await message.save();
        await message.populate('sender', 'username avatar');
        
        // 广播消息
        io.to(roomId).emit('chat message', {
          id: message._id,
          user: message.sender.username,
          avatar: message.sender.avatar,
          content: message.content,
          type: message.type,
          timestamp: message.createdAt
        });
      } catch (error) {
        socket.emit('error', { message: error.message });
      }
    });
    
    // 💡 私聊
    socket.on('private message', async (data) => {
      try {
        const { toUserId, content } = data;
        
        // 获取目标用户的socketId
        const targetSocketId = await redis.hget(`user:${toUserId}`, 'socketId');
        
        if (targetSocketId) {
          const messageData = {
            from: socket.username,
            fromUserId: socket.userId,
            content,
            timestamp: Date.now()
          };
          
          // 发送给目标用户
          io.to(targetSocketId).emit('private message', messageData);
          
          // 发送给自己（确认）
          socket.emit('private message', {
            ...messageData,
            to: toUserId
          });
        } else {
          socket.emit('error', { message: '用户不在线' });
        }
      } catch (error) {
        socket.emit('error', { message: error.message });
      }
    });
    
    // 💡 正在输入...
    socket.on('typing', (roomId) => {
      socket.to(roomId).emit('typing', {
        user: socket.username
      });
    });
    
    socket.on('stop typing', (roomId) => {
      socket.to(roomId).emit('stop typing', {
        user: socket.username
      });
    });
    
    // 💡 离开房间
    socket.on('leave room', (roomId) => {
      socket.leave(roomId);
      socket.to(roomId).emit('user left', {
        userId: socket.userId,
        username: socket.username
      });
      
      socket.currentRoom = null;
    });
    
    // 💡 断开连接
    socket.on('disconnect', async () => {
      console.log(`用户 ${socket.username} 断开连接`);
      
      // 用户下线
      await redis.srem('online_users', socket.userId);
      await redis.del(`user:${socket.userId}`);
      
      // 广播在线用户列表
      const onlineUserIds = await redis.smembers('online_users');
      const onlineUsers = await Promise.all(
        onlineUserIds.map(id => redis.hgetall(`user:${id}`))
      );
      
      io.emit('online users', onlineUsers);
      
      // 通知当前房间
      if (socket.currentRoom) {
        socket.to(socket.currentRoom).emit('user left', {
          userId: socket.userId,
          username: socket.username
        });
      }
    });
  });
  
  return io;
}
```

### 第七步：主应用

**server/app.js**：

```javascript
import Fastify from 'fastify';
import cors from '@fastify/cors';
import multipart from '@fastify/multipart';
import dotenv from 'dotenv';
import { connectDB } from './utils/db.js';
import { setupSocketIO } from './socket.js';
import authRoutes from './routes/auth.js';
import roomRoutes from './routes/rooms.js';

dotenv.config();

const fastify = Fastify({ logger: true });

// 中间件
await fastify.register(cors, { origin: '*' });
await fastify.register(multipart);

// 路由
await fastify.register(authRoutes, { prefix: '/api/auth' });
await fastify.register(roomRoutes, { prefix: '/api' });

// 健康检查
fastify.get('/health', async () => {
  return { status: 'ok', timestamp: Date.now() };
});

// 启动服务器
const start = async () => {
  try {
    // 连接数据库
    await connectDB();
    
    // 启动HTTP服务器
    await fastify.listen({ 
      port: process.env.PORT || 3000,
      host: '0.0.0.0'
    });
    
    // 启动Socket.IO
    setupSocketIO(fastify.server);
    
    console.log(`🚀 服务器运行在 http://localhost:${process.env.PORT || 3000}`);
  } catch (error) {
    fastify.log.error(error);
    process.exit(1);
  }
};

start();
```
