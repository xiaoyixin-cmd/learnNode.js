## 📖 知识点概述

### 核心概念

**ws**是Node.js中最流行的WebSocket库（GitHub 21K+ stars），提供快速、简洁的WebSocket服务器和客户端实现。相比原生实现，ws提供了更友好的API和更好的性能优化。

### 解决的问题

原生WebSocket的不足：
- **API复杂**：需要手动处理HTTP升级
- **连接管理困难**：需要手动维护客户端列表
- **没有房间概念**：需要自己实现消息分组
- **缺少工具方法**：广播、心跳等需要手写

ws库的解决方案：
- **简洁API**：一行代码创建WebSocket服务器
- **自动连接管理**：内置clients集合
- **性能优化**：高效的帧处理和内存管理
- **灵活扩展**：易于集成到Express/Fastify等框架

### 主要特点

1. **高性能**：C++扩展优化（可选）
2. **标准兼容**：完全符合RFC 6455规范
3. **易于使用**：API简洁直观
4. **零依赖**：核心库无依赖
5. **广泛应用**：Socket.IO底层使用ws

### 典型应用场景

- **实时聊天**：聊天室、即时通讯
- **游戏服务器**：多人在线游戏
- **实时监控**：系统监控、日志流
- **协作工具**：在线编辑器、白板
- **IoT通信**：物联网设备控制

// server-basic.js
import { WebSocketServer } from 'ws';

// 💡 创建WebSocket服务器
const wss = new WebSocketServer({ 
  port: 3000,
  // 可选配置
  clientTracking: true,  // 跟踪客户端（wss.clients）

  maxPayload: 100 * 1024 * 1024 // 最大消息100MB

// 监听连接
wss.on('connection', (ws, request) => {
  console.log('新客户端连接');
  console.log('来源IP:', request.socket.remoteAddress);
  console.log('请求URL:', request.url);
  console.log('当前连接数:', wss.clients.size);
  
  // 发送欢迎消息
  ws.send('欢迎连接到WebSocket服务器');
  
  // 接收消息
  ws.on('message', (data, isBinary) => {
    console.log('收到消息:', data.toString());
    
    // 回显消息
    ws.send(`服务器收到: ${data.toString()}`);
  });
  
  // 连接关闭
  ws.on('close', (code, reason) => {
    console.log('连接关闭:', code, reason.toString());
  });
  
  // 错误处理
  ws.on('error', (error) => {
    console.error('WebSocket错误:', error);
  });
});

// 服务器错误
wss.on('error', (error) => {
  console.error('服务器错误:', error);

console.log('WebSocket服务器运行在 ws://localhost:3000');

      text: message.text,
      timestamp: Date.now()
    });
  });
  
  ws.on('close', () => {
    console.log('客户端断开，剩余连接数:', wss.clients.size);

    broadcast({
      type: 'system',
      message: '用户离开',
      userCount: wss.clients.size
    });
  });

// 💡 广播函数：发送给所有连接的客户端
function broadcast(data) {
  const message = JSON.stringify(data);
  
  wss.clients.forEach((client) => {
    if (client.readyState === client.OPEN) {
      client.send(message);
    }
  });
}

// 💡 广播函数2：发送给除自己外的客户端
function broadcastExcept(ws, data) {
  const message = JSON.stringify(data);
  
  wss.clients.forEach((client) => {
    if (client !== ws && client.readyState === client.OPEN) {
      client.send(message);
    }
  });
}

#### 第三步：房间管理

```javascript
// server-rooms.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

// 💡 房间管理：roomId -> Set<ws>
const rooms = new Map();

wss.on('connection', (ws) => {
  // 为每个连接添加自定义属性
  ws.userId = generateId();
  ws.username = null;
  ws.currentRoom = null;
  
  ws.send(JSON.stringify({
    type: 'welcome',
    userId: ws.userId
  }));
  
  ws.on('message', (data) => {
    try {
      const message = JSON.parse(data.toString());
      
      switch (message.type) {
        case 'setUsername':
          ws.username = message.username;
          ws.send(JSON.stringify({
            type: 'usernameSet',
            username: ws.username
          }));
          break;
          
        case 'joinRoom':
          joinRoom(ws, message.roomId);
          break;
          
        case 'leaveRoom':
          leaveRoom(ws);
          break;
          
        case 'chat':
          if (ws.currentRoom) {
            broadcastToRoom(ws.currentRoom, {
              type: 'chat',
              user: ws.username || ws.userId,
              text: message.text,
              timestamp: Date.now()
            }, ws);
          }
          break;
      }
    } catch (error) {
      console.error('消息处理错误:', error);
    }
  });
  
  ws.on('close', () => {
    leaveRoom(ws);
    console.log(`用户${ws.userId}断开连接`);
  });
});

// 💡 加入房间
function joinRoom(ws, roomId) {
  // 先离开当前房间
  if (ws.currentRoom) {
    leaveRoom(ws);
  }
  
  // 创建房间（如果不存在）
  if (!rooms.has(roomId)) {
    rooms.set(roomId, new Set());
  }
  
  // 加入房间
  rooms.get(roomId).add(ws);
  ws.currentRoom = roomId;
  
  console.log(`用户${ws.userId}加入房间${roomId}`);
  
  // 通知用户
  ws.send(JSON.stringify({
    type: 'roomJoined',
    roomId: roomId,
    userCount: rooms.get(roomId).size
  }));
  
  // 通知房间其他人
  broadcastToRoom(roomId, {
    type: 'userJoined',
    user: ws.username || ws.userId,
    userCount: rooms.get(roomId).size
  }, ws);
}

// 💡 离开房间
function leaveRoom(ws) {
  if (!ws.currentRoom) return;
  
  const roomId = ws.currentRoom;
  const room = rooms.get(roomId);
  
  if (room) {
    room.delete(ws);
    
    // 通知房间其他人
    broadcastToRoom(roomId, {
      type: 'userLeft',
      user: ws.username || ws.userId,
      userCount: room.size
    }, ws);
    
    // 如果房间为空，删除房间
    if (room.size === 0) {
      rooms.delete(roomId);
      console.log(`房间${roomId}已删除（无用户）`);
    }
  }
  
  ws.currentRoom = null;
}

// 💡 房间广播
function broadcastToRoom(roomId, data, excludeWs = null) {
  const room = rooms.get(roomId);
  if (!room) return;
  
  const message = JSON.stringify(data);
  
  room.forEach((client) => {
    if (client !== excludeWs && client.readyState === client.OPEN) {
      client.send(message);
    }
  });
}

function generateId() {
  return Math.random().toString(36).substring(7);
}

console.log('房间服务器运行在 ws://localhost:3000');
```

import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

const HEARTBEAT_INTERVAL = 30000;

**客户端断线重连**：

```javascript
// client-reconnect.js
class ReconnectingWebSocket {
  constructor(url, options = {}) {
    this.url = url;
    this.reconnectInterval = options.reconnectInterval || 3000;
    this.maxReconnectAttempts = options.maxReconnectAttempts || 10;
    this.reconnectAttempts = 0;
    
    this.ws = null;
    this.messageQueue = [];
    this.listeners = {};
    
    this.connect();
  }
  
  connect() {
    console.log('连接WebSocket...');
    
    this.ws = new WebSocket(this.url);
    
    this.ws.addEventListener('open', () => {
      console.log('WebSocket已连接');
      this.reconnectAttempts = 0;
      
      // 发送队列中的消息
      while (this.messageQueue.length > 0) {
        const message = this.messageQueue.shift();
        this.ws.send(message);
      }
      
      this.emit('open');
    });
    
    this.ws.addEventListener('message', (event) => {
      this.emit('message', event.data);
    });
    
    this.ws.addEventListener('close', () => {
      console.log('WebSocket已关闭');
      this.emit('close');
      
      // 💡 自动重连
      if (this.reconnectAttempts < this.maxReconnectAttempts) {
        this.reconnectAttempts++;
        console.log(`${this.reconnectInterval / 1000}秒后尝试重连（第${this.reconnectAttempts}次）...`);
        setTimeout(() => this.connect(), this.reconnectInterval);
      } else {
        console.error('达到最大重连次数，放弃重连');
        this.emit('maxReconnectAttemptsReached');
      }
    });
    
    this.ws.addEventListener('error', (error) => {
      console.error('WebSocket错误:', error);
      this.emit('error', error);
    });
  }
  
  send(data) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(data);
    } else {
      // 💡 连接未就绪，加入队列
      console.log('连接未就绪，消息加入队列');
      this.messageQueue.push(data);
    }
  }
  
  on(event, callback) {
    if (!this.listeners[event]) {
      this.listeners[event] = [];
    }
    this.listeners[event].push(callback);
  }
  
  emit(event, ...args) {
    if (this.listeners[event]) {
      this.listeners[event].forEach(callback => callback(...args));
    }
  }
  
  close() {
    this.maxReconnectAttempts = 0; // 禁止重连
    if (this.ws) {
      this.ws.close();
    }
  }
}

// 使用示例
const ws = new ReconnectingWebSocket('ws://localhost:3000', {
  reconnectInterval: 3000,
  maxReconnectAttempts: 10
});

ws.on('open', () => {
  console.log('连接成功');
});

ws.on('message', (data) => {
  console.log('收到消息:', data);
});

ws.on('close', () => {
  console.log('连接关闭');
});

ws.send('Hello');
```

### 错误处理：3个高频问题

#### 问题1：连接泄漏（clients持续增长）

**错误表现**：wss.clients.size持续增长，内存占用不断上升

**原因**：未正确处理异常断开的连接

**解决方案**：
```javascript
// ✅ 方案1：使用心跳检测清理死连接
const interval = setInterval(() => {
  wss.clients.forEach((ws) => {
    if (ws.isAlive === false) {
      return ws.terminate();
    }
    ws.isAlive = false;
    ws.ping();
  });
}, 30000);

// ✅ 方案2：设置连接超时
const TIMEOUT = 60000; // 60秒

wss.on('connection', (ws) => {
  ws.lastActivity = Date.now();
  
  ws.on('message', () => {
    ws.lastActivity = Date.now();
  });
  
  ws.on('pong', () => {
    ws.lastActivity = Date.now();
  });
});

// 定期检查超时连接
setInterval(() => {
  const now = Date.now();
  wss.clients.forEach((ws) => {
    if (now - ws.lastActivity > TIMEOUT) {
      console.log('连接超时，断开');
      ws.terminate();
    }
  });
}, 30000);
```

#### 问题2：消息背压（发送速度>处理速度）

**错误表现**：内存暴涨，客户端收不到消息

**原因**：发送缓冲区满，消息堆积

**解决方案**：
```javascript
// ✅ 检查发送缓冲区
function safeSend(ws, data) {
  // 💡 bufferedAmount：待发送字节数
  if (ws.bufferedAmount > 1024 * 1024) { // 1MB
    console.warn('发送缓冲区过大，跳过消息');
    return false;
  }
  
  ws.send(data);
  return true;
}

// ✅ 使用drain事件
ws.on('message', (data) => {
  const response = processData(data);
  
  if (ws.bufferedAmount > 0) {
    // 等待缓冲区排空
    ws.once('drain', () => {
      ws.send(response);
    });
  } else {
    ws.send(response);
  }
});
```

#### 问题3：并发连接数过多导致崩溃

**错误表现**：连接数过多时服务器崩溃

**原因**：系统资源耗尽（文件描述符、内存）

## 🎯 应用场景

### 场景1：实时聊天室

#### 完整实现

```javascript
// chat-server.js
import { WebSocketServer } from 'ws';
import crypto from 'crypto';

const wss = new WebSocketServer({ port: 3000 });

// 数据结构
const users = new Map(); // userId -> { ws, username, room }
const rooms = new Map(); // roomId -> Set<userId>

wss.on('connection', (ws) => {
  const userId = crypto.randomUUID();
  users.set(userId, { ws, username: null, room: null });
  
  console.log(`用户${userId}连接`);
  
  ws.send(JSON.stringify({
    type: 'connected',
    userId: userId
  }));
  
  ws.on('message', (data) => {
    try {
      const message = JSON.parse(data.toString());
      handleMessage(userId, message);
    } catch (error) {
      console.error('消息解析错误:', error);
    }
  });
  
  ws.on('close', () => {
    handleDisconnect(userId);
  });
});

function handleMessage(userId, message) {
  const user = users.get(userId);
  if (!user) return;
  
  switch (message.type) {
    case 'setUsername':
      user.username = message.username;
      user.ws.send(JSON.stringify({
        type: 'usernameSet',
        username: user.username
      }));
      break;
      
    case 'joinRoom':
      joinRoom(userId, message.roomId);
      break;
      
    case 'leaveRoom':
      leaveRoom(userId);
      break;
      
    case 'chat':
      if (user.room) {
        broadcastToRoom(user.room, {
          type: 'message',
          user: user.username || userId,
          text: message.text,
          timestamp: Date.now()
        }, userId);
      }
      break;
      
    case 'typing':
      if (user.room) {
        broadcastToRoom(user.room, {
          type: 'typing',
          user: user.username || userId
        }, userId);
      }
      break;
  }
}

function joinRoom(userId, roomId) {
  const user = users.get(userId);
  if (!user) return;
  
  // 离开当前房间
  if (user.room) {
    leaveRoom(userId);
  }
  
  // 创建房间
  if (!rooms.has(roomId)) {
    rooms.set(roomId, new Set());
  }
  
  // 加入房间
  rooms.get(roomId).add(userId);
  user.room = roomId;
  
  // 通知用户
  user.ws.send(JSON.stringify({
    type: 'joinedRoom',
    roomId: roomId,
    users: getRoomUsers(roomId)
  }));
  
  // 通知其他人
  broadcastToRoom(roomId, {
    type: 'userJoined',
    user: user.username || userId,
    userCount: rooms.get(roomId).size
  }, userId);
}

function leaveRoom(userId) {
  const user = users.get(userId);
  if (!user || !user.room) return;
  
  const roomId = user.room;
  const room = rooms.get(roomId);
  
  if (room) {
    room.delete(userId);
    
    // 通知其他人
    broadcastToRoom(roomId, {
      type: 'userLeft',
      user: user.username || userId,
      userCount: room.size
    }, userId);
    
    // 删除空房间
    if (room.size === 0) {
      rooms.delete(roomId);
    }
  }
  
  user.room = null;
}

function handleDisconnect(userId) {
  const user = users.get(userId);
  if (user) {
    leaveRoom(userId);
    users.delete(userId);
  }
  console.log(`用户${userId}断开`);
}

function broadcastToRoom(roomId, data, excludeUserId = null) {
  const room = rooms.get(roomId);
  if (!room) return;
  
  const message = JSON.stringify(data);
  
  room.forEach((uid) => {
    if (uid !== excludeUserId) {
      const user = users.get(uid);
      if (user && user.ws.readyState === user.ws.OPEN) {
        user.ws.send(message);
      }
    }
  });
}

function getRoomUsers(roomId) {
  const room = rooms.get(roomId);
  if (!room) return [];
  
  return Array.from(room).map(userId => {
    const user = users.get(userId);
    return {
      userId,
      username: user?.username || userId
    };
  });
}

console.log('聊天服务器运行在 ws://localhost:3000');
```

- **性能**：内存占用<500MB

#### 优化前：JSON序列化开销

// ❌ 每次都序列化
wss.clients.forEach(ws => {

## 📋 总结

### 核心要点

1. **ws库**：高性能、零依赖的WebSocket库
2. **广播消息**：wss.clients遍历所有连接
3. **房间管理**：使用Map存储房间和用户关系
4. **心跳检测**：ping/pong保持连接活跃

### 实现关键

1. **创建服务器**：`new WebSocketServer({ port })`
2. **监听连接**：`wss.on('connection', (ws) => {})`
3. **发送消息**：`ws.send(data)`
4. **广播消息**：遍历`wss.clients`
5. **心跳检测**：`ws.ping()` + `ws.on('pong')`

### 应用价值

- **高性能**：原生实现，性能优异
- **简洁API**：易于上手
- **灵活性**：完全控制连接管理
- **标准兼容**：完全符合RFC 6455

### 面试重点

1. **如何实现广播消息？**
   - 遍历wss.clients，对每个ws.send()

2. **如何实现房间功能？**
   - 使用Map存储roomId -> Set<ws>

3. **如何处理断线重连？**
   - 客户端监听close事件，setTimeout重连

4. **如何防止连接泄漏？**
   - 心跳检测 + 超时清理

### 注意事项

⚠️ **必须检查readyState**：
```javascript
if (ws.readyState === ws.OPEN) {
  ws.send(data);
}
```

⚠️ **使用terminate()而非close()**：
```javascript
ws.terminate(); // 立即关闭，不发送close帧
```
