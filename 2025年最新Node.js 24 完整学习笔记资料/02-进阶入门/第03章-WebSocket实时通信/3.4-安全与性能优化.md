## 📖 知识点概述

### 核心概念

WebSocket的安全和性能优化涉及认证授权、限流防护、连接管理、内存优化等多个方面。在生产环境中，这些措施必不可少。

### 解决的问题

- **安全漏洞**：CSRF、XSS、未授权访问
- **DDoS攻击**：恶意连接耗尽资源
- **性能瓶颈**：连接数过多、内存泄漏
- **资源浪费**：死连接占用资源

### 主要特点

1. **多层防护**：Origin验证、Token认证、限流
2. **性能监控**：连接数、消息量、内存占用
3. **资源管理**：连接池、心跳检测
4. **优雅降级**：过载保护、熔断机制

## 🚀 安全加固

### 1. Origin验证（防止CSRF）

```javascript
// server-origin.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ 
  port: 3000,
  verifyClient: (info, callback) => {
    const origin = info.origin;
    
    // 💡 白名单验证
    const allowedOrigins = [
      'http://localhost:8080',
      'https://example.com',
      'https://www.example.com'
    ];
    
    if (allowedOrigins.includes(origin)) {
      callback(true);
    } else {
      console.warn('拒绝来自不受信任来源的连接:', origin);
      callback(false, 403, 'Forbidden');
    }
  }
});

wss.on('connection', (ws) => {
  console.log('连接已验证');
});
```

### 2. Token认证（JWT）

```javascript
// server-jwt.js
import { WebSocketServer } from 'ws';
import jwt from 'jsonwebtoken';
import { URL } from 'url';

const JWT_SECRET = 'your-secret-key';

const wss = new WebSocketServer({ 
  port: 3000,
  verifyClient: (info, callback) => {
    try {
      // 💡 从查询参数获取token
      const url = new URL(info.req.url, `ws://localhost:3000`);
      const token = url.searchParams.get('token');
      
      if (!token) {
        callback(false, 401, 'No token provided');
        return;
      }
      
      // 验证token
      const decoded = jwt.verify(token, JWT_SECRET);
      info.req.user = decoded;
      callback(true);
    } catch (error) {
      console.error('Token验证失败:', error);
      callback(false, 401, 'Invalid token');
    }
  }
});

wss.on('connection', (ws, request) => {
  const user = request.user;
  console.log(`用户 ${user.username} 连接`);
  
  ws.userId = user.userId;
  ws.username = user.username;
  
  ws.on('message', (data) => {
    console.log(`来自 ${ws.username} 的消息:`, data.toString());
  });
});

// 客户端连接示例
// const token = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...';
// const ws = new WebSocket(`ws://localhost:3000?token=${token}`);
```

### 3. 限流防护

```javascript
// server-ratelimit.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

// 💡 连接限流：每个IP最多5个连接
const connectionsPerIP = new Map();
const MAX_CONNECTIONS_PER_IP = 5;

wss.on('connection', (ws, request) => {
  const ip = request.socket.remoteAddress;
  
  // 检查IP连接数
  const count = connectionsPerIP.get(ip) || 0;
  if (count >= MAX_CONNECTIONS_PER_IP) {
    console.warn(`IP ${ip} 连接数已达上限`);
    ws.close(1008, 'Too many connections');
    return;
  }
  
  // 增加计数
  connectionsPerIP.set(ip, count + 1);
  
  // 💡 消息限流：每秒最多10条消息
  let messageCount = 0;
  const MESSAGE_LIMIT = 10;
  
  const resetInterval = setInterval(() => {
    messageCount = 0;
  }, 1000);
  
  ws.on('message', (data) => {
    messageCount++;
    
    if (messageCount > MESSAGE_LIMIT) {
      console.warn(`用户 ${ws.id} 消息过多，限流`);
      ws.send(JSON.stringify({
        error: 'Rate limit exceeded',
        retryAfter: 1000
      }));
      return;
    }
    
    // 处理消息
    console.log('消息:', data.toString());
  });
  
  ws.on('close', () => {
    // 减少计数
    const currentCount = connectionsPerIP.get(ip) || 1;
    if (currentCount <= 1) {
      connectionsPerIP.delete(ip);
    } else {
      connectionsPerIP.set(ip, currentCount - 1);
    }
    
    clearInterval(resetInterval);
  });
});

console.log('限流服务器运行在 ws://localhost:3000');
```

## ⚡ 性能优化

### 1. 连接池管理

```javascript
// server-pool.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

// 💡 连接池
class ConnectionPool {
  constructor(maxConnections = 10000) {
    this.maxConnections = maxConnections;
    this.connections = new Map();
  }
  
  add(ws, metadata) {
    if (this.connections.size >= this.maxConnections) {
      console.warn('连接池已满');
      return false;
    }
    
    const id = this.generateId();
    this.connections.set(id, { ws, metadata, lastActivity: Date.now() });
    ws.id = id;
    return true;
  }
  
  remove(id) {
    this.connections.delete(id);
  }
  
  get(id) {
    return this.connections.get(id);
  }
  
  broadcast(data) {
    const message = JSON.stringify(data);
    this.connections.forEach(({ ws }) => {
      if (ws.readyState === ws.OPEN) {
        ws.send(message);
      }
    });
  }
  
  // 💡 清理不活跃连接
  cleanup(maxIdleTime = 300000) { // 5分钟
    const now = Date.now();
    let cleaned = 0;
    
    this.connections.forEach(({ ws, lastActivity }, id) => {
      if (now - lastActivity > maxIdleTime) {
        ws.close(1000, 'Idle timeout');
        this.connections.delete(id);
        cleaned++;
      }
    });
    
    console.log(`清理了 ${cleaned} 个不活跃连接`);
  }
  
  generateId() {
    return Math.random().toString(36).substring(7);
  }
}

const pool = new ConnectionPool(10000);

wss.on('connection', (ws) => {
  if (!pool.add(ws, { connectedAt: Date.now() })) {
    ws.close(1008, 'Server is full');
    return;
  }
  
  console.log(`连接已添加到池，当前连接数: ${pool.connections.size}`);
  
  ws.on('message', () => {
    const conn = pool.get(ws.id);
    if (conn) {
      conn.lastActivity = Date.now();
    }
  });
  
  ws.on('close', () => {
    pool.remove(ws.id);
  });
});

// 💡 定期清理
setInterval(() => {
  pool.cleanup();
}, 60000); // 每分钟
```

### 2. 消息批处理

```javascript
// server-batch.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

class MessageBatcher {
  constructor(interval = 100) {
    this.queue = [];
    this.interval = interval;
    this.timer = null;
  }
  
  add(message) {
    this.queue.push(message);
    
    if (!this.timer) {
      this.timer = setTimeout(() => {
        this.flush();
      }, this.interval);
    }
  }
  
  flush() {
    if (this.queue.length === 0) return;
    
    // 💡 批量发送
    const batch = {
      type: 'batch',
      messages: this.queue,
      count: this.queue.length
    };
    
    wss.clients.forEach((ws) => {
      if (ws.readyState === ws.OPEN) {
        ws.send(JSON.stringify(batch));
      }
    });
    
    console.log(`批量发送 ${this.queue.length} 条消息`);
    
    this.queue = [];
    this.timer = null;
  }
}

const batcher = new MessageBatcher(100);

// 模拟高频消息
setInterval(() => {
  batcher.add({
    type: 'update',
    data: Math.random(),
    timestamp: Date.now()
  });
}, 10);
```

### 3. 二进制优化

```javascript
// server-binary.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

wss.on('connection', (ws) => {
  // 💡 发送二进制数据
  const buffer = Buffer.from('Hello');
  ws.send(buffer);
  
  // 💡 使用Protocol Buffers（更高效）
  // 定义消息格式
  const messageSchema = {
    type: 1,    // uint8
    userId: 123, // uint32
    text: 'Hello' // string
  };
  
  // 手动编码（简化版）
  const binaryMessage = encodeToBinary(messageSchema);
  ws.send(binaryMessage);
});

function encodeToBinary(data) {
  // 简化的二进制编码
  const buffer = Buffer.alloc(100);
  let offset = 0;
  
  buffer.writeUInt8(data.type, offset); offset += 1;
  buffer.writeUInt32LE(data.userId, offset); offset += 4;
  buffer.write(data.text, offset);
  
  return buffer.slice(0, offset + data.text.length);
}

// 对比：
// JSON: {"type":1,"userId":123,"text":"Hello"} = 42字节
// 二进制: 30字节（节省30%）
```

## 📊 监控和诊断

### 性能监控

```javascript
// server-monitor.js
import { WebSocketServer } from 'ws';

const wss = new WebSocketServer({ port: 3000 });

class PerformanceMonitor {
  constructor() {
    this.metrics = {
      connections: 0,
      totalMessages: 0,
      messagesPerSecond: 0,
      bytesReceived: 0,
      bytesSent: 0,
      errors: 0
    };
    
    this.messageCountLastSecond = 0;
    
    // 每秒更新指标
    setInterval(() => {
      this.metrics.messagesPerSecond = this.messageCountLastSecond;
      this.messageCountLastSecond = 0;
      
      this.printMetrics();
    }, 1000);
  }
  
  recordConnection() {
    this.metrics.connections++;
  }
  
  recordDisconnection() {
    this.metrics.connections--;
  }
  
  recordMessage(size) {
    this.metrics.totalMessages++;
    this.messageCountLastSecond++;
    this.metrics.bytesReceived += size;
  }
  
  recordSent(size) {
    this.metrics.bytesSent += size;
  }
  
  recordError() {
    this.metrics.errors++;
  }
  
  printMetrics() {
    console.log('===== 性能指标 =====');
    console.log(`当前连接数: ${this.metrics.connections}`);
    console.log(`总消息数: ${this.metrics.totalMessages}`);
    console.log(`消息/秒: ${this.metrics.messagesPerSecond}`);
    console.log(`接收字节: ${(this.metrics.bytesReceived / 1024 / 1024).toFixed(2)}MB`);
    console.log(`发送字节: ${(this.metrics.bytesSent / 1024 / 1024).toFixed(2)}MB`);
    console.log(`错误数: ${this.metrics.errors}`);
    console.log(`内存占用: ${(process.memoryUsage().heapUsed / 1024 / 1024).toFixed(2)}MB`);
    console.log('====================\n');
  }
}

const monitor = new PerformanceMonitor();

wss.on('connection', (ws) => {
  monitor.recordConnection();
  
  ws.on('message', (data) => {
    monitor.recordMessage(data.length);
    
    // 处理消息...
    const response = 'OK';
    ws.send(response);
    monitor.recordSent(response.length);
  });
  
  ws.on('close', () => {
    monitor.recordDisconnection();
  });
  
  ws.on('error', (error) => {
    monitor.recordError();
  });
});
```

## 📋 总结

### 核心要点

1. **安全三要素**：Origin验证、Token认证、限流
2. **性能优化**：连接池、消息批处理、二进制、压缩
3. **资源管理**：心跳检测、死连接清理
4. **监控诊断**：性能指标、错误追踪

### 实现关键

1. **Origin白名单**：`verifyClient`验证来源
2. **JWT认证**：从查询参数或Cookie获取token
3. **限流**：IP限流 + 消息频率限流
4. **连接池**：限制最大连接数
5. **监控**：记录连接数、消息量、错误数

### 应用价值

- **安全性**：防止CSRF、DDoS等攻击
- **性能**：支持更多并发连接
- **稳定性**：避免资源耗尽
- **可观测**：实时监控系统状态

### 面试重点

1. **WebSocket如何防止CSRF？**
   - Origin验证 + Token认证

2. **如何限制WebSocket连接数？**
   - verifyClient中检查连接数，超过限制返回false

3. **如何优化WebSocket性能？**
   - 二进制数据、压缩、批处理、连接池

4. **如何监控WebSocket性能？**
   - 记录连接数、消息量、延迟、错误率
