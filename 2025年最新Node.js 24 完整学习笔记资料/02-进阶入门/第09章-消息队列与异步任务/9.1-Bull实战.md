## 📖 知识点概述

### 核心概念

**Bull**是Node.js中最流行的基于Redis的队列库，用于处理异步任务、后台作业和分布式任务调度。Bull基于Redis实现，具有高性能、可靠性和易用性。

### Bull架构

```
┌──────────────────────────────────────────────────────────────┐
│                    Bull Queue System                          │
├──────────────────────────────────────────────────────────────┤
│  1. Producer（生产者）                                         │
│     └─ 创建任务并添加到队列                                    │
│                                                                │
│  2. Queue（队列）                                              │
│     ├─ 存储在Redis中                                          │
│     ├─ 任务状态：waiting → active → completed/failed         │
│     └─ 支持优先级、延迟、重试                                  │
│                                                                │
│  3. Worker（消费者）                                           │
│     ├─ 从队列取出任务                                          │
│     ├─ 执行任务逻辑                                           │
│     └─ 支持并发控制                                           │
│                                                                │
│  4. Events（事件）                                             │
│     ├─ completed - 任务完成                                   │
│     ├─ failed - 任务失败                                      │
│     ├─ progress - 任务进度                                    │
│     └─ stalled - 任务卡死                                     │
│                                                                │
│  5. Redis（存储层）                                            │
│     ├─ List（队列）                                           │
│     ├─ Sorted Set（延迟/优先级）                              │
│     └─ Hash（任务数据）                                       │
└──────────────────────────────────────────────────────────────┘
```

### 任务生命周期

```
┌─────────┐
│  创建任务 │
└────┬────┘
     │
     ▼
┌─────────┐
│ waiting  │ ← 等待被处理
└────┬────┘
     │
     ▼
┌─────────┐
│ active   │ ← Worker正在处理
└────┬────┘
     │
     ├─────────────┬─────────────┐
     ▼             ▼             ▼
┌──────────┐ ┌─────────┐  ┌─────────┐
│completed │ │ failed  │  │ delayed │
│(成功)    │ │(失败)   │  │(延迟)   │
└──────────┘ └─────────┘  └─────────┘
```

```javascript
  await sendSMS(job.data.phone);
```

### 实践：Bull完整实现

#### 第一步：基础队列使用

```javascript
// bull-basic.js
import Queue from 'bull';
import Redis from 'ioredis';

// 💡 Redis连接配置
const redisConfig = {
  host: process.env.REDIS_HOST || 'localhost',
  port: process.env.REDIS_PORT || 6379,
  password: process.env.REDIS_PASSWORD,
  db: 0,
  // 💡 连接池配置
  maxRetriesPerRequest: 3,
  enableReadyCheck: true,
  retryStrategy: (times) => {
    const delay = Math.min(times * 50, 2000);
    return delay;
  }
};

// 💡 创建队列（生产者）
const emailQueue = new Queue('email', {
  redis: redisConfig,
  // 💡 队列配置
  defaultJobOptions: {
    attempts: 3,              // 失败重试3次
    backoff: {
      type: 'exponential',    // 指数退避
      delay: 2000             // 初始延迟2秒
    },
    removeOnComplete: 100,    // 只保留最近100个完成任务
    removeOnFail: 200         // 只保留最近200个失败任务
  },
  settings: {
    lockDuration: 30000,      // 任务锁定时间30秒
    stalledInterval: 30000,   // 检查卡死任务间隔
    maxStalledCount: 1        // 最大卡死次数
  }
});

// 💡 添加任务（生产者）
async function sendEmail(to, subject, body) {
  const job = await emailQueue.add(
    'send-email',              // 任务名称
    {
      to,
      subject,
      body,
      timestamp: Date.now()
    },
    {
      // 💡 任务选项（可覆盖默认配置）
      priority: 1,             // 优先级（1-10，数字越小优先级越高）
      delay: 0,                // 延迟（毫秒）
      attempts: 5,             // 重试次数
      timeout: 60000,          // 超时时间60秒
      jobId: `email-${Date.now()}` // 自定义任务ID（用于去重）
    }
  );
  
  console.log(`✅ 任务已添加: ${job.id}`);
  return job;
}

// 💡 Worker（消费者）- 处理任务
emailQueue.process('send-email', async (job) => {
  const { to, subject, body } = job.data;
  
  console.log(`📧 开始发送邮件: ${job.id}`);
  
  // 💡 更新进度
  job.progress(10);
  
  // 模拟发送邮件
  await new Promise(resolve => setTimeout(resolve, 2000));
  
  job.progress(50);
  
  // 实际发送逻辑
  // await nodemailer.sendMail({ to, subject, html: body });
  
  job.progress(100);
  
  console.log(`✅ 邮件发送成功: ${to}`);
  
  return { success: true, to, sentAt: Date.now() };
});

// 💡 事件监听
emailQueue.on('completed', (job, result) => {
  console.log(`✅ 任务完成: ${job.id}`, result);
});

emailQueue.on('failed', (job, err) => {
  console.error(`❌ 任务失败: ${job.id}`, err.message);
});

emailQueue.on('progress', (job, progress) => {
  console.log(`📊 任务进度: ${job.id} - ${progress}%`);
});

emailQueue.on('stalled', (job) => {
  console.warn(`⚠️ 任务卡死: ${job.id}`);
});

emailQueue.on('error', (error) => {
  console.error('❌ 队列错误:', error);
});

// 💡 全局错误处理
emailQueue.on('global:completed', (jobId, result) => {
  console.log(`🌍 全局：任务完成 ${jobId}`);
});

emailQueue.on('global:failed', (jobId, err) => {
  console.log(`🌍 全局：任务失败 ${jobId}`);
});

// 使用示例
await sendEmail('user@example.com', '欢迎注册', '<h1>欢迎！</h1>');
```

#### 第四步：任务管理与查询

```javascript
// bull-management.js
import Queue from 'bull';

const queue = new Queue('tasks', { redis: redisConfig });

// 💡 获取任务状态
async function getJobStatus(jobId) {
  const job = await queue.getJob(jobId);
  
  if (!job) {
    return { error: '任务不存在' };
  }
  
  return {
    id: job.id,
    name: job.name,
    data: job.data,
    progress: job.progress(),
    state: await job.getState(),        // waiting/active/completed/failed
    attempts: job.attemptsMade,
    failedReason: job.failedReason,
    finishedOn: job.finishedOn,
    processedOn: job.processedOn,
    timestamp: job.timestamp
  };
}

// 💡 查询队列统计
async function getQueueStats() {
  const [
    waiting,
    active,
    completed,
    failed,
    delayed
  ] = await Promise.all([
    queue.getWaitingCount(),
    queue.getActiveCount(),
    queue.getCompletedCount(),
    queue.getFailedCount(),
    queue.getDelayedCount()
  ]);
  
  return {
    waiting,
    active,
    completed,
    failed,
    delayed,
    total: waiting + active + completed + failed + delayed
  };
}

// 💡 获取失败任务
async function getFailedJobs(start = 0, end = 10) {
  const jobs = await queue.getFailed(start, end);
  
  return jobs.map(job => ({
    id: job.id,
    name: job.name,
    data: job.data,
    failedReason: job.failedReason,
    stacktrace: job.stacktrace,
    attemptsMade: job.attemptsMade,
    timestamp: job.timestamp
  }));
}

// 💡 重试失败任务
async function retryFailedJob(jobId) {
  const job = await queue.getJob(jobId);
  
  if (!job) {
    throw new Error('任务不存在');
  }
  
  const state = await job.getState();
  
  if (state === 'failed') {
    await job.retry();
    return { success: true, message: '任务已重试' };
  }
  
  return { success: false, message: '任务状态不是failed' };
}

// 💡 批量重试所有失败任务
async function retryAllFailed() {
  const failedJobs = await queue.getFailed();
  
  for (const job of failedJobs) {
    await job.retry();
  }
  
  return { success: true, count: failedJobs.length };
}

// 💡 清理完成的任务
async function cleanCompletedJobs(grace = 3600000) {
  // grace: 保留时间（毫秒），默认1小时
  await queue.clean(grace, 'completed');
  return { success: true };
}

// 💡 清理所有任务
async function clearQueue() {
  await queue.empty(); // 清空waiting任务
  await queue.clean(0, 'completed');
  await queue.clean(0, 'failed');
  await queue.clean(0, 'delayed');
  
  return { success: true, message: '队列已清空' };
}

// 💡 暂停/恢复队列
async function pauseQueue() {
  await queue.pause();
  return { success: true, message: '队列已暂停' };
}

async function resumeQueue() {
  await queue.resume();
  return { success: true, message: '队列已恢复' };
}

// 💡 关闭队列
async function closeQueue() {
  await queue.close();
  return { success: true, message: '队列已关闭' };
}

// API接口示例
import express from 'express';
const app = express();

app.get('/api/queue/stats', async (req, res) => {
  const stats = await getQueueStats();
  res.json(stats);
});

app.get('/api/jobs/:jobId', async (req, res) => {

  res.json(status);
});

app.get('/api/jobs/failed', async (req, res) => {

  res.json({ jobs });
});

  res.json(result);
});

app.post('/api/queue/pause', async (req, res) => {
  const result = await pauseQueue();
  res.json(result);
});

app.listen(3000);
```

### 高频错误1：Redis连接失败

**现象：**
```
Error: connect ECONNREFUSED 127.0.0.1:6379
```

**原因：**
Redis服务未启动或连接配置错误

**解决方案：**

```javascript
// ✅ 方案1：健康检查
import Redis from 'ioredis';

async function checkRedisConnection() {
  const redis = new Redis(redisConfig);
  
  try {
    await redis.ping();
    console.log('✅ Redis连接成功');
    redis.disconnect();
    return true;
  } catch (error) {
    console.error('❌ Redis连接失败:', error.message);
    return false;
  }
}

// 启动前检查
if (!(await checkRedisConnection())) {
  console.error('Redis未就绪，程序退出');
  process.exit(1);
}

// ✅ 方案2：连接重试
const redisConfig = {
  host: 'localhost',
  port: 6379,
  retryStrategy: (times) => {
    if (times > 10) {
      console.error('Redis重试次数超限，退出');
      return null; // 停止重试
    }
    
    const delay = Math.min(times * 100, 3000);
    console.log(`⏳ Redis重连中... 第${times}次，${delay}ms后重试`);
    return delay;
  },
  reconnectOnError: (err) => {
    const targetError = 'READONLY';
    if (err.message.includes(targetError)) {
      return true; // 特定错误重连
    }
    return false;
  }
};

// ✅ 方案3：监听连接事件
const queue = new Queue('tasks', { redis: redisConfig });

queue.on('error', (error) => {
  console.error('❌ 队列错误:', error.message);
  
  if (error.message.includes('ECONNREFUSED')) {
    console.error('Redis连接被拒绝，请检查Redis服务');
  }
});

queue.on('waiting', (jobId) => {
  // 连接恢复后会触发
  console.log(`✅ 连接恢复，任务 ${jobId} 等待处理`);
});
```

### 高频错误2：任务卡死（Stalled Jobs）

**现象：**
任务一直处于active状态，不完成也不失败

**原因：**
1. Worker进程崩溃
2. 任务执行超时
3. 未处理的异常

**解决方案：**

```javascript
// ✅ 方案1：设置任务超时
const queue = new Queue('tasks', {
  redis: redisConfig,
  settings: {
    lockDuration: 30000,    // 任务锁30秒
    stalledInterval: 30000, // 每30秒检查卡死任务
    maxStalledCount: 2      // 卡死2次后标记为失败
  }
});

queue.process(async (job) => {
  // 设置任务级别的超时
  const timeoutMs = 60000; // 60秒
  
  const result = await Promise.race([
    processTask(job.data),
    new Promise((_, reject) => 
      setTimeout(() => reject(new Error('任务超时')), timeoutMs)
    )
  ]);
  
  return result;
});

// ✅ 方案2：监听stalled事件
queue.on('stalled', async (job) => {
  console.warn(`⚠️ 任务卡死: ${job.id}`);
  
  // 发送告警
  await sendAlert(`任务 ${job.id} 卡死`, {
    jobId: job.id,
    data: job.data,
    attemptsMade: job.attemptsMade
  });
});

// ✅ 方案3：优雅退出
process.on('SIGTERM', async () => {
  console.log('🛑 收到SIGTERM信号，优雅退出...');
  
  // 1. 停止接收新任务
  await queue.pause(true, true); // pause(local, doNotWaitActive)
  
  // 2. 等待当前任务完成
  await queue.whenCurrentJobsFinished();
  
  // 3. 关闭队列
  await queue.close();
  
  console.log('✅ 队列已关闭');
  process.exit(0);
});

// ✅ 方案4：使用PM2自动重启
/*
// ecosystem.config.js
module.exports = {
  apps: [{
    name: 'worker',
    script: 'worker.js',
    instances: 4,
    autorestart: true,
    watch: false,
    max_memory_restart: '1G',
    // 优雅退出
    kill_timeout: 30000,
    wait_ready: true,
    listen_timeout: 10000
  }]
};
*/

// worker.js中通知PM2就绪
process.send('ready');
```

### 高频错误3：内存泄漏

**现象：**
Worker进程内存持续增长，最终崩溃

**原因：**
1. 任务数据过大
2. 事件监听器未移除
3. 完成任务未清理

**解决方案：**

```javascript
// ✅ 方案1：限制任务数据大小
await queue.add('task', data, {
  // 检查数据大小
  ...(JSON.stringify(data).length > 100000 && {
    error: '任务数据过大（>100KB）'
  })
});

// ✅ 方案2：自动清理完成任务
const queue = new Queue('tasks', {
  redis: redisConfig,
  defaultJobOptions: {
    removeOnComplete: 100,  // 只保留最近100个
    removeOnFail: 200       // 只保留最近200个
  }
});

// ✅ 方案3：定期清理
import cron from 'node-cron';

// 每小时清理1小时前的任务
cron.schedule('0 * * * *', async () => {
  console.log('🧹 开始清理旧任务...');
  
  const oneHour = 3600000;
  
  await queue.clean(oneHour, 'completed');
  await queue.clean(oneHour * 24, 'failed'); // 失败任务保留24小时
  
  console.log('✅ 清理完成');
});

// ✅ 方案4：监控内存使用
setInterval(() => {
  const usage = process.memoryUsage();
  const mb = (bytes) => Math.round(bytes / 1024 / 1024);
  
  console.log(`💾 内存使用: ${mb(usage.heapUsed)}MB / ${mb(usage.heapTotal)}MB`);
  
  // 内存超过阈值告警
  if (usage.heapUsed > 800 * 1024 * 1024) { // 800MB
    console.warn('⚠️ 内存使用过高！');
    
    // 强制垃圾回收（需要--expose-gc启动）
    if (global.gc) {
      global.gc();
    }
  }
}, 60000); // 每分钟检查

// ✅ 方案5：移除事件监听器
// ❌ 错误：每次都添加新监听器
queue.process(async (job) => {
  queue.on('completed', handleCompleted); // 内存泄漏！
});

// ✅ 正确：只添加一次
queue.on('completed', handleCompleted);

queue.process(async (job) => {
  // 处理任务
});

// 或使用once
queue.once('completed', handleCompleted);
```

## 🎯 应用场景

### 场景1：电商订单处理系统

**问题背景：**
某电商平台日订单量10万+，订单创建后需要：扣减库存 → 发送支付链接 → 15分钟未支付自动取消。直接同步处理会导致接口超时，用户体验差。

**技术选型理由：**
1. **Bull + Redis**：高性能，支持延迟任务
2. **任务流程编排**：多步骤异步处理
3. **自动重试**：网络抖动自动恢复

**完整实现：**

```javascript
// order-queue-system.js
import Queue from 'bull';
import express from 'express';

const app = express();
app.use(express.json());

// 💡 创建多个队列
const orderQueue = new Queue('order', { redis: redisConfig });
const inventoryQueue = new Queue('inventory', { redis: redisConfig });
const paymentQueue = new Queue('payment', { redis: redisConfig });
const notificationQueue = new Queue('notification', { redis: redisConfig });

// 💡 订单创建接口
app.post('/api/orders', async (req, res) => {
  const { userId, productId, quantity } = req.body;
  
  try {
    // 1. 快速创建订单记录
    const order = await db.orders.create({
      id: generateOrderId(),
      userId,
      productId,
      quantity,
      status: 'pending', // 待支付
      createdAt: Date.now()
    });
    
    // 2. 💡 添加到订单处理队列
    await orderQueue.add('process-order', {
      orderId: order.id,
      userId,
      productId,
      quantity
    }, {
      jobId: order.id, // 使用订单ID防重复
      attempts: 3,
      backoff: {
        type: 'exponential',
        delay: 2000
      }
    });
    
    // 3. 💡 添加延迟取消任务（15分钟后）
    await orderQueue.add('cancel-unpaid', {
      orderId: order.id
    }, {
      delay: 15 * 60 * 1000, // 15分钟
      jobId: `cancel-${order.id}`
    });
    
    // 4. 立即返回
    res.status(201).json({
      success: true,
      order: {
        id: order.id,
        status: 'pending'
      },
      message: '订单创建成功，请在15分钟内完成支付'
    });
  } catch (error) {
    console.error('订单创建失败:', error);
    res.status(500).json({ error: '订单创建失败' });
  }
});

// 💡 Worker 1：处理订单（扣减库存 → 生成支付链接）
orderQueue.process('process-order', async (job) => {
  const { orderId, productId, quantity } = job.data;
  
  console.log(`📦 处理订单: ${orderId}`);
  
  try {
    // 1. 扣减库存
    job.progress(20);
    const inventoryResult = await inventoryQueue.add('deduct', {
      productId,
      quantity,
      orderId
    }).finished(); // 等待库存任务完成
    
    if (!inventoryResult.success) {
      throw new Error('库存不足');
    }
    
    // 2. 生成支付链接
    job.progress(60);
    const paymentResult = await paymentQueue.add('create-payment', {
      orderId,
      amount: inventoryResult.totalPrice
    }).finished();
    
    // 3. 发送通知
    job.progress(80);
    await notificationQueue.add('send-payment-link', {
      orderId,
      paymentUrl: paymentResult.paymentUrl
    });
    
    // 4. 更新订单状态
    await db.orders.update(orderId, {
      status: 'awaiting_payment',
      paymentUrl: paymentResult.paymentUrl
    });
    
    job.progress(100);
    
    return {
      success: true,
      orderId,
      paymentUrl: paymentResult.paymentUrl
    };
  } catch (error) {
    console.error(`订单处理失败: ${orderId}`, error);
    
    // 回滚库存
    if (error.message !== '库存不足') {
      await inventoryQueue.add('rollback', {
        productId,
        quantity,
        orderId
      });
    }
    
    // 更新订单状态
    await db.orders.update(orderId, {
      status: 'failed',
      failReason: error.message
    });
    
    throw error;
  }
});

// 💡 Worker 2：取消未支付订单
orderQueue.process('cancel-unpaid', async (job) => {
  const { orderId } = job.data;
  
  console.log(`⏰ 检查订单支付状态: ${orderId}`);
  
  const order = await db.orders.findOne({ id: orderId });
  
  // 如果还是待支付状态，则取消
  if (order.status === 'awaiting_payment') {
    console.log(`❌ 订单超时未支付，自动取消: ${orderId}`);
    
    // 1. 恢复库存
    await inventoryQueue.add('rollback', {
      productId: order.productId,
      quantity: order.quantity,
      orderId
    });
    
    // 2. 更新订单状态
    await db.orders.update(orderId, {
      status: 'cancelled',
      cancelReason: '超时未支付',
      cancelledAt: Date.now()
    });
    
    // 3. 发送通知
    await notificationQueue.add('send-cancel-notice', {
      orderId,
      userId: order.userId
    });
    
    return { success: true, cancelled: true };
  }
  
  return { success: true, cancelled: false, status: order.status };
});

// 💡 Worker 3：库存操作
inventoryQueue.process('deduct', async (job) => {
  const { productId, quantity, orderId } = job.data;
  
  const product = await db.products.findOne({ id: productId });
  
  if (product.stock < quantity) {
    throw new Error('库存不足');
  }
  
  // 扣减库存
  await db.products.update(productId, {
    stock: product.stock - quantity
  });
  
  // 记录库存日志
  await db.inventoryLogs.create({
    orderId,
    productId,
    quantity: -quantity,
    type: 'deduct',
    timestamp: Date.now()
  });
  
  return {
    success: true,
    totalPrice: product.price * quantity
  };
});

inventoryQueue.process('rollback', async (job) => {
  const { productId, quantity, orderId } = job.data;
  
  // 恢复库存
  await db.products.increment(productId, { stock: quantity });
  
  // 记录库存日志
  await db.inventoryLogs.create({
    orderId,
    productId,
    quantity: +quantity,
    type: 'rollback',
    timestamp: Date.now()
  });
  
  return { success: true };
});

// 💡 支付回调
app.post('/api/payment/callback', async (req, res) => {
  const { orderId, status } = req.body;
  
  if (status === 'success') {
    // 支付成功，移除取消任务
    const cancelJob = await orderQueue.getJob(`cancel-${orderId}`);
    if (cancelJob) {
      await cancelJob.remove();
      console.log(`✅ 已移除取消任务: ${orderId}`);
    }
    
    // 更新订单状态
    await db.orders.update(orderId, {
      status: 'paid',
      paidAt: Date.now()
    });
    
    // 添加发货任务
    await orderQueue.add('ship-order', { orderId });
  }
  
  res.json({ success: true });
});

app.listen(3000);
```

**可量化效果：**
- ✅ 接口响应时间：从5000ms降至50ms（降低99%）
- ✅ 订单处理成功率：99.5%
- ✅ 超时取消准确性：100%（15分钟±5秒）
- ✅ 库存准确性：100%（有回滚机制）
- ✅ 峰值处理能力：1000订单/秒

## 📋 总结

### 核心要点

1. **Bull架构**：Producer → Queue → Worker → Events
2. **Redis存储**：List（队列） + Sorted Set（优先级/延迟） + Hash（数据）
3. **并发控制**：process(concurrency)设置并发数
4. **事件监听**：completed、failed、progress、stalled

### 实现关键

1. **连接配置**：retryStrategy、连接池
2. **任务选项**：attempts、backoff、removeOnComplete
3. **Worker并发**：根据CPU核心数和任务类型
4. **清理策略**：定期清理完成/失败任务

### 应用价值

- **异步处理**：接口响应时间降低90%+
- **削峰填谷**：平滑处理突发流量
- **可靠性**：自动重试+死信队列
- **可扩展**：水平扩展Worker节点

### 面试重点

**Q1：Bull的任务生命周期？**
答：waiting → active → completed/failed，支持延迟（delayed）和卡死（stalled）状态。

**Q2：如何防止任务重复执行？**
答：使用jobId去重，相同jobId的任务只会执行一次。

**Q3：Bull如何保证任务不丢失？**
答：Redis持久化（AOF/RDB）+ 任务重试机制 + 卡死检测。

**Q4：如何监控队列健康？**
答：Bull Board可视化 + Prometheus指标 + 事件监听。

### 注意事项

⚠️ **Redis持久化**：生产环境必须开启AOF
⚠️ **连接池**：避免为每个任务创建新连接
⚠️ **内存限制**：定期清理完成任务
⚠️ **优雅退出**：SIGTERM信号处理
