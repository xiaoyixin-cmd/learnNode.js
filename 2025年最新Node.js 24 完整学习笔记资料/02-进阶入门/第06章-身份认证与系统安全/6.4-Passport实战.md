# 6.4 Passport.js 实战

> LocalStrategy、JWTStrategy 自定义，中间件集成到Express/Koa

## 📖 知识点概述

### 核心概念

**Passport.js**是Node.js中最流行的身份认证中间件，提供500+种认证策略（Strategies），包括本地认证、OAuth、OpenID、JWT等。它通过**策略模式**（Strategy Pattern）抽象认证逻辑，使你可以灵活组合多种认证方式。

### 核心架构

```
┌──────────────────────────────────────────────────────────┐
│                    Passport.js                            │
├──────────────────────────────────────────────────────────┤
│  1. Strategies（策略）                                     │
│     ├─ passport-local (用户名+密码)                        │
│     ├─ passport-jwt (JWT Token)                           │
│     ├─ passport-google-oauth20 (Google登录)               │
│     ├─ passport-github2 (GitHub登录)                      │
│     └─ 500+ 其他策略...                                    │
│                                                            │
│  2. Serialize/Deserialize（序列化）                        │
│     ├─ serializeUser: User对象 → Session ID               │
│     └─ deserializeUser: Session ID → User对象             │
│                                                            │
│  3. Middleware（中间件）                                   │
│     ├─ passport.authenticate() - 认证                     │
│     ├─ passport.initialize() - 初始化                     │
│     └─ passport.session() - Session支持                   │
└──────────────────────────────────────────────────────────┘
```

### 工作流程

```
                    请求认证
                       │
                       ▼
         ┌──────────────────────┐
         │  passport.initialize()│
         │  (初始化Passport)      │
         └──────────┬───────────┘
                    │
                    ▼
         ┌──────────────────────┐
         │  passport.session()   │
         │  (恢复用户Session)     │
         └──────────┬───────────┘
                    │
                    ▼
         ┌──────────────────────┐
         │passport.authenticate()│
         │  (执行认证策略)        │
         └──────────┬───────────┘
                    │
           ┌────────┴────────┐
           │                 │
           ▼                 ▼
        认证成功           认证失败
           │                 │
           │                 ▼
           │          返回401错误
           │
           ▼
   ┌────────────────┐
   │ serializeUser  │  ← 将用户对象存入Session
   │ (user → id)    │
   └────────────────┘
           │
           ▼
     用户已认证
```

### 解决的问题

- **统一认证接口**：多种认证方式共用一套API
- **策略可插拔**：轻松切换或组合认证方式
- **Session管理**：自动处理用户序列化/反序列化
- **代码复用**：认证逻辑与业务逻辑分离

### Passport vs 手写认证

| 维度 | 手写认证 | Passport.js |
|------|---------|-------------|

| **维护成本** | 高 | 低 |

### 典型应用场景

- **多种登录方式**：邮箱+密码、手机+验证码、Google、GitHub
- **微服务认证**：JWT Token在服务间传递
- **企业SSO**：SAML、LDAP集成

## 🚀 使用指南

### 理解：Passport的三大核心概念

#### 1. Strategy（策略）

```javascript
// 策略定义了"如何验证用户身份"
passport.use(new LocalStrategy(
  // 💡 验证函数：从数据库查用户，验证密码
  async function verify(username, password, done) {
    try {
      const user = await db.getUserByUsername(username);
      
      if (!user) {
        return done(null, false, { message: '用户不存在' });
      }
      
      const validPassword = await bcrypt.compare(password, user.password);
      
      if (!validPassword) {
        return done(null, false, { message: '密码错误' });
      }
      
      // 验证通过
      return done(null, user);
    } catch (error) {
      return done(error);
    }
  }
));
```

#### 2. Serialize/Deserialize（序列化）

```javascript
// 💡 登录成功后，将完整User对象压缩为ID存入Session
passport.serializeUser((user, done) => {
  done(null, user.id); // 只存储ID
});

// 💡 后续请求时，从ID恢复完整User对象
passport.deserializeUser(async (id, done) => {
  try {
    const user = await db.getUserById(id);
    done(null, user);
  } catch (error) {
    done(error);
  }
});

// Session存储示例：
// req.session = {
//   passport: {
//     user: 123  ← 只存ID，不存整个对象
//   }
// }
```

#### 3. Middleware（中间件）

```javascript
// 💡 认证中间件
app.post('/login',
  passport.authenticate('local', {
    successRedirect: '/profile',
    failureRedirect: '/login',
    failureFlash: true
  })
);

// 💡 保护路由
function isAuthenticated(req, res, next) {
  if (req.isAuthenticated()) {
    return next();
  }
  res.redirect('/login');
}

app.get('/profile', isAuthenticated, (req, res) => {
  res.json({ user: req.user });
});
```

```javascript
// passport-local-example.js
import express from 'express';
import passport from 'passport';
import { Strategy as LocalStrategy } from 'passport-local';
import session from 'express-session';
import bcrypt from 'bcrypt';

const app = express();
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// 💡 Session配置（Passport需要）
app.use(session({
  secret: process.env.SESSION_SECRET || 'keyboard-cat',
  resave: false,
  saveUninitialized: false,
  cookie: {
    secure: process.env.NODE_ENV === 'production',
    httpOnly: true,
    maxAge: 24 * 60 * 60 * 1000 // 24小时
  }
}));

// 💡 初始化Passport
app.use(passport.initialize());
app.use(passport.session()); // 必须在session后面

// 模拟数据库
const users = [
  {
    id: 1,
    username: 'john',
    email: 'john@example.com',
    // bcrypt hash of 'password123'
    password: '$2b$10$CwTycUXWue0Thq9StjUM0uJ8qDXx5w5mJZNKJzH.JW.Fj/bXJ5Lx6',
    role: 'user'
  },
  {
    id: 2,
    username: 'admin',
    email: 'admin@example.com',
    password: '$2b$10$CwTycUXWue0Thq9StjUM0uJ8qDXx5w5mJZNKJzH.JW.Fj/bXJ5Lx6',
    role: 'admin'
  }
];

// 💡 配置LocalStrategy
passport.use(new LocalStrategy(
  {
    usernameField: 'email', // 默认是'username'，这里改为'email'
    passwordField: 'password',
    passReqToCallback: false // 是否将req传给verify函数
  },
  // 💡 验证函数
  async function verify(email, password, done) {
    try {
      console.log('🔍 验证用户:', email);
      
      // 1. 查找用户
      const user = users.find(u => u.email === email);
      
      if (!user) {
        console.log('❌ 用户不存在');
        return done(null, false, { message: '邮箱或密码错误' });
      }
      
      // 2. 验证密码
      const validPassword = await bcrypt.compare(password, user.password);
      
      if (!validPassword) {
        console.log('❌ 密码错误');
        return done(null, false, { message: '邮箱或密码错误' });
      }
      
      // 3. 验证通过
      console.log('✅ 验证成功:', user.username);
      return done(null, user);
    } catch (error) {
      console.error('验证出错:', error);
      return done(error);
    }
  }
));

// 💡 序列化：User对象 → Session ID
passport.serializeUser((user, done) => {
  console.log('📦 序列化用户:', user.id);
  done(null, user.id);
});

// 💡 反序列化：Session ID → User对象
passport.deserializeUser(async (id, done) => {
  try {
    console.log('📂 反序列化用户:', id);
    const user = users.find(u => u.id === id);
    
    if (!user) {
      return done(new Error('用户不存在'));
    }
    
    // 💡 从对象中移除敏感信息
    const { password, ...userWithoutPassword } = user;
    done(null, userWithoutPassword);
  } catch (error) {
    done(error);
  }
});

// 💡 登录接口
app.post('/auth/login',
  // passport.authenticate()返回中间件
  passport.authenticate('local', {
    successRedirect: '/profile',
    failureRedirect: '/login',
    failureFlash: false
  })
);

// 💡 自定义回调（更灵活）
app.post('/auth/login-custom', (req, res, next) => {
  passport.authenticate('local', (err, user, info) => {
    if (err) {
      return res.status(500).json({ error: '服务器错误' });
    }
    
    if (!user) {
      return res.status(401).json({
        success: false,
        error: info.message || '登录失败'
      });
    }
    
    // 💡 手动登录
    req.login(user, (err) => {
      if (err) {
        return res.status(500).json({ error: '登录失败' });
      }
      
      res.json({
        success: true,
        message: '登录成功',
        user: {
          id: user.id,
          username: user.username,
          email: user.email,
          role: user.role
        }
      });
    });
  })(req, res, next);
});

// 💡 注册接口
app.post('/auth/register', async (req, res) => {
  try {
    const { username, email, password } = req.body;
    
    // 验证输入
    if (!username || !email || !password) {
      return res.status(400).json({
        success: false,
        error: '缺少必填字段'
      });
    }
    
    // 检查邮箱是否已存在
    if (users.find(u => u.email === email)) {
      return res.status(409).json({
        success: false,
        error: '邮箱已被注册'
      });
    }
    
    // 密码加密
    const hashedPassword = await bcrypt.hash(password, 10);
    
    // 创建用户
    const newUser = {
      id: users.length + 1,
      username,
      email,
      password: hashedPassword,
      role: 'user'
    };
    
    users.push(newUser);
    
    // 💡 自动登录
    req.login(newUser, (err) => {
      if (err) {
        return res.status(500).json({ error: '自动登录失败' });
      }
      
      const { password, ...userWithoutPassword } = newUser;
      
      res.status(201).json({
        success: true,
        message: '注册成功',
        user: userWithoutPassword
      });
    });
  } catch (error) {
    console.error('注册失败:', error);
    res.status(500).json({
      success: false,
      error: '服务器错误'
    });
  }
});

// 💡 受保护的路由
app.get('/profile', (req, res) => {
  // 💡 Passport自动将用户附加到req.user
  if (!req.isAuthenticated()) {
    return res.status(401).json({
      success: false,
      error: '需要登录'
    });
  }
  
  res.json({
    success: true,
    user: req.user
  });
});

// 💡 登出
app.post('/auth/logout', (req, res) => {
  req.logout((err) => {
    if (err) {
      return res.status(500).json({ error: '登出失败' });
    }
    
    res.json({
      success: true,
      message: '登出成功'
    });
  });
});

// 💡 认证中间件（可复用）
function isAuthenticated(req, res, next) {
  if (req.isAuthenticated()) {
    return next();
  }
  res.status(401).json({ error: '需要登录' });
}

function hasRole(role) {
  return (req, res, next) => {
    if (!req.isAuthenticated()) {
      return res.status(401).json({ error: '需要登录' });
    }
    
    if (req.user.role !== role) {
      return res.status(403).json({ error: '权限不足' });
    }
    
    next();
  };
}

// 使用
app.get('/admin', hasRole('admin'), (req, res) => {
  res.json({ message: '管理员页面' });
});

app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});
```

#### 第二步：JWTStrategy（无状态认证）

```javascript
// passport-jwt-example.js
import express from 'express';
import passport from 'passport';
import { Strategy as JwtStrategy, ExtractJwt } from 'passport-jwt';
import jwt from 'jsonwebtoken';
import bcrypt from 'bcrypt';

const app = express();
app.use(express.json());

const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

// 模拟数据库
const users = [
  {
    id: 1,
    username: 'john',
    email: 'john@example.com',
    password: '$2b$10$CwTycUXWue0Thq9StjUM0uJ8qDXx5w5mJZNKJzH.JW.Fj/bXJ5Lx6',
    role: 'user'
  }
];

// 💡 配置JWTStrategy
const jwtOptions = {
  // 💡 从哪里提取JWT
  jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(), // "Authorization: Bearer TOKEN"
  // 或其他方式：
  // jwtFromRequest: ExtractJwt.fromHeader('x-access-token'),
  // jwtFromRequest: ExtractJwt.fromBodyField('token'),
  // jwtFromRequest: ExtractJwt.fromUrlQueryParameter('token'),
  
  // 💡 验证密钥
  secretOrKey: JWT_SECRET,
  
  // 💡 可选：签发者和接收者
  issuer: 'my-app',
  audience: 'my-app-users',
  
  // 💡 可选：允许的算法
  algorithms: ['HS256']
};

passport.use(new JwtStrategy(
  jwtOptions,
  // 💡 验证函数：JWT已验证通过，这里只需查询用户
  async function verify(jwtPayload, done) {
    try {
      console.log('🔍 JWT Payload:', jwtPayload);
      
      // 1. 从Payload中提取用户ID
      const userId = jwtPayload.sub || jwtPayload.userId;
      
      // 2. 查询用户
      const user = users.find(u => u.id === userId);
      
      if (!user) {
        return done(null, false);
      }
      
      // 3. 返回用户
      const { password, ...userWithoutPassword } = user;
      return done(null, userWithoutPassword);
    } catch (error) {
      return done(error, false);
    }
  }
));

// 💡 登录接口（生成JWT）
app.post('/auth/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    // 验证用户
    const user = users.find(u => u.email === email);
    
    if (!user) {
      return res.status(401).json({
        success: false,
        error: '邮箱或密码错误'
      });
    }
    
    const validPassword = await bcrypt.compare(password, user.password);
    
    if (!validPassword) {
      return res.status(401).json({
        success: false,
        error: '邮箱或密码错误'
      });
    }
    
    // 💡 生成JWT
    const token = jwt.sign(
      {
        sub: user.id,        // Subject（用户ID）
        username: user.username,
        email: user.email,
        role: user.role
      },
      JWT_SECRET,
      {
        expiresIn: '1h',
        issuer: 'my-app',
        audience: 'my-app-users'
      }
    );
    
    res.json({
      success: true,
      token,
      tokenType: 'Bearer',
      expiresIn: 3600,
      user: {
        id: user.id,
        username: user.username,
        email: user.email,
        role: user.role
      }
    });
  } catch (error) {
    console.error('登录失败:', error);
    res.status(500).json({
      success: false,
      error: '服务器错误'
    });
  }
});

// 💡 受保护的路由（使用JWT认证）
app.get('/profile',
  passport.authenticate('jwt', { session: false }), // session: false表示不使用Session
  (req, res) => {
    res.json({
      success: true,
      user: req.user
    });
  }
);

// 💡 认证中间件（可复用）
const authenticateJWT = passport.authenticate('jwt', { session: false });

app.get('/api/users', authenticateJWT, (req, res) => {
  res.json({
    success: true,
    users: users.map(({ password, ...user }) => user)
  });
});

// 💡 角色权限中间件
function requireRole(role) {
  return [
    authenticateJWT,
    (req, res, next) => {
      if (req.user.role !== role) {
        return res.status(403).json({
          success: false,
          error: '权限不足'
        });
      }
      next();
    }
  ];
}

app.delete('/api/users/:id', requireRole('admin'), (req, res) => {
  res.json({ message: '用户已删除' });
});

app.listen(3000);
```

#### 第三步：多策略组合

```javascript
// passport-multi-strategy.js
import express from 'express';
import passport from 'passport';
import { Strategy as LocalStrategy } from 'passport-local';
import { Strategy as JwtStrategy, ExtractJwt } from 'passport-jwt';
import { Strategy as GoogleStrategy } from 'passport-google-oauth20';
import session from 'express-session';
import jwt from 'jsonwebtoken';
import bcrypt from 'bcrypt';

const app = express();
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

app.use(session({
  secret: process.env.SESSION_SECRET,
  resave: false,
  saveUninitialized: false
}));

app.use(passport.initialize());
app.use(passport.session());

const users = [];
const JWT_SECRET = process.env.JWT_SECRET;

// 💡 策略1：LocalStrategy（传统登录）
passport.use('local', new LocalStrategy(
  { usernameField: 'email' },
  async (email, password, done) => {
    try {
      const user = users.find(u => u.email === email && u.provider === 'local');
      
      if (!user) {
        return done(null, false, { message: '邮箱或密码错误' });
      }
      
      const validPassword = await bcrypt.compare(password, user.password);
      
      if (!validPassword) {
        return done(null, false, { message: '邮箱或密码错误' });
      }
      
      return done(null, user);
    } catch (error) {
      return done(error);
    }
  }
));

// 💡 策略2：JWTStrategy（API认证）
passport.use('jwt', new JwtStrategy(
  {
    jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
    secretOrKey: JWT_SECRET
  },
  async (jwtPayload, done) => {
    try {
      const user = users.find(u => u.id === jwtPayload.sub);
      
      if (!user) {
        return done(null, false);
      }
      
      return done(null, user);
    } catch (error) {
      return done(error);
    }
  }
));

// 💡 策略3：GoogleStrategy（OAuth登录）
passport.use('google', new GoogleStrategy(
  {
    clientID: process.env.GOOGLE_CLIENT_ID,
    clientSecret: process.env.GOOGLE_CLIENT_SECRET,
    callbackURL: 'http://localhost:3000/auth/google/callback'
  },
  async (accessToken, refreshToken, profile, done) => {
    try {
      // 查找或创建用户
      let user = users.find(u => u.googleId === profile.id);
      
      if (!user) {
        user = {
          id: users.length + 1,
          googleId: profile.id,
          username: profile.displayName,
          email: profile.emails[0].value,
          avatar: profile.photos[0].value,
          provider: 'google'
        };
        users.push(user);
      }
      
      return done(null, user);
    } catch (error) {
      return done(error);
    }
  }
));

// 💡 序列化（所有策略共用）
passport.serializeUser((user, done) => {
  done(null, { id: user.id, provider: user.provider });
});

passport.deserializeUser(async (data, done) => {
  try {
    const user = users.find(u => u.id === data.id && u.provider === data.provider);
    
    if (!user) {
      return done(new Error('用户不存在'));
    }
    
    const { password, ...userWithoutPassword } = user;
    done(null, userWithoutPassword);
  } catch (error) {
    done(error);
  }
});

// 💡 路由1：传统登录（使用LocalStrategy + Session）
app.post('/auth/login', (req, res, next) => {
  passport.authenticate('local', (err, user, info) => {
    if (err) return res.status(500).json({ error: '服务器错误' });
    if (!user) return res.status(401).json({ error: info.message });
    
    req.login(user, (err) => {
      if (err) return res.status(500).json({ error: '登录失败' });
      
      res.json({
        success: true,
        message: '登录成功',
        user: req.user
      });
    });
  })(req, res, next);
});

// 💡 路由2：生成JWT（使用LocalStrategy验证 + 返回JWT）
app.post('/auth/token', (req, res, next) => {
  passport.authenticate('local', (err, user, info) => {
    if (err) return res.status(500).json({ error: '服务器错误' });
    if (!user) return res.status(401).json({ error: info.message });
    
    // 生成JWT
    const token = jwt.sign(
      { sub: user.id, username: user.username },
      JWT_SECRET,
      { expiresIn: '1h' }
    );
    
    res.json({
      success: true,
      token,
      user: { id: user.id, username: user.username, email: user.email }
    });
  })(req, res, next);
});

// 💡 路由3：Google登录
app.get('/auth/google',
  passport.authenticate('google', {
    scope: ['profile', 'email']
  })
);

app.get('/auth/google/callback',
  passport.authenticate('google', {
    successRedirect: '/profile',
    failureRedirect: '/login'
  })
);

// 💡 受保护路由（支持Session和JWT双重认证）
function authenticateMultiple(req, res, next) {
  // 尝试JWT认证
  passport.authenticate('jwt', { session: false }, (err, user) => {
    if (user) {
      req.user = user;
      return next();
    }
    
    // JWT失败，尝试Session认证
    if (req.isAuthenticated()) {
      return next();
    }
    
    res.status(401).json({ error: '需要登录' });
  })(req, res, next);
}

app.get('/profile', authenticateMultiple, (req, res) => {
  res.json({
    success: true,
    user: req.user,
    authMethod: req.headers.authorization ? 'JWT' : 'Session'
  });
});

app.listen(3000);
```

## 📋 总结

### 核心要点

1. **Strategy（策略）**：定义"如何验证身份"
2. **Serialize/Deserialize**：User对象与Session ID相互转换
3. **Middleware**：`passport.authenticate()`执行认证
4. **无状态JWT**：使用`JwtStrategy`+ `session: false`

### 实现关键

1. **初始化顺序**：`session()` → `passport.initialize()` → `passport.session()`
2. **LocalStrategy**：适合传统Web应用（HTML表单）
3. **JWTStrategy**：适合API和移动端
4. **自定义回调**：使用`(err, user, info) => {}`获得更多控制

### 应用价值

- **快速集成**：1小时实现完整认证系统
- **策略可插拔**：轻松添加Google/GitHub登录
- **代码复用**：统一的`req.user`和`req.isAuthenticated()`
- **社区支持**：500+ 认证策略

### 面试重点

**Q1：Passport的工作原理？**
答：通过Strategy模式抽象认证逻辑，验证成功后调用`serializeUser`将用户存入Session，后续请求通过`deserializeUser`恢复用户对象。

**Q2：serializeUser和deserializeUser的作用？**
答：`serializeUser`将完整User对象压缩为ID存入Session（节省空间），`deserializeUser`从ID恢复完整对象。

**Q3：JWT认证为什么不需要Session？**
答：JWT是无状态的，Token本身包含用户信息。使用`session: false`告诉Passport不要序列化/反序列化。

**Q4：如何实现"记住我"功能？**
答：使用长期Cookie存储Token，配置Session的`maxAge`为30天，或使用Refresh Token机制。

### 注意事项

⚠️ **初始化顺序**：`passport.session()`必须在`session()`之后
⚠️ **JWT不需要Session**：使用`session: false`
⚠️ **密码不要存明文**：使用bcrypt加密
⚠️ **验证错误不要泄露信息**：统一返回"邮箱或密码错误"
