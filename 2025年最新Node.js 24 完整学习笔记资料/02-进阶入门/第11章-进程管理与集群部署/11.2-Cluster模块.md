## 📖 知识点概述

### 核心概念

**Cluster模块**是Node.js内置的多进程解决方案，允许创建共享同一端口的子进程（工作进程），实现负载均衡和高可用。主进程负责管理工作进程，工作进程处理实际请求。

### Cluster架构

```
┌──────────────────────────────────────────────────────────┐
│              Cluster 多进程架构                            │
├──────────────────────────────────────────────────────────┤
│                                                            │
│  主进程（Master Process）                                  │
│  ├─ 不处理请求                                            │
│  ├─ 管理工作进程生命周期                                   │
│  ├─ 监听工作进程异常                                       │
│  ├─ 负载均衡调度                                          │
│  └─ 进程间通信（IPC）                                      │
│                                                            │
│  ─────────────────────────────────────                    │
│                                                            │
│  工作进程1（Worker 1）      [Port 3000]                    │
│  ├─ 处理客户端请求                                         │
│  ├─ 运行应用代码                                          │
│  └─ 独立V8实例和内存                                       │
│                                                            │
│  工作进程2（Worker 2）      [Port 3000]                    │
│  ├─ 处理客户端请求                                         │
│  ├─ 运行应用代码                                          │
│  └─ 独立V8实例和内存                                       │
│                                                            │
│  工作进程3（Worker 3）      [Port 3000]                    │
│  工作进程4（Worker 4）      [Port 3000]                    │
│  ...                                                       │
│                                                            │
│  💡 所有工作进程共享同一端口                                │
│  💡 主进程使用Round-Robin分发请求                          │
│                                                            │
└──────────────────────────────────────────────────────────┘
```

### 负载均衡策略

| 策略 | 说明 | 平台 |
|------|------|------|
| **Round-Robin（轮询）** | 依次分发请求到各工作进程 | 非Windows |
| **操作系统调度** | 由操作系统决定分发 | Windows |

### 理解：基础架构

#### 最简单的Cluster示例

```javascript
// cluster-basic.js
import cluster from 'cluster';
import http from 'http';
import os from 'os';

const numCPUs = os.cpus().length;

if (cluster.isPrimary) {
  console.log(`主进程 ${process.pid} 正在运行`);
  
  // 💡 创建工作进程
  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }
  
  // 监听工作进程退出
  cluster.on('exit', (worker, code, signal) => {
    console.log(`工作进程 ${worker.process.pid} 退出`);
  });
} else {
  // 💡 工作进程运行应用
  http.createServer((req, res) => {
    res.writeHead(200);
    res.end(`来自进程 ${process.pid} 的响应\n`);
  }).listen(8000);
  
  console.log(`工作进程 ${process.pid} 已启动`);
}

// 运行：node cluster-basic.js
```

### 实践：进程间通信（IPC）

#### 主进程向工作进程发送消息

```javascript
// cluster-ipc-master-to-worker.js
import cluster from 'cluster';
import os from 'os';

if (cluster.isPrimary) {
  const workers = [];
  
  for (let i = 0; i < os.cpus().length; i++) {
    const worker = cluster.fork();
    workers.push(worker);
  }
  
  // 💡 主进程向所有工作进程广播消息
  function broadcast(message) {
    workers.forEach(worker => {
      worker.send(message);
    });
  }
  
  // 定期发送配置更新
  setInterval(() => {
    broadcast({
      type: 'config-update',
      data: {
        maxConnections: 1000,
        timeout: 30000
      }
    });
  }, 10000);
  
  // 💡 向特定工作进程发送消息
  setTimeout(() => {
    const targetWorker = workers[0];
    targetWorker.send({
      type: 'custom-task',
      data: { task: 'cleanup' }
    });
  }, 5000);
} else {
  // 💡 工作进程接收消息
  process.on('message', (msg) => {
    console.log(`工作进程 ${process.pid} 收到消息:`, msg);
    
    if (msg.type === 'config-update') {
      // 更新配置
      console.log('更新配置:', msg.data);
    } else if (msg.type === 'custom-task') {
      // 执行特定任务
      console.log('执行任务:', msg.data.task);
    }
  });
  
  // 工作进程运行应用
  require('./app.js');
}
```

#### 工作进程向主进程发送消息

```javascript
// cluster-ipc-worker-to-master.js
import cluster from 'cluster';
import express from 'express';
import os from 'os';

if (cluster.isPrimary) {
  const stats = {
    totalRequests: 0,
    workerRequests: {}
  };
  
  for (let i = 0; i < os.cpus().length; i++) {
    const worker = cluster.fork();
    stats.workerRequests[worker.id] = 0;
    
    // 💡 主进程接收工作进程消息
    worker.on('message', (msg) => {
      if (msg.type === 'request-count') {
        stats.totalRequests++;
        stats.workerRequests[worker.id]++;
        
        console.log(`总请求数: ${stats.totalRequests}`);
        console.log('各工作进程请求数:', stats.workerRequests);
      } else if (msg.type === 'error') {
        console.error(`工作进程 ${worker.process.pid} 报告错误:`, msg.error);
      }
    });
  }
  
  // 定期显示统计信息
  setInterval(() => {
    console.log('\n=== 统计信息 ===');
    console.log(`总请求数: ${stats.totalRequests}`);
    console.log('各工作进程请求数:', stats.workerRequests);
  }, 10000);
} else {
  const app = express();
  
  app.get('/', (req, res) => {
    // 💡 向主进程报告请求
    process.send({ type: 'request-count' });
    
    res.json({
      pid: process.pid,
      message: 'Hello from cluster'
    });
  });
  
  app.get('/error', (req, res) => {
    try {
      throw new Error('测试错误');
    } catch (error) {
      // 💡 向主进程报告错误
      process.send({
        type: 'error',
        error: {
          message: error.message,
          stack: error.stack
        }
      });
      
      res.status(500).json({ error: error.message });
    }
  });
  
  app.listen(3000);
}
```

### 实践：优雅重启（Zero-Downtime）

#### 无缝重启实现

```javascript
// cluster-graceful-restart.js
import cluster from 'cluster';
import express from 'express';
import os from 'os';

if (cluster.isPrimary) {
  const workers = new Map();
  const numCPUs = os.cpus().length;
  
  console.log(`主进程 ${process.pid} 正在运行`);
  
  // 创建工作进程
  for (let i = 0; i < numCPUs; i++) {
    createWorker();
  }
  
  function createWorker() {
    const worker = cluster.fork();
    workers.set(worker.id, worker);
    
    worker.on('message', (msg) => {
      if (msg.cmd === 'ready') {
        console.log(`工作进程 ${worker.process.pid} 已就绪`);
      }
    });
  }
  
  // 💡 优雅重启函数
  function gracefulRestart() {
    const workerIds = Array.from(workers.keys());
    let currentIndex = 0;
    
    function restartNext() {
      if (currentIndex >= workerIds.length) {
        console.log('所有工作进程已重启完成');
        return;
      }
      
      const workerId = workerIds[currentIndex];
      const worker = workers.get(workerId);
      
      if (!worker) {
        currentIndex++;
        restartNext();
        return;
      }
      
      console.log(`正在重启工作进程 ${worker.process.pid}...`);
      
      // 创建新工作进程
      const newWorker = cluster.fork();
      
      // 等待新工作进程就绪
      newWorker.once('message', (msg) => {
        if (msg.cmd === 'ready') {
          console.log(`新工作进程 ${newWorker.process.pid} 已就绪`);
          
          // 💡 关闭旧工作进程
          worker.send({ cmd: 'shutdown' });
          
          // 等待旧进程关闭或超时强制关闭
          const timeout = setTimeout(() => {
            console.log(`工作进程 ${worker.process.pid} 超时，强制关闭`);
            worker.kill('SIGKILL');
          }, 5000);
          
          worker.once('exit', () => {
            clearTimeout(timeout);
            workers.delete(workerId);
            workers.set(newWorker.id, newWorker);
            
            console.log(`工作进程 ${worker.process.pid} 已关闭`);
            
            // 继续重启下一个
            currentIndex++;
            setTimeout(restartNext, 1000);
          });
        }
      });
    }
    
    restartNext();
  }
  
  // 💡 监听SIGUSR2信号触发优雅重启
  process.on('SIGUSR2', () => {
    console.log('收到SIGUSR2信号，开始优雅重启...');
    gracefulRestart();
  });
  
  // 监听工作进程异常退出
  cluster.on('exit', (worker, code, signal) => {
    if (signal !== 'SIGTERM') {
      console.log(`工作进程 ${worker.process.pid} 异常退出，自动重启`);
      createWorker();
    }
  });
} else {
  const app = express();
  
  // 模拟长连接
  const activeConnections = new Set();
  
  app.get('/', (req, res) => {
    activeConnections.add(res);
    
    // 模拟长连接（10秒后响应）
    setTimeout(() => {
      res.json({
        pid: process.pid,
        message: 'Long request completed'
      });
      activeConnections.delete(res);
    }, 10000);
  });
  
  app.get('/quick', (req, res) => {
    res.json({ pid: process.pid, message: 'Quick response' });
  });
  
  const server = app.listen(3000);
  
  // 通知主进程已就绪
  process.send({ cmd: 'ready' });
  
  // 💡 优雅关闭
  process.on('message', (msg) => {
    if (msg.cmd === 'shutdown') {
      console.log(`工作进程 ${process.pid} 开始优雅关闭...`);
      
      // 停止接受新连接
      server.close(() => {
        console.log(`工作进程 ${process.pid} 已停止接受新连接`);
        
        // 等待现有连接完成
        const checkConnections = setInterval(() => {
          if (activeConnections.size === 0) {
            clearInterval(checkConnections);
            console.log(`工作进程 ${process.pid} 所有连接已关闭`);
            process.exit(0);
          } else {
            console.log(`工作进程 ${process.pid} 还有 ${activeConnections.size} 个连接`);
          }
        }, 1000);
      });
    }
  });
}

// 💡 使用方法：
// 1. 启动：node cluster-graceful-restart.js
// 2. 触发优雅重启：kill -SIGUSR2 <主进程PID>
// 3. 观察日志：工作进程逐个重启，不中断服务
```

## 🔧 错误处理

### 高频错误1：工作进程崩溃导致服务中断

**现象：**
```javascript
// ❌ 问题：工作进程崩溃后未自动重启
if (cluster.isPrimary) {
  cluster.fork();
  cluster.fork();
} else {
  // 工作进程代码
  throw new Error('工作进程崩溃'); // 崩溃后不会重启
}
```

**解决方案：**
```javascript
// ✅ 方案1：监听exit事件自动重启
if (cluster.isPrimary) {
  const numWorkers = os.cpus().length;
  
  for (let i = 0; i < numWorkers; i++) {
    cluster.fork();
  }
  
  // 💡 自动重启崩溃的工作进程
  cluster.on('exit', (worker, code, signal) => {
    console.log(`工作进程 ${worker.process.pid} 退出 (code: ${code}, signal: ${signal})`);
    
    if (code !== 0 && !worker.exitedAfterDisconnect) {
      console.log('工作进程异常退出，正在重启...');
      cluster.fork();
    }
  });
} else {
  // 工作进程代码
  const app = express();
  
  // 💡 捕获未处理的异常
  process.on('uncaughtException', (err) => {
    console.error(`工作进程 ${process.pid} 发生未捕获异常:`, err);
    
    // 记录错误后退出，让主进程重启
    setTimeout(() => {
      process.exit(1);
    }, 1000);
  });
  
  process.on('unhandledRejection', (reason, promise) => {
    console.error(`工作进程 ${process.pid} 发生未处理的Promise拒绝:`, reason);
  });
  
  app.listen(3000);
}
```

### 高频错误2：共享状态导致数据不一致

**现象：**
```javascript
// ❌ 问题：每个工作进程的计数器独立
let requestCount = 0;

app.get('/', (req, res) => {
  requestCount++;
  res.json({ count: requestCount, pid: process.pid });
});

// 问题：4个工作进程，每个进程的count独立计数
// Worker 1: count=5
// Worker 2: count=3
// Worker 3: count=7
// Worker 4: count=2
```

**解决方案：**
```javascript
// ✅ 方案1：使用Redis共享状态
import Redis from 'ioredis';

const redis = new Redis();

app.get('/', async (req, res) => {
  const count = await redis.incr('request_count');
  res.json({ count, pid: process.pid });
});

// ✅ 方案2：通过主进程协调
if (cluster.isPrimary) {
  let globalCount = 0;
  
  cluster.on('message', (worker, msg) => {
    if (msg.type === 'increment') {
      globalCount++;
      worker.send({ type: 'count', value: globalCount });
    }
  });
  
  // ...创建工作进程
} else {
  app.get('/', (req, res) => {
    process.send({ type: 'increment' });
    
    process.once('message', (msg) => {
      if (msg.type === 'count') {
        res.json({ count: msg.value, pid: process.pid });
      }
    });
  });
}
```

```javascript
if (cluster.isPrimary) {

  cluster.fork();
  cluster.fork();

  // ✅ 只有工作进程监听端口
```

## 📚 应用场景

### 场景1：高并发Web服务

```javascript
// production-cluster.js
import cluster from 'cluster';
import express from 'express';
import os from 'os';
import Redis from 'ioredis';

const redis = new Redis();
const numWorkers = process.env.WEB_CONCURRENCY || os.cpus().length;

if (cluster.isPrimary) {
  console.log(`🚀 启动集群模式，主进程 PID: ${process.pid}`);
  console.log(`📊 CPU核心数: ${os.cpus().length}`);
  console.log(`👷 启动 ${numWorkers} 个工作进程`);
  
  // 创建工作进程
  for (let i = 0; i < numWorkers; i++) {
    const worker = cluster.fork();
    console.log(`✅ 工作进程 ${worker.process.pid} 已创建`);
  }
  
  // 监控工作进程
  cluster.on('online', (worker) => {
    console.log(`🟢 工作进程 ${worker.process.pid} 已上线`);
  });
  
  cluster.on('exit', (worker, code, signal) => {
    console.log(`🔴 工作进程 ${worker.process.pid} 退出 (code: ${code})`);
    
    if (!worker.exitedAfterDisconnect) {
      console.log(`⚠️  工作进程异常退出，正在重启...`);
      setTimeout(() => {
        cluster.fork();
      }, 1000);
    }
  });
  
  // 优雅关闭
  process.on('SIGTERM', () => {
    console.log('📥 收到SIGTERM信号，开始优雅关闭...');
    
    for (const id in cluster.workers) {
      cluster.workers[id].send({ cmd: 'shutdown' });
    }
    
    setTimeout(() => {
      console.log('⏱️  超时，强制关闭');
      process.exit(0);
    }, 30000);
  });
} else {
  const app = express();
  
  // 中间件
  app.use(express.json());
  
  // 健康检查
  app.get('/health', (req, res) => {
    res.json({
      status: 'ok',
      pid: process.pid,
      uptime: process.uptime(),
      memory: process.memoryUsage()
    });
  });
  
  // API路由
  app.get('/api/products', async (req, res) => {
    try {
      // 使用Redis缓存
      const cacheKey = 'products:all';
      const cached = await redis.get(cacheKey);
      
      if (cached) {
        return res.json({
          data: JSON.parse(cached),
          cached: true,
          pid: process.pid
        });
      }
      
      // 模拟数据库查询
      const products = await fetchProductsFromDB();
      
      // 缓存1小时
      await redis.setex(cacheKey, 3600, JSON.stringify(products));
      
      res.json({
        data: products,
        cached: false,
        pid: process.pid
      });
    } catch (error) {
      res.status(500).json({ error: error.message });
    }
  });
  
  // 错误处理
  app.use((err, req, res, next) => {
    console.error(`[Worker ${process.pid}] Error:`, err);
    res.status(500).json({ error: 'Internal Server Error' });
  });
  
  const server = app.listen(3000, () => {
    console.log(`🎯 工作进程 ${process.pid} 监听端口 3000`);
    process.send({ cmd: 'ready' });
  });
  
  // 优雅关闭
  process.on('message', (msg) => {
    if (msg.cmd === 'shutdown') {
      console.log(`📤 工作进程 ${process.pid} 开始优雅关闭...`);
      
      server.close(() => {
        console.log(`✅ 工作进程 ${process.pid} 已关闭`);
        redis.disconnect();
        process.exit(0);
      });
      
      // 超时强制关闭
      setTimeout(() => {
        console.log(`⏱️  工作进程 ${process.pid} 关闭超时，强制退出`);
        process.exit(1);
      }, 10000);
    }
  });
  
  // 异常处理
  process.on('uncaughtException', (err) => {
    console.error(`[Worker ${process.pid}] Uncaught Exception:`, err);
    process.exit(1);
  });
  
  process.on('unhandledRejection', (reason) => {
    console.error(`[Worker ${process.pid}] Unhandled Rejection:`, reason);
  });
}
```

### 场景2：定时任务集群

```javascript
// cron-cluster.js
import cluster from 'cluster';
import cron from 'node-cron';
import Redis from 'ioredis';

const redis = new Redis();

if (cluster.isPrimary) {
  console.log(`主进程 ${process.pid} 正在运行`);
  
  // 创建4个工作进程
  for (let i = 0; i < 4; i++) {
    cluster.fork();
  }
  
  cluster.on('exit', (worker) => {
    console.log(`工作进程 ${worker.process.pid} 退出`);
    cluster.fork();
  });
} else {
  console.log(`工作进程 ${process.pid} 已启动`);
  
  // 💡 使用分布式锁确保定时任务只执行一次
  async function runWithLock(taskName, fn, lockTTL = 60) {
    const lockKey = `lock:${taskName}`;
    const lockValue = `${process.pid}:${Date.now()}`;
    
    // 尝试获取锁
    const acquired = await redis.set(
      lockKey,
      lockValue,
      'EX',
      lockTTL,
      'NX'
    );
    
    if (!acquired) {
      console.log(`[${process.pid}] 任务 ${taskName} 已被其他进程执行`);
      return;
    }
    
    try {
      console.log(`[${process.pid}] 开始执行任务 ${taskName}`);
      await fn();
      console.log(`[${process.pid}] 任务 ${taskName} 执行完成`);
    } catch (error) {
      console.error(`[${process.pid}] 任务 ${taskName} 执行失败:`, error);
    } finally {
      // 释放锁
      const currentLock = await redis.get(lockKey);
      if (currentLock === lockValue) {
        await redis.del(lockKey);
      }
    }
  }
  
  // 每分钟执行一次
  cron.schedule('* * * * *', () => {
    runWithLock('sync-data', async () => {
      // 同步数据任务
      console.log('同步数据...');
      await new Promise(resolve => setTimeout(resolve, 5000));
    });
  });
  
  // 每小时执行一次
  cron.schedule('0 * * * *', () => {
    runWithLock('generate-report', async () => {
      // 生成报告任务
      console.log('生成报告...');
      await new Promise(resolve => setTimeout(resolve, 10000));
    });
  });
  
  // 保持进程运行
  setInterval(() => {}, 1000);
}

// 💡 结果：
// - 4个工作进程同时运行
// - 每个定时任务只有1个进程执行（通过Redis锁）
// - 执行进程异常退出时，其他进程可以接管
```

## 📋 总结

### 核心要点

1. **多进程架构**：主进程管理，工作进程处理请求
2. **负载均衡**：Round-Robin算法均匀分发请求
3. **进程间通信**：通过IPC实现消息传递
4. **优雅重启**：Zero-Downtime逐个重启工作进程

### 实现关键

1. **主进程职责**：创建/管理工作进程，不处理请求
2. **工作进程职责**：运行应用代码，处理请求
3. **自动重启**：监听exit事件，崩溃后自动重启
4. **共享状态**：使用Redis/数据库，不依赖进程内存

### 应用价值

- **充分利用多核**：进程数=CPU核心数
- **高可用**：单个进程崩溃不影响整体服务
- **负载均衡**：请求均匀分发到各进程
- **零停机部署**：优雅重启实现无缝更新

### 面试重点

**Q1：Cluster模块如何实现负载均衡？**
答：主进程监听端口，收到请求后通过Round-Robin算法依次分发给工作进程。在非Windows系统上使用Round-Robin，Windows上由操作系统调度。

**Q2：如何实现Zero-Downtime Deployment？**
答：逐个重启工作进程。先启动新进程，等待就绪后，再关闭旧进程。旧进程优雅关闭（等待现有连接完成）后才退出。

**Q3：Cluster模式下如何共享状态？**
答：不能依赖进程内存共享。使用Redis、数据库或主进程协调。Session可以用Redis存储，计数器用Redis INCR命令。

**Q4：工作进程崩溃怎么办？**
答：主进程监听exit事件，自动重启崩溃的工作进程。同时捕获uncaughtException和unhandledRejection，记录日志后退出。

### 注意事项

⚠️ **不要在主进程处理请求**：主进程只负责管理
⚠️ **不要依赖进程内存**：使用Redis/DB共享状态
⚠️ **优雅关闭**：监听SIGTERM，完成现有请求后退出
⚠️ **日志聚合**：多进程日志需要集中管理（如Pino）
