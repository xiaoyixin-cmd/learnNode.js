## 📖 知识点概述

### 核心概念

**进程守护（Daemon）**是指让应用在后台持续运行，并在崩溃时自动重启，确保服务高可用。Node.js应用需要进程守护来应对异常退出、系统重启等情况。

### 异常类型

| 异常类型 | 处理策略 | 示例 |
|---------|---------|------|
| **可恢复异常** | 记录日志，继续运行 | 单个请求失败 |
| **不可恢复异常** | 优雅关闭，自动重启 | 数据库连接失败 |
| **资源耗尽** | 限制资源，超限重启 | 内存泄漏 |
| **外部依赖故障** | 重试+降级 | Redis不可用 |

### 为什么需要进程守护？

```bash
# ❌ 不使用守护：进程崩溃后服务停止

# 问题：
# 1. 未捕获异常导致进程退出
# 2. 服务器重启后不会自动启动
# 3. 无法监控进程状态
# 4. 日志管理困难

# ✅ 使用守护：自动恢复服务

# 优点：
# 1. 崩溃后自动重启
# 2. 开机自动启动
# 3. 实时监控
```

#### 健康检查

```javascript
// health-check.js
import pm2 from 'pm2';
import axios from 'axios';

const APP_NAME = 'my-app';
const HEALTH_CHECK_URL = 'http://localhost:8080/health';
const CHECK_INTERVAL = 30000; // 30秒
const MAX_FAILURES = 3;

let failureCount = 0;

async function healthCheck() {
  try {
    const response = await axios.get(HEALTH_CHECK_URL, {
      timeout: 5000
    });
    
    if (response.status === 200) {
      console.log('✅ 健康检查通过');
      failureCount = 0;
    } else {
      console.log(`⚠️  健康检查失败: ${response.status}`);
      handleFailure();
    }
  } catch (error) {
    console.error(`❌ 健康检查错误: ${error.message}`);
    handleFailure();
  }
}

function handleFailure() {
  failureCount++;
  
  if (failureCount >= MAX_FAILURES) {
    console.error(`连续失败 ${MAX_FAILURES} 次，重启应用`);
    restartApp();
    failureCount = 0;
  }
}

function restartApp() {
  pm2.connect((err) => {
    if (err) {
      console.error('连接PM2失败:', err);
      return;
    }
    
    pm2.restart(APP_NAME, (err) => {
      if (err) {
        console.error('重启应用失败:', err);
      } else {
        console.log('应用已重启');
      }
      pm2.disconnect();
    });
  });
}

// 定时健康检查
setInterval(healthCheck, CHECK_INTERVAL);

console.log(`开始健康检查，间隔 ${CHECK_INTERVAL}ms`);
```

### 实践4：指数退避重启策略

#### 实现指数退避

```javascript
// restart-manager.js
class RestartManager {
  constructor(maxRetries = 5, baseDelay = 1000, maxDelay = 60000) {
    this.maxRetries = maxRetries;
    this.baseDelay = baseDelay;      // 基础延迟：1秒
    this.maxDelay = maxDelay;        // 最大延迟：60秒
    this.retryCount = 0;
    this.lastCrashTime = null;
  }
  
  // 💡 计算下次重启延迟（指数退避）
  getNextDelay() {
    const delay = Math.min(
      this.baseDelay * Math.pow(2, this.retryCount),
      this.maxDelay
    );
    
    // 添加随机抖动（避免雪崩）
    const jitter = Math.random() * delay * 0.1;
    
    return delay + jitter;
  }
  
  // 💡 处理崩溃
  handleCrash(error) {
    const now = Date.now();
    
    // 如果距上次崩溃超过5分钟，重置计数器
    if (this.lastCrashTime && (now - this.lastCrashTime) > 5 * 60 * 1000) {
      console.log('距上次崩溃超过5分钟，重置重启计数器');
      this.retryCount = 0;
    }
    
    this.lastCrashTime = now;
    this.retryCount++;
    
    console.error(`应用崩溃 (${this.retryCount}/${this.maxRetries}):`, error.message);
    
    if (this.retryCount > this.maxRetries) {
      console.error('重启次数超过限制，停止重启');
      process.exit(1);
    }
    
    const delay = this.getNextDelay();
    console.log(`将在 ${(delay / 1000).toFixed(1)} 秒后重启...`);
    
    return delay;
  }
  
  // 💡 成功启动后重置
  reset() {
    console.log('应用稳定运行，重置重启计数器');
    this.retryCount = 0;
    this.lastCrashTime = null;
  }
}

// 使用示例
import { spawn } from 'child_process';

const manager = new RestartManager();

function startApp() {
  console.log('启动应用...');
  
  const child = spawn('node', ['app.js'], {
    stdio: 'inherit'
  });
  
  // 应用稳定运行10秒后重置计数器
  const resetTimer = setTimeout(() => {
    manager.reset();
  }, 10000);
  
  child.on('exit', (code, signal) => {
    clearTimeout(resetTimer);
    
    console.log(`应用退出 (code: ${code}, signal: ${signal})`);
    
    if (code !== 0) {
      const delay = manager.handleCrash(new Error(`Exit code: ${code}`));
      
      setTimeout(() => {
        startApp();
      }, delay);
    }
  });
}

startApp();
```

```javascript
process.on('SIGINT', gracefulShutdown);

    return;
  }
  
  isShuttingDown = true;

  // 💡 异步关闭资源
  Promise.all([
    closeDatabase(),
    closeRedis(),
    flushPendingTasks()

      process.exit(1);

  }, 30000);
```

### 场景：生产环境完整监控方案

```javascript
// monitor.js
import pm2 from 'pm2';
import nodemailer from 'nodemailer';

const APP_NAME = 'my-app';
const EMAIL_CONFIG = {
  host: 'smtp.example.com',
  port: 587,
  secure: false,
  auth: {
    user: 'alert@example.com',
    pass: 'password'
  }
};

const transporter = nodemailer.createTransporter(EMAIL_CONFIG);

// 💡 监控指标阈值
const THRESHOLDS = {
  memory: 400 * 1024 * 1024,      // 400MB
  cpu: 80,                        // 80%
  restarts: 5,                    // 5次重启
  errorRate: 0.05                 // 5%错误率
};

async function checkHealth() {
  return new Promise((resolve, reject) => {
    pm2.connect((err) => {
      if (err) return reject(err);
      
      pm2.describe(APP_NAME, (err, processDescription) => {
        if (err) {
          pm2.disconnect();
          return reject(err);
        }
        
        const proc = processDescription[0];
        const { monit, pm2_env } = proc;
        
        // 检查各项指标
        const alerts = [];
        
        // 内存检查
        if (monit.memory > THRESHOLDS.memory) {
          alerts.push({
            type: 'memory',
            value: `${(monit.memory / 1024 / 1024).toFixed(2)} MB`,
            threshold: `${THRESHOLDS.memory / 1024 / 1024} MB`
          });
        }
        
        // CPU检查
        if (monit.cpu > THRESHOLDS.cpu) {
          alerts.push({
            type: 'cpu',
            value: `${monit.cpu}%`,
            threshold: `${THRESHOLDS.cpu}%`
          });
        }
        
        // 重启次数检查
        if (pm2_env.unstable_restarts > THRESHOLDS.restarts) {
          alerts.push({
            type: 'restarts',
            value: pm2_env.unstable_restarts,
            threshold: THRESHOLDS.restarts
          });
        }
        
        pm2.disconnect();
        
        if (alerts.length > 0) {
          sendAlert(alerts);
        }
        
        resolve(alerts);
      });
    });
  });
}

async function sendAlert(alerts) {
  const alertsText = alerts.map(alert => 
    `${alert.type}: ${alert.value} (阈值: ${alert.threshold})`
  ).join('\n');
  
  try {
    await transporter.sendMail({
      from: 'alert@example.com',
      to: 'admin@example.com',
      subject: `⚠️  ${APP_NAME} 应用告警`,
      text: `检测到以下异常:\n\n${alertsText}`
    });
    
    console.log('告警邮件已发送');
  } catch (error) {
    console.error('发送告警失败:', error);
  }
}

// 每5分钟检查一次
setInterval(checkHealth, 5 * 60 * 1000);

console.log('监控服务已启动');
```

## 📋 总结

### 核心要点

1. **异常监控**：uncaughtException、unhandledRejection
2. **优雅关闭**：SIGTERM/SIGINT信号处理
3. **自动重启**：PM2/Systemd/Docker守护
4. **健康检查**：定期检测应用状态

### 实现关键

1. **捕获异常**：监听process事件
2. **资源清理**：关闭数据库、Redis连接
3. **超时处理**：避免关闭挂起
4. **指数退避**：避免频繁重启

### 应用价值

- **高可用**：崩溃后自动恢复
- **零停机**：优雅关闭不丢失请求
- **可观测**：监控+告警及时发现问题
- **稳定性**：避免频繁重启

### 面试重点

**Q1：如何实现Node.js应用的进程守护？**
答：1) PM2自动重启 2) Systemd系统级守护 3) Docker重启策略 4) 自定义守护脚本。

**Q2：什么是优雅关闭？如何实现？**
答：收到关闭信号后：1) 停止接受新连接 2) 等待现有请求完成 3) 关闭数据库连接等 4) 退出进程。监听SIGTERM/SIGINT信号实现。

**Q3：uncaughtException和unhandledRejection的区别？**
答：uncaughtException是未捕获的同步异常，unhandledRejection是未处理的Promise拒绝。都应记录日志并优雅关闭。

**Q4：如何避免应用频繁重启？**
答：1) 设置min_uptime 2) 限制max_restarts 3) 指数退避延迟 4) 修复根本问题（端口冲突、配置错误等）。

### 注意事项

⚠️ **不要忽略uncaughtException**：应用可能处于不一致状态
⚠️ **优雅关闭要设超时**：避免挂起
⚠️ **重启策略要合理**：避免频繁重启循环
⚠️ **监控+告警**：及时发现问题
