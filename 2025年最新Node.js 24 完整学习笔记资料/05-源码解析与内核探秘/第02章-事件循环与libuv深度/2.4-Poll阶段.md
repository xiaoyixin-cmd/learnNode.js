# 2.4 Poll 阶段：I/O 多路复用深度解析

> 深入理解 epoll/kqueue/IOCP 在 Node.js 中的应用

---

## 📋 核心概念

Poll 阶段是事件循环中**最复杂**的阶段，负责处理 I/O 事件。

**主要功能：**
- 执行 I/O 回调（除了 timers、check、close）
- 计算阻塞时间（等待 I/O）
- 处理文件描述符就绪事件

**多路复用技术：**
```
Linux   → epoll
macOS   → kqueue
Windows → IOCP (I/O Completion Port)
```

```c
// deps/uv/src/unix/core.c
int uv_run(uv_loop_t* loop, uv_run_mode mode) {
  int timeout;
  int r;
  int can_sleep;

  r = uv__loop_alive(loop);
  if (!r)
    uv__update_time(loop);

  while (r != 0 && loop->stop_flag == 0) {
    uv__update_time(loop);
    uv__run_timers(loop);
    
    ran_pending = uv__run_pending(loop);
    uv__run_idle(loop);
    uv__run_prepare(loop);

    timeout = 0;
    
    // 计算 poll 超时时间
    if ((mode == UV_RUN_ONCE && !ran_pending) || mode == UV_RUN_DEFAULT)
      timeout = uv__backend_timeout(loop);

    // 🔥 核心：Poll 阶段
    uv__io_poll(loop, timeout);
    
    uv__metrics_update_idle_time(loop);
    uv__run_check(loop);
    uv__run_closing_handles(loop);

    if (mode == UV_RUN_ONCE) {
      uv__update_time(loop);
      uv__run_timers(loop);
    }

    r = uv__loop_alive(loop);
    if (mode == UV_RUN_ONCE || mode == UV_RUN_NOWAIT)
      break;
  }

  return r;
}
```

## 💻 超时时间计算

### 1. 计算逻辑

```c
// deps/uv/src/unix/core.c
int uv__backend_timeout(const uv_loop_t* loop) {
  // 🔥 情况1：有 closing handles，不等待
  if (loop->closing_handles)
    return 0;

  // 🔥 情况2：循环停止，不等待
  if (loop->stop_flag)
    return 0;

  // 🔥 情况3：没有活跃句柄/请求，不等待
  if (!uv__has_active_handles(loop) && !uv__has_active_reqs(loop))
    return 0;

  // 🔥 情况4：有 idle handles，不等待
  if (!QUEUE_EMPTY(&loop->idle_handles))
    return 0;

  // 🔥 情况5：有 pending 回调，不等待
  if (!QUEUE_EMPTY(&loop->pending_queue))
    return 0;

  // 🔥 情况6：有 closing handles，不等待
  if (loop->closing_handles)
    return 0;

  // 🔥 情况7：计算到下一个定时器的时间
  uint64_t diff;
  uv_timer_t* timer;
  struct heap_node* heap_node;

  heap_node = heap_min(timer_heap(loop));
  if (heap_node == NULL)
    return -1;  // 无限等待

  timer = container_of(heap_node, uv_timer_t, heap_node);
  
  if (timer->timeout <= loop->time)
    return 0;  // 定时器已到期

  diff = timer->timeout - loop->time;
  
  if (diff > INT_MAX)
    diff = INT_MAX;

  return (int) diff;
}
```

**超时时间决策树：**
```
有 closing handles? ─YES→ 0ms
    │
    NO
    │
有 idle handles? ─YES→ 0ms
    │
    NO
    │
有 pending? ─YES→ 0ms
    │
    NO
    │
有定时器? ─YES→ 计算差值
    │
    NO
    │
   -1 (无限等待)
```

```c
// deps/uv/src/unix/linux-core.c
void uv__io_poll(uv_loop_t* loop, int timeout) {
  struct epoll_event events[1024];
  struct epoll_event* pe;
  struct epoll_event e;
  int real_timeout;
  QUEUE* q;
  uv__io_t* w;
  uint64_t base;
  int have_signals;
  int nevents;
  int count;
  int nfds;
  int fd;
  int op;
  int i;
  int user_timeout;
  int reset_timeout;

  // 如果没有 I/O 观察者，直接返回
  if (loop->nfds == 0) {
    assert(QUEUE_EMPTY(&loop->watcher_queue));
    return;
  }

  // 处理观察者队列
  while (!QUEUE_EMPTY(&loop->watcher_queue)) {
    q = QUEUE_HEAD(&loop->watcher_queue);
    QUEUE_REMOVE(q);
    QUEUE_INIT(q);

    w = QUEUE_DATA(q, uv__io_t, watcher_queue);
    assert(w->pevents != 0);
    assert(w->fd >= 0);
    assert(w->fd < (int) loop->nwatchers);

    // 🔥 设置 epoll 事件
    e.events = w->pevents;
    e.data.fd = w->fd;

    if (w->events == 0)
      op = EPOLL_CTL_ADD;
    else
      op = EPOLL_CTL_MOD;

    // 🔥 注册/修改文件描述符
    if (epoll_ctl(loop->backend_fd, op, w->fd, &e)) {
      if (errno != EEXIST)
        abort();

      assert(op == EPOLL_CTL_ADD);

      if (epoll_ctl(loop->backend_fd, EPOLL_CTL_MOD, w->fd, &e))
        abort();
    }

    w->events = w->pevents;
  }

  assert(timeout >= -1);
  base = loop->time;
  count = 48; // 最多执行48次

  real_timeout = timeout;

  for (;;) {
    // 🔥 等待事件就绪
    nfds = epoll_wait(loop->backend_fd,
                      events,
                      ARRAY_SIZE(events),
                      real_timeout);

    // 更新时间
    SAVE_ERRNO(uv__update_time(loop));

    if (nfds == 0) {
      assert(timeout != -1);
      
      if (reset_timeout != 0) {
        timeout = user_timeout;
        reset_timeout = 0;
      }

      if (timeout == -1)
        continue;

      if (timeout == 0)
        return;

      goto update_timeout;
    }

    if (nfds == -1) {
      if (errno == EINTR)
        continue;

      abort();
    }

    have_signals = 0;
    nevents = 0;

    // 🔥 处理就绪事件
    for (i = 0; i < nfds; i++) {
      pe = events + i;
      fd = pe->data.fd;

      if (fd == -1) {
        have_signals = 1;
        continue;
      }

      assert(fd >= 0);
      assert((unsigned) fd < loop->nwatchers);

      w = loop->watchers[fd];

      if (w == NULL) {
        epoll_ctl(loop->backend_fd, EPOLL_CTL_DEL, fd, pe);
        continue;
      }

      // 🔥 设置就绪事件
      w->rnevents |= pe->events & (POLLIN | POLLOUT | POLLHUP | POLLERR);
      nevents++;
    }

    // 处理信号
    if (have_signals != 0) {
      uv__io_feed(loop, &loop->signal_io_watcher);
    }

    // 🔥 执行 I/O 回调
    loop->watchers[loop->nwatchers] = (void*) events;
    loop->watchers[loop->nwatchers + 1] = (void*) (uintptr_t) nfds;

    for (i = 0; i < (int) loop->nwatchers; i++) {
      w = loop->watchers[i];

      if (w == NULL)
        continue;

      if (w->rnevents == 0)
        continue;

      w->cb(loop, w, w->rnevents);
      nevents--;
    }

    loop->watchers[loop->nwatchers] = NULL;
    loop->watchers[loop->nwatchers + 1] = NULL;

    if (nevents != 0) {
      if (nfds == ARRAY_SIZE(events) && --count != 0) {
        real_timeout = 0;
        continue;
      }
      return;
    }

    if (timeout == 0)
      return;

    if (timeout == -1)
      continue;

update_timeout:
    assert(timeout > 0);

    real_timeout = timeout - (loop->time - base);
    if (real_timeout <= 0)
      return;
  }
}
```

### 3. epoll 事件类型

```c
// /usr/include/sys/epoll.h
enum EPOLL_EVENTS {
  EPOLLIN      = 0x001,  // 可读
  EPOLLPRI     = 0x002,  // 紧急数据
  EPOLLOUT     = 0x004,  // 可写
  EPOLLERR     = 0x008,  // 错误
  EPOLLHUP     = 0x010,  // 挂断
  EPOLLRDHUP   = 0x2000, // 对端关闭连接
  
  EPOLLET      = 1u << 31,  // 边缘触发（Edge Triggered）
  EPOLLONESHOT = 1u << 30   // 一次性监听
};
```

**触发模式：**
```c
// ❌ 水平触发（Level Triggered，默认）
// 只要条件满足，每次 epoll_wait 都返回

// ✅ 边缘触发（Edge Triggered）
e.events = EPOLLIN | EPOLLET;
// 只在状态变化时触发一次，需要一次性读完数据
```

```c
// deps/uv/src/unix/kqueue.c
void uv__io_poll(uv_loop_t* loop, int timeout) {
  struct kevent events[1024];
  struct kevent* ev;
  struct timespec spec;
  unsigned int nevents;
  unsigned int revents;
  QUEUE* q;
  uv__io_t* w;
  uv__io_t* dup;
  int filter;
  int fflags;
  int count;
  int nfds;
  int fd;
  int op;
  int i;

  if (loop->nfds == 0) {
    assert(QUEUE_EMPTY(&loop->watcher_queue));
    return;
  }

  nevents = 0;

  // 🔥 注册/修改事件
  while (!QUEUE_EMPTY(&loop->watcher_queue)) {
    q = QUEUE_HEAD(&loop->watcher_queue);
    QUEUE_REMOVE(q);
    QUEUE_INIT(q);

    w = QUEUE_DATA(q, uv__io_t, watcher_queue);
    assert(w->pevents != 0);
    assert(w->fd >= 0);
    assert(w->fd < (int) loop->nwatchers);

    // 🔥 可读事件
    if ((w->pevents & POLLIN) != 0) {
      filter = EVFILT_READ;
      fflags = 0;
      op = EV_ADD;

      if (w->cb == uv__fs_event)
        fflags = NOTE_EXTEND | NOTE_WRITE | NOTE_DELETE | NOTE_RENAME;

      EV_SET(events + nevents, w->fd, filter, op, fflags, 0, 0);
      
      if (++nevents == ARRAY_SIZE(events)) {
        kevent(loop->backend_fd, events, nevents, NULL, 0, NULL);
        nevents = 0;
      }
    }

    // 🔥 可写事件
    if ((w->pevents & POLLOUT) != 0) {
      filter = EVFILT_WRITE;
      op = EV_ADD;

      EV_SET(events + nevents, w->fd, filter, op, 0, 0, 0);
      
      if (++nevents == ARRAY_SIZE(events)) {
        kevent(loop->backend_fd, events, nevents, NULL, 0, NULL);
        nevents = 0;
      }
    }

    w->events = w->pevents;
  }

  // 提交修改
  if (nevents != 0) {
    kevent(loop->backend_fd, events, nevents, NULL, 0, NULL);
  }

  assert(timeout >= -1);

  if (timeout != -1) {
    spec.tv_sec = timeout / 1000;
    spec.tv_nsec = (timeout % 1000) * 1000000;
  }

  // 🔥 等待事件就绪
  nfds = kevent(loop->backend_fd,
                NULL,
                0,
                events,
                ARRAY_SIZE(events),
                timeout == -1 ? NULL : &spec);

  if (nfds == -1) {
    if (errno != EINTR)
      abort();
    return;
  }

  SAVE_ERRNO(uv__update_time(loop));

  // 🔥 处理就绪事件
  for (i = 0; i < nfds; i++) {
    ev = events + i;
    fd = ev->ident;

    if (fd == -1)
      continue;

    assert(fd >= 0);
    assert((unsigned) fd < loop->nwatchers);

    w = loop->watchers[fd];

    if (w == NULL) {
      // 删除事件
      struct kevent changelist;
      EV_SET(&changelist, fd, ev->filter, EV_DELETE, 0, 0, 0);
      kevent(loop->backend_fd, &changelist, 1, NULL, 0, NULL);
      continue;
    }

    // 🔥 设置就绪事件
    revents = 0;

    if (ev->filter == EVFILT_READ) {
      if (w->pevents & POLLIN) {
        revents |= POLLIN;
        w->rnevents |= POLLIN;
      }
    }

    if (ev->filter == EVFILT_WRITE) {
      if (w->pevents & POLLOUT) {
        revents |= POLLOUT;
        w->rnevents |= POLLOUT;
      }
    }

    if (ev->flags & EV_EOF)
      w->rnevents |= POLLHUP;

    if (ev->flags & EV_ERROR)
      w->rnevents |= POLLERR;
  }

  // 🔥 执行 I/O 回调
  for (i = 0; i < (int) loop->nwatchers; i++) {
    w = loop->watchers[i];

    if (w == NULL)
      continue;

    if (w->rnevents == 0)
      continue;

    w->cb(loop, w, w->rnevents);
  }
}
```

```c
// deps/uv/src/win/core.c
void uv__poll(uv_loop_t* loop, int timeout) {
  DWORD bytes;
  ULONG_PTR key;
  OVERLAPPED* overlapped;
  uv_req_t* req;
  int repeat;
  uint64_t timeout_time;

  timeout_time = loop->time + timeout;

  for (repeat = 0; ; repeat++) {
    // 🔥 获取完成状态
    GetQueuedCompletionStatus(loop->iocp,
                              &bytes,
                              &key,
                              &overlapped,
                              timeout);

    uv__update_time(loop);

    if (overlapped) {
      // 🔥 处理 I/O 完成
      req = uv_overlapped_to_req(overlapped);
      uv_insert_pending_req(loop, req);

    } else if (GetLastError() != WAIT_TIMEOUT) {
      // 错误处理
      abort();
    }

    // 超时检查
    if (timeout > 0 && loop->time >= timeout_time)
      break;

    // 避免无限循环
    if (repeat >= 200)
      break;
  }
}
```

## 💻 I/O 回调执行

### 1. 回调注册

```c
// deps/uv/src/unix/core.c
void uv__io_start(uv_loop_t* loop, uv__io_t* w, unsigned int events) {
  assert(0 == (events & ~(POLLIN | POLLOUT | UV__POLLRDHUP | UV__POLLPRI)));
  assert(0 != events);
  assert(w->fd >= 0);
  assert(w->fd < INT_MAX);

  w->pevents |= events;
  maybe_resize(loop, w->fd + 1);

  if (w->events == w->pevents)
    return;

  if (QUEUE_EMPTY(&w->watcher_queue))
    QUEUE_INSERT_TAIL(&loop->watcher_queue, &w->watcher_queue);

  if (loop->watchers[w->fd] == NULL) {
    loop->watchers[w->fd] = w;
    loop->nfds++;
  }
}
```

## 📝 总结

### 核心要点

1. **多路复用**：epoll（Linux）、kqueue（macOS）、IOCP（Windows）
2. **超时计算**：动态调整，避免饥饿
3. **事件处理**：注册→等待→触发→回调
4. **性能优化**：批量 I/O、连接池、减少调用

### 实现关键

1. **文件描述符管理**：watchers 数组
