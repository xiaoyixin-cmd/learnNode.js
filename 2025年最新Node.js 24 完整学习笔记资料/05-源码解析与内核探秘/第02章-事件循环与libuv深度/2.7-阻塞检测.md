# 2.7 阻塞检测原理与实现

> 深入理解事件循环阻塞检测机制（blocked-at 原理）

---

## 📋 核心概念

事件循环阻塞检测用于**发现长时间运行的同步代码**，帮助定位性能瓶颈。

**阻塞定义：**
- 单个任务执行时间 > 阈值（如 100ms）
- 阻塞事件循环，影响响应性

**检测原理：**
```
定时器检查 + 事件循环延迟测量
```

## 💻 blocked-at 库原理

### 1. 核心实现

```javascript
// blocked-at.js
const async_hooks = require('async_hooks');
const { performance } = require('perf_hooks');

class BlockedDetector {
  constructor(threshold = 100) {
    this.threshold = threshold;
    this.lastCheck = performance.now();
    this.interval = null;
    this.stack = null;
  }

  start() {
    // 🔥 使用 setInterval 检查
    this.interval = setInterval(() => {
      const now = performance.now();
      const delay = now - this.lastCheck;

      // 🔥 检测到阻塞
      if (delay > this.threshold) {
        console.warn(`⚠️  Event loop blocked for ${delay.toFixed(2)}ms`);
        
        if (this.stack) {
          console.warn('Stack trace when blocked:');
          console.warn(this.stack);
        }
      }

      this.lastCheck = now;
    }, this.threshold / 2);  // 检查间隔为阈值的一半

    // 🔥 捕获调用栈
    this.captureStack();
  }

  captureStack() {
    const hook = async_hooks.createHook({
      init: (asyncId, type, triggerAsyncId) => {
        // 捕获异步操作的调用栈
        const err = new Error();
        this.stack = err.stack;
      }
    });
    hook.enable();
  }

  stop() {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }
}

module.exports = BlockedDetector;
```

## 💻 精确阻塞检测

### 1. 基于 async_hooks 的实现

```javascript
// precise-blocked.js
const async_hooks = require('async_hooks');
const { performance } = require('perf_hooks');

class PreciseBlockedDetector {
  constructor(threshold = 50) {
    this.threshold = threshold;
    this.operations = new Map();
    this.hook = null;
  }

  start() {
    // 🔥 使用 async_hooks 追踪所有异步操作
    this.hook = async_hooks.createHook({
      init: (asyncId, type, triggerAsyncId) => {
        this.operations.set(asyncId, {
          type,
          triggerAsyncId,
          initTime: performance.now(),
          stack: this.captureStack()
        });
      },

      before: (asyncId) => {
        const op = this.operations.get(asyncId);
        if (op) {
          op.beforeTime = performance.now();
        }
      },

      after: (asyncId) => {
        const op = this.operations.get(asyncId);
        if (!op || !op.beforeTime) return;

        const duration = performance.now() - op.beforeTime;

        // 🔥 检测到阻塞
        if (duration > this.threshold) {
          console.warn(`⚠️  ${op.type} blocked for ${duration.toFixed(2)}ms`);
          console.warn('Initiated at:', op.stack);
        }
      },

      destroy: (asyncId) => {
        this.operations.delete(asyncId);
      }
    });

    this.hook.enable();
  }

  captureStack() {
    const obj = {};
    Error.captureStackTrace(obj);
    return obj.stack;
  }

  stop() {
    if (this.hook) {
      this.hook.disable();
      this.hook = null;
    }
    this.operations.clear();
  }
}

module.exports = PreciseBlockedDetector;
```

## 💻 事件循环延迟测量

### 1. 简单延迟测量

```javascript
// event-loop-lag.js
const { performance } = require('perf_hooks');

class EventLoopLag {
  constructor(interval = 100) {
    this.interval = interval;
    this.lastTime = performance.now();
    this.timer = null;
    this.samples = [];
  }

  start() {
    this.timer = setInterval(() => {
      const now = performance.now();
      const expected = this.interval;
      const actual = now - this.lastTime;
      const lag = actual - expected;

      // 🔥 记录延迟
      this.samples.push(lag);

      if (lag > 10) {  // 延迟超过 10ms
        console.warn(`Event loop lag: ${lag.toFixed(2)}ms`);
      }

      this.lastTime = now;

      // 保留最近 100 个样本
      if (this.samples.length > 100) {
        this.samples.shift();
      }
    }, this.interval);

    this.lastTime = performance.now();
  }

  getStats() {
    if (this.samples.length === 0) {
      return { min: 0, max: 0, avg: 0, p95: 0, p99: 0 };
    }

    const sorted = [...this.samples].sort((a, b) => a - b);
    const sum = sorted.reduce((acc, val) => acc + val, 0);

    return {
      min: sorted[0],
      max: sorted[sorted.length - 1],
      avg: sum / sorted.length,
      p95: sorted[Math.floor(sorted.length * 0.95)],
      p99: sorted[Math.floor(sorted.length * 0.99)]
    };
  }

  stop() {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
    }
  }
}

// 使用
const lag = new EventLoopLag(100);
lag.start();

// 模拟负载
setInterval(() => {
  const start = Date.now();
  while (Date.now() - start < Math.random() * 50) {}
}, 200);

// 定期打印统计
setInterval(() => {
  const stats = lag.getStats();
  console.log('Event Loop Lag Stats:', {
    min: stats.min.toFixed(2) + 'ms',
    max: stats.max.toFixed(2) + 'ms',
    avg: stats.avg.toFixed(2) + 'ms',
    p95: stats.p95.toFixed(2) + 'ms',
    p99: stats.p99.toFixed(2) + 'ms'
  });
}, 5000);
```

```javascript
// apm-monitoring.js
const { performance, monitorEventLoopDelay } = require('perf_hooks');

class APMMonitor {
  constructor() {
    this.histogram = monitorEventLoopDelay({ resolution: 10 });
    this.metrics = {
      eventLoopLag: [],
      cpu: [],
      memory: []
    };
  }

  start() {
    this.histogram.enable();

    // 🔥 每秒收集指标
    this.interval = setInterval(() => {
      this.collect();
    }, 1000);

    // 🔥 每分钟发送指标
    this.reportInterval = setInterval(() => {
      this.report();
    }, 60000);
  }

  collect() {
    // 事件循环延迟
    this.metrics.eventLoopLag.push({
      timestamp: Date.now(),
      value: this.histogram.mean / 1e6  // 转换为毫秒
    });

    // CPU 使用率
    const usage = process.cpuUsage();
    this.metrics.cpu.push({
      timestamp: Date.now(),
      user: usage.user / 1000,
      system: usage.system / 1000
    });

    // 内存使用
    const mem = process.memoryUsage();
    this.metrics.memory.push({
      timestamp: Date.now(),
      heapUsed: mem.heapUsed,
      heapTotal: mem.heapTotal,
      external: mem.external
    });

    this.histogram.reset();

    // 保留最近 60 个样本
    if (this.metrics.eventLoopLag.length > 60) {
      this.metrics.eventLoopLag.shift();
      this.metrics.cpu.shift();
      this.metrics.memory.shift();
    }
  }

  report() {
    // 计算统计
    const lagStats = this.calculateStats(
      this.metrics.eventLoopLag.map(m => m.value)
    );

    console.log('📊 APM Report:');
    console.log('Event Loop Lag:', {
      min: lagStats.min.toFixed(2) + 'ms',
      max: lagStats.max.toFixed(2) + 'ms',
      avg: lagStats.avg.toFixed(2) + 'ms',
      p95: lagStats.p95.toFixed(2) + 'ms'
    });

    // 这里可以发送到 APM 服务（如 Elastic APM, Datadog 等）
    // this.sendToAPM(this.metrics);
  }

  calculateStats(values) {
    if (values.length === 0) return { min: 0, max: 0, avg: 0, p95: 0 };

    const sorted = [...values].sort((a, b) => a - b);
    const sum = sorted.reduce((acc, val) => acc + val, 0);

      min: sorted[0],

      avg: sum / sorted.length,
      p95: sorted[Math.floor(sorted.length * 0.95)]
    };
  }

  stop() {
    if (this.interval) clearInterval(this.interval);
    if (this.reportInterval) clearInterval(this.reportInterval);
    this.histogram.disable();
  }
}
```

```javascript
// chunked-processing.js
async function processLargeArray(arr, chunkSize = 1000) {
  const chunks = Math.ceil(arr.length / chunkSize);
  
  for (let i = 0; i < chunks; i++) {
    const start = i * chunkSize;
    const end = Math.min(start + chunkSize, arr.length);
    
    // 处理分片
    for (let j = start; j < end; j++) {
      // 处理 arr[j]
      arr[j] = arr[j] * 2;

    // 🔥 让出控制权
    await new Promise(resolve => setImmediate(resolve));
  }
}
```

### 2. Worker Threads

```javascript
// worker-offload.js
const { Worker } = require('worker_threads');

function offloadWork(data) {
  return new Promise((resolve, reject) => {
    // 🔥 在 Worker 线程中执行
    const worker = new Worker(`
      const { parentPort, workerData } = require('worker_threads');
      
      // 执行 CPU 密集型任务
      function fibonacci(n) {
        if (n <= 1) return n;
        return fibonacci(n - 1) + fibonacci(n - 2);
      }
      
      const result = fibonacci(workerData);
      parentPort.postMessage(result);
    `, {
      eval: true,
      workerData: data
    });

    worker.on('message', resolve);
    worker.on('error', reject);
    worker.on('exit', (code) => {
      if (code !== 0) {
        reject(new Error(`Worker stopped with exit code ${code}`));
      }
    });
  });
}

// 使用
const detector = require('./blocked-at');
detector.start();

offloadWork(40).then(result => {
  console.log('Fibonacci result:', result);
  detector.stop();
});

// ✅ 主线程不阻塞
```

## 📝 总结

### 核心要点

1. **阻塞定义**：同步代码执行时间超过阈值
2. **检测方法**：定时器 + 延迟测量 + async_hooks
3. **性能指标**：事件循环延迟、CPU、内存

1. **定时器检查**：setInterval 周期性检测

3. **perf_hooks**：内置事件循环延迟监控
4. **Inspector API**：CPU 分析和性能剖析

### 应用价值

1. **性能监控**：实时检测阻塞问题
2. **问题定位**：捕获阻塞时的调用栈
