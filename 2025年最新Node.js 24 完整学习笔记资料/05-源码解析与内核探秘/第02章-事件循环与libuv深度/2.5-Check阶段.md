# 2.5 Check 阶段：setImmediate 的执行时机

> 深入理解 setImmediate 与 setTimeout(0) 的本质区别

---

## 📋 核心概念

Check 阶段是事件循环中**专门执行 setImmediate 回调**的阶段。

**执行时机：**
- 在 Poll 阶段之后
- 在 Close 阶段之前
- 每次事件循环只执行一次

**关键特性：**
```
setImmediate → Check 阶段
setTimeout(0) → Timers 阶段
process.nextTick → 微任务队列（优先级最高）
```

## 💻 setImmediate 源码实现

### 1. JavaScript 层实现

```javascript
// lib/timers.js
function setImmediate(callback, arg1, arg2, arg3) {
  validateFunction(callback, 'callback');

  let i, args;
  
  // 处理额外参数
  switch (arguments.length) {
    case 1:
      break;
    case 2:
      args = [arg1];
      break;
    case 3:
      args = [arg1, arg2];
      break;
    default:
      args = new Array(arguments.length - 1);
      for (i = 1; i < arguments.length; i++)
        args[i - 1] = arguments[i];
  }

  // 🔥 创建 Immediate 对象
  return new Immediate(callback, args);
}

// Immediate 类定义
class Immediate {
  constructor(callback, args) {
    this._idleNext = null;
    this._idlePrev = null;
    this._onImmediate = callback;
    this._argv = args;
    this._destroyed = false;
    this._requiresTicketWorkaround = false;

    // 🔥 加入队列
    this[kRefed] = true;

    initAsyncResource(this, 'Immediate');

    // 插入到 immediateQueue
    immediateQueue.append(this);
  }
}
```

```c
// deps/uv/src/unix/core.c
void uv__run_check(uv_loop_t* loop) {
  uv_check_t* h;
  QUEUE queue;
  QUEUE* q;

  // 🔥 获取 check 队列
  QUEUE_MOVE(&loop->check_handles, &queue);
  
  // 🔥 遍历执行
  while (!QUEUE_EMPTY(&queue)) {
    q = QUEUE_HEAD(&queue);
    h = QUEUE_DATA(q, uv_check_t, queue);

    QUEUE_REMOVE(q);
    QUEUE_INSERT_TAIL(&loop->check_handles, q);

    // 🔥 执行回调
    h->check_cb(h);
  }
}
```

## 💻 setImmediate vs setTimeout(0)

### 1. 执行时机对比

```javascript
// immediate-vs-timeout.js

// 情况1：主模块中（不在 I/O 回调中）
setTimeout(() => {
  console.log('timeout');
}, 0);

setImmediate(() => {
  console.log('immediate');
});

// 输出顺序不确定！
// 可能是：timeout → immediate
// 也可能是：immediate → timeout
```

**原因分析：**
```
主模块启动时：
1. loop->time = uv_now()  // 假设为 100ms
2. setTimeout(0) → 插入堆，timeout = 100 + 1 = 101ms
3. setImmediate() → 插入 check 队列

第一轮事件循环：
- Timers 阶段：loop->time = 100ms
  - 如果系统调度快，仍为 100ms，定时器未到期 ✗
  - 如果系统调度慢，已是 101ms，定时器到期 ✓
- Poll 阶段：无 I/O
- Check 阶段：执行 setImmediate ✓

所以顺序不确定！
```

---

### 2. I/O 回调中的执行顺序

```javascript
// io-callback-order.js
const fs = require('fs');

fs.readFile(__filename, () => {
  // 🔥 在 I/O 回调中（Poll 阶段）
  
  setTimeout(() => {
    console.log('timeout');
  }, 0);

  setImmediate(() => {
    console.log('immediate');
  });
});

// 输出顺序固定：
// immediate
// timeout
```

**原因分析：**
```
I/O 回调执行时（Poll 阶段）：
1. setTimeout(0) → 插入定时器堆（下一轮 Timers 阶段）
2. setImmediate() → 插入 check 队列（本轮 Check 阶段）

事件循环继续：
- Poll 阶段 → Check 阶段（执行 setImmediate）✓
- 本轮循环结束
- 下一轮 Timers 阶段（执行 setTimeout）✓

所以 immediate 一定先执行！
```

## 💻 执行优先级

### 1. 完整优先级链

```
1. process.nextTick  (最高优先级)
   ↓
2. Promise.then/catch/finally
   ↓
3. queueMicrotask
   ↓
─────── 事件循环开始 ───────
   ↓
4. Timers 阶段
   - setTimeout
   - setInterval
   ↓
5. Pending 阶段
   - I/O 错误回调
   ↓
6. Idle/Prepare 阶段
   - (内部使用)
   ↓
7. Poll 阶段
   - I/O 回调
   - fs.readFile
   - http.request
   ↓
8. Check 阶段
   - setImmediate  ← 我们在这里
   ↓
9. Close 阶段
   - socket.on('close')
```

### 2. Check 启动

```c
// deps/uv/src/unix/loop-watcher.c
int uv_check_start(uv_check_t* handle, uv_check_cb cb) {
  if (uv__is_active(handle)) return 0;

  // 🔥 设置回调
  handle->check_cb = cb;

  // 🔥 插入 check 队列
  QUEUE_INSERT_HEAD(&handle->loop->check_handles, &handle->queue);

  // 🔥 激活句柄
  uv__handle_start(handle);

  return 0;
}
```

---

### 3. Check 执行

```c
// deps/uv/src/unix/core.c
void uv__run_check(uv_loop_t* loop) {
  uv_check_t* h;
  QUEUE queue;
  QUEUE* q;

  // 🔥 移动队列（避免在迭代中修改）
  QUEUE_MOVE(&loop->check_handles, &queue);
  
  // 🔥 遍历执行
  while (!QUEUE_EMPTY(&queue)) {
    q = QUEUE_HEAD(&queue);
    h = QUEUE_DATA(q, uv_check_t, queue);

    QUEUE_REMOVE(q);
    
    // 🔥 重新插入（下次继续执行）
    QUEUE_INSERT_TAIL(&loop->check_handles, q);

    // 🔥 执行回调
    h->check_cb(h);
  }
}
```

**执行特点：**
- 每轮循环执行一次
- 在队列中的会重新加入队列
- 新加入的会在下一轮执行

## 💻 应用场景

### 1. I/O 后的任务

```javascript
// io-then-task.js
const fs = require('fs');

fs.readFile('/data.json', (err, data) => {
  if (err) throw err;
  
  const parsed = JSON.parse(data);
  
  // 🔥 I/O 完成后，下一阶段执行
  setImmediate(() => {
    processData(parsed);
  });
});

function processData(data) {
  // 处理数据，不阻塞当前 I/O 回调
  console.log('Processing:', data);
}
```

### 2. 分片处理大任务

```javascript
// chunked-processing.js
function processLargeArray(arr, chunkSize = 1000) {
  let index = 0;
  
  function processChunk() {
    const end = Math.min(index + chunkSize, arr.length);
    
    // 处理当前分片
    for (let i = index; i < end; i++) {
      // 处理 arr[i]
      arr[i] = arr[i] * 2;
    }
    
    index = end;
    
    // 🔥 还有剩余，下一阶段继续
    if (index < arr.length) {
      setImmediate(processChunk);
    } else {
      console.log('Processing complete');
    }
  }
  
  processChunk();
}

// 使用
const largeArray = new Array(1000000).fill(1);
processLargeArray(largeArray);
```

## 💻 性能优化

### 1. 避免过度使用

```javascript
// ❌ 过度使用 setImmediate
for (let i = 0; i < 10000; i++) {
  setImmediate(() => {
    console.log(i);
  });
}
// 创建了 10000 个 immediate，性能差

// ✅ 批量处理

  for (let i = start; i < end; i++) {
    console.log(i);
  }
}

const batchSize = 100;
for (let i = 0; i < 10000; i += batchSize) {
  setImmediate(() => logBatch(i, i + batchSize));
}
// 只创建 100 个 immediate
```

```javascript
  console.log('immediate');
```

### 2. 递归爆栈

```javascript
// trap-2.js

// ❌ 会堆栈溢出
function recurse(n) {
  if (n <= 0) return;
  recurse(n - 1);
}
recurse(100000);  // RangeError: Maximum call stack size exceeded

// ✅ 使用 setImmediate
function recurseImmediate(n) {
  if (n <= 0) return;
  
  setImmediate(() => {
    recurseImmediate(n - 1);
  });
}
recurseImmediate(100000);  // ✅ 正常执行
```

### 3. 内存泄漏

```javascript
// trap-3.js
const tasks = [];

// ❌ 无限累积
function addTask() {
  tasks.push({ data: new Array(1000000) });
  setImmediate(addTask);
}

addTask();  // 内存会一直增长

// ✅ 控制队列大小
const MAX_TASKS = 100;

function addTaskSafe() {
  if (tasks.length < MAX_TASKS) {
    tasks.push({ data: new Array(1000000) });
  } else {
    tasks.shift();  // 移除最老的
    tasks.push({ data: new Array(1000000) });
  }
  
  setImmediate(addTaskSafe);
}
```

## 📝 总结

### 核心要点

1. **执行时机**：Poll 阶段后，Close 阶段前
2. **与 setTimeout(0) 区别**：I/O 回调中固定顺序
3. **应用场景**：I/O 后任务、分片处理、递归
4. **性能考虑**：批量处理、避免过度使用

### 实现关键

1. **队列管理**：check_handles 链表
2. **回调执行**：遍历队列执行 check_cb
3. **优先级**：低于微任务，高于下一轮 timers
4. **unref 支持**：不阻止进程退出

### 应用价值

1. **理解事件循环**：掌握各阶段执行顺序
2. **性能优化**：选择合适的调度方式
3. **避免陷阱**：正确理解执行时机
4. **最佳实践**：I/O 后使用 setImmediate
