## 📋 核心概念

`uv_run()` 是 libuv 事件循环的**核心入口函数**，所有异步操作都围绕它展开。

**主要职责：**
- 管理事件循环的生命周期
- 协调各个阶段的执行
- 处理异步任务和 I/O 事件
- 控制循环的退出条件

**执行模式：**
```c
UV_RUN_DEFAULT  // 持续运行直到没有活跃句柄
UV_RUN_ONCE     // 执行一次循环
UV_RUN_NOWAIT   // 非阻塞执行一次
```

## 💻 uv_run() 完整源码

### 1. 函数签名

```c
// deps/uv/src/unix/core.c
int uv_run(uv_loop_t* loop, uv_run_mode mode) {
  int timeout;
  int r;
  int can_sleep;

  // 🔥 检查循环是否存活
  r = uv__loop_alive(loop);
  if (!r)
    uv__update_time(loop);

  // 🔥 主循环
  while (r != 0 && loop->stop_flag == 0) {
    // 更新当前时间
    uv__update_time(loop);
    
    // 🔥 阶段 1: Timers
    uv__run_timers(loop);
    
    // 🔥 阶段 2: Pending callbacks
    ran_pending = uv__run_pending(loop);
    
    // 🔥 阶段 3: Idle handles
    uv__run_idle(loop);
    
    // 🔥 阶段 4: Prepare handles
    uv__run_prepare(loop);

    timeout = 0;
    
    // 🔥 计算 poll 超时时间
    if ((mode == UV_RUN_ONCE && !ran_pending) || mode == UV_RUN_DEFAULT)
      timeout = uv__backend_timeout(loop);

    // 🔥 阶段 5: Poll (I/O)
    uv__io_poll(loop, timeout);
    
    // 🔥 Metrics 更新（可选）
    uv__metrics_update_idle_time(loop);

    // 🔥 阶段 6: Check handles
    uv__run_check(loop);
    
    // 🔥 阶段 7: Close callbacks
    uv__run_closing_handles(loop);

    // 🔥 UV_RUN_ONCE 模式特殊处理
    if (mode == UV_RUN_ONCE) {
      uv__update_time(loop);
      uv__run_timers(loop);
    }

    // 🔥 更新循环存活状态
    r = uv__loop_alive(loop);
    
    // 🔥 退出条件
    if (mode == UV_RUN_ONCE || mode == UV_RUN_NOWAIT)
      break;
  }

  // 🔥 返回循环是否仍然存活
  if (loop->stop_flag != 0)
    loop->stop_flag = 0;

  return r;
}
```

## 💻 循环存活检查

### 1. uv__loop_alive() 实现

```c
// deps/uv/src/unix/core.c
static int uv__loop_alive(const uv_loop_t* loop) {
  // 🔥 有活跃的句柄
  return uv__has_active_handles(loop) ||
         // 🔥 或有活跃的请求
         uv__has_active_reqs(loop) ||
         // 🔥 或有 closing handles
         loop->closing_handles != NULL;
}
```

**存活条件：**
```
1. 有活跃句柄（active handles）
   - Timer、TCP、UDP、Pipe 等
   
2. 有活跃请求（active reqs）
   - 文件 I/O、DNS 查询等
   
3. 有待关闭句柄（closing handles）
   - 正在关闭但未完成的句柄
```

### 2. 活跃句柄检查

```c
// deps/uv/include/uv.h
#define uv__has_active_handles(loop)                                        \
  ((loop)->active_handles > 0)

#define uv__has_active_reqs(loop)                                           \
  ((loop)->active_reqs.count > 0)
```

**引用计数机制：**
```c
// 句柄激活
void uv__handle_start(uv_handle_t* handle) {
  if (uv__is_active(handle)) return;
  
  // 🔥 增加活跃句柄计数
  loop->active_handles++;
  
  if (uv__is_ref(handle))
    uv__active_handle_add(handle);
}

// 句柄停止
void uv__handle_stop(uv_handle_t* handle) {
  if (!uv__is_active(handle)) return;
  
  // 🔥 减少活跃句柄计数
  loop->active_handles--;
  
  if (uv__is_ref(handle))
    uv__active_handle_rm(handle);
}
```

```c
// deps/uv/src/unix/linux-core.c (Linux)
uint64_t uv__hrtime(uv_clocktype_t type) {
  struct timespec ts;
  
  // 🔥 使用 clock_gettime
  clock_gettime(CLOCK_MONOTONIC, &ts);
  
  // 🔥 转换为纳秒
  return ts.tv_sec * 1e9 + ts.tv_nsec;
}

// deps/uv/src/unix/darwin-proctitle.c (macOS)
uint64_t uv__hrtime(uv_clocktype_t type) {
  mach_timebase_info_data_t info;
  
  // 🔥 使用 mach_absolute_time
  if (mach_timebase_info(&info) != KERN_SUCCESS)
    abort();
  
  return mach_absolute_time() * info.numer / info.denom;
}
```

**时间缓存优化：**
```
循环开始时更新一次 loop->time
各阶段使用缓存的时间，避免频繁系统调用
```

## 💻 Timers 阶段

### 1. uv__run_timers() 实现

```c
// deps/uv/src/timer.c
void uv__run_timers(uv_loop_t* loop) {
  struct heap_node* heap_node;
  uv_timer_t* handle;

  for (;;) {
    // 🔥 获取最小堆堆顶（最早到期的定时器）
    heap_node = heap_min(timer_heap(loop));
    if (heap_node == NULL)
      break;

    // 🔥 转换为 timer 句柄
    handle = container_of(heap_node, uv_timer_t, heap_node);

    // 🔥 检查是否到期
    if (handle->timeout > loop->time)
      break;

    // 🔥 停止定时器（从堆移除）
    uv_timer_stop(handle);
    
    // 🔥 如果是 setInterval，重新启动
    uv_timer_again(handle);

    // 🔥 执行回调
    handle->timer_cb(handle);
  }
}
```

**执行流程：**
```
1. 获取堆顶定时器
2. 检查是否到期（timeout <= loop->time）
3. 从堆移除
4. setInterval 的重新插入堆
5. 执行回调
6. 重复直到没有到期定时器
```

## 💻 Pending 阶段

### 1. uv__run_pending() 实现

```c
// deps/uv/src/unix/core.c
static int uv__run_pending(uv_loop_t* loop) {
  QUEUE* q;
  QUEUE pq;
  uv__io_t* w;

  // 🔥 移动 pending_queue 到临时队列
  if (QUEUE_EMPTY(&loop->pending_queue))
    return 0;

  QUEUE_MOVE(&loop->pending_queue, &pq);

  // 🔥 遍历执行
  while (!QUEUE_EMPTY(&pq)) {
    q = QUEUE_HEAD(&pq);
    QUEUE_REMOVE(q);
    QUEUE_INIT(q);

    w = QUEUE_DATA(q, uv__io_t, pending_queue);
    
    // 🔥 执行回调
    w->cb(loop, w, POLLOUT);
  }

  return 1;
}
```

**应用场景：**
```
处理上一轮循环中延迟的 I/O 回调
例如：TCP 连接错误、写入错误等
```

## 💻 Poll 阶段

### 1. 超时计算

```c
// deps/uv/src/unix/core.c
int uv__backend_timeout(const uv_loop_t* loop) {
  // 1. 有 closing handles，立即返回
  if (loop->closing_handles)
    return 0;

  // 2. 循环停止标志，立即返回
  if (loop->stop_flag)
    return 0;

  // 3. 没有活跃句柄/请求，立即返回
  if (!uv__has_active_handles(loop) && !uv__has_active_reqs(loop))
    return 0;

  // 4. 有 idle handles，立即返回
  if (!QUEUE_EMPTY(&loop->idle_handles))
    return 0;

  // 5. 有 pending 回调，立即返回
  if (!QUEUE_EMPTY(&loop->pending_queue))
    return 0;

  // 6. 计算到下一个定时器的时间
  return uv__next_timeout(loop);
}
```

```c
// deps/uv/src/unix/loop-watcher.c
void uv__run_check(uv_loop_t* loop) {
  uv_check_t* h;
  QUEUE queue;
  QUEUE* q;

  QUEUE_MOVE(&loop->check_handles, &queue);

  while (!QUEUE_EMPTY(&queue)) {
    q = QUEUE_HEAD(&queue);
    h = QUEUE_DATA(q, uv_check_t, queue);

    QUEUE_REMOVE(q);
    QUEUE_INSERT_TAIL(&loop->check_handles, q);

    h->check_cb(h);
  }
}
```

## 💻 Close 阶段

### 1. uv__run_closing_handles() 实现

```c
// deps/uv/src/unix/core.c
static void uv__run_closing_handles(uv_loop_t* loop) {
  uv_handle_t* p;
  uv_handle_t* q;

  p = loop->closing_handles;
  loop->closing_handles = NULL;

  // 🔥 遍历 closing_handles 链表
  while (p) {
    q = p->next_closing;
    
    // 🔥 执行 close 回调
    uv__finish_close(p);
    
    p = q;
  }
}

static void uv__finish_close(uv_handle_t* handle) {
  assert(handle->flags & UV_HANDLE_CLOSING);
  assert(!(handle->flags & UV_HANDLE_CLOSED));
  
  // 🔥 设置关闭标志
  handle->flags |= UV_HANDLE_CLOSED;

  // 🔥 根据句柄类型执行清理
  switch (handle->type) {
    case UV_NAMED_PIPE:
    case UV_TCP:
    case UV_TTY:
      uv__stream_destroy((uv_stream_t*)handle);
      break;
      
    case UV_UDP:
      uv__udp_finish_close((uv_udp_t*)handle);
      break;
      
    // ... 其他类型
  }

  // 🔥 减少活跃句柄计数
  uv__handle_unref(handle);
  QUEUE_REMOVE(&handle->handle_queue);

  // 🔥 执行用户 close 回调
  if (handle->close_cb) {
    handle->close_cb(handle);
  }
}
```

### 2. UV_RUN_ONCE

```c
// 执行一次模式
uv_run(loop, UV_RUN_ONCE);

// 执行流程：
{
  // 执行所有阶段
  // 如果没有 pending，计算 poll 超时
  // 等待 I/O
  
  // 特殊：再执行一次 timers
  if (mode == UV_RUN_ONCE) {
    uv__update_time(loop);
    uv__run_timers(loop);
  }
  
  break;  // 退出
}

// 应用：集成到其他事件循环
while (external_loop_running) {
  uv_run(loop, UV_RUN_ONCE);

}
```

## 📝 总结

### 核心要点

1. **主循环**：`uv_run()` 是事件循环的核心
2. **7 个阶段**：Timers → Pending → Idle → Prepare → Poll → Check → Close
3. **存活检查**：活跃句柄 + 活跃请求 + 待关闭句柄
4. **运行模式**：DEFAULT（持续）、ONCE（一次）、NOWAIT（非阻塞）

### 实现关键

1. **时间缓存**：避免频繁系统调用
2. **队列管理**：移动队列避免迭代冲突
3. **最小堆**：O(1) 获取最早定时器
4. **引用计数**：管理循环生命周期

### 应用价值

1. **理解异步**：深入事件循环机制
2. **性能调优**：优化循环执行效率
3. **问题排查**：定位事件循环问题
4. **架构设计**：合理使用异步模式
