# 2.3 定时器阶段：最小堆（min-heap）实现原理

> 深入理解定时器的数据结构与算法实现

---

## 📋 核心概念

定时器是事件循环的第一个阶段，使用**最小堆**数据结构实现高效的定时器管理。

**设计目标：**
- 快速查找最早到期的定时器（O(1)）
- 快速插入新定时器（O(log n)）
- 快速删除定时器（O(log n)）
- 内存效率高

**核心数据结构：**
```
最小堆 (Min Heap)
    - 完全二叉树
    - 父节点 ≤ 子节点
    - 堆顶为最小值
```

```
- 父节点：(i - 1) / 2
- 左子节点：2 * i + 1
- 右子节点：2 * i + 2

         /   \
        1     2
       / \   / \
      3   4 5   6

数组：[0, 1, 2, 3, 4, 5, 6]
```

## 💻 堆操作算法

### 1. 插入操作（向上调整）

```c
// deps/uv/src/heap-inl.h
static void heap_insert(struct heap* heap,
                        struct heap_node* newnode,
                        heap_compare_fn less_than) {
  unsigned int index;
  
  // 插入到末尾
  index = heap->nelts++;
  
  // 向上调整（上浮）
  while (index != 0) {
    unsigned int parent = (index - 1) / 2;
    
    // 满足堆性质，停止
    if (less_than(heap->base[parent], newnode))
      break;
    
    // 父节点下移
    heap->base[index] = heap->base[parent];
    heap->base[index]->heap_index = index;
    
    // 继续向上
    index = parent;
  }
  
  // 插入新节点
  heap->base[index] = newnode;
  newnode->heap_index = index;
}
```

**时间复杂度：O(log n)**

**可视化：**
```
插入 1 到堆 [5, 7, 6, 9, 8]

步骤1：插入到末尾
[5, 7, 6, 9, 8, 1]
               ↑

步骤2：向上调整（1 < 6）
[5, 7, 1, 9, 8, 6]
         ↑

步骤3：继续向上（1 < 5）
[1, 7, 5, 9, 8, 6]
 ↑

结果：[1, 7, 5, 9, 8, 6]
```

### 2. 删除最小值（向下调整）

```c
// deps/uv/src/heap-inl.h
static struct heap_node* heap_min(const struct heap* heap) {
  if (heap->nelts == 0)
    return NULL;
  
  // 堆顶即为最小值
  return heap->base[0];
}

static void heap_remove(struct heap* heap,
                        struct heap_node* node,
                        heap_compare_fn less_than) {
  unsigned int index = node->heap_index;
  
  // 如果是最后一个元素
  if (index == heap->nelts - 1) {
    heap->nelts--;
    return;
  }
  
  // 用最后一个元素替换
  heap->base[index] = heap->base[--heap->nelts];
  heap->base[index]->heap_index = index;
  
  // 向下调整（下沉）
  heap_siftdown(heap, index, less_than);
  
  // 可能需要向上调整
  heap_siftup(heap, index, less_than);
}
```

---

### 3. 向下调整（下沉）

```c
// deps/uv/src/heap-inl.h
static void heap_siftdown(struct heap* heap,
                          unsigned int index,
                          heap_compare_fn less_than) {
  struct heap_node* node = heap->base[index];
  
  for (;;) {
    unsigned int min_index = index;
    unsigned int child_index;
    
    // 左子节点
    child_index = 2 * index + 1;
    if (child_index < heap->nelts &&
        less_than(heap->base[child_index], heap->base[min_index])) {
      min_index = child_index;
    }
    
    // 右子节点
    child_index = 2 * index + 2;
    if (child_index < heap->nelts &&
        less_than(heap->base[child_index], heap->base[min_index])) {
      min_index = child_index;
    }
    
    // 已满足堆性质
    if (min_index == index)
      break;
    
    // 交换
    heap->base[index] = heap->base[min_index];
    heap->base[index]->heap_index = index;
    
    // 继续向下
    index = min_index;
  }
  
  heap->base[index] = node;
  node->heap_index = index;
}
```

**可视化：**
```
删除最小值 1 从堆 [1, 7, 5, 9, 8, 6]

步骤1：用最后元素替换
[6, 7, 5, 9, 8]
 ↑

步骤2：向下调整（6 > 5）
[5, 7, 6, 9, 8]
       ↑

结果：[5, 7, 6, 9, 8]
```

### 4. 定时器比较函数

```c
// deps/uv/src/timer.c
static int timer_less_than(const struct heap_node* ha,
                           const struct heap_node* hb) {
  const uv_timer_t* a;
  const uv_timer_t* b;

  a = container_of(ha, uv_timer_t, heap_node);
  b = container_of(hb, uv_timer_t, heap_node);

  // 比较超时时间
  if (a->timeout < b->timeout)
    return 1;
  
  if (b->timeout < a->timeout)
    return 0;

  // 超时时间相同，比较启动ID（FIFO）
  return a->start_id < b->start_id;
}
```

**排序规则：**
1. 超时时间小的优先
2. 时间相同，先启动的优先（FIFO）

## 💻 定时器API实现

### 1. setTimeout 实现

```c
// deps/uv/src/timer.c
int uv_timer_start(uv_timer_t* handle,
                   uv_timer_cb cb,
                   uint64_t timeout,
                   uint64_t repeat) {
  uint64_t clamped_timeout;

  // 如果已激活，先停止
  if (uv__is_active(handle))
    uv_timer_stop(handle);

  // 计算绝对超时时间
  clamped_timeout = handle->loop->time + timeout;
  
  if (clamped_timeout < timeout)
    clamped_timeout = (uint64_t) -1;

  // 设置定时器属性
  handle->timer_cb = cb;
  handle->timeout = clamped_timeout;
  handle->repeat = repeat;
  handle->start_id = handle->loop->timer_counter++;

  // 插入堆
  heap_insert(timer_heap(handle->loop),
              (struct heap_node*) &handle->heap_node,
              timer_less_than);
  
  // 激活句柄
  uv__handle_start(handle);

  return 0;
}
```

**JavaScript 调用：**
```javascript
// lib/timers.js
function setTimeout(callback, after, ...args) {

  const timer = new Timeout(callback, after, args, false);
```

### 2. setInterval 实现

```c
// deps/uv/src/timer.c
int uv_timer_again(uv_timer_t* handle) {
  if (handle->timer_cb == NULL)
    return UV_EINVAL;

  // 如果有 repeat，重新启动
  if (handle->repeat) {
    uv_timer_stop(handle);
    uv_timer_start(handle,
                   handle->timer_cb,
                   handle->repeat,
                   handle->repeat);
  }

  return 0;
}
```

**执行流程：**
```c
void uv__run_timers(uv_loop_t* loop) {
  for (;;) {
    struct heap_node* heap_node = heap_min(timer_heap(loop));
    
    if (heap_node == NULL)
      break;

    uv_timer_t* handle = container_of(heap_node, uv_timer_t, heap_node);
    
    // 未到期
    if (handle->timeout > loop->time)
      break;

    // 停止定时器（从堆移除）
    uv_timer_stop(handle);
    
    // 如果是 setInterval，重新插入
    uv_timer_again(handle);

    // 执行回调
    handle->timer_cb(handle);
  }
}
```

## 💻 定时器精度问题

### 1. 精度限制

```javascript
// timer-precision.js
const tests = [];

function measure(delay) {
  const start = Date.now();
  
  setTimeout(() => {
    const actual = Date.now() - start;
    const error = actual - delay;
    
    tests.push({ delay, actual, error });
    console.log(`Delay: ${delay}ms, Actual: ${actual}ms, Error: ${error}ms`);
  }, delay);
}

// 测试不同延迟
measure(0);    // 最小1ms
measure(1);
measure(10);
measure(100);
measure(1000);

setTimeout(() => {
  const avgError = tests.reduce((sum, t) => sum + Math.abs(t.error), 0) / tests.length;
  console.log(`\nAverage error: ${avgError.toFixed(2)}ms`);
}, 2000);

// 输出示例：
// Delay: 0ms, Actual: 1ms, Error: 1ms
// Delay: 1ms, Actual: 2ms, Error: 1ms
// Delay: 10ms, Actual: 11ms, Error: 1ms
// Delay: 100ms, Actual: 101ms, Error: 1ms
// Delay: 1000ms, Actual: 1001ms, Error: 1ms
//
// Average error: 1.00ms
```

**原因：**
1. 系统调度延迟
2. 事件循环其他阶段耗时
3. 时间缓存机制

## 💻 性能优化

### 1. 批量定时器

```javascript
// ❌ 性能差：创建大量定时器
for (let i = 0; i < 10000; i++) {
  setTimeout(() => {
    console.log(i);
  }, i * 10);

// ✅ 优化：使用队列
const tasks = Array.from({ length: 10000 }, (_, i) => i);

function processBatch() {
  const batch = tasks.splice(0, 100);
  
  batch.forEach(i => {
    console.log(i);
  });
  
  if (tasks.length > 0) {
    setTimeout(processBatch, 0);
  }
}
```

## 💻 高级特性

### 1. 未引用定时器

```c
// deps/uv/src/uv-common.c
void uv_unref(uv_handle_t* handle) {
  uv__handle_unref(handle);
}

void uv__handle_unref(uv_handle_t* handle) {
  if (uv__is_active(handle) && uv__is_ref(handle)) {
    uv__active_handle_rm(handle);
  }
  
  uv__handle_ref(handle);
}
```

**使用场景：**
```javascript

const timer = setTimeout(() => {

```
