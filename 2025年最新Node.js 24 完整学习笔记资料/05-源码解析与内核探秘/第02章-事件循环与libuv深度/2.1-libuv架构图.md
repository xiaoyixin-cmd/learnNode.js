# 2.1 libuv 2025 架构图：线程池、I/O 多路复用、定时器堆

> 深入理解 libuv 架构设计，掌握异步 I/O 的核心原理

## 📋 核心概念

libuv 是一个跨平台的异步 I/O 库，是 Node.js 的核心依赖，负责处理所有异步操作。

**设计目标：**
- 跨平台抽象（Windows/Linux/macOS）
- 高性能事件循环
- 异步文件 I/O
- 异步网络 I/O
- 线程池支持
- 定时器管理

**核心组件：**
1. **事件循环** - 主循环调度
2. **I/O 多路复用** - epoll/kqueue/IOCP
3. **线程池** - 文件 I/O 和 DNS
4. **定时器堆** - 最小堆实现
5. **异步句柄** - 统一接口

### 1. 事件循环（Event Loop）

```c
// deps/uv/include/uv.h
typedef struct uv_loop_s {
  /* User data - use this for whatever. */
  void* data;
  
  /* Loop reference counting. */
  unsigned int active_handles;
  void* handle_queue[2];
  void* active_reqs_queue[2];
  
  /* Internal storage for closing handles. */
  void* closing_handles;
  
  /* 各阶段队列 */
  void* timer_heap;          // 定时器最小堆
  uv__io_t watcher_queue[2]; // I/O 观察者队列
  void* pending_queue;       // 待处理队列
  void* idle_handles;        // idle 句柄
  void* prepare_handles;     // prepare 句柄
  void* check_handles;       // check 句柄
  
  /* Platform-specific I/O polling. */
  uv__io_t watcher_queue[2];
  unsigned int nwatchers;
  unsigned int nfds;
  
  /* Platform-specific event loop. */
#if defined(__linux__)
  int epoll_fd;              // epoll 文件描述符
#elif defined(__APPLE__)
  int kqueue_fd;             // kqueue 文件描述符
#elif defined(_WIN32)
  HANDLE iocp;               // IOCP 句柄
#endif
  
  uint64_t time;             // 当前时间缓存
  int stop_flag;             // 停止标志
} uv_loop_t;
```

**关键字段说明：**
- `active_handles` - 活跃句柄计数
- `timer_heap` - 定时器最小堆
- `pending_queue` - 待处理回调队列
- `epoll_fd/kqueue_fd/iocp` - 平台特定 I/O
- `time` - 时间缓存（避免频繁系统调用）

### 2. 线程池（Thread Pool）

```c
// deps/uv/src/threadpool.c
static uv_thread_t default_threads[4];  // 默认4个线程
static QUEUE wq;                         // 工作队列
static uv_mutex_t mutex;                 // 互斥锁
static uv_cond_t cond;                   // 条件变量

// 线程池大小（环境变量）
// UV_THREADPOOL_SIZE=8 node app.js

static void init_threads(void) {
  unsigned int i;
  const char* val;
  uv_sem_t sem;

  // 从环境变量获取线程数
  val = getenv("UV_THREADPOOL_SIZE");
  nthreads = ARRAY_SIZE(default_threads);
  
  if (val != NULL)
    nthreads = atoi(val);
  
  if (nthreads == 0)
    nthreads = 1;
  
  if (nthreads > MAX_THREADPOOL_SIZE)
    nthreads = MAX_THREADPOOL_SIZE;

  // 创建线程
  for (i = 0; i < nthreads; i++) {
    if (uv_thread_create(threads + i, worker, &sem))
      abort();
  }
}

// 工作线程函数
static void worker(void* arg) {
  struct uv__work* w;
  QUEUE* q;

  for (;;) {
    uv_mutex_lock(&mutex);

    // 等待工作
    while (QUEUE_EMPTY(&wq))
      uv_cond_wait(&cond, &mutex);

    // 获取任务
    q = QUEUE_HEAD(&wq);
    QUEUE_REMOVE(q);
    QUEUE_INIT(q);

    uv_mutex_unlock(&mutex);

    w = QUEUE_DATA(q, struct uv__work, wq);
    
    // 执行任务
    w->work(w);

    // 通知主线程
    uv_mutex_lock(&w->loop->wq_mutex);
    w->work = NULL;
    QUEUE_INSERT_TAIL(&w->loop->wq, &w->wq);
    uv_async_send(&w->loop->wq_async);
    uv_mutex_unlock(&w->loop->wq_mutex);
  }
}
```

**线程池用途：**
- 文件 I/O（fs.readFile, fs.writeFile）
- DNS 解析（dns.lookup）
- CPU 密集任务（crypto.pbkdf2）
- 用户自定义（uv_queue_work）

**性能考虑：**
```javascript
// 默认4个线程可能不够
// UV_THREADPOOL_SIZE=8 node app.js

const crypto = require('crypto');

// 如果有8个并发操作，默认4个线程会导致排队

  crypto.pbkdf2('password', 'salt', 100000, 512, 'sha512', () => {
    console.log(`Task ${i} done`);
  });
}
```

### 3. 定时器堆（Timer Heap）

```c
// deps/uv/src/heap-inl.h
// 最小堆实现

// 插入定时器 - O(log n)
static void heap_insert(struct heap* heap,
                        struct heap_node* newnode,
                        heap_compare_fn less_than) {
  unsigned int index = heap->nelts++;
  
  // 向上调整
  while (index != 0) {
    unsigned int parent = (index - 1) / 2;
    
    if (less_than(heap->base[parent], newnode))
      break;
    
    heap->base[index] = heap->base[parent];
    index = parent;
  }
  
  heap->base[index] = newnode;
}

// 删除最小值 - O(log n)
static void heap_remove(struct heap* heap,
                        struct heap_node* node,
                        heap_compare_fn less_than) {
  unsigned int index = node->heap_index;
  
  if (index == heap->nelts - 1) {
    heap->nelts--;
    return;
  }
  
  // 用最后一个元素替换
  heap->base[index] = heap->base[--heap->nelts];
  
  // 向下调整
  heap_siftdown(heap, index, less_than);
}

// 定时器比较函数
static int timer_less_than(const struct heap_node* ha,
                           const struct heap_node* hb) {
  const uv_timer_t* a = container_of(ha, uv_timer_t, heap_node);
  const uv_timer_t* b = container_of(hb, uv_timer_t, heap_node);

  if (a->timeout < b->timeout)
    return 1;
  if (b->timeout < a->timeout)
    return 0;

  // 相同超时时间，按启动顺序
  return a->start_id < b->start_id;
}
```

**时间复杂度：**
- 插入：O(log n)
- 删除最小值：O(log n)
- 查找最小值：O(1)
- 堆化：O(n)

### 4. I/O 多路复用（Platform Specific）

#### Linux - epoll

```c
// deps/uv/src/unix/linux-core.c
int uv__io_poll(uv_loop_t* loop, int timeout) {
  struct epoll_event events[1024];
  struct epoll_event* pe;
  int nevents;
  int i;

  // epoll_wait 阻塞等待事件
  nevents = epoll_wait(loop->epoll_fd,
                       events,
                       ARRAY_SIZE(events),
                       timeout);

  if (nevents == -1) {
    if (errno != EINTR)
      abort();
    return;
  }

  // 处理就绪事件
  for (i = 0; i < nevents; i++) {
    pe = events + i;
    
    uv__io_t* w = pe->data.ptr;

    // 触发回调
    if (pe->events & (EPOLLIN | EPOLLOUT | EPOLLERR | EPOLLHUP)) {
      w->cb(loop, w, pe->events);
    }
  }
}
```

**epoll 特点：**
- 边缘触发（ET）模式
- 高效（O(1) 通知）
- 支持海量连接

#### macOS - kqueue

```c
// deps/uv/src/unix/kqueue.c
void uv__io_poll(uv_loop_t* loop, int timeout) {
  struct kevent events[1024];
  struct kevent* ev;
  int nevents;
  int i;

  // kevent 阻塞等待事件
  nevents = kevent(loop->kqueue_fd,
                   NULL,
                   0,
                   events,
                   ARRAY_SIZE(events),
                   timeout == -1 ? NULL : &spec);

  for (i = 0; i < nevents; i++) {
    ev = events + i;
    
    uv__io_t* w = ev->udata;

    // 触发回调
    if (ev->filter == EVFILT_READ || ev->filter == EVFILT_WRITE) {
      w->cb(loop, w, ev->filter);
    }
  }
}
```

**kqueue 特点：**
- 事件过滤器机制
- 支持多种事件类型
- 高性能

#### Windows - IOCP

```c
// deps/uv/src/win/core.c
void uv__poll(uv_loop_t* loop, DWORD timeout) {
  OVERLAPPED_ENTRY overlappeds[128];
  ULONG count;
  BOOL success;

  // GetQueuedCompletionStatusEx 批量获取
  success = GetQueuedCompletionStatusEx(
    loop->iocp,
    overlappeds,
    ARRAY_SIZE(overlappeds),
    &count,
    timeout,
    FALSE
  );

  for (i = 0; i < count; i++) {
    OVERLAPPED_ENTRY* oe = &overlappeds[i];
    uv_req_t* req = CONTAINING_RECORD(oe->lpOverlapped, uv_req_t, u.io.overlapped);

    // 执行完成回调
    req->u.io.queued_bytes = oe->dwNumberOfBytesTransferred;
    uv_insert_pending_req(loop, req);
  }
}
```

**IOCP 特点：**
- 完成端口模型
- 主动通知完成
- Windows 最佳性能

// timer-precision.js
const start = Date.now();
const delays = [];

function test(expected) {
  return new Promise(resolve => {
    const startTime = Date.now();
    
    setTimeout(() => {
      const actual = Date.now() - startTime;
      const delay = actual - expected;

  await test(50);

## 📝 总结

### 核心要点

1. **libuv 架构**：事件循环 + 线程池 + I/O 多路复用
2. **线程池**：默认4个线程，处理文件 I/O 和 DNS
3. **定时器堆**：最小堆，O(log n) 性能
4. **I/O 多路复用**：epoll/kqueue/IOCP 平台特定
5. **异步句柄**：统一接口，继承体系

### 实现关键

1. **跨平台抽象**：统一 API，平台特定实现
2. **性能优化**：最小堆、边缘触发、批量处理
3. **资源管理**：句柄生命周期、内存池
4. **线程安全**：线程池队列、互斥锁

### 应用价值

1. **理解异步**：深入 Node.js 异步原理
2. **性能调优**：合理设置线程池大小
3. **调试技巧**：监控事件循环状态
4. **扩展开发**：使用 libuv API 开发

### 注意事项

1. **线程池大小**：根据 I/O 密集度调整
2. **定时器精度**：受系统调度影响
3. **平台差异**：不同平台行为可能不同
