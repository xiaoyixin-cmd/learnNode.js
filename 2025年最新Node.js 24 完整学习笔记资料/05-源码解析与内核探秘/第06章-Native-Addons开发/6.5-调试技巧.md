## 💻 gdb 调试（Linux）

### 1. 基础用法

```bash
# 启动调试
gdb --args node test.js

# 或附加到运行中的进程
gdb -p <pid>

# 常用命令
(gdb) run                 # 运行
(gdb) continue (c)        # 继续
(gdb) next (n)            # 单步（跳过函数）
(gdb) step (s)            # 单步（进入函数）
(gdb) finish              # 执行到函数返回
(gdb) quit                # 退出
```

---

### 2. 断点

```bash
# 设置断点
(gdb) break addon.cc:42           # 文件:行号
(gdb) break MyFunction            # 函数名
(gdb) break addon.cc:42 if x==0   # 条件断点

# 查看断点
(gdb) info breakpoints

# 删除断点
(gdb) delete 1                    # 删除断点1
(gdb) clear addon.cc:42           # 清除特定位置断点
```

---

### 3. 查看变量

```bash
# 打印变量
(gdb) print variable_name
(gdb) print *pointer
(gdb) print array[0]

# 查看类型
(gdb) ptype variable_name

# 查看内存
(gdb) x/10xw 0x7fff5fbff000   # 查看10个word

# 查看所有局部变量
(gdb) info locals

# 查看参数
(gdb) info args
```

---

### 4. 调用栈

```bash
# 查看调用栈
(gdb) backtrace (bt)
(gdb) backtrace full       # 显示局部变量

# 切换栈帧
(gdb) frame 2              # 切换到帧2
(gdb) up                   # 向上一帧
(gdb) down                 # 向下一帧
```

---

### 5. 实战示例

```cpp
// addon.cc (有bug)
#include <node_api.h>

napi_value Divide(napi_env env, napi_callback_info info) {
  size_t argc = 2;
  napi_value args[2];
  napi_get_cb_info(env, info, &argc, args, nullptr, nullptr);

  double a, b;
  napi_get_value_double(env, args[0], &a);
  napi_get_value_double(env, args[1], &b);

  double result = a / b;  // Bug: 未检查 b == 0

  napi_value result_value;
  napi_create_double(env, result, &result_value);
  return result_value;
}
```

**调试：**
```bash
# 编译 Debug 版本
node-gyp rebuild --debug

# 启动调试
gdb --args node test.js

(gdb) break addon.cc:13      # 设置断点
(gdb) run                    # 运行

Breakpoint 1, Divide (...) at addon.cc:13
13        double result = a / b;

(gdb) print a
$1 = 10

(gdb) print b
$2 = 0                       # 发现问题！

(gdb) continue               # 继续运行
# 程序返回 inf 或崩溃
```

## 💻 lldb 调试（macOS）

### 1. 基础用法

```bash
# 启动调试
lldb -- node test.js

# 或附加到进程
lldb -p <pid>

# 常用命令
(lldb) run (r)               # 运行
(lldb) continue (c)          # 继续
(lldb) next (n)              # 单步
(lldb) step (s)              # 进入函数
(lldb) finish                # 执行到返回
(lldb) quit                  # 退出
```

---

### 2. 断点与查看

```bash
# 断点
(lldb) breakpoint set -f addon.cc -l 42
(lldb) breakpoint set -n MyFunction
(lldb) breakpoint list

# 查看变量
(lldb) frame variable         # 所有局部变量
(lldb) p variable_name        # 打印变量
(lldb) p *pointer             # 解引用

# 调用栈
(lldb) thread backtrace (bt)
(lldb) frame select 2         # 切换帧
```

---

### 3. 崩溃分析

```bash
# 程序崩溃后
(lldb) thread list            # 查看所有线程
(lldb) thread backtrace all   # 所有线程的调用栈
(lldb) frame info             # 当前帧信息
(lldb) register read          # 寄存器状态
```

## 💻 Valgrind 内存泄漏检测

### 1. 基本用法

```bash
# 安装 Valgrind
sudo apt-get install valgrind  # Linux
brew install valgrind          # macOS (需 Rosetta)

# 检测内存泄漏
valgrind --leak-check=full \
         --show-leak-kinds=all \
         --track-origins=yes \
         --verbose \
         node test.js
```

---

### 2. 输出解读

```bash
==12345== HEAP SUMMARY:
==12345==     in use at exit: 1,024 bytes in 1 blocks
==12345==   total heap usage: 100 allocs, 99 frees, 10,240 bytes allocated
==12345==
==12345== 1,024 bytes in 1 blocks are definitely lost
==12345==    at malloc (vg_replace_malloc.c:309)
==12345==    by MyFunction (addon.cc:42)
==12345==    by main (node_main.cc:123)
==12345==
==12345== LEAK SUMMARY:
==12345==    definitely lost: 1,024 bytes in 1 blocks
==12345==    indirectly lost: 0 bytes in 0 blocks
```

**分析：**
- `definitely lost`: 确定泄漏（需修复）
- `indirectly lost`: 间接泄漏
- `possibly lost`: 可能泄漏
- `still reachable`: 仍可达（程序退出时未释放）

---

### 3. 修复示例

```cpp
// Bug: 内存泄漏
napi_value ProcessData(napi_env env, napi_callback_info info) {
  char* buffer = new char[1024];  // 分配
  
  // 处理数据...
  
  // 忘记释放！
  return result;
}

// Fix: 正确释放
napi_value ProcessData(napi_env env, napi_callback_info info) {
  char* buffer = new char[1024];
  
  // 处理数据...
  
  delete[] buffer;  // 释放
  return result;
}

// 更好：使用 RAII
napi_value ProcessData(napi_env env, napi_callback_info info) {
  std::vector<char> buffer(1024);  // 自动管理
  
  // 处理数据...
  
  return result;  // 自动释放
}
```

## 💻 AddressSanitizer（ASan）

### 1. 编译启用

```python
# binding.gyp
{
  "targets": [{
    "target_name": "addon",
    "sources": [ "addon.cc" ],
    "cflags": [ "-fsanitize=address", "-fno-omit-frame-pointer" ],
    "ldflags": [ "-fsanitize=address" ]
  }]
}
```

```bash
# 构建
node-gyp rebuild

# 运行（自动检测）
node test.js
```

---

### 2. 检测类型

**缓冲区溢出：**
```cpp
char buffer[10];
buffer[10] = 'x';  // Out of bounds!

// ASan 输出：
// ==12345==ERROR: AddressSanitizer: stack-buffer-overflow
// WRITE of size 1 at 0x7fff5fbff00a
```

**Use-after-free：**
```cpp
char* ptr = new char[10];
delete[] ptr;
char c = ptr[0];  // Use after free!

// ASan 输出：
// ==12345==ERROR: AddressSanitizer: heap-use-after-free
```

**内存泄漏：**
```cpp
char* ptr = new char[10];
// 忘记 delete

// ASan 输出：
// Direct leak of 10 byte(s)
```

## 💻 实战案例

### 案例1：Segmentation Fault

**问题代码：**
```cpp
napi_value GetProperty(napi_env env, napi_callback_info info) {
  size_t argc = 1;
  napi_value args[1];
  napi_get_cb_info(env, info, &argc, args, nullptr, nullptr);

  char* key = nullptr;
  size_t key_len;
  napi_get_value_string_utf8(env, args[0], key, 0, &key_len);  // Bug!

  return result;
}
```

**调试：**
```bash
gdb --args node test.js

(gdb) run
Program received signal SIGSEGV, Segmentation fault.

(gdb) backtrace
#0  napi_get_value_string_utf8 (...)
#1  GetProperty (addon.cc:10)

(gdb) frame 1
(gdb) print key
$1 = 0x0       # 空指针！

# 修复：分配内存
char* key = new char[key_len + 1];
napi_get_value_string_utf8(env, args[0], key, key_len + 1, &key_len);
```

---

### 案例2：内存泄漏

**问题代码：**
```cpp
napi_value CreateObjects(napi_env env, napi_callback_info info) {
  for (int i = 0; i < 1000; i++) {
    MyObject* obj = new MyObject();  // 泄漏！
    // 处理...
  }
  return nullptr;
}
```

**检测：**
```bash
valgrind --leak-check=full node test.js

# 输出：
# 1,000,000 bytes in 1,000 blocks are definitely lost
# at new (addon.cc:42)
```

**修复：**
```cpp
napi_value CreateObjects(napi_env env, napi_callback_info info) {
  for (int i = 0; i < 1000; i++) {
    MyObject* obj = new MyObject();
    // 处理...
    delete obj;  // 修复！
  }
  
  // 更好：使用智能指针
  std::unique_ptr<MyObject> obj = std::make_unique<MyObject>();
  
  return nullptr;
}
```

---

### 案例3：数据竞争

**问题代码：**
```cpp
int global_counter = 0;

void WorkerThread() {
  for (int i = 0; i < 1000; i++) {
    global_counter++;  // 数据竞争！
  }
}
```

**检测：**
```bash
# 使用 ThreadSanitizer
# binding.gyp 添加: "-fsanitize=thread"

node test.js

# 输出：
# WARNING: ThreadSanitizer: data race
```

**修复：**
```cpp
#include <mutex>

std::mutex mtx;
int global_counter = 0;

void WorkerThread() {
  for (int i = 0; i < 1000; i++) {
    std::lock_guard<std::mutex> lock(mtx);
    global_counter++;  // 安全！
  }
}
```

## 💻 调试技巧总结

### 1. 编译选项

```python
# Debug 模式
{
  "cflags": [
    "-g",                  # 调试符号
    "-O0",                 # 禁用优化
    "-Wall",               # 所有警告
    "-Wextra"              # 额外警告
  ]
}

# Release 模式
{
  "cflags": [
    "-O3",                 # 最高优化
    "-DNDEBUG"             # 禁用断言
  ]
}
```

---

### 2. 日志记录

```cpp
// 添加调试日志
#ifdef DEBUG
  #define LOG(msg) printf("[DEBUG] %s:%d: %s\n", __FILE__, __LINE__, msg)
#else
  #define LOG(msg)
#endif

napi_value MyFunction(napi_env env, napi_callback_info info) {
  LOG("Entering MyFunction");
  
  // 处理...
  
  LOG("Exiting MyFunction");
  return result;
}
```

---

### 3. 断言

```cpp
#include <cassert>

napi_value Divide(napi_env env, napi_callback_info info) {
  // ...
  assert(b != 0 && "Divisor cannot be zero");
  
  double result = a / b;
  return result_value;
}
```

## 📝 总结

### 核心工具

1. **gdb/lldb**：断点调试、崩溃分析
2. **Valgrind**：内存泄漏检测
3. **ASan**：内存错误检测
4. **perf/Instruments**：性能分析

### 常见问题

1. **Segmentation Fault**：空指针、越界访问
2. **内存泄漏**：忘记释放、智能指针
3. **数据竞争**：多线程访问共享数据
4. **性能瓶颈**：算法、内存分配

### 最佳实践

1. **Debug 模式开发**：启用调试符号
2. **RAII**：自动资源管理
3. **智能指针**：避免手动 delete
4. **代码审查**：多人检查
5. **单元测试**：覆盖边界情况
