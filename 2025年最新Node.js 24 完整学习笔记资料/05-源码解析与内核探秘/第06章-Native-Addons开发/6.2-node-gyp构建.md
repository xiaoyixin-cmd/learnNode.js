# 6.2 使用 node-gyp 构建原生模块

> 深入理解 Node.js 原生模块的构建工具链

---

## 📋 核心概念

**node-gyp** 是 Node.js 原生模块的**官方构建工具**，基于 Google 的 GYP（Generate Your Projects）。

**核心问题：**
- node-gyp 如何工作？
- binding.gyp 如何配置？
- 如何实现跨平台构建？
- 如何处理第三方依赖？

**构建流程：**
```
binding.gyp → GYP → Makefile/MSBuild → 编译 → .node 文件
```

## 💻 binding.gyp 基础

### 1. 最简配置

```python
# binding.gyp
{
  "targets": [{
    "target_name": "addon",
    "sources": [ "addon.cc" ]
  }]
}
```

**说明：**
- `target_name`: 输出文件名（addon.node）
- `sources`: 源文件列表

---

### 2. 完整配置

```python
{
  "targets": [{
    "target_name": "addon",
    "sources": [
      "src/addon.cc",
      "src/helper.cc"
    ],
    "include_dirs": [
      "<!@(node -p \"require('node-addon-api').include\")",
      "include/"
    ],
    "dependencies": [
      "<!(node -p \"require('node-addon-api').gyp\")"
    ],
    "defines": [
      "NAPI_DISABLE_CPP_EXCEPTIONS",
      "MY_DEFINE=1"
    ],
    "cflags": [
      "-Wall",
      "-O3"
    ],
    "cflags!": [ "-fno-exceptions" ],
    "cflags_cc!": [ "-fno-exceptions" ],
    "conditions": [
      ['OS=="mac"', {
        "xcode_settings": {
          "GCC_ENABLE_CPP_EXCEPTIONS": "YES",
          "CLANG_CXX_LIBRARY": "libc++",
          "MACOSX_DEPLOYMENT_TARGET": "10.13"
        }
      }],
      ['OS=="win"', {
        "msvs_settings": {
          "VCCLCompilerTool": {
            "ExceptionHandling": 1
          }
        }
      }]
    ]
  }]
}
```

---

### 3. 配置项详解

| 配置项 | 说明 | 示例 |
|--------|------|------|
| `target_name` | 目标名称 | `"addon"` |
| `sources` | 源文件 | `["src/*.cc"]` |
| `include_dirs` | 头文件目录 | `["include/"]` |
| `libraries` | 链接库 | `["-lpng"]` |
| `defines` | 预处理宏 | `["DEBUG=1"]` |
| `cflags` | C 编译标志 | `["-Wall"]` |
| `cflags_cc` | C++ 编译标志 | `["-std=c++17"]` |
| `ldflags` | 链接标志 | `["-L/usr/local/lib"]` |

## 💻 构建命令

### 1. 基本命令

```bash
# 配置
node-gyp configure

# 构建
node-gyp build

# 重新构建
node-gyp rebuild

# 清理
node-gyp clean
```

---

### 2. 指定配置

```bash
# Debug 模式
node-gyp rebuild --debug

# Release 模式（默认）
node-gyp rebuild --release

# 指定架构
node-gyp rebuild --arch=x64

# 详细输出
node-gyp rebuild --verbose
```

---

### 3. NPM Scripts 集成

```json
// package.json
{
  "name": "my-addon",
  "version": "1.0.0",
  "main": "index.js",
  "scripts": {
    "install": "node-gyp rebuild",
    "build": "node-gyp rebuild",
    "build:debug": "node-gyp rebuild --debug",
    "clean": "node-gyp clean"
  },
  "dependencies": {
    "node-addon-api": "^5.0.0"
  },

}
```

---

### 2. 动态库链接

```python
{
  "targets": [{
    "target_name": "addon",
    "sources": [ "addon.cc" ],
    "libraries": [
      "-lz",       # zlib
      "-lpng",     # libpng
      "-lcrypto"   # OpenSSL
    ],
    "conditions": [
      ['OS=="mac"', {

      }],
      ['OS=="linux"', {
        "libraries": [
          "-Wl,-rpath,'$$ORIGIN'"
        ]
      }]
    ]
  }]
}
```

## 💻 跨平台构建

### 1. 条件编译

```python
{
  "targets": [{
    "target_name": "addon",
    "sources": [ "addon.cc" ],
    "conditions": [
      ['OS=="mac"', {
        "sources": [ "src/mac_specific.cc" ],
        "xcode_settings": {
          "OTHER_CFLAGS": [ "-arch x86_64", "-arch arm64" ]
        }
      }],
      ['OS=="linux"', {
        "sources": [ "src/linux_specific.cc" ],
        "cflags": [ "-fPIC" ]
      }],
      ['OS=="win"', {
        "sources": [ "src/win_specific.cc" ],
        "msvs_settings": {
          "VCCLCompilerTool": {
            "RuntimeLibrary": 2  # MultiThreadedDLL
          }
        }
      }]
    ]
  }]
}
```

---

### 2. C++ 代码条件编译

```cpp
// addon.cc
#ifdef _WIN32
  #include <windows.h>
  // Windows 特定代码
#elif __APPLE__
  #include <TargetConditionals.h>
  // macOS 特定代码
#elif __linux__
  #include <unistd.h>
  // Linux 特定代码
#endif
```

```json

  "binary": {
```

## 💻 实战案例

### 案例1：集成 OpenSSL

```python
# binding.gyp
{
  "targets": [{
    "target_name": "crypto_addon",
    "sources": [ "crypto.cc" ],
    "conditions": [
      ['OS=="mac"', {
        "include_dirs": [
          "/usr/local/opt/openssl/include"
        ],
        "libraries": [
          "/usr/local/opt/openssl/lib/libcrypto.a"
        ]
      }],
      ['OS=="linux"', {
        "cflags": [ "<!@(pkg-config --cflags openssl)" ],
        "libraries": [ "<!@(pkg-config --libs openssl)" ]
      }],
      ['OS=="win"', {
        "include_dirs": [
          "C:/OpenSSL-Win64/include"
        ],
        "libraries": [
          "C:/OpenSSL-Win64/lib/libcrypto.lib"
        ]
      }]
    ]
  }]
}
```

```cpp
// crypto.cc
#include <node_api.h>
#include <openssl/sha.h>

napi_value Sha256(napi_env env, napi_callback_info info) {
  // ... SHA256 实现
}

NAPI_MODULE_INIT() {
  napi_value fn;
  napi_create_function(env, "sha256", NAPI_AUTO_LENGTH, Sha256, nullptr, &fn);
  napi_set_named_property(env, exports, "sha256", fn);
  return exports;
}
```

## 💻 常见问题

### 1. Python 版本问题

```bash
# 指定 Python 版本
npm config set python /usr/bin/python3

# 或环境变量
export PYTHON=/usr/bin/python3
node-gyp rebuild
```

---

### 2. Visual Studio 版本

```bash
# Windows: 指定 VS 版本
npm config set msvs_version 2019

# 或命令行
node-gyp rebuild --msvs_version=2019
```

---

### 3. 缓存问题

```bash
# 清除 node-gyp 缓存
rm -rf ~/.node-gyp

# 清除 npm 缓存
npm cache clean --force
```

---

### 4. 编译错误调试

```bash
# 详细输出
node-gyp rebuild --verbose

# 保留临时文件
node-gyp rebuild --debug --verbose
```

## 💻 高级技巧

### 1. 多目标构建

```python
{
  "targets": [
    {
      "target_name": "addon",
      "sources": [ "addon.cc" ]
    },
    {
      "target_name": "tests",
      "type": "executable",
      "sources": [ "tests/test.cc" ],
      "dependencies": [ "addon" ]
    }
  ]
}
```

---

### 2. 自定义变量

```python
{
  "variables": {
    "openssl_root": "/usr/local/opt/openssl"
  },
  "targets": [{
    "target_name": "addon",
    "sources": [ "addon.cc" ],
    "include_dirs": [ "<(openssl_root)/include" ],
    "libraries": [ "<(openssl_root)/lib/libcrypto.a" ]
  }]
}
```

---

### 3. 构建后脚本

```json
// package.json
{
  "scripts": {
    "install": "node-gyp rebuild && node scripts/post-build.js"
  }
}
```

```javascript
// scripts/post-build.js
const fs = require('fs');
const path = require('path');

// 复制 .node 文件到 lib/
const source = path.join(__dirname, '../build/Release/addon.node');
const target = path.join(__dirname, '../lib/addon.node');

fs.copyFileSync(source, target);
console.log('Post-build: Copied addon.node to lib/');
```

## 📝 总结

### 核心要点

1. **node-gyp**：Node.js 原生模块构建工具
2. **binding.gyp**：GYP 配置文件，定义构建规则
3. **跨平台**：条件编译支持 Windows/macOS/Linux
4. **依赖管理**：静态库/动态库链接

### 实现关键

1. **配置文件**：binding.gyp 完整配置
2. **构建命令**：configure、build、rebuild
3. **预编译**：node-pre-gyp 分发二进制
4. **条件编译**：OS、架构特定代码

### 应用价值

1. **自动化**：统一的构建流程
2. **跨平台**：一份代码多平台编译
3. **可维护**：清晰的配置文件
4. **易分发**：预编译二进制支持
