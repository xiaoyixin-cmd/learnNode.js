# 3.2 http 模块：从 net.Server 到 HTTPParser（llhttp）

> 深入理解 HTTP 服务器如何基于 TCP 构建协议解析层

---

## 📋 核心概念

http 模块是基于 **net.Server**（TCP）之上的应用层协议实现，展示了网络分层的完整过程。

**核心问题：**
- HTTP 服务器如何接受 TCP 连接？
- HTTP 协议如何解析（llhttp）？
- Keep-Alive 如何实现？
- HTTP/2 如何集成？

**技术栈：**
```
Application Layer:  http.Server (HTTP/1.1)
                    ↓
Transport Layer:    net.Server (TCP)
                    ↓
libuv:             uv_tcp_t (TCP handle)
                    ↓
OS:                socket (系统调用)
```

## 💻 HTTPParser 详解

### 1. llhttp 解析器

**历史演进：**
```
Node.js < 12:  http-parser (C)
Node.js ≥ 12:  llhttp (C)  ← 更快、更安全
```

**llhttp 特点：**
- 基于状态机的解析器
- 零拷贝设计
- 比 http-parser 快 156%
- 支持 HTTP/1.0、HTTP/1.1

### 2. C++ Binding

```cpp
// src/node_http_parser.cc
namespace node {

using v8::Context;
using v8::FunctionCallbackInfo;
using v8::FunctionTemplate;
using v8::Local;
using v8::Object;
using v8::Value;

// 🔥 Parser 类
class Parser : public AsyncWrap {
 public:
  Parser(Environment* env, Local<Object> wrap, enum http_parser_type type)
      : AsyncWrap(env, wrap, AsyncWrap::PROVIDER_HTTPPARSER) {
    // 🔥 初始化 llhttp
    llhttp_init(&parser_, type, &settings);
    llhttp_set_lenient(&parser_, true);
  }

  // 🔥 执行解析
  static void Execute(const FunctionCallbackInfo<Value>& args) {
    Environment* env = Environment::GetCurrent(args);
    Parser* parser;
    ASSIGN_OR_RETURN_UNWRAP(&parser, args.Holder());

    CHECK(Buffer::HasInstance(args[0]));
    Local<Object> buffer_obj = args[0].As<Object>();
    
    // 🔥 获取数据
    char* data = Buffer::Data(buffer_obj);
    size_t len = Buffer::Length(buffer_obj);

    // 🔥 执行 llhttp 解析
    size_t nread = llhttp_execute(&parser->parser_, data, len);

    // 🔥 检查错误
    if (parser->parser_.error != HPE_OK) {
      enum llhttp_errno err = llhttp_get_errno(&parser->parser_);
      Local<Value> e = Exception::Error(env->parse_error_string());
      args.GetReturnValue().Set(e);
      return;
    }

    args.GetReturnValue().Set(nread);
  }

  // 🔥 完成解析
  static void Finish(const FunctionCallbackInfo<Value>& args) {
    Environment* env = Environment::GetCurrent(args);
    Parser* parser;
    ASSIGN_OR_RETURN_UNWRAP(&parser, args.Holder());

    // 🔥 调用 llhttp_finish
    int rv = llhttp_finish(&parser->parser_);

    if (rv != HPE_OK) {
      enum llhttp_errno err = llhttp_get_errno(&parser->parser_);
      Local<Value> e = Exception::Error(env->parse_error_string());
      args.GetReturnValue().Set(e);
    }
  }

 private:
  llhttp_t parser_;
  llhttp_settings_t settings;
};

} // namespace node
```

## 💻 IncomingMessage 与 ServerResponse

### 1. IncomingMessage（请求对象）

```javascript
// lib/_http_incoming.js
const { Readable } = require('stream');

class IncomingMessage extends Readable {
  constructor(socket) {
    super();

    this.socket = socket;
    this.httpVersion = null;
    this.headers = {};
    this.rawHeaders = [];
    this.trailers = {};
    this.rawTrailers = [];
    this.url = '';
    this.method = null;
    this.statusCode = null;
    this.statusMessage = null;
    this.complete = false;
  }

  // 🔥 读取 Body
  _read(n) {
    if (!this.socket) {
      this.push(null);
      return;
    }

    // 从 socket 读取数据（由 parser 推送）
    const state = this.socket._readableState;
    if (!state.reading) {
      this.socket.read(0);
    }
  }

  // 🔥 添加 Headers
  _addHeaderLines(headers, n) {
    if (headers && headers.length) {
      for (let i = 0; i < n; i += 2) {
        this._addHeaderLine(headers[i], headers[i + 1]);
      }
    }
  }

  _addHeaderLine(field, value) {
    field = field.toLowerCase();
    
    // 🔥 处理多值 Header
    const dest = this.complete ? this.trailers : this.headers;
    
    if (dest[field] === undefined) {
      dest[field] = value;
    } else if (Array.isArray(dest[field])) {
      dest[field].push(value);
    } else {
      dest[field] = [dest[field], value];
    }

    // 保存原始 Header
    if (this.complete) {
      this.rawTrailers.push(field, value);
    } else {
      this.rawHeaders.push(field, value);
    }
  }
}
```

### 2. ServerResponse（响应对象）

```javascript
// lib/_http_server.js
const { OutgoingMessage } = require('_http_outgoing');

class ServerResponse extends OutgoingMessage {
  constructor(req) {
    super();

    this.req = req;
    this.statusCode = 200;
    this.statusMessage = undefined;
    this.sendDate = true;
    this._sent100 = false;
    this._defaultKeepAlive = true;
    this._removedConnection = false;
    this._removedContLen = false;
    this._removedTE = false;
    this._contentLength = null;
    this._hasBody = true;
    this._trailer = '';
  }

  // 🔥 设置状态码
  writeHead(statusCode, reason, obj) {
    let headers;
    
    if (typeof reason === 'string') {
      this.statusMessage = reason;
      headers = obj;
    } else {
      this.statusMessage = getStatusMessage(statusCode);
      headers = reason;
    }

    this.statusCode = statusCode;

    // 🔥 设置 Headers
    if (headers) {
      const keys = Object.keys(headers);
      for (let i = 0; i < keys.length; i++) {
        const k = keys[i];
        this.setHeader(k, headers[k]);
      }
    }

    // 🔥 写入状态行和 Headers
    this._implicitHeader();
    
    return this;
  }

  // 🔥 写入数据
  write(chunk, encoding, callback) {
    if (this.finished) {
      // 已完成，报错
      const err = new Error('write after end');
      process.nextTick(() => this.emit('error', err));
      return false;
    }

    if (!this._header) {
      this._implicitHeader();
    }

    return this._writeRaw(chunk, encoding, callback);
  }

  // 🔥 结束响应
  end(data, encoding, callback) {
    if (this.finished) {
      return this;
    }

    if (data) {
      this.write(data, encoding);
    }

    this.finished = true;
    
    // 🔥 发送最终数据
    this._finish();

    if (callback) {
      this.once('finish', callback);
    }

    return this;
  }

  // 🔥 隐式写入状态行和 Headers
  _implicitHeader() {
    this.writeHead(this.statusCode);
  }

  // 🔥 生成状态行
  _storeHeader(firstLine, headers) {
    const statusLine = `HTTP/${this.req.httpVersion} ${this.statusCode} ${this.statusMessage}\r\n`;

    let header = statusLine + headers;

    if (this.sendDate) {
      header += 'Date: ' + utcDate() + '\r\n';
    }

    // Connection: keep-alive
    if (this.shouldKeepAlive) {
      header += 'Connection: keep-alive\r\n';
    }

    header += '\r\n';  // 空行分隔 Headers 和 Body

    this._header = header;
    this._headerSent = true;
    
    // 🔥 写入 socket
    this._writeRaw(this._header, 'latin1', null);
  }
}
```

## 💻 Keep-Alive 机制

### 1. 连接复用

```javascript
// lib/_http_server.js
function connectionListener(socket) {
  // 🔥 设置 Keep-Alive 超时
  const keepAliveTimeout = server.keepAliveTimeout;
  if (keepAliveTimeout > 0) {
    socket.setTimeout(keepAliveTimeout);
  }

  // 🔥 监听超时
  socket.on('timeout', () => {
    const req = socket.parser && socket.parser.incoming;
    const reqTimeout = req && !req.complete && req.emit('timeout', socket);
    const res = socket._httpMessage;
    const resTimeout = res && res.emit('timeout', socket);
    const serverTimeout = server.emit('timeout', socket);

    if (!reqTimeout && !resTimeout && !serverTimeout)
      socket.destroy();
  });

  // 🔥 响应完成后检查是否保持连接
  parser.onIncoming = (req, res) => {
    res.on('finish', () => {
      if (res.shouldKeepAlive) {
        // 🔥 保持连接，等待下一个请求
        parser.reinitialize(HTTPParser.REQUEST);
        socket.removeAllListeners('timeout');
        socket.setTimeout(keepAliveTimeout);
      } else {
        // 🔥 关闭连接
        socket.destroy();
      }
    });
  };
}
```

### 2. Keep-Alive 条件

// lib/_http_outgoing.js
function shouldKeepAlive(req, res) {
  // 🔥 HTTP/1.0 需要显式指定
  if (req.httpVersionMajor === 1 && req.httpVersionMinor === 0) {
    return req.headers.connection === 'keep-alive';
  }

  // 🔥 HTTP/1.1 默认 Keep-Alive
  if (req.httpVersionMajor === 1 && req.httpVersionMinor === 1) {
    return req.headers.connection !== 'close';
  }

  return false;
}

    nghttp2_session_callbacks_set_on_data_chunk_recv_callback(
      callbacks, OnDataChunkRecv);

    // 创建 server session
    nghttp2_session_server_new(&session_, callbacks, this);
    
    nghttp2_session_callbacks_del(callbacks);
  }

  // 🔥 接收数据
  static void Consume(const FunctionCallbackInfo<Value>& args) {
    Http2Session* session;
    ASSIGN_OR_RETURN_UNWRAP(&session, args.Holder());

    Local<ArrayBuffer> buf = args[0].As<ArrayBuffer>();
    size_t len = buf->ByteLength();
    
    // 🔥 提交给 nghttp2 解析
    ssize_t nread = nghttp2_session_mem_recv(
      session->session_,
      static_cast<const uint8_t*>(buf->GetBackingStore()->Data()),
      len
    );

    args.GetReturnValue().Set(static_cast<double>(nread));
  }

 private:
  nghttp2_session* session_;
};

## 📝 总结

### 核心要点

1. **分层设计**：HTTP → TCP → libuv → OS
2. **llhttp解析**：状态机、零拷贝、高性能
3. **Keep-Alive**：连接复用、超时管理

1. **net.Server**: TCP 服务器基础
2. **HTTPParser**: 协议解析（llhttp）
3. **IncomingMessage**: 请求对象（Readable Stream）
4. **ServerResponse**: 响应对象（Writable Stream）

1. **理解网络**：深入HTTP协议实现
