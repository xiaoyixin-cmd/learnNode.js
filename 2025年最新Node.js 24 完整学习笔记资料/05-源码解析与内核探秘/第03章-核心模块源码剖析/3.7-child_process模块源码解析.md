# 3.7 child_process 模块：spawn/fork 底层实现与 IPC

> 深入理解 Node.js 进程管理与进程间通信机制

---

## 📋 核心概念

child_process 模块实现了 **进程创建与管理**，是实现多进程架构的基础。

**核心问题：**
- `spawn()` vs `fork()` vs `exec()` 的区别？
- IPC（进程间通信）如何实现？
- stdio 如何继承与重定向？
- 如何处理子进程信号？

**方法对比：**
```
spawn()    → 最底层，返回 Stream
fork()     → spawn() 特例，启动 Node.js 进程，内置 IPC
exec()     → 创建 shell，缓冲输出
execFile() → 不创建 shell，缓冲输出
```

### 2. ChildProcess 类

```javascript
// lib/internal/child_process.js
const { Process } = internalBinding('process_wrap');

class ChildProcess extends EventEmitter {
  constructor() {
    super();

    // 🔥 底层进程句柄
    this._handle = null;

    // 🔥 进程信息
    this.pid = null;
    this.exitCode = null;
    this.signalCode = null;
    this.killed = false;

    // 🔥 stdio 流
    this.stdin = null;
    this.stdout = null;
    this.stderr = null;
    this.stdio = null;

    // 🔥 IPC 通道
    this.channel = null;
    this.connected = false;

    // 🔥 状态标志
    this._closesNeeded = 1;
    this._closesGot = 0;
  }

  // 🔥 spawn 子进程
  spawn(options) {
    // 🔥 创建底层进程句柄
    this._handle = new Process();

    // 🔥 设置 onexit 回调
    this._handle.onexit = (exitCode, signalCode) => {
      this._onExit(exitCode, signalCode);
    };

    // 🔥 准备 stdio
    const stdio = getValidStdio(options.stdio);

    // 🔥 调用底层 spawn
    const err = this._handle.spawn({
      file: options.file,
      args: options.args,
      cwd: options.cwd,
      envPairs: options.envPairs,
      stdio,
      uid: options.uid,
      gid: options.gid,
      detached: options.detached,
      windowsHide: options.windowsHide
    });

    if (err) {
      // 🔥 spawn 失败
      process.nextTick(() => {
        this._handle.onexit(err);
      });
      return err;
    }

    // 🔥 获取 PID
    this.pid = this._handle.pid;

    // 🔥 设置 stdio 流
    for (let i = 0; i < stdio.length; i++) {
      const stream = stdio[i];
      
      if (stream.type === 'pipe') {
        if (i === 0) {
          this.stdin = stream.handle;
        } else {
          if (i === 1) {
            this.stdout = stream.handle;
          } else {
            this.stderr = stream.handle;
          }
        }
      } else if (stream.type === 'ipc') {
        // 🔥 IPC 通道
        this.channel = stream.handle;
        this.connected = true;
      }
    }

    // 🔥 stdio 数组
    this.stdio = [this.stdin, this.stdout, this.stderr, ...];

    return 0;
  }

  // 🔥 进程退出回调
  _onExit(exitCode, signalCode) {
    this.exitCode = exitCode;
    this.signalCode = signalCode;

    // 🔥 清理资源
    if (this._handle) {
      this._handle.close();
      this._handle = null;
    }

    // 🔥 触发 exit 事件
    this.emit('exit', exitCode, signalCode);

    // 🔥 触发 close 事件
    if (++this._closesGot === this._closesNeeded) {
      this.emit('close', exitCode, signalCode);
    }
  }

  // 🔥 kill 子进程
  kill(signal) {
    if (this._handle == null) {
      return false;
    }

    const err = this._handle.kill(signal || 'SIGTERM');

    if (err === 0) {
      this.killed = true;
      return true;
    }

    return false;
  }
}
```

```cpp
namespace node {

using v8::Context;
using v8::FunctionCallbackInfo;
using v8::Local;
using v8::Object;
using v8::Value;

class ProcessWrap : public HandleWrap {
 public:
  // 🔥 Spawn 子进程
  static void Spawn(const FunctionCallbackInfo<Value>& args) {
    Environment* env = Environment::GetCurrent(args);
    ProcessWrap* wrap;
    ASSIGN_OR_RETURN_UNWRAP(&wrap, args.Holder());

    // 🔥 获取参数
    Local<Object> js_options = args[0].As<Object>();
    
    // file, args, cwd, env, stdio, uid, gid, detached...
    Local<Value> file_v = js_options->Get(...);
    node::Utf8Value file(env->isolate(), file_v);

    // 🔥 准备 uv_process_options_t
    uv_process_options_t options;
    memset(&options, 0, sizeof(uv_process_options_t));
    
    options.exit_cb = OnExit;
    options.file = *file;
    options.args = ...;   // char**
    options.env = ...;    // char**
    options.cwd = ...;    // const char*
    options.flags = ...;  // UV_PROCESS_DETACHED, etc.
    options.stdio_count = ...;
    options.stdio = ...;  // uv_stdio_container_t*
    options.uid = ...;
    options.gid = ...;

    // 🔥 调用 libuv spawn
    int err = uv_spawn(env->event_loop(), &wrap->process_, &options);

    if (err == 0) {
      // 🔥 获取 PID
      args.GetReturnValue().Set(wrap->process_.pid);
    }

    args.GetReturnValue().Set(err);
  }

  // 🔥 退出回调
  static void OnExit(uv_process_t* handle, int64_t exit_status, int term_signal) {
    ProcessWrap* wrap = ContainerOf(&ProcessWrap::process_, handle);
    Environment* env = wrap->env();

    HandleScope handle_scope(env->isolate());
    Context::Scope context_scope(env->context());

    Local<Value> argv[] = {
      Number::New(env->isolate(), static_cast<double>(exit_status)),
      OneByteString(env->isolate(), signo_string(term_signal))
    };

    // 🔥 调用 JavaScript 回调
    wrap->MakeCallback(env->onexit_string(), arraysize(argv), argv);
  }

  // 🔥 Kill 子进程
  static void Kill(const FunctionCallbackInfo<Value>& args) {
    ProcessWrap* wrap;
    ASSIGN_OR_RETURN_UNWRAP(&wrap, args.Holder());

    int signal = args[0]->Int32Value(...);

    // 🔥 发送信号
    int err = uv_process_kill(&wrap->process_, signal);

    args.GetReturnValue().Set(err);
  }

 private:
  uv_process_t process_;
};

} // namespace node
```

### 4. libuv 进程创建

```c
// deps/uv/src/unix/process.c
int uv_spawn(uv_loop_t* loop,
             uv_process_t* process,
             const uv_process_options_t* options) {
  int err;
  int signal_pipe[2] = { -1, -1 };
  int pipes_storage[8][2];
  int (*pipes)[2];
  int stdio_count;
  pid_t pid;

  // 🔥 准备 stdio
  stdio_count = options->stdio_count;
  if (stdio_count < 3) stdio_count = 3;

  pipes = pipes_storage;
  if (stdio_count > (int) ARRAY_SIZE(pipes_storage)) {
    pipes = malloc(stdio_count * sizeof(*pipes));
  }

  // 🔥 创建管道
  for (int i = 0; i < stdio_count; i++) {
    pipes[i][0] = -1;
    pipes[i][1] = -1;
  }

  for (int i = 0; i < stdio_count; i++) {
    if (options->stdio[i].flags & UV_CREATE_PIPE) {
      // 🔥 创建管道
      if (socketpair(AF_UNIX, SOCK_STREAM, 0, pipes[i]) == -1) {
        err = UV__ERR(errno);
        goto error;
      }
    }
  }

  // 🔥 创建信号管道
  if (socketpair(AF_UNIX, SOCK_STREAM, 0, signal_pipe) == -1) {
    err = UV__ERR(errno);
    goto error;
  }

  // 🔥 fork 子进程
  pid = fork();

  if (pid == -1) {
    // fork 失败
    err = UV__ERR(errno);
    goto error;
  }

  if (pid == 0) {
    // 🔥 子进程
    uv__process_child_init(options, stdio_count, pipes, signal_pipe[1]);
    abort();  // 不应该到达这里
  }

  // 🔥 父进程
  process->pid = pid;

  // 🔥 关闭子进程端的管道
  for (int i = 0; i < stdio_count; i++) {
    if (pipes[i][1] >= 0) {
      close(pipes[i][1]);
      pipes[i][1] = -1;
    }
  }

  close(signal_pipe[1]);

  // 🔥 设置父进程端的管道
  for (int i = 0; i < stdio_count; i++) {
    if (options->stdio[i].flags & UV_CREATE_PIPE) {
      if (pipes[i][0] >= 0) {
        // 将 fd 关联到 uv_pipe_t
        uv_pipe_open((uv_pipe_t*) options->stdio[i].data.stream, pipes[i][0]);
      }
    }
  }

  return 0;

error:
  // 清理资源
  for (int i = 0; i < stdio_count; i++) {
    close(pipes[i][0]);
    close(pipes[i][1]);
  }
  
  if (pipes != pipes_storage) {
    free(pipes);
  }

  return err;
}

// 🔥 子进程初始化
static void uv__process_child_init(const uv_process_options_t* options,
                                    int stdio_count,
                                    int (*pipes)[2],
                                    int error_fd) {
  sigset_t set;
  int close_fd;
  int use_fd;
  int fd;
  int n;

  // 🔥 重置信号处理
  for (n = 1; n < 32; n++) {
    signal(n, SIG_DFL);
  }

  sigemptyset(&set);
  sigprocmask(SIG_SETMASK, &set, NULL);

  // 🔥 设置 stdio
  for (fd = 0; fd < stdio_count; fd++) {
    close_fd = pipes[fd][0];
    use_fd = pipes[fd][1];

    if (use_fd < 0) {
      if (options->stdio[fd].flags & UV_IGNORE) {
        // 忽略
        use_fd = open("/dev/null", fd == 0 ? O_RDONLY : O_RDWR);
        close_fd = use_fd;
      } else if (options->stdio[fd].flags & UV_INHERIT_FD) {
        // 继承
        use_fd = options->stdio[fd].data.fd;
      }
    }

    if (use_fd >= 0) {
      // 🔥 dup2 重定向
      if (use_fd != fd) {
        dup2(use_fd, fd);
        close(use_fd);
      }
    }

    if (close_fd >= 0) {
      close(close_fd);
    }
  }

  // 🔥 设置工作目录
  if (options->cwd != NULL && chdir(options->cwd)) {
    uv__write_int(error_fd, UV__ERR(errno));
    _exit(127);
  }

  // 🔥 设置 uid/gid
  if (options->flags & UV_PROCESS_SETGID) {
    if (setgid(options->gid)) {
      uv__write_int(error_fd, UV__ERR(errno));
      _exit(127);
    }
  }

  if (options->flags & UV_PROCESS_SETUID) {
    if (setuid(options->uid)) {
      uv__write_int(error_fd, UV__ERR(errno));
      _exit(127);
    }
  }

  // 🔥 设置环境变量
  if (options->env != NULL) {
    environ = options->env;
  }

  // 🔥 执行程序
  execvp(options->file, options->args);

  // execvp 失败
  uv__write_int(error_fd, UV__ERR(errno));
  _exit(127);
}
```

## 💻 fork() 实现

### 1. JavaScript 层

```javascript
// lib/child_process.js
function fork(modulePath, args, options) {
  // 🔥 参数归一化
  options = options || {};
  args = args || [];

  // 🔥 强制使用当前 Node.js 可执行文件
  options.execPath = options.execPath || process.execPath;

  // 🔥 添加模块路径参数
  const execArgv = options.execArgv || process.execArgv;
  const argsWithModule = execArgv.concat([modulePath], args);

  // 🔥 强制创建 IPC 通道
  if (options.stdio == null) {
    options.stdio = ['pipe', 'pipe', 'pipe', 'ipc'];
  } else if (!Array.isArray(options.stdio)) {
    throw new TypeError('stdio must be an array');
  } else {
    // 确保有 IPC 通道
    if (options.stdio.indexOf('ipc') === -1) {
      options.stdio.push('ipc');
    }
  }

  // 🔥 调用 spawn
  return spawn(options.execPath, argsWithModule, options);
}
```

```javascript
// lib/internal/child_process.js
// 🔥 IPC 通道基于 Unix Domain Socket 或 Named Pipe (Windows)

// 设置 IPC 通道
function setupChannel(target, channel) {
  target.channel = channel;
  target.connected = true;

  // 🔥 send() 发送消息
  target.send = function(message, sendHandle, options, callback) {
    if (typeof sendHandle === 'function') {
      callback = sendHandle;
      sendHandle = undefined;
      options = undefined;
    } else if (typeof options === 'function') {
      callback = options;
      options = undefined;
    }

    if (options && options.keepOpen) {
      // ...
    }

    if (!this.connected) {
      const err = new Error('Channel closed');
      if (callback) {
        process.nextTick(callback, err);
      }
      return false;
    }

    // 🔥 序列化消息
    const req = {
      message,
      handle: sendHandle
    };

    // 🔥 发送到 IPC 通道
    return channel.writeQueuedMessage(req, callback);
  };

  // 🔥 disconnect() 断开连接
  target.disconnect = function() {
    if (!this.connected) {
      return;
    }

    this.connected = false;
    
    // 🔥 关闭 IPC 通道
    if (channel) {
      channel.close();
      channel = null;
    }
    
    this.emit('disconnect');
  };

  // 🔥 接收消息
  channel.onread = (nread, pool, recvHandle) => {
    if (nread < 0) {
      return;
    }

    // 🔥 反序列化消息
    const message = JSON.parse(pool.toString('utf8', 0, nread));

    // 🔥 触发 message 事件
    target.emit('message', message, recvHandle);
  };
}
```

```javascript
function exec(command, options, callback) {
  const opts = normalizeExecArgs(command, options, callback);
  
  // 🔥 使用 shell 执行命令
  return module.exports.execFile(opts.file, opts.options, opts.callback);

function normalizeExecArgs(command, options, callback) {
  if (typeof options === 'function') {
    callback = options;
    options = undefined;
  }

  options = { ...options };

  // 🔥 确定 shell
  options.shell = typeof options.shell === 'string' ? options.shell : true;

  return {
    file: command,
    options: options,
    callback: callback
```

### 2. execFile() - 缓冲输出

```javascript
// lib/child_process.js
function execFile(file, args, options, callback) {
  // 参数归一化
  if (args != null && typeof args === 'object' && !Array.isArray(args)) {
    callback = options;
    options = args;
    args = null;
  }

  if (args == null) {
    args = [];
  }

  options = { ...options };
  callback = callback || function() {};

  // 🔥 spawn 子进程
  const child = spawn(file, args, {
    cwd: options.cwd,
    env: options.env,
    gid: options.gid,
    uid: options.uid,
    shell: options.shell,
    windowsHide: options.windowsHide,
    windowsVerbatimArguments: options.windowsVerbatimArguments
  });

  // 🔥 缓冲输出
  let stdout = '';
  let stderr = '';
  let stdoutLen = 0;
  let stderrLen = 0;
  const maxBuffer = options.maxBuffer || 1024 * 1024;

  if (child.stdout) {
    child.stdout.on('data', (data) => {
      stdoutLen += data.length;

      if (stdoutLen > maxBuffer) {
        // 🔥 超过缓冲区限制
        child.kill();
        callback(new Error('maxBuffer exceeded'));
      } else {
        stdout += data;
      }
    });
  }

  if (child.stderr) {
    child.stderr.on('data', (data) => {
      stderrLen += data.length;

      if (stderrLen > maxBuffer) {
        child.kill();
        callback(new Error('maxBuffer exceeded'));
      } else {
        stderr += data;
      }
    });
  }

  // 🔥 等待子进程退出
  child.on('close', (code, signal) => {
    if (code !== 0) {
      const err = new Error(`Command failed: ${file}`);
      err.code = code;
      err.signal = signal;
      err.stdout = stdout;
      err.stderr = stderr;
      callback(err);
    } else {
      callback(null, stdout, stderr);
    }
  });

  child.on('error', callback);

  return child;
}
```

## 📝 总结

### 核心要点

1. **spawn**: 最底层，返回 Stream，适合大量数据
2. **fork**: spawn 特例，启动 Node.js，内置 IPC
3. **exec**: 创建 shell，缓冲输出，适合简单命令
4. **execFile**: 不创建 shell，缓冲输出，更高效

### 实现关键

1. **uv_spawn**: libuv 进程创建
2. **fork + execvp**: Unix 进程模型
3. **IPC 通道**: Unix Domain Socket / Named Pipe
4. **stdio 管道**: pipe() + dup2()

### 应用价值

1. **多进程架构**：充分利用多核CPU
2. **任务隔离**：独立进程空间
3. **IPC 通信**：进程间协作
4. **系统集成**：调用外部命令
