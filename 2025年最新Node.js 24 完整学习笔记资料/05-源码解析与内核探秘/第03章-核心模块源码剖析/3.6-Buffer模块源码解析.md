# 3.6 Buffer 模块：内存池分配策略与 TypedArray

> 深入理解 Node.js 二进制数据处理与内存管理

---

## 📋 核心概念

Buffer 是 Node.js 中处理**二进制数据**的核心类，基于 V8 的 **TypedArray** 实现。

**核心问题：**
- Buffer 如何管理内存？
- 内存池（poolSize = 8KB）如何工作？
- `Buffer.alloc()` vs `Buffer.allocUnsafe()` 的区别？
- Buffer 与 TypedArray 的关系？

**设计原则：**
```
性能优先：内存池复用，避免频繁分配
安全可控：alloc 清零，allocUnsafe 不清零
TypedArray：共享ArrayBuffer，零拷贝
```

## 💻 内存池分配策略

### 1. 内存池设计

```javascript
// lib/buffer.js
// 🔥 内存池大小：8KB
Buffer.poolSize = 8 * 1024;

// 🔥 当前内存池
let poolSize, poolOffset, allocPool;

// 🔥 创建新内存池
function createPool() {
  poolSize = Buffer.poolSize;
  
  // 🔥 分配 8KB 内存
  allocPool = createUnsafeBuffer(poolSize);
  
  poolOffset = 0;
}

createPool();  // 初始化
```

---

### 2. Buffer.allocUnsafe() - 快速分配

```javascript
// lib/buffer.js
Buffer.allocUnsafe = function(size) {
  assertSize(size);
  
  // 🔥 分配 Buffer
  return allocate(size);
};

function allocate(size) {
  // 🔥 大对象（≥ 4KB）：直接分配
  if (size >= (Buffer.poolSize >>> 1)) {
    return createUnsafeBuffer(size);
  }

  // 🔥 小对象（< 4KB）：从内存池分配
  if (size < 0) {
    size = 0;
  }

  // 🔥 内存池空间不足
  if (poolSize - poolOffset < size) {
    createPool();  // 创建新内存池
  }

  // 🔥 从内存池切片
  const b = new FastBuffer(allocPool, poolOffset, size);
  poolOffset += size;
  
  // 🔥 对齐到 8 字节
  alignPool();
  
  return b;
}

// 🔥 内存对齐
function alignPool() {
  // 对齐到 8 字节边界
  if (poolOffset & 7) {
    poolOffset |= 7;
    poolOffset++;
  }
}
```

**分配策略可视化：**
```
poolSize = 8192 bytes

┌─────────────────────────────┐
│ allocPool (8KB)              │
├─────────┬───────┬───────────┤
│ buf1    │ buf2  │ 剩余空间   │
│ 100B    │ 200B  │ 7892B     │
└─────────┴───────┴───────────┘
    ↑       ↑
    |       poolOffset = 300

大对象（≥ 4KB）:
  直接分配独立内存

小对象（< 4KB）:
  从 allocPool 切片
```

### 3. Buffer.alloc() - 安全分配

```javascript
// lib/buffer.js
Buffer.alloc = function(size, fill, encoding) {
  assertSize(size);

  // 🔥 分配并清零
  if (size > 0 && fill !== undefined) {
    // 🔥 填充指定值
    if (encoding !== undefined && typeof encoding !== 'string') {
      throw new TypeError('encoding must be a string');
    }
    return createUnsafeBuffer(size).fill(fill, encoding);
  }

  // 🔥 分配并清零为 0
  return new FastBuffer(size);
};
```

### 4. Buffer.from() - 从数据创建

```javascript
// lib/buffer.js
Buffer.from = function(value, encodingOrOffset, length) {
  // 🔥 从字符串创建
  if (typeof value === 'string') {
    return fromString(value, encodingOrOffset);
  }

  // 🔥 从数组创建
  if (Array.isArray(value)) {
    return fromArrayLike(value);
  }

  // 🔥 从 ArrayBuffer 创建
  if (value instanceof ArrayBuffer ||
      SharedArrayBuffer !== undefined && value instanceof SharedArrayBuffer) {
    return fromArrayBuffer(value, encodingOrOffset, length);
  }

  // 🔥 从 Buffer 创建（拷贝）
  if (value instanceof Buffer) {
    return fromBuffer(value);
  }

  // 🔥 从类数组对象创建
  if (value != null && typeof value === 'object' && value.length !== undefined) {
    return fromArrayLike(value);
  }

  throw new TypeError('...');
};

// 🔥 从字符串创建
function fromString(string, encoding) {
  if (typeof encoding !== 'string' || encoding === '') {
    encoding = 'utf8';
  }

  const length = byteLengthUtf8(string);

  if (length >= (Buffer.poolSize >>> 1)) {
    // 🔥 大字符串：直接分配
    return binding.createFromString(string, encoding);
  }

  // 🔥 小字符串：从内存池分配
  if (length > (poolSize - poolOffset)) {
    createPool();
  }

  const b = new FastBuffer(allocPool, poolOffset, length);
  const actual = b.write(string, encoding);

  if (actual !== length) {
    // byteLength() 可能不准确，调整
    return new FastBuffer(allocPool, poolOffset, actual);
  }

  poolOffset += length;
  alignPool();
  
  return b;
}

// 🔥 从数组创建
function fromArrayLike(array) {
  const length = array.length;
  const buf = Buffer.allocUnsafe(length);

  for (let i = 0; i < length; i++) {
    buf[i] = array[i];
  }

  return buf;
}
```

## 💻 C++ Binding 实现

### 1. FastBuffer 类

```cpp
// src/node_buffer.cc
namespace node {
namespace Buffer {

using v8::ArrayBuffer;
using v8::BackingStore;
using v8::Local;
using v8::Uint8Array;

// 🔥 创建 Buffer
void New(const FunctionCallbackInfo<Value>& args) {
  Environment* env = Environment::GetCurrent(args);
  
  // 🔥 获取大小
  size_t length;
  if (!args[0]->IsUndefined() && !args[0]->IsNull()) {
    length = args[0]->IntegerValue(env->context()).ToChecked();
  }

  // 🔥 分配内存
  char* data = Allocate(env, length);

  // 🔥 创建 ArrayBuffer
  std::unique_ptr<BackingStore> store = ArrayBuffer::NewBackingStore(
    data, length, [](void* data, size_t length, void* deleter_data) {
      // 🔥 释放内存
      free(data);
    }, nullptr);

  Local<ArrayBuffer> ab = ArrayBuffer::New(env->isolate(), std::move(store));

  // 🔥 创建 Uint8Array
  Local<Uint8Array> ui = Uint8Array::New(ab, 0, length);
  
  args.GetReturnValue().Set(ui);
}

// 🔥 分配内存
char* Allocate(Environment* env, size_t length) {
  // 🔥 使用 malloc 分配
  char* data = static_cast<char*>(malloc(length));
  
  if (data == nullptr) {
    // 内存不足
    env->ThrowError("Cannot allocate memory");
    return nullptr;
  }

  return data;
}

} // namespace Buffer
} // namespace node
```

### 2. 字符串编码

```cpp
// src/node_buffer.cc
MaybeLocal<Object> New(Isolate* isolate,
                       Local<String> string,
                       enum encoding enc) {
  EscapableHandleScope scope(isolate);

  // 🔥 计算字节长度
  size_t length;
  switch (enc) {
    case UTF8:
      length = string->Utf8Length(isolate);
      break;
    case ASCII:
    case LATIN1:
      length = string->Length();
      break;
    case BASE64:
      // Base64 解码后长度
      length = base64_decoded_size(string->Length());
      break;
    // ...
  }

  // 🔥 分配内存
  char* data = Allocate(env, length);

  // 🔥 写入字符串
  switch (enc) {
    case UTF8:
      string->WriteUtf8(isolate, data, length, nullptr, String::NO_NULL_TERMINATION);
      break;
    case ASCII:
      string->WriteOneByte(isolate, reinterpret_cast<uint8_t*>(data), 0, length);
      break;
    // ...
  }

  // 🔥 返回 Buffer
  Local<Object> buf;
  if (New(isolate, data, length).ToLocal(&buf)) {
    return scope.Escape(buf);
  }

  return MaybeLocal<Object>();
}
```

### 2. toString() 编码转换

```javascript
// lib/buffer.js
Buffer.prototype.toString = function(encoding, start, end) {
  if (arguments.length === 0) {
    return this.utf8Slice(0, this.length);
  }

  encoding = String(encoding || 'utf8').toLowerCase();

  if (start === undefined || start < 0) {
    start = 0;
  }

  if (end === undefined || end > this.length) {
    end = this.length;
  }

  if (end <= start) {
    return '';
  }

  // 🔥 根据编码调用不同方法
  switch (encoding) {
    case 'hex':
      return this.hexSlice(start, end);
    
    case 'utf8':
    case 'utf-8':
      return this.utf8Slice(start, end);
    
    case 'ascii':
      return this.asciiSlice(start, end);
    
    case 'latin1':
    case 'binary':
      return this.latin1Slice(start, end);
    
    case 'base64':
      return this.base64Slice(start, end);
    
    case 'ucs2':
    case 'ucs-2':
    case 'utf16le':
    case 'utf-16le':
      return this.ucs2Slice(start, end);
    
    default:
      throw new TypeError('Unknown encoding: ' + encoding);
  }
};
```

### 3. slice() 切片（零拷贝）

```javascript
// lib/buffer.js
Buffer.prototype.slice = function(start, end) {
  const length = this.length;

  // 🔥 参数归一化
  start = start >>> 0;
  end = end === undefined ? length : end >>> 0;

  if (start > end || end > length) {
    throw new RangeError('...');
  }

  // 🔥 创建新视图（共享内存）
  return new FastBuffer(this.buffer, this.byteOffset + start, end - start);
};

// 使用示例
const buf1 = Buffer.from('hello world');
const buf2 = buf1.slice(0, 5);  // 🔥 零拷贝！

console.log(buf2.toString());  // 'hello'

// 🔥 修改 buf2 会影响 buf1
buf2[0] = 72;  // 'H'
console.log(buf1.toString());  // 'Hello world'
```

### 4. copy() 拷贝

```javascript
// lib/buffer.js
Buffer.prototype.copy = function(target, targetStart, sourceStart, sourceEnd) {
  // 🔥 参数归一化
  if (targetStart === undefined) {
    targetStart = 0;
  }

  if (sourceStart === undefined || sourceStart < 0) {
    sourceStart = 0;
  }

  if (sourceEnd === undefined || sourceEnd > this.length) {
    sourceEnd = this.length;
  }

  // 🔥 调用底层拷贝
  return binding.copy(this, target, targetStart, sourceStart, sourceEnd);
};

// C++ 实现
void Copy(const FunctionCallbackInfo<Value>& args) {
  // 🔥 获取源和目标
  char* source = node::Buffer::Data(args[0]);
  char* target = node::Buffer::Data(args[1]);

  size_t target_start = args[2]->Uint32Value(...);
  size_t source_start = args[3]->Uint32Value(...);
  size_t source_end = args[4]->Uint32Value(...);

  // 🔥 内存拷贝
  memmove(target + target_start, source + source_start, source_end - source_start);

  args.GetReturnValue().Set(source_end - source_start);
}
```

```javascript
  const buf = Buffer.alloc(100);
```

### 2. 使用 allocUnsafe

```javascript
// ❌ alloc 会清零（慢）
const buf1 = Buffer.alloc(1024 * 1024);  // 1MB

// ✅ allocUnsafe 不清零（快）
const buf2 = Buffer.allocUnsafe(1024 * 1024);

```

---

### 3. 避免字符串拼接

```javascript
// ❌ 字符串拼接（慢）
let str = '';
for (let i = 0; i < 1000; i++) {
  str += 'data';
}

// ✅ Buffer 拼接（快）
const buffers = [];
for (let i = 0; i < 1000; i++) {
  buffers.push(Buffer.from('data'));
```

## 📝 总结

### 核心要点

1. **内存池**：8KB poolSize，小对象复用
2. **分配策略**：< 4KB 池化，≥ 4KB 直接分配
3. **安全性**：alloc 清零，allocUnsafe 不清零
4. **TypedArray**：共享 ArrayBuffer，零拷贝

### 实现关键

1. **allocPool**: 8KB 内存池
2. **poolOffset**: 当前分配位置
3. **FastBuffer**: Uint8Array 子类
4. **binding**: C++ 内存操作

### 应用价值

1. **性能优化**：理解内存分配策略
2. **安全使用**：alloc vs allocUnsafe
3. **内存管理**：避免泄漏
4. **二进制处理**：文件、网络、加密
