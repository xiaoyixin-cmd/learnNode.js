# 3.5 events 模块：EventEmitter 监听器数组与内存管理

> 深入理解 Node.js 事件驱动架构的基石

## 📋 核心概念

EventEmitter 是 Node.js **事件驱动模型的核心**，几乎所有异步 API 都继承自它。

**核心问题：**
- 监听器如何存储？
- `once()` 如何实现？
- 内存泄漏如何检测？
- `error` 事件为何特殊？

**设计特点：**
```
简单高效：基于 JavaScript 对象和数组
内存优化：单个监听器直接存储，多个使用数组
安全机制：内存泄漏检测、错误处理
```

### 2. on() 添加监听器

```javascript
// lib/events.js
EventEmitter.prototype.on = function(type, listener) {
  return _addListener(this, type, listener, false);
};

// 🔥 核心添加逻辑
function _addListener(target, type, listener, prepend) {
  let m;
  let events;
  let existing;

  // 🔥 检查 listener 是否为函数
  if (typeof listener !== 'function') {
    throw new TypeError('The "listener" argument must be of type Function');
  }

  events = target._events;
  
  // 🔥 初始化 _events
  if (events === undefined) {
    events = target._events = Object.create(null);
    target._eventsCount = 0;
  } else {
    // 🔥 检查是否有 newListener 监听器
    if (events.newListener !== undefined) {
      target.emit('newListener', type, listener.listener || listener);
      events = target._events;
    }
    
    existing = events[type];
  }

  // 🔥 第一个监听器：直接存储函数
  if (existing === undefined) {
    events[type] = listener;
    ++target._eventsCount;
  } 
  // 🔥 第二个监听器：转为数组
  else {
    if (typeof existing === 'function') {
      existing = events[type] = prepend ? [listener, existing] : [existing, listener];
    } 
    // 🔥 已是数组：添加到数组
    else {
      if (prepend) {
        existing.unshift(listener);
      } else {
        existing.push(listener);
      }
    }

    // 🔥 检查内存泄漏
    m = _getMaxListeners(target);
    if (m > 0 && existing.length > m && !existing.warned) {
      existing.warned = true;
      
      const w = new Error('Possible EventEmitter memory leak detected. ' +
                          `${existing.length} ${String(type)} listeners added. ` +
                          'Use emitter.setMaxListeners() to increase limit');
      w.name = 'MaxListenersExceededWarning';
      w.emitter = target;
      w.type = type;
      w.count = existing.length;
      
      // 🔥 触发警告
      process.emitWarning(w);
    }
  }

  return target;
};
```

**监听器存储结构：**
```javascript
{

    'event1': Function,              // 单个监听器
    'event2': [fn1, fn2, fn3],       // 多个监听器（数组）
    'error': Function
  },
  _eventsCount: 2,
  _maxListeners: 10
}
```

### 3. emit() 触发事件

```javascript
// lib/events.js
EventEmitter.prototype.emit = function(type, ...args) {
  const events = this._events;

  // 🔥 特殊处理 error 事件
  if (type === 'error') {
    if (events === undefined || events.error === undefined) {
      const err = args[0];
      
      // 🔥 没有 error 监听器，抛出错误
      if (err instanceof Error) {
        throw err;
      }
      
      const error = new Error('Unhandled error.' + (err ? ' (' + err + ')' : ''));
      error.context = err;
      throw error;
    }
  }

  const handler = events[type];

  // 🔥 没有监听器
  if (handler === undefined) {
    return false;
  }

  // 🔥 单个监听器
  if (typeof handler === 'function') {
    ReflectApply(handler, this, args);
  } 
  // 🔥 多个监听器（数组）
  else {
    const len = handler.length;
    const listeners = arrayClone(handler, len);
    
    for (let i = 0; i < len; ++i) {
      ReflectApply(listeners[i], this, args);
    }
  }

  return true;
};

// 🔥 优化：复制数组避免在迭代中修改
function arrayClone(arr, n) {
  const copy = new Array(n);
  for (let i = 0; i < n; ++i) {
    copy[i] = arr[i];
  }
  return copy;
}
```

### 4. once() 一次性监听

```javascript
// lib/events.js
EventEmitter.prototype.once = function(type, listener) {
  if (typeof listener !== 'function') {
    throw new TypeError('The "listener" argument must be of type Function');
  }

  // 🔥 包装监听器
  this.on(type, _onceWrap(this, type, listener));
  
  return this;
};

// 🔥 once 包装器
function _onceWrap(target, type, listener) {
  const state = { fired: false, wrapFn: undefined, target, type, listener };
  
  // 🔥 包装函数
  const wrapped = onceWrapper.bind(state);
  wrapped.listener = listener;
  state.wrapFn = wrapped;
  
  return wrapped;
}

// 🔥 执行一次后自动移除
function onceWrapper(...args) {
  if (!this.fired) {
    this.target.removeListener(this.type, this.wrapFn);
    this.fired = true;
    
    // 🔥 调用原始监听器
    return ReflectApply(this.listener, this.target, args);
  }
}
```

### 5. off() / removeListener() 移除监听器

```javascript
// lib/events.js
EventEmitter.prototype.off = EventEmitter.prototype.removeListener;

EventEmitter.prototype.removeListener = function(type, listener) {
  let list, events, position, i, originalListener;

  if (typeof listener !== 'function') {
    throw new TypeError('The "listener" argument must be of type Function');
  }

  events = this._events;
  if (events === undefined) {
    return this;
  }

  list = events[type];
  if (list === undefined) {
    return this;
  }

  // 🔥 单个监听器
  if (list === listener || list.listener === listener) {
    if (--this._eventsCount === 0) {
      this._events = Object.create(null);
    } else {
      delete events[type];
      
      // 🔥 触发 removeListener 事件
      if (events.removeListener) {
        this.emit('removeListener', type, list.listener || listener);
      }
    }
  } 
  // 🔥 多个监听器（数组）
  else if (typeof list !== 'function') {
    position = -1;

    // 🔥 查找监听器位置
    for (i = list.length - 1; i >= 0; i--) {
      if (list[i] === listener || list[i].listener === listener) {
        originalListener = list[i].listener;
        position = i;
        break;
      }
    }

    if (position < 0) {
      return this;
    }

    // 🔥 移除监听器
    if (position === 0) {
      list.shift();
    } else {
      spliceOne(list, position);
    }

    // 🔥 只剩一个，转为函数
    if (list.length === 1) {
      events[type] = list[0];
    }

    // 🔥 触发 removeListener 事件
    if (events.removeListener !== undefined) {
      this.emit('removeListener', type, originalListener || listener);
    }
  }

  return this;
};

// 🔥 优化：删除数组元素
function spliceOne(list, index) {
  for (; index + 1 < list.length; index++) {
    list[index] = list[index + 1];
  }
  list.pop();
}
```

### 6. removeAllListeners() 移除所有监听器

```javascript
// lib/events.js
EventEmitter.prototype.removeAllListeners = function(type) {
  const events = this._events;

  if (events === undefined) {
    return this;
  }

  // 🔥 移除所有类型的监听器
  if (type === undefined) {
    for (const key of Object.keys(events)) {
      if (key === 'removeListener') continue;
      this.removeAllListeners(key);
    }
    
    this.removeAllListeners('removeListener');
    
    this._events = Object.create(null);
    this._eventsCount = 0;
    
    return this;
  }

  // 🔥 移除指定类型的监听器
  const listeners = events[type];

  if (typeof listeners === 'function') {
    this.removeListener(type, listeners);
  } else if (listeners !== undefined) {
    // 🔥 从后往前移除（避免数组索引问题）
    for (let i = listeners.length - 1; i >= 0; i--) {
      this.removeListener(type, listeners[i]);
    }
  }

  return this;
};
```

## 💻 内存泄漏检测

### 1. 最大监听器限制

```javascript
// lib/events.js
EventEmitter.defaultMaxListeners = 10;

EventEmitter.prototype.setMaxListeners = function(n) {
  if (typeof n !== 'number' || n < 0 || Number.isNaN(n)) {
    throw new TypeError('n must be a non-negative number');
  }
  
  this._maxListeners = n;
  return this;
};

function _getMaxListeners(that) {
  if (that._maxListeners === undefined) {
    return EventEmitter.defaultMaxListeners;
  }
  return that._maxListeners;
}
```

### 2. rawListeners() 获取原始监听器

```javascript
// lib/events.js
EventEmitter.prototype.rawListeners = function(type) {
  const events = this._events;

  if (events === undefined) {
    return [];
  }

  const evlistener = events[type];
  
  // 🔥 单个监听器
  if (typeof evlistener === 'function') {
    return [evlistener];
  }
  
  // 🔥 多个监听器
  if (evlistener !== undefined) {
    return arrayClone(evlistener, evlistener.length);
  }
  
  return [];
};

// 使用示例
const emitter = new EventEmitter();

const handler1 = () => {};
const handler2 = () => {};

emitter.on('data', handler1);
emitter.once('data', handler2);

console.log(emitter.rawListeners('data'));
// [ [Function: handler1], [Function: onceWrapper] ]
```

```javascript
EventEmitter.prototype.listenerCount = function(type) {
```

### 3. 避免内存泄漏

```javascript
// ❌ 忘记移除监听器
class MyClass {
  constructor() {
    this.emitter = new EventEmitter();
    
    // 🔥 监听器持有 this 引用
    this.emitter.on('data', this.handleData.bind(this));
  }
  
  handleData(data) {
    console.log(data);
  }
  
  // ❌ 没有清理监听器
}

// ✅ 正确清理
class MyClass {
  constructor() {
    this.emitter = new EventEmitter();
    this.boundHandler = this.handleData.bind(this);
    
    this.emitter.on('data', this.boundHandler);
  }
  
  handleData(data) {
    console.log(data);
  }
  
  // ✅ 清理监听器
  cleanup() {
    this.emitter.removeListener('data', this.boundHandler);
  }
}
```

```javascript
const { EventEmitter } = require('events');

class MyEmitter extends EventEmitter {
```

### 2. 异步事件处理

```javascript
// async-emitter.js
const { EventEmitter } = require('events');
const { promisify } = require('util');

class AsyncEmitter extends EventEmitter {
  async emitAsync(event, ...args) {
    const listeners = this.listeners(event);
    
    // 🔥 串行执行所有监听器
    for (const listener of listeners) {
      await listener(...args);
    }
  }

  async emitParallel(event, ...args) {
    const listeners = this.listeners(event);
    
    // 🔥 并行执行所有监听器
    await Promise.all(
      listeners.map(listener => listener(...args))
    );
  }
}

// 使用
const emitter = new AsyncEmitter();

emitter.on('data', async (data) => {
  await new Promise(resolve => setTimeout(resolve, 100));
  console.log('Listener 1:', data);
});

emitter.on('data', async (data) => {
  await new Promise(resolve => setTimeout(resolve, 50));
  console.log('Listener 2:', data);
});

// 串行执行
await emitter.emitAsync('data', 'test');
// Listener 1: test（100ms后）
// Listener 2: test（再50ms后）

// 并行执行
await emitter.emitParallel('data', 'test');
// Listener 2: test（50ms后）
// Listener 1: test（同时100ms后）
```

### 3. 事件命名空间

```javascript
// namespaced-emitter.js
const { EventEmitter } = require('events');

class NamespacedEmitter extends EventEmitter {
  onNamespace(namespace, event, listener) {
    // 🔥 使用命名空间前缀
    return this.on(`${namespace}:${event}`, listener);
  }

  emitNamespace(namespace, event, ...args) {
    // 🔥 触发命名空间事件
    return this.emit(`${namespace}:${event}`, ...args);
  }

  removeNamespace(namespace) {
    // 🔥 移除整个命名空间的监听器
    const events = this.eventNames();
    
    events.forEach(event => {
      if (event.startsWith(`${namespace}:`)) {
        this.removeAllListeners(event);
      }
    });
  }
}

// 使用
const emitter = new NamespacedEmitter();

emitter.onNamespace('user', 'login', (data) => {
  console.log('User logged in:', data);
});

emitter.onNamespace('user', 'logout', (data) => {
  console.log('User logged out:', data);
});

emitter.emitNamespace('user', 'login', { id: 1, name: 'John' });
// 输出：User logged in: { id: 1, name: 'John' }

emitter.removeNamespace('user');  // 移除所有 user 事件
```
