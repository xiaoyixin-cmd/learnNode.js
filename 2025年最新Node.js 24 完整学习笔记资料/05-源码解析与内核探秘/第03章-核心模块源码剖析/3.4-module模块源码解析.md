# 3.4 module 模块：CommonJS 加载全过程

> 深入理解 `require()` 的完整执行流程与模块缓存机制

---

## 📋 核心概念

module 模块实现了 Node.js 的 **CommonJS 模块系统**，是理解模块加载的关键。

**核心问题：**
- `require()` 如何查找模块？
- 模块如何被编译执行？
- 循环依赖如何处理？
- 模块缓存如何工作？

**执行流程：**
```
require('module')
  ↓
Module._load()        // 加载模块
  ↓
Module._resolveFilename()  // 解析文件路径
  ↓
Module._cache 查找     // 检查缓存
  ↓
new Module()          // 创建模块实例
  ↓
Module._compile()     // 编译执行
  ↓
runInThisContext()    // V8 执行
```

### 2. Module._load() 核心加载逻辑

```javascript
// lib/internal/modules/cjs/loader.js
Module._load = function(request, parent, isMain) {
  let relResolveCacheIdentifier;

  if (parent) {
    // 🔥 计算缓存key
    relResolveCacheIdentifier = `${parent.path}\x00${request}`;
    
    // 🔥 检查相对路径解析缓存
    const filename = relativeResolveCache[relResolveCacheIdentifier];
    
    if (filename !== undefined) {
      // 🔥 从模块缓存加载
      const cachedModule = Module._cache[filename];
      if (cachedModule !== undefined) {
        updateChildren(parent, cachedModule, true);
        return cachedModule.exports;
      }
      
      // 🔥 直接使用缓存的文件名
      delete relativeResolveCache[relResolveCacheIdentifier];
    }
  }

  // 🔥 解析文件路径
  const filename = Module._resolveFilename(request, parent, isMain);

  // 🔥 检查模块缓存
  const cachedModule = Module._cache[filename];
  if (cachedModule !== undefined) {
    // 🔥 更新父子关系
    updateChildren(parent, cachedModule, true);
    
    return cachedModule.exports;
  }

  // 🔥 检查原生模块
  const mod = loadNativeModule(filename, request);
  if (mod && mod.canBeRequiredByUsers) {
    return mod.exports;
  }

  // 🔥 创建新模块
  const module = cachedModule || new Module(filename, parent);

  // 🔥 标记为main模块
  if (isMain) {
    process.mainModule = module;
    module.id = '.';
  }

  // 🔥 加入缓存（循环依赖需要提前缓存）
  Module._cache[filename] = module;

  // 🔥 更新父子关系
  if (parent !== undefined) {
    relativeResolveCache[relResolveCacheIdentifier] = filename;
  }

  // 🔥 加载模块
  let threw = true;
  try {
    module.load(filename);
    threw = false;
  } finally {
    if (threw) {
      // 🔥 加载失败，删除缓存
      delete Module._cache[filename];
      if (parent !== undefined) {
        delete relativeResolveCache[relResolveCacheIdentifier];
      }
    }
  }

  return module.exports;
};
```

### 3. Module._resolveFilename() 路径解析

```javascript
// lib/internal/modules/cjs/loader.js
Module._resolveFilename = function(request, parent, isMain, options) {
  // 🔥 原生模块优先
  if (StringPrototypeStartsWith(request, 'node:')) {
    const id = StringPrototypeSlice(request, 5);
    const module = loadNativeModule(id, request);
    
    if (module && module.canBeRequiredByUsers) {
      return request;
    }
    
    throw new Error(`Cannot find module '${request}'`);
  }

  // 🔥 检查 package.json 的 exports 字段
  if (request.startsWith('.') ||
      request.startsWith('/') ||
      request.startsWith('\\')) {
    // 相对/绝对路径
  } else {
    // 🔥 node_modules 查找
    return Module._findPath(request, Module._nodeModulePaths(parent.filename));
  }

  // 🔥 常规路径解析
  let filename;
  const paths = Module._resolveLookupPaths(request, parent);
  
  filename = Module._findPath(request, paths, isMain);
  
  if (!filename) {
    throw new Error(`Cannot find module '${request}'`);
  }
  
  return filename;
};
```

### 4. Module._findPath() 查找逻辑

```javascript
// lib/internal/modules/cjs/loader.js
Module._findPath = function(request, paths, isMain) {
  const absoluteRequest = path.isAbsolute(request);
  
  // 🔥 绝对路径
  if (absoluteRequest) {
    paths = [''];
  } else if (!paths || paths.length === 0) {
    return false;
  }

  // 🔥 缓存key
  const cacheKey = request + '\x00' + (paths.length === 1 ? paths[0] : paths.join('\x00'));
  const entry = Module._pathCache[cacheKey];
  
  if (entry) {
    return entry;
  }

  let exts;
  const trailingSlash = request.length > 0 && request.charCodeAt(request.length - 1) === CHAR_FORWARD_SLASH;

  // 🔥 遍历所有路径
  for (let i = 0; i < paths.length; i++) {
    const curPath = paths[i];
    
    if (curPath && stat(curPath) < 1) continue;

    // 🔥 拼接路径
    const basePath = path.resolve(curPath, request);

    let filename;

    // 🔥 尝试直接作为文件
    const rc = stat(basePath);
    if (!trailingSlash) {
      if (rc === 0) {  // 文件存在
        if (!isMain) {
          filename = basePath;
        }
      }

      if (!filename) {
        // 🔥 尝试添加扩展名 .js, .json, .node
        exts = Object.keys(Module._extensions);
        filename = tryExtensions(basePath, exts, isMain);
      }
    }

    // 🔥 尝试作为目录（查找 index.js）
    if (!filename && rc === 1) {  // 目录存在
      exts = Object.keys(Module._extensions);
      filename = tryPackage(basePath, exts, isMain, request);
    }

    if (filename) {
      // 🔥 缓存结果
      Module._pathCache[cacheKey] = filename;
      return filename;
    }
  }

  return false;
};

// 🔥 尝试不同扩展名
function tryExtensions(path, exts, isMain) {
  for (let i = 0; i < exts.length; i++) {
    const filename = tryFile(path + exts[i], isMain);
    
    if (filename) {
      return filename;
    }
  }
  
  return false;
}

// 🔥 尝试 package.json
function tryPackage(requestPath, exts, isMain, originalPath) {
  // 🔥 读取 package.json
  const pkg = readPackage(requestPath, isMain, originalPath);

  if (!pkg) return tryExtensions(path.resolve(requestPath, 'index'), exts, isMain);

  // 🔥 使用 main 字段
  const filename = path.resolve(requestPath, pkg.main);
  
  let actual = tryFile(filename, isMain) ||
               tryExtensions(filename, exts, isMain) ||
               tryExtensions(path.resolve(filename, 'index'), exts, isMain);

  if (actual === false) {
    // 🔥 fallback 到 index.js
    actual = tryExtensions(path.resolve(requestPath, 'index'), exts, isMain);
  }

  return actual;
}
```

### 5. Module.load() 加载文件

```javascript
// lib/internal/modules/cjs/loader.js
Module.prototype.load = function(filename) {
  assert(!this.loaded);
  this.filename = filename;
  
  // 🔥 设置 module.paths（node_modules 查找路径）
  this.paths = Module._nodeModulePaths(path.dirname(filename));

  // 🔥 根据扩展名选择加载器
  const extension = findLongestRegisteredExtension(filename);
  
  if (StringPrototypeEndsWith(filename, '.mjs')) {
    throw new Error('Cannot load ES Module from require()');
  }

  // 🔥 调用对应的加载器
  Module._extensions[extension](this, filename);
  
  this.loaded = true;
};
```

### 7. Module._compile() 核心编译

```javascript
// lib/internal/modules/cjs/loader.js
Module.prototype._compile = function(content, filename) {
  // 🔥 包装代码
  const wrapper = Module.wrap(content);

  // 🔥 编译成函数
  const compiledWrapper = vm.runInThisContext(wrapper, {
    filename,
    lineOffset: 0,
    displayErrors: true,
    importModuleDynamically: async (specifier, script) => {
      const loader = asyncESM.ESMLoader;
      return loader.import(specifier, normalizeReferrerURL(filename));
    },
  });

  // 🔥 准备参数
  const dirname = path.dirname(filename);
  const require = makeRequireFunction(this, redirects);
  let result;
  
  const exports = this.exports;
  const thisValue = exports;
  const module = this;

  // 🔥 执行编译后的函数
  if (requireDepth === 0) statCache = new SafeMap();
  
  result = ReflectApply(compiledWrapper, thisValue, [
    exports,      // exports
    require,      // require
    module,       // module
    filename,     // __filename
    dirname       // __dirname
  ]);

  if (requireDepth === 0) statCache = null;
  
  return result;
};
```

## 💻 循环依赖处理

### 1. 循环依赖场景

```javascript
// a.js
console.log('a starting');
exports.done = false;

const b = require('./b.js');  // 🔥 加载 b
console.log('in a, b.done =', b.done);

exports.done = true;
console.log('a done');

// b.js
console.log('b starting');
exports.done = false;

const a = require('./a.js');  // 🔥 循环依赖！
console.log('in b, a.done =', a.done);

exports.done = true;
console.log('b done');

// main.js
console.log('main starting');
const a = require('./a.js');
const b = require('./b.js');

console.log('in main, a.done =', a.done);
console.log('in main, b.done =', b.done);
```

**执行结果：**
```
main starting
a starting
b starting
in b, a.done = false  ← a 还未完成
b done
in a, b.done = true
a done
in main, a.done = true
in main, b.done = true
```

---

### 2. 循环依赖解决原理

```javascript
// Module._load 伪代码
Module._load = function(request, parent, isMain) {
  const filename = Module._resolveFilename(request, parent, isMain);

  // 🔥 1. 检查缓存
  const cachedModule = Module._cache[filename];
  if (cachedModule !== undefined) {
    // 🔥 直接返回已有的 exports（可能未完成）
    return cachedModule.exports;
  }

  // 🔥 2. 创建模块
  const module = new Module(filename, parent);

  // 🔥 3. 提前加入缓存（关键！）
  Module._cache[filename] = module;

  // 🔥 4. 加载模块（此时缓存中已有空的 exports）
  module.load(filename);

  // 🔥 5. 返回 exports
  return module.exports;
};
```

**关键点：**
- 模块在执行前就加入缓存
- 循环依赖时返回未完成的 exports
- 依赖方需处理未完成的 exports

## 💻 node_modules 查找算法

### 1. Module._nodeModulePaths() 生成查找路径

```javascript
// lib/internal/modules/cjs/loader.js
Module._nodeModulePaths = function(from) {
  // 🔥 从当前目录向上查找
  // from: '/home/user/project/lib/utils'
  
  const paths = [];
  let current = from;
  
  while (true) {
    const basename = path.basename(current);
    
    // 🔥 跳过 node_modules 目录本身
    if (basename !== 'node_modules') {
      paths.push(path.resolve(current, 'node_modules'));
    }
    
    const parent = path.dirname(current);
    
    // 🔥 到达根目录
    if (parent === current) {
      break;
    }
    
    current = parent;
  }
  
  return paths;
};

// 🔥 结果示例
// [
//   '/home/user/project/lib/utils/node_modules',
//   '/home/user/project/lib/node_modules',
//   '/home/user/project/node_modules',
//   '/home/user/node_modules',
//   '/home/node_modules',
//   '/node_modules'
// ]
```

## 💻 原生模块加载

### 1. NativeModule 实现

```javascript
// lib/internal/bootstrap/loaders.js
class NativeModule {
  constructor(id) {
    this.id = id;
    this.filename = `${id}.js`;
    this.exports = {};
    this.loaded = false;
    this.loading = false;
  }

  // 🔥 编译原生模块
  compile() {
    let source = NativeModule.getSource(this.id);
    
    // 🔥 包装代码
    source = NativeModule.wrap(source);

    // 🔥 编译
    const fn = vm.runInThisContext(source, {
      filename: this.filename,
      displayErrors: true
    });

    // 🔥 执行
    const requireFn = this.makeRequireFunction();
    fn(this.exports, requireFn, this, this.filename);

    this.loaded = true;
  }

  // 🔥 加载原生模块
  static require(id) {
    if (id === 'native_module') {
      return NativeModule;
    }

    const cached = NativeModule.getCached(id);
    if (cached) {
      return cached.exports;
    }

    const nativeModule = new NativeModule(id);
    nativeModule.cache();
    nativeModule.compile();

    return nativeModule.exports;
  }
}
```

## 📝 总结

### 核心要点

1. **加载流程**：resolve → cache → compile → execute
2. **路径查找**：node_modules 逐级向上查找
3. **模块缓存**：Module._cache 避免重复加载
4. **循环依赖**：提前缓存 + 返回未完成的 exports

### 实现关键

1. **Module._load()**: 核心加载逻辑
2. **Module._resolveFilename()**: 路径解析
3. **Module._compile()**: 代码包装与执行
4. **Module.wrap()**: 函数包装器

### 应用价值

1. **理解加载**：深入模块系统原理
2. **性能优化**：合理使用缓存
3. **问题排查**：定位加载问题
