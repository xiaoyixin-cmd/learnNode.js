# 3.3 stream 模块：Readable/Writable 状态机设计

> 深入理解 Node.js 流式处理的核心机制与背压控制

---

## 📋 核心概念

Stream 是 Node.js 中处理**大数据流**的核心抽象，基于精妙的**状态机设计**。

**核心问题：**
- 为什么需要 Stream？
- flowing mode vs paused mode 的区别？
- 背压（backpressure）如何实现？
- pipe() 如何自动处理背压？

**Stream 类型：**
```
Readable  → 可读流（fs.createReadStream）
Writable  → 可写流（fs.createWriteStream）
Duplex    → 双工流（net.Socket）
Transform → 转换流（zlib.createGzip）
```

```javascript
// lib/internal/streams/readable.js
const { EventEmitter } = require('events');

class Readable extends Stream {
  constructor(options = {}) {
    super(options);

    // 🔥 创建状态对象
    this._readableState = new ReadableState(options, this);

    // 🔥 自动销毁
    if (options.destroy) {
      this._destroy = options.destroy;
    }

    // 🔥 实现 _read 方法
    if (options.read) {
      this._read = options.read;
    }
  }

  // 🔥 核心：push 数据到内部缓冲区
  push(chunk, encoding) {
    return readableAddChunk(this, chunk, encoding, false);
  }

  // 🔥 核心：从缓冲区读取数据
  read(n) {
    const state = this._readableState;
    
    // 🔥 触发 _read() 填充缓冲区
    if (state.length === 0 && !state.ended) {
      this._read(state.highWaterMark);
    }

    // 🔥 从缓冲区读取
    const data = fromList(n, state);
    
    if (data === null) {
      state.needReadable = true;
      n = 0;
    }

    state.length -= n;

    // 🔥 缓冲区低于阈值，触发 _read
    if (state.length < state.highWaterMark) {
      this._read(state.highWaterMark);
    }

    if (data !== null) {
      this.emit('data', data);
    }

    return data;
  }

  // 🔥 用户实现：如何读取底层数据
  _read(size) {
    throw new Error('_read() is not implemented');
  }

  // 🔥 切换到 flowing mode
  resume() {
    const state = this._readableState;
    
    if (!state.flowing) {
      state.flowing = true;
      resume(this, state);
    }

    return this;
  }

  // 🔥 切换到 paused mode
  pause() {
    if (this._readableState.flowing !== false) {
      this._readableState.flowing = false;
      this.emit('pause');
    }

    return this;
  }

  // 🔥 管道到 Writable
  pipe(dest, options) {
    const src = this;
    const state = this._readableState;

    // 🔥 监听 data 事件（切换到 flowing mode）
    src.on('data', ondata);
    
    function ondata(chunk) {
      const ret = dest.write(chunk);
      
      // 🔥 背压：目标写不下，暂停源
      if (ret === false) {
        src.pause();
      }
    }

    // 🔥 目标可写时，恢复源
    dest.on('drain', ondrain);
    
    function ondrain() {
      if (src.isPaused()) {
        src.resume();
      }
    }

    // 🔥 源结束时，结束目标
    src.on('end', onend);
    
    function onend() {
      dest.end();
    }

    return dest;
  }
}
```

```javascript
// lib/internal/streams/state.js
class ReadableState {
  constructor(options, stream) {
    // 🔥 核心状态
    this.flowing = null;           // null / false / true
    this.ended = false;            // 是否已结束
    this.endEmitted = false;       // 是否已触发 end 事件
    this.reading = false;          // 是否正在读取
    this.constructed = true;
    
    // 🔥 缓冲区
    this.buffer = new BufferList(); // 内部缓冲区（链表）
    this.length = 0;               // 缓冲区长度
    this.highWaterMark = options.highWaterMark || 16384; // 16KB

    // 🔥 编码
    this.encoding = options.encoding || null;
    this.decoder = null;

    // 🔥 状态标志
    this.sync = true;              // 同步模式
    this.needReadable = false;
    this.emittedReadable = false;
    this.readableListening = false;
    this.resumeScheduled = false;

    // 🔥 管道
    this.pipes = [];               // 管道目标列表
    this.multiAwaitDrain = false;

    // 🔥 错误处理
    this.errored = null;
    this.closeEmitted = false;
    this.destroyed = false;
  }
}
```

**状态转换：**
```
flowing 状态：
  null  → 初始状态（未监听 data，未调用 pipe）
  false → paused mode（调用 pause）
  true  → flowing mode（调用 resume 或监听 data）

ended 状态：
  false → 未结束
  true  → 已推送 null（end of stream）

reading 状态：
  false → 未在读取
  true  → 正在调用 _read()
```

### 1. Writable 实现

```javascript
// lib/internal/streams/writable.js
class Writable extends Stream {
  constructor(options = {}) {
    super(options);

    // 🔥 创建状态对象
    this._writableState = new WritableState(options, this);

    // 🔥 实现 _write 方法
    if (options.write) {
      this._write = options.write;
    }

    // 🔥 实现 _writev（批量写入）
    if (options.writev) {
      this._writev = options.writev;
    }

    // 🔥 实现 _final（结束前调用）
    if (options.final) {
      this._final = options.final;
    }
  }

  // 🔥 核心：写入数据
  write(chunk, encoding, callback) {
    const state = this._writableState;

    // 🔥 参数归一化
    if (typeof encoding === 'function') {
      callback = encoding;
      encoding = state.defaultEncoding;
    }

    if (typeof callback !== 'function') {
      callback = nop;
    }

    // 🔥 检查是否已结束
    if (state.ending) {
      writeAfterEnd(this, callback);
      return false;
    }

    // 🔥 写入到缓冲区
    return writeOrBuffer(this, state, chunk, encoding, callback);
  }

  // 🔥 用户实现：如何写入底层
  _write(chunk, encoding, callback) {
    throw new Error('_write() is not implemented');
  }

  // 🔥 批量写入（可选）
  _writev(chunks, callback) {
    throw new Error('_writev() is not implemented');
  }

  // 🔥 结束写入
  end(chunk, encoding, callback) {
    const state = this._writableState;

    // 🔥 参数归一化
    if (typeof chunk === 'function') {
      callback = chunk;
      chunk = null;
      encoding = null;
    } else if (typeof encoding === 'function') {
      callback = encoding;
      encoding = null;
    }

    // 🔥 写入最后的数据
    if (chunk !== null && chunk !== undefined) {
      this.write(chunk, encoding);
    }

    // 🔥 标记结束
    if (!state.ending) {
      endWritable(this, state, callback);
    }

    return this;
  }

  // 🔥 结束前调用（可选）
  _final(callback) {
    callback();
  }

  // 🔥 软木塞：暂停写入（优化）
  cork() {
    this._writableState.corked++;
  }

  // 🔥 拔出软木塞：恢复写入
  uncork() {
    const state = this._writableState;

    if (state.corked) {
      state.corked--;

      if (!state.writing && !state.corked && !state.bufferProcessing && state.bufferedRequest) {
        clearBuffer(this, state);
      }
    }
  }
}
```

### 3. 写入缓冲区处理

```javascript
// lib/internal/streams/writable.js
function writeOrBuffer(stream, state, chunk, encoding, callback) {
  const len = state.objectMode ? 1 : chunk.length;

  // 🔥 增加长度
  state.length += len;

  // 🔥 检查是否超过 highWaterMark
  const ret = state.length < state.highWaterMark;
  
  if (!ret) {
    state.needDrain = true;  // 需要 drain 事件
  }

  // 🔥 正在写入或软木塞，加入缓冲区
  if (state.writing || state.corked || state.errored || !state.constructed) {
    state.buffered.push({ chunk, encoding, callback });
    
    if (state.allBuffers && encoding !== 'buffer') {
      state.allBuffers = false;
    }
    if (state.allNoop && callback !== nop) {
      state.allNoop = false;
    }
  } else {
    // 🔥 直接写入
    state.writelen = len;
    state.writecb = callback;
    state.writing = true;
    state.sync = true;
    stream._write(chunk, encoding, state.onwrite);
    state.sync = false;
  }

  // 🔥 返回 false 表示需要背压
  return ret && !state.errored && !state.destroyed;
}

// 🔥 写入完成回调
function onWriteComplete(stream, state, err) {
  state.writing = false;
  state.writecb = null;
  state.length -= state.writelen;
  state.writelen = 0;

  if (err) {
    onwriteError(stream, state, err);
    return;
  }

  // 🔥 触发 drain 事件
  if (state.needDrain && state.length === 0 && !state.destroyed) {
    state.needDrain = false;
    stream.emit('drain');
  }

  // 🔥 继续处理缓冲区
  if (!state.ending) {
    clearBuffer(stream, state);
  } else if (state.length === 0 && !state.destroyed) {
    finishMaybe(stream, state);
  }
}

// 🔥 清空缓冲区
function clearBuffer(stream, state) {
  state.bufferProcessing = true;
  let entry = state.buffered[0];

  // 🔥 批量写入（如果支持 _writev）
  if (stream._writev && state.buffered.length > 1) {
    const chunks = state.buffered.splice(0);
    state.writing = true;
    stream._writev(chunks, state.onwrite);
    return;
  }

  // 🔥 逐个写入
  while (entry) {
    const { chunk, encoding, callback } = entry;
    const len = state.objectMode ? 1 : chunk.length;

    doWrite(stream, state, len, chunk, encoding, callback);

    entry = state.buffered.shift();
    
    // 🔥 一次只处理一个（避免阻塞）
    if (state.writing) {
      break;
    }
  }

  state.bufferProcessing = false;
}
```

## 💻 背压（Backpressure）机制

### 1. 背压原理

**场景：**
```
快速读取 → 慢速写入
  ↓
写入缓冲区堆积
  ↓
内存占用过高
  ↓
需要暂停读取（背压）
```

**实现：**
```javascript
// 快速读取文件
const readable = fs.createReadStream('input.txt', {
  highWaterMark: 64 * 1024  // 64KB
});

// 慢速写入文件
const writable = fs.createWriteStream('output.txt', {
  highWaterMark: 16 * 1024  // 16KB（小于读取）
});

// 🔥 手动处理背压
readable.on('data', (chunk) => {
  const ret = writable.write(chunk);
  
  if (!ret) {
    // 🔥 写入缓冲区满，暂停读取
    console.log('Backpressure! Pausing read...');
    readable.pause();
  }
});

// 🔥 写入缓冲区清空，恢复读取
writable.on('drain', () => {
  console.log('Drained! Resuming read...');
  readable.resume();
});

readable.on('end', () => {
  writable.end();
});
```

### 2. pipe() 自动背压

```javascript
// lib/internal/streams/readable.js
Readable.prototype.pipe = function(dest, options) {
  const src = this;
  const state = src._readableState;

  // 🔥 增加管道计数
  state.pipes.push(dest);

  // 🔥 监听 data 事件（自动切换到 flowing mode）
  src.on('data', ondata);

  function ondata(chunk) {
    const ret = dest.write(chunk);
    
    // 🔥 自动背压
    if (ret === false && src.listenerCount('data')) {
      src.pause();
    }
  }

  // 🔥 自动恢复
  dest.on('drain', ondrain);

  function ondrain() {
    if (src.listenerCount('data') && src.isPaused()) {
      src.resume();
    }
  }

  // 🔥 源结束，结束目标
  src.on('end', onend);

  function onend() {
    dest.end();
  }

  // 🔥 源错误，传播到目标
  src.on('error', onerror);

  function onerror(err) {
    dest.destroy(err);
  }

  // 🔥 清理
  src.on('close', cleanup);
  dest.on('close', cleanup);

  function cleanup() {
    src.removeListener('data', ondata);
    dest.removeListener('drain', ondrain);
    // ...
  }

  return dest;
};
```

```javascript
class Duplex extends Readable {

    // 🔥 混入 Writable
    this._writableState = new WritableState(options, this);
    
    if (options.write) {
      this._write = options.write;

    return Writable.prototype.write.call(this, chunk, encoding, callback);

  }

    // 实现读取
  }

    // 实现写入
  }
}
```

### 2. Transform（转换流）

```javascript
// lib/internal/streams/transform.js
const { Duplex } = require('internal/streams/duplex');

class Transform extends Duplex {
  constructor(options = {}) {
    super(options);

    // 🔥 Transform 特定状态
    this._transformState = {
      afterTransform: afterTransform.bind(this),
      needTransform: false,
      transforming: false,
      writecb: null,
      writechunk: null,
      writeencoding: null
    };

    // 🔥 实现 _transform 方法
    if (options.transform) {
      this._transform = options.transform;
    }

    // 🔥 实现 _flush 方法（可选）
    if (options.flush) {
      this._flush = options.flush;
    }
  }

  // 🔥 写入时自动调用 _transform
  _write(chunk, encoding, callback) {
    const ts = this._transformState;
    
    ts.writecb = callback;
    ts.writechunk = chunk;
    ts.writeencoding = encoding;

    if (!ts.transforming) {
      ts.transforming = true;
      this._transform(chunk, encoding, ts.afterTransform);
    }
  }

  // 🔥 Transform 完成
  afterTransform(err, data) {
    const ts = this._transformState;
    ts.transforming = false;

    const cb = ts.writecb;
    ts.writecb = null;

    if (data !== null && data !== undefined) {
      // 🔥 推送转换后的数据到可读端
      this.push(data);
    }

    if (cb) {
      cb(err);
    }

    // 🔥 继续处理缓冲区
    const rs = this._readableState;
    rs.reading = false;
    
    if (rs.needReadable || rs.length < rs.highWaterMark) {
      this._read(rs.highWaterMark);
    }
  }

  // 🔥 用户实现：如何转换数据
  _transform(chunk, encoding, callback) {
    throw new Error('_transform() is not implemented');
  }

  // 🔥 结束前刷新（可选）
  _flush(callback) {
    callback();
  }
}

// 🔥 应用：大写转换流
const { Transform } = require('stream');

class UpperCaseTransform extends Transform {
  _transform(chunk, encoding, callback) {
    // 🔥 转换为大写
    this.push(chunk.toString().toUpperCase());
    callback();
  }
}

// 使用
const upperCase = new UpperCaseTransform();

upperCase.write('hello');
upperCase.write(' world');
upperCase.end();

upperCase.on('data', (chunk) => {
  console.log(chunk.toString());  // HELLO WORLD
});
```

## 💻 stream.pipeline() 实现

### 1. pipeline 完整错误处理

```javascript
// lib/internal/streams/pipeline.js
function pipeline(...streams) {
  const callback = streams[streams.length - 1];
  const streamsArray = streams.slice(0, -1);

  if (streamsArray.length < 2) {
    throw new Error('Pipeline requires at least 2 streams');
  }

  let error;
  let value;
  const destroys = [];

  // 🔥 错误传播
  function onError(err) {
    if (!error) {
      error = err;
      
      // 🔥 销毁所有流
      for (const destroy of destroys) {
        destroy(err);
      }
    }
  }

  // 🔥 完成处理
  function onFinish() {
    if (!error) {
      callback(null, value);
    } else {
      callback(error);
    }
  }

  // 🔥 连接所有流
  for (let i = 0; i < streamsArray.length - 1; i++) {
    const source = streamsArray[i];
    const destination = streamsArray[i + 1];

    // 🔥 pipe 连接
    source.pipe(destination);

    // 🔥 错误处理
    source.on('error', onError);

    // 🔥 记录销毁函数
    destroys.push((err) => {
      if (typeof source.destroy === 'function') {
        source.destroy(err);
      }
    });
  }

  // 🔥 最后一个流完成
  const lastStream = streamsArray[streamsArray.length - 1];
  lastStream.on('finish', onFinish);
  lastStream.on('error', onError);

  return lastStream;
}

// 🔥 应用
const { pipeline } = require('stream');
const fs = require('fs');
const zlib = require('zlib');

pipeline(
  fs.createReadStream('input.txt'),
  zlib.createGzip(),
  fs.createWriteStream('input.txt.gz'),
  (err) => {
    if (err) {
      console.error('Pipeline failed:', err);
    } else {
      console.log('Pipeline succeeded');
    }
  }
);
```

## 📝 总结

### 核心要点

1. **状态机设计**：flowing/paused模式、ended/finished状态
2. **背压机制**：write()返回false、drain事件、自动pause/resume

### 实现关键

1. **ReadableState/WritableState**: 状态对象管理
2. **push()/write()**: 数据入口
3. **_read()/_write()**: 用户实现
4. **pipe()/pipeline()**: 流式组合

### 应用价值

1. **内存优化**：流式处理大文件
2. **性能提升**：避免大块数据积压
3. **错误处理**：pipeline统一管理
4. **架构设计**：流式思想、组合模式
