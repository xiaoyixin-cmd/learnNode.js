# 3.1 fs 模块：JS 调用转 C++ 线程池任务

> 深入理解文件系统操作如何从 JavaScript 转换为底层 C++ 执行

---

## 📋 核心概念

fs 模块是 Node.js 中**使用频率最高的核心模块之一**，它展示了完整的 JavaScript → C++ → libuv 调用链。

**关键问题：**
- `fs.readFile()` 如何实现异步？
- 为什么文件操作不阻塞事件循环？
- Promise 化是如何实现的？
- 同步 API 与异步 API 的底层差异？

**核心机制：**
```
JavaScript 调用
  ↓
C++ Binding 包装
  ↓
libuv 线程池任务
  ↓
系统调用（read/write）
  ↓
回调到 JavaScript
```

## 💻 fs.readFile() 完整调用链

### 1. JavaScript 层入口

```javascript
// lib/fs.js
const { readFile } = require('fs');

readFile('/path/to/file', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

**内部实现：**
```javascript
// lib/fs.js
function readFile(path, options, callback) {
  // 🔥 参数归一化
  callback = maybeCallback(callback || options);
  options = getOptions(options, { flag: 'r' });

  // 🔥 创建读取上下文
  const context = new ReadFileContext(callback, options.encoding);
  context.isUserFd = isFd(path);

  // 🔥 打开文件
  const req = new FSReqCallback();
  req.context = context;
  req.oncomplete = readFileAfterOpen;

  if (context.isUserFd) {
    process.nextTick(function tick(context) {
      req.oncomplete(null, path);
    }, context);
  } else {
    // 🔥 调用 C++ binding
    binding.open(pathModule.toNamespacedPath(path),
                 stringToFlags(options.flag || 'r'),
                 0o666,
                 req);
  }
}
```

### 3. C++ Binding 层

```cpp
// src/node_file.cc
namespace node {
namespace fs {

// 🔥 Open 文件
static void Open(const FunctionCallbackInfo<Value>& args) {
  Environment* env = Environment::GetCurrent(args);

  const int argc = args.Length();
  CHECK_GE(argc, 3);

  // 🔥 参数解析
  BufferValue path(env->isolate(), args[0]);
  CHECK_NOT_NULL(*path);

  int flags = args[1]->Int32Value(env->context()).ToChecked();
  int mode = args[2]->Int32Value(env->context()).ToChecked();

  // 🔥 创建请求对象
  FSReqBase* req_wrap_async = GetReqWrap(args, 3);
  
  if (req_wrap_async != nullptr) {  // 异步
    AsyncCall(env, req_wrap_async, args, "open", UTF8, 
              // 🔥 提交到线程池
              AfterInteger,
              uv_fs_open, *path, flags, mode);
  } else {  // 同步
    // 🔥 直接调用（阻塞）
    FSReqWrapSync req_wrap_sync;
    FS_SYNC_TRACE_BEGIN(open);
    int result = SyncCall(env, args[3], &req_wrap_sync, "open",
                          uv_fs_open, *path, flags, mode);
    FS_SYNC_TRACE_END(open);
    args.GetReturnValue().Set(result);
  }
}

// 🔥 异步调用封装
template <typename Func, typename... Args>
inline FSReqBase* AsyncCall(Environment* env,
                            FSReqBase* req_wrap,
                            const v8::FunctionCallbackInfo<v8::Value>& args,
                            const char* syscall,
                            enum encoding enc,
                            uv_fs_cb after,
                            Func fn,
                            Args... fn_args) {
  return AsyncDestCall(env, req_wrap, args,
                       syscall, nullptr, 0, enc,
                       after, fn, fn_args...);
}

template <typename Func, typename... Args>
inline FSReqBase* AsyncDestCall(Environment* env,
                                FSReqBase* req_wrap,
                                const v8::FunctionCallbackInfo<v8::Value>& args,
                                const char* syscall,
                                const char* dest,
                                size_t len,
                                enum encoding enc,
                                uv_fs_cb after,
                                Func fn,
                                Args... fn_args) {
  CHECK_NOT_NULL(req_wrap);
  req_wrap->Init(syscall, dest, len, enc);
  
  // 🔥 调用 libuv 函数
  int err = req_wrap->Dispatch(fn, fn_args..., after);
  
  if (err < 0) {
    // 错误处理
    uv_fs_t* uv_req = req_wrap->req();
    uv_req->result = err;
    uv_req->path = nullptr;
    after(uv_req);  // 立即回调
  }

  return req_wrap;
}

} // namespace fs
} // namespace node
```

### 4. libuv 线程池执行

```c
// deps/uv/src/unix/fs.c
int uv_fs_open(uv_loop_t* loop,
               uv_fs_t* req,
               const char* path,
               int flags,
               int mode,
               uv_fs_cb cb) {
  // 🔥 初始化请求
  INIT(OPEN);
  PATH;
  req->flags = flags;
  req->mode = mode;
  
  // 🔥 同步执行或提交到线程池
  if (cb != NULL) {
    // 异步：提交到线程池
    POST;
  } else {
    // 同步：直接执行
    uv__fs_work(&req->work_req);
    return req->result;
  }
  
  return 0;
}

// 🔥 线程池工作函数
static void uv__fs_work(struct uv__work* w) {
  int retry_on_eintr;
  uv_fs_t* req;

  req = container_of(w, uv_fs_t, work_req);
  retry_on_eintr = !(req->fs_type == UV_FS_CLOSE ||
                     req->fs_type == UV_FS_READ);

#define X(type, action)                                                       \
  case UV_FS_##type:                                                         \
    do {                                                                      \
      r = action;                                                             \
    }                                                                         \
    while (r == -1 && errno == EINTR && retry_on_eintr);                     \
    break;

  switch (req->fs_type) {
    // 🔥 执行实际的文件操作
    X(OPEN, open(req->path, req->flags, req->mode));
    X(READ, uv__fs_read(req));
    X(WRITE, uv__fs_write(req));
    // ... 其他操作
  }

#undef X
}

// 🔥 完成回调（在主线程）
static void uv__fs_done(struct uv__work* w, int status) {
  uv_fs_t* req;

  req = container_of(w, uv_fs_t, work_req);
  uv__req_unregister(req->loop);

  // 🔥 调用用户回调
  if (req->cb != NULL)
    req->cb(req);
}
```

```cpp
// src/node_file.cc
static void AfterInteger(uv_fs_t* req) {
  FSReqBase* req_wrap = FSReqBase::from_req(req);
  FSReqAfterScope after(req_wrap, req);

  if (after.Proceed()) {
    // 🔥 调用 JavaScript 回调
    req_wrap->Resolve(Integer::New(req_wrap->env()->isolate(),
                                   static_cast<int>(req->result)));
  }
}
```

### 2. FileHandle 类

```javascript
// lib/internal/fs/promises.js
class FileHandle extends EventEmitter {
  constructor(filehandle) {
    super();
    this[kHandle] = filehandle;
    this[kFd] = filehandle.fd;
  }

  // 🔥 读取文件
  async readFile(options) {
    const statFields = binding.fstat(this[kFd], false, kUsePromises);
    
    if (statFields === undefined)
      return undefined;

    const size = statFields[kIoMaxLength];
    
    if (size === 0) {
      // 读取到 EOF
      return this[kReadFileGeneric]('utf8', 0);
    }

    // 🔥 一次性读取
    const buffer = Buffer.allocUnsafe(size);
    const bytesRead = await binding.read(this[kFd], buffer, 0, size, -1, kUsePromises);
    
    if (bytesRead !== size) {
      // 部分读取，调整 buffer
      return buffer.slice(0, bytesRead);
    }
    
    return buffer;
  }

  // 🔥 关闭文件
  async close() {
    if (this[kFd] === -1) {
      return;
    }
    
    const fd = this[kFd];
    this[kFd] = -1;
    
    // 🔥 调用 C++ binding
    await binding.close(fd, kUsePromises);
  }
}
```

```c
// deps/uv/src/threadpool.c
void uv__work_submit(uv_loop_t* loop,
                     struct uv__work* w,
                     enum uv__work_kind kind,
                     void (*work)(struct uv__work* w),
                     void (*done)(struct uv__work* w, int status)) {
  uv_once(&once, init_once);

  w->loop = loop;
  w->work = work;      // 🔥 uv__fs_work
  w->done = done;      // 🔥 uv__fs_done

  // 🔥 加入工作队列
  post(&w->wq, kind);
}

static void post(QUEUE* q, enum uv__work_kind kind) {
  uv_mutex_lock(&mutex);
  
  // 🔥 加入队列
  if (kind == UV__WORK_SLOW_IO) {
    QUEUE_INSERT_TAIL(&slow_io_pending_wq, q);
  } else {
    QUEUE_INSERT_TAIL(&wq, q);
  }
  
  // 🔥 唤醒工作线程
  if (idle_threads > 0)
    uv_cond_signal(&cond);
  
  uv_mutex_unlock(&mutex);
}
```

## 💻 线程池详解

### 1. 线程池初始化

```c
// deps/uv/src/threadpool.c
static void init_once(void) {
  unsigned int i;
  const char* val;
  uv_sem_t sem;

  // 🔥 从环境变量获取线程池大小
  val = getenv("UV_THREADPOOL_SIZE");
  if (val != NULL)
    nthreads = atoi(val);
  
  // 🔥 限制范围
  if (nthreads == 0)
    nthreads = 1;
  if (nthreads > MAX_THREADPOOL_SIZE)
    nthreads = MAX_THREADPOOL_SIZE;

  // 🔥 创建工作线程
  threads = uv__malloc(nthreads * sizeof(threads[0]));

  for (i = 0; i < nthreads; i++)
    if (uv_thread_create(threads + i, worker, &sem))
      abort();
}
```

**默认配置：**
```
默认线程数：4
最大线程数：1024
环境变量：UV_THREADPOOL_SIZE
```

### 2. 工作线程循环

```c
// deps/uv/src/threadpool.c
static void worker(void* arg) {
  struct uv__work* w;
  QUEUE* q;
  int is_slow_work;

  // 🔥 通知主线程线程已启动
  uv_sem_post((uv_sem_t*) arg);
  arg = NULL;

  uv_mutex_lock(&mutex);
  for (;;) {
    // 🔥 等待任务
    while (QUEUE_EMPTY(&wq) &&
           QUEUE_EMPTY(&slow_io_pending_wq)) {
      idle_threads += 1;
      uv_cond_wait(&cond, &mutex);
      idle_threads -= 1;
    }

    // 🔥 获取任务
    q = QUEUE_HEAD(&wq);
    if (q == &wq)
      q = QUEUE_HEAD(&slow_io_pending_wq);

    QUEUE_REMOVE(q);
    QUEUE_INIT(q);  // 标记已取出

    // 🔥 检查是否为慢 I/O
    is_slow_work = 0;
    if (q == &exit_message)
      uv_cond_signal(&cond);
    else {
      w = QUEUE_DATA(q, struct uv__work, wq);
      w->work(w);  // 🔥 执行任务

      // 🔥 标记完成，通知主线程
      uv_mutex_lock(&w->loop->wq_mutex);
      w->work = NULL;  // 标记完成
      QUEUE_INSERT_TAIL(&w->loop->wq, &w->wq);
      uv_async_send(&w->loop->wq_async);
      uv_mutex_unlock(&w->loop->wq_mutex);
    }

    // 🔥 慢 I/O 任务完成后，处理等待队列
    if (is_slow_work) {
      // ...
    }
  }
}
```

### 3. 完成回调处理

```c
// deps/uv/src/unix/core.c
void uv__work_done(uv_async_t* handle) {
  struct uv__work* w;
  uv_loop_t* loop;
  QUEUE* q;
  QUEUE wq;
  int err;

  loop = container_of(handle, uv_loop_t, wq_async);
  uv_mutex_lock(&loop->wq_mutex);
  
  // 🔥 移动完成队列
  QUEUE_MOVE(&loop->wq, &wq);
  uv_mutex_unlock(&loop->wq_mutex);

  // 🔥 遍历执行完成回调
  while (!QUEUE_EMPTY(&wq)) {
    q = QUEUE_HEAD(&wq);
    QUEUE_REMOVE(q);

    w = container_of(q, struct uv__work, wq);
    err = (w->work == uv__cancelled) ? UV_ECANCELED : 0;
    
    // 🔥 执行 done 回调（在主线程）
    w->done(w, err);
  }
}
```

## 📝 总结

### 核心要点

1. **异步机制**：libuv 线程池实现真正的非阻塞 I/O
2. **调用链路**：JS → C++ Binding → libuv → System Call
3. **Promise 化**：通过 kUsePromises 标记实现
4. **线程池**：默认4线程，可配置（UV_THREADPOOL_SIZE）

### 实现关键

1. **FSReqCallback**: 包装 JavaScript 回调
2. **uv_fs_t**: libuv 文件操作请求
3. **线程池**: 异步执行，不阻塞主线程
4. **完成回调**: uv__work_done 在主线程执行

### 应用价值

1. **理解异步**: 深入 Node.js 异步 I/O 原理
2. **性能优化**: 合理使用线程池资源
3. **问题排查**: 定位文件操作瓶颈
4. **架构设计**: 借鉴线程池设计模式
