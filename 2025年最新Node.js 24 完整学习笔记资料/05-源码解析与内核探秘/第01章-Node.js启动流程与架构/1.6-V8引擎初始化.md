# 1.6 V8 引擎初始化与 Isolate/Context 创建

> 深入理解 V8 引擎的初始化流程和核心概念

---

## 📋 知识点概述

### 核心概念

V8 是 Google 开发的高性能 JavaScript 引擎，Node.js 的核心执行引擎。理解 V8 的初始化过程对深入掌握 Node.js 至关重要。

**核心概念：**
- **Platform** - 平台抽象层（线程、任务调度）
- **Isolate** - 隔离实例（独立堆内存）
- **Context** - 执行上下文（全局对象、作用域）
- **Snapshot** - 快照机制（快速启动）

**初始化流程：**
```
V8::InitializePlatform()
    ↓
V8::Initialize()
    ↓
Isolate::New()
    ↓
Context::New()
    ↓
LoadSnapshot() (可选)
    ↓
Execute JavaScript
```

### 2. NodePlatform 实现

```cpp
// src/node_platform.cc
namespace node {

using v8::Task;
using v8::TracingController;

class NodePlatform : public MultiIsolatePlatform {
 public:
  NodePlatform(int thread_pool_size, TracingController* tracing)
    : tracing_controller_(tracing) {
    // 创建工作线程池
    for (int i = 0; i < thread_pool_size; i++) {
      auto worker = std::make_unique<WorkerThread>(this);
      worker->Start();
      workers_.push_back(std::move(worker));
    }
  }
  
  ~NodePlatform() override {
    // 停止所有工作线程
    for (auto& worker : workers_) {
      worker->Stop();
    }
  }
  
  // 获取线程池大小
  size_t NumberOfWorkerThreads() override {
    return workers_.size();
  }
  
  // 提交任务到线程池
  void CallOnWorkerThread(std::unique_ptr<Task> task) override {
    // 轮询分配任务
    size_t index = next_worker_.fetch_add(1) % workers_.size();
    workers_[index]->PostTask(std::move(task));
  }
  
  // 延迟任务
  void CallDelayedOnWorkerThread(
    std::unique_ptr<Task> task,
    double delay_in_seconds
  ) override {
    // 使用 libuv 定时器
    DelayedTaskScheduler::Use(GetForegroundTaskRunner(isolate))
      ->PostDelayedTask(
        std::move(task),
        delay_in_seconds
      );
  }
  
  // 获取追踪控制器
  TracingController* GetTracingController() override {
    return tracing_controller_;
  }
  
  // 获取前台任务运行器
  std::shared_ptr<v8::TaskRunner> GetForegroundTaskRunner(
    Isolate* isolate
  ) override {
    return per_isolate_[isolate];
  }
  
  // 注册 Isolate
  void RegisterIsolate(Isolate* isolate, uv_loop_t* loop) override {
    per_isolate_[isolate] = 
      std::make_shared<PerIsolatePlatformData>(isolate, loop);
  }
  
  // 注销 Isolate
  void UnregisterIsolate(Isolate* isolate) override {
    per_isolate_.erase(isolate);
  }
  
 private:
  std::vector<std::unique_ptr<WorkerThread>> workers_;
  std::atomic<size_t> next_worker_{0};
  TracingController* tracing_controller_;
  std::unordered_map<Isolate*, std::shared_ptr<PerIsolatePlatformData>> 
    per_isolate_;
};

} // namespace node
```

### 3. WorkerThread 实现

```cpp
// src/node_platform.cc
class WorkerThread {
 public:
  explicit WorkerThread(NodePlatform* platform)
    : platform_(platform),
      running_(false) {}
  
  void Start() {
    running_ = true;
    
    // 创建 libuv 线程
    uv_thread_create(&thread_, [](void* arg) {
      static_cast<WorkerThread*>(arg)->Run();
    }, this);
  }
  
  void Stop() {
    running_ = false;
    
    // 发送退出任务
    PostTask(nullptr);
    
    // 等待线程结束
    uv_thread_join(&thread_);
  }
  
  void PostTask(std::unique_ptr<Task> task) {
    // 添加到任务队列
    std::lock_guard<std::mutex> lock(queue_mutex_);
    task_queue_.push(std::move(task));
    queue_cond_.notify_one();
  }
  
 private:
  void Run() {
    while (running_) {
      std::unique_ptr<Task> task;
      
      {
        std::unique_lock<std::mutex> lock(queue_mutex_);
        
        // 等待任务
        queue_cond_.wait(lock, [this]() {
          return !task_queue_.empty() || !running_;
        });
        
        if (!running_) break;
        
        task = std::move(task_queue_.front());
        task_queue_.pop();
      }
      
      if (task) {
        // 执行任务
        task->Run();
      }
    }
  }
  
  NodePlatform* platform_;
  uv_thread_t thread_;
  std::atomic<bool> running_;
  std::mutex queue_mutex_;
  std::condition_variable queue_cond_;
  std::queue<std::unique_ptr<Task>> task_queue_;
};
```

## 💻 Isolate 创建

### 1. 创建 Isolate

```cpp
// src/node.cc
namespace node {

Isolate* NewIsolate(
  ArrayBufferAllocator* allocator,
  uv_loop_t* event_loop,
  MultiIsolatePlatform* platform
) {
  // 1. 配置 Isolate 参数
  Isolate::CreateParams params;
  
  // 内存分配器
  params.array_buffer_allocator = allocator;
  
  // 外部引用表（C++ 函数地址）
  params.external_references = per_process::external_references;
  
  // 2. 加载 Snapshot
  const SnapshotData* snapshot_data = 
    SnapshotBuilder::GetEmbeddedSnapshotData();
  
  if (snapshot_data != nullptr) {
    params.snapshot_blob = snapshot_data->AsBlob();
  }
  
  // 3. 创建 Isolate
  Isolate* isolate = Isolate::New(params);
  
  // 4. 设置内存限制
  if (per_process::cli_options->max_old_space_size > 0) {
    size_t limit = per_process::cli_options->max_old_space_size * 1024 * 1024;
    isolate->SetHeapLimit(limit);
  }
  
  // 5. 注册到 Platform
  platform->RegisterIsolate(isolate, event_loop);
  
  // 6. 设置回调
  isolate->SetPromiseRejectCallback(PromiseRejectCallback);
  isolate->SetPrepareStackTraceCallback(PrepareStackTraceCallback);
  
  // 7. 添加 GC 回调
  isolate->AddGCPrologueCallback(GCPrologueCallback, 
                                   static_cast<void*>(isolate));
  isolate->AddGCEpilogueCallback(GCEpilogueCallback, 
                                  static_cast<void*>(isolate));
  
  return isolate;
}

} // namespace node
```

### 3. ArrayBufferAllocator 实现

```cpp
// src/node.cc
namespace node {

class NodeArrayBufferAllocator : public ArrayBuffer::Allocator {
 public:
  NodeArrayBufferAllocator() = default;
  
  // 分配内存（初始化为 0）
  void* Allocate(size_t length) override {
    if (length > kMaxLength) return nullptr;
    
    void* data = UncheckedMalloc(length);
    if (data == nullptr) {
      return nullptr;
    }
    
    // 初始化为 0
    memset(data, 0, length);
    
    // 更新统计
    total_allocated_.fetch_add(length, std::memory_order_relaxed);
    
    return data;
  }
  
  // 分配内存（不初始化）
  void* AllocateUninitialized(size_t length) override {
    if (length > kMaxLength) return nullptr;
    
    void* data = UncheckedMalloc(length);
    if (data == nullptr) {
      return nullptr;
    }
    
    // 更新统计
    total_allocated_.fetch_add(length, std::memory_order_relaxed);
    
    return data;
  }
  
  // 释放内存
  void Free(void* data, size_t length) override {
    if (data == nullptr) return;
    
    // 更新统计
    total_allocated_.fetch_sub(length, std::memory_order_relaxed);
    
    free(data);
  }
  
  // 获取总分配量
  size_t total_allocated() const {
    return total_allocated_.load(std::memory_order_relaxed);
  }
  
 private:
  static constexpr size_t kMaxLength = 0x7fffffff;
  std::atomic<size_t> total_allocated_{0};
};

} // namespace node
```

### 1. 创建 Context

```cpp
// src/api/environment.cc
namespace node {

using v8::Context;
using v8::EscapableHandleScope;
using v8::Isolate;
using v8::Local;
using v8::Object;
using v8::ObjectTemplate;

Local<Context> NewContext(
  Isolate* isolate,
  Local<ObjectTemplate> object_template
) {
  EscapableHandleScope scope(isolate);
  
  // 1. 创建 Context
  Local<Context> context = Context::New(
    isolate,
    nullptr,  // 扩展
    object_template  // 全局对象模板
  );
  
  if (context.IsEmpty()) {
    return Local<Context>();
  }
  
  // 2. 进入 Context
  Context::Scope context_scope(context);
  
  // 3. 设置安全 Token
  context->SetSecurityToken(Undefined(isolate));
  
  // 4. 设置 Embedder Data
  context->SetEmbedderData(
    ContextEmbedderIndex::kContextTag,
    Number::New(isolate, Environment::kNodeContextTag)
  );
  
  // 5. 允许代码生成
  context->AllowCodeGenerationFromStrings(true);
  
  return scope.Escape(context);
}

} // namespace node
```

### 2. Context 概念详解

**Context 是什么？**
- JavaScript 执行环境
- 包含全局对象（global）
- 包含内置对象（Object、Array、Function 等）
- 独立的作用域链

**多 Context 示例：**

## 💻 Snapshot 机制

### 1. Snapshot 原理

Snapshot 是 V8 的快照机制，将初始化后的堆状态序列化，下次启动时直接加载，大幅提升启动速度。

**工作流程：**
```
第一次启动：
  初始化 V8 → 执行 Bootstrap → 生成 Snapshot

后续启动：
  加载 Snapshot → 恢复堆状态 → 直接执行用户代码
```

**性能对比：**
- 无 Snapshot：~200ms
- 有 Snapshot：~50ms（提升 4 倍）

---

### 2. 生成 Snapshot

```cpp
// tools/snapshot/node_mksnapshot.cc
namespace node {

int BuildSnapshot(int argc, char* argv[]) {
  // 1. 初始化 V8
  std::unique_ptr<Platform> platform = 
    v8::platform::NewDefaultPlatform();
  V8::InitializePlatform(platform.get());
  V8::Initialize();
  
  // 2. 创建 Isolate
  SnapshotCreator creator;
  Isolate* isolate = creator.GetIsolate();
  
  {
    Isolate::Scope isolate_scope(isolate);
    HandleScope handle_scope(isolate);
    
    // 3. 创建 Context
    Local<Context> context = Context::New(isolate);
    Context::Scope context_scope(context);
    
    // 4. 执行 Bootstrap 代码
    Environment* env = CreateEnvironment(
      isolate,
      uv_default_loop(),
      context,
      argc,
      argv
    );
    
    LoadEnvironment(env);
    
    // 5. 设置默认 Context
    creator.SetDefaultContext(context);
    
    // 6. 添加外部引用
    creator.AddData(env);
  }
  
  // 7. 创建 Blob
  StartupData blob = 
    creator.CreateBlob(SnapshotCreator::FunctionCodeHandling::kClear);
  
  // 8. 写入文件
  FILE* fp = fopen("node_snapshot.blob", "wb");
  fwrite(blob.data, blob.raw_size, 1, fp);
  fclose(fp);
  
  // 9. 清理
  delete[] blob.data;
  V8::Dispose();
  V8::ShutdownPlatform();
  
  return 0;
}

} // namespace node
```

### 2. HandleScope 作用域

```cpp
void Example() {
  Isolate* isolate = Isolate::GetCurrent();
  
  // 1. HandleScope - 栈上作用域
  {
    HandleScope scope(isolate);
    
    Local<String> str1 = String::NewFromUtf8Literal(isolate, "a");
    Local<String> str2 = String::NewFromUtf8Literal(isolate, "b");
    
    // str1 和 str2 在 scope 结束时自动清理
  }
  
  // 2. EscapableHandleScope - 可逃逸作用域
  Local<String> CreateString() {
    EscapableHandleScope scope(isolate);
    
    Local<String> str = String::NewFromUtf8Literal(isolate, "hello");
    
    // 逃逸出作用域
    return scope.Escape(str);
  }
  
  // 3. 无 HandleScope - 内存泄漏！
  for (int i = 0; i < 1000000; i++) {
    // ❌ 内存泄漏
    Local<String> str = String::NewFromUtf8Literal(isolate, "leak");
  }
  
  // 4. 正确用法
  for (int i = 0; i < 1000000; i++) {
    HandleScope scope(isolate);
    
    // ✅ 每次循环后自动清理
    Local<String> str = String::NewFromUtf8Literal(isolate, "ok");
  }
}
```

## 📝 总结

### 核心要点

1. **Platform** - 线程池、任务调度
2. **Isolate** - 独立实例、堆内存隔离
3. **Context** - 执行环境、全局对象
4. **Snapshot** - 快照机制、快速启动
5. **Handle** - 句柄管理、内存安全

### 实现关键

1. **分层架构**：Platform → Isolate → Context
2. **资源管理**：HandleScope 自动清理
3. **性能优化**：Snapshot 减少启动时间
4. **内存隔离**：多 Isolate 安全隔离

### 应用价值

1. **性能优化**：理解内存管理，避免泄漏
2. **扩展开发**：使用 V8 API 开发原生模块
3. **调试技巧**：使用 V8 工具定位问题
4. **架构设计**：理解 Node.js 底层架构

### 注意事项

1. **Handle 管理**：必须在 HandleScope 中使用
2. **线程安全**：Isolate 同时只能被一个线程使用
3. **内存限制**：合理设置 `--max-old-space-size`
4. **Snapshot 限制**：不能包含外部数据
