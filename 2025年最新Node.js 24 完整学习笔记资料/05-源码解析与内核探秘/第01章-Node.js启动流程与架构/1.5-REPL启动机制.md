# 1.5 REPL 启动机制：lib/repl.js 与 readline 集成

> 深入理解 Node.js 交互式环境的实现原理

---

## 📋 知识点概述

### 核心概念

REPL（Read-Eval-Print-Loop）是 Node.js 提供的交互式命令行环境，用于快速测试 JavaScript 代码。

**REPL 循环：**
1. **Read** - 读取用户输入
2. **Eval** - 执行 JavaScript 代码
3. **Print** - 打印执行结果
4. **Loop** - 循环等待下一次输入

**关键组件：**
- `lib/repl.js` - REPL 核心实现
- `lib/readline.js` - 行读取和编辑
- `lib/internal/repl/history.js` - 历史记录
- `lib/vm.js` - 代码执行沙箱

### 2. repl.js - REPLServer 类

```javascript
// lib/repl.js（简化版）
'use strict';

const {
  ArrayPrototypeJoin,
  ArrayPrototypePop,
  ArrayPrototypePush,
  ArrayPrototypeShift,
  ObjectDefineProperty,
  ObjectGetPrototypeOf,
  ObjectKeys,
  ObjectSetPrototypeOf,
  ReflectApply,
  RegExpPrototypeTest,
  StringPrototypeEndsWith,
  StringPrototypeIncludes,
  StringPrototypeStartsWith,
  StringPrototypeTrim,
} = primordials;

const { Interface } = require('readline');
const { inspect } = require('util');
const vm = require('vm');
const path = require('path');
const fs = require('fs');
const { inherits } = require('util');

// REPLServer 类
function REPLServer(prompt, stream, eval_, useGlobal, ignoreUndefined) {
  if (!(this instanceof REPLServer)) {
    return new REPLServer(prompt, stream, eval_, useGlobal, ignoreUndefined);
  }

  // 1. 继承 readline.Interface
  Interface.call(this, {
    input: stream.input || process.stdin,
    output: stream.output || process.stdout,
    terminal: stream.terminal || process.stdout.isTTY,
    completer: this.complete.bind(this),
    prompt: prompt || '> '
  });

  // 2. 设置属性
  this.eval = eval_ || defaultEval;
  this.useGlobal = !!useGlobal;
  this.ignoreUndefined = !!ignoreUndefined;
  this.replMode = exports.REPL_MODE_SLOPPY;

  // 3. 创建执行上下文
  if (this.useGlobal) {
    this.context = global;
  } else {
    this.context = vm.createContext();
    
    // 复制全局对象
    for (const name of ObjectKeys(global)) {
      if (!this.context.hasOwnProperty(name)) {
        ObjectDefineProperty(this.context, name, {
          configurable: true,
          enumerable: true,
          value: global[name],
          writable: true
        });
      }
    }
  }

  // 4. 历史记录
  this.history = [];
  this.historyIndex = -1;

  // 5. 命令缓冲
  this.bufferedCommand = '';
  this.lines = {
    level: [],
    buffer: ''
  };

  // 6. 设置事件监听器
  this.on('line', (cmd) => {
    this._onLine(cmd);
  });

  this.on('close', () => {
    this.emit('exit');
  });

  // 7. 显示提示符
  this.displayPrompt();
}

// 继承 readline.Interface
inherits(REPLServer, Interface);

// 导出
module.exports = {
  REPLServer,
  start: function start(options) {
    return new REPLServer(options);
  },
  REPL_MODE_SLOPPY: Symbol('repl-sloppy'),
  REPL_MODE_STRICT: Symbol('repl-strict'),
};
```

## 💻 源码分析：代码执行

### 1. _onLine() - 处理输入

```javascript
// lib/repl.js
REPLServer.prototype._onLine = function(cmd) {
  const self = this;
  
  // 1. 去除空白
  cmd = StringPrototypeTrim(cmd);
  
  // 2. 检查是否是 REPL 命令
  if (cmd && cmd[0] === '.' && cmd[1] !== '.' && !NumberIsNaN(Number(cmd[1]))) {
    this._parseREPLCommand(cmd);
    return;
  }
  
  // 3. 处理多行输入
  if (this._inTemplateLiteral(cmd)) {
    this.bufferedCommand += cmd + '\n';
    this._refreshLine();
    return;
  }
  
  // 4. 执行代码
  this.eval(
    this.bufferedCommand + cmd,
    this.context,
    this.inputStream.filename,
    (err, result) => {
      if (err) {
        // 错误处理
        if (self._domain) {
          self._domain.emit('error', err);
        } else {
          self.outputStream.write(err.stack + '\n');
        }
      } else if (!self.ignoreUndefined || result !== undefined) {
        // 打印结果
        self.outputStream.write(
          self.writer(result) + '\n'
        );
      }
      
      // 清空缓冲
      self.bufferedCommand = '';
      self.lines.level = [];
      
      // 显示下一个提示符
      self.displayPrompt();
    }
  );
};
```

```javascript
// lib/repl.js
function defaultEval(code, context, file, cb) {
  const async = false;
  let err = null;
  let result;

  try {
    // 1. 检查语法错误
    const { isRecoverableError } = require('internal/repl/utils');
    
    try {
      // 尝试作为表达式执行
      result = vm.runInContext(
        `(${code})`,
        context,
        {
          filename: file,
          displayErrors: false,
        }
      );
    } catch (e) {
      // 如果失败，作为语句执行
      result = vm.runInContext(
        code,
        context,
        {
          filename: file,
          displayErrors: false,
        }
      );
    }

  } catch (e) {
    err = e;

    // 2. 检查是否可恢复（多行输入）
    if (isRecoverableError(e, code)) {
      err.recoverable = true;
    }
  }

  // 3. 回调
  cb(err, result);
}
```

### 1. readline.Interface

```javascript
// lib/readline.js（简化版）
'use strict';

const {
  ArrayPrototypePush,
  StringPrototypeSlice,
} = primordials;

const { Interface: ReadlineInterface } = internalBinding('readline');

function Interface(options) {
  if (!(this instanceof Interface)) {
    return new Interface(options);
  }

  // 配置选项
  this.input = options.input || process.stdin;
  this.output = options.output || process.stdout;
  this.terminal = options.terminal !== undefined ? 
    options.terminal : this.output.isTTY;
  this.completer = options.completer;
  this.prompt = options.prompt || '> ';
  
  // 内部状态
  this.line = '';
  this.cursor = 0;
  this.history = [];
  this.historyIndex = -1;
  
  // TTY 模式
  if (this.terminal) {
    this._setupTTY();
  }
  
  // 监听输入
  this.input.on('data', (data) => {
    this._normalWrite(data);
  });
  
  this.input.on('end', () => {
    this.close();
  });
}

// 设置 TTY 模式
Interface.prototype._setupTTY = function() {
  const { emitKeypressEvents } = require('internal/readline/emitKeypressEvents');
  
  // 启用原始模式
  this.input.setRawMode(true);
  
  // 监听按键事件
  emitKeypressEvents(this.input);
  
  this.input.on('keypress', (s, key) => {
    this._ttyWrite(s, key);
  });
};

// 处理 TTY 输入
Interface.prototype._ttyWrite = function(s, key) {
  // 特殊键处理
  if (key.ctrl && key.name === 'c') {
    // Ctrl+C
    this.emit('SIGINT');
    return;
  }
  
  if (key.ctrl && key.name === 'd') {
    // Ctrl+D（EOF）
    if (this.line.length === 0) {
      this.close();
      return;
    }
  }
  
  if (key.name === 'return' || key.name === 'enter') {
    // 回车
    this._line();
    return;
  }
  
  if (key.name === 'up') {
    // 上箭头（历史记录）
    this._historyPrev();
    return;
  }
  
  if (key.name === 'down') {
    // 下箭头
    this._historyNext();
    return;
  }
  
  if (key.name === 'tab') {
    // Tab（自动补全）
    this._tabComplete();
    return;
  }
  
  if (key.name === 'backspace') {
    // 退格
    this._deleteLeft();
    return;
  }
  
  // 普通字符
  this._insertString(s);
};

// 显示提示符
Interface.prototype.displayPrompt = function(preserveCursor) {
  if (this.terminal) {
    this.output.write(this.prompt);
  }
};

// 处理一行输入
Interface.prototype._line = function() {
  const line = this.line;
  
  this.output.write('\n');
  this.line = '';
  this.cursor = 0;
  
  this._addHistory(line);
  
  this.emit('line', line);
};

// 添加到历史记录
Interface.prototype._addHistory = function(line) {
  if (line.length === 0 || /^\s*$/.test(line)) {
    return;
  }
  
  // 避免重复
  if (this.history.length > 0 && this.history[0] === line) {
    return;
  }
  
  ArrayPrototypePush(this.history, line);
  
  // 限制历史记录数量
  if (this.history.length > 1000) {
    ArrayPrototypeShift(this.history);
  }
  
  this.historyIndex = -1;
};

module.exports = { Interface };
```

## 💻 源码分析：自动补全

### 1. complete() - 自动补全

```javascript
// lib/repl.js
REPLServer.prototype.complete = function(line, callback) {
  // 1. 获取当前上下文的所有键
  let completionGroups = [];
  let completeOn = line;
  
  // 2. 检查是否是属性访问
  const match = line.match(/([a-zA-Z_$][a-zA-Z0-9_$]*\.)*([a-zA-Z_$][a-zA-Z0-9_$]*)$/);
  
  if (match) {
    const expr = match[0];
    const propertyPath = expr.split('.');
    const obj = this._resolve(propertyPath.slice(0, -1));
    
    if (obj != null) {
      const prefix = propertyPath[propertyPath.length - 1];
      const keys = this._getKeys(obj);
      
      completionGroups.push(
        keys.filter(key => key.startsWith(prefix))
      );
    }
  } else {
    // 3. 全局对象补全
    const keys = ObjectKeys(this.context);
    completionGroups.push(
      keys.filter(key => key.startsWith(line))
    );
  }
  
  // 4. 返回补全结果
  const completions = [];
  for (const group of completionGroups) {
    for (const item of group) {
      completions.push(item);
    }
  }
  
  callback(null, [completions, completeOn]);
};

// 解析属性路径
REPLServer.prototype._resolve = function(path) {
  let obj = this.context;
  
  for (const key of path) {
    if (obj == null || !obj.hasOwnProperty(key)) {
      return null;
    }
    obj = obj[key];
  }
  
  return obj;
};

// 获取对象的所有键（包括原型链）
REPLServer.prototype._getKeys = function(obj) {
  const keys = new Set();
  
  while (obj != null) {
    for (const key of ObjectKeys(obj)) {
      keys.add(key);
    }
    obj = ObjectGetPrototypeOf(obj);
  }
  
  return Array.from(keys);
};
```

## 💻 源码分析：历史记录

### 1. setupHistory() - 设置历史记录

```javascript
// lib/internal/repl/history.js
'use strict';

const fs = require('fs');
const path = require('path');
const os = require('os');

// 默认历史文件路径
const kHistoryPath = path.join(
  os.homedir(),
  '.node_repl_history'
);

function setupHistory(repl, historyPath, callback) {
  historyPath = historyPath || process.env.NODE_REPL_HISTORY || kHistoryPath;
  
  // 1. 读取历史文件
  fs.readFile(historyPath, 'utf8', (err, data) => {
    if (err && err.code !== 'ENOENT') {
      return callback(err);
    }
    
    // 2. 解析历史记录
    if (data) {
      const lines = data.split(/\r?\n/).reverse();
      
      for (const line of lines) {
        if (line.trim()) {
          repl.history.push(line);
        }
      }
    }
    
    // 3. 监听新命令
    repl.on('line', (line) => {
      if (line.trim()) {
        // 追加到历史文件
        fs.appendFile(historyPath, line + '\n', (err) => {
          if (err) {
            console.error('Error writing to history file:', err);
          }
        });
      }
    });
    
    callback(null);
  });
}

module.exports = { setupHistory };
```

### 1. 内置命令

```javascript
// lib/repl.js
REPLServer.prototype._parseREPLCommand = function(cmd) {
  switch (cmd) {
    case '.break':
      // 清空缓冲区
      this.bufferedCommand = '';
      this.lines.level = [];
      this.displayPrompt();
      break;
      
    case '.clear':
      // 重置上下文
      if (this.useGlobal) {
        this.outputStream.write('Cannot clear global context\n');
      } else {
        this.context = vm.createContext();
      }
      this.displayPrompt();
      break;
      
    case '.exit':
      // 退出
      this.close();
      break;
      
    case '.help':
      // 显示帮助
      this.outputStream.write(
        '.break    Sometimes you get stuck in a place you can\'t get out...\n' +
        '.clear    Clear the REPL context\n' +
        '.exit     Exit the REPL\n' +
        '.help     Print this help message\n' +
        '.save     Save all evaluated commands to a file\n' +
        '.load     Load JS from a file into the REPL session\n'
      );
      this.displayPrompt();
      break;
      
    case '.save':
      // 保存会话
      this._saveSes session(cmd);
      break;
      
    case '.load':
      // 加载文件
      this._loadFile(cmd);
      break;
      
    default:
      this.outputStream.write('Unknown command: ' + cmd + '\n');
      this.displayPrompt();
  }
};
```

## 🎯 实践：自定义 REPL

### 1. 创建自定义 REPL

```javascript
// my-repl.js
const repl = require('repl');

// 自定义 eval 函数
function myEval(cmd, context, filename, callback) {
  let result;
  
  try {
    // 添加自定义功能
    if (cmd.startsWith('$')) {
      // 特殊命令
      result = handleSpecialCommand(cmd.slice(1));
    } else {
      // 正常执行
      result = eval(cmd);
    }
    
    callback(null, result);
  } catch (err) {
    callback(err);
  }
}

function handleSpecialCommand(cmd) {
  switch (cmd.trim()) {
    case 'time':
      return new Date().toLocaleString();
    case 'version':
      return process.version;
    default:
      throw new Error(`Unknown command: $${cmd}`);
  }
}

// 启动自定义 REPL
const r = repl.start({
  prompt: 'my-repl> ',
  eval: myEval,
  useGlobal: false,
  ignoreUndefined: true
});

// 添加自定义命令
r.defineCommand('greet', {
  help: 'Greet the user',
  action(name) {
    this.clearBufferedCommand();
    console.log(`Hello, ${name || 'World'}!`);
    this.displayPrompt();
  }
});
```
