# 1.3 Bootstrap 过程：如何加载 internal/bootstrap/loaders.js

> 深入理解 Node.js 的 JavaScript 引导加载机制

---

## 📋 知识点概述

### 核心概念

Bootstrap（引导）是 Node.js 启动过程中最关键的阶段，负责将 C++ 环境转换为 JavaScript 执行环境。

**Bootstrap 阶段：**
1. **Pre-execution**：准备执行环境
2. **Setup Loaders**：设置模块加载器
3. **Setup Node.js**：初始化 Node.js 核心
4. **Setup User Code**：加载用户代码

**关键文件：**
- `lib/internal/bootstrap/loaders.js` - 模块加载器
- `lib/internal/bootstrap/node.js` - Node.js 核心
- `lib/internal/bootstrap/pre_execution.js` - 预执行
- `lib/internal/main/run_main_module.js` - 主模块运行

## 💻 源码分析：ExecuteBootstrapper()

### 1. C++ 执行引导代码

```cpp
// src/node.cc
MaybeLocal<Value> ExecuteBootstrapper(
  Isolate* isolate,
  const char* id,
  std::vector<Local<String>>* parameters,
  std::vector<Local<Value>>* arguments
) {
  EscapableHandleScope scope(isolate);
  
  // 1. 获取引导代码源码（从 C++ 嵌入）
  Local<String> source = 
    node::per_process::builtins.GetSource(isolate, id);
  
  if (source.IsEmpty()) {
    return MaybeLocal<Value>();
  }
  
  // 2. 编译为函数
  ScriptOrigin origin(
    FIXED_ONE_BYTE_STRING(isolate, id),
    0,  // line offset
    0,  // column offset
    false,  // is cross origin
    -1,  // script id
    Local<Value>(),  // source map URL
    false,  // is opaque
    false,  // is WASM
    false,  // is ES module
    Local<PrimitiveArray>()  // host defined options
  );
  
  ScriptCompiler::Source script_source(source, origin);
  
  Local<Function> fn;
  if (!ScriptCompiler::CompileFunctionInContext(
        isolate->GetCurrentContext(),
        &script_source,
        parameters->size(),
        parameters->data(),
        0,
        nullptr
      ).ToLocal(&fn)) {
    return MaybeLocal<Value>();
  }
  
  // 3. 执行函数
  Local<Value> result;
  if (!fn->Call(
        isolate->GetCurrentContext(),
        Undefined(isolate),
        arguments->size(),
        arguments->data()
      ).ToLocal(&result)) {
    return MaybeLocal<Value>();
  }
  
  return scope.Escape(result);
}
```

**工作原理：**
1. 从 C++ 嵌入的字符串获取源码
2. 编译为 JavaScript 函数
3. 传入参数并执行
4. 返回结果

```cpp
// 获取源码
Local<String> BuiltinLoader::GetSource(
  Isolate* isolate,
  const char* id
) {
  // 查找模块
  for (size_t i = 0; i < arraysize(builtin_modules); i++) {
    if (strcmp(id, builtin_modules[i].id) == 0) {
      // 转换为 V8 String
      return String::NewFromOneByte(
        isolate,
        builtin_modules[i].source,
        NewStringType::kNormal,
        builtin_modules[i].source_len
      ).ToLocalChecked();
    }
  }
  
  return Local<String>();
}

} // namespace builtins
} // namespace node
```

### 1. NativeModule 系统

```javascript
// lib/internal/bootstrap/loaders.js
'use strict';

// 使用 primordials 保护内置对象
const {
  ArrayPrototypeForEach,
  ArrayPrototypeIncludes,
  ArrayPrototypePush,
  ArrayPrototypeSlice,
  Error,
  ObjectCreate,
  ObjectDefineProperty,
  ObjectGetOwnPropertyDescriptor,
  ObjectKeys,
  ObjectPrototypeHasOwnProperty,
  ReflectApply,
  SafeArrayIterator,
  SafeMap,
  SafeSet,
  SafeWeakMap,
  StringPrototypeStartsWith,
  StringPrototypeSlice,
} = primordials;

// NativeModule 类
class NativeModule {
  constructor(id) {
    this.filename = `${id}.js`;
    this.id = id;
    this.exports = {};
    this.loaded = false;
    this.loading = false;
    this.canBeRequiredByUsers = 
      !StringPrototypeStartsWith(id, 'internal/');
  }

  // 静态缓存
  static map = new SafeMap();

  // 编译模块
  compile() {
    let source = NativeModule.getSource(this.id);
    source = NativeModule.wrap(source);

    this.loading = true;

    const fn = NativeModule.runScriptInThisContext(
      source, 
      this.filename
    );

    const requireFn = NativeModule.makeRequireFunction(this);

    ReflectApply(fn, this.exports, [
      this.exports,
      requireFn,
      this,
      this.filename,
      NativeModule.dirname(this.id),
    ]);

    this.loaded = true;
    this.loading = false;
  }

  // 包装源码
  static wrap(script) {
    return `(function (exports, require, module, __filename, __dirname) {
      ${script}
    });`;
  }

  // 在当前上下文执行
  static runScriptInThisContext(code, filename) {
    const { compileFunction } = internalBinding('contextify');
    
    return compileFunction(
      code,
      filename,
      0,  // line offset
      0,  // column offset
      undefined,  // cached data
      false,  // produce cached data
      undefined,  // parsing context
      [],  // context extensions
      [
        'exports',
        'require',
        'module',
        '__filename',
        '__dirname',
      ]
    );
  }

  // 获取源码（从 C++）
  static getSource(id) {
    const { getCacheUsage, compileFunction } = 
      internalBinding('native_module');
    
    return getCacheUsage(id) || internalBinding('natives')[id];
  }

  // 创建 require 函数
  static makeRequireFunction(mod) {
    const Module = NativeModule;

    function require(id) {
      if (id === 'module') {
        return mod;
      }

      return Module.require(id);
    }

    function resolve(request) {
      return request;
    }

    require.resolve = resolve;

    return require;
  }

  // 加载模块
  static require(id) {
    if (id === 'native_module') {
      return NativeModule;
    }

    const cached = NativeModule.getCached(id);
    if (cached && (cached.loaded || cached.loading)) {
      return cached.exports;
    }

    const nativeModule = new NativeModule(id);

    nativeModule.cache();
    nativeModule.compile();

    return nativeModule.exports;
  }

  // 缓存模块
  cache() {
    NativeModule.map.set(this.id, this);
  }

  // 获取缓存
  static getCached(id) {
    return NativeModule.map.get(id);
  }

  // 获取目录名
  static dirname(id) {
    if (StringPrototypeIncludes(id, '/')) {
      return StringPrototypeSlice(
        id, 0, 
        id.lastIndexOf('/')
      );
    }
    return '';
  }

  // 检查是否存在
  static exists(id) {
    return NativeModule.map.has(id) || 
           internalBinding('natives').hasOwnProperty(id);
  }
}

// 导出
return {
  NativeModule,
  require: NativeModule.require.bind(NativeModule),
};
```

```javascript
  let mod = bindingObj[module];
  if (typeof mod !== 'object') {
```

```cpp
// src/node_binding.cc
void GetInternalBinding(
  const FunctionCallbackInfo<Value>& args
) {
  Environment* env = Environment::GetCurrent(args);
  Isolate* isolate = env->isolate();
  
  // 获取模块名
  Local<String> module = args[0].As<String>();
  Utf8Value module_v(isolate, module);
  
  // 查找模块
  node_module* mod = get_linked_module(*module_v);
  
  if (mod == nullptr) {
    return;  // 模块不存在
  }
  
  // 初始化模块
  Local<Object> exports = Object::New(isolate);
  mod->nm_context_register_func(
    exports,
    Undefined(isolate),
    env->context(),
    mod->nm_priv
  );
  
  args.GetReturnValue().Set(exports);
}
```

### 3. primordials 保护

```javascript
// lib/internal/bootstrap/primordials.js
'use strict';

// 保存原始内置对象，防止用户代码篡改
function copyPrototype(src, dest, prefix) {
  const keys = Reflect.ownKeys(src);
  
  for (let i = 0; i < keys.length; i++) {
    const key = keys[i];
    
    if (typeof key === 'symbol') continue;
    
    const desc = Reflect.getOwnPropertyDescriptor(src, key);
    
    if (typeof desc.value === 'function') {
      const name = `${prefix}${key}`;
      dest[name] = desc.value;
    }
  }
}

// 创建 primordials 对象
function makePrimordials() {
  const primordials = {};

  // Array
  copyPrototype(Array.prototype, primordials, 'ArrayPrototype');
  primordials.Array = Array;

  // Object
  copyPrototype(Object.prototype, primordials, 'ObjectPrototype');
  copyPrototype(Object, primordials, 'Object');

  // String
  copyPrototype(String.prototype, primordials, 'StringPrototype');
  primordials.String = String;

  // Number
  copyPrototype(Number.prototype, primordials, 'NumberPrototype');
  primordials.Number = Number;

  // ... 更多内置对象

  // 冻结 primordials
  Object.freeze(primordials);

  return primordials;
}

return makePrimordials();
```

**为什么需要 primordials？**
```javascript
// ❌ 用户代码可能污染内置对象

  console.log('Hacked!');
```

```javascript
// lib/internal/process/pre_execution.js
function setupProcessObject(rawMethods) {
  const {
    _rawDebug,
    hrtime,
    hrtimeBigInt,
    cpuUsage,
    memoryUsage,
    resourceUsage,
    umask,
  } = rawMethods;

  // 绑定方法到 process
  process._rawDebug = _rawDebug;
  process.hrtime = hrtime;
  process.hrtime.bigint = hrtimeBigInt;
  process.cpuUsage = cpuUsage;
  process.memoryUsage = memoryUsage;
  process.resourceUsage = resourceUsage;
  process.umask = umask;

  // 设置 process.title
  const title = `node ${process.argv.join(' ')}`;
  process.title = title;

  // 设置 process.version
  process.version = NODE_VERSION;
  process.versions = {
    node: NODE_VERSION,
    v8: V8_VERSION,
    uv: UV_VERSION,
    zlib: ZLIB_VERSION,
    openssl: OPENSSL_VERSION,
    modules: MODULE_VERSION,
    // ...
  };

  // 设置 process.env（代理到 C++）
  const envProxy = new Proxy({}, {
    get(target, property) {
      return internalBinding('env').get(property);
    },
    set(target, property, value) {
      internalBinding('env').set(property, String(value));
      return true;
    },
    deleteProperty(target, property) {
      internalBinding('env').delete(property);
      return true;
    },
    ownKeys() {
      return internalBinding('env').keys();
    },
  });

  process.env = envProxy;
}
```

```javascript
// lib/internal/main/run_main_module.js
'use strict';

const {
  prepareMainThreadExecution,
} = require('internal/process/pre_execution');

const { isESM } = require('internal/modules/cjs/loader');

prepareMainThreadExecution(true);

// 标记应用启动
markBootstrapComplete();

// 获取用户主模块路径
const mainScript = process.argv[1];

if (mainScript === undefined) {
  // REPL 模式
  const repl = require('internal/repl');
  repl.start();
} else {
  // 判断模块类型
  if (isESM(mainScript)) {
    // ES Module
    const { loadESM } = require('internal/process/esm_loader');
    loadESM(() => {
      const esmLoader = require('internal/modules/esm/loader');
      return esmLoader.import(pathToFileURL(mainScript).href);
    });
  } else {
    // CommonJS Module
    const CJSModule = require('internal/modules/cjs/loader').Module;
    CJSModule._load(mainScript, null, true);
  }
}
```

## 📝 总结

### 核心要点

1. **ExecuteBootstrapper**：从 C++ 执行 JavaScript 引导代码
2. **NativeModule**：内置模块加载系统
3. **internalBinding**：JavaScript 访问 C++ 模块
4. **primordials**：保护内置对象不被篡改
5. **Bootstrap 阶段**：loaders → node → run_main_module

### 实现关键

1. **源码嵌入**：C++ 编译时将 JavaScript 嵌入二进制
2. **模块缓存**：NativeModule 使用 Map 缓存已加载模块
3. **安全隔离**：primordials 防止用户代码污染
4. **分阶段加载**：逐步初始化 Node.js 环境

### 应用价值

1. **性能优化**：理解启动流程，优化启动速度
2. **安全加固**：学习 primordials 保护机制
3. **模块开发**：理解模块系统工作原理
4. **调试技巧**：追踪引导过程定位问题

### 注意事项

1. **版本差异**：Bootstrap 流程在不同版本可能不同
2. **内部 API**：`internal/*` 模块不对外暴露
3. **性能影响**：Bootstrap 过程影响启动速度
4. **安全风险**：避免覆盖 primordials
