#### 2. node.cc - 核心启动逻辑

```cpp
// src/node.cc（简化版）
namespace node {

// 主启动函数
int Start(int argc, char** argv) {
  // 1. 初始化 V8 Platform
  std::unique_ptr<MultiIsolatePlatform> platform = 
    MultiIsolatePlatform::Create(4);  // 4个线程
  
  // 2. 初始化 V8
  V8::InitializePlatform(platform.get());
  V8::Initialize();
  
  // 3. 创建 Isolate
  Isolate::CreateParams params;
  params.array_buffer_allocator = 
    ArrayBuffer::Allocator::NewDefaultAllocator();
  
  Isolate* isolate = Isolate::New(params);
  
  {
    Isolate::Scope isolate_scope(isolate);
    HandleScope handle_scope(isolate);
    
    // 4. 创建 Context
    Local<Context> context = Context::New(isolate);
    Context::Scope context_scope(context);
    
    // 5. 创建 Environment
    Environment* env = CreateEnvironment(
      isolate, 
      context, 
      argc, 
      argv
    );
    
    // 6. 加载环境
    LoadEnvironment(env);
    
    // 7. 启动事件循环
    do {
      uv_run(env->event_loop(), UV_RUN_DEFAULT);
    } while (more_work_to_do());
  }
  
  // 8. 清理
  isolate->Dispose();
  V8::Dispose();
  V8::ShutdownPlatform();
  
  return exit_code;
}

} // namespace node
```

**启动流程：**
1. 初始化 V8 Platform（多线程支持）
2. 初始化 V8 引擎
3. 创建 Isolate（V8 隔离实例）
4. 创建 Context（执行上下文）
5. 创建 Environment（Node.js 环境）
6. 加载 JavaScript 引导代码
7. 启动 libuv 事件循环
8. 清理资源

#### 3. node_binding.cc - Binding 机制

```cpp
// src/node_binding.cc
namespace node {

// 内置模块注册表
std::map<std::string, node::binding::InitializerFunction> 
  binding_map;

// 注册内置模块
void RegisterBuiltinModule(
  const char* name,
  binding::InitializerFunction fn
) {
  binding_map[name] = fn;
}

// 获取内置模块
void GetInternalBinding(
  const FunctionCallbackInfo<Value>& args
) {
  Environment* env = Environment::GetCurrent(args);
  Isolate* isolate = env->isolate();
  
  // 获取模块名
  std::string module_name(*String::Utf8Value(isolate, args[0]));
  
  // 查找模块
  auto it = binding_map.find(module_name);
  if (it == binding_map.end()) {
    return;  // 模块不存在
  }
  
  // 初始化模块
  Local<Object> exports = Object::New(isolate);
  it->second(exports, Local<Value>(), env->context(), nullptr);
  
  args.GetReturnValue().Set(exports);
}

} // namespace node
```

**Binding 流程：**
```
JavaScript                    C++
    ↓                          ↑
internalBinding('fs')    GetInternalBinding()
    ↓                          ↑
查找 binding_map         RegisterBuiltinModule()
    ↓                          ↑
执行初始化函数           fs_module::Initialize()
    ↓                          ↑
返回 exports 对象         返回 C++ 函数包装
```

### 关键模块详解

#### 1. internal/bootstrap/loaders.js

```javascript
// lib/internal/bootstrap/loaders.js（简化版）
'use strict';

const {
  ObjectCreate,
  ObjectDefineProperty,
  SafeMap
} = primordials;

// NativeModule 系统
class NativeModule {
  constructor(id) {
    this.id = id;
    this.exports = {};
    this.loaded = false;
    this.loading = false;
  }

  // 缓存
  static map = new SafeMap();

  // 编译模块
  compile() {
    const id = this.id;
    
    // 获取源码（从 C++ 传入）
    const source = NativeModule.getSource(id);
    
    // 包装成函数
    const fn = NativeModule.wrap(source);
    
    // 执行
    const exports = this.exports;
    const module = this;
    
    fn(exports, require, module, __filename, __dirname);
    
    this.loaded = true;
  }

  // 包装源码
  static wrap(script) {
    return `(function (exports, require, module, __filename, __dirname) {
      ${script}
    });`;
  }

  // 从 C++ 获取源码
  static getSource(id) {
    return internalBinding('natives')[id];
  }

  // 加载模块
  static require(id) {
    if (NativeModule.map.has(id)) {
      return NativeModule.map.get(id).exports;
    }

    const nativeModule = new NativeModule(id);
    nativeModule.compile();
    NativeModule.map.set(id, nativeModule);

    return nativeModule.exports;
  }
}

// 导出
module.exports = { NativeModule };
```

**工作原理：**
1. C++ 将 JavaScript 源码编译进二进制
2. `internalBinding('natives')` 获取源码
3. 包装成函数并执行
4. 缓存导出对象

  SafeMap,

const path = require('path');
const fs = require('fs');
const vm = require('vm');

function Module(id, parent) {
  this.id = id;
  this.path = path.dirname(id);
  this.exports = {};
  this.parent = parent;
  this.filename = null;
  this.loaded = false;
  this.children = [];
}

// 模块缓存
Module._cache = ObjectCreate(null);

// 加载模块
Module.prototype.load = function(filename) {
  this.filename = filename;
  
  const extension = path.extname(filename) || '.js';
  
  // 根据扩展名选择加载器
  Module._extensions[extension](this, filename);
  
  this.loaded = true;
};

// .js 文件加载器
Module._extensions['.js'] = function(module, filename) {
  // 读取文件
  const content = fs.readFileSync(filename, 'utf8');
  
  // 编译
  module._compile(content, filename);
};

// 编译模块
Module.prototype._compile = function(content, filename) {
  const wrapper = Module.wrap(content);
  
  const compiledWrapper = vm.runInThisContext(wrapper, {
    filename,
    lineOffset: 0,
    displayErrors: true
  });
  
  const dirname = path.dirname(filename);
  const require = makeRequireFunction(this);
  
  compiledWrapper.call(
    this.exports,
    this.exports,
    require,
    this,
    filename,
    dirname
  );
};

// 包装函数
Module.wrap = function(script) {
  return `(function (exports, require, module, __filename, __dirname) {
    ${script}
  });`;
};

module.exports = Module;
