# 4.4 与 CommonJS 互操作：createRequire 与互操作机制

> 深入理解 ESM 与 CJS 的互操作实现与限制

---

## 📋 核心概念

ESM 与 CommonJS 的互操作是 Node.js 模块系统中的**关键挑战**之一。

**核心问题：**
- ESM 如何导入 CJS 模块？
- CJS 如何导入 ESM 模块？
- Named Exports 如何检测？
- createRequire 如何工作？

**互操作规则：**
```
ESM → CJS：✅ 支持（默认导出 + Named Exports检测）
CJS → ESM：❌ 不支持（需要 import() 动态导入）
```

---

## 💻 ESM 导入 CJS

### 1. 默认导出

```javascript
// cjs-module.cjs
module.exports = {
  foo: 'bar',
  method() {
    return 'hello';
  }
};

// esm-consumer.mjs
import pkg from './cjs-module.cjs';

console.log(pkg.foo);      // 'bar'
console.log(pkg.method()); // 'hello'
```

**实现原理：**
```javascript
// lib/internal/modules/esm/translators.js

// 🔥 CJS → ESM 翻译器
function cjsModuleToESM(module) {
  const namespace = Object.create(null);
  
  // 🔥 默认导出为 module.exports
  namespace.default = module.exports;
  
  return namespace;
}
```

---

### 2. Named Exports 检测

```javascript
// cjs-with-named.cjs
exports.foo = 'bar';
exports.method = function() {
  return 'hello';
};

// esm-consumer.mjs
// 🔥 自动检测 Named Exports
import { foo, method } from './cjs-with-named.cjs';

console.log(foo);      // 'bar'
console.log(method()); // [Function]
```

**检测算法：**
```javascript
// lib/internal/modules/cjs/loader.js

/**
 * 分析 CJS 模块的导出
 */
function analyzeModuleExports(source) {
  const ast = parseSource(source);
  const exports = [];

  // 🔥 遍历 AST 查找 exports.xxx
  walkAST(ast, (node) => {
    if (node.type === 'AssignmentExpression') {
      const { left, right } = node;
      
      // 🔥 exports.name = ...
      if (
        left.type === 'MemberExpression' &&
        left.object.name === 'exports' &&
        left.property.type === 'Identifier'
      ) {
        exports.push(left.property.name);
      }
      
      // 🔥 module.exports.name = ...
      if (
        left.type === 'MemberExpression' &&
        left.object.type === 'MemberExpression' &&
        left.object.object.name === 'module' &&
        left.object.property.name === 'exports' &&
        left.property.type === 'Identifier'
      ) {
        exports.push(left.property.name);
      }
    }
  });

  return exports;
}
```

---

### 3. 源码实现

```javascript
// lib/internal/modules/esm/translators.js

/**
 * CJS 模块翻译器
 */
translators.set('commonjs', async function(url, source, isMain) {
  const filename = fileURLToPath(url);
  
  // 🔥 加载 CJS 模块
  const module = loadCJSModule(filename, source, isMain);

  // 🔥 分析 Named Exports
  const namedExports = analyzeModuleExports(source);

  // 🔥 创建 ESM 命名空间
  const namespace = Object.create(null);

  // 🔥 默认导出
  Object.defineProperty(namespace, 'default', {
    enumerable: true,
    get() {
      return module.exports;
    }
  });

  // 🔥 Named Exports
  for (const name of namedExports) {
    if (name === 'default') continue;

    Object.defineProperty(namespace, name, {
      enumerable: true,
      get() {
        return module.exports[name];
      }
    });
  }

  return namespace;
});
```

## 💻 CJS 导入 ESM

### 1. 限制与解决方案

```javascript
// esm-module.mjs
export const foo = 'bar';
export function method() {
  return 'hello';
}

// cjs-consumer.cjs
// ❌ 不支持同步 require()
const pkg = require('./esm-module.mjs');  // Error!

// ✅ 使用动态 import()
(async () => {
  const pkg = await import('./esm-module.mjs');
  console.log(pkg.foo);      // 'bar'
  console.log(pkg.method()); // [Function]
})();
```

**原因：**
- CJS `require()` 是同步的
- ESM 可能包含 Top-Level Await（异步）
- 无法将异步转为同步

---

### 2. 条件导出解决

```json
// package.json
{
  "exports": {
    ".": {
      "import": "./index.mjs",
      "require": "./index.cjs"
    }
  }
}
```

```javascript
// index.mjs (ESM)
export const foo = 'bar';

// index.cjs (CJS)
module.exports = { foo: 'bar' };

// 🔥 两种方式都可用
// ESM
import { foo } from 'my-package';

// CJS
const { foo } = require('my-package');
```

## 💻 createRequire API

### 1. 基本用法

```javascript
// esm-module.mjs
import { createRequire } from 'module';
const require = createRequire(import.meta.url);

// 🔥 在 ESM 中使用 require
const fs = require('fs');
const lodash = require('lodash');

console.log(__dirname);  // ❌ Error
console.log(require.resolve('lodash'));  // ✅ 'path/to/lodash'
```

---

### 2. 源码实现

```javascript
// lib/module.js

/**
 * 创建 require 函数
 */
function createRequire(filename) {
  // 🔥 转换为文件路径
  if (filename instanceof URL) {
    filename = fileURLToPath(filename);
  }

  // 🔥 创建模拟模块
  const module = new Module(filename);
  module.filename = filename;
  module.paths = Module._nodeModulePaths(path.dirname(filename));

  // 🔥 绑定 require 函数
  function require(id) {
    return module.require(id);
  }

  // 🔥 添加 resolve 方法
  require.resolve = function resolve(request, options) {
    return Module._resolveFilename(request, module, false, options);
  };

  require.resolve.paths = function paths(request) {
    return Module._resolveLookupPaths(request, module);
  };

  // 🔥 添加 cache 和 extensions
  require.cache = Module._cache;
  require.extensions = Module._extensions;

  // 🔥 添加 main
  require.main = process.mainModule;

  return require;
}

module.exports = { createRequire };
```

## 💻 互操作陷阱

### 1. module.exports 覆盖

```javascript
// cjs-module.cjs
exports.foo = 'bar';

// ❌ 覆盖 exports
module.exports = function() {
  return 'hello';
};

// esm-consumer.mjs
import pkg from './cjs-module.cjs';
import { foo } from './cjs-module.cjs';  // ❌ Error! foo 不存在

console.log(pkg);      // [Function]
console.log(pkg.foo);  // undefined
```

**解决：**
```javascript
// ✅ 保持一致
module.exports = {
  default: function() {
    return 'hello';
  },
  foo: 'bar'
};

// 或者
function main() {
  return 'hello';
}
main.foo = 'bar';
module.exports = main;
```

---

### 2. this 绑定问题

```javascript
// cjs-module.cjs
module.exports = {
  value: 42,
  getValue() {
    return this.value;  // 🔥 this 依赖
  }
};

// esm-consumer.mjs
import { getValue } from './cjs-module.cjs';

// ❌ this 丢失
console.log(getValue());  // undefined

// ✅ 导入整个对象
import pkg from './cjs-module.cjs';
console.log(pkg.getValue());  // 42
```

---

### 3. 循环依赖差异

```javascript
// ESM：Live Binding
// a.mjs
export let value = 'initial';
import { bValue } from './b.mjs';
console.log('a:', bValue);
value = 'updated';

// b.mjs
import { value } from './a.mjs';
console.log('b:', value);  // undefined (TDZ)
export const bValue = 'b value';

// CJS：值拷贝
// a.js
exports.value = 'initial';
const b = require('./b.js');
console.log('a:', b.bValue);  // 'b value'
exports.value = 'updated';

// b.js
const a = require('./a.js');
console.log('b:', a.value);  // 'initial'
exports.bValue = 'b value';
```

## 📝 总结

### 核心要点

1. **ESM→CJS**：支持，默认导出+Named Exports检测
2. **CJS→ESM**：不支持同步require，需用import()
3. **createRequire**：在ESM中使用require
4. **互操作陷阱**：module.exports覆盖、this绑定、循环依赖

### 实现关键

1. **cjsModuleToESM**：CJS→ESM翻译器
2. **analyzeModuleExports**：Named Exports检测
3. **createRequire**：创建require函数
4. **条件导出**：双模式包支持

### 应用价值

1. **平滑迁移**：CJS→ESM渐进式迁移
2. **生态兼容**：支持现有CJS包
3. **工具集成**：Bundlers、TypeScript支持
4. **性能优化**：合理选择模块格式
