# 4.3 条件导出（exports 字段）解析算法

> 深入理解 package.json exports 字段的完整解析机制

---

## 📋 核心概念

`exports` 字段是 package.json 中定义**模块导出**的现代方式，支持条件导出、子路径导出等高级特性。

**核心问题：**
- exports 字段如何解析？
- 条件导出的优先级？
- Subpath Exports 如何工作？
- 与 main 字段的关系？

**支持的条件：**
```json
{
  "import": "ESM 导入",
  "require": "CommonJS 导入", 
  "node": "Node.js 环境",
  "default": "默认回退",
  "types": "TypeScript 类型",
  "development": "开发环境",
  "production": "生产环境"
}
```

```json

  "name": "my-package",
  "exports": "./index.js"
```

```json
  "exports": {
```

## 💻 解析算法源码

### 1. packageExportsResolve 核心算法

```javascript
// lib/internal/modules/esm/resolve.js

/**
 * 解析 package exports
 * @param {string} packageURL - package.json 所在目录 URL
 * @param {string} subpath - 子路径（如 ".", "./utils"）
 * @param {object} packageConfig - package.json 配置
 * @param {string} base - 基础 URL
 * @param {Set<string>} conditions - 条件集合
 */
function packageExportsResolve(
  packageURL,
  subpath,
  packageConfig,
  base,
  conditions
) {
  // 🔥 获取 exports 字段
  const exports = packageConfig.exports;

  if (exports == null) {
    throw new Error('Package exports not defined');
  }

  // 🔥 解析 exports
  const resolved = packageResolve(
    subpath,
    exports,
    packageURL,
    base,
    conditions
  );

  if (resolved == null) {
    throwExportsNotFound(packageURL, subpath, base);
  }

  return resolved;
}

/**
 * 递归解析 exports 对象
 */
function packageResolve(
  subpath,
  exports,
  packageURL,
  base,
  conditions
) {
  // 🔥 情况1：exports 是字符串
  if (typeof exports === 'string') {
    return resolveExportsTarget(packageURL, exports, '', false, conditions);
  }

  // 🔥 情况2：exports 是数组（回退列表）
  if (Array.isArray(exports)) {
    for (const item of exports) {
      const resolved = packageResolve(subpath, item, packageURL, base, conditions);
      if (resolved != null) {
        return resolved;
      }
    }
    return null;
  }

  // 🔥 情况3：exports 是对象
  if (typeof exports === 'object' && exports !== null) {
    // 🔥 检查是否有条件键
    if (hasConditionalExports(exports)) {
      const target = conditionalExportsResolve(
        exports,
        conditions,
        packageURL
      );
      
      if (target !== undefined) {
        return packageResolve(subpath, target, packageURL, base, conditions);
      }
      
      return null;
    }

    // 🔥 子路径匹配
    if (subpath in exports) {
      const target = exports[subpath];
      return packageResolve('', target, packageURL, base, conditions);
    }

    // 🔥 通配符匹配（如 "./features/*"）
    const matches = getWildcardMatches(subpath, exports);
    if (matches) {
      const { key, pattern, target } = matches;
      const resolved = resolveExportsTarget(
        packageURL,
        target,
        pattern,
        false,
        conditions
      );
      return resolved;
    }
  }

  return null;
}
```

### 2. 条件导出解析

```javascript
// lib/internal/modules/esm/resolve.js

/**
 * 解析条件导出
 */
function conditionalExportsResolve(
  exports,
  conditions,
  packageURL
) {
  // 🔥 遍历所有键
  for (const key of Object.keys(exports)) {
    // 🔥 检查是否为条件键
    if (isConditionalExport(key)) {
      // 🔥 检查条件是否满足
      if (conditions.has(key)) {
        const target = exports[key];
        
        // 🔥 递归解析嵌套条件
        if (typeof target === 'object' && target !== null) {
          const resolved = conditionalExportsResolve(
            target,
            conditions,
            packageURL
          );
          
          if (resolved !== undefined) {
            return resolved;
          }
        } else {
          return target;
        }
      }
    }
  }

  // 🔥 检查 default 回退
  if ('default' in exports) {
    return exports.default;
  }

  return undefined;
}

/**
 * 检查是否为条件键
 */
function isConditionalExport(key) {
  const conditionalKeys = [
    'import',
    'require',
    'node',
    'node-addons',
    'default',
    'types',
    'browser',
    'development',
    'production'
  ];

  return conditionalKeys.includes(key);
}

/**
 * 获取默认条件集合
 */
function getDefaultConditions() {
  const conditions = new Set(['node', 'import']);

  // 🔥 根据环境添加条件
  if (process.env.NODE_ENV === 'development') {
    conditions.add('development');
  } else if (process.env.NODE_ENV === 'production') {
    conditions.add('production');
  }

  // 🔥 总是添加 default
  conditions.add('default');

  return conditions;
}
```

### 3. 通配符匹配

```javascript
// lib/internal/modules/esm/resolve.js

/**
 * 获取通配符匹配
 */
function getWildcardMatches(subpath, exports) {
  for (const [key, target] of Object.entries(exports)) {
    // 🔥 检查是否包含通配符
    if (!key.includes('*')) {
      continue;
    }

    // 🔥 生成匹配正则
    const pattern = key.split('*').map(escapeRegExp).join('(.*)');
    const regex = new RegExp(`^${pattern}$`);

    const match = subpath.match(regex);
    
    if (match) {
      // 🔥 提取通配符部分
      const wildcardValue = match[1];

      // 🔥 替换目标中的通配符
      let resolvedTarget = target;
      if (typeof target === 'string') {
        resolvedTarget = target.replace('*', wildcardValue);
      }

      return {
        key,
        pattern: wildcardValue,
        target: resolvedTarget
      };
    }
  }

  return null;
}

function escapeRegExp(string) {
  return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
```

### 4. 解析目标路径

```javascript
// lib/internal/modules/esm/resolve.js

/**
 * 解析 exports 目标
 */
function resolveExportsTarget(
  packageURL,
  target,
  pattern,
  isImports,
  conditions
) {
  // 🔥 null 或 undefined：未找到
  if (target == null) {
    return null;
  }

  // 🔥 字符串：直接解析
  if (typeof target === 'string') {
    // 🔥 检查是否为有效路径
    if (!target.startsWith('./')) {
      throw new Error('Invalid exports target: must start with "./"');
    }

    // 🔥 替换通配符
    if (pattern) {
      target = target.replace('*', pattern);
    }

    // 🔥 解析为绝对 URL
    const resolved = new URL(target, packageURL);

    // 🔥 检查是否在 package 目录内
    if (!resolved.href.startsWith(packageURL.href)) {
      throw new Error('Invalid exports target: escapes package');
    }

    return resolved.href;
  }

  // 🔥 数组：尝试每个选项
  if (Array.isArray(target)) {
    for (const item of target) {
      const resolved = resolveExportsTarget(
        packageURL,
        item,
        pattern,
        isImports,
        conditions
      );
      
      if (resolved != null) {
        return resolved;
      }
    }
    
    return null;
  }

  // 🔥 对象：条件导出
  if (typeof target === 'object' && target !== null) {
    const resolved = conditionalExportsResolve(
      target,
      conditions,
      packageURL
    );
    
    if (resolved !== undefined) {
      return resolveExportsTarget(
        packageURL,
        resolved,
        pattern,
        isImports,
        conditions
      );
    }
  }

  return null;
}
```

## 💻 条件优先级

### 1. 条件匹配顺序

```javascript
// 条件按定义顺序匹配，第一个满足的条件生效

{
  "exports": {
    ".": {
      "node": "./node.js",        // 1. 优先匹配
      "import": "./esm.mjs",      // 2. 其次
      "require": "./cjs.cjs",     // 3. 再次
      "default": "./default.js"   // 4. 最后回退
    }
  }
}
```

**匹配逻辑：**
```javascript
// 如果 conditions = ['node', 'import', 'default']
// 结果：./node.js

// 如果 conditions = ['import', 'default']
// 结果：./esm.mjs

// 如果 conditions = ['require', 'default']
// 结果：./cjs.cjs

// 如果 conditions = ['default']
// 结果：./default.js
```

### 2. 解析算法

```javascript
// lib/internal/modules/esm/resolve.js

/**
 * 解析 package imports
 */
function packageImportsResolve(
  specifier,
  parentURL,
  conditions
) {
  // 🔥 只支持 # 开头的标识符
  if (!specifier.startsWith('#')) {
    throw new Error('Invalid imports specifier');
  }

  // 🔥 查找 package.json
  const packageConfig = getPackageConfig(parentURL);

  if (!packageConfig || !packageConfig.imports) {
    throw new Error('Package imports not defined');
  }

  // 🔥 解析 imports
  return packageImportsExportsResolve(
    specifier,
    packageConfig.imports,
    packageConfig.url,
    conditions
  );
}
```

## 💻 常见错误

### 1. 路径必须以 ./ 开头

```json
{
  "exports": {
    // ❌ 错误
    ".": "index.js",
    
    // ✅ 正确
    ".": "./index.js"
  }
}
```

---

### 2. 不能逃逸包目录

```json
{
  "exports": {
    // ❌ 错误：向上逃逸
    ".": "../outside.js",
    
    // ✅ 正确：在包内
    ".": "./src/index.js"
  }
}
```

---

### 3. 条件必须是对象

```json
{
  "exports": {
    ".": {
      // ❌ 错误：条件值不能是 null
      "import": null,
      
      // ✅ 正确
      "import": "./index.mjs"
    }
  }
}
```

```javascript
// resolve-cache.js
const resolveCache = new Map();

function cachedResolve(specifier, parentURL, conditions) {
  const cacheKey = JSON.stringify({ specifier, parentURL, conditions });

  if (resolveCache.has(cacheKey)) {
    return resolveCache.get(cacheKey);
  }

  const result = packageExportsResolve(specifier, parentURL, conditions);
  resolveCache.set(cacheKey, result);

  return result;
}
```

## 📝 总结

### 核心要点

1. **exports字段**：现代模块导出规范
2. **条件导出**：支持import/require/node等条件
3. **子路径导出**：支持通配符和精确匹配
4. **优先级**：按定义顺序，第一个满足的条件生效

### 实现关键

1. **packageExportsResolve**：核心解析算法
2. **conditionalExportsResolve**：条件匹配
3. **getWildcardMatches**：通配符处理
4. **resolveExportsTarget**：目标路径解析

### 应用价值

1. **多环境支持**：Node/Browser/Development/Production
2. **模块封装**：精确控制导出内容
3. **类型支持**：TypeScript类型导出
4. **性能优化**：按需加载不同版本
