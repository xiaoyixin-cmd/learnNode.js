# 5.5 如何优化代码以适应 V8（避免去优化）

> 编写 V8 友好的高性能 JavaScript 代码

---

## 📋 核心概念

理解 V8 的优化机制，编写**对 V8 友好的代码**，避免反优化。

**核心问题：**
- 哪些代码模式会导致反优化？
- 如何保持函数单态？
- 什么是 Fast Elements？
- 如何减少 GC 压力？

**优化原则：**
```
1. 保持类型稳定
2. 保持隐藏类稳定
3. 避免动态操作
4. 减少内存分配
```

---

## 💻 避免反优化

### 1. 常见反优化原因

```javascript
// 使用 --allow-natives-syntax 查看优化状态

function add(a, b) {
  return a + b;
}

// 预热
for (let i = 0; i < 10000; i++) {
  add(i, i);
}

// 强制优化
%OptimizeFunctionOnNextCall(add);
add(1, 2);

console.log(%GetOptimizationStatus(add));
// 1: 已优化

// 🔥 触发反优化：类型改变
add('hello', 'world');

console.log(%GetOptimizationStatus(add));
// 2: 已反优化
```

---

### 2. 类型不稳定

```javascript
// ❌ 坏：类型不稳定
function process(value) {
  if (typeof value === 'number') {
    return value * 2;
  } else if (typeof value === 'string') {
    return value.toUpperCase();
  }
  return value;
}

// TurboFan 优化假设：value 总是数字
// 传入字符串 → 假设失效 → 反优化

// ✅ 好：类型稳定
function processNumber(value) {
  return value * 2;
}

function processString(value) {
  return value.toUpperCase();
}

// 调用方根据类型选择函数
if (typeof value === 'number') {
  processNumber(value);
} else if (typeof value === 'string') {
  processString(value);
}
```

### 3. Try-Catch 性能影响

```javascript
// ❌ 坏：try-catch 导致反优化
function riskyOperation(data) {
  try {
    return JSON.parse(data);
  } catch (e) {
    return null;
  }
}

// TurboFan 不会优化包含 try-catch 的函数

// ✅ 好：隔离 try-catch
function parseJSON(data) {
  return JSON.parse(data);  // 可被优化
}

function safeParseJSON(data) {
  try {
    return parseJSON(data);
  } catch (e) {
    return null;
  }
}
```

### 4. Arguments 对象

```javascript
// ❌ 坏：使用 arguments
function sum() {
  let total = 0;
  for (let i = 0; i < arguments.length; i++) {
    total += arguments[i];  // 🔥 arguments 导致反优化
  }
  return total;
}

// ✅ 好：使用 rest 参数
function sum(...args) {
  let total = 0;
  for (let i = 0; i < args.length; i++) {
    total += args[i];
  }
  return total;
}

// ✅ 更好：使用 reduce
function sum(...args) {
  return args.reduce((acc, val) => acc + val, 0);
}
```

### 5. for-in 循环

```javascript
// ❌ 坏：for-in 遍历数组
const arr = [1, 2, 3, 4, 5];

for (let key in arr) {
  console.log(arr[key]);  // 🔥 慢！检查原型链
}

// ✅ 好：for-of 或传统 for
for (let item of arr) {
  console.log(item);
}

// 或
for (let i = 0; i < arr.length; i++) {
  console.log(arr[i]);
}
```

## 💻 保持隐藏类稳定

### 1. 固定属性初始化

```javascript
// ❌ 坏：条件属性
class Point {
  constructor(x, y, z) {
    this.x = x;
    this.y = y;
    if (z !== undefined) {
      this.z = z;  // 🔥 隐藏类分裂！
    }
  }
}

const p1 = new Point(1, 2);       // Map1: {x, y}
const p2 = new Point(1, 2, 3);    // Map2: {x, y, z}

// ✅ 好：总是初始化所有属性
class Point {
  constructor(x, y, z) {
    this.x = x;
    this.y = y;
    this.z = z ?? null;  // 总是初始化
  }
}

const p1 = new Point(1, 2);       // Map1: {x, y, z}
const p2 = new Point(1, 2, 3);    // Map1: {x, y, z}
```

---

### 2. 固定属性添加顺序

```javascript
// ❌ 坏：不同顺序
function createUser1(name, age) {
  return { name, age };  // Map1: name, age
}

function createUser2(name, age) {
  return { age, name };  // Map2: age, name
}

// ✅ 好：统一顺序
function createUser1(name, age) {
  return { name, age };  // Map1: name, age
}

function createUser2(name, age) {
  return { name, age };  // Map1: name, age
}
```

### 3. 避免删除属性

```javascript
// ❌ 坏：删除属性
const obj = { x: 1, y: 2, z: 3 };
delete obj.y;  // 🔥 破坏隐藏类，转为 Dictionary 模式

// ✅ 好：设为 undefined 或 null
const obj = { x: 1, y: 2, z: 3 };
obj.y = undefined;  // 保持隐藏类
```

---

### 4. 使用类而非对象字面量

```javascript
// ❌ 坏：对象字面量（难以共享隐藏类）
function createPoints(n) {
  const points = [];
  for (let i = 0; i < n; i++) {
    points.push({ x: i, y: i * 2 });  // 每次可能创建新 Map
  }
  return points;
}

// ✅ 好：使用类
class Point {
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }
}

function createPoints(n) {
  const points = [];
  for (let i = 0; i < n; i++) {
    points.push(new Point(i, i * 2));  // 共享同一个 Map
  }
  return points;
}
```

## 💻 Fast Elements 优化

### 1. 数组元素类型

V8 对数组使用不同的内部表示：

| 类型 | 描述 | 性能 |
|------|------|------|
| **PACKED_SMI_ELEMENTS** | 密集、小整数 | ⚡️ 最快 |
| **PACKED_DOUBLE_ELEMENTS** | 密集、浮点数 | ⚡️ 快 |
| **PACKED_ELEMENTS** | 密集、任意类型 | ✅ 中 |
| **HOLEY_SMI_ELEMENTS** | 稀疏、小整数 | ⚠️ 慢 |
| **HOLEY_ELEMENTS** | 稀疏、任意类型 | ❌ 很慢 |
| **DICTIONARY_ELEMENTS** | 哈希表 | ❌❌ 最慢 |

---

### 2. 保持数组密集

```javascript
// ❌ 坏：创建稀疏数组
const arr = [];
arr[0] = 1;
arr[1000] = 2;  // 🔥 变成 HOLEY_ELEMENTS 或 DICTIONARY_ELEMENTS

// ✅ 好：保持密集
const arr = [];
for (let i = 0; i < 1001; i++) {
  arr[i] = i === 1000 ? 2 : (i === 0 ? 1 : 0);
}
```

---

### 3. 保持类型一致

```javascript
// ❌ 坏：混合类型
const arr = [1, 2, 3];           // PACKED_SMI_ELEMENTS
arr.push(4.5);                   // 🔥 降级为 PACKED_DOUBLE_ELEMENTS
arr.push('hello');               // 🔥 降级为 PACKED_ELEMENTS

// ✅ 好：类型一致
const numbers = [1, 2, 3, 4, 5];        // PACKED_SMI_ELEMENTS
const strings = ['a', 'b', 'c'];         // PACKED_ELEMENTS
```

---

### 4. 避免数组空洞

```javascript
// ❌ 坏：delete 创建空洞
const arr = [1, 2, 3, 4, 5];
delete arr[2];  // 🔥 [1, 2, <empty>, 4, 5] → HOLEY_ELEMENTS

// ✅ 好：使用 splice
const arr = [1, 2, 3, 4, 5];
arr.splice(2, 1);  // [1, 2, 4, 5] → 保持 PACKED
```

---

### 5. 预分配数组

```javascript
// ❌ 坏：动态增长
const arr = [];
for (let i = 0; i < 100000; i++) {
  arr.push(i);  // 🔥 多次重新分配内存
}

// ✅ 好：预分配
const arr = new Array(100000);
for (let i = 0; i < 100000; i++) {
  arr[i] = i;
}
```

## 💻 减少 GC 压力

### 1. 对象池

```javascript
class ObjectPool {
  constructor(factory, size = 100) {
    this.factory = factory;
    this.pool = [];
    
    // 预分配
    for (let i = 0; i < size; i++) {
      this.pool.push(factory());
    }
  }

  acquire() {
    return this.pool.pop() || this.factory();
  }

  release(obj) {
    if (obj.reset) obj.reset();
    if (this.pool.length < 1000) {  // 限制池大小
      this.pool.push(obj);
    }
  }
}

// 使用
const vectorPool = new ObjectPool(() => ({ x: 0, y: 0 }));

function simulate() {
  const v = vectorPool.acquire();
  v.x = Math.random();
  v.y = Math.random();
  // 使用 v
  vectorPool.release(v);  // 归还
}
```

### 2. 复用 Buffer

```javascript
// ❌ 坏：频繁分配
function processData(chunks) {
  for (const chunk of chunks) {
    const buffer = Buffer.alloc(1024);  // 🔥 每次都分配
    // 处理 buffer
  }
}

// ✅ 好：复用 Buffer
function processData(chunks) {
  const buffer = Buffer.alloc(1024);  // 只分配一次
  
  for (const chunk of chunks) {
    buffer.fill(0);  // 重置
    // 处理 buffer
  }
}
```

### 3. 避免闭包陷阱

```javascript
// ❌ 坏：闭包持有大对象
function createHandlers() {
  const largeData = new Array(1000000).fill(0);  // 1MB
  
  return {
    get: () => largeData.length,  // 🔥 闭包持有整个数组
    set: (i, v) => { largeData[i] = v; }
  };
}

const handlers = createHandlers();  // 1MB 内存泄漏

// ✅ 好：只保留必要数据
function createHandlers() {
  const data = new Array(1000000).fill(0);
  const length = data.length;
  
  return {
    get: () => length,  // 只持有长度（8 bytes）
    set: (i, v) => { data[i] = v; }  // 只在需要时访问
  };
}
```

### 4. WeakMap/WeakSet

```javascript
// ❌ 坏：Map 持有强引用
const cache = new Map();

function process(obj) {
  if (!cache.has(obj)) {
    cache.set(obj, heavyComputation(obj));  // 🔥 obj 永不释放
  }
  return cache.get(obj);
}

// ✅ 好：WeakMap 弱引用
const cache = new WeakMap();

function process(obj) {
  if (!cache.has(obj)) {
    cache.set(obj, heavyComputation(obj));  // obj 可被 GC
  }
  return cache.get(obj);
}
```

## 💻 函数优化

### 1. 保持函数单态

```javascript
// ❌ 坏：多态函数
function getX(obj) {
  return obj.x;  // IC 点
}

getX({ x: 1, y: 2 });        // Map1
getX({ x: 2, z: 3 });        // Map2
getX({ x: 3, a: 4 });        // Map3
getX({ x: 4, b: 5 });        // Map4
getX({ x: 5, c: 6 });        // Map5  🔥 超多态！

// ✅ 好：单态函数
class Point {
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }
}

function getX(point) {
  return point.x;  // 总是 Point 类型 → 单态
}

getX(new Point(1, 2));
getX(new Point(3, 4));
```

### 2. 避免巨型函数

// ❌ 坏：巨型函数（> 600 字节字节码）

## 💻 实战案例

### 案例1：优化数据处理

```javascript
// ❌ 原始版本
function processUsers(users) {
  const result = [];
  
  for (let i = 0; i < users.length; i++) {
    const user = users[i];
    
    if (user.age > 18) {
      result.push({
        id: user.id,
        name: user.name.toUpperCase(),
        email: user.email.toLowerCase(),
        age: user.age
      });
    }
  }
  
  return result;
}

// ✅ 优化版本
class ProcessedUser {
  constructor(id, name, email, age) {
    this.id = id;
    this.name = name;
    this.email = email;
    this.age = age;
  }
}

function processUsers(users) {
  const result = [];
  const len = users.length;  // 缓存长度
  
  for (let i = 0; i < len; i++) {
    const user = users[i];
    
    if (user.age > 18) {
      // 使用类，共享隐藏类
      result.push(new ProcessedUser(
        user.id,
        user.name.toUpperCase(),
        user.email.toLowerCase(),
        user.age
      ));
    }
  }
  
  return result;
}

// 性能提升：~30%
```

### 案例2：优化矩阵运算

```javascript
// ❌ 原始版本
function multiplyMatrices(a, b) {
  const result = [];
  
  for (let i = 0; i < a.length; i++) {
    result[i] = [];
    for (let j = 0; j < b[0].length; j++) {
      let sum = 0;
      for (let k = 0; k < a[0].length; k++) {
        sum += a[i][k] * b[k][j];
      }
      result[i][j] = sum;
    }
  }
  
  return result;
}

// ✅ 优化版本
function multiplyMatrices(a, b) {
  const rows = a.length;
  const cols = b[0].length;
  const inner = a[0].length;
  
  // 预分配结果数组
  const result = new Array(rows);
  
  for (let i = 0; i < rows; i++) {
    result[i] = new Array(cols);
    const aRow = a[i];  // 缓存行
    
    for (let j = 0; j < cols; j++) {
      let sum = 0;
      
      for (let k = 0; k < inner; k++) {
        sum += aRow[k] * b[k][j];
      }
      
      result[i][j] = sum;
    }
  }
  
  return result;
}

// 性能提升：~50%
```

## 📝 总结

### 核心要点

1. **避免反优化**：类型稳定、避免 try-catch、不用 arguments
2. **隐藏类稳定**：固定属性、统一顺序、不删除属性
3. **Fast Elements**：密集数组、类型一致、避免空洞
4. **减少 GC**：对象池、复用 Buffer、避免闭包陷阱

### 实现关键

1. **单态函数**：参数类型一致，IC 优化
2. **小函数**：容易内联，容易优化
3. **预分配**：减少内存重新分配
4. **类型一致**：数组元素类型统一

### 应用价值

1. **高性能**：充分利用 V8 优化
2. **低延迟**：减少 GC 停顿
3. **可维护**：清晰的代码结构
4. **可测量**：使用 Benchmark 验证
