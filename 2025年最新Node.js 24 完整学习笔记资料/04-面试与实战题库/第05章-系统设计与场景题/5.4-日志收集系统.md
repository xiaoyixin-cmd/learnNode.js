## 💻 系统架构

### 1. 整体架构

```
应用服务（100+）
    ↓
Fluentd Agent（每台机器）
    ↓
Kafka Cluster（3节点）
    ↓
Logstash/Fluentd Consumer
    ↓
Elasticsearch Cluster（3节点）
    ↓
Kibana（查询、可视化）
```

### 2. 数据流

```
1. 应用输出日志 → 本地文件/stdout
2. Fluentd 采集 → 解析、过滤、标签
3. 发送到 Kafka → 缓冲、解耦
4. Consumer 消费 → 格式化、清洗
5. 写入 ES → 索引、存储
6. Kibana 查询 → 检索、聚合
```

### 3. 消费者（Node.js）

```javascript
const { Kafka } = require('kafkajs');
const { Client } = require('@elastic/elasticsearch');

const kafka = new Kafka({
  clientId: 'log-consumer',
  brokers: ['kafka1:9092', 'kafka2:9092', 'kafka3:9092']
});

const consumer = kafka.consumer({ groupId: 'log-group' });
const esClient = new Client({ node: 'http://localhost:9200' });

async function run() {
  await consumer.connect();
  await consumer.subscribe({ topic: 'app-logs', fromBeginning: false });
  
  await consumer.run({
    eachBatch: async ({ batch, resolveOffset, heartbeat }) => {
      const operations = [];
      
      for (let message of batch.messages) {
        const log = JSON.parse(message.value.toString());
        
        // 构建批量操作
        operations.push(
          { index: { _index: 'app-logs-' + getDateString(), _id: log.id } },
          log
        );
        
        // 每100条发送一次
        if (operations.length >= 200) {
          await esClient.bulk({ operations });
          operations.length = 0;
        }
        
        resolveOffset(message.offset);
        await heartbeat();
      }
      
      // 发送剩余数据
      if (operations.length > 0) {
        await esClient.bulk({ operations });
      }
    }
  });
}

function getDateString() {
  const now = new Date();
  return now.toISOString().split('T')[0];  // 2025-01-01
}

run().catch(console.error);
```

```javascript
// 全文检索
async function searchLogs(keyword) {
  const result = await client.search({
    index: 'app-logs-*',
    body: {
      query: {
        match: {
          message: keyword
        }
      },
      sort: [{ timestamp: 'desc' }],

// 聚合统计（按服务统计日志数）
async function aggregateLogs() {
  const result = await client.search({
    index: 'app-logs-*',
    body: {
      size: 0,
      aggs: {
        by_service: {
          terms: { field: 'service_name', size: 10 },
          aggs: {
            by_level: {
              terms: { field: 'level' }
            }
          }
        }
      }
    }
  });
  
  return result.aggregations.by_service.buckets;
}

    index: 'app-logs-*',
    body: {
      query: {
        term: { level: 'error' }
      }
```

## 💻 链路追踪

### 1. 生成 Trace ID

```javascript
const crypto = require('crypto');

function generateTraceId() {
  return crypto.randomBytes(16).toString('hex');  // 32位
}

function generateSpanId() {
  return crypto.randomBytes(8).toString('hex');   // 16位
}

// Express 中间件
function traceMiddleware(req, res, next) {
  // 从请求头获取或生成新的 Trace ID
  req.traceId = req.headers['x-trace-id'] || generateTraceId();
  req.spanId = generateSpanId();
  
  // 传递给下游服务
  res.setHeader('x-trace-id', req.traceId);
  
  next();
}

app.use(traceMiddleware);
```

### 2. 异步批量写入

```javascript
const { Client } = require('@elastic/elasticsearch');

class LogBuffer {
  constructor(esClient, maxSize = 100, flushInterval = 5000) {
    this.esClient = esClient;
    this.buffer = [];
    this.maxSize = maxSize;
    
    // 定时刷新
    setInterval(() => this.flush(), flushInterval);
  }
  
  add(log) {
    this.buffer.push(
      { index: { _index: 'app-logs-' + getDateString() } },
      log
    );
    
    if (this.buffer.length >= this.maxSize * 2) {
      this.flush();
    }
  }
  
  async flush() {
    if (this.buffer.length === 0) return;
    
    const operations = this.buffer.splice(0);
    
    try {
      await this.esClient.bulk({ operations });
      console.log(`Flushed ${operations.length / 2} logs`);
    } catch (error) {
      console.error('Flush error:', error);
      // 失败重试
      this.buffer.unshift(...operations);
    }
  }
}

const logBuffer = new LogBuffer(esClient);

// 使用
logBuffer.add({
  timestamp: new Date(),
  level: 'info',
  message: 'User login',
  userId: 123
});
```

## 🎓 面试标准答题

### 问题："如何设计一个分布式日志收集系统？"

**标准答案（10分钟）：**

**1. 整体架构（3分钟）**
```
应用 → Fluentd → Kafka → Consumer → ES → Kibana

核心组件：
- Fluentd：日志采集（轻量、插件丰富）
- Kafka：消息缓冲（高吞吐、持久化）
- Elasticsearch：存储查询（全文检索）
- Kibana：可视化
```

**2. 数据流（2分钟）**
```
1. 应用输出 JSON 日志
2. Fluentd 采集、解析、过滤
3. 发送到 Kafka（3副本、压缩）
4. Consumer 消费、批量写入 ES
5. Kibana 查询、聚合
```

**3. 关键技术（3分钟）**

**防止丢失：**
- Fluentd：Buffer 持久化
- Kafka：acks=-1（等待所有副本）
- Consumer：手动提交 offset

**链路追踪：**
```javascript
// 生成 Trace ID
req.traceId = generateTraceId();

// 日志携带
logger.info('...', { trace_id: req.traceId });
```

**4. 性能优化（2分钟）**
> - 采样：10% 日志
> - 批量写入：100条/批
> - 压缩：Gzip（减少70%）
