## 💻 短链生成算法

### 算法 1：自增 ID + Base62 编码

**原理：**
```
1. 为每个长链接分配唯一的自增 ID
2. 将 ID 转换为 Base62 编码（0-9, a-z, A-Z）
3. 作为短链的唯一标识

示例：
ID = 125 → Base62 = "1Z"
ID = 1000000 → Base62 = "4c92"
```

**实现：**
```javascript
// Base62 编码
const BASE62 = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';

function toBase62(num) {
  if (num === 0) return '0';
  
  let result = '';
  while (num > 0) {
    result = BASE62[num % 62] + result;
    num = Math.floor(num / 62);
  }
  return result;
}

function fromBase62(str) {
  let result = 0;
  for (let i = 0; i < str.length; i++) {
    result = result * 62 + BASE62.indexOf(str[i]);
  }
  return result;
}

// 使用
const id = 1000000;
const shortCode = toBase62(id);  // "4c92"
console.log(`https://short.url/${shortCode}`);
```

**优点：**
- ✅ 简单高效
- ✅ 保证唯一性（ID 唯一）
- ✅ 短链长度可预测

**缺点：**
- ❌ ID 可被预测（安全性低）
- ❌ 依赖数据库自增 ID（单点）

### 算法 2：Hash + 冲突检测

**原理：**
```
1. 对长链接进行 Hash（MD5/SHA256）
2. 取 Hash 值的前 N 位作为短链
3. 检测冲突，若冲突则加盐重新 Hash

示例：
URL = "https://example.com/very-long-url"
MD5 = "a1b2c3d4e5f6..."
短链 = "a1b2c3"（前 6 位）
```

**实现：**
```javascript
const crypto = require('crypto');

async function generateShortCode(longUrl) {
  let shortCode;
  let attempt = 0;
  
  while (true) {
    // Hash（加盐避免冲突）
    const hash = crypto
      .createHash('md5')
      .update(longUrl + attempt)
      .digest('hex');
    
    shortCode = toBase62(parseInt(hash.substring(0, 8), 16))
      .substring(0, 6);  // 取前 6 位
    
    // 检查冲突
    const exists = await redis.exists(`short:${shortCode}`);
    if (!exists) break;
    
    attempt++;
  }
  
  return shortCode;
}
```

**优点：**
- ✅ 无需依赖数据库自增
- ✅ 分布式友好

**缺点：**
- ❌ 可能冲突（需要检测）
- ❌ 无法预测长度

### 算法 3：雪花算法（Snowflake）

**原理：**
```
64 位 ID 结构：
| 1位符号 | 41位时间戳 | 10位机器ID | 12位序列号 |

优势：
- 全局唯一
- 趋势递增
- 分布式生成
```

**实现：**
```javascript
class SnowflakeIdGenerator {
  constructor(workerId) {
    this.workerId = workerId;
    this.sequence = 0;
    this.lastTimestamp = -1;
    
    this.workerIdBits = 10;
    this.sequenceBits = 12;
    this.maxWorkerId = -1 ^ (-1 << this.workerIdBits);
    this.maxSequence = -1 ^ (-1 << this.sequenceBits);
    this.workerIdShift = this.sequenceBits;
    this.timestampShift = this.sequenceBits + this.workerIdBits;
    this.epoch = 1609459200000; // 2021-01-01
  }
  
  nextId() {
    let timestamp = Date.now();
    
    if (timestamp < this.lastTimestamp) {
      throw new Error('Clock moved backwards');
    }
    
    if (timestamp === this.lastTimestamp) {
      this.sequence = (this.sequence + 1) & this.maxSequence;
      if (this.sequence === 0) {
        timestamp = this.waitNextMillis(timestamp);
      }
    } else {
      this.sequence = 0;
    }
    
    this.lastTimestamp = timestamp;
    
    return ((timestamp - this.epoch) << this.timestampShift) |
           (this.workerId << this.workerIdShift) |
           this.sequence;
  }
  
  waitNextMillis(timestamp) {
    while (timestamp === this.lastTimestamp) {
      timestamp = Date.now();
    }
    return timestamp;
  }
}

// 使用
const generator = new SnowflakeIdGenerator(1);
const id = generator.nextId();
const shortCode = toBase62(id);
```

**优点：**
- ✅ 分布式唯一
- ✅ 高性能（本地生成）
- ✅ 趋势递增

**缺点：**
- ❌ 依赖时钟（时钟回拨问题）
- ❌ 需要分配机器 ID

### 3. 完整实现

```javascript
const express = require('express');
const Redis = require('ioredis');
const mysql = require('mysql2/promise');
const crypto = require('crypto');

const app = express();
app.use(express.json());

// Redis 连接
const redis = new Redis({
  host: 'localhost',
  port: 6379
});

// MySQL 连接池
const pool = mysql.createPool({
  host: 'localhost',
  user: 'root',
  password: 'password',
  database: 'shortlink',
  connectionLimit: 10
});

// Base62 编码
const BASE62 = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';

function toBase62(num) {
  if (num === 0) return '0';
  let result = '';
  while (num > 0) {
    result = BASE62[num % 62] + result;
    num = Math.floor(num / 62);
  }
  return result;
}

// 生成短链
app.post('/api/shorten', async (req, res) => {
  const { longUrl, customCode, expiresIn } = req.body;
  
  try {
    // 验证 URL
    new URL(longUrl);
    
    let shortCode;
    
    if (customCode) {
      // 自定义短链
      const exists = await redis.exists(`short:${customCode}`);
      if (exists) {
        return res.status(400).json({ error: 'Custom code already exists' });
      }
      shortCode = customCode;
    } else {
      // 自动生成（自增 ID + Base62）
      const [result] = await pool.query(
        'INSERT INTO short_links (long_url, expires_at) VALUES (?, ?)',
        [longUrl, expiresIn ? new Date(Date.now() + expiresIn * 1000) : null]
      );
      
      const id = result.insertId;
      shortCode = toBase62(id);
      
      // 更新短链码
      await pool.query(
        'UPDATE short_links SET short_code = ? WHERE id = ?',
        [shortCode, id]
      );
    }
    
    // 缓存到 Redis（1小时）
    await redis.setex(
      `short:${shortCode}`,
      3600,
      JSON.stringify({ longUrl, createdAt: Date.now() })
    );
    
    res.json({
      shortUrl: `https://short.url/${shortCode}`,
      shortCode,
      longUrl
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// 重定向
app.get('/:shortCode', async (req, res) => {
  const { shortCode } = req.params;
  
  try {
    // 1. 从 Redis 读取（缓存）
    let data = await redis.get(`short:${shortCode}`);
    
    if (data) {
      const { longUrl } = JSON.parse(data);
      
      // 异步更新点击统计（不阻塞响应）
      setImmediate(() => {
        updateClickStats(shortCode, req);
      });
      
      return res.redirect(301, longUrl);
    }
    
    // 2. 从数据库读取
    const [rows] = await pool.query(
      'SELECT long_url, expires_at FROM short_links WHERE short_code = ?',
      [shortCode]
    );
    
    if (rows.length === 0) {
      return res.status(404).send('Short link not found');
    }
    
    const { long_url, expires_at } = rows[0];
    
    // 检查过期
    if (expires_at && new Date(expires_at) < new Date()) {
      return res.status(410).send('Short link expired');
    }
    
    // 缓存到 Redis
    await redis.setex(
      `short:${shortCode}`,
      3600,
      JSON.stringify({ longUrl: long_url, createdAt: Date.now() })
    );
    
    // 异步更新点击统计
    setImmediate(() => {
      updateClickStats(shortCode, req);
    });
    
    res.redirect(301, long_url);
  } catch (error) {
    console.error(error);
    res.status(500).send('Internal server error');
  }
});

// 更新点击统计
async function updateClickStats(shortCode, req) {
  try {
    // 1. 更新 Redis 计数（快）
    await redis.hincrby(`stats:${shortCode}`, 'clicks', 1);
    
    // 2. 批量写入数据库（每100次）
    const clicks = await redis.hget(`stats:${shortCode}`, 'clicks');
    
    if (clicks % 100 === 0) {
      await pool.query(
        'UPDATE short_links SET click_count = click_count + 100 WHERE short_code = ?',
        [shortCode]
      );
    }
    
    // 3. 记录详细统计（采样，10% 概率）
    if (Math.random() < 0.1) {
      await pool.query(
        'INSERT INTO click_stats (short_code, ip, user_agent, referer) VALUES (?, ?, ?, ?)',
        [shortCode, req.ip, req.headers['user-agent'], req.headers['referer']]
      );
    }
  } catch (error) {
    console.error('Update stats error:', error);
  }
}

// 查询统计
app.get('/api/stats/:shortCode', async (req, res) => {
  const { shortCode } = req.params;
  
  try {
    const [rows] = await pool.query(
      'SELECT click_count, created_at FROM short_links WHERE short_code = ?',
      [shortCode]
    );
    
    if (rows.length === 0) {
      return res.status(404).json({ error: 'Short link not found' });
    }
    
    // 从 Redis 获取实时计数
    const realtimeClicks = await redis.hget(`stats:${shortCode}`, 'clicks') || 0;
    
    res.json({
      shortCode,
      clicks: rows[0].click_count + parseInt(realtimeClicks),
      createdAt: rows[0].created_at
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

app.listen(3000, () => {
  console.log('Short link service running on port 3000');
});
```

## 💻 性能优化

### 1. 多级缓存

```javascript
// L1 缓存：本地内存（LRU）
const LRU = require('lru-cache');

const localCache = new LRU({
  max: 10000,  // 最多 10000 条
  ttl: 60000   // 1 分钟
});

app.get('/:shortCode', async (req, res) => {
  const { shortCode } = req.params;
  
  // L1：本地缓存
  let longUrl = localCache.get(shortCode);
  if (longUrl) {
    return res.redirect(301, longUrl);
  }
  
  // L2：Redis 缓存
  let data = await redis.get(`short:${shortCode}`);
  if (data) {
    longUrl = JSON.parse(data).longUrl;
    localCache.set(shortCode, longUrl);
    return res.redirect(301, longUrl);
  }
  
  // L3：数据库
  const [rows] = await pool.query(
    'SELECT long_url FROM short_links WHERE short_code = ?',
    [shortCode]
  );
  
  if (rows.length > 0) {
    longUrl = rows[0].long_url;
    localCache.set(shortCode, longUrl);
    await redis.setex(`short:${shortCode}`, 3600, JSON.stringify({ longUrl }));
    return res.redirect(301, longUrl);
  }
  
  res.status(404).send('Not found');
});
```

**性能提升：**
- L1 缓存：< 1ms
- L2 缓存：< 5ms
- L3 数据库：< 20ms

### 3. 布隆过滤器（防止缓存穿透）

```javascript
const { BloomFilter } = require('bloom-filters');

// 初始化布隆过滤器
const bloomFilter = new BloomFilter(1000000, 4);

// 加载所有短链到布隆过滤器
async function loadBloomFilter() {
  const [rows] = await pool.query('SELECT short_code FROM short_links');
  rows.forEach(row => bloomFilter.add(row.short_code));
}

// 重定向（增加布隆过滤器）
app.get('/:shortCode', async (req, res) => {
  const { shortCode } = req.params;
  
  // 1. 布隆过滤器检查（可能存在）
  if (!bloomFilter.has(shortCode)) {
    return res.status(404).send('Not found');
  }
  
  // 2. Redis 缓存
  let data = await redis.get(`short:${shortCode}`);
  if (data) {
    return res.redirect(301, JSON.parse(data).longUrl);
  }
  
  // 3. 数据库
  const [rows] = await pool.query(
    'SELECT long_url FROM short_links WHERE short_code = ?',
    [shortCode]
  );
  
  if (rows.length === 0) {
    return res.status(404).send('Not found');
  }
  
  res.redirect(301, rows[0].long_url);
});
```
