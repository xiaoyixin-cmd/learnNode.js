### 1. 认证鉴权

```javascript
const express = require('express');
const jwt = require('jsonwebtoken');
const Redis = require('ioredis');

const app = express();
const redis = new Redis();

// JWT 配置
const JWT_SECRET = 'your-secret-key';
const JWT_EXPIRES_IN = '1h';

// 认证中间件
async function authenticate(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  
  if (!token) {
    return res.status(401).json({ error: 'Missing token' });
  }
  
  try {
    // 验证 Token
    const decoded = jwt.verify(token, JWT_SECRET);
    
    // 检查黑名单（用户登出的 Token）
    const isBlacklisted = await redis.exists(`blacklist:${token}`);
    if (isBlacklisted) {
      return res.status(401).json({ error: 'Token revoked' });
    }
    
    // 检查权限
    const hasPermission = await checkPermission(decoded.userId, req.path);
    if (!hasPermission) {
      return res.status(403).json({ error: 'Forbidden' });
    }
    
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
}

// 权限检查
async function checkPermission(userId, path) {
  // 从 Redis 获取用户角色
  const roles = await redis.smembers(`user:${userId}:roles`);
  
  // 检查路径权限
  const permissions = {
    '/api/admin': ['admin'],
    '/api/users': ['admin', 'user'],
    '/api/public': ['*']
  };
  
  const requiredRoles = permissions[path] || [];
  
  if (requiredRoles.includes('*')) return true;
  
  return roles.some(role => requiredRoles.includes(role));
}

// 使用
app.use('/api', authenticate);
```

### 2. 限流（令牌桶算法）

```javascript
// 令牌桶限流（Redis 实现）
const rateLimitScript = `
local key = KEYS[1]
local capacity = tonumber(ARGV[1])    -- 桶容量
local rate = tonumber(ARGV[2])        -- 填充速率（tokens/秒）
local requested = tonumber(ARGV[3])   -- 请求令牌数
local now = tonumber(ARGV[4])         -- 当前时间

local bucket = redis.call('HMGET', key, 'tokens', 'last_refill')
local tokens = tonumber(bucket[1]) or capacity
local last_refill = tonumber(bucket[2]) or now

-- 填充令牌
local elapsed = (now - last_refill) / 1000
local tokens_to_add = elapsed * rate
tokens = math.min(capacity, tokens + tokens_to_add)

-- 消费令牌
if tokens >= requested then
  tokens = tokens - requested
  redis.call('HMSET', key, 'tokens', tokens, 'last_refill', now)
  redis.call('EXPIRE', key, 60)
  return 1  -- 允许
else
  return 0  -- 拒绝
end
`;

async function rateLimit(req, res, next) {
  const key = `rate_limit:${req.user.userId}:${req.path}`;
  
  const allowed = await redis.eval(
    rateLimitScript,
    1,
    key,
    100,      // capacity: 100 tokens
    10,       // rate: 10 tokens/秒
    1,        // requested: 1 token
    Date.now()
  );
  
  if (allowed === 1) {
    next();
  } else {
    res.status(429).json({ error: 'Rate limit exceeded' });
  }
}

// 使用
app.use('/api', rateLimit);
```

### 3. 熔断器

```javascript
class CircuitBreaker {
  constructor(options = {}) {
    this.failureThreshold = options.failureThreshold || 5;   // 失败阈值
    this.successThreshold = options.successThreshold || 2;   // 成功阈值
    this.timeout = options.timeout || 60000;                 // 超时时间
    
    this.state = 'CLOSED';     // CLOSED | OPEN | HALF_OPEN
    this.failures = 0;
    this.successes = 0;
    this.nextAttempt = Date.now();
  }
  
  async exec(fn) {
    if (this.state === 'OPEN') {
      if (Date.now() < this.nextAttempt) {
        throw new Error('Circuit breaker is OPEN');
      }
      
      // 尝试半开
      this.state = 'HALF_OPEN';
    }
    
    try {
      const result = await fn();
      
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }
  
  onSuccess() {
    this.failures = 0;
    
    if (this.state === 'HALF_OPEN') {
      this.successes++;
      
      if (this.successes >= this.successThreshold) {
        this.state = 'CLOSED';
        this.successes = 0;
      }
    }
  }
  
  onFailure() {
    this.failures++;
    this.successes = 0;
    
    if (this.failures >= this.failureThreshold) {
      this.state = 'OPEN';
      this.nextAttempt = Date.now() + this.timeout;
    }
  }
}

// 使用
const breaker = new CircuitBreaker({
  failureThreshold: 5,
  successThreshold: 2,
  timeout: 30000
});

app.get('/api/users/:id', async (req, res) => {
  try {
    const user = await breaker.exec(() => 
      axios.get(`http://user-service/users/${req.params.id}`)
    );
    
    res.json(user.data);
  } catch (error) {
    if (error.message === 'Circuit breaker is OPEN') {
      return res.status(503).json({ error: 'Service unavailable' });
    }
    
    res.status(500).json({ error: error.message });
  }
});
```

### 4. 负载均衡

```javascript
class LoadBalancer {
  constructor(servers) {
    this.servers = servers;  // [{ url, weight }]
    this.currentIndex = 0;
  }
  
  // 轮询
  roundRobin() {
    const server = this.servers[this.currentIndex];
    this.currentIndex = (this.currentIndex + 1) % this.servers.length;
    return server;
  }
  
  // 加权轮询
  weightedRoundRobin() {
    const totalWeight = this.servers.reduce((sum, s) => sum + s.weight, 0);
    let random = Math.random() * totalWeight;
    
    for (const server of this.servers) {
      random -= server.weight;
      if (random <= 0) {
        return server;
      }
    }
    
    return this.servers[0];
  }
  
  // 最少连接
  leastConnections() {
    return this.servers.reduce((min, server) => 
      server.connections < min.connections ? server : min
    );
  }
}

// 使用
const lb = new LoadBalancer([
  { url: 'http://service-1:3000', weight: 5, connections: 0 },
  { url: 'http://service-2:3000', weight: 3, connections: 0 },
  { url: 'http://service-3:3000', weight: 2, connections: 0 }
]);

app.use('/api/users', async (req, res) => {
  const server = lb.weightedRoundRobin();
  
  try {
    server.connections++;
    
    const response = await axios({
      method: req.method,
      url: `${server.url}${req.path}`,
      data: req.body,
      headers: req.headers
    });
    
    res.status(response.status).json(response.data);
  } finally {
    server.connections--;
  }
});
```

### 5. 动态路由

```javascript
const routes = new Map();

// 加载路由配置（从数据库/配置中心）
async function loadRoutes() {
  const configs = await db.query('SELECT * FROM api_routes WHERE enabled = true');
  
  for (const config of configs) {
    routes.set(config.path, {
      target: config.target,
      method: config.method,
      auth: config.auth,
      rateLimit: config.rateLimit
    });
  }
}

// 动态路由中间件
app.use('*', async (req, res, next) => {
  const route = routes.get(req.path);
  
  if (!route) {
    return res.status(404).json({ error: 'Route not found' });
  }
  
  // 应用路由配置
  if (route.auth) {
    await authenticate(req, res, () => {});
  }
  
  if (route.rateLimit) {
    await rateLimit(req, res, () => {});
  }
  
  // 代理请求
  try {
    const response = await axios({
      method: req.method,
      url: `${route.target}${req.path}`,
      data: req.body,
      headers: req.headers
    });
    
    res.status(response.status).json(response.data);
  } catch (error) {
    next(error);
  }
});

// 定时刷新路由
setInterval(loadRoutes, 60000);
loadRoutes();
```

### 7. 响应缓存

```javascript
// 缓存中间件
function cache(duration = 60) {
  return async (req, res, next) => {
    if (req.method !== 'GET') {
      return next();
    }
    
    const key = `cache:${req.path}:${JSON.stringify(req.query)}`;
    
    // 检查缓存
    const cached = await redis.get(key);
    if (cached) {
      res.setHeader('X-Cache', 'HIT');
      return res.json(JSON.parse(cached));
    }
    
    // 劫持 res.json
    const originalJson = res.json.bind(res);
    res.json = (data) => {
      redis.setex(key, duration, JSON.stringify(data));
      res.setHeader('X-Cache', 'MISS');
      return originalJson(data);
    };
    
    next();
  };
}

// 使用
app.get('/api/users/:id', cache(300), async (req, res) => {
  const user = await getUserById(req.params.id);
  res.json(user);
});
```

```lua
local tokens = redis.call('GET', key) or capacity
```

**3. 熔断实现（3分钟）**
> 状态转换：
```
CLOSED（正常）
  ↓ 失败 >= 5次
OPEN（熔断）
  ↓ 等待 30秒
HALF_OPEN（半开）
  ↓ 成功 >= 2次
CLOSED（恢复）
```

> - 加权：按权重分配
