## 💻 系统架构

### 1. 整体架构

```
调度中心（Master）
    ↓
任务队列（Redis）
    ↓
Worker 集群（10台）
    ↓
任务分片执行
    ↓
执行结果（MySQL）
```

### 2. 核心组件

```
1. 调度器（Scheduler）：触发任务
2. 分片器（Sharding）：任务分片
3. 执行器（Worker）：执行任务
4. 协调器（Coordinator）：故障转移
5. 监控器（Monitor）：任务监控
```

### 1. 任务调度器

```javascript
const schedule = require('node-schedule');
const Redis = require('ioredis');

const redis = new Redis();

// 任务配置
const jobs = [
  {
    name: 'daily-report',
    cron: '0 0 * * *',       // 每天凌晨
    handler: 'reportHandler',
    sharding: true,          // 是否分片
    shardingCount: 10        // 分片数
  },
  {
    name: 'send-email',
    cron: '*/5 * * * *',     // 每5分钟
    handler: 'emailHandler',
    sharding: false
  }
];

// 调度器
class JobScheduler {
  constructor() {
    this.jobs = new Map();
  }
  
  async start() {
    for (const job of jobs) {
      const scheduleJob = schedule.scheduleJob(job.cron, async () => {
        await this.triggerJob(job);
      });
      
      this.jobs.set(job.name, scheduleJob);
    }
    
    console.log('Scheduler started');
  }
  
  async triggerJob(job) {
    console.log(`Triggering job: ${job.name}`);
    
    if (job.sharding) {
      // 分片任务
      await this.shardingJob(job);
    } else {
      // 普通任务
      await this.enqueueJob(job);
    }
  }
  
  async shardingJob(job) {
    const shards = [];
    
    for (let i = 0; i < job.shardingCount; i++) {
      shards.push({
        jobName: job.name,
        shardIndex: i,
        shardTotal: job.shardingCount,
        handler: job.handler,
        createdAt: Date.now()
      });
    }
    
    // 推送到任务队列
    for (const shard of shards) {
      await redis.lpush('job:queue', JSON.stringify(shard));
    }
    
    console.log(`Job sharded: ${job.name}, ${job.shardingCount} shards`);
  }
  
  async enqueueJob(job) {
    await redis.lpush('job:queue', JSON.stringify({
      jobName: job.name,
      handler: job.handler,
      createdAt: Date.now()
    }));
  }
}

const scheduler = new JobScheduler();
scheduler.start();
```

```javascript
// 分片策略
class ShardingStrategy {
  // 按 ID 范围分片
  static byIdRange(totalRecords, shardIndex, shardTotal) {
    const pageSize = Math.ceil(totalRecords / shardTotal);
    const start = shardIndex * pageSize;
    const end = Math.min(start + pageSize, totalRecords);
    
    return { start, end };
  }
  
  // 按哈希分片
  static byHash(id, shardTotal) {
    return id % shardTotal;
  }
  
  // 按时间分片
  static byTime(startTime, endTime, shardIndex, shardTotal) {
    const duration = endTime - startTime;
    const shardDuration = duration / shardTotal;
    const start = startTime + shardIndex * shardDuration;
    const end = start + shardDuration;
    
    return { start, end };
  }
}

// 任务处理器
const handlers = {
  async reportHandler(shard) {
    console.log(`Processing shard: ${shard.shardIndex}/${shard.shardTotal}`);
    
    // 获取总用户数
    const totalUsers = await db.users.count();
    
    // 计算分片范围
    const { start, end } = ShardingStrategy.byIdRange(
      totalUsers,
      shard.shardIndex,
      shard.shardTotal
    );
    
    // 处理分片数据
    const users = await db.users.find({
      id: { $gte: start, $lt: end }
    });
    
    for (const user of users) {
      await generateReport(user);
    }
    
    console.log(`Shard completed: ${start}-${end}`);
  }
};
```

### 3. Worker 执行器

```javascript
const os = require('os');
const Redis = require('ioredis');

const redis = new Redis();

class Worker {
  constructor(workerId) {
    this.workerId = workerId || os.hostname();
    this.isRunning = false;
  }
  
  async start() {
    this.isRunning = true;
    
    // 注册 Worker
    await this.register();
    
    // 心跳
    setInterval(() => this.heartbeat(), 5000);
    
    // 消费任务
    while (this.isRunning) {
      await this.consumeJob();
    }
  }
  
  async register() {
    await redis.hset('workers', this.workerId, JSON.stringify({
      workerId: this.workerId,
      status: 'active',
      lastHeartbeat: Date.now()
    }));
    
    console.log(`Worker registered: ${this.workerId}`);
  }
  
  async heartbeat() {
    await redis.hset('workers', this.workerId, JSON.stringify({
      workerId: this.workerId,
      status: 'active',
      lastHeartbeat: Date.now()
    }));
  }
  
  async consumeJob() {
    // 阻塞式获取任务
    const result = await redis.brpop('job:queue', 5);
    
    if (!result) return;
    
    const job = JSON.parse(result[1]);
    
    try {
      // 执行任务
      await this.executeJob(job);
    } catch (error) {
      console.error('Job failed:', error);
      
      // 重试
      await this.retryJob(job, error);
    }
  }
  
  async executeJob(job) {
    console.log(`Executing job: ${job.jobName}`);
    
    // 获取分布式锁（防止重复执行）
    const lockKey = `lock:${job.jobName}:${job.shardIndex || 0}`;
    const lockValue = this.workerId;
    
    const locked = await redis.set(lockKey, lockValue, 'EX', 60, 'NX');
    
    if (!locked) {
      console.log('Job already locked');
      return;
    }
    
    try {
      // 执行任务处理器
      const handler = handlers[job.handler];
      await handler(job);
      
      // 记录成功
      await this.recordSuccess(job);
    } finally {
      // 释放锁
      const currentValue = await redis.get(lockKey);
      if (currentValue === lockValue) {
        await redis.del(lockKey);
      }
    }
  }
  
  async retryJob(job, error) {
    job.retryCount = (job.retryCount || 0) + 1;
    
    if (job.retryCount >= 3) {
      // 记录失败
      await this.recordFailure(job, error);
      return;
    }
    
    // 延时重试
    await redis.lpush('job:queue', JSON.stringify(job));
  }
  
  async recordSuccess(job) {
    await db.jobLogs.insert({
      jobName: job.jobName,
      shardIndex: job.shardIndex,
      workerId: this.workerId,
      status: 'success',
      createdAt: Date.now()
    });
  }
  
  async recordFailure(job, error) {
    await db.jobLogs.insert({
      jobName: job.jobName,
      shardIndex: job.shardIndex,
      workerId: this.workerId,
      status: 'failed',
      error: error.message,
      createdAt: Date.now()
    });
  }
  
  async stop() {
    this.isRunning = false;
    await redis.hdel('workers', this.workerId);
  }
}

const worker = new Worker();
worker.start();
```

### 4. 故障转移

```javascript
// 协调器（检测故障、转移任务）
class Coordinator {
  constructor() {
    this.checkInterval = 10000;  // 10秒检测一次
  }
  
  async start() {
    setInterval(() => this.checkWorkers(), this.checkInterval);
  }
  
  async checkWorkers() {
    const workers = await redis.hgetall('workers');
    const now = Date.now();
    
    for (const [workerId, data] of Object.entries(workers)) {
      const worker = JSON.parse(data);
      
      // 检查心跳超时（30秒）
      if (now - worker.lastHeartbeat > 30000) {
        console.log(`Worker timeout: ${workerId}`);
        
        // 标记失败
        await redis.hset('workers', workerId, JSON.stringify({
          ...worker,
          status: 'failed'
        }));
        
        // 转移任务
        await this.migrateJobs(workerId);
      }
    }
  }
  
  async migrateJobs(workerId) {
    // 查找该 Worker 正在执行的任务
    const jobs = await db.jobLogs.find({
      workerId,
      status: 'running'
    });
    
    // 重新入队
    for (const job of jobs) {
      await redis.lpush('job:queue', JSON.stringify({
        jobName: job.jobName,
        shardIndex: job.shardIndex,
        handler: job.handler,
        retryCount: (job.retryCount || 0) + 1
      }));
      
      console.log(`Job migrated: ${job.jobName}`);
    }
  }
}

const coordinator = new Coordinator();
coordinator.start();
```

```javascript
// 延时任务（Redis Sorted Set）
class DelayedJobQueue {
  async add(job, delay) {
    const executeTime = Date.now() + delay;
    
    await redis.zadd('delayed:jobs', executeTime, JSON.stringify(job));
  }
  
  async consume() {
    while (true) {
      // 获取到期任务
      const now = Date.now();
      const jobs = await redis.zrangebyscore('delayed:jobs', 0, now, 'LIMIT', 0, 10);
      
      if (jobs.length === 0) {
        await sleep(1000);
        continue;
      }
      
      for (const jobStr of jobs) {
        const job = JSON.parse(jobStr);
        
        // 移除
        await redis.zrem('delayed:jobs', jobStr);
        
        // 执行
        await this.executeJob(job);
      }
    }
  }
  
  async executeJob(job) {
    await redis.lpush('job:queue', JSON.stringify(job));
  }
}

// 使用
const delayedQueue = new DelayedJobQueue();
delayedQueue.consume();

// 添加延时任务（订单30分钟自动取消）
await delayedQueue.add({
  jobName: 'cancel-order',
  orderId: 123,
  handler: 'cancelOrderHandler'
}, 30 * 60 * 1000);  // 30分钟
```

### 6. 工作流（DAG）

```javascript
// 有向无环图（DAG）
class WorkflowEngine {
  constructor(workflow) {
    this.workflow = workflow;  // { tasks, dependencies }
  }
  
  // 拓扑排序
  topologicalSort() {
    const { tasks, dependencies } = this.workflow;
    const inDegree = new Map();
    const graph = new Map();
    
    // 初始化
    for (const task of tasks) {
      inDegree.set(task.name, 0);
      graph.set(task.name, []);
    }
    
    // 构建图
    for (const [from, to] of dependencies) {
      graph.get(from).push(to);
      inDegree.set(to, inDegree.get(to) + 1);
    }
    
    // 拓扑排序
    const queue = [];
    const result = [];
    
    for (const [task, degree] of inDegree) {
      if (degree === 0) {
        queue.push(task);
      }
    }
    
    while (queue.length > 0) {
      const task = queue.shift();
      result.push(task);
      
      for (const neighbor of graph.get(task)) {
        inDegree.set(neighbor, inDegree.get(neighbor) - 1);
        
        if (inDegree.get(neighbor) === 0) {
          queue.push(neighbor);
        }
      }
    }
    
    // 检测循环依赖
    if (result.length !== tasks.length) {
      throw new Error('Circular dependency detected');
    }
    
    return result;
  }
  
  async execute() {
    const order = this.topologicalSort();
    
    for (const taskName of order) {
      const task = this.workflow.tasks.find(t => t.name === taskName);
      
      console.log(`Executing task: ${taskName}`);
      await task.handler();
    }
  }
}

// 使用
const workflow = new WorkflowEngine({
  tasks: [
    { name: 'task-a', handler: async () => console.log('A') },
    { name: 'task-b', handler: async () => console.log('B') },
    { name: 'task-c', handler: async () => console.log('C') }
  ],
  dependencies: [
    ['task-a', 'task-b'],
    ['task-a', 'task-c'],
    ['task-b', 'task-c']
  ]
});

await workflow.execute();  // 输出：A → B → C
```
