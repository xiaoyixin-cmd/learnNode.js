## 💻 基础实现（Socket.io）

### 1. 服务端实现

```javascript
const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const Redis = require('ioredis');

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  cors: { origin: '*' }
});

const redis = new Redis();

// 用户-Socket 映射
const users = new Map();

// 连接事件
io.on('connection', (socket) => {
  console.log('User connected:', socket.id);
  
  // 用户登录
  socket.on('login', async (data) => {
    const { userId, username } = data;
    
    // 保存用户信息
    socket.userId = userId;
    socket.username = username;
    users.set(userId, socket.id);
    
    // 加入个人房间（用于私聊）
    socket.join(`user:${userId}`);
    
    // 通知其他用户
    socket.broadcast.emit('user-online', {
      userId,
      username
    });
    
    // 返回在线用户列表
    const onlineUsers = Array.from(users.entries()).map(([id, socketId]) => ({
      userId: id,
      username: io.sockets.sockets.get(socketId)?.username
    }));
    
    socket.emit('online-users', onlineUsers);
  });
  
  // 加入房间
  socket.on('join-room', async (data) => {
    const { roomId } = data;
    
    socket.join(roomId);
    socket.currentRoom = roomId;
    
    // 通知房间内其他用户
    socket.to(roomId).emit('user-joined', {
      userId: socket.userId,
      username: socket.username,
      roomId
    });
    
    // 加载历史消息
    const messages = await loadRoomMessages(roomId);
    socket.emit('room-messages', messages);
  });
  
  // 发送消息
  socket.on('send-message', async (data) => {
    const { roomId, content, type = 'text' } = data;
    
    const message = {
      id: generateMessageId(),
      roomId,
      userId: socket.userId,
      username: socket.username,
      content,
      type,
      timestamp: Date.now()
    };
    
    // 保存消息到 Redis
    await redis.lpush(
      `room:${roomId}:messages`,
      JSON.stringify(message)
    );
    
    // 广播到房间内所有用户
    io.to(roomId).emit('new-message', message);
  });
  
  // 私聊
  socket.on('private-message', async (data) => {
    const { toUserId, content } = data;
    
    const message = {
      id: generateMessageId(),
      fromUserId: socket.userId,
      fromUsername: socket.username,
      toUserId,
      content,
      timestamp: Date.now()
    };
    
    // 保存私聊消息
    await redis.lpush(
      `private:${socket.userId}:${toUserId}`,
      JSON.stringify(message)
    );
    
    // 发送给目标用户
    io.to(`user:${toUserId}`).emit('private-message', message);
    
    // 确认发送成功
    socket.emit('message-sent', { messageId: message.id });
  });
  
  // 输入状态
  socket.on('typing', (data) => {
    const { roomId } = data;
    socket.to(roomId).emit('user-typing', {
      userId: socket.userId,
      username: socket.username
    });
  });
  
  // 离开房间
  socket.on('leave-room', (data) => {
    const { roomId } = data;
    
    socket.leave(roomId);
    
    socket.to(roomId).emit('user-left', {
      userId: socket.userId,
      username: socket.username
    });
  });
  
  // 断开连接
  socket.on('disconnect', () => {
    console.log('User disconnected:', socket.id);
    
    users.delete(socket.userId);
    
    // 通知其他用户
    socket.broadcast.emit('user-offline', {
      userId: socket.userId,
      username: socket.username
    });
  });
});

// 加载房间消息
async function loadRoomMessages(roomId, limit = 50) {
  const messages = await redis.lrange(
    `room:${roomId}:messages`,
    0,
    limit - 1
  );
  
  return messages.map(msg => JSON.parse(msg)).reverse();
}

// 生成消息 ID
function generateMessageId() {
  return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
}

server.listen(3000, () => {
  console.log('Chat server running on port 3000');
});
```

## 💻 断线重连机制

### 1. 客户端重连

```javascript
class ChatClient {
  constructor(url) {
    this.url = url;
    this.socket = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
    this.reconnectDelay = 1000;
    this.messageQueue = [];
    
    this.connect();
  }
  
  connect() {
    this.socket = io(this.url, {
      reconnection: true,
      reconnectionAttempts: this.maxReconnectAttempts,
      reconnectionDelay: this.reconnectDelay,
      reconnectionDelayMax: 5000,
      timeout: 10000
    });
    
    this.socket.on('connect', () => {
      console.log('Connected to server');
      this.reconnectAttempts = 0;
      
      // 重新登录
      this.reLogin();
      
      // 发送缓存的消息
      this.flushMessageQueue();
    });
    
    this.socket.on('disconnect', (reason) => {
      console.log('Disconnected:', reason);
      
      if (reason === 'io server disconnect') {
        // 服务器主动断开，需要手动重连
        this.socket.connect();
      }
    });
    
    this.socket.on('reconnect_attempt', (attempt) => {
      console.log(`Reconnecting... Attempt ${attempt}`);
      this.reconnectAttempts = attempt;
    });
    
    this.socket.on('reconnect_failed', () => {
      console.error('Reconnection failed');
      // 通知用户网络问题
      this.onConnectionLost();
    });
  }
  
  reLogin() {
    // 重新登录并恢复状态
    if (this.userId && this.username) {
      this.socket.emit('login', {
        userId: this.userId,
        username: this.username
      });
      
      // 重新加入房间
      if (this.currentRoom) {
        this.socket.emit('join-room', {
          roomId: this.currentRoom
        });
      }
    }
  }
  
  sendMessage(content) {
    const message = {
      roomId: this.currentRoom,
      content,
      timestamp: Date.now()
    };
    
    if (this.socket.connected) {
      this.socket.emit('send-message', message);
    } else {
      // 离线时缓存消息
      this.messageQueue.push(message);
      console.log('Message queued (offline)');
    }
  }
  
  flushMessageQueue() {
    while (this.messageQueue.length > 0) {
      const message = this.messageQueue.shift();
      this.socket.emit('send-message', message);
    }
  }
  
  onConnectionLost() {
    // 显示离线提示
    alert('Connection lost. Please check your network.');
  }
}

// 使用
const chat = new ChatClient('http://localhost:3000');
```

### 2. 消息可靠性保证

```javascript
// 消息确认机制
class ReliableChat {
  constructor() {
    this.pendingMessages = new Map();
    this.messageTimeout = 5000;
  }
  
  sendMessage(content) {
    const messageId = this.generateId();
    const message = {
      id: messageId,
      content,
      timestamp: Date.now()
    };
    
    // 添加到待确认列表
    this.pendingMessages.set(messageId, {
      message,
      retries: 0,
      timer: setTimeout(() => {
        this.onMessageTimeout(messageId);
      }, this.messageTimeout)
    });
    
    // 发送消息
    this.socket.emit('send-message', message);
  }
  
  onMessageAck(messageId) {
    const pending = this.pendingMessages.get(messageId);
    if (pending) {
      clearTimeout(pending.timer);
      this.pendingMessages.delete(messageId);
      console.log('Message delivered:', messageId);
    }
  }
  
  onMessageTimeout(messageId) {
    const pending = this.pendingMessages.get(messageId);
    if (pending && pending.retries < 3) {
      // 重试
      pending.retries++;
      console.log(`Retrying message ${messageId} (${pending.retries}/3)`);
      
      this.socket.emit('send-message', pending.message);
      
      pending.timer = setTimeout(() => {
        this.onMessageTimeout(messageId);
      }, this.messageTimeout);
    } else {
      // 重试失败
      console.error('Message failed:', messageId);
      this.pendingMessages.delete(messageId);
    }
  }
  
  generateId() {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }
}

// 服务端确认
socket.on('send-message', async (message) => {
  // 保存消息
  await saveMessage(message);
  
  // 广播消息
  io.to(message.roomId).emit('new-message', message);
  
  // 确认消息
  socket.emit('message-ack', { messageId: message.id });
});
```

```javascript
const mongoose = require('mongoose');

// 消息 Schema
const MessageSchema = new mongoose.Schema({
  roomId: { type: String, required: true, index: true },
  userId: { type: String, required: true },
  username: { type: String, required: true },
  content: { type: String, required: true },
  type: { type: String, enum: ['text', 'image', 'file'], default: 'text' },
  timestamp: { type: Date, default: Date.now, index: true },
  deleted: { type: Boolean, default: false }
});

const Message = mongoose.model('Message', MessageSchema);

// 保存消息
async function saveMessage(data) {
  const message = new Message(data);
  await message.save();
  return message;
}

// 加载历史消息（分页）
async function loadMessages(roomId, limit = 50, before = null) {
  const query = {
    roomId,
    deleted: false
  };
  
  if (before) {
    query.timestamp = { $lt: new Date(before) };
  }
  
  return await Message
    .find(query)
    .sort({ timestamp: -1 })
    .limit(limit)
    .lean();
}

// 服务端集成
socket.on('send-message', async (data) => {
  // 保存到数据库
  const message = await saveMessage({
    ...data,
    userId: socket.userId,
    username: socket.username
  });
  
  // 广播
  io.to(data.roomId).emit('new-message', message);
});

// 加载历史消息
socket.on('load-messages', async (data) => {
  const { roomId, before } = data;
  const messages = await loadMessages(roomId, 50, before);
  socket.emit('history-messages', messages);
});
```

```javascript
// 使用 Redis 存储在线状态
class OnlineStatusManager {
  constructor(redis) {
    this.redis = redis;
    this.serverId = process.env.SERVER_ID || 'server1';
  }
  
  async userOnline(userId, socketId) {
    // 存储用户在线状态
    await this.redis.hset(
      `online:${userId}`,
      'socketId', socketId,
      'serverId', this.serverId,
      'timestamp', Date.now()
    );
    
    // 加入在线用户集合
    await this.redis.sadd('online:users', userId);
    
    // 发布上线事件
    await this.redis.publish('user:online', JSON.stringify({
      userId,
      serverId: this.serverId
    }));
  }
  
  async userOffline(userId) {
    // 删除在线状态
    await this.redis.del(`online:${userId}`);
    await this.redis.srem('online:users', userId);
    
    // 发布下线事件
    await this.redis.publish('user:offline', JSON.stringify({
      userId
    }));
  }
  
  async getOnlineUsers() {
    return await this.redis.smembers('online:users');
  }
  
  async getUserServer(userId) {
    const data = await this.redis.hgetall(`online:${userId}`);
    return data.serverId;
  }
}

// 订阅在线状态变化
const statusManager = new OnlineStatusManager(redis);

subClient.subscribe('user:online', 'user:offline');

subClient.on('message', (channel, message) => {
  const data = JSON.parse(message);
  
  if (channel === 'user:online') {
    io.emit('user-online', data);
  } else if (channel === 'user:offline') {
    io.emit('user-offline', data);
  }
});
```

## 💻 性能优化

### 1. 消息批量处理

```javascript
class MessageBatcher {
  constructor(redis) {
    this.redis = redis;
    this.batch = [];
    this.batchSize = 100;
    this.flushInterval = 1000;
    
    setInterval(() => this.flush(), this.flushInterval);
  }
  
  add(message) {
    this.batch.push(message);
    
    if (this.batch.length >= this.batchSize) {
      this.flush();
    }
  }
  
  async flush() {
    if (this.batch.length === 0) return;
    
    const messages = this.batch.splice(0);
    
    // 批量插入 MongoDB
    await Message.insertMany(messages);
    
    // 批量缓存到 Redis
    const pipeline = this.redis.pipeline();
    messages.forEach(msg => {
      pipeline.lpush(
        `room:${msg.roomId}:messages`,
        JSON.stringify(msg)
      );
    });
    await pipeline.exec();
    
    console.log(`Flushed ${messages.length} messages`);
  }
}

const batcher = new MessageBatcher(redis);

socket.on('send-message', (data) => {
  const message = {
    ...data,
    userId: socket.userId,
    timestamp: Date.now()
  };
  
  // 添加到批处理
  batcher.add(message);
  
  // 立即广播（不等待持久化）
  io.to(data.roomId).emit('new-message', message);
});
```

```javascript
const zlib = require('zlib');

// 压缩大消息
function compressMessage(message) {
  if (message.content.length > 1024) {
    return {
      ...message,
      content: zlib.gzipSync(message.content).toString('base64'),
      compressed: true
    };
  }
  return message;
}

// 解压消息
function decompressMessage(message) {
  if (message.compressed) {
    return {
      ...message,
      content: zlib.gunzipSync(
        Buffer.from(message.content, 'base64')
      ).toString(),
      compressed: false
    };
  }
  return message;
}
```
