## 📚 问题分析

### 1. 什么是超卖？

**超卖场景：**
```javascript
// 商品库存：10 件

// 用户 A 查询库存
const stock = await getStock(productId);  // 10
if (stock > 0) {
  // 用户 B 同时查询库存
  const stock = await getStock(productId);  // 10
  
  // 用户 A 扣减库存
  await updateStock(productId, stock - 1);  // 9
  
  // 用户 B 扣减库存
  await updateStock(productId, stock - 1);  // 9（错误！应该是8）
}

// 问题：
// 1. 两个用户同时读取库存（都是10）
// 2. 两个用户同时扣减库存（都扣成9）
// 3. 实际应该卖出2件，库存应该是8，但结果是9
// 4. 少卖了1件（超卖）
```

**超卖的本质：**
- 并发读写导致数据不一致
- 缺少原子性操作
- 没有加锁保护

## 💻 方案1：Redis + Lua（推荐）

### 1. 原理

**核心思想：**
- 使用 Lua 脚本在 Redis 中原子执行
- 单线程执行，天然保证原子性
- 高性能（内存操作）

**流程：**
```
1. 客户端执行 Lua 脚本
2. Redis 单线程执行整个脚本
3. 检查库存 → 扣减库存 → 返回结果（原子操作）
4. 没有并发冲突
```

---

### 2. 完整实现

```javascript
const Redis = require('ioredis');
const redis = new Redis();

// Lua 脚本：原子扣减库存
const decreaseStockScript = `
local key = KEYS[1]
local quantity = tonumber(ARGV[1])

-- 获取当前库存
local stock = tonumber(redis.call('GET', key) or 0)

-- 检查库存是否足够
if stock >= quantity then
  -- 扣减库存
  redis.call('DECRBY', key, quantity)
  return stock - quantity  -- 返回剩余库存
else
  return -1  -- 库存不足
end
`;

// 预加载脚本
let decreaseStockSHA;
async function loadScript() {
  decreaseStockSHA = await redis.script('LOAD', decreaseStockScript);
}

// 扣减库存
async function decreaseStock(productId, quantity) {
  const key = `stock:${productId}`;
  
  // 执行 Lua 脚本（原子操作）
  const result = await redis.evalsha(
    decreaseStockSHA,
    1,              // KEYS 数量
    key,            // KEYS[1]
    quantity        // ARGV[1]
  );
  
  if (result === -1) {
    throw new Error('Stock insufficient');
  }
  
  return result;  // 返回剩余库存
}

// 创建订单
async function createOrder(userId, productId, quantity) {
  try {
    // 1. 扣减库存（原子操作）
    const remainingStock = await decreaseStock(productId, quantity);
    
    // 2. 创建订单
    const order = {
      id: generateOrderId(),
      userId,
      productId,
      quantity,
      status: 'pending',
      createdAt: Date.now()
    };
    
    // 3. 保存订单到数据库
    await db.orders.insert(order);
    
    console.log(`Order created. Remaining stock: ${remainingStock}`);
    return order;
  } catch (error) {
    if (error.message === 'Stock insufficient') {
      throw new Error('Product out of stock');
    }
    throw error;
  }
}

// 初始化库存
async function initStock(productId, stock) {
  await redis.set(`stock:${productId}`, stock);
}

// 使用
await loadScript();
await initStock('product-123', 100);

// 并发下单（不会超卖）
const orders = await Promise.all([
  createOrder('user-1', 'product-123', 1),
  createOrder('user-2', 'product-123', 1),
  createOrder('user-3', 'product-123', 1)
]);
```

### 3. 优化：预减库存

```javascript
// Lua 脚本：预减库存（支持回滚）
const preDecreaseStockScript = `
local key = KEYS[1]
local tempKey = KEYS[2]
local quantity = tonumber(ARGV[1])
local orderId = ARGV[2]

local stock = tonumber(redis.call('GET', key) or 0)

if stock >= quantity then
  -- 扣减库存
  redis.call('DECRBY', key, quantity)
  
  -- 记录预扣减（用于回滚）
  redis.call('HSET', tempKey, orderId, quantity)
  redis.call('EXPIRE', tempKey, 600)  -- 10分钟过期
  
  return stock - quantity
else
  return -1
end
`;

// 回滚库存
const rollbackStockScript = `
local key = KEYS[1]
local tempKey = KEYS[2]
local orderId = ARGV[1]

local quantity = tonumber(redis.call('HGET', tempKey, orderId) or 0)

if quantity > 0 then
  -- 回滚库存
  redis.call('INCRBY', key, quantity)
  redis.call('HDEL', tempKey, orderId)
  return quantity
else
  return 0
end
`;

// 预扣减
async function preDecreaseStock(productId, quantity, orderId) {
  const result = await redis.evalsha(
    preDecreaseStockSHA,
    2,
    `stock:${productId}`,
    `stock:temp:${productId}`,
    quantity,
    orderId
  );
  
  return result;
}

// 回滚
async function rollbackStock(productId, orderId) {
  await redis.evalsha(
    rollbackStockSHA,
    2,
    `stock:${productId}`,
    `stock:temp:${productId}`,
    orderId
  );
}

// 使用
async function createOrderWithRollback(userId, productId, quantity) {
  const orderId = generateOrderId();
  
  try {
    // 1. 预扣减库存
    await preDecreaseStock(productId, quantity, orderId);
    
    // 2. 创建订单
    const order = await db.orders.insert({
      id: orderId,
      userId,
      productId,
      quantity
    });
    
    // 3. 支付（模拟）
    await processPayment(order);
    
    return order;
  } catch (error) {
    // 4. 失败回滚
    await rollbackStock(productId, orderId);
    throw error;
  }
}
```

## 💻 方案2：数据库行锁

### 1. SELECT FOR UPDATE

```javascript
const mysql = require('mysql2/promise');

async function createOrderWithRowLock(userId, productId, quantity) {
  const connection = await pool.getConnection();
  
  try {
    await connection.beginTransaction();
    
    // 1. 查询库存（加行锁）
    const [rows] = await connection.query(
      'SELECT stock FROM products WHERE id = ? FOR UPDATE',
      [productId]
    );
    
    if (rows.length === 0) {
      throw new Error('Product not found');
    }
    
    const stock = rows[0].stock;
    
    // 2. 检查库存
    if (stock < quantity) {
      throw new Error('Stock insufficient');
    }
    
    // 3. 扣减库存
    await connection.query(
      'UPDATE products SET stock = stock - ? WHERE id = ?',
      [quantity, productId]
    );
    
    // 4. 创建订单
    const [result] = await connection.query(
      'INSERT INTO orders (user_id, product_id, quantity) VALUES (?, ?, ?)',
      [userId, productId, quantity]
    );
    
    await connection.commit();
    
    return { orderId: result.insertId };
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}
```

**优点：**
- ✅ 强一致性（ACID）
- ✅ 实现简单

**缺点：**
- ❌ 性能差（锁等待）
- ❌ 不适合高并发

### 2. 直接 UPDATE（推荐）

```javascript
async function createOrderWithUpdate(userId, productId, quantity) {
  const connection = await pool.getConnection();
  
  try {
    await connection.beginTransaction();
    
    // 1. 直接扣减库存（带条件）
    const [result] = await connection.query(
      'UPDATE products SET stock = stock - ? WHERE id = ? AND stock >= ?',
      [quantity, productId, quantity]
    );
    
    // 2. 检查是否更新成功
    if (result.affectedRows === 0) {
      throw new Error('Stock insufficient or product not found');
    }
    
    // 3. 创建订单
    const [orderResult] = await connection.query(
      'INSERT INTO orders (user_id, product_id, quantity) VALUES (?, ?, ?)',
      [userId, productId, quantity]
    );
    
    await connection.commit();
    
    return { orderId: orderResult.insertId };
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}
```

**优点：**
- ✅ 无需 SELECT FOR UPDATE

## 💻 方案3：乐观锁（版本号）

### 1. 实现

```javascript
// 数据库表结构
/*
CREATE TABLE products (
  id INT PRIMARY KEY,
  stock INT NOT NULL,
  version INT NOT NULL DEFAULT 0,  -- 版本号
  INDEX idx_version (version)
);
*/

async function createOrderWithOptimisticLock(userId, productId, quantity) {
  const maxRetries = 3;
  let retries = 0;
  
  while (retries < maxRetries) {
    try {
      const connection = await pool.getConnection();
      
      try {
        await connection.beginTransaction();
        
        // 1. 查询库存和版本号
        const [rows] = await connection.query(
          'SELECT stock, version FROM products WHERE id = ?',
          [productId]
        );
        
        if (rows.length === 0) {
          throw new Error('Product not found');
        }
        
        const { stock, version } = rows[0];
        
        // 2. 检查库存
        if (stock < quantity) {
          throw new Error('Stock insufficient');
        }
        
        // 3. 扣减库存（检查版本号）
        const [result] = await connection.query(
          'UPDATE products SET stock = stock - ?, version = version + 1 WHERE id = ? AND version = ?',
          [quantity, productId, version]
        );
        
        // 4. 检查是否更新成功
        if (result.affectedRows === 0) {
          // 版本号冲突，重试
          throw new Error('Version conflict');
        }
        
        // 5. 创建订单
        const [orderResult] = await connection.query(
          'INSERT INTO orders (user_id, product_id, quantity) VALUES (?, ?, ?)',
          [userId, productId, quantity]
        );
        
        await connection.commit();
        
        return { orderId: orderResult.insertId };
      } finally {
        connection.release();
      }
    } catch (error) {
      if (error.message === 'Version conflict') {
        retries++;
        console.log(`Retry ${retries}/${maxRetries}`);
        
        // 指数退避
        await new Promise(resolve => 
          setTimeout(resolve, Math.pow(2, retries) * 100)
        );
        
        continue;
      }
      
      throw error;
    }
  }
  
  throw new Error('Max retries exceeded');
}
```

**优点：**
- ✅ 无锁，性能好
- ✅ 适合中等并发

**缺点：**
- ❌ 冲突时需要重试
- ❌ 高并发时重试次数多

## 💻 方案4：分布式锁（Redlock）

### 1. Redis 分布式锁

```javascript
const Redlock = require('redlock');

const redlock = new Redlock(
  [redis1, redis2, redis3],  // 多个 Redis 实例
  {
    driftFactor: 0.01,
    retryCount: 3,
    retryDelay: 200,
    retryJitter: 200
  }
);

async function createOrderWithDistributedLock(userId, productId, quantity) {
  const lockKey = `lock:product:${productId}`;
  const lockTTL = 10000;  // 10 秒
  
  // 获取分布式锁
  const lock = await redlock.lock(lockKey, lockTTL);
  
  try {
    // 1. 查询库存
    const stock = await redis.get(`stock:${productId}`);
    
    if (stock < quantity) {
      throw new Error('Stock insufficient');
    }
    
    // 2. 扣减库存
    await redis.decrby(`stock:${productId}`, quantity);
    
    // 3. 创建订单
    const order = await db.orders.insert({
      userId,
      productId,
      quantity
    });
    
    return order;
  } finally {
    // 释放锁
    await lock.unlock();
  }
}
```

**优点：**
- ✅ 分布式一致性
- ✅ 适合分布式场景

**缺点：**
- ❌ 实现复杂
- ❌ 性能中等

```javascript
const Queue = require('bull');

const orderQueue = new Queue('orders', {
  redis: { host: 'localhost', port: 6379 }
});

// 生产者：添加订单到队列
async function submitOrder(userId, productId, quantity) {
  const job = await orderQueue.add({
    userId,
    productId,
    quantity
  });
  
  return { jobId: job.id };

// 消费者：串行处理订单（不会超卖）
orderQueue.process(async (job) => {
  const { userId, productId, quantity } = job.data;
  
  // 1. 查询库存
  const stock = await redis.get(`stock:${productId}`);
  
  if (stock < quantity) {
    throw new Error('Stock insufficient');
  }
  
  // 2. 扣减库存
  await redis.decrby(`stock:${productId}`, quantity);
  
  // 3. 创建订单
  const order = await db.orders.insert({
    userId,
    productId,
    quantity
  });
  
  console.log('Order created:', order.id);
  return order;
});

// 使用
const result = await submitOrder('user-1', 'product-123', 1);
console.log('Order submitted:', result.jobId);
```

**优点：**
- ✅ 削峰填谷

### 问题："如何防止商品超卖？"

**标准答案（10分钟）：**

**1. 问题分析（2分钟）**
> 超卖原因：并发读写 → 数据不一致
> 
> 示例：
> - 用户A查询库存：10
> - 用户B查询库存：10
> - 用户A扣减：9
> - 用户B扣减：9（错误，应该是8）

**2. 解决方案（5分钟）**

**方案1：Redis + Lua（推荐）**
```lua
-- 原子扣减
local stock = redis.call('GET', KEYS[1])
if stock >= ARGV[1] then
  redis.call('DECRBY', KEYS[1], ARGV[1])
  return 0
else
  return -1
end
```
- 优点：高性能、原子性
- 适用：秒杀、高并发

**方案2：数据库行锁**
```sql
UPDATE products 
SET stock = stock - 1 
WHERE id = 1 AND stock >= 1
```
- 优点：强一致性
- 适用：低并发订单

**方案3：乐观锁**
```sql
UPDATE products 
SET stock = stock - 1, version = version + 1
WHERE id = 1 AND version = 10
```
- 优点：无锁、高性能
- 缺点：冲突重试

**3. 方案选择（2分钟）**
> - 秒杀场景：Redis + Lua
> - 普通订单：数据库直接 UPDATE
> - 分布式：Redlock 分布式锁

**4. 一致性保证（1分钟）**
> - Redis → MySQL：定时同步
> - MySQL → Redis：启动预热
> - 双写一致：Canal 监听 binlog
