### 2. 令牌桶限流

```javascript
class TokenBucket {
  constructor(capacity, refillRate) {
    this.capacity = capacity;        // 桶容量
    this.tokens = capacity;          // 当前令牌数
    this.refillRate = refillRate;    // 每秒填充速率
    this.lastRefillTime = Date.now();
  }
  
  refill() {
    const now = Date.now();
    const elapsed = (now - this.lastRefillTime) / 1000;
    const tokensToAdd = elapsed * this.refillRate;
    
    this.tokens = Math.min(this.capacity, this.tokens + tokensToAdd);
    this.lastRefillTime = now;
  }
  
  consume(tokens = 1) {
    this.refill();
    
    if (this.tokens >= tokens) {
      this.tokens -= tokens;
      return true;
    }
    
    return false;
  }
}

// Redis 实现（分布式）
const limitScript = `
local key = KEYS[1]
local capacity = tonumber(ARGV[1])
local rate = tonumber(ARGV[2])
local requested = tonumber(ARGV[3])
local now = tonumber(ARGV[4])

local bucket = redis.call('HMGET', key, 'tokens', 'last_refill')
local tokens = tonumber(bucket[1]) or capacity
local last_refill = tonumber(bucket[2]) or now

-- 填充令牌
local elapsed = (now - last_refill) / 1000
local tokens_to_add = elapsed * rate
tokens = math.min(capacity, tokens + tokens_to_add)

-- 消费令牌
if tokens >= requested then
  tokens = tokens - requested
  redis.call('HMSET', key, 'tokens', tokens, 'last_refill', now)
  redis.call('EXPIRE', key, 60)
  return 1
else
  return 0
end
`;

async function tryAcquire(userId, activityId) {
  const key = `limit:${activityId}:${userId}`;
  const result = await redis.eval(
    limitScript,
    1,
    key,
    10,      // capacity
    1,       // rate (1/秒)
    1,       // requested
    Date.now()
  );
  
  return result === 1;
}
```

### 3. 防刷策略

```javascript
// 多维度防刷
class AntiBrushSystem {
  // 1. 用户行为验证
  async validateUserBehavior(userId, activityId) {
    // 检查用户是否为新注册（防止批量注册）
    const user = await db.users.findById(userId);
    if (Date.now() - user.createdAt < 24 * 3600 * 1000) {
      throw new Error('New user not allowed');
    }
    
    // 检查是否实名认证
    if (!user.verified) {
      throw new Error('User not verified');
    }
    
    // 检查历史订单（防止异常账号）
    const orderCount = await db.orders.count({ userId });
    if (orderCount < 3) {
      throw new Error('Insufficient order history');
    }
  }
  
  // 2. 设备指纹
  async checkDeviceFingerprint(deviceId) {
    const key = `device:${deviceId}`;
    const count = await redis.incr(key);
    await redis.expire(key, 3600);
    
    if (count > 5) {
      throw new Error('Device limit exceeded');
    }
  }
  
  // 3. IP 限流
  async checkIpLimit(ip, activityId) {
    const key = `ip:${activityId}:${ip}`;
    const count = await redis.incr(key);
    await redis.expire(key, 60);
    
    if (count > 10) {
      throw new Error('IP limit exceeded');
    }
  }
  
  // 4. 验证码
  async validateCaptcha(userId, captcha) {
    const key = `captcha:${userId}`;
    const expected = await redis.get(key);
    
    if (captcha !== expected) {
      throw new Error('Invalid captcha');
    }
    
    await redis.del(key);
  }
}
```

### 4. 秒杀核心流程

```javascript
const express = require('express');
const Redis = require('ioredis');
const Queue = require('bull');

const app = express();
const redis = new Redis.Cluster([...]);
const orderQueue = new Queue('seckill-orders', {...});

// 秒杀 Lua 脚本
const seckillScript = `
local stockKey = KEYS[1]
local userKey = KEYS[2]
local userId = ARGV[1]
local quantity = tonumber(ARGV[2])

-- 检查是否已购买
if redis.call('SISMEMBER', userKey, userId) == 1 then
  return -2  -- 已购买
end

-- 检查库存
local stock = tonumber(redis.call('GET', stockKey) or 0)
if stock >= quantity then
  -- 扣减库存
  redis.call('DECRBY', stockKey, quantity)
  -- 记录购买用户
  redis.call('SADD', userKey, userId)
  return stock - quantity
else
  return -1  -- 库存不足
end
`;

// 秒杀接口
app.post('/api/seckill/:activityId', async (req, res) => {
  const { activityId } = req.params;
  const { userId, captcha } = req.body;
  
  try {
    // 1. 限流（令牌桶）
    const allowed = await tryAcquire(userId, activityId);
    if (!allowed) {
      return res.status(429).json({ error: 'Rate limit exceeded' });
    }
    
    // 2. 防刷验证
    const antiBrush = new AntiBrushSystem();
    await antiBrush.validateUserBehavior(userId, activityId);
    await antiBrush.checkDeviceFingerprint(req.headers['device-id']);
    await antiBrush.checkIpLimit(req.ip, activityId);
    await antiBrush.validateCaptcha(userId, captcha);
    
    // 3. 检查活动时间
    const activity = await redis.hgetall(`seckill:${activityId}`);
    const now = Date.now();
    
    if (now < activity.startTime) {
      return res.status(400).json({ error: 'Activity not started' });
    }
    
    if (now > activity.endTime) {
      return res.status(400).json({ error: 'Activity ended' });
    }
    
    // 4. 秒杀（原子扣减库存）
    const result = await redis.evalsha(
      seckillScriptSHA,
      2,
      `seckill:stock:${activityId}`,
      `seckill:users:${activityId}`,
      userId,
      1
    );
    
    if (result === -1) {
      return res.status(400).json({ error: 'Sold out' });
    }
    
    if (result === -2) {
      return res.status(400).json({ error: 'Already purchased' });
    }
    
    // 5. 异步创建订单（消息队列）
    await orderQueue.add({
      activityId,
      userId,
      quantity: 1,
      timestamp: Date.now()
    });
    
    res.json({
      success: true,
      message: 'Order queued',
      remainingStock: result
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: error.message });
  }
});

// 订单队列消费者
orderQueue.process(async (job) => {
  const { activityId, userId, quantity } = job.data;
  
  try {
    // 创建订单
    const order = await db.orders.insert({
      activityId,
      userId,
      quantity,
      status: 'pending',
      createdAt: Date.now()
    });
    
    console.log('Order created:', order.id);
    
    // 通知用户（WebSocket/推送）
    await notifyUser(userId, {
      type: 'order_created',
      orderId: order.id
    });
    
    return order;
  } catch (error) {
    // 失败回滚库存
    await redis.eval(`
      local stockKey = KEYS[1]
      local userKey = KEYS[2]
      local userId = ARGV[1]
      
      redis.call('INCRBY', stockKey, 1)
      redis.call('SREM', userKey, userId)
    `, 2,
      `seckill:stock:${activityId}`,
      `seckill:users:${activityId}`,
      userId
    );
    
    throw error;
  }
});
```

### 5. 系统降级

```javascript
// 降级策略配置
const degradeConfig = {
  level1: {  // 轻度降级
    disableFeatures: ['recommend', 'comments'],
    cacheOnly: true
  },
  level2: {  // 中度降级
    disableFeatures: ['recommend', 'comments', 'history'],
    staticResponse: true
  },
  level3: {  // 重度降级
    maintenanceMode: true,
    message: 'System is busy, please try again later'
  }
};

// 降级中间件
function degradeMiddleware(req, res, next) {
  const currentLevel = redis.get('degrade:level');
  
  if (currentLevel === 'level3') {
    return res.status(503).json({
      error: degradeConfig.level3.message
    });
  }
  
  if (currentLevel === 'level2') {
    // 返回静态响应
    return res.json({
      success: false,
      message: 'Please wait and try again'
    });
  }
  
  if (currentLevel === 'level1') {
    // 禁用非核心功能
    req.degradedFeatures = degradeConfig.level1.disableFeatures;
  }
  
  next();
}

// 自动降级（基于系统负载）
setInterval(async () => {
  const metrics = await getSystemMetrics();
  
  if (metrics.cpu > 90 || metrics.qps > 100000) {
    await redis.set('degrade:level', 'level3');
    console.log('Degraded to level 3');
  } else if (metrics.cpu > 70 || metrics.qps > 50000) {
    await redis.set('degrade:level', 'level2');
    console.log('Degraded to level 2');
  } else if (metrics.cpu > 50 || metrics.qps > 20000) {
    await redis.set('degrade:level', 'level1');
    console.log('Degraded to level 1');
  } else {
    await redis.del('degrade:level');
  }
}, 5000);
```
