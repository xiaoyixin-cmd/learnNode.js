## 💻 基础实现

### 1. Redis Sorted Set 原理

```
Sorted Set = Hash + Skip List

Hash：存储 member → score 映射
Skip List：按 score 排序（支持范围查询）

时间复杂度：
- ZADD：O(log N)
- ZRANGE：O(log N + M)
- ZRANK：O(log N)
```

```javascript
const Redis = require('ioredis');
const redis = new Redis();

// 添加/更新分数
async function updateScore(userId, score) {
  await redis.zadd('rank:global', score, userId);

// 获取排行榜（Top 100）
async function getTopN(n = 100) {
  const result = await redis.zrevrange('rank:global', 0, n - 1, 'WITHSCORES');
  
  const ranking = [];
  for (let i = 0; i < result.length; i += 2) {
    ranking.push({
      userId: result[i],
      score: parseInt(result[i + 1]),
      rank: Math.floor(i / 2) + 1
    });
  }
  
  return ranking;
}

// 获取用户排名
async function getUserRank(userId) {
  const rank = await redis.zrevrank('rank:global', userId);
  const score = await redis.zscore('rank:global', userId);
  
  if (rank === null) {
    return null;
  }
  
  return {
    userId,
    rank: rank + 1,  // 从1开始
    score: parseInt(score)
  };
}

async function incrScore(userId, delta) {
  const newScore = await redis.zincrby('rank:global', delta, userId);
```

### 1. 多维度排行榜

```javascript
class RankingSystem {
  constructor(redis) {
    this.redis = redis;
  }
  
  // 更新分数（同时更新日/周/月榜）
  async updateScore(userId, score) {
    const now = new Date();
    const day = now.toISOString().split('T')[0];      // 2025-01-01
    const week = this.getWeekKey(now);                // 2025-W01
    const month = now.toISOString().slice(0, 7);      // 2025-01
    
    await Promise.all([
      this.redis.zadd('rank:global', score, userId),
      this.redis.zadd(`rank:day:${day}`, score, userId),
      this.redis.zadd(`rank:week:${week}`, score, userId),
      this.redis.zadd(`rank:month:${month}`, score, userId)
    ]);
  }
  
  // 获取周榜 Key
  getWeekKey(date) {
    const year = date.getFullYear();
    const startOfYear = new Date(year, 0, 1);
    const days = Math.floor((date - startOfYear) / (24 * 60 * 60 * 1000));
    const week = Math.ceil((days + startOfYear.getDay() + 1) / 7);
    
    return `${year}-W${String(week).padStart(2, '0')}`;
  }
  
  // 获取日榜
  async getDailyRank(date, limit = 100) {
    const day = date.toISOString().split('T')[0];
    return this.getTopN(`rank:day:${day}`, limit);
  }
  
  // 获取周榜
  async getWeeklyRank(date, limit = 100) {
    const week = this.getWeekKey(date);
    return this.getTopN(`rank:week:${week}`, limit);
  }
  
  // 获取月榜
  async getMonthlyRank(date, limit = 100) {
    const month = date.toISOString().slice(0, 7);
    return this.getTopN(`rank:month:${month}`, limit);
  }
  
  async getTopN(key, limit) {
    const result = await this.redis.zrevrange(key, 0, limit - 1, 'WITHSCORES');
    
    const ranking = [];
    for (let i = 0; i < result.length; i += 2) {
      ranking.push({
        userId: result[i],
        score: parseInt(result[i + 1]),
        rank: Math.floor(i / 2) + 1
      });
    }
    
    return ranking;
  }
}

// 使用
const ranking = new RankingSystem(redis);

await ranking.updateScore('user-123', 1000);

const dailyRank = await ranking.getDailyRank(new Date(), 100);
console.log(dailyRank);
```

```javascript
const NodeCache = require('node-cache');
const localCache = new NodeCache({ stdTTL: 60 });

class CachedRanking {
  constructor(redis) {
    this.redis = redis;
  }
  
  // 带缓存的 Top N
  async getTopN(key, limit = 100) {
    const cacheKey = `ranking:${key}:${limit}`;
    
    // 检查本地缓存
    const cached = localCache.get(cacheKey);
    if (cached) {
      return cached;
    }
    
    // 从 Redis 获取
    const result = await this.redis.zrevrange(key, 0, limit - 1, 'WITHSCORES');
    
    const ranking = [];
    for (let i = 0; i < result.length; i += 2) {
      ranking.push({
        userId: result[i],
        score: parseInt(result[i + 1]),
        rank: Math.floor(i / 2) + 1
      });
    }
    
    // 缓存
    localCache.set(cacheKey, ranking);
    
    return ranking;
  }
  
  // 更新分数（清除缓存）
  async updateScore(userId, score) {
    await this.redis.zadd('rank:global', score, userId);
    
    // 清除缓存
    localCache.del('ranking:rank:global:100');
  }
}
```

### 3. 分片优化（千万级用户）

```javascript
// 一致性哈希分片
class ShardedRanking {
  constructor(redisCluster) {
    this.redis = redisCluster;
    this.shardCount = 10;
  }
  
  // 计算分片
  getShardId(userId) {
    let hash = 0;
    for (let i = 0; i < userId.length; i++) {
      hash = ((hash << 5) - hash) + userId.charCodeAt(i);
    }
    
    return Math.abs(hash) % this.shardCount;
  }
  
  // 更新分数
  async updateScore(userId, score) {
    const shardId = this.getShardId(userId);
    await this.redis.zadd(`rank:shard:${shardId}`, score, userId);
  }
  
  // 获取全局 Top N（合并分片）
  async getGlobalTopN(limit = 100) {
    const shardResults = [];
    
    // 从每个分片获取 Top N
    for (let i = 0; i < this.shardCount; i++) {
      const result = await this.redis.zrevrange(
        `rank:shard:${i}`,
        0,
        limit - 1,
        'WITHSCORES'
      );
      
      for (let j = 0; j < result.length; j += 2) {
        shardResults.push({
          userId: result[j],
          score: parseInt(result[j + 1])
        });
      }
    }
    
    // 排序并取 Top N
    shardResults.sort((a, b) => b.score - a.score);
    
    return shardResults.slice(0, limit).map((item, index) => ({
      ...item,
      rank: index + 1
    }));
  }
  
  // 获取用户排名（跨分片统计）
  async getUserRank(userId) {
    const shardId = this.getShardId(userId);
    
    const score = await this.redis.zscore(`rank:shard:${shardId}`, userId);
    
    if (!score) return null;
    
    // 统计所有分片中比该用户分数高的数量
    let higherCount = 0;
    
    for (let i = 0; i < this.shardCount; i++) {
      const count = await this.redis.zcount(
        `rank:shard:${i}`,
        parseInt(score) + 1,
        '+inf'
      );
      higherCount += count;
    }
    
    return {
      userId,
      score: parseInt(score),
      rank: higherCount + 1
    };
  }
}
```

### 4. 定时更新（批量）

```javascript
// 批量更新分数
class BatchUpdater {
  constructor(redis) {
    this.redis = redis;
    this.buffer = [];
    this.maxSize = 1000;
    this.flushInterval = 5000;
    
    setInterval(() => this.flush(), this.flushInterval);
  }
  
  // 添加更新
  add(userId, score) {
    this.buffer.push([score, userId]);
    
    if (this.buffer.length >= this.maxSize) {
      this.flush();
    }
  }
  
  // 批量刷新
  async flush() {
    if (this.buffer.length === 0) return;
    
    const updates = this.buffer.splice(0);
    
    // 使用 pipeline
    const pipeline = this.redis.pipeline();
    
    for (const [score, userId] of updates) {
      pipeline.zadd('rank:global', score, userId);
    }
    
    await pipeline.exec();
    
    console.log(`Flushed ${updates.length} updates`);
  }
}

const updater = new BatchUpdater(redis);

// 使用
updater.add('user-123', 1000);
updater.add('user-456', 2000);
```

### 5. 实时推送

```javascript
const io = require('socket.io')(server);
const Redis = require('ioredis');

const redis = new Redis();
const pub = new Redis();
const sub = new Redis();

// 订阅排行榜更新
sub.subscribe('ranking:update');

sub.on('message', async (channel, message) => {
  const { userId, score } = JSON.parse(message);
  
  // 获取排名
  const rank = await redis.zrevrank('rank:global', userId);
  
  // 推送给在线用户
  io.to(userId).emit('rank:update', {
    userId,
    score,
    rank: rank + 1
  });
  
  // 推送 Top 10 变化
  if (rank < 10) {
    const top10 = await getTopN(10);
    io.emit('top10:update', top10);
  }
});

// 更新分数并发布
async function updateScoreWithNotify(userId, score) {
  await redis.zadd('rank:global', score, userId);
  
  await pub.publish('ranking:update', JSON.stringify({ userId, score }));
}

// WebSocket 连接
io.on('connection', (socket) => {
  const userId = socket.handshake.query.userId;
  
  socket.join(userId);
  
  // 发送当前排名
  getUserRank(userId).then((rank) => {
    socket.emit('rank:current', rank);
  });
});
```

```javascript
async function updateScoreWithSegment(userId, score) {
  const segment = Math.floor(score / 1000);

// 获取 Top N（从高分段开始）
async function getTopNFromSegments(limit) {
  const ranking = [];
  let segment = 10;  // 从最高段开始
  
  while (ranking.length < limit && segment >= 0) {
    const result = await redis.zrevrange(
      `rank:segment:${segment}`,
      0,
      limit - ranking.length - 1,
      'WITHSCORES'
    );
    
    for (let i = 0; i < result.length; i += 2) {
      ranking.push({
        userId: result[i],
        score: parseInt(result[i + 1])
      });
    }
    
    segment--;
  }
  
  return ranking;
}
```
