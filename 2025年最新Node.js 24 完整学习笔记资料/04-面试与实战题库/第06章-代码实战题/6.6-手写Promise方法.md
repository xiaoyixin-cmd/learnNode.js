## 🎯 题目描述

实现以下 Promise 静态方法：
1. `Promise.all` - 所有成功才成功，一个失败就失败

3. `Promise.allSettled` - 所有Promise都完成（无论成败）
4. `Promise.any` - 第一个成功的结果（进阶）

## 💻 Promise.all 实现

### 基础实现（60分）

```javascript
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    
    promises.forEach((promise, index) => {
      Promise.resolve(promise).then(value => {
        results[index] = value;
        count++;
        
        if (count === promises.length) {
          resolve(results);
        }
      }).catch(reject);
    });
  });
}

// 使用
const p1 = Promise.resolve(1);
const p2 = Promise.resolve(2);
const p3 = Promise.resolve(3);

promiseAll([p1, p2, p3]).then(results => {
  console.log(results);  // [1, 2, 3]
});
```

### 完整实现（100分）

```javascript
/**
 * Promise.all 实现
 * @param {Iterable} iterable - 可迭代对象
 * @returns {Promise}
 */
function promiseAll(iterable) {
  return new Promise((resolve, reject) => {
    // 转为数组
    const promises = Array.from(iterable);
    
    // 空数组直接返回
    if (promises.length === 0) {
      resolve([]);
      return;
    }
    
    const results = [];
    let completedCount = 0;
    
    promises.forEach((promise, index) => {
      // 包装为 Promise（支持非Promise值）
      Promise.resolve(promise)
        .then(value => {
          results[index] = value;
          completedCount++;
          
          // 所有完成
          if (completedCount === promises.length) {
            resolve(results);
          }
        })
        .catch(error => {
          // 任一失败，立即 reject
          reject(error);
        });
    });
  });
}

// 测试
async function test() {
  // 测试1：全部成功
  const result1 = await promiseAll([
    Promise.resolve(1),
    Promise.resolve(2),
    Promise.resolve(3)
  ]);
  console.log(result1);  // [1, 2, 3]
  
  // 测试2：一个失败
  try {
    await promiseAll([
      Promise.resolve(1),
      Promise.reject(new Error('失败')),
      Promise.resolve(3)
    ]);
  } catch (error) {
    console.log(error.message);  // 失败
  }
  
  // 测试3：空数组
  const result3 = await promiseAll([]);
  console.log(result3);  // []
  
  // 测试4：非Promise值
  const result4 = await promiseAll([1, 2, 3]);
  console.log(result4);  // [1, 2, 3]
}

test();
```

## 💻 Promise.race 实现

### 完整实现（100分）

```javascript
/**
 * Promise.race 实现
 * @param {Iterable} iterable - 可迭代对象
 * @returns {Promise}
 */
function promiseRace(iterable) {
  return new Promise((resolve, reject) => {
    const promises = Array.from(iterable);
    
    // 空数组，永远 pending
    if (promises.length === 0) {
      return;
    }
    
    promises.forEach(promise => {
      Promise.resolve(promise)
        .then(resolve)   // 第一个成功
        .catch(reject);  // 第一个失败
    });
  });
}

// 测试
async function test() {
  // 测试1：第一个最快
  const result1 = await promiseRace([
    new Promise(resolve => setTimeout(() => resolve(1), 100)),
    new Promise(resolve => setTimeout(() => resolve(2), 200))
  ]);
  console.log(result1);  // 1
  
  // 测试2：第一个失败
  try {
    await promiseRace([
      Promise.reject(new Error('失败')),
      Promise.resolve(2)
    ]);
  } catch (error) {
    console.log(error.message);  // 失败
  }
}

test();
```

## 💻 Promise.allSettled 实现

### 完整实现（100分）

```javascript
/**
 * Promise.allSettled 实现
 * @param {Iterable} iterable - 可迭代对象
 * @returns {Promise<Array<{status, value|reason}>>}
 */
function promiseAllSettled(iterable) {
  return new Promise(resolve => {
    const promises = Array.from(iterable);
    
    // 空数组
    if (promises.length === 0) {
      resolve([]);
      return;
    }
    
    const results = [];
    let completedCount = 0;
    
    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then(value => {
          results[index] = {
            status: 'fulfilled',
            value
          };
        })
        .catch(reason => {
          results[index] = {
            status: 'rejected',
            reason
          };
        })
        .finally(() => {
          completedCount++;
          
          if (completedCount === promises.length) {
            resolve(results);
          }
        });
    });
  });
}

// 测试
async function test() {
  const result = await promiseAllSettled([
    Promise.resolve(1),
    Promise.reject(new Error('失败')),
    Promise.resolve(3)
  ]);
  
  console.log(result);
  // [
  //   { status: 'fulfilled', value: 1 },
  //   { status: 'rejected', reason: Error: 失败 },
  //   { status: 'fulfilled', value: 3 }
  // ]
}

test();
```

## 💻 Promise.any 实现（进阶）

```javascript
/**
 * Promise.any 实现
 * @param {Iterable} iterable - 可迭代对象
 * @returns {Promise}
 */
function promiseAny(iterable) {
  return new Promise((resolve, reject) => {
    const promises = Array.from(iterable);
    
    // 空数组
    if (promises.length === 0) {
      reject(new AggregateError([], 'All promises were rejected'));
      return;
    }
    
    const errors = [];
    let rejectedCount = 0;
    
    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then(value => {
          // 任一成功，立即 resolve
          resolve(value);
        })
        .catch(reason => {
          errors[index] = reason;
          rejectedCount++;
          
          // 全部失败
          if (rejectedCount === promises.length) {
            reject(new AggregateError(errors, 'All promises were rejected'));
          }
        });
    });
  });
}

// 测试
async function test() {
  // 测试1：有成功的
  const result1 = await promiseAny([
    Promise.reject(new Error('失败1')),
    Promise.resolve(2),
    Promise.reject(new Error('失败3'))
  ]);
  console.log(result1);  // 2
  
  // 测试2：全部失败
  try {
    await promiseAny([
      Promise.reject(new Error('失败1')),
      Promise.reject(new Error('失败2'))
    ]);
  } catch (error) {
    console.log(error);  // AggregateError
  }
}

test();
```

## ⚠️ 常见错误

### 错误1：使用 for 循环导致闭包问题

```javascript
// ❌ 错误
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    
    for (let i = 0; i < promises.length; i++) {
      promises[i].then(value => {
        results[i] = value;  // i 变量会变化
        count++;
        
        if (count === promises.length) {
          resolve(results);
        }
      });
    }
  });
}

// ✅ 正确（使用 forEach 或箭头函数）
promises.forEach((promise, index) => {
  promise.then(value => {
    results[index] = value;
    // ...
  });
});
```

### 错误2：忘记处理空数组

```javascript
// ❌ 错误（永远不会 resolve）
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    if (promises.length === 0) {
      // 忘记 return
    }
    
    // ...
  });
}

// ✅ 正确
if (promises.length === 0) {
  resolve([]);
  return;
}
```

### 错误3：未包装非Promise值

```javascript
// ❌ 错误
promises.forEach((promise, index) => {
  promise.then(value => {  // 非Promise会报错
    // ...
  });
});

// ✅ 正确
Promise.resolve(promise).then(value => {
  // ...
});
```

### 2. 并发限制（p-limit）

```javascript
function promiseAllWithLimit(promises, limit) {
  return new Promise((resolve, reject) => {
    const results = [];
    let index = 0;
    let activeCount = 0;
    let completedCount = 0;
    
    function next() {
      if (index >= promises.length) return;
      
      const currentIndex = index++;
      activeCount++;
      
      Promise.resolve(promises[currentIndex])
        .then(value => {
          results[currentIndex] = value;
        })
        .catch(reject)
        .finally(() => {
          activeCount--;
          completedCount++;
          
          if (completedCount === promises.length) {
            resolve(results);
          } else if (activeCount < limit) {
            next();
          }
        });
    }
    
    // 启动 limit 个并发
    for (let i = 0; i < limit && i < promises.length; i++) {
      next();
    }
  });
}
```

### 3. 超时控制

```javascript
function promiseWithTimeout(promise, timeout) {
  return Promise.race([
    promise,
    new Promise((_, reject) => 
      setTimeout(() => reject(new Error('Timeout')), timeout)
    )
  ]);
}

// 使用
await promiseWithTimeout(fetch('/api'), 5000);
```

## 💡 总结

### 核心要点
1. **Promise.all**：计数器 + 全部成功
2. **Promise.race**：第一个完成
3. **Promise.allSettled**：状态记录 + 全部完成
4. **Promise.any**：第一个成功 + AggregateError

### 实现关键
```javascript
// Promise.all
function promiseAll(iterable) {
  return new Promise((resolve, reject) => {
    const promises = Array.from(iterable);
    
    if (promises.length === 0) {
      resolve([]);
      return;
    }
    
    const results = [];
    let count = 0;
    
    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then(value => {
          results[index] = value;
          if (++count === promises.length) {
            resolve(results);
          }
        })
        .catch(reject);
    });
  });
}

// Promise.race
function promiseRace(iterable) {
  return new Promise((resolve, reject) => {
    Array.from(iterable).forEach(promise => {
      Promise.resolve(promise).then(resolve).catch(reject);
    });
  });
}

      Promise.resolve(promise)
        .then(value => {
          results[index] = { status: 'fulfilled', value };
        })
        .catch(reason => {
          results[index] = { status: 'rejected', reason };
        })
        .finally(() => {
          if (++count === promises.length) {

          }

    });
  });
}
```
