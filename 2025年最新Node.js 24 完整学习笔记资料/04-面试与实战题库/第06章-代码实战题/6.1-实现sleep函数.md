# 6.1 实现 sleep 函数

> 掌握基于 Promise 的延时函数实现，支持 async/await 和取消功能

---

## 🎯 题目描述

实现一个 `sleep` 函数，要求：
1. 支持延时指定毫秒数
2. 返回 Promise，可用于 async/await
3. 支持取消延时（进阶）

## 💻 基础实现（60分）

### 方法1：Promise + setTimeout

```javascript
function sleep(ms) {
  return new Promise(resolve => {
    setTimeout(resolve, ms);
  });
}

// 使用
async function demo() {
  console.log('开始');
  await sleep(1000);
  console.log('1秒后');
}

demo();
```

## 💻 进阶实现（80分）

### 方法2：支持取消

```javascript
function sleep(ms) {
  let timeoutId;
  
  const promise = new Promise((resolve, reject) => {
    timeoutId = setTimeout(resolve, ms);
  });
  
  // 添加 cancel 方法
  promise.cancel = () => {
    clearTimeout(timeoutId);
  };
  
  return promise;
}

// 使用
async function demo() {
  const sleepPromise = sleep(5000);
  
  // 1秒后取消
  setTimeout(() => {
    sleepPromise.cancel();
    console.log('已取消');
  }, 1000);
  
  await sleepPromise;
  console.log('5秒后（不会执行）');
}
```

**评分：80分**
- ✅ 支持取消
- ❌ 取消后 Promise 未 reject

## 💻 完美实现（100分）

### 方法3：完整版（支持取消 + 错误处理）

```javascript
/**
 * 延时函数
 * @param {number} ms - 延时毫秒数
 * @param {AbortSignal} [signal] - 取消信号（可选）
 * @returns {Promise<void>}
 */
function sleep(ms, signal) {
  return new Promise((resolve, reject) => {
    // 参数校验
    if (typeof ms !== 'number' || ms < 0) {
      reject(new TypeError('ms must be a non-negative number'));
      return;
    }
    
    // 检查是否已取消
    if (signal?.aborted) {
      reject(new DOMException('Aborted', 'AbortError'));
      return;
    }
    
    const timeoutId = setTimeout(resolve, ms);
    
    // 监听取消事件
    if (signal) {
      signal.addEventListener('abort', () => {
        clearTimeout(timeoutId);
        reject(new DOMException('Aborted', 'AbortError'));
      }, { once: true });
    }
  });
}

// 使用示例1：基础用法
async function demo1() {
  console.log('开始');
  await sleep(1000);
  console.log('1秒后');
}

// 使用示例2：取消
async function demo2() {
  const controller = new AbortController();
  
  setTimeout(() => {
    controller.abort();
    console.log('已取消');
  }, 500);
  
  try {
    await sleep(2000, controller.signal);
    console.log('2秒后（不会执行）');
  } catch (error) {
    if (error.name === 'AbortError') {
      console.log('睡眠被取消');
    }
  }
}

// 使用示例3：错误处理
async function demo3() {
  try {
    await sleep(-100);
  } catch (error) {
    console.error(error.message);  // ms must be a non-negative number
  }
}
```

**评分：100分**
- ✅ 完整的参数校验
- ✅ 标准 AbortController API
- ✅ 正确的错误处理
- ✅ JSDoc 注释

## ⚠️ 常见错误

### 错误1：忘记返回 Promise

```javascript
// ❌ 错误
function sleep(ms) {
  setTimeout(() => {
    console.log('时间到');
  }, ms);
}

// ✅ 正确
function sleep(ms) {
  return new Promise(resolve => {
    setTimeout(resolve, ms);
  });
}
```

### 错误2：取消后未 reject

```javascript
// ❌ 错误（Promise 永不 resolve）
function sleep(ms) {
  let timeoutId;
  
  const promise = new Promise(resolve => {
    timeoutId = setTimeout(resolve, ms);
  });
  
  promise.cancel = () => {
    clearTimeout(timeoutId);
    // 没有 reject，Promise 永远挂起
  };
  
  return promise;
}

// ✅ 正确
function sleep(ms) {
  return new Promise((resolve, reject) => {
    const timeoutId = setTimeout(resolve, ms);
    
    // ... 取消时 reject
  });
}
```

### 错误3：未清理定时器

```javascript
// ❌ 错误（内存泄漏）
async function demo() {
  const controller = new AbortController();
  
  controller.abort();
  
  await sleep(10000, controller.signal);  // setTimeout 未清理
}

// ✅ 正确
function sleep(ms, signal) {
  return new Promise((resolve, reject) => {
    const timeoutId = setTimeout(resolve, ms);
    
    if (signal) {
      signal.addEventListener('abort', () => {
        clearTimeout(timeoutId);  // 清理定时器
        reject(new DOMException('Aborted', 'AbortError'));
      });
    }
  });
}
```
