# 6.5 用原生 http 模块实现 RESTful CRUD

> 掌握原生 http 模块，实现完整的 REST API，含路由、JSON 解析、状态码

---

## 🎯 题目描述

用 Node.js 原生 `http` 模块实现 RESTful CRUD API，要求：
1. `GET /users` - 获取所有用户
2. `GET /users/:id` - 获取单个用户
3. `POST /users` - 创建用户
4. `PUT /users/:id` - 更新用户
5. `DELETE /users/:id` - 删除用户
6. 数据存储在内存（Map/Array）
7. 正确的状态码（200/201/404/400/500）

---

- ⭐⭐⭐ 路由解析

## 💻 完整实现（100分）

```javascript
const http = require('http');
const url = require('url');

// 内存存储
const users = new Map();
let nextId = 1;

// 初始数据
users.set(1, { id: 1, name: 'Alice', email: 'alice@example.com' });
users.set(2, { id: 2, name: 'Bob', email: 'bob@example.com' });
nextId = 3;

/**
 * 解析请求体（JSON）
 */
function parseBody(req) {
  return new Promise((resolve, reject) => {
    let body = '';
    
    req.on('data', chunk => {
      body += chunk.toString();
    });
    
    req.on('end', () => {
      try {
        resolve(body ? JSON.parse(body) : {});
      } catch (error) {
        reject(new Error('Invalid JSON'));
      }
    });
    
    req.on('error', reject);
  });
}

/**
 * 发送 JSON 响应
 */
function sendJSON(res, statusCode, data) {
  res.writeHead(statusCode, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify(data));
}

/**
 * 路由处理器
 */
const routes = {
  // GET /users - 获取所有用户
  'GET /users': (req, res, params) => {
    const userList = Array.from(users.values());
    sendJSON(res, 200, { data: userList });
  },
  
  // GET /users/:id - 获取单个用户
  'GET /users/:id': (req, res, params) => {
    const id = parseInt(params.id);
    const user = users.get(id);
    
    if (!user) {
      sendJSON(res, 404, { error: 'User not found' });
      return;
    }
    
    sendJSON(res, 200, { data: user });
  },
  
  // POST /users - 创建用户
  'POST /users': async (req, res, params) => {
    try {
      const body = await parseBody(req);
      
      // 验证
      if (!body.name || !body.email) {
        sendJSON(res, 400, { error: 'Missing name or email' });
        return;
      }
      
      // 创建
      const user = {
        id: nextId++,
        name: body.name,
        email: body.email
      };
      
      users.set(user.id, user);
      
      sendJSON(res, 201, { data: user });
    } catch (error) {
      sendJSON(res, 400, { error: error.message });
    }
  },
  
  // PUT /users/:id - 更新用户
  'PUT /users/:id': async (req, res, params) => {
    try {
      const id = parseInt(params.id);
      const user = users.get(id);
      
      if (!user) {
        sendJSON(res, 404, { error: 'User not found' });
        return;
      }
      
      const body = await parseBody(req);
      
      // 更新
      if (body.name) user.name = body.name;
      if (body.email) user.email = body.email;
      
      users.set(id, user);
      
      sendJSON(res, 200, { data: user });
    } catch (error) {
      sendJSON(res, 400, { error: error.message });
    }
  },
  
  // DELETE /users/:id - 删除用户
  'DELETE /users/:id': (req, res, params) => {
    const id = parseInt(params.id);
    
    if (!users.has(id)) {
      sendJSON(res, 404, { error: 'User not found' });
      return;
    }
    
    users.delete(id);
    
    sendJSON(res, 204, null);
  }
};

/**
 * 路由匹配
 */
function matchRoute(method, pathname) {
  for (const [pattern, handler] of Object.entries(routes)) {
    const [routeMethod, routePath] = pattern.split(' ');
    
    if (routeMethod !== method) continue;
    
    // 构建正则
    const paramNames = [];
    const regexPath = routePath.replace(/:([^/]+)/g, (match, paramName) => {
      paramNames.push(paramName);
      return '([^/]+)';
    });
    
    const regex = new RegExp(`^${regexPath}$`);
    const match = pathname.match(regex);
    
    if (match) {
      const params = {};
      paramNames.forEach((name, index) => {
        params[name] = match[index + 1];
      });
      
      return { handler, params };
    }
  }
  
  return null;
}

/**
 * 创建服务器
 */
const server = http.createServer(async (req, res) => {
  const { pathname } = url.parse(req.url);
  const method = req.method;
  
  console.log(`${method} ${pathname}`);
  
  // CORS
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');
  
  if (method === 'OPTIONS') {
    res.writeHead(204);
    res.end();
    return;
  }
  
  // 路由匹配
  const route = matchRoute(method, pathname);
  
  if (!route) {
    sendJSON(res, 404, { error: 'Not found' });
    return;
  }
  
  try {
    await route.handler(req, res, route.params);
  } catch (error) {
    console.error(error);
    sendJSON(res, 500, { error: 'Internal server error' });
  }
});

const PORT = 3000;

server.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);
});
```

### 使用 Node.js

```javascript
const http = require('http');

async function testAPI() {
  // GET /users
  const getAll = await request('GET', '/users');
  console.log('GET /users:', getAll);
  
  // POST /users
  const created = await request('POST', '/users', {
    name: 'Charlie',
    email: 'charlie@example.com'
  });
  console.log('POST /users:', created);
  
  // GET /users/:id
  const getOne = await request('GET', `/users/${created.data.id}`);
  console.log('GET /users/:id:', getOne);
  
  // PUT /users/:id
  const updated = await request('PUT', `/users/${created.data.id}`, {
    name: 'Charlie Updated'
  });
  console.log('PUT /users/:id:', updated);
  
  // DELETE /users/:id
  await request('DELETE', `/users/${created.data.id}`);
  console.log('DELETE /users/:id: 成功');
}

function request(method, path, body) {
  return new Promise((resolve, reject) => {
    const options = {
      hostname: 'localhost',
      port: 3000,
      path,
      method,
      headers: {
        'Content-Type': 'application/json'
      }
    };
    
    const req = http.request(options, (res) => {
      let data = '';
      
      res.on('data', chunk => {
        data += chunk;
      });
      
      res.on('end', () => {
        resolve(data ? JSON.parse(data) : null);
      });
    });
    
    req.on('error', reject);
    
    if (body) {
      req.write(JSON.stringify(body));
    }
    
    req.end();
  });
}

testAPI();
```

## ⚠️ 常见错误

### 错误1：未监听 data 事件

```javascript
// ❌ 错误（请求体为空）
function parseBody(req) {
  return JSON.parse(req.body);  // req.body 不存在
}

// ✅ 正确
function parseBody(req) {
  return new Promise((resolve, reject) => {
    let body = '';
    
    req.on('data', chunk => {
      body += chunk.toString();
    });
    
    req.on('end', () => {
      resolve(JSON.parse(body));
    });
  });
}
```

### 错误2：状态码不规范

```javascript
// ❌ 错误
res.writeHead(200);  // 创建成功应该 201

// ✅ 正确
res.writeHead(201);  // Created
```

### 错误3：路由参数解析错误

```javascript
// ❌ 错误（硬编码）
if (pathname === '/users/1') {
  // ...
}

// ✅ 正确（正则提取）
const regex = /^\/users\/([^/]+)$/;
const match = pathname.match(regex);
const id = match ? match[1] : null;
```

## 📝 RESTful 规范

### 状态码

| 状态码 | 含义 | 使用场景 |
|-------|------|---------|
| 200 | OK | 成功获取/更新 |
| 201 | Created | 成功创建 |
| 204 | No Content | 成功删除 |
| 400 | Bad Request | 请求参数错误 |
| 404 | Not Found | 资源不存在 |
| 500 | Internal Server Error | 服务器错误 |

### 请求方法

| 方法 | 操作 | 幂等性 |
|------|------|--------|
| GET | 查询 | ✅ |
| POST | 创建 | ❌ |
| PUT | 更新（全量） | ✅ |
| PATCH | 更新（部分） | ❌ |
| DELETE | 删除 | ✅ |
