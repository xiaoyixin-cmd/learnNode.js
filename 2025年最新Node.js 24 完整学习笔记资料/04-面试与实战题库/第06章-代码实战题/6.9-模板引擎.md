# 6.9 实现简单的模板引擎

> 掌握模板引擎原理，使用正则替换 + with 语句实现变量插值和逻辑控制

---

## 🎯 题目描述

实现一个简单的模板引擎，要求：
1. 支持 `{{ variable }}` 变量插值
2. 支持 `{% if condition %}...{% endif %}` 条件判断
3. 支持 `{% for item in list %}...{% endfor %}` 循环
4. 支持转义（防止 XSS）
5. 支持嵌套

```javascript
function render(template, data) {
```

## 💻 进阶实现（80分）

### 支持条件和循环

```javascript
function render(template, data) {
  // 1. 处理 if 语句
  template = template.replace(
    /\{% if (.+?) %\}(.+?)\{% endif %\}/g,
    (match, condition, content) => {
      const result = evalWithContext(condition, data);
      return result ? render(content, data) : '';
    }
  );
  
  // 2. 处理 for 循环
  template = template.replace(
    /\{% for (.+?) in (.+?) %\}(.+?)\{% endfor %\}/g,
    (match, item, list, content) => {
      const array = evalWithContext(list, data);
      
      return array
        .map(value => render(content, { ...data, [item.trim()]: value }))
        .join('');
    }
  );
  
  // 3. 处理变量插值
  template = template.replace(/\{\{(.+?)\}\}/g, (match, key) => {
    return evalWithContext(key, data);
  });
  
  return template;
}

function evalWithContext(expr, context) {
  try {
    with (context) {
      return eval(expr);
    }
  } catch (error) {
    return '';
  }
}

// 使用
const template = `
  {% if isAdmin %}
    <h1>管理员：{{ name }}</h1>
  {% endif %}
  
  <ul>
    {% for user in users %}
      <li>{{ user }}</li>
    {% endfor %}
  </ul>
`;

const result = render(template, {
  isAdmin: true,
  name: 'Alice',
  users: ['Bob', 'Charlie']
});

console.log(result);
```

**评分：80分**
- ✅ 条件和循环
- ✅ 嵌套支持
- ❌ 无 XSS 防护

## 💻 完整实现（100分）

```javascript
/**
 * 简单模板引擎
 */
class TemplateEngine {
  constructor(options = {}) {
    this.escapeHTML = options.escapeHTML !== false;
  }
  
  /**
   * 渲染模板
   * @param {string} template - 模板字符串
   * @param {Object} data - 数据对象
   * @returns {string} 渲染结果
   */
  render(template, data) {
    // 1. 处理 if 语句
    template = this.handleIf(template, data);
    
    // 2. 处理 for 循环
    template = this.handleFor(template, data);
    
    // 3. 处理变量插值
    template = this.handleVariable(template, data);
    
    return template;
  }
  
  /**
   * 处理 if 语句
   */
  handleIf(template, data) {
    const regex = /\{%\s*if\s+(.+?)\s*%\}(.+?)\{%\s*endif\s*%\}/gs;
    
    return template.replace(regex, (match, condition, content) => {
      const result = this.evaluate(condition, data);
      return result ? this.render(content, data) : '';
    });
  }
  
  /**
   * 处理 for 循环
   */
  handleFor(template, data) {
    const regex = /\{%\s*for\s+(\w+)\s+in\s+(.+?)\s*%\}(.+?)\{%\s*endfor\s*%\}/gs;
    
    return template.replace(regex, (match, item, list, content) => {
      const array = this.evaluate(list, data);
      
      if (!Array.isArray(array)) {
        return '';
      }
      
      return array
        .map(value => {
          const itemData = { ...data, [item]: value };
          return this.render(content, itemData);
        })
        .join('');
    });
  }
  
  /**
   * 处理变量插值
   */
  handleVariable(template, data) {
    const regex = /\{\{\s*(.+?)\s*\}\}/g;
    
    return template.replace(regex, (match, expr) => {
      const value = this.evaluate(expr, data);
      
      if (value === null || value === undefined) {
        return '';
      }
      
      return this.escapeHTML ? this.escape(String(value)) : String(value);
    });
  }
  
  /**
   * 执行表达式
   */
  evaluate(expr, context) {
    try {
      // 使用 Function 替代 eval（更安全）
      const func = new Function(...Object.keys(context), `return ${expr}`);
      return func(...Object.values(context));
    } catch (error) {
      console.error(`Template error: ${error.message}`);
      return '';
    }
  }
  
  /**
   * HTML 转义
   */
  escape(html) {
    const map = {
      '&': '&amp;',
      '<': '&lt;',
      '>': '&gt;',
      '"': '&quot;',
      "'": '&#39;'
    };
    
    return html.replace(/[&<>"']/g, char => map[char]);
  }
}

// 使用示例
const engine = new TemplateEngine({ escapeHTML: true });

const template = `
<!DOCTYPE html>
<html>
<head>
  <title>{{ title }}</title>
</head>
<body>
  {% if isAdmin %}
    <h1>欢迎，管理员 {{ name }}！</h1>
  {% endif %}
  
  <h2>用户列表：</h2>
  <ul>
    {% for user in users %}
      <li>{{ user.name }} - {{ user.email }}</li>
    {% endfor %}
  </ul>
  
  {% if users.length > 0 %}
    <p>共 {{ users.length }} 个用户</p>
  {% endif %}
</body>
</html>
`;

const result = engine.render(template, {
  title: '用户管理',
  isAdmin: true,
  name: 'Alice',
  users: [
    { name: 'Bob', email: 'bob@example.com' },
    { name: 'Charlie', email: 'charlie@example.com' }
  ]
});

console.log(result);
```

```javascript
function render(template, data) {
```

## ⚠️ 常见错误

### 错误1：正则贪婪匹配

```javascript
// ❌ 错误（会匹配到最后一个 }}）
template.replace(/\{\{(.+)\}\}/g, ...);

// ✅ 正确（非贪婪）
template.replace(/\{\{(.+?)\}\}/g, ...);
```

### 错误2：忘记转义 HTML

```javascript
// ❌ 错误（XSS 风险）
return data[key];

// ✅ 正确
return escapeHTML(data[key]);
```

### 错误3：使用 eval

```javascript
// ❌ 错误（不安全）
eval(expr);

// ✅ 正确（使用 Function）
new Function(...keys, `return ${expr}`)(...values);
```

```javascript
// 词法分析
function tokenize(template) {
  const tokens = [];
  const regex = /\{\{(.+?)\}\}|\{%(.+?)%\}/g;
  let lastIndex = 0;
  
  template.replace(regex, (match, variable, logic, offset) => {
    // 添加文本 token
    if (offset > lastIndex) {
      tokens.push({
        type: 'text',
        value: template.slice(lastIndex, offset)
      });
    }
    
    // 添加变量或逻辑 token
    if (variable) {
      tokens.push({ type: 'variable', value: variable.trim() });
    } else if (logic) {
      tokens.push({ type: 'logic', value: logic.trim() });
    }
    
    lastIndex = offset + match.length;
  });
  
  return tokens;
}
```
