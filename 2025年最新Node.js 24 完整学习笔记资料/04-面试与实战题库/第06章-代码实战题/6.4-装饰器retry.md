## 💻 基础实现（60分）

```javascript
function retry(maxRetries = 3, delay = 1000) {
  return function(target, propertyKey, descriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function(...args) {
      let lastError;
      
      for (let i = 0; i <= maxRetries; i++) {
        try {
          return await originalMethod.apply(this, args);
        } catch (error) {
          lastError = error;
          
          if (i < maxRetries) {
            await sleep(delay);
          }
        }
      }
      
      throw lastError;
    };
    
    return descriptor;
  };
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// 使用
class ApiClient {
  @retry(3, 1000)
  async fetchData(url) {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }
    
    return response.json();
  }
}

const client = new ApiClient();
await client.fetchData('https://api.example.com/data');
```

## 💻 完整实现（100分）

### JavaScript 版本

```javascript
/**
 * 重试装饰器
 * @param {Object} options - 配置
 * @param {number} options.maxRetries - 最大重试次数
 * @param {number} options.delay - 延迟时间（ms）
 * @param {boolean} options.exponentialBackoff - 是否指数退避
 * @param {Function} options.shouldRetry - 重试条件判断
 * @param {Function} options.onRetry - 重试回调
 */
function retry(options = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    exponentialBackoff = false,
    shouldRetry = () => true,
    onRetry = () => {}
  } = options;
  
  return function(target, propertyKey, descriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function(...args) {
      let lastError;
      let attempt = 0;
      
      while (attempt <= maxRetries) {
        try {
          const result = await originalMethod.apply(this, args);
          return result;
        } catch (error) {
          lastError = error;
          attempt++;
          
          // 检查是否还能重试
          const canRetry = attempt <= maxRetries && shouldRetry(error, attempt);
          
          if (!canRetry) {
            break;
          }
          
          // 计算延迟
          const retryDelay = exponentialBackoff
            ? delay * Math.pow(2, attempt - 1)
            : delay;
          
          // 触发回调
          onRetry(error, attempt, maxRetries, retryDelay);
          
          // 延迟
          await sleep(retryDelay);
        }
      }
      
      // 所有重试失败
      throw new Error(
        `Failed after ${attempt} attempts: ${lastError.message}`,
        { cause: lastError }
      );
    };
    
    return descriptor;
  };
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// 使用示例
class ApiClient {
  @retry({
    maxRetries: 5,
    delay: 1000,
    exponentialBackoff: true,
    shouldRetry: (error) => {
      // 只重试网络错误和超时
      return error.message.includes('timeout') || 
             error.message.includes('network');
    },
    onRetry: (error, attempt, maxRetries, delay) => {
      console.log(`[Retry ${attempt}/${maxRetries}] ${error.message} (delay: ${delay}ms)`);
    }
  })
  async fetchData(url) {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }
    
    return response.json();
  }
}

const client = new ApiClient();

try {
  const data = await client.fetchData('https://api.example.com/data');
  console.log(data);
} catch (error) {
  console.error('最终失败:', error.message);
}
```

```typescript
interface RetryOptions {
  maxRetries?: number;
  delay?: number;
  exponentialBackoff?: boolean;
  shouldRetry?: (error: Error, attempt: number) => boolean;
  onRetry?: (error: Error, attempt: number, maxRetries: number, delay: number) => void;
}

function retry(options: RetryOptions = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    exponentialBackoff = false,
    shouldRetry = () => true,
    onRetry = () => {}
  } = options;
  
  return function(
    target: any,
    propertyKey: string,
    descriptor: PropertyDescriptor
  ): PropertyDescriptor {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function(...args: any[]) {
      let lastError: Error;
      let attempt = 0;
      
      while (attempt <= maxRetries) {
        try {
          return await originalMethod.apply(this, args);
        } catch (error) {
          lastError = error as Error;
          attempt++;
          
          const canRetry = attempt <= maxRetries && shouldRetry(lastError, attempt);
          
          if (!canRetry) {
            break;
          }
          
          const retryDelay = exponentialBackoff
            ? delay * Math.pow(2, attempt - 1)
            : delay;
          
          onRetry(lastError, attempt, maxRetries, retryDelay);
          
          await new Promise(resolve => setTimeout(resolve, retryDelay));
        }
      }
      
      throw new Error(
        `Failed after ${attempt} attempts: ${lastError!.message}`,
        { cause: lastError }
      );
    };
    
    return descriptor;
  };
}

// 使用
class ApiClient {
  @retry({
    maxRetries: 3,
    delay: 1000,
    exponentialBackoff: true
  })
  async fetchData(url: string): Promise<any> {
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }
    
    return response.json();
  }
}
```

## 💻 函数式实现（不使用装饰器）

```javascript
function withRetry(fn, options = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    exponentialBackoff = false,
    shouldRetry = () => true
  } = options;
  
  return async function(...args) {
    let lastError;
    let attempt = 0;
    
    while (attempt <= maxRetries) {
      try {
        return await fn.apply(this, args);
      } catch (error) {
        lastError = error;
        attempt++;
        
        if (attempt <= maxRetries && shouldRetry(error, attempt)) {
          const retryDelay = exponentialBackoff
            ? delay * Math.pow(2, attempt - 1)
            : delay;
          
          await sleep(retryDelay);
        } else {
          break;
        }
      }
    }
    
    throw lastError;
  };
}

// 使用
const fetchData = withRetry(
  async (url) => {
    const response = await fetch(url);
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return response.json();
  },
  {
    maxRetries: 3,
    delay: 1000,
    exponentialBackoff: true
  }
);

await fetchData('https://api.example.com/data');
```

## ⚠️ 常见错误

### 错误1：忘记 await sleep

```javascript
// ❌ 错误（不会真正延迟）
if (i < maxRetries) {
  sleep(delay);  // 忘记 await
}

// ✅ 正确
if (i < maxRetries) {
  await sleep(delay);
}
```

### 错误2：指数退避计算错误

```javascript
// ❌ 错误
const retryDelay = delay * Math.pow(2, i);
// i=0: delay*1
// i=1: delay*2
// i=2: delay*4

// ✅ 正确（第一次重试应该是 delay）
const retryDelay = delay * Math.pow(2, i - 1);
// i=1: delay*1
// i=2: delay*2
// i=3: delay*4
```

### 错误3：重试次数理解错误

```javascript
// ❌ 错误（实际只重试2次）
for (let i = 0; i < maxRetries; i++) {
  // maxRetries=3: i=0,1,2（3次尝试，2次重试）
}

// ✅ 正确（重试3次）
for (let i = 0; i <= maxRetries; i++) {
  // maxRetries=3: i=0,1,2,3（4次尝试，3次重试）
}
```

```javascript
function retry(options) {
  return function(target, key, descriptor) {
    const original = descriptor.value;
    
    descriptor.value = async function(...args) {
      let attempt = 0;
      
      while (attempt <= options.maxRetries) {
        try {
          return await original.apply(this, args);
        } catch (error) {
          attempt++;
          
          if (attempt > options.maxRetries || !options.shouldRetry(error)) {
            throw error;
          }
          
          const delay = options.delay * Math.pow(2, attempt - 1);
          await sleep(delay);
        }
      }
    };
    
    return descriptor;
  };
}
```
