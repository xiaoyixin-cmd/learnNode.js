# 6.3 实现 pipeline 流式管道

> 掌握 Node.js Stream pipeline 实现，含错误传播和自动清理

---

## 🎯 题目描述

实现一个 `pipeline` 函数，连接多个流（Readable → Transform → Writable），要求：
1. 自动管道连接（pipe）
2. 错误传播（任一流错误，停止所有流）
3. 自动清理（destroy 所有流）
4. 支持回调和 Promise

## 💻 基础实现（60分）

```javascript
function pipeline(...streams) {
  if (streams.length < 2) {
    throw new Error('pipeline requires at least 2 streams');
  }
  
  // 连接流
  for (let i = 0; i < streams.length - 1; i++) {
    streams[i].pipe(streams[i + 1]);
  }
  
  // 返回最后一个流
  return streams[streams.length - 1];
}

// 使用

const writable = fs.createWriteStream('output.txt');
```

## 💻 进阶实现（80分）

```javascript
function pipeline(...args) {
  // 分离流和回调
  const callback = typeof args[args.length - 1] === 'function' 
    ? args.pop() 
    : null;
  
  const streams = args;
  
  if (streams.length < 2) {
    throw new Error('pipeline requires at least 2 streams');
  }
  
  let error;
  
  // 错误处理
  const onError = (err) => {
    if (error) return;  // 只处理第一个错误
    
    error = err;
    
    // 销毁所有流
    for (const stream of streams) {
      if (stream && typeof stream.destroy === 'function') {
        stream.destroy(err);
      }
    }
    
    if (callback) {
      callback(err);
    }
  };
  
  // 监听错误
  for (const stream of streams) {
    stream.on('error', onError);
  }
  
  // 连接流
  for (let i = 0; i < streams.length - 1; i++) {
    streams[i].pipe(streams[i + 1]);
  }
  
  // 监听最后一个流的完成
  const lastStream = streams[streams.length - 1];
  
  lastStream.on('finish', () => {
    if (!error && callback) {
      callback();
    }
  });
  
  return lastStream;
}

// 使用
pipeline(readable, uppercase, writable, (err) => {
  if (err) {
    console.error('Pipeline 错误:', err);
  } else {
    console.log('Pipeline 完成');
  }
});
```

**评分：80分**
- ✅ 错误处理
- ✅ 自动清理
- ✅ 回调支持
- ❌ 无 Promise 支持

## 💻 完整实现（100分）

```javascript
const { promisify } = require('util');
const { destroyer } = require('stream');

/**
 * Pipeline 实现
 * @param {...Stream|Function} args - 流和回调
 * @returns {Stream|Promise}
 */
function pipeline(...args) {
  const callback = typeof args[args.length - 1] === 'function' 
    ? args.pop() 
    : null;
  
  const streams = args;
  
  // 参数校验
  if (streams.length < 2) {
    throw new Error('pipeline requires at least 2 streams');
  }
  
  let error;
  let completed = false;
  
  // 清理函数
  const cleanup = () => {
    if (completed) return;
    completed = true;
    
    for (const stream of streams) {
      if (stream && typeof stream.destroy === 'function') {
        stream.destroy(error);
      }
    }
  };
  
  // 错误处理
  const onError = (err) => {
    if (error) return;
    error = err;
    cleanup();
    
    if (callback) {
      callback(err);
    }
  };
  
  // 完成处理
  const onFinish = () => {
    if (error) return;
    cleanup();
    
    if (callback) {
      callback();
    }
  };
  
  // 监听所有流的错误
  for (const stream of streams) {
    stream.on('error', onError);
  }
  
  // 连接流
  for (let i = 0; i < streams.length - 1; i++) {
    const source = streams[i];
    const destination = streams[i + 1];
    
    source.pipe(destination);
    
    // 监听 unpipe（手动断开）
    source.on('unpipe', () => {
      onError(new Error('Stream unpipe'));
    });
  }
  
  // 监听最后一个流
  const lastStream = streams[streams.length - 1];
  
  if (lastStream.writable) {
    lastStream.on('finish', onFinish);
  } else {
    lastStream.on('end', onFinish);
  }
  
  // Promise 支持
  if (!callback) {
    return new Promise((resolve, reject) => {
      pipeline(...streams, (err) => {
        if (err) {
          reject(err);
        } else {
          resolve();
        }
      });
    });
  }
  
  return lastStream;
}

// 使用示例1：回调
pipeline(readable, uppercase, writable, (err) => {
  if (err) {
    console.error('错误:', err);
  } else {
    console.log('完成');
  }
});

// 使用示例2：Promise
async function run() {
  try {
    await pipeline(readable, uppercase, writable);
    console.log('完成');
  } catch (error) {
    console.error('错误:', error);
  }
}

run();
```

**评分：100分**
- ✅ 完整错误处理
- ✅ 自动清理资源
- ✅ 回调和 Promise 支持
- ✅ unpipe 事件监听

## ⚠️ 常见错误

### 错误1：忘记销毁流

```javascript
// ❌ 错误（内存泄漏）
const onError = (err) => {
  if (callback) {
    callback(err);
  }
  // 忘记 destroy
};

// ✅ 正确
const onError = (err) => {
  for (const stream of streams) {
    stream.destroy(err);
  }
  
  if (callback) {
    callback(err);
  }
};
```

### 错误2：重复处理错误

```javascript
// ❌ 错误（多次触发回调）
const onError = (err) => {
  cleanup();
  callback(err);  // 可能多次调用
};

// ✅ 正确
let error;

const onError = (err) => {
  if (error) return;  // 防止重复
  error = err;
  cleanup();
  callback(err);
};
```

### 错误3：未处理 unpipe

```javascript
// ❌ 错误（手动 unpipe 未检测）
source.pipe(destination);

// ✅ 正确
source.pipe(destination);

source.on('unpipe', () => {
```

### 核心要点

### 实现关键
```javascript
function pipeline(...args) {
  const callback = typeof args[args.length - 1] === 'function' 
    ? args.pop() 
    : null;
  
  const streams = args;
  let error;
  
  const cleanup = () => {
    streams.forEach(s => s.destroy(error));
  };
  
  const onError = (err) => {
    if (error) return;
    error = err;
    cleanup();
    callback?.(err);
  };
  
  const onFinish = () => {
    if (!error) {
      cleanup();
      callback?.();
    }
  };
  
  streams.forEach(s => s.on('error', onError));
  
  for (let i = 0; i < streams.length - 1; i++) {
    streams[i].pipe(streams[i + 1]);
  }
  
  streams[streams.length - 1].on('finish', onFinish);
  
  if (!callback) {
    return new Promise((resolve, reject) => {
      pipeline(...streams, err => err ? reject(err) : resolve());
    });
  }
}
```
