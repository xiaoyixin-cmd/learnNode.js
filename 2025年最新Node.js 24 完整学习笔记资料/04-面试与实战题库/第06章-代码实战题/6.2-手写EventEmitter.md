## 🎯 题目描述

实现一个 `EventEmitter` 类，要求：
1. `on(event, listener)` - 注册事件监听器
2. `emit(event, ...args)` - 触发事件
3. `once(event, listener)` - 注册一次性监听器
4. `off(event, listener)` - 移除监听器
5. `removeAllListeners(event)` - 移除所有监听器
6. 错误处理（error 事件）
7. 内存泄漏检测（进阶）

## 💻 基础实现（60分）

```javascript
class EventEmitter {
  constructor() {
    this.events = {};
  }
  
  // 注册事件
  on(event, listener) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    
    this.events[event].push(listener);
  }
  
  // 触发事件
  emit(event, ...args) {
    const listeners = this.events[event];
    
    if (!listeners) return false;
    
    listeners.forEach(listener => {
      listener(...args);
    });
    
    return true;
  }
  
  // 移除事件
  off(event, listener) {
    const listeners = this.events[event];
    
    if (!listeners) return;
    
    const index = listeners.indexOf(listener);
    if (index !== -1) {
      listeners.splice(index, 1);
    }
  }
}

// 使用
const emitter = new EventEmitter();

emitter.on('data', (msg) => {
  console.log('收到:', msg);
});

emitter.emit('data', 'Hello');  // 输出: 收到: Hello
```

## 💻 进阶实现（80分）

```javascript
class EventEmitter {
  constructor() {
    this.events = new Map();
  }
  
  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    
    this.events.get(event).push(listener);
    return this;  // 支持链式调用
  }
  
  emit(event, ...args) {
    const listeners = this.events.get(event);
    
    if (!listeners) return false;
    
    // 克隆数组，避免在遍历时修改
    const clone = [...listeners];
    
    clone.forEach(listener => {
      listener.apply(this, args);
    });
    
    return true;
  }
  
  once(event, listener) {
    const wrapper = (...args) => {
      listener.apply(this, args);
      this.off(event, wrapper);
    };
    
    // 保存原始 listener（用于 off）
    wrapper.listener = listener;
    
    this.on(event, wrapper);
    return this;
  }
  
  off(event, listener) {
    const listeners = this.events.get(event);
    
    if (!listeners) return this;
    
    const index = listeners.findIndex(l => 
      l === listener || l.listener === listener
    );
    
    if (index !== -1) {
      listeners.splice(index, 1);
    }
    
    return this;
  }
  
  removeAllListeners(event) {
    if (event) {
      this.events.delete(event);
    } else {
      this.events.clear();
    }
    
    return this;
  }
}

// 使用

});

emitter.emit('login', 'Alice');  // 输出: 用户登录: Alice
emitter.emit('login', 'Bob');    // 无输出
```

## 💻 完美实现（100分）

```javascript
class EventEmitter {
  constructor() {
    this.events = new Map();
    this.maxListeners = 10;  // 默认最大监听器数
  }
  
  /**
   * 注册事件监听器
   * @param {string} event - 事件名
   * @param {Function} listener - 监听器函数
   */
  on(event, listener) {
    if (typeof listener !== 'function') {
      throw new TypeError('listener must be a function');
    }
    
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    
    const listeners = this.events.get(event);
    listeners.push(listener);
    
    // 内存泄漏警告
    if (listeners.length > this.maxListeners) {
      console.warn(
        `MaxListenersExceededWarning: ${listeners.length} ${event} listeners added. ` +
        `Use emitter.setMaxListeners() to increase limit`
      );
    }
    
    return this;
  }
  
  /**
   * 触发事件
   * @param {string} event - 事件名
   * @param {...any} args - 参数
   */
  emit(event, ...args) {
    // 错误事件特殊处理
    if (event === 'error') {
      if (!this.events.has('error') || this.events.get('error').length === 0) {
        const error = args[0];
        
        if (error instanceof Error) {
          throw error;
        } else {
          throw new Error('Uncaught, unspecified "error" event');
        }
      }
    }
    
    const listeners = this.events.get(event);
    
    if (!listeners) return false;
    
    // 克隆数组，避免在遍历时修改
    const clone = [...listeners];
    
    clone.forEach(listener => {
      try {
        listener.apply(this, args);
      } catch (error) {
        // 监听器内部错误
        this.emit('error', error);
      }
    });
    
    return true;
  }
  
  /**
   * 注册一次性监听器
   * @param {string} event - 事件名
   * @param {Function} listener - 监听器函数
   */
  once(event, listener) {
    if (typeof listener !== 'function') {
      throw new TypeError('listener must be a function');
    }
    
    const wrapper = (...args) => {
      listener.apply(this, args);
      this.off(event, wrapper);
    };
    
    // 保存原始 listener（用于 off）
    wrapper.listener = listener;
    
    this.on(event, wrapper);
    return this;
  }
  
  /**
   * 移除监听器
   * @param {string} event - 事件名
   * @param {Function} listener - 监听器函数
   */
  off(event, listener) {
    const listeners = this.events.get(event);
    
    if (!listeners) return this;
    
    const index = listeners.findIndex(l => 
      l === listener || l.listener === listener
    );
    
    if (index !== -1) {
      listeners.splice(index, 1);
    }
    
    // 如果没有监听器了，删除事件
    if (listeners.length === 0) {
      this.events.delete(event);
    }
    
    return this;
  }
  
  /**
   * 移除所有监听器
   * @param {string} [event] - 事件名（可选）
   */
  removeAllListeners(event) {
    if (event) {
      this.events.delete(event);
    } else {
      this.events.clear();
    }
    
    return this;
  }
  
  /**
   * 获取监听器列表
   * @param {string} event - 事件名
   */
  listeners(event) {
    return this.events.get(event) || [];
  }
  
  /**
   * 获取监听器数量
   * @param {string} event - 事件名
   */
  listenerCount(event) {
    const listeners = this.events.get(event);
    return listeners ? listeners.length : 0;
  }
  
  /**
   * 设置最大监听器数
   * @param {number} n - 数量
   */
  setMaxListeners(n) {
    if (typeof n !== 'number' || n < 0) {
      throw new TypeError('n must be a non-negative number');
    }
    
    this.maxListeners = n;
    return this;
  }
  
  /**
   * 获取事件名列表
   */
  eventNames() {
    return Array.from(this.events.keys());
  }
}

// 使用示例
const emitter = new EventEmitter();

// 基础使用
emitter.on('data', (msg) => {
  console.log('收到:', msg);
});

emitter.emit('data', 'Hello');  // 输出: 收到: Hello

// once 使用
emitter.once('login', (user) => {
  console.log('用户登录:', user);
});

emitter.emit('login', 'Alice');  // 输出: 用户登录: Alice
emitter.emit('login', 'Bob');    // 无输出

// 错误处理
emitter.on('error', (error) => {
  console.error('错误:', error.message);
});

emitter.on('process', () => {
  throw new Error('处理失败');
});

emitter.emit('process');  // 输出: 错误: 处理失败

// 链式调用
emitter
  .on('start', () => console.log('开始'))
  .on('end', () => console.log('结束'))
  .emit('start')
  .emit('end');
```

- ✅ 错误处理（error 事件）
- ✅ 内存泄漏警告

## ⚠️ 常见错误

### 错误1：遍历时修改数组

```javascript
// ❌ 错误
emit(event, ...args) {
  const listeners = this.events.get(event);
  
  listeners.forEach(listener => {
    listener(...args);
    // 如果 listener 内部调用 off，会修改正在遍历的数组
  });
}

// ✅ 正确
emit(event, ...args) {
  const listeners = this.events.get(event);
  const clone = [...listeners];  // 克隆数组
  
  clone.forEach(listener => {
    listener(...args);
  });
}
```

### 错误2：once 无法 off

```javascript
// ❌ 错误
once(event, listener) {
  const wrapper = (...args) => {
    listener(...args);
    this.off(event, wrapper);
  };
  
  this.on(event, wrapper);  // 无法通过原始 listener 移除
}

// ✅ 正确
once(event, listener) {
  const wrapper = (...args) => {
    listener(...args);
    this.off(event, wrapper);
  };
  
  wrapper.listener = listener;  // 保存原始 listener
  this.on(event, wrapper);
}

off(event, listener) {
  const index = listeners.findIndex(l => 
    l === listener || l.listener === listener  // 比较原始 listener
  );
  
  // ...
}
```

### 错误3：error 事件未处理

```javascript
// ❌ 错误（未捕获的错误会导致进程崩溃）
emit(event, ...args) {
  listeners.forEach(listener => {
    listener(...args);  // 如果抛出错误，未捕获
  });
}

// ✅ 正确
emit(event, ...args) {
  listeners.forEach(listener => {
    try {
      listener(...args);
    } catch (error) {
      this.emit('error', error);
    }
  });
}
```

```javascript
class EventEmitter {
  constructor() {
    this.events = new Map();
    this.maxListeners = 10;
  }
  
  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(listener);
    
    // 检查内存泄漏
    if (this.events.get(event).length > this.maxListeners) {
      console.warn('MaxListenersExceededWarning');
    }
    
    return this;
  }
  
  emit(event, ...args) {
    const listeners = this.events.get(event);
    if (!listeners) return false;
    
    [...listeners].forEach(listener => {
      try {
        listener.apply(this, args);
      } catch (error) {
        this.emit('error', error);
      }
    });
    
    return true;
  }
  
  once(event, listener) {
    const wrapper = (...args) => {
      listener.apply(this, args);
      this.off(event, wrapper);
    };
    wrapper.listener = listener;
    this.on(event, wrapper);
    return this;
  }
}
```
