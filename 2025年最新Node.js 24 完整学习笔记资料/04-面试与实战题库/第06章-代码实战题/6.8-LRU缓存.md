# 6.8 实现 LRU 缓存

> 掌握 LRU (Least Recently Used) 缓存算法，使用 Map + 双向链表实现 O(1) 操作

---

## 🎯 题目描述

实现一个 LRU (最近最少使用) 缓存，要求：
1. `get(key)` - 获取值，O(1) 时间复杂度
2. `put(key, value)` - 设置值，O(1) 时间复杂度
3. 容量限制，超出时淘汰最久未使用的项
4. 访问或设置时更新"最近使用"状态

---

## 📋 核心考点

- ⭐⭐⭐ 双向链表数据结构
- ⭐⭐⭐ 哈希表（Map）
- ⭐⭐ 时间复杂度 O(1)
- ⭐⭐ 链表节点操作

## 💻 方法1：Map（ES2015+ 特性）

### 完整实现（90分，推荐）

```javascript
/**
 * LRU 缓存（使用 Map 顺序特性）
 */
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
  }
  
  /**
   * 获取值
   * @param {*} key - 键
   * @returns {*} 值，不存在返回 -1
   */
  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }
    
    // 更新访问顺序：删除后重新插入
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    
    return value;
  }
  
  /**
   * 设置值
   * @param {*} key - 键
   * @param {*} value - 值
   */
  put(key, value) {
    // 如果存在，先删除（更新顺序）
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }
    
    // 插入新值
    this.cache.set(key, value);
    
    // 超出容量，删除最旧的（第一个）
    if (this.cache.size > this.capacity) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
  }
}

// 使用
const cache = new LRUCache(2);

cache.put(1, 1);  // 缓存: {1=1}
cache.put(2, 2);  // 缓存: {1=1, 2=2}
console.log(cache.get(1));  // 返回 1，缓存: {2=2, 1=1}

cache.put(3, 3);  // 删除 key 2，缓存: {1=1, 3=3}
console.log(cache.get(2));  // 返回 -1（未找到）

cache.put(4, 4);  // 删除 key 1，缓存: {3=3, 4=4}
console.log(cache.get(1));  // 返回 -1
console.log(cache.get(3));  // 返回 3
console.log(cache.get(4));  // 返回 4
```

**评分：90分**
- ✅ O(1) 时间复杂度
- ✅ 简洁易懂
- ⚠️ 依赖 Map 的顺序特性（ES2015+）

## 💻 方法2：Map + 双向链表（100分）

### 双向链表节点

```javascript
class ListNode {
  constructor(key, value) {
    this.key = key;
    this.value = value;
    this.prev = null;
    this.next = null;
  }
}
```

### 完整实现

```javascript
/**
 * LRU 缓存（Map + 双向链表）
 */
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();  // key → ListNode
    
    // 虚拟头尾节点
    this.head = new ListNode(0, 0);
    this.tail = new ListNode(0, 0);
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }
  
  /**
   * 获取值
   * @param {*} key - 键
   * @returns {*} 值，不存在返回 -1
   */
  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }
    
    const node = this.cache.get(key);
    
    // 移动到头部（最近使用）
    this.moveToHead(node);
    
    return node.value;
  }
  
  /**
   * 设置值
   * @param {*} key - 键
   * @param {*} value - 值
   */
  put(key, value) {
    if (this.cache.has(key)) {
      // 更新值并移到头部
      const node = this.cache.get(key);
      node.value = value;
      this.moveToHead(node);
    } else {
      // 创建新节点
      const node = new ListNode(key, value);
      this.cache.set(key, node);
      this.addToHead(node);
      
      // 超出容量，删除尾部
      if (this.cache.size > this.capacity) {
        const removed = this.removeTail();
        this.cache.delete(removed.key);
      }
    }
  }
  
  /**
   * 添加节点到头部
   * @param {ListNode} node - 节点
   */
  addToHead(node) {
    node.prev = this.head;
    node.next = this.head.next;
    this.head.next.prev = node;
    this.head.next = node;
  }
  
  /**
   * 移除节点
   * @param {ListNode} node - 节点
   */
  removeNode(node) {
    node.prev.next = node.next;
    node.next.prev = node.prev;
  }
  
  /**
   * 移动节点到头部
   * @param {ListNode} node - 节点
   */
  moveToHead(node) {
    this.removeNode(node);
    this.addToHead(node);
  }
  
  /**
   * 移除尾部节点
   * @returns {ListNode} 被移除的节点
   */
  removeTail() {
    const node = this.tail.prev;
    this.removeNode(node);
    return node;
  }
  
  /**
   * 获取所有键（从最近到最旧）
   * @returns {Array} 键数组
   */
  keys() {
    const keys = [];
    let node = this.head.next;
    
    while (node !== this.tail) {
      keys.push(node.key);
      node = node.next;
    }
    
    return keys;
  }
  
  /**
   * 清空缓存
   */
  clear() {
    this.cache.clear();
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }
}

// 使用
const cache = new LRUCache(2);

cache.put(1, 1);
cache.put(2, 2);
console.log(cache.get(1));  // 1

cache.put(3, 3);
console.log(cache.get(2));  // -1

cache.put(4, 4);
console.log(cache.get(1));  // -1
console.log(cache.get(3));  // 3
console.log(cache.get(4));  // 4
```

**评分：100分**
- ✅ O(1) 时间复杂度
- ✅ 标准双向链表实现
- ✅ 不依赖 Map 顺序特性

## 💻 方法3：纯对象实现（60分）

```javascript
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = {};
    this.keys = [];  // 记录访问顺序
  }
  
  get(key) {
    if (!(key in this.cache)) {
      return -1;
    }
    
    // 更新访问顺序（移到末尾）
    const index = this.keys.indexOf(key);
    this.keys.splice(index, 1);
    this.keys.push(key);
    
    return this.cache[key];
  }
  
  put(key, value) {
    if (key in this.cache) {
      const index = this.keys.indexOf(key);
      this.keys.splice(index, 1);
    } else if (this.keys.length >= this.capacity) {
      // 删除最旧的
      const oldKey = this.keys.shift();
      delete this.cache[oldKey];
    }
    
    this.cache[key] = value;
    this.keys.push(key);
  }
}
```

**评分：60分**
- ⚠️ indexOf/splice 时间复杂度 O(n)
- ⚠️ 性能较差

## ⚠️ 常见错误

### 错误1：使用数组导致 O(n)

```javascript
// ❌ 错误（indexOf 是 O(n)）
get(key) {
  const index = this.keys.indexOf(key);  // O(n)
  this.keys.splice(index, 1);            // O(n)
  this.keys.push(key);
  
  return this.cache[key];
}

// ✅ 正确（Map O(1)）
get(key) {
  const value = this.cache.get(key);
  this.cache.delete(key);
  this.cache.set(key, value);
  return value;
}
```

### 错误2：忘记更新顺序

```javascript
// ❌ 错误（get 未更新顺序）
get(key) {
  return this.cache.get(key) || -1;
}

// ✅ 正确
get(key) {
  if (!this.cache.has(key)) return -1;
  
  const value = this.cache.get(key);
  this.cache.delete(key);
  this.cache.set(key, value);  // 更新顺序
  return value;
}
```

### 错误3：双向链表操作错误

```javascript
// ❌ 错误（未处理虚拟节点）
removeTail() {
  return this.tail;  // 错误，应该是 tail.prev
}

// ✅ 正确
removeTail() {
  const node = this.tail.prev;
  this.removeNode(node);
  return node;
}
```

## 📝 扩展知识

### 1. LRU vs LFU vs FIFO

| 策略 | 淘汰规则 | 使用场景 |
|------|---------|---------|
| **LRU** | 最久未使用 | 通用缓存 |
| **LFU** | 使用频率最低 | 热点数据 |
| **FIFO** | 最先进入 | 简单队列 |

### 2. Redis LRU 实现

```
Redis LRU 是近似 LRU：
1. 随机采样 N 个 key
2. 淘汰其中最久未使用的
3. 时间复杂度 O(1)，节省内存
```

### 3. 实际应用

```javascript
// HTTP 缓存
class HttpCache {
  constructor(maxSize = 100) {
    this.cache = new LRUCache(maxSize);
  }
  
  async fetch(url) {
    const cached = this.cache.get(url);
    
    if (cached) {
      return cached;
    }
    
    const response = await fetch(url);
    const data = await response.json();
    
    this.cache.put(url, data);
    return data;
  }
}
```

## 💡 总结

### 核心要点
1. **Map 方案**：利用顺序特性，delete + set
2. **链表方案**：双向链表 + 虚拟头尾节点
3. **时间复杂度**：get/put O(1)
4. **空间复杂度**：O(capacity)

### 实现关键

**Map 方案（推荐）：**
```javascript
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
  }
  
  get(key) {
    if (!this.cache.has(key)) return -1;
    
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }
  
  put(key, value) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }
    
    this.cache.set(key, value);
    
    if (this.cache.size > this.capacity) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
  }
}
```

**双向链表方案：**
```javascript
// Map: key → ListNode
// 链表: head ↔ node1 ↔ node2 ↔ tail

get(key) {
  const node = this.cache.get(key);
  this.moveToHead(node);
  return node.value;
}

put(key, value) {
  const node = new ListNode(key, value);
  this.addToHead(node);
  
  if (this.cache.size > this.capacity) {
    const removed = this.removeTail();
    this.cache.delete(removed.key);
  }
}
```
