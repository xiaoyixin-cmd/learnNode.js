## 📚 核心知识点

### 1. 执行优先级对比

```
当前操作完成后：
  ↓
1. process.nextTick 队列（优先级最高）
  ↓
2. Promise 微任务队列
  ↓
3. queueMicrotask 队列
  ↓
进入下一个事件循环阶段
```

**优先级顺序：**
```javascript
process.nextTick > Promise.then > queueMicrotask
```

---

### 2. 基础示例

```javascript
console.log('1');

Promise.resolve().then(() => console.log('2'));

process.nextTick(() => console.log('3'));

console.log('4');

// 输出：
// 1
// 4
// 3  ← nextTick 先执行
// 2  ← Promise 后执行
```

**执行流程：**
1. 同步代码：`1` → `4`
2. nextTick 队列：`3`
3. Promise 队列：`2`

---

### 3. 为什么 nextTick 更快？

#### **原因 1：独立队列**
- nextTick 有**专属队列**，在微任务之前执行
- Promise 使用**标准微任务队列**

```javascript
// Node.js 内部实现（简化版）
function runMicrotasks() {
  // 1. 先执行 nextTick 队列
  while (nextTickQueue.length > 0) {
    const callback = nextTickQueue.shift();
    callback();
  }
  
  // 2. 再执行微任务队列
  while (microtaskQueue.length > 0) {
    const microtask = microtaskQueue.shift();
    microtask();
  }
}
```

---

#### **原因 2：执行时机更早**
- nextTick 在**当前操作结束后立即执行**
- Promise 在 nextTick 队列清空后才执行

```javascript
function foo() {
  console.log('foo start');
  
  process.nextTick(() => console.log('nextTick'));
  Promise.resolve().then(() => console.log('promise'));
  
  console.log('foo end');
}

foo();

// 输出：
// foo start
// foo end
// nextTick  ← 当前操作结束后立即执行
// promise   ← nextTick 队列清空后执行
```

---

### 4. 嵌套调用对比

```javascript
// nextTick 嵌套
process.nextTick(() => {
  console.log('nextTick 1');
  process.nextTick(() => {
    console.log('nextTick 2');
  });
});

Promise.resolve().then(() => {
  console.log('promise 1');
  Promise.resolve().then(() => {
    console.log('promise 2');
  });
});

// 输出：
// nextTick 1
// nextTick 2  ← 递归 nextTick 立即执行
// promise 1
// promise 2
```

**执行流程：**
1. nextTick 队列：`nextTick 1`（注册 `nextTick 2`）
2. nextTick 队列：`nextTick 2`（队列清空）
3. Promise 队列：`promise 1`（注册 `promise 2`）
4. Promise 队列：`promise 2`

### 案例 2：递归 nextTick 导致 I/O 饥饿

```javascript
let i = 0;

function recursiveNextTick() {
  if (i < 10000) {
    i++;
    process.nextTick(recursiveNextTick);
  }
}

recursiveNextTick();

setTimeout(() => {
  console.log('timeout', i);  // 延迟执行
}, 0);

setImmediate(() => {
  console.log('immediate', i);  // 延迟执行
}, 0);

// 输出：
// timeout 10000  （延迟约 10ms）
// immediate 10000（延迟约 10ms）
```

**问题：**
- nextTick 递归阻塞事件循环
- setTimeout 和 setImmediate 无法执行
- 导致 **I/O 饥饿**

**解决方案：**
```javascript
// ✅ 使用 setImmediate 代替
let i = 0;

function recursiveImmediate() {
  if (i < 10000) {
    i++;
    setImmediate(recursiveImmediate);  // 让出控制权
  }
}

recursiveImmediate();

setTimeout(() => {
  console.log('timeout', i);  // 正常执行
}, 0);

// 输出：
// timeout 0（或小于 10000 的值）
```

### 案例 3：错误处理中的应用

```javascript
class EventEmitter {
  emit(event, ...args) {
    const listeners = this.listeners[event] || [];
    
    // ❌ 错误：同步执行可能抛出异常
    listeners.forEach(listener => {
      listener(...args);  // 可能抛出异常
    });
  }
  
  // ✅ 正确：使用 nextTick 异步执行
  emitAsync(event, ...args) {
    const listeners = this.listeners[event] || [];
    
    listeners.forEach(listener => {
      process.nextTick(() => {
        try {
          listener(...args);
        } catch (error) {
          console.error('Listener error:', error);
        }
      });
    });
  }
}

// 使用示例
const emitter = new EventEmitter();

emitter.on('data', (data) => {
  throw new Error('Listener error');
});

try {
  emitter.emit('data', 'hello');  // 同步，会抛出异常
} catch (error) {
  console.error('Caught:', error);
}

emitter.emitAsync('data', 'hello');  // 异步，不会抛出异常
console.log('Continue execution');
```

## 🎓 面试标准答题

### 问题 1："process.nextTick 和 Promise.then 哪个先执行？"

**标准答案：**
> process.nextTick **总是先执行**。
> 
> **原因：**
> 1. **独立队列**：nextTick 有专属队列，在微任务队列之前执行
> 2. **执行时机**：nextTick 在当前操作完成后立即执行，Promise 在 nextTick 队列清空后执行
> 3. **优先级设计**：Node.js 有意设计 nextTick 优先级最高，用于处理关键回调
> 
> **执行流程：**
> ```
> 当前操作完成
>   ↓
> 执行 nextTick 队列
>   ↓
> 执行 Promise 微任务队列
>   ↓
> 进入下一个事件循环阶段
> ```
> 
> **示例：**
> ```javascript
> Promise.resolve().then(() => console.log('promise'));
> process.nextTick(() => console.log('nextTick'));
> // 输出：nextTick → promise
> ```

**加分项：**
> 需要注意的是，如果递归调用 nextTick，会导致 I/O 饥饿，因为事件循环无法进入下一个阶段。建议在循环场景使用 setImmediate 代替。

---

### 问题 2："递归调用 nextTick 会有什么问题？"

**标准答案：**
> 递归调用 nextTick 会导致 **I/O 饥饿**（I/O Starvation）。
> 
> **原因：**
> 1. nextTick 队列必须清空才能进入下一个事件循环阶段
> 2. 递归调用导致队列永远无法清空
> 3. setTimeout、setImmediate、I/O 回调无法执行
> 
> **示例问题：**
> ```javascript
> let i = 0;
> function loop() {
>   if (i < 10000) {
>     i++;
>     process.nextTick(loop);  // 阻塞事件循环
>   }
> }
> loop();
> 
> setTimeout(() => console.log('timeout'), 0);  // 延迟 10ms+
> ```
> 
> **解决方案：**
> ```javascript
> function loop() {
>   if (i < 10000) {
>     i++;
>     setImmediate(loop);  // ✅ 让出控制权
>   }
> }
> ```
> 
> **对比：**
> - **nextTick**：立即执行，不让出控制权
> - **setImmediate**：在 check 阶段执行，让出控制权

## 💡 应用场景

### 场景 1：确保回调异步执行

```javascript
// ❌ 问题：同步执行回调
class Database {
  query(sql, callback) {
    if (this.cache[sql]) {
      callback(null, this.cache[sql]);  // 同步执行
    } else {
      this.db.query(sql, callback);  // 异步执行
    }
  }
}

// 使用
db.query('SELECT *', (err, data) => {
  console.log('callback');
});
console.log('after query');

// 问题：有时输出 "callback → after query"（缓存命中）
//      有时输出 "after query → callback"（缓存未命中）

// ✅ 解决：使用 nextTick 确保异步
class Database {
  query(sql, callback) {
    if (this.cache[sql]) {
      process.nextTick(() => {
        callback(null, this.cache[sql]);  // 异步执行
      });
    } else {
      this.db.query(sql, callback);
    }
  }
}

// 现在总是输出："after query → callback"
```

```javascript
const EventEmitter = require('events');

    super();

  }
}

emitter.on('ready', () => {
  console.log('ready');  // ✅ 执行
});
```

### 场景 3：错误优先处理

```javascript
function readFile(path, callback) {
  if (!path) {
    // ❌ 同步抛出错误
    return callback(new Error('Path required'));
  }
  
  fs.readFile(path, callback);
}

// ✅ 使用 nextTick 统一异步
function readFile(path, callback) {
  if (!path) {
    process.nextTick(() => {
      callback(new Error('Path required'));
    });
    return;
  }
  
  fs.readFile(path, callback);
}
```

## 🔬 性能对比

### 测试 1：执行速度

```javascript
const { performance } = require('perf_hooks');

// 测试 nextTick
function testNextTick(count) {
  return new Promise((resolve) => {
    const start = performance.now();
    let i = 0;
    
    function next() {
      i++;
      if (i < count) {
        process.nextTick(next);
      } else {
        resolve(performance.now() - start);
      }
    }
    
    next();
  });
}

// 测试 Promise
function testPromise(count) {
  return new Promise((resolve) => {
    const start = performance.now();
    let i = 0;
    
    function next() {
      i++;
      if (i < count) {
        Promise.resolve().then(next);
      } else {
        resolve(performance.now() - start);
      }
    }
    
    next();
  });
}

(async () => {
  const count = 10000;
  
  const nextTickTime = await testNextTick(count);
  console.log(`nextTick: ${nextTickTime.toFixed(2)}ms`);
  
  const promiseTime = await testPromise(count);
  console.log(`Promise: ${promiseTime.toFixed(2)}ms`);
  
  console.log(`nextTick 快 ${(promiseTime / nextTickTime).toFixed(2)} 倍`);
})();

// 输出示例：
// nextTick: 12.34ms
// Promise: 15.67ms
// nextTick 快 1.27 倍
```

## 📝 总结

本节学习了：

✅ **执行优先级**：nextTick > Promise > queueMicrotask  
✅ **快的原因**：独立队列、执行时机更早  
✅ **递归风险**：I/O 饥饿、事件循环阻塞  
✅ **应用场景**：确保异步执行、构造函数发出事件、错误优先处理

✅ **性能对比**：nextTick 比 Promise 快约 20-30%
