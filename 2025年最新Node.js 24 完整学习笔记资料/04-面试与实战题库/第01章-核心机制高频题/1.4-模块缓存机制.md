## 📚 核心知识点

### 1. 缓存机制原理

**基本规则：**
- 模块**第一次加载**时执行代码并缓存
- 后续 require 直接返回缓存的 `module.exports`
- 缓存键是**模块的绝对路径**

```javascript
// module.js
console.log('Module loaded');
module.exports = { value: 42 };

// app.js
const mod1 = require('./module');  // 输出：Module loaded
const mod2 = require('./module');  // 无输出（使用缓存）

console.log(mod1 === mod2);  // true（同一个对象）
```

---

### 2. require.cache 数据结构

```javascript
// require.cache 结构
{
  '/path/to/module.js': Module {
    id: '/path/to/module.js',
    exports: { value: 42 },
    parent: Module { ... },
    filename: '/path/to/module.js',
    loaded: true,
    children: [],
    paths: [ ... ]
  },
  ...
}
```

**查看缓存：**
```javascript
console.log(Object.keys(require.cache));
// [
//   '/path/to/app.js',
//   '/path/to/module.js',
//   ...
// ]

// 查看特定模块
console.log(require.cache[require.resolve('./module')]);
```

---

### 3. 模块加载流程

```
require('./module')
   ↓
1. 解析路径（require.resolve）
   ↓
2. 检查 require.cache
   ├── 有缓存 → 返回 module.exports
   ↓
   └── 无缓存
      ↓
3. 创建新 Module 对象
   ↓
4. 加载文件内容
   ↓
5. 编译执行（包装函数）
   ↓
6. 缓存到 require.cache
   ↓
7. 返回 module.exports
```

## 💻 实战案例

### 案例 1：基础缓存行为

```javascript
// counter.js
let count = 0;

module.exports = {
  increment() {
    return ++count;
  },
  getCount() {
    return count;
  }
};

// app.js
const counter1 = require('./counter');
const counter2 = require('./counter');

console.log(counter1.increment());  // 1
console.log(counter2.increment());  // 2（共享状态）

console.log(counter1.getCount());  // 2
console.log(counter2.getCount());  // 2

console.log(counter1 === counter2);  // true
```

**关键点：**
- 两次 require 返回同一个对象
- 状态在多个引用之间共享

### 案例 2：清除缓存

```javascript
// config.js
module.exports = {
  port: 3000,
  host: 'localhost'
};

// app.js
const config = require('./config');
console.log(config.port);  // 3000

// 修改配置文件后，清除缓存重新加载
delete require.cache[require.resolve('./config')];

const newConfig = require('./config');
console.log(newConfig.port);  // 读取新值
```

### 案例 3：递归清除缓存

```javascript
function clearRequireCache(modulePath) {
  const resolvedPath = require.resolve(modulePath);
  const module = require.cache[resolvedPath];
  
  if (!module) return;
  
  // 递归清除子模块缓存
  module.children.forEach(child => {
    clearRequireCache(child.id);
  });
  
  // 清除当前模块缓存
  delete require.cache[resolvedPath];
}

// 使用
clearRequireCache('./module');
const freshModule = require('./module');  // 重新加载
```

### 案例 4：模块热更新（Hot Reload）

```javascript
const fs = require('fs');
const path = require('path');

// 监听文件变化
function watchModule(modulePath) {
  const resolvedPath = require.resolve(modulePath);
  
  fs.watch(resolvedPath, (eventType) => {
    if (eventType === 'change') {
      console.log(`Module ${modulePath} changed, reloading...`);
      
      // 清除缓存
      delete require.cache[resolvedPath];
      
      // 重新加载
      try {
        const newModule = require(modulePath);
        console.log('Module reloaded:', newModule);
      } catch (error) {
        console.error('Failed to reload module:', error);
      }
    }
  });
}

// 使用
watchModule('./config');
```

### 案例 5：条件缓存

```javascript
// 开发环境：禁用缓存
if (process.env.NODE_ENV === 'development') {
  const Module = require('module');
  const originalRequire = Module.prototype.require;
  
  Module.prototype.require = function(id) {
    // 清除特定模块的缓存
    if (id.startsWith('./config')) {
      delete require.cache[require.resolve(id)];
    }
    
    return originalRequire.call(this, id);
  };
}

// 现在每次 require('./config') 都会重新加载
```

## 🎓 面试标准答题

### 问题 1："require 的缓存机制是什么？"

**标准答案：**
> Node.js 的 require 缓存机制基于 **require.cache** 对象实现。
> 
> **工作原理：**
> 1. **首次加载**：
>    - 解析模块路径为绝对路径
>    - 执行模块代码
>    - 将 `module.exports` 缓存到 `require.cache[绝对路径]`
> 
> 2. **后续加载**：
>    - 检查 require.cache 中是否存在
>    - 如果存在，直接返回缓存的 `module.exports`
>    - 不再执行模块代码
> 
> **优点：**
> - 提高性能（避免重复执行）
> - 保持单例模式（同一个对象实例）
> 
> **缓存键：**
> - 使用模块的**绝对路径**作为键
> - Windows：`C:\\path\\to\\module.js`
> - Unix：`/path/to/module.js`
> 
> **示例：**
> ```javascript
> const mod1 = require('./module');  // 执行代码，缓存结果
> const mod2 = require('./module');  // 直接返回缓存
> console.log(mod1 === mod2);  // true
> ```

**加分项：**
> 在实际项目中，我利用这个机制实现了配置热更新。通过监听配置文件变化，清除 require.cache，重新加载配置，无需重启服务器。

---

### 问题 2："如何清除 require 缓存？"

**标准答案：**
> 清除 require 缓存有**三种方法**：
> 
> **方法 1：直接删除（最常用）**
> ```javascript
> delete require.cache[require.resolve('./module')];
> const freshModule = require('./module');  // 重新加载
> ```
> 
> **方法 2：递归清除（清除依赖）**
> ```javascript
> function clearRequireCache(modulePath) {
>   const resolvedPath = require.resolve(modulePath);
>   const module = require.cache[resolvedPath];
>   
>   if (module) {
>     // 递归清除子模块
>     module.children.forEach(child => {
>       clearRequireCache(child.id);
>     });
>     
>     delete require.cache[resolvedPath];
>   }
> }
> ```
> 
> **方法 3：清除所有缓存**
> ```javascript
> Object.keys(require.cache).forEach(key => {
>   delete require.cache[key];
> });
> ```
> 
> **注意事项：**
> 1. 核心模块（如 `fs`、`path`）无法清除缓存
> 2. 清除缓存后，新旧模块可能存在不兼容
> 3. 开发环境可用，生产环境需谨慎

## ⚠️ 常见误区

### 误区 1：以为每次 require 都会执行

```javascript
// module.js
console.log('Module code executed');
module.exports = { value: 42 };

// app.js
require('./module');  // 输出：Module code executed
require('./module');  // 无输出（缓存）
require('./module');  // 无输出（缓存）

// ❌ 错误理解：每次 require 都执行
// ✅ 正确理解：只有第一次执行，后续返回缓存
```

---

### 误区 2：修改 exports 影响其他引用

```javascript
// module.js
module.exports = { value: 42 };

// app.js
const mod1 = require('./module');
mod1.value = 100;  // 修改对象属性

const mod2 = require('./module');
console.log(mod2.value);  // 100（共享同一个对象）

// ❌ 错误理解：mod2 不受影响
// ✅ 正确理解：mod1 和 mod2 是同一个对象
```

---

### 误区 3：清除缓存会影响已有引用

```javascript
const mod1 = require('./module');

delete require.cache[require.resolve('./module')];

const mod2 = require('./module');

console.log(mod1 === mod2);  // false

// ❌ 错误理解：mod1 会变成 mod2
// ✅ 正确理解：mod1 仍然是旧对象，mod2 是新对象
```

## 💡 高级应用

### 应用 1：配置热更新

```javascript
const fs = require('fs');

class ConfigManager {
  constructor(configPath) {
    this.configPath = configPath;
    this.config = null;
    this.loadConfig();
    this.watchConfig();
  }
  
  loadConfig() {
    delete require.cache[require.resolve(this.configPath)];
    this.config = require(this.configPath);
    console.log('Config loaded:', this.config);
  }
  
  watchConfig() {
    fs.watch(this.configPath, (eventType) => {
      if (eventType === 'change') {
        console.log('Config changed, reloading...');
        this.loadConfig();
        this.emit('configChanged', this.config);
      }
    });
  }
  
  get(key) {
    return this.config[key];
  }
}

// 使用
const configManager = new ConfigManager('./config.js');
console.log(configManager.get('port'));  // 3000

// 修改 config.js 后自动重新加载
```

### 应用 2：插件热加载

```javascript
class PluginManager {
  constructor() {
    this.plugins = new Map();
  }
  
  loadPlugin(pluginPath) {
    const resolvedPath = require.resolve(pluginPath);
    
    // 清除旧缓存
    delete require.cache[resolvedPath];
    
    // 加载插件
    const plugin = require(pluginPath);
    
    this.plugins.set(pluginPath, plugin);
    
    // 激活插件
    if (plugin.activate) {
      plugin.activate();
    }
    
    return plugin;
  }
  
  reloadPlugin(pluginPath) {
    const plugin = this.plugins.get(pluginPath);
    
    // 停用旧插件
    if (plugin && plugin.deactivate) {
      plugin.deactivate();
    }
    
    // 重新加载
    return this.loadPlugin(pluginPath);
  }
}

// 使用
const pm = new PluginManager();
pm.loadPlugin('./plugins/logger');

// 修改插件后重新加载
pm.reloadPlugin('./plugins/logger');
```

## 🔍 缓存与性能

### 性能测试

```javascript
const { performance } = require('perf_hooks');

// 测试：首次加载 vs 缓存加载
function testRequirePerformance() {
  const modulePath = './large-module';
  
  // 清除缓存
  delete require.cache[require.resolve(modulePath)];
  
  // 首次加载
  const start1 = performance.now();
  require(modulePath);
  const time1 = performance.now() - start1;
  
  // 缓存加载
  const start2 = performance.now();
  require(modulePath);
  const time2 = performance.now() - start2;
  
  console.log(`首次加载: ${time1.toFixed(2)}ms`);
  console.log(`缓存加载: ${time2.toFixed(2)}ms`);
  console.log(`性能提升: ${(time1 / time2).toFixed(2)}倍`);
}

testRequirePerformance();

// 输出示例：
// 首次加载: 15.23ms
// 缓存加载: 0.03ms
// 性能提升: 507.67倍
```

## 📝 总结

本节学习了：

✅ **缓存机制**：首次加载执行代码并缓存，后续返回缓存  
✅ **require.cache**：键是绝对路径，值是 Module 对象  
✅ **清除缓存**：delete require.cache[path]、递归清除  
✅ **模块热更新**：监听文件变化、清除缓存、重新加载  
✅ **性能优势**：缓存加载比首次加载快 100-1000 倍  
✅ **注意事项**：清除缓存不影响已有引用、核心模块不可清除
