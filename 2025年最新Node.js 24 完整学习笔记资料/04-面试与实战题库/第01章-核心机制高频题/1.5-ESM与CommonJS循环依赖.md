## 📚 核心知识点

### 1. CommonJS 循环依赖

**原理：**
- 模块第一次加载时，`module.exports` 为**空对象** `{}`
- 立即缓存到 `require.cache`
- 执行模块代码，逐步填充 `exports`
- 循环依赖时返回**未完成的 exports**

```javascript
// a.js
console.log('a starting');
exports.done = false;
const b = require('./b');
console.log('in a, b.done =', b.done);
exports.done = true;
console.log('a done');

// b.js
console.log('b starting');
exports.done = false;
const a = require('./a');
console.log('in b, a.done =', a.done);
exports.done = true;
console.log('b done');

// main.js
const a = require('./a');
const b = require('./b');
console.log('in main, a.done =', a.done);
console.log('in main, b.done =', b.done);
```

**输出：**
```
a starting
b starting
in b, a.done = false  ← a 未完成
b done
in a, b.done = true
a done
in main, a.done = true
in main, b.done = true
```

**执行流程：**
```
1. main.js require('./a')
   ↓
2. a.js 开始执行，exports.done = false
   ↓
3. a.js require('./b')
   ↓
4. b.js 开始执行，exports.done = false
   ↓
5. b.js require('./a')  ← 循环依赖
   ↓
6. 返回 a.js 的未完成 exports（a.done = false）
   ↓
7. b.js 继续执行，exports.done = true
   ↓
8. 返回到 a.js，a.js 继续执行，exports.done = true
   ↓
9. 返回到 main.js
```

---

### 2. ESM 循环依赖

**原理：**
- ESM 使用**静态分析**，在执行前建立模块图
- 支持**动态绑定**（Live Binding）
- 循环依赖时可以访问变量，但受 **TDZ（暂时性死区）**限制

```javascript
// a.mjs
console.log('a starting');
import { bDone } from './b.mjs';
console.log('in a, bDone =', bDone);
export let aDone = false;
aDone = true;
console.log('a done');

// b.mjs
console.log('b starting');
import { aDone } from './a.mjs';
console.log('in b, aDone =', aDone);  // ❌ ReferenceError: Cannot access 'aDone' before initialization
export let bDone = false;
bDone = true;
console.log('b done');
```

**输出：**
```
b starting
ReferenceError: Cannot access 'aDone' before initialization
```

**原因：**
- ESM 先构建模块图（分析所有 import/export）
- a.mjs 导入 b.mjs，b.mjs 导入 a.mjs
- b.mjs 执行时，aDone 尚未初始化（TDZ）

---

### 3. ESM 成功的循环依赖

**使用函数延迟访问：**
```javascript
// a.mjs
console.log('a starting');
import { getB } from './b.mjs';
export function getA() {
  return 'A';
}
console.log('in a, getB() =', getB());
console.log('a done');

// b.mjs
console.log('b starting');
import { getA } from './a.mjs';
export function getB() {
  return 'B';
}
console.log('in b, getA() =', getA());
console.log('b done');

// main.mjs
import './a.mjs';
```

**输出：**
```
b starting
a starting
in a, getB() = B  ← 成功调用
a done
in b, getA() = A  ← 成功调用
b done
```

**为什么成功？**
- 函数声明会被提升
- 调用时（非导入时）函数已定义

## 💻 实战案例

### 案例 1：CommonJS 经典陷阱

```javascript
// user.js
const { getOrders } = require('./order');

class User {
  constructor(id) {
    this.id = id;
  }
  
  getOrders() {
    return getOrders(this.id);
  }
}

module.exports = { User };

// order.js
const { User } = require('./user');

function getOrders(userId) {
  // 使用 User（循环依赖）
  const user = new User(userId);  // ❌ TypeError: User is not a constructor
  return [];
}

module.exports = { getOrders };

// main.js
const { User } = require('./user');
const user = new User(1);
user.getOrders();  // 报错
```

**问题分析：**
1. `user.js` require `order.js`
2. `order.js` require `user.js`（循环依赖）
3. `order.js` 拿到的是未完成的 `exports`（空对象）
4. `User` 为 `undefined`

**解决方案 1：延迟 require**
```javascript
// order.js
function getOrders(userId) {
  const { User } = require('./user');  // ✅ 延迟 require
  const user = new User(userId);
  return [];
}

module.exports = { getOrders };
```

**解决方案 2：重构代码**
```javascript
// user.js
class User {
  constructor(id) {
    this.id = id;
  }
}

module.exports = { User };

// order.js
const { User } = require('./user');

function getOrders(userId) {
  const user = new User(userId);
  return [];
}

module.exports = { getOrders };

// user-service.js
const { User } = require('./user');
const { getOrders } = require('./order');

User.prototype.getOrders = function() {
  return getOrders(this.id);
};
```

### 案例 2：ESM TDZ 问题

```javascript
// config.mjs
import { db } from './database.mjs';

export const config = {
  dbHost: 'localhost',
  getConnection() {
    return db.connect();  // ❌ ReferenceError
  }
};

// database.mjs
import { config } from './config.mjs';

export const db = {
  connect() {
    return `Connected to ${config.dbHost}`;
  }
};

// main.mjs
import { config } from './config.mjs';
config.getConnection();  // 报错
```

**解决方案 1：函数包装**
```javascript
// config.mjs
import { getDb } from './database.mjs';

export const config = {
  dbHost: 'localhost',
  getConnection() {
    return getDb().connect();  // ✅ 函数调用时已初始化
  }
};

// database.mjs
import { config } from './config.mjs';

export function getDb() {
  return {
    connect() {
      return `Connected to ${config.dbHost}`;
    }
  };
}
```

**解决方案 2：动态 import**
```javascript
// config.mjs
export const config = {
  dbHost: 'localhost',
  async getConnection() {
    const { db } = await import('./database.mjs');  // ✅ 动态导入
    return db.connect();
  }
};
```

### 案例 3：依赖注入解决循环依赖

```javascript
// Before（循环依赖）
// user.service.js
const orderService = require('./order.service');

class UserService {
  getUserOrders(userId) {
    return orderService.getOrdersByUser(userId);
  }
}

// order.service.js
const userService = require('./user.service');

class OrderService {
  getOrdersByUser(userId) {
    const user = userService.getUser(userId);  // 循环依赖
    return [];
  }
}

// After（依赖注入）
// user.service.js
class UserService {
  constructor(orderService) {
    this.orderService = orderService;
  }
  
  getUserOrders(userId) {
    return this.orderService.getOrdersByUser(userId);
  }
}

// order.service.js
class OrderService {
  constructor(userService) {
    this.userService = userService;
  }
  
  getOrdersByUser(userId) {
    const user = this.userService.getUser(userId);
    return [];
  }
}

// container.js
const userService = new UserService(null);
const orderService = new OrderService(userService);
userService.orderService = orderService;  // 手动注入
```

## 🎓 面试标准答题

### 问题 1："ESM 和 CommonJS 的循环依赖处理有什么区别？"

**标准答案：**
> **CommonJS 循环依赖：**
> 1. **返回未完成的 exports**
>    - 模块首次加载时，立即缓存空对象 `{}`
>    - 循环依赖时返回未完成的 `exports`
>    - 后续代码会填充 `exports`
> 
> 2. **特点：**
>    - 运行时加载（动态）
>    - 可以拿到部分导出
>    - 需要注意访问顺序
> 
> **ESM 循环依赖：**
> 1. **动态绑定 + TDZ**
>    - 静态分析，提前建立模块图
>    - 支持动态绑定（Live Binding）
>    - 受暂时性死区（TDZ）限制
> 
> 2. **特点：**
>    - 编译时确定依赖关系
>    - 访问未初始化变量抛出 ReferenceError
>    - 函数声明可以被提升
> 
> **关键区别：**
> 
> | 特性 | CommonJS | ESM |
> |------|----------|-----|
> | **加载时机** | 运行时 | 编译时 |
> | **循环依赖** | 返回未完成 exports | 抛出 TDZ 错误 |
> | **解决方案** | 延迟 require | 函数包装/动态 import |
> | **灵活性** | 高（动态加载） | 低（静态分析） |
> 
> **示例对比：**
> ```javascript
> // CommonJS
> const a = require('./a');  // 拿到未完成的 exports
> console.log(a.value);  // undefined 或部分值
> 
> // ESM
> import { value } from './a.mjs';
> console.log(value);  // ReferenceError（TDZ）
> ```

**加分项：**
> 在实际项目中，我们通过**依赖注入**和**事件驱动**模式避免循环依赖，将模块依赖关系扁平化，提高代码可维护性。

---

### 问题 2："如何避免循环依赖问题？"

**标准答案：**
> **避免循环依赖的最佳实践：**
> 
> **1. 重构代码结构**
> - 提取共同依赖到第三个模块
> - 将双向依赖改为单向依赖
> - 使用中介者模式（Mediator）
> 
> **2. 延迟加载**
> - CommonJS：函数内 require
> - ESM：动态 import()
> 
> **3. 依赖注入**
> - 构造函数注入
> - 属性注入
> - 使用 DI 容器（InversifyJS）
> 
> **4. 事件驱动**
> - 使用 EventEmitter 解耦
> - 发布订阅模式
> 
> **5. 接口隔离**
> - 定义抽象接口
> - 依赖接口而非具体实现
> 
> **示例：**
> ```javascript
> // ❌ 循环依赖
> // a.js → b.js → a.js
> 
> // ✅ 提取共同依赖
> // a.js → common.js
> // b.js → common.js
> 
> // ✅ 依赖注入
> class A {
>   constructor(b) {
>     this.b = b;
>   }
> }
> 
> class B {
>   constructor(a) {
>     this.a = a;
>   }
> }
> 
> // 在容器中组装
> const a = new A(null);
> const b = new B(a);
> a.b = b;
> ```

## ⚠️ 常见误区

### 误区 1：以为 ESM 不支持循环依赖

```javascript
// ❌ 错误理解：ESM 完全不支持循环依赖
// ✅ 正确理解：ESM 支持，但受 TDZ 限制

// 成功的 ESM 循环依赖
// a.mjs
import { getB } from './b.mjs';
export function getA() {
  return 'A';
}
console.log(getB());  // ✅ 成功

// b.mjs
import { getA } from './a.mjs';
export function getB() {
  return 'B';
}
console.log(getA());  // ✅ 成功
```

---

### 误区 2：以为 CommonJS 循环依赖一定有问题

```javascript
// ❌ 错误理解：CommonJS 循环依赖总是出错
// ✅ 正确理解：只要访问已导出的属性就没问题

// a.js
exports.name = 'A';  // ← 立即导出
const b = require('./b');
console.log(b.name);  // ✅ 成功

// b.js
exports.name = 'B';  // ← 立即导出
const a = require('./a');
console.log(a.name);  // ✅ 成功
```
