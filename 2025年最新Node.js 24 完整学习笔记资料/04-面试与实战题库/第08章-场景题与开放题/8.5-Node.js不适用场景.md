## 💻 不适用场景

### 1. CPU 密集型任务

#### 问题示例

```javascript
// ❌ 不适合：大量计算阻塞事件循环
app.get('/api/calculate', (req, res) => {
  const n = req.query.n;
  
  // 计算斐波那契数列（递归）
  function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
  }
  
  const result = fibonacci(n);  // 阻塞事件循环！
  
  res.json({ result });
});

// 问题：
// - fibonacci(40) 耗时 ~2秒
// - 阻塞所有其他请求
// - CPU 100%，其他请求超时
```

#### 性能对比

```javascript
// 测试：计算 fibonacci(40)
const start = Date.now();

// Node.js（单线程）
fibonacci(40);  // ~2000ms

// Go（协程）：~200ms
// Java（多线程）：~100ms
// Rust（零成本抽象）：~50ms
```

#### 解决方案

```javascript
// 方案1：Worker Threads
const { Worker } = require('worker_threads');

app.get('/api/calculate', async (req, res) => {
  const worker = new Worker('./worker.js', {
    workerData: { n: req.query.n }
  });
  
  worker.on('message', (result) => {
    res.json({ result });
  });
  
  worker.on('error', (error) => {
    res.status(500).json({ error: error.message });
  });
});

// 方案2：调用其他语言服务
app.get('/api/calculate', async (req, res) => {
  const result = await fetch(`http://go-service:8080/calculate?n=${req.query.n}`);
  res.json(result);
});
```

### 2. 大数据处理

#### 问题示例

```javascript
// ❌ 不适合：处理大文件
app.get('/api/export-csv', async (req, res) => {
  const users = await db.users.findAll();  // 100万条数据
  
  // 构建CSV（占用大量内存）
  let csv = 'id,name,email\n';
  
  for (const user of users) {
    csv += `${user.id},${user.name},${user.email}\n`;  // 字符串拼接
  }
  
  res.send(csv);  // OOM!
});

// 问题：
// - 100万条数据 ~500MB 内存
// - 字符串拼接性能差
// - 超过V8堆限制（~2GB）
```

#### 推荐方案

```javascript
// ✅ 使用流式处理
const { pipeline } = require('stream');
const { Transform } = require('stream');

app.get('/api/export-csv', (req, res) => {
  res.setHeader('Content-Type', 'text/csv');
  res.setHeader('Content-Disposition', 'attachment; filename=users.csv');
  
  // 数据库流
  const dbStream = db.users.findAllStream();
  
  // 转换流
  const csvTransform = new Transform({
    objectMode: true,
    transform(user, encoding, callback) {
      const row = `${user.id},${user.name},${user.email}\n`;
      callback(null, row);
    }
  });
  
  // 流式传输
  pipeline(dbStream, csvTransform, res, (err) => {
    if (err) {
      console.error('Export failed:', err);
    }
  });
});

// 或使用专业工具：
// - Spark（大数据处理）
// - Hadoop（分布式计算）
// - Python + Pandas（数据分析）
```

### 3. 实时计算

#### 问题示例

```javascript
// ❌ 不适合：实时数据分析
app.get('/api/realtime-stats', async (req, res) => {
  // 查询100万条日志
  const logs = await db.logs.findAll({
    where: {
      created_at: { $gte: new Date(Date.now() - 3600000) }
    }
  });
  
  // 实时聚合（耗时长）
  const stats = {};
  
  for (const log of logs) {
    if (!stats[log.level]) {
      stats[log.level] = 0;
    }
    stats[log.level]++;
  }
  
  res.json(stats);  // 响应慢！
});

// 问题：
// - 100万条数据查询慢
// - 内存计算效率低
// - 实时性差
```

#### 推荐方案

```javascript
// ✅ 使用流式计算引擎
// - Apache Flink（流处理）
// - Kafka Streams（实时流）
// - Redis（预聚合）

// Node.js 只负责查询结果
app.get('/api/realtime-stats', async (req, res) => {
  const stats = await redis.hgetall('logs:stats:hourly');
  res.json(stats);
});

// Flink 实时计算
// Java/Scala:
// stream
//   .keyBy(log -> log.level)
//   .timeWindow(Time.hours(1))
//   .reduce((a, b) -> a + b)
//   .addSink(new RedisSink());
```

### 4. 传统企业应用

#### 不适合原因

| 原因 | 说明 |
|------|------|
| **生态不成熟** | 缺少企业级框架（如Java的Spring） |
| **强类型需求** | TypeScript弥补，但不如Java/C#原生 |
| **长期维护** | 技术栈变化快，维护成本高 |
| **团队技能** | 传统企业团队更熟悉Java/.NET |

#### 推荐技术

```java
// ✅ 传统企业应用：Spring Boot
@RestController
public class UserController {
    @Autowired
    private UserService userService;
    
    @GetMapping("/api/users/{id}")
    public User getUser(@PathVariable Long id) {
        return userService.findById(id);
    }
    
    @Transactional
    @PostMapping("/api/users")
    public User createUser(@RequestBody User user) {
        return userService.save(user);
    }
}

// 优势：
// - 强类型、编译检查
// - 完善的ORM（Hibernate）
// - 成熟的事务管理
// - 丰富的企业级组件
```

### 5. 需要强一致性的系统

#### 问题示例

```javascript
// ❌ 不适合：金融交易系统
app.post('/api/transfer', async (req, res) => {
  const { fromAccount, toAccount, amount } = req.body;
  
  try {
    // 扣款
    await db.accounts.update(
      { balance: sequelize.literal(`balance - ${amount}`) },
      { where: { id: fromAccount } }
    );
    
    // 加款
    await db.accounts.update(
      { balance: sequelize.literal(`balance + ${amount}`) },
      { where: { id: toAccount } }
    );
    
    res.json({ success: true });
  } catch (error) {
    // 事务回滚困难！
    res.status(500).json({ error: error.message });
  }
});

// 问题：
// - 分布式事务难以保证
// - ACID特性支持弱
// - 数据一致性风险高
```

#### 推荐方案

```java
// ✅ 金融系统：Java + Spring + MySQL
@Service
public class TransferService {
    @Transactional(isolation = Isolation.SERIALIZABLE)
    public void transfer(Long fromAccount, Long toAccount, BigDecimal amount) {
        // 锁定账户（行锁）
        Account from = accountRepository.findByIdForUpdate(fromAccount);
        Account to = accountRepository.findByIdForUpdate(toAccount);
        
        // 检查余额
        if (from.getBalance().compareTo(amount) < 0) {
            throw new InsufficientBalanceException();
        }
        
        // 扣款
        from.setBalance(from.getBalance().subtract(amount));
        
        // 加款
        to.setBalance(to.getBalance().add(amount));
        
        // 记录流水
        transactionRepository.save(new Transaction(from, to, amount));
    }
}

// 优势：
// - 强事务支持（ACID）
// - 行级锁（FOR UPDATE）
// - BigDecimal（精确计算）
// - 成熟的事务管理
```

## 💻 Node.js 适用场景

### ✅ 推荐场景

| 场景 | 原因 | 典型应用 |
|------|------|---------|
| **I/O密集型** | 异步非阻塞 | API网关、BFF |
| **实时通信** | WebSocket、SSE | 聊天室、推送 |
| **微服务** | 轻量、快速启动 | 中间层服务 |
| **前端工具链** | JavaScript生态 | Webpack、Vite |
| **Serverless** | 冷启动快 | Lambda、云函数 |

---

### 示例：适合的应用

```javascript
// ✅ API网关（I/O密集）
app.get('/api/user/:id', async (req, res) => {
  const [user, posts, comments] = await Promise.all([
    fetch(`http://user-service/users/${req.params.id}`),
    fetch(`http://post-service/posts?userId=${req.params.id}`),
    fetch(`http://comment-service/comments?userId=${req.params.id}`)
  ]);
  
  res.json({ user, posts, comments });
});

// ✅ WebSocket（实时通信）
io.on('connection', (socket) => {
  socket.on('chat', (msg) => {
    io.emit('chat', msg);  // 广播消息
  });
});

// ✅ SSE（服务器推送）
app.get('/api/events', (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  
  const intervalId = setInterval(() => {
    res.write(`data: ${JSON.stringify({ time: Date.now() })}\n\n`);
  }, 1000);
  
  req.on('close', () => {
    clearInterval(intervalId);
  });
});
```

## 🎯 面试标准答题

### 问题："Node.js 不适合什么场景？"

**标准答案（10-15分钟）：**

**1. 核心观点（2分钟）**
> "Node.js 是单线程、事件驱动的，适合I/O密集型任务，  
> 但不适合CPU密集、大数据处理、强一致性等场景。"

**2. 不适用场景（6分钟）**

**CPU密集型：**
> - 例如：图像处理、视频转码、复杂计算  
> - 原因：单线程，计算阻塞事件循环  
> - 推荐：Go、Rust、C++

**大数据处理：**
> - 例如：TB级数据分析、实时计算  
> - 原因：内存限制（~2GB），计算效率低  
> - 推荐：Spark、Flink、Python

**强一致性：**
> - 例如：金融交易、支付系统  
> - 原因：分布式事务弱，ACID支持差  
> - 推荐：Java + Spring、Go

**3. 适用场景（3分钟）**
> "Node.js 擅长：  
> 1. API网关：聚合多个服务  
> 2. BFF：前端数据聚合  
> 3. WebSocket：实时通信  
> 4. Serverless：冷启动快  
> 5. 工具链：Webpack、ESLint"

**4. 技术选型建议（2分钟）**
> "选择技术要考虑：  
> 1. 业务特点：I/O密集 vs CPU密集  
> 2. 团队技能：是否熟悉Node.js  
> 3. 生态成熟度：是否有现成方案  
> 4. 长期维护：是否容易招人  
>   
> 不要盲目追新，合适最重要。"

## 📝 总结

### 核心要点
1. **不适用**：CPU密集、大数据、强一致性
2. **适用**：I/O密集、实时通信、微服务
3. **选型**：考虑业务、团队、生态

### 判断标准
- ✅ I/O密集型 → Node.js
- ❌ CPU密集型 → Go/Rust
- ❌ 大数据处理 → Spark/Flink
- ❌ 强一致性 → Java/Go
- ✅ 实时通信 → Node.js
- ✅ Serverless → Node.js
