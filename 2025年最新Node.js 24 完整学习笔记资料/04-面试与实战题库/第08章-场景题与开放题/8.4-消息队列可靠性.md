## 💻 生产者可靠性

### 1. 事务消息

```javascript
const amqp = require('amqplib');

class Producer {
  constructor() {
    this.connection = null;
    this.channel = null;
  }
  
  async connect() {
    this.connection = await amqp.connect('amqp://localhost');
    this.channel = await this.connection.createConfirmChannel();
    
    // 设置发送确认
    this.channel.on('return', (msg) => {
      console.error('Message returned:', msg);
    });
  }
  
  async sendWithConfirm(queue, message) {
    try {
      // 确保队列存在
      await this.channel.assertQueue(queue, { durable: true });
      
      // 发送消息（等待确认）
      await this.channel.sendToQueue(
        queue,
        Buffer.from(JSON.stringify(message)),
        {
          persistent: true,  // 持久化
          mandatory: true    // 路由失败返回
        }
      );
      
      // 等待Broker确认
      await this.channel.waitForConfirms();
      
      console.log('Message confirmed by broker');
      return true;
    } catch (error) {
      console.error('Send failed:', error);
      return false;
    }
  }
}

// 使用
const producer = new Producer();
await producer.connect();

const success = await producer.sendWithConfirm('orders', {
  orderId: '123',
  amount: 100
});
```

### 2. 重试机制

```javascript
class ReliableProducer {
  constructor(maxRetries = 3) {
    this.maxRetries = maxRetries;
  }
  
  async sendWithRetry(queue, message) {
    let lastError;
    
    for (let i = 0; i < this.maxRetries; i++) {
      try {
        await this.sendWithConfirm(queue, message);
        
        console.log(`Message sent on attempt ${i + 1}`);
        return true;
      } catch (error) {
        lastError = error;
        
        console.warn(`Attempt ${i + 1} failed:`, error.message);
        
        // 指数退避
        const delay = Math.pow(2, i) * 1000;
        await this.sleep(delay);
      }
    }
    
    // 所有重试失败，保存到失败队列
    await this.saveToDLQ(queue, message, lastError);
    
    throw new Error(`Failed after ${this.maxRetries} retries`);
  }
  
  async saveToDLQ(queue, message, error) {
    // 保存到死信队列
    await this.sendWithConfirm('dlq', {
      originalQueue: queue,
      message,
      error: error.message,
      timestamp: Date.now()
    });
  }
  
  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### 3. 本地消息表

```javascript
// 结合数据库事务
class TransactionalProducer {
  async sendWithTransaction(db, message) {
    const connection = await db.getConnection();
    
    try {
      await connection.beginTransaction();
      
      // 1. 执行业务操作
      await connection.query(
        'UPDATE orders SET status = ? WHERE id = ?',
        ['paid', message.orderId]
      );
      
      // 2. 保存消息到本地表
      await connection.query(
        'INSERT INTO outbox_messages (id, queue, message, status) VALUES (?, ?, ?, ?)',
        [message.id, 'orders', JSON.stringify(message), 'pending']
      );
      
      await connection.commit();
      
      // 3. 异步发送消息
      setImmediate(() => {
        this.sendMessage(message);
      });
    } catch (error) {
      await connection.rollback();
      throw error;
    } finally {
      connection.release();
    }
  }
  
  async sendMessage(message) {
    try {
      await this.sendWithConfirm('orders', message);
      
      // 标记为已发送
      await db.query(
        'UPDATE outbox_messages SET status = ? WHERE id = ?',
        ['sent', message.id]
      );
    } catch (error) {
      console.error('Send failed:', error);
    }
  }
  
  // 定时任务：重发未成功的消息
  async retryPendingMessages() {
    const [messages] = await db.query(`
      SELECT * FROM outbox_messages
      WHERE status = 'pending'
      AND created_at < DATE_SUB(NOW(), INTERVAL 1 MINUTE)
      LIMIT 100
    `);
    
    for (const msg of messages) {
      await this.sendMessage(JSON.parse(msg.message));
    }
  }
}

// 定时任务
setInterval(() => {
  producer.retryPendingMessages();
}, 60000);
```

### 2. 幂等处理

```javascript
const Redis = require('ioredis');
const redis = new Redis();

class IdempotentConsumer {
  async processMessage(message) {
    const messageId = message.id;
    
    // 1. 检查是否已处理
    const processed = await redis.get(`processed:${messageId}`);
    
    if (processed) {
      console.log('Message already processed:', messageId);
      return;
    }
    
    try {
      // 2. 处理消息
      await this.handleMessage(message);
      
      // 3. 标记为已处理（24小时过期）
      await redis.setex(`processed:${messageId}`, 86400, '1');
      
      console.log('Message processed:', messageId);
    } catch (error) {
      console.error('Process failed:', error);
      throw error;
    }
  }
  
  async handleMessage(message) {
    // 业务逻辑（必须支持重试）
    const db = await getConnection();
    
    try {
      await db.beginTransaction();
      
      // 更新订单状态
      await db.query(
        'UPDATE orders SET status = ? WHERE id = ?',
        ['processing', message.orderId]
      );
      
      // 减库存
      await db.query(
        'UPDATE inventory SET stock = stock - ? WHERE product_id = ?',
        [message.quantity, message.productId]
      );
      
      await db.commit();
    } catch (error) {
      await db.rollback();
      throw error;
    }
  }
}
```

### 3. 重试队列

```javascript
class RetryConsumer {
  constructor() {
    this.maxRetries = 3;
  }
  
  async consume() {
    await channel.consume('orders', async (msg) => {
      const message = JSON.parse(msg.content.toString());
      const retryCount = msg.properties.headers?.['x-retry-count'] || 0;
      
      try {
        await this.processMessage(message);
        
        // 成功：ACK
        channel.ack(msg);
      } catch (error) {
        if (retryCount < this.maxRetries) {
          // 重试：发送到重试队列
          await this.sendToRetryQueue(message, retryCount + 1);
          
          // ACK原消息
          channel.ack(msg);
        } else {
          // 超过最大重试：发送到死信队列
          await this.sendToDLQ(message, error);
          
          // ACK原消息
          channel.ack(msg);
        }
      }
    });
  }
  
  async sendToRetryQueue(message, retryCount) {
    const delay = Math.pow(2, retryCount) * 1000;  // 指数退避
    
    await channel.sendToQueue(
      'orders-retry',
      Buffer.from(JSON.stringify(message)),
      {
        headers: {
          'x-retry-count': retryCount,
          'x-original-queue': 'orders'
        },
        expiration: delay.toString()  // 延迟
      }
    );
  }
  
  async sendToDLQ(message, error) {
    await channel.sendToQueue(
      'orders-dlq',
      Buffer.from(JSON.stringify({
        message,
        error: error.message,
        timestamp: Date.now()
      })),
      { persistent: true }
    );
  }
}
```

## 🎯 面试标准答题

### 问题："如何保证消息队列可靠性？"

**标准答案（20-30分钟）：**

**1. 整体思路（5分钟）**
> "消息队列可靠性需要从三个阶段保证：  
> 1. 生产者：事务消息、重试  
> 2. Broker：持久化、集群  
> 3. 消费者：ACK、幂等"

**2. 生产者可靠性（8分钟）**
> "确保消息不丢失：  
> 1. 事务消息：等待Broker确认  
> 2. 重试机制：失败重发3次  
> 3. 本地消息表：结合DB事务  
>   
> 代码示例：  
> - 使用confirmChannel  
> - waitForConfirms等待确认  
> - 失败保存到DLQ"

**3. Broker可靠性（5分钟）**
> "防止数据丢失：  
> 1. 持久化：队列和消息都持久化  
> 2. 集群：3个Broker节点  
> 3. 副本：acks=-1（所有副本确认）"

**4. 消费者可靠性（8分钟）**
> "确保消费成功：  
> 1. 手动ACK：处理成功才ACK  
> 2. 幂等处理：Redis去重  
> 3. 重试队列：失败重试3次  
> 4. 死信队列：最终失败保存  
>   
> 案例：  
> - 订单系统，消息重复消费导致重复扣款  
> - 解决：Redis记录已处理消息ID  
> - 效果：重复率从5%降到0%"

**5. 监控告警（3分钟）**
> "监控关键指标：  
> 1. 队列深度：>10000告警  
> 2. 消费延迟：>5秒告警  
> 3. 失败率：>1%告警  
> 4. DLQ大小：>100告警"

## 📝 总结

### 核心要点
1. **生产者**：事务消息 + 重试
2. **Broker**：持久化 + 集群
3. **消费者**：手动ACK + 幂等
4. **监控**：队列深度 + 延迟

### 可靠性方案
- ✅ 事务消息（Broker确认）
- ✅ 本地消息表（DB事务）
- ✅ 手动ACK（处理成功才确认）
- ✅ 幂等处理（Redis去重）
- ✅ 重试队列（失败重试）
- ✅ 死信队列（最终失败）
