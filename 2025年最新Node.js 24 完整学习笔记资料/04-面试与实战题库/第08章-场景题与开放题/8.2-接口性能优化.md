| 层级 | 可能原因 | 占比 |
|------|---------|------|
| **数据库** | 慢查询、无索引、N+1查询 | 60% |

| **缓存** | 缓存未命中、缓存雪崩 | 5% |
| **资源** | CPU/内存/磁盘瓶颈 | 5% |

### 1. 数据库慢查询

```javascript
// 慢查询日志分析
const mysql = require('mysql2/promise');

const pool = mysql.createPool({
  host: 'localhost',
  user: 'root',
  password: 'password',
  database: 'mydb',
  
  // 启用查询日志
  multipleStatements: true,
  debug: false,
  
  // 慢查询阈值
  queryTimeout: 1000
});

// 拦截查询
const originalQuery = pool.query;
pool.query = async function(...args) {
  const start = Date.now();
  
  try {
    const result = await originalQuery.apply(this, args);
    const duration = Date.now() - start;
    
    if (duration > 100) {
      console.warn('[SLOW QUERY]', {
        sql: args[0],
        duration: `${duration}ms`,
        stack: new Error().stack
      });
    }
    
    return result;
  } catch (error) {
    console.error('[QUERY ERROR]', args[0], error);
    throw error;
  }
};

// 使用 EXPLAIN 分析
async function analyzeQuery(sql) {
  const [explain] = await pool.query(`EXPLAIN ${sql}`);
  
  console.log('Query Plan:', explain);
  
  // 检查是否有全表扫描
  for (const row of explain) {
    if (row.type === 'ALL') {
      console.warn('Full table scan detected:', row.table);
    }
    
    if (!row.key) {
      console.warn('No index used:', row.table);
    }
  }
}

// 分析示例
analyzeQuery('SELECT * FROM users WHERE email = "test@example.com"');
```

// ❌ N+1 查询
async function getBadUsers() {
  const users = await db.users.findAll();
  
  for (const user of users) {
    user.posts = await db.posts.findAll({ userId: user.id });
  }
  
  return users;
}

// ✅ JOIN 查询
async function getGoodUsers() {
  const users = await db.query(`
    SELECT 
      u.*,
      JSON_ARRAYAGG(
        JSON_OBJECT('id', p.id, 'title', p.title)
      ) as posts
    FROM users u
    LEFT JOIN posts p ON p.user_id = u.id
    GROUP BY u.id
  `);
  
  return users;
}

// ✅ 使用 ORM 的 include
async function getGoodUsersORM() {
  const users = await db.users.findAll({
    include: [{ model: db.posts }]
  });
  
  return users;
}

### 2. 缓存优化

```javascript
const Redis = require('ioredis');
const redis = new Redis();

// 多级缓存
class CacheService {
  constructor() {
    this.localCache = new Map();
    this.localCacheTTL = 60 * 1000;  // 1分钟
  }
  
  async get(key, fetcher, ttl = 3600) {
    // 1. 本地缓存
    const localData = this.localCache.get(key);
    if (localData && Date.now() < localData.expireAt) {
      return localData.value;
    }
    
    // 2. Redis 缓存
    const redisData = await redis.get(key);
    if (redisData) {
      const value = JSON.parse(redisData);
      
      // 更新本地缓存
      this.localCache.set(key, {
        value,
        expireAt: Date.now() + this.localCacheTTL
      });
      
      return value;
    }
    
    // 3. 数据库查询
    const value = await fetcher();
    
    // 更新 Redis
    await redis.setex(key, ttl, JSON.stringify(value));
    
    // 更新本地缓存
    this.localCache.set(key, {
      value,
      expireAt: Date.now() + this.localCacheTTL
    });
    
    return value;
  }
}

const cache = new CacheService();

// 使用
app.get('/api/users/:id', async (req, res) => {
  const user = await cache.get(
    `user:${req.params.id}`,
    () => db.users.findById(req.params.id),
    3600
  );
  
  res.json(user);
});
```

### 4. 分页优化

```javascript
// ❌ OFFSET 分页（慢）
async function getBadPosts(page, pageSize) {
  const offset = (page - 1) * pageSize;
  
  const posts = await db.query(`
    SELECT * FROM posts
    ORDER BY id DESC
    LIMIT ${pageSize} OFFSET ${offset}
  `);
  
  return posts;
}

// ✅ 游标分页（快）
async function getGoodPosts(lastId = 0, pageSize = 20) {
  const posts = await db.query(`
    SELECT * FROM posts
    WHERE id > ${lastId}
    ORDER BY id ASC
    LIMIT ${pageSize}
  `);
  
  return posts;
}
```
