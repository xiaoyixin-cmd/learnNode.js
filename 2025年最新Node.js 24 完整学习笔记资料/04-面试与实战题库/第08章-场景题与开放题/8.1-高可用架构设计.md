### 1. 负载均衡层

#### Nginx 配置

```nginx
# /etc/nginx/nginx.conf
upstream nodejs_backend {
    # 负载均衡策略
    least_conn;  # 最少连接数
    
    # 节点列表
    server 192.168.1.101:3000 weight=5 max_fails=3 fail_timeout=30s;
    server 192.168.1.102:3000 weight=5 max_fails=3 fail_timeout=30s;
    server 192.168.1.103:3000 weight=3 max_fails=3 fail_timeout=30s backup;
    
    # 健康检查
    check interval=3000 rise=2 fall=5 timeout=1000;
}

server {
    listen 80;
    server_name api.example.com;
    
    # 重定向到 HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name api.example.com;
    
    # SSL 配置
    ssl_certificate /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    
    # 限流
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=100r/s;
    limit_req zone=api_limit burst=200 nodelay;
    
    location / {
        proxy_pass http://nodejs_backend;
        
        # 代理配置
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # 超时配置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
        
        # 缓存配置
        proxy_cache api_cache;
        proxy_cache_valid 200 5m;
        proxy_cache_key $scheme$request_method$host$request_uri;
    }
}
```

#### 优雅关闭

```javascript
// server.js
const express = require('express');
const app = express();

let isShuttingDown = false;

// 健康检查
app.get('/health', (req, res) => {
  if (isShuttingDown) {
    res.status(503).send('Shutting down');
  } else {
    res.status(200).send('OK');
  }
});

const server = app.listen(3000, () => {
  console.log('Server started on port 3000');
  
  // 通知 PM2 进程已准备好
  if (process.send) {
    process.send('ready');
  }
});

// 优雅关闭
process.on('SIGINT', gracefulShutdown);
process.on('SIGTERM', gracefulShutdown);

function gracefulShutdown() {
  console.log('Received shutdown signal');
  
  isShuttingDown = true;
  
  // 停止接收新请求
  server.close(() => {
    console.log('Server closed');
    
    // 关闭数据库连接
    closeDatabase();
    
    // 关闭Redis连接
    closeRedis();
    
    // 退出进程
    process.exit(0);
  });
  
  // 强制退出（30秒后）
  setTimeout(() => {
    console.error('Forced shutdown');
    process.exit(1);
  }, 30000);
}
```

### 3. 缓存层

#### Redis Cluster

```javascript
const Redis = require('ioredis');

// Redis Cluster 配置
const redis = new Redis.Cluster([
  { host: '192.168.1.201', port: 7000 },
  { host: '192.168.1.202', port: 7001 },
  { host: '192.168.1.203', port: 7002 }
], {
  redisOptions: {
    password: 'your-password',
    db: 0
  },
  enableReadyCheck: true,
  maxRetriesPerRequest: 3,
  retryDelayOnFailover: 100,
  retryDelayOnClusterDown: 300
});

// 缓存封装
class CacheService {
  async get(key) {
    try {
      const data = await redis.get(key);
      return data ? JSON.parse(data) : null;
    } catch (error) {
      console.error('Cache get error:', error);
      return null;
    }
  }
  
  async set(key, value, ttl = 3600) {
    try {
      await redis.setex(key, ttl, JSON.stringify(value));
    } catch (error) {
      console.error('Cache set error:', error);
    }
  }
  
  async del(key) {
    try {
      await redis.del(key);
    } catch (error) {
      console.error('Cache del error:', error);
    }
  }
}

module.exports = new CacheService();
```

### 4. 数据库层

#### MySQL 主从复制

```javascript
const mysql = require('mysql2/promise');

// 主从连接池
const masterPool = mysql.createPool({
  host: '192.168.1.101',
  user: 'root',
  password: 'password',
  database: 'mydb',
  connectionLimit: 50,
  queueLimit: 0,
  waitForConnections: true
});

const slavePool = mysql.createPool({
  host: '192.168.1.102',
  user: 'readonly',
  password: 'password',
  database: 'mydb',
  connectionLimit: 100,
  queueLimit: 0,
  waitForConnections: true
});

// 读写分离
class Database {
  // 写操作（主库）
  async write(sql, params) {
    const [rows] = await masterPool.execute(sql, params);
    return rows;
  }
  
  // 读操作（从库）
  async read(sql, params) {
    try {
      const [rows] = await slavePool.execute(sql, params);
      return rows;
    } catch (error) {
      // 从库故障，降级到主库
      console.error('Slave DB error, fallback to master:', error);
      const [rows] = await masterPool.execute(sql, params);
      return rows;
    }
  }
}

module.exports = new Database();
```

### 5. 熔断与降级

```javascript
const CircuitBreaker = require('opossum');

// 熔断器配置
const options = {
  timeout: 3000,           // 超时时间
  errorThresholdPercentage: 50,  // 错误率阈值
  resetTimeout: 30000      // 重置时间
};

// API 熔断
const apiBreaker = new CircuitBreaker(async (url) => {
  const response = await fetch(url);
  
  if (!response.ok) {
    throw new Error(`HTTP ${response.status}`);
  }
  
  return response.json();
}, options);

// 降级处理
apiBreaker.fallback(() => {
  return {
    status: 'degraded',
    message: 'Service temporarily unavailable',
    data: null
  };
});

// 监听事件
apiBreaker.on('open', () => {
  console.log('Circuit breaker opened');
});

apiBreaker.on('halfOpen', () => {
  console.log('Circuit breaker half-opened');
});

apiBreaker.on('close', () => {
  console.log('Circuit breaker closed');
});

// 使用
app.get('/api/data', async (req, res) => {
  try {
    const data = await apiBreaker.fire('http://external-api.com/data');
    res.json(data);
  } catch (error) {
    res.status(503).json({ error: 'Service unavailable' });
  }
});
```

### 6. 监控告警

#### Prometheus 指标

```javascript
const promClient = require('prom-client');

// 创建注册表
const register = new promClient.Registry();

// 默认指标
promClient.collectDefaultMetrics({ register });

// 自定义指标
const httpRequestDuration = new promClient.Histogram({
  name: 'http_request_duration_seconds',
  help: 'HTTP request duration in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.1, 0.5, 1, 2, 5]
});

const httpRequestTotal = new promClient.Counter({
  name: 'http_requests_total',
  help: 'Total HTTP requests',
  labelNames: ['method', 'route', 'status_code']
});

register.registerMetric(httpRequestDuration);
register.registerMetric(httpRequestTotal);

// 中间件
app.use((req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    
    httpRequestDuration
      .labels(req.method, req.route?.path || req.path, res.statusCode)
      .observe(duration);
    
    httpRequestTotal
      .labels(req.method, req.route?.path || req.path, res.statusCode)
      .inc();
  });
  
  next();
});

// 指标暴露
app.get('/metrics', async (req, res) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});
```

### 1. 多可用区部署

```yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: api-server
spec:
  replicas: 6
  selector:
    matchLabels:
      app: api-server
  template:
    metadata:
      labels:
        app: api-server

      # 反亲和性（不同节点）
      affinity:
        podAntiAffinity:
          requiredDuringSchedulingIgnoredDuringExecution:
          - labelSelector:
              matchExpressions:
              - key: app
                operator: In
                values:
                - api-server
            topologyKey: kubernetes.io/hostname
      
      containers:
      - name: api-server
        image: api-server:latest
        ports:
        - containerPort: 3000

        livenessProbe:
          httpGet:
```
