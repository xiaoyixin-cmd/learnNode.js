### 内存泄漏的常见表现

| 现象 | 说明 |
|------|------|
| **内存持续增长** | RSS不断上升，GC无法回收 |
| **OOM崩溃** | 内存超限，进程被kill |
| **性能下降** | GC频繁，CPU占用高 |
| **响应变慢** | 内存不足，swap频繁 |

## 💻 识别内存泄漏

### 1. 监控内存使用

```javascript
const v8 = require('v8');

// 定时监控
setInterval(() => {
  const heapStats = v8.getHeapStatistics();
  const usage = process.memoryUsage();
  
  console.log({
    rss: `${(usage.rss / 1024 / 1024).toFixed(2)} MB`,
    heapTotal: `${(usage.heapTotal / 1024 / 1024).toFixed(2)} MB`,
    heapUsed: `${(usage.heapUsed / 1024 / 1024).toFixed(2)} MB`,
    external: `${(usage.external / 1024 / 1024).toFixed(2)} MB`,
    heapLimit: `${(heapStats.heap_size_limit / 1024 / 1024).toFixed(2)} MB`
  });
  
  // 告警阈值
  if (usage.heapUsed > heapStats.heap_size_limit * 0.9) {
    console.error('Memory usage is too high!');
    // 发送告警
    sendAlert('High memory usage');
  }
}, 10000);
```

## 💻 常见内存泄漏模式

### 1. 全局变量

```javascript
// ❌ 内存泄漏
let cache = [];

app.get('/api/data', (req, res) => {
  // 不断向全局数组添加数据
  cache.push({
    timestamp: Date.now(),
    data: req.body
  });
  
  res.json({ success: true });
});

// ✅ 修复：使用 LRU 缓存
const LRU = require('lru-cache');

const cache = new LRU({
  max: 1000,  // 最多1000项
  ttl: 60000  // 60秒过期
});

app.get('/api/data', (req, res) => {
  cache.set(Date.now(), req.body);
  res.json({ success: true });
});
```

---

### 2. 闭包引用

```javascript
// ❌ 内存泄漏
function createHandler() {
  const largeData = new Array(1000000).fill('x');
  
  return function(req, res) {
    // 闭包持有 largeData 引用
    res.json({ length: largeData.length });
  };
}

app.get('/api/handler', createHandler());

// ✅ 修复：只保留需要的数据
function createHandler() {
  const largeData = new Array(1000000).fill('x');
  const length = largeData.length;
  
  return function(req, res) {
    res.json({ length });
  };
}
```

---

### 3. 事件监听器

```javascript
// ❌ 内存泄漏
function processData() {
  const emitter = new EventEmitter();
  
  emitter.on('data', (data) => {
    console.log(data);
  });
  
  // 监听器未移除
}

setInterval(processData, 1000);

// ✅ 修复：移除监听器
function processData() {
  const emitter = new EventEmitter();
  
  const handler = (data) => {
    console.log(data);
  };
  
  emitter.on('data', handler);
  
  // 使用完后移除
  emitter.removeListener('data', handler);
}
```

---

### 4. 定时器未清理

```javascript
// ❌ 内存泄漏
app.get('/api/polling', (req, res) => {
  setInterval(() => {
    // 定时器永不清理
    fetch('http://api.example.com/data');
  }, 1000);
  
  res.json({ success: true });
});

// ✅ 修复：清理定时器
const timers = new Map();

app.get('/api/polling', (req, res) => {
  const id = req.query.id;
  
  const timerId = setInterval(() => {
    fetch('http://api.example.com/data');
  }, 1000);
  
  timers.set(id, timerId);
  
  res.json({ success: true, id });
});

app.delete('/api/polling/:id', (req, res) => {
  const timerId = timers.get(req.params.id);
  
  if (timerId) {
    clearInterval(timerId);
    timers.delete(req.params.id);
  }
  
  res.json({ success: true });
});
```

## 💻 线上处理流程

### 1. 紧急处理

```javascript
// 监控告警
const threshold = 0.9;

setInterval(() => {
  const heapStats = v8.getHeapStatistics();
  const usage = process.memoryUsage();
  
  if (usage.heapUsed > heapStats.heap_size_limit * threshold) {
    console.error('Memory threshold exceeded');
    
    // 1. 生成快照
    takeSnapshot(`emergency-${Date.now()}.heapsnapshot`);
    
    // 2. 发送告警
    sendAlert('Memory leak detected');
    
    // 3. 手动GC（如果启用了 --expose-gc）
    if (global.gc) {
      global.gc();
    }
    
    // 4. 重启进程（PM2）
    setTimeout(() => {
      process.exit(1);
    }, 5000);
  }
}, 10000);
```

memwatch.on('leak', (info) => {

### 3. 单元测试

```javascript
const assert = require('assert');

describe('Memory leak test', () => {
  it('should not leak memory', async () => {
    const before = process.memoryUsage().heapUsed;
    
    // 执行多次操作
    for (let i = 0; i < 10000; i++) {
      await processData();
    }
    
    // 手动GC
    if (global.gc) {
      global.gc();
    }
    
    const after = process.memoryUsage().heapUsed;
    const diff = after - before;
    
    // 内存增长不应超过10MB
    assert(diff < 10 * 1024 * 1024, `Memory leak: ${diff} bytes`);
  });
});
```

## 🎯 面试标准答题

### 问题："如何处理内存泄漏？"

**标准答案（20-30分钟）：**

**1. 识别（5分钟）**
> "通过监控发现：  
> - 内存持续增长（RSS不降）  
> - GC频繁但无效  
> - 最终OOM崩溃"

**2. 定位（8分钟）**
> "使用工具定位：  
> 1. Heap Snapshot：对比快照，找增长对象  
> 2. Chrome DevTools：分析引用链  
> 3. clinic.js：火焰图分析  
>   
> 常见原因：  
> - 全局变量无限增长  
> - 定时器未清理  
> - 事件监听器未移除  
> - 闭包持有大对象"

**3. 修复案例（10分钟）**
> "我之前遇到过：  
>   
> **问题：** 内存每小时增长200MB，最终OOM  
>   
> **定位：**  
> 1. Heap Snapshot发现大量缓存对象  
> 2. 代码审查发现全局数组无限增长  
> 3. 定时器累积未清理  
>   
> **修复：**  
> 1. 全局数组改为LRU缓存（限制1000项）  
> 2. 定时器添加清理逻辑  
> 3. 添加内存监控告警  
>   
> **效果：**  
> - 内存稳定在500MB  
> - 无OOM崩溃  
> - GC时间降低80%"

**4. 预防（5分钟）**
> "预防措施：  
> 1. 代码审查：检查定时器、监听器  
> 2. 自动化检测：memwatch-next  
> 3. 单元测试：内存增长测试  
> 4. 监控告警：Prometheus + Grafana  
> 5. 资源限制：PM2 max_memory_restart"

## 📝 总结

### 核心要点
1. **识别**：监控内存持续增长
2. **定位**：Heap Snapshot + Chrome DevTools
3. **修复**：清理定时器/监听器/连接
4. **预防**：代码审查 + 自动化检测

### 常见泄漏
- 全局变量无限增长
- 定时器未清理
- 事件监听器未移除
- 闭包持有大对象
- 连接未关闭
