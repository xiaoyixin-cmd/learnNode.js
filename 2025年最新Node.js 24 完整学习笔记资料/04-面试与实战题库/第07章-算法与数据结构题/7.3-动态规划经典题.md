## 💻 基础题型

### 1. 斐波那契数列

#### 问题
```
F(0) = 0, F(1) = 1
F(n) = F(n-1) + F(n-2)
```

#### 递归（低效）

```javascript
/**
 * 斐波那契（递归）
 * @param {number} n
 * @returns {number}
 * 时间复杂度：O(2^n) - 指数级
 * 空间复杂度：O(n) - 递归栈
 */
function fibRecursive(n) {
  if (n <= 1) return n;
  
  return fibRecursive(n - 1) + fibRecursive(n - 2);
}

// fibRecursive(40) 需要几秒钟
```

---

#### 记忆化搜索（优化）

```javascript
/**
 * 斐波那契（记忆化）
 * @param {number} n
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)
 */
function fibMemo(n, memo = {}) {
  if (n <= 1) return n;
  
  if (memo[n]) return memo[n];
  
  memo[n] = fibMemo(n - 1, memo) + fibMemo(n - 2, memo);
  
  return memo[n];
}

// fibMemo(40) 瞬间完成
console.log(fibMemo(40));  // 102334155
```

---

#### 动态规划（最优）

```javascript
/**
 * 斐波那契（DP）
 * @param {number} n
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)
 */
function fibDP(n) {
  if (n <= 1) return n;
  
  const dp = new Array(n + 1);
  dp[0] = 0;
  dp[1] = 1;
  
  for (let i = 2; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }
  
  return dp[n];
}

console.log(fibDP(40));  // 102334155
```

---

#### 空间优化（O(1)）

```javascript
/**
 * 斐波那契（空间优化）
 * @param {number} n
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function fibOptimized(n) {
  if (n <= 1) return n;
  
  let prev2 = 0;  // F(n-2)
  let prev1 = 1;  // F(n-1)
  
  for (let i = 2; i <= n; i++) {
    const current = prev1 + prev2;
    prev2 = prev1;
    prev1 = current;
  }
  
  return prev1;
}

console.log(fibOptimized(40));  // 102334155
```

### 2. 爬楼梯

#### 问题
```
爬n阶楼梯，每次可以爬1或2阶，有多少种方法？

示例：
n = 2: 2种 (1+1, 2)
n = 3: 3种 (1+1+1, 1+2, 2+1)
```

#### 状态转移方程
```
dp[i] = dp[i-1] + dp[i-2]

解释：
- 爬到第i阶的方法 = 从第i-1阶爬1步 + 从第i-2阶爬2步
```

#### 实现

```javascript
/**
 * 爬楼梯
 * @param {number} n
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function climbStairs(n) {
  if (n <= 2) return n;
  
  let prev2 = 1;  // dp[1]
  let prev1 = 2;  // dp[2]
  
  for (let i = 3; i <= n; i++) {
    const current = prev1 + prev2;
    prev2 = prev1;
    prev1 = current;
  }
  
  return prev1;
}

// 测试
console.log(climbStairs(5));  // 8
```

### 3. 最长公共子序列（LCS）

#### 问题
```
找出两个字符串的最长公共子序列

示例：
text1 = "abcde"
text2 = "ace"
LCS = "ace" (长度3)
```

#### 状态转移方程
```
dp[i][j] = 
  - dp[i-1][j-1] + 1          (text1[i] == text2[j])
  - max(dp[i-1][j], dp[i][j-1])  (text1[i] != text2[j])
```

#### 实现

```javascript
/**
 * 最长公共子序列
 * @param {string} text1
 * @param {string} text2
 * @returns {number}
 * 时间复杂度：O(m*n)
 * 空间复杂度：O(m*n)
 */
function longestCommonSubsequence(text1, text2) {
  const m = text1.length;
  const n = text2.length;
  
  // dp[i][j]: text1[0...i-1] 和 text2[0...j-1] 的LCS长度
  const dp = Array.from({ length: m + 1 }, () => 
    new Array(n + 1).fill(0)
  );
  
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  
  return dp[m][n];
}

// 测试
console.log(longestCommonSubsequence("abcde", "ace"));  // 3
```

---

#### 空间优化（滚动数组）

```javascript
/**
 * LCS（空间优化）
 * @param {string} text1
 * @param {string} text2
 * @returns {number}
 * 时间复杂度：O(m*n)
 * 空间复杂度：O(n)
 */
function lcsOptimized(text1, text2) {
  const m = text1.length;
  const n = text2.length;
  
  let prev = new Array(n + 1).fill(0);
  let curr = new Array(n + 1).fill(0);
  
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        curr[j] = prev[j - 1] + 1;
      } else {
        curr[j] = Math.max(prev[j], curr[j - 1]);
      }
    }
    
    [prev, curr] = [curr, prev];
  }
  
  return prev[n];
}

console.log(lcsOptimized("abcde", "ace"));  // 3
```

## 💻 背包问题

### 1. 0-1背包

#### 问题
```
n个物品，每个物品有重量w[i]和价值v[i]
背包容量W，每个物品只能选一次，求最大价值

示例：
weights = [1, 3, 4]
values = [15, 20, 30]
capacity = 4
最大价值 = 35 (选第1和第2个)
```

#### 状态转移方程
```
dp[i][w] = max(
  dp[i-1][w],                    // 不选第i个
  dp[i-1][w-weights[i]] + values[i]  // 选第i个
)
```

#### 实现

```javascript
/**
 * 0-1背包
 * @param {number[]} weights
 * @param {number[]} values
 * @param {number} capacity
 * @returns {number}
 * 时间复杂度：O(n*W)
 * 空间复杂度：O(n*W)
 */
function knapsack(weights, values, capacity) {
  const n = weights.length;
  
  // dp[i][w]: 前i个物品，容量w时的最大价值
  const dp = Array.from({ length: n + 1 }, () =>
    new Array(capacity + 1).fill(0)
  );
  
  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= capacity; w++) {
      // 不选第i个物品
      dp[i][w] = dp[i - 1][w];
      
      // 选第i个物品（如果放得下）
      if (w >= weights[i - 1]) {
        dp[i][w] = Math.max(
          dp[i][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]
        );
      }
    }
  }
  
  return dp[n][capacity];
}

// 测试
console.log(knapsack([1, 3, 4], [15, 20, 30], 4));  // 35
```

---

#### 空间优化（一维数组）

```javascript
/**
 * 0-1背包（空间优化）
 * @param {number[]} weights
 * @param {number[]} values
 * @param {number} capacity
 * @returns {number}
 * 时间复杂度：O(n*W)
 * 空间复杂度：O(W)
 */
function knapsackOptimized(weights, values, capacity) {
  const n = weights.length;
  const dp = new Array(capacity + 1).fill(0);
  
  for (let i = 0; i < n; i++) {
    // 从后往前遍历（避免覆盖）
    for (let w = capacity; w >= weights[i]; w--) {
      dp[w] = Math.max(
        dp[w],
        dp[w - weights[i]] + values[i]
      );
    }
  }
  
  return dp[capacity];
}

console.log(knapsackOptimized([1, 3, 4], [15, 20, 30], 4));  // 35
```

### 2. 完全背包

#### 问题
```
每个物品可以选无限次
```

#### 实现

```javascript
/**
 * 完全背包
 * @param {number[]} weights
 * @param {number[]} values
 * @param {number} capacity
 * @returns {number}
 * 时间复杂度：O(n*W)
 * 空间复杂度：O(W)
 */
function unboundedKnapsack(weights, values, capacity) {
  const n = weights.length;
  const dp = new Array(capacity + 1).fill(0);
  
  for (let i = 0; i < n; i++) {
    // 从前往后遍历（可以重复选）
    for (let w = weights[i]; w <= capacity; w++) {
      dp[w] = Math.max(
        dp[w],
        dp[w - weights[i]] + values[i]
      );
    }
  }
  
  return dp[capacity];
}

// 测试
console.log(unboundedKnapsack([1, 3, 4], [15, 20, 30], 4));  // 60 (4个第1个)
```

## 💻 股票买卖

### 1. 买卖一次

#### 问题
```
只能买卖一次，求最大利润

示例：
prices = [7, 1, 5, 3, 6, 4]
最大利润 = 5 (第2天买，第5天卖)
```

#### 实现

```javascript
/**
 * 买卖股票I（一次）
 * @param {number[]} prices
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function maxProfit(prices) {
  let minPrice = Infinity;
  let maxProfit = 0;
  
  for (const price of prices) {
    minPrice = Math.min(minPrice, price);
    maxProfit = Math.max(maxProfit, price - minPrice);
  }
  
  return maxProfit;
}

// 测试
console.log(maxProfit([7, 1, 5, 3, 6, 4]));  // 5
```

### 2. 买卖多次

#### 问题
```
可以买卖多次，求最大利润

示例：
prices = [7, 1, 5, 3, 6, 4]
最大利润 = 7 (第2天买第3天卖 + 第4天买第5天卖)
```

#### 实现

```javascript
/**
 * 买卖股票II（多次）
 * @param {number[]} prices
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function maxProfitMultiple(prices) {
  let profit = 0;
  
  for (let i = 1; i < prices.length; i++) {
    // 只要今天价格高于昨天，就交易
    if (prices[i] > prices[i - 1]) {
      profit += prices[i] - prices[i - 1];
    }
  }
  
  return profit;
}

// 测试
console.log(maxProfitMultiple([7, 1, 5, 3, 6, 4]));  // 7
```

### 3. 买卖两次

#### 问题
```
最多买卖两次，求最大利润
```

#### 状态转移方程
```
buy1 = max(buy1, -prices[i])
sell1 = max(sell1, buy1 + prices[i])
buy2 = max(buy2, sell1 - prices[i])
sell2 = max(sell2, buy2 + prices[i])
```

#### 实现

```javascript
/**
 * 买卖股票III（两次）
 * @param {number[]} prices
 * @returns {number}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function maxProfitTwoTimes(prices) {
  let buy1 = -Infinity, sell1 = 0;
  let buy2 = -Infinity, sell2 = 0;
  
  for (const price of prices) {
    buy1 = Math.max(buy1, -price);
    sell1 = Math.max(sell1, buy1 + price);
    buy2 = Math.max(buy2, sell1 - price);
    sell2 = Math.max(sell2, buy2 + price);
  }
  
  return sell2;
}

// 测试
console.log(maxProfitTwoTimes([3, 3, 5, 0, 0, 3, 1, 4]));  // 6
```

## 🎯 面试标准答题

### 问题："最长公共子序列"

**标准答案（20-25分钟）：**

**1. 问题理解（3分钟）**
> "LCS是找两个字符串的最长公共子序列（不要求连续）  
> 例如：'abcde'和'ace'的LCS是'ace'，长度3"

**2. 状态定义（5分钟）**
> "定义dp[i][j]：text1前i个字符和text2前j个字符的LCS长度  
>   
> 状态转移：  
> - 如果text1[i-1] == text2[j-1]：dp[i][j] = dp[i-1][j-1] + 1  
> - 否则：dp[i][j] = max(dp[i-1][j], dp[i][j-1])"

**3. 代码实现（10分钟）**
```javascript
function longestCommonSubsequence(text1, text2) {
  const m = text1.length;
  const n = text2.length;
  
  const dp = Array.from({ length: m + 1 }, () =>
    new Array(n + 1).fill(0)
  );
  
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  
  return dp[m][n];
}
```

**4. 复杂度分析（2分钟）**
> "时间复杂度：O(m*n) - 双重循环  
> 空间复杂度：O(m*n) - 二维数组"

**5. 优化方向（3分钟）**
> "空间优化：滚动数组，只保留两行  
> 空间复杂度降到O(n)"
