## 💻 回文字符串

### 1. 判断回文

```javascript
/**
 * 判断回文字符串
 * @param {string} s
 * @returns {boolean}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function isPalindrome(s) {
  // 预处理：去除非字母数字，转小写
  s = s.replace(/[^a-zA-Z0-9]/g, '').toLowerCase();
  
  let left = 0;
  let right = s.length - 1;
  
  while (left < right) {
    if (s[left] !== s[right]) {
      return false;
    }
    
    left++;
    right--;
  }
  
  return true;
}

// 测试
console.log(isPalindrome("A man, a plan, a canal: Panama"));  // true
console.log(isPalindrome("race a car"));  // false
```

### 2. 验证回文字符串II（最多删除一个字符）

```javascript
/**
 * 验证回文字符串II
 * @param {string} s
 * @returns {boolean}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function validPalindrome(s) {
  let left = 0;
  let right = s.length - 1;
  
  while (left < right) {
    if (s[left] !== s[right]) {
      // 尝试删除左边或右边的字符
      return isPalindromeRange(s, left + 1, right) || 
             isPalindromeRange(s, left, right - 1);
    }
    
    left++;
    right--;
  }
  
  return true;
}

function isPalindromeRange(s, left, right) {
  while (left < right) {
    if (s[left] !== s[right]) {
      return false;
    }
    
    left++;
    right--;
  }
  
  return true;
}

// 测试
console.log(validPalindrome("aba"));   // true
console.log(validPalindrome("abca"));  // true (删除c)
console.log(validPalindrome("abc"));   // false
```

### 3. 最长回文子串

#### 中心扩展法

```javascript
/**
 * 最长回文子串（中心扩展）
 * @param {string} s
 * @returns {string}
 * 时间复杂度：O(n²)
 * 空间复杂度：O(1)
 */
function longestPalindrome(s) {
  if (!s || s.length < 2) return s;
  
  let start = 0;
  let maxLen = 0;
  
  for (let i = 0; i < s.length; i++) {
    // 奇数长度回文（中心是一个字符）
    const len1 = expandAroundCenter(s, i, i);
    
    // 偶数长度回文（中心是两个字符）
    const len2 = expandAroundCenter(s, i, i + 1);
    
    const len = Math.max(len1, len2);
    
    if (len > maxLen) {
      maxLen = len;
      start = i - Math.floor((len - 1) / 2);
    }
  }
  
  return s.substring(start, start + maxLen);
}

function expandAroundCenter(s, left, right) {
  while (left >= 0 && right < s.length && s[left] === s[right]) {
    left--;
    right++;
  }
  
  return right - left - 1;
}

// 测试
console.log(longestPalindrome("babad"));  // "bab" 或 "aba"
console.log(longestPalindrome("cbbd"));   // "bb"
```

#### 动态规划

```javascript
/**
 * 最长回文子串（DP）
 * @param {string} s
 * @returns {string}
 * 时间复杂度：O(n²)
 * 空间复杂度：O(n²)
 */
function longestPalindromeDP(s) {
  const n = s.length;
  if (n < 2) return s;
  
  // dp[i][j]: s[i...j]是否为回文
  const dp = Array.from({ length: n }, () => new Array(n).fill(false));
  
  let start = 0;
  let maxLen = 1;
  
  // 单个字符
  for (let i = 0; i < n; i++) {
    dp[i][i] = true;
  }
  
  // 两个字符
  for (let i = 0; i < n - 1; i++) {
    if (s[i] === s[i + 1]) {
      dp[i][i + 1] = true;
      start = i;
      maxLen = 2;
    }
  }
  
  // 三个及以上字符
  for (let len = 3; len <= n; len++) {
    for (let i = 0; i <= n - len; i++) {
      const j = i + len - 1;
      
      if (s[i] === s[j] && dp[i + 1][j - 1]) {
        dp[i][j] = true;
        start = i;
        maxLen = len;
      }
    }
  }
  
  return s.substring(start, start + maxLen);
}

// 测试
console.log(longestPalindromeDP("babad"));  // "bab" 或 "aba"
```

## 💻 子串匹配

### 1. 暴力匹配

```javascript
/**
 * 子串匹配（暴力）
 * @param {string} text
 * @param {string} pattern
 * @returns {number} - 第一次出现的索引，-1表示不存在
 * 时间复杂度：O(m*n)
 * 空间复杂度：O(1)
 */
function strStr(text, pattern) {
  if (!pattern) return 0;
  
  const m = text.length;
  const n = pattern.length;
  
  for (let i = 0; i <= m - n; i++) {
    let j = 0;
    
    while (j < n && text[i + j] === pattern[j]) {
      j++;
    }
    
    if (j === n) {
      return i;  // 找到匹配
    }
  }
  
  return -1;  // 未找到
}

// 测试
console.log(strStr("hello", "ll"));  // 2
console.log(strStr("aaaaa", "bba"));  // -1
```

### 2. KMP算法

```javascript
/**
 * KMP算法
 * @param {string} text
 * @param {string} pattern
 * @returns {number}
 * 时间复杂度：O(m + n)
 * 空间复杂度：O(n)
 */
function kmp(text, pattern) {
  if (!pattern) return 0;
  
  const m = text.length;
  const n = pattern.length;
  
  // 1. 构建next数组（部分匹配表）
  const next = buildNext(pattern);
  
  // 2. 匹配
  let i = 0;  // text指针
  let j = 0;  // pattern指针
  
  while (i < m) {
    if (text[i] === pattern[j]) {
      i++;
      j++;
    } else if (j > 0) {
      j = next[j - 1];  // 回退
    } else {
      i++;
    }
    
    if (j === n) {
      return i - j;  // 找到匹配
    }
  }
  
  return -1;
}

function buildNext(pattern) {
  const n = pattern.length;
  const next = new Array(n).fill(0);
  
  let j = 0;
  
  for (let i = 1; i < n; i++) {
    while (j > 0 && pattern[i] !== pattern[j]) {
      j = next[j - 1];
    }
    
    if (pattern[i] === pattern[j]) {
      j++;
    }
    
    next[i] = j;
  }
  
  return next;
}

// 测试
console.log(kmp("hello", "ll"));  // 2
console.log(kmp("aaaaa", "bba"));  // -1
```

### 3. Rabin-Karp算法（滚动哈希）

```javascript
/**
 * Rabin-Karp算法
 * @param {string} text
 * @param {string} pattern
 * @returns {number}
 * 时间复杂度：O(m + n) 平均
 * 空间复杂度：O(1)
 */
function rabinKarp(text, pattern) {
  const m = text.length;
  const n = pattern.length;
  
  if (n > m) return -1;
  
  const base = 256;
  const mod = 101;
  
  // 计算pattern的哈希值
  let patternHash = 0;
  let textHash = 0;
  let h = 1;
  
  // h = base^(n-1) % mod
  for (let i = 0; i < n - 1; i++) {
    h = (h * base) % mod;
  }
  
  // 计算初始哈希
  for (let i = 0; i < n; i++) {
    patternHash = (base * patternHash + pattern.charCodeAt(i)) % mod;
    textHash = (base * textHash + text.charCodeAt(i)) % mod;
  }
  
  // 滚动哈希
  for (let i = 0; i <= m - n; i++) {
    // 哈希值相等，验证字符串
    if (patternHash === textHash) {
      let j = 0;
      
      while (j < n && text[i + j] === pattern[j]) {
        j++;
      }
      
      if (j === n) {
        return i;
      }
    }
    
    // 计算下一个哈希值
    if (i < m - n) {
      textHash = (base * (textHash - text.charCodeAt(i) * h) + 
                  text.charCodeAt(i + n)) % mod;
      
      if (textHash < 0) {
        textHash += mod;
      }
    }
  }
  
  return -1;
}

// 测试
console.log(rabinKarp("hello", "ll"));  // 2
```

## 💻 字符串操作

### 1. 字符串压缩

```javascript
/**
 * 字符串压缩
 * @param {string} s
 * @returns {string}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)
 */
function compressString(s) {
  if (!s) return s;
  
  const result = [];
  let count = 1;
  
  for (let i = 1; i <= s.length; i++) {
    if (i < s.length && s[i] === s[i - 1]) {
      count++;
    } else {
      result.push(s[i - 1]);
      
      if (count > 1) {
        result.push(count.toString());
      }
      
      count = 1;
    }
  }
  
  const compressed = result.join('');
  
  return compressed.length < s.length ? compressed : s;
}

// 测试
console.log(compressString("aabcccccaaa"));  // "a2bc5a3"
console.log(compressString("abcd"));         // "abcd"
```

### 2. 翻转字符串中的单词

```javascript
/**
 * 翻转字符串中的单词
 * @param {string} s
 * @returns {string}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)
 */
function reverseWords(s) {
  // 去除多余空格，分割，翻转，合并
  return s.trim()
          .split(/\s+/)
          .reverse()
          .join(' ');
}

// 测试
console.log(reverseWords("the sky is blue"));  // "blue is sky the"
console.log(reverseWords("  hello world  "));  // "world hello"
```

### 3. 字符串相加

```javascript
/**
 * 字符串相加（大数加法）
 * @param {string} num1
 * @param {string} num2
 * @returns {string}
 * 时间复杂度：O(max(m, n))
 * 空间复杂度：O(max(m, n))
 */
function addStrings(num1, num2) {
  let i = num1.length - 1;
  let j = num2.length - 1;
  let carry = 0;
  const result = [];
  
  while (i >= 0 || j >= 0 || carry > 0) {
    const digit1 = i >= 0 ? parseInt(num1[i]) : 0;
    const digit2 = j >= 0 ? parseInt(num2[j]) : 0;
    
    const sum = digit1 + digit2 + carry;
    
    result.push(sum % 10);
    carry = Math.floor(sum / 10);
    
    i--;
    j--;
  }
  
  return result.reverse().join('');
}

// 测试
console.log(addStrings("123", "456"));  // "579"
console.log(addStrings("999", "1"));    // "1000"
```

## 💻 实战应用

### 1. URL解析

```javascript
/**
 * URL参数解析
 * @param {string} url
 * @returns {Object}
 */
function parseURL(url) {
  const params = {};
  
  // 找到?的位置
  const queryIndex = url.indexOf('?');
  
  if (queryIndex === -1) {
    return params;
  }
  
  // 提取查询字符串
  const queryString = url.substring(queryIndex + 1);
  
  // 分割参数
  const pairs = queryString.split('&');
  
  for (const pair of pairs) {
    const [key, value] = pair.split('=');
    params[decodeURIComponent(key)] = decodeURIComponent(value || '');
  }
  
  return params;
}

// 测试
console.log(parseURL("https://example.com?name=John&age=30"));
// { name: 'John', age: '30' }
```

### 2. 简单模板引擎

```javascript
/**
 * 简单模板引擎
 * @param {string} template
 * @param {Object} data
 * @returns {string}
 */
function renderTemplate(template, data) {
  return template.replace(/\{\{(\w+)\}\}/g, (match, key) => {
    return data[key] || '';
  });
}

// 测试
const template = "Hello, {{name}}! You are {{age}} years old.";
const data = { name: 'Alice', age: 25 };

console.log(renderTemplate(template, data));
// "Hello, Alice! You are 25 years old."
```

### 3. 路径简化

```javascript
/**
 * 简化路径
 * @param {string} path
 * @returns {string}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)
 */
function simplifyPath(path) {
  const stack = [];
  const parts = path.split('/');
  
  for (const part of parts) {
    if (part === '' || part === '.') {
      continue;
    } else if (part === '..') {
      stack.pop();
    } else {
      stack.push(part);
    }
  }
  
  return '/' + stack.join('/');
}

// 测试
console.log(simplifyPath("/home/"));           // "/home"
console.log(simplifyPath("/../"));             // "/"
console.log(simplifyPath("/home//foo/"));      // "/home/foo"
console.log(simplifyPath("/a/./b/../../c/")); // "/c"
```

## 🎯 面试标准答题

### 问题："最长回文子串"

**标准答案（15-25分钟）：**

**1. 问题理解（2分钟）**
> "找字符串中最长的回文子串  
> 例如：'babad' → 'bab'或'aba'"

**2. 思路分析（5分钟）**
> "有两种方法：  
> 1. 中心扩展：O(n²)时间，O(1)空间  
> 2. 动态规划：O(n²)时间，O(n²)空间  
>   
> 推荐中心扩展，空间更优"

**3. 代码实现（12分钟）**
```javascript

```

**4. 复杂度分析（2分钟）**
> "时间复杂度：O(n²) - 每个中心扩展O(n)  
> 空间复杂度：O(1) - 只用常量空间"
