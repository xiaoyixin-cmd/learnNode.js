## 💻 0-1背包问题

### 问题描述

给定 n 个物品，每个物品有重量 `w[i]` 和价值 `v[i]`。  
背包容量为 `W`，求最大价值。  
**约束：** 每个物品只能选0次或1次。

### 递归实现（60分）

```javascript
/**
 * 0-1背包（暴力递归）
 * @param {number[]} weights - 重量数组
 * @param {number[]} values - 价值数组
 * @param {number} W - 背包容量
 */
function knapsack(weights, values, W) {
  const n = weights.length;
  
  function dp(i, w) {
    // base case
    if (i < 0 || w <= 0) return 0;
    
    // 选择1：不放入第i个物品
    const notTake = dp(i - 1, w);
    
    // 选择2：放入第i个物品
    let take = 0;
    if (w >= weights[i]) {
      take = dp(i - 1, w - weights[i]) + values[i];
    }
    
    return Math.max(notTake, take);
  }
  
  return dp(n - 1, W);
}

// 测试
const weights = [1, 3, 4];
const values = [15, 20, 30];
const W = 4;

console.log(knapsack(weights, values, W));  // 35（选物品0和1）
```

**时间复杂度：** O(2^n) - 指数级  
**空间复杂度：** O(n) - 递归栈

### 记忆化搜索（80分）

```javascript
/**
 * 0-1背包（记忆化搜索）
 */
function knapsack(weights, values, W) {
  const n = weights.length;
  const memo = Array.from({ length: n }, () => Array(W + 1).fill(-1));
  
  function dp(i, w) {
    if (i < 0 || w <= 0) return 0;
    
    if (memo[i][w] !== -1) return memo[i][w];
    
    const notTake = dp(i - 1, w);
    
    let take = 0;
    if (w >= weights[i]) {
      take = dp(i - 1, w - weights[i]) + values[i];
    }
    
    memo[i][w] = Math.max(notTake, take);
    return memo[i][w];
  }
  
  return dp(n - 1, W);
}
```

**时间复杂度：** O(n * W)  
**空间复杂度：** O(n * W)

### 动态规划（二维DP）（90分）

```javascript
/**
 * 0-1背包（动态规划）
 */
function knapsack(weights, values, W) {
  const n = weights.length;
  const dp = Array.from({ length: n + 1 }, () => Array(W + 1).fill(0));
  
  // dp[i][w] = 前i个物品，容量为w时的最大价值
  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= W; w++) {
      // 不选第i个物品
      dp[i][w] = dp[i - 1][w];
      
      // 选第i个物品
      if (w >= weights[i - 1]) {
        dp[i][w] = Math.max(
          dp[i][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]
        );
      }
    }
  }
  
  return dp[n][W];
}
```

**状态转移方程：**
```
dp[i][w] = max(
  dp[i-1][w],                              // 不选
  dp[i-1][w - weights[i-1]] + values[i-1]  // 选
)
```

### 空间优化（一维DP）（100分）

```javascript
/**
 * 0-1背包（滚动数组优化）
 */
function knapsack(weights, values, W) {
  const n = weights.length;
  const dp = Array(W + 1).fill(0);
  
  for (let i = 0; i < n; i++) {
    // 从右向左更新（避免覆盖）
    for (let w = W; w >= weights[i]; w--) {
      dp[w] = Math.max(
        dp[w],
        dp[w - weights[i]] + values[i]
      );
    }
  }
  
  return dp[W];
}
```

**空间复杂度：** O(W)

### 路径还原（进阶）

```javascript
/**
 * 0-1背包（还原选择的物品）
 */
function knapsackWithPath(weights, values, W) {
  const n = weights.length;
  const dp = Array.from({ length: n + 1 }, () => Array(W + 1).fill(0));
  
  // 构建DP表
  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= W; w++) {
      dp[i][w] = dp[i - 1][w];
      
      if (w >= weights[i - 1]) {
        dp[i][w] = Math.max(
          dp[i][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]
        );
      }
    }
  }
  
  // 还原路径
  const selected = [];
  let i = n, w = W;
  
  while (i > 0 && w > 0) {
    if (dp[i][w] !== dp[i - 1][w]) {
      selected.push(i - 1);
      w -= weights[i - 1];
    }
    i--;
  }
  
  return {
    maxValue: dp[n][W],
    items: selected.reverse()
  };
}

// 测试
const result = knapsackWithPath([1, 3, 4], [15, 20, 30], 4);
console.log(result);  // { maxValue: 35, items: [0, 1] }
```

## 💻 最长公共子序列（LCS）

### 问题描述

给定两个字符串 `s1` 和 `s2`，求它们的最长公共子序列长度。

**示例：**
```
s1 = "abcde"
s2 = "ace"
LCS = "ace"，长度为3
```

### 递归实现（60分）

```javascript
/**
 * LCS（暴力递归）
 */
function longestCommonSubsequence(s1, s2) {
  function dp(i, j) {
    // base case
    if (i < 0 || j < 0) return 0;
    
    if (s1[i] === s2[j]) {
      return dp(i - 1, j - 1) + 1;
    } else {
      return Math.max(
        dp(i - 1, j),
        dp(i, j - 1)
      );
    }
  }
  
  return dp(s1.length - 1, s2.length - 1);
}

// 测试
console.log(longestCommonSubsequence("abcde", "ace"));  // 3
```

**时间复杂度：** O(2^(m+n))

---

### 动态规划（二维DP）（90分）

```javascript
/**
 * LCS（动态规划）
 */
function longestCommonSubsequence(s1, s2) {
  const m = s1.length;
  const n = s2.length;
  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));
  
  // dp[i][j] = s1[0..i-1] 和 s2[0..j-1] 的LCS长度
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (s1[i - 1] === s2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  
  return dp[m][n];
}
```

**状态转移方程：**
```
dp[i][j] = {
  dp[i-1][j-1] + 1,              if s1[i-1] == s2[j-1]
  max(dp[i-1][j], dp[i][j-1]),   otherwise
}
```

**时间复杂度：** O(m * n)  
**空间复杂度：** O(m * n)

### 路径还原（100分）

```javascript
/**
 * LCS（还原子序列）
 */
function longestCommonSubsequence(s1, s2) {
  const m = s1.length;
  const n = s2.length;
  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0));
  
  // 构建DP表
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (s1[i - 1] === s2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  
  // 还原路径
  const lcs = [];
  let i = m, j = n;
  
  while (i > 0 && j > 0) {
    if (s1[i - 1] === s2[j - 1]) {
      lcs.unshift(s1[i - 1]);
      i--;
      j--;
    } else if (dp[i - 1][j] > dp[i][j - 1]) {
      i--;
    } else {
      j--;
    }
  }
  
  return {
    length: dp[m][n],
    sequence: lcs.join('')
  };
}

// 测试
const result = longestCommonSubsequence("abcde", "ace");
console.log(result);  // { length: 3, sequence: 'ace' }
```

### 空间优化（滚动数组）

```javascript
/**
 * LCS（空间优化）
 */
function longestCommonSubsequence(s1, s2) {
  const m = s1.length;
  const n = s2.length;
  let dp = Array(n + 1).fill(0);
  
  for (let i = 1; i <= m; i++) {
    let prev = 0;
    
    for (let j = 1; j <= n; j++) {
      const temp = dp[j];
      
      if (s1[i - 1] === s2[j - 1]) {
        dp[j] = prev + 1;
      } else {
        dp[j] = Math.max(dp[j], dp[j - 1]);
      }
      
      prev = temp;
    }
  }
  
  return dp[n];
}
```

**空间复杂度：** O(min(m, n))

## 💻 其他经典DP题

### 1. 爬楼梯

```javascript
/**
 * 爬楼梯（LeetCode 70）
 * 每次可以爬1或2个台阶，求爬到第n阶的方法数
 */
function climbStairs(n) {
  if (n <= 2) return n;
  
  let dp1 = 1, dp2 = 2;
  
  for (let i = 3; i <= n; i++) {
    const dp = dp1 + dp2;
    dp1 = dp2;
    dp2 = dp;
  }
  
  return dp2;
}

// 测试
console.log(climbStairs(5));  // 8
```

### 2. 打家劫舍

```javascript
/**
 * 打家劫舍（LeetCode 198）
 * 不能偷相邻的房子，求最大金额
 */
function rob(nums) {
  if (nums.length === 0) return 0;
  if (nums.length === 1) return nums[0];
  
  let dp1 = 0, dp2 = nums[0];
  
  for (let i = 1; i < nums.length; i++) {
    const dp = Math.max(dp2, dp1 + nums[i]);
    dp1 = dp2;
    dp2 = dp;
  }
  
  return dp2;
}

// 测试
console.log(rob([2, 7, 9, 3, 1]));  // 12（2 + 9 + 1）
```

### 3. 最小路径和

```javascript
/**
 * 最小路径和（LeetCode 64）
 */
function minPathSum(grid) {
  const m = grid.length;
  const n = grid[0].length;
  const dp = Array.from({ length: m }, () => Array(n).fill(0));
  
  dp[0][0] = grid[0][0];
  
  // 第一行
  for (let j = 1; j < n; j++) {
    dp[0][j] = dp[0][j - 1] + grid[0][j];
  }
  
  // 第一列
  for (let i = 1; i < m; i++) {
    dp[i][0] = dp[i - 1][0] + grid[i][0];
  }
  
  // 其他
  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];
    }
  }
  
  return dp[m - 1][n - 1];
}
```

## 📊 DP解题步骤

### 1. 定义状态
```
0-1背包：dp[i][w] = 前i个物品，容量为w的最大价值
LCS：dp[i][j] = s1[0..i-1]和s2[0..j-1]的LCS长度
```

### 2. 状态转移方程
```
背包：dp[i][w] = max(dp[i-1][w], dp[i-1][w-w[i]] + v[i])
LCS：dp[i][j] = s1[i]==s2[j] ? dp[i-1][j-1]+1 : max(...)
```

### 3. 初始状态
```
背包：dp[0][w] = 0
LCS：dp[0][j] = dp[i][0] = 0
```

### 4. 计算顺序
```
从前往后（递推）
或从后往前（递归+记忆化）
```
