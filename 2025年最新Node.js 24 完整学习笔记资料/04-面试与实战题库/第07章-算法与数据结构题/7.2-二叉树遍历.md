| **层序遍历** | 逐层遍历 | 最短路径、树的层数 |

## 💻 前序遍历（根→左→右）

### 1. 递归实现

```javascript
/**
 * 前序遍历（递归）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h) - h为树高
 */
function preorderTraversal(root) {
  const result = [];
  
  function traverse(node) {
    if (!node) return;
    
    result.push(node.val);      // 根
    traverse(node.left);         // 左
    traverse(node.right);        // 右
  }
  
  traverse(root);
  return result;
}

// 测试
console.log(preorderTraversal(root));
// [1, 2, 4, 5, 3]
```

---

### 2. 迭代实现（栈）

```javascript
/**
 * 前序遍历（迭代）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h)
 */
function preorderTraversalIterative(root) {
  if (!root) return [];
  
  const result = [];
  const stack = [root];
  
  while (stack.length > 0) {
    const node = stack.pop();
    result.push(node.val);
    
    // 先压右子树（后访问）
    if (node.right) stack.push(node.right);
    // 再压左子树（先访问）
    if (node.left) stack.push(node.left);
  }
  
  return result;
}

// 测试
console.log(preorderTraversalIterative(root));
// [1, 2, 4, 5, 3]
```

## 💻 中序遍历（左→根→右）

### 1. 递归实现

```javascript
/**
 * 中序遍历（递归）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h)
 */
function inorderTraversal(root) {
  const result = [];
  
  function traverse(node) {
    if (!node) return;
    
    traverse(node.left);         // 左
    result.push(node.val);       // 根
    traverse(node.right);        // 右
  }
  
  traverse(root);
  return result;
}

// 测试
console.log(inorderTraversal(root));
// [4, 2, 5, 1, 3]
```

---

### 2. 迭代实现（栈）

```javascript
/**
 * 中序遍历（迭代）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h)
 */
function inorderTraversalIterative(root) {
  const result = [];
  const stack = [];
  let current = root;
  
  while (current || stack.length > 0) {
    // 一直往左走，将所有左子节点入栈
    while (current) {
      stack.push(current);
      current = current.left;
    }
    
    // 弹出栈顶（最左节点）
    current = stack.pop();
    result.push(current.val);
    
    // 转向右子树
    current = current.right;
  }
  
  return result;
}

// 测试
console.log(inorderTraversalIterative(root));
// [4, 2, 5, 1, 3]
```

### 3. Morris 遍历（O(1)空间）

```javascript
/**
 * 中序遍历（Morris）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function inorderMorris(root) {
  const result = [];
  let current = root;
  
  while (current) {
    if (!current.left) {
      // 没有左子树，直接访问
      result.push(current.val);
      current = current.right;
    } else {
      // 找到前驱节点（左子树的最右节点）
      let predecessor = current.left;
      
      while (predecessor.right && predecessor.right !== current) {
        predecessor = predecessor.right;
      }
      
      if (!predecessor.right) {
        // 建立线索
        predecessor.right = current;
        current = current.left;
      } else {
        // 断开线索
        predecessor.right = null;
        result.push(current.val);
        current = current.right;
      }
    }
  }
  
  return result;
}

// 测试
console.log(inorderMorris(root));
// [4, 2, 5, 1, 3]
```

## 💻 后序遍历（左→右→根）

### 1. 递归实现

```javascript
/**
 * 后序遍历（递归）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h)
 */
function postorderTraversal(root) {
  const result = [];
  
  function traverse(node) {
    if (!node) return;
    
    traverse(node.left);         // 左
    traverse(node.right);        // 右
    result.push(node.val);       // 根
  }
  
  traverse(root);
  return result;
}

// 测试
console.log(postorderTraversal(root));
// [4, 5, 2, 3, 1]
```

---

### 2. 迭代实现（双栈）

```javascript
/**
 * 后序遍历（迭代 - 双栈）
 * @param {TreeNode} root
 * @returns {number[]}
 * 时间复杂度：O(n)
 * 空间复杂度：O(h)
 */
function postorderTraversalIterative(root) {
  if (!root) return [];
  
  const stack1 = [root];
  const stack2 = [];
  
  // 前序遍历的变体：根→右→左
  while (stack1.length > 0) {
    const node = stack1.pop();
    stack2.push(node);
    
    if (node.left) stack1.push(node.left);
    if (node.right) stack1.push(node.right);
  }
  
  // 反转结果：左→右→根
  const result = [];
  
  while (stack2.length > 0) {
    result.push(stack2.pop().val);
  }
  
  return result;
}

// 测试
console.log(postorderTraversalIterative(root));
// [4, 5, 2, 3, 1]
```

### 3. 迭代实现（单栈）

```javascript
/**
 * 后序遍历（迭代 - 单栈）
 * @param {TreeNode} root
 * @returns {number[]}
 */
function postorderTraversalSingleStack(root) {
  if (!root) return [];
  
  const result = [];
  const stack = [];
  let current = root;
  let lastVisited = null;
  
  while (current || stack.length > 0) {
    // 一直往左走
    while (current) {
      stack.push(current);
      current = current.left;
    }
    
    // 查看栈顶（不弹出）
    current = stack[stack.length - 1];
    
    // 如果右子树不存在或已访问，则访问根节点
    if (!current.right || current.right === lastVisited) {
      result.push(current.val);
      lastVisited = current;
      stack.pop();
      current = null;  // 重置current
    } else {
      // 转向右子树
      current = current.right;
    }
  }
  
  return result;
}

// 测试
console.log(postorderTraversalSingleStack(root));
// [4, 5, 2, 3, 1]
```

## 💻 层序遍历（BFS）

### 1. 队列实现

```javascript
/**
 * 层序遍历
 * @param {TreeNode} root
 * @returns {number[][]} - 按层返回
 * 时间复杂度：O(n)
 * 空间复杂度：O(w) - w为树的最大宽度
 */
function levelOrder(root) {
  if (!root) return [];
  
  const result = [];
  const queue = [root];
  
  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel = [];
    
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      currentLevel.push(node.val);
      
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    
    result.push(currentLevel);
  }
  
  return result;
}

// 测试
console.log(levelOrder(root));
// [[1], [2, 3], [4, 5]]
```

### 3. 之字形层序遍历

```javascript
/**
 * 之字形层序遍历
 * @param {TreeNode} root
 * @returns {number[][]}
 */
function zigzagLevelOrder(root) {
  if (!root) return [];
  
  const result = [];
  const queue = [root];
  let leftToRight = true;
  
  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel = [];
    
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      
      if (leftToRight) {
        currentLevel.push(node.val);
      } else {
        currentLevel.unshift(node.val);  // 反向
      }
      
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    
    result.push(currentLevel);
    leftToRight = !leftToRight;
  }
  
  return result;
}

// 测试
console.log(zigzagLevelOrder(root));
// [[1], [3, 2], [4, 5]]
```

## 💻 实战应用

### 1. 验证二叉搜索树（中序遍历）

```javascript
/**
 * 验证BST
 * @param {TreeNode} root
 * @returns {boolean}
 */
function isValidBST(root) {
  let prev = null;
  
  function inorder(node) {
    if (!node) return true;
    
    // 左子树
    if (!inorder(node.left)) return false;
    
    // 检查顺序
    if (prev !== null && node.val <= prev) {
      return false;
    }
    
    prev = node.val;
    
    // 右子树
    return inorder(node.right);
  }
  
  return inorder(root);
}

// 测试
const bst = new TreeNode(2,
  new TreeNode(1),
  new TreeNode(3)
);
console.log(isValidBST(bst));  // true
```

### 2. 二叉树的最大深度（后序遍历）

```javascript
/**
 * 最大深度
 * @param {TreeNode} root
 * @returns {number}
 */
function maxDepth(root) {
  if (!root) return 0;
  
  const leftDepth = maxDepth(root.left);
  const rightDepth = maxDepth(root.right);
  
  return Math.max(leftDepth, rightDepth) + 1;
}

// 测试
console.log(maxDepth(root));  // 3
```

### 3. 二叉树的最小深度（层序遍历）

```javascript
/**
 * 最小深度
 * @param {TreeNode} root
 * @returns {number}
 */
function minDepth(root) {
  if (!root) return 0;
  
  const queue = [{ node: root, depth: 1 }];
  
  while (queue.length > 0) {
    const { node, depth } = queue.shift();
    
    // 叶子节点
    if (!node.left && !node.right) {
      return depth;
    }
    
    if (node.left) queue.push({ node: node.left, depth: depth + 1 });
    if (node.right) queue.push({ node: node.right, depth: depth + 1 });
  }
  
  return 0;
}

// 测试
console.log(minDepth(root));  // 2
```

### 4. 路径总和（前序遍历）

```javascript
/**
 * 路径总和
 * @param {TreeNode} root
 * @param {number} targetSum
 * @returns {boolean}
 */
function hasPathSum(root, targetSum) {
  if (!root) return false;
  
  // 叶子节点
  if (!root.left && !root.right) {
    return root.val === targetSum;
  }
  
  const remaining = targetSum - root.val;
  
  return hasPathSum(root.left, remaining) || 
         hasPathSum(root.right, remaining);
}

// 测试
const tree = new TreeNode(5,
  new TreeNode(4,
    new TreeNode(11,
      new TreeNode(7),
      new TreeNode(2)
    )
  ),
  new TreeNode(8,
    new TreeNode(13),
    new TreeNode(4,
      null,
      new TreeNode(1)
    )
  )
);
console.log(hasPathSum(tree, 22));  // true (5→4→11→2)
```

## 📝 总结

### 核心要点
1. **前序**：根→左→右（复制树）
2. **中序**：左→根→右（BST有序）
3. **后序**：左→右→根（删除树）
4. **层序**：BFS（最短路径）

### 实现方式
- 递归：简洁，空间O(h)
- 迭代：栈/队列，空间O(h/w)
- Morris：O(1)空间（中序）
