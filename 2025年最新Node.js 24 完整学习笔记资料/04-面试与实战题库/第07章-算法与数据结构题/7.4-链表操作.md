## 💻 链表节点定义

```javascript
/**
 * 单链表节点
 */
class ListNode {
  constructor(val, next = null) {
    this.val = val;
    this.next = next;
  }
}

// 创建链表：1 → 2 → 3 → 4 → 5
const head = new ListNode(1,
  new ListNode(2,
    new ListNode(3,
      new ListNode(4,
        new ListNode(5)
      )
    )
  )
);

// 辅助函数：打印链表
function printList(head) {
  const values = [];
  let current = head;
  
  while (current) {
    values.push(current.val);
    current = current.next;
  }
  
  console.log(values.join(' → '));
}

printList(head);  // 1 → 2 → 3 → 4 → 5
```

## 💻 反转链表

### 1. 迭代实现

```javascript
/**
 * 反转链表（迭代）
 * @param {ListNode} head
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function reverseList(head) {
  let prev = null;
  let current = head;
  
  while (current) {
    const next = current.next;  // 保存下一个节点
    current.next = prev;        // 反转指针
    prev = current;             // 移动prev
    current = next;             // 移动current
  }
  
  return prev;  // 新的头节点
}

// 测试
const list1 = new ListNode(1, new ListNode(2, new ListNode(3)));
printList(reverseList(list1));  // 3 → 2 → 1
```

---

### 2. 递归实现

```javascript
/**
 * 反转链表（递归）
 * @param {ListNode} head
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(n) - 递归栈
 */
function reverseListRecursive(head) {
  // 边界条件
  if (!head || !head.next) {
    return head;
  }
  
  // 递归反转后面的链表
  const newHead = reverseListRecursive(head.next);
  
  // 反转当前节点
  head.next.next = head;
  head.next = null;
  
  return newHead;
}

// 测试
const list2 = new ListNode(1, new ListNode(2, new ListNode(3)));
printList(reverseListRecursive(list2));  // 3 → 2 → 1
```

### 3. 反转前N个节点

```javascript
/**
 * 反转前N个节点
 * @param {ListNode} head
 * @param {number} n
 * @returns {ListNode}
 */
function reverseN(head, n) {
  if (n === 1) {
    return head;
  }
  
  const newHead = reverseN(head.next, n - 1);
  
  const nextNode = head.next.next;
  head.next.next = head;
  head.next = nextNode;
  
  return newHead;
}

// 测试：反转前3个
const list3 = new ListNode(1, new ListNode(2, new ListNode(3, new ListNode(4, new ListNode(5)))));
printList(reverseN(list3, 3));  // 3 → 2 → 1 → 4 → 5
```

### 4. 反转区间[m, n]

```javascript
/**
 * 反转链表II（区间反转）
 * @param {ListNode} head
 * @param {number} m
 * @param {number} n
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function reverseBetween(head, m, n) {
  if (m === 1) {
    return reverseN(head, n);
  }
  
  head.next = reverseBetween(head.next, m - 1, n - 1);
  
  return head;
}

// 测试：反转[2, 4]
const list4 = new ListNode(1, new ListNode(2, new ListNode(3, new ListNode(4, new ListNode(5)))));
printList(reverseBetween(list4, 2, 4));  // 1 → 4 → 3 → 2 → 5
```

## 💻 环检测

### 1. 检测环（快慢指针）

```javascript
/**
 * 检测环
 * @param {ListNode} head
 * @returns {boolean}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function hasCycle(head) {
  let slow = head;
  let fast = head;
  
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    
    if (slow === fast) {
      return true;  // 有环
    }
  }
  
  return false;  // 无环
}

// 测试：创建带环链表
const cycleNode = new ListNode(3);
const cycleList = new ListNode(1,
  new ListNode(2, cycleNode)
);
cycleNode.next = new ListNode(4, cycleList.next);

console.log(hasCycle(cycleList));  // true
```

---

### 2. 找环的入口

```javascript
/**
 * 找环的入口
 * @param {ListNode} head
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function detectCycle(head) {
  let slow = head;
  let fast = head;
  
  // 1. 检测环
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    
    if (slow === fast) {
      // 2. 找入口
      let ptr = head;
      
      while (ptr !== slow) {
        ptr = ptr.next;
        slow = slow.next;
      }
      
      return ptr;
    }
  }
  
  return null;
}

// 测试
console.log(detectCycle(cycleList));  // 返回环入口节点
```

## 💻 合并链表

### 1. 合并两个有序链表

```javascript
/**
 * 合并两个有序链表
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @returns {ListNode}
 * 时间复杂度：O(m + n)
 * 空间复杂度：O(1)
 */
function mergeTwoLists(l1, l2) {
  const dummy = new ListNode(0);
  let current = dummy;
  
  while (l1 && l2) {
    if (l1.val <= l2.val) {
      current.next = l1;
      l1 = l1.next;
    } else {
      current.next = l2;
      l2 = l2.next;
    }
    
    current = current.next;
  }
  
  // 连接剩余节点
  current.next = l1 || l2;
  
  return dummy.next;
}

// 测试
const l1 = new ListNode(1, new ListNode(3, new ListNode(5)));
const l2 = new ListNode(2, new ListNode(4, new ListNode(6)));
printList(mergeTwoLists(l1, l2));  // 1 → 2 → 3 → 4 → 5 → 6
```

---

### 2. 递归实现

```javascript
/**
 * 合并两个有序链表（递归）
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @returns {ListNode}
 * 时间复杂度：O(m + n)
 * 空间复杂度：O(m + n) - 递归栈
 */
function mergeTwoListsRecursive(l1, l2) {
  if (!l1) return l2;
  if (!l2) return l1;
  
  if (l1.val <= l2.val) {
    l1.next = mergeTwoListsRecursive(l1.next, l2);
    return l1;
  } else {
    l2.next = mergeTwoListsRecursive(l1, l2.next);
    return l2;
  }
}

// 测试
const l3 = new ListNode(1, new ListNode(3, new ListNode(5)));
const l4 = new ListNode(2, new ListNode(4, new ListNode(6)));
printList(mergeTwoListsRecursive(l3, l4));  // 1 → 2 → 3 → 4 → 5 → 6
```

## 💻 删除节点

### 1. 删除倒数第N个节点

```javascript
/**
 * 删除倒数第N个节点
 * @param {ListNode} head
 * @param {number} n
 * @returns {ListNode}
 * 时间复杂度：O(L)
 * 空间复杂度：O(1)
 */
function removeNthFromEnd(head, n) {
  const dummy = new ListNode(0, head);
  let fast = dummy;
  let slow = dummy;
  
  // 1. fast先走n+1步
  for (let i = 0; i <= n; i++) {
    fast = fast.next;
  }
  
  // 2. fast和slow一起走
  while (fast) {
    fast = fast.next;
    slow = slow.next;
  }
  
  // 3. 删除倒数第n个节点
  slow.next = slow.next.next;
  
  return dummy.next;
}

// 测试
const list5 = new ListNode(1, new ListNode(2, new ListNode(3, new ListNode(4, new ListNode(5)))));
printList(removeNthFromEnd(list5, 2));  // 1 → 2 → 3 → 5
```

---

### 2. 删除重复元素

```javascript
/**
 * 删除排序链表中的重复元素
 * @param {ListNode} head
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function deleteDuplicates(head) {
  let current = head;
  
  while (current && current.next) {
    if (current.val === current.next.val) {
      current.next = current.next.next;
    } else {
      current = current.next;
    }
  }
  
  return head;
}

// 测试
const list6 = new ListNode(1, new ListNode(1, new ListNode(2, new ListNode(3, new ListNode(3)))));
printList(deleteDuplicates(list6));  // 1 → 2 → 3
```

## 💻 其他操作

### 1. 找中间节点

```javascript
/**
 * 找链表中间节点
 * @param {ListNode} head
 * @returns {ListNode}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function middleNode(head) {
  let slow = head;
  let fast = head;
  
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  
  return slow;  // 中间节点
}

// 测试
const list7 = new ListNode(1, new ListNode(2, new ListNode(3, new ListNode(4, new ListNode(5)))));
console.log(middleNode(list7).val);  // 3
```

---

### 2. 判断回文链表

```javascript
/**
 * 判断回文链表
 * @param {ListNode} head
 * @returns {boolean}
 * 时间复杂度：O(n)
 * 空间复杂度：O(1)
 */
function isPalindrome(head) {
  if (!head || !head.next) return true;
  
  // 1. 找中间节点
  let slow = head;
  let fast = head;
  
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  
  // 2. 反转后半部分
  let secondHalf = reverseList(slow);
  
  // 3. 比较
  let p1 = head;
  let p2 = secondHalf;
  
  while (p2) {
    if (p1.val !== p2.val) {
      return false;
    }
    
    p1 = p1.next;
    p2 = p2.next;
  }
  
  return true;
}

// 测试
const palindrome = new ListNode(1, new ListNode(2, new ListNode(2, new ListNode(1))));
console.log(isPalindrome(palindrome));  // true
```

### 3. 相交链表

```javascript
/**
 * 找两个链表的交点
 * @param {ListNode} headA
 * @param {ListNode} headB
 * @returns {ListNode}
 * 时间复杂度：O(m + n)
 * 空间复杂度：O(1)
 */
function getIntersectionNode(headA, headB) {
  if (!headA || !headB) return null;
  
  let pA = headA;
  let pB = headB;
  
  // 两个指针走相同的路程
  while (pA !== pB) {
    pA = pA ? pA.next : headB;
    pB = pB ? pB.next : headA;
  }
  
  return pA;  // 交点或null
}

// 测试
const common = new ListNode(8, new ListNode(4, new ListNode(5)));
const listA = new ListNode(4, new ListNode(1, common));
const listB = new ListNode(5, new ListNode(6, new ListNode(1, common)));

console.log(getIntersectionNode(listA, listB));  // 返回节点8
```

## 💻 双指针技巧总结

### 快慢指针

```javascript
// 模式：快指针走2步，慢指针走1步
let slow = head;
let fast = head;

while (fast && fast.next) {
  slow = slow.next;
  fast = fast.next.next;
}

// 应用：
// 1. 找中间节点：slow即为中间节点
// 2. 检测环：slow === fast则有环
// 3. 找倒数第k个：fast先走k步
```

---

### 双指针间隔

```javascript
// 模式：两个指针间隔n
let first = head;
let second = head;

// first先走n步
for (let i = 0; i < n; i++) {
  first = first.next;
}

// 一起走
while (first) {
  first = first.next;
  second = second.next;
}

// 应用：
// 1. 删除倒数第n个：second.next为目标
// 2. 找倒数第n个：second即为目标
```

## 🎯 面试标准答题

### 问题："反转链表"

**标准答案（15-20分钟）：**

**1. 算法思路（3分钟）**
> "反转链表需要：  
> 1. 保存next节点（避免丢失）  
> 2. 反转current.next指针  
> 3. 移动prev和current"

**2. 代码实现（10分钟）**
```javascript
function reverseList(head) {
  let prev = null;
  let current = head;
  
  while (current) {
    const next = current.next;  // 保存
    current.next = prev;        // 反转
    prev = current;             // 移动
    current = next;
  }
  
  return prev;
}
```

**3. 复杂度分析（2分钟）**
> "时间复杂度：O(n) - 遍历一次  
> 空间复杂度：O(1) - 只用三个指针"

**4. 递归实现（3分钟）**
```javascript
function reverseListRecursive(head) {
  if (!head || !head.next) return head;
  
  const newHead = reverseListRecursive(head.next);
  
  head.next.next = head;
  head.next = null;
  
  return newHead;
}
```

**5. 边界情况（2分钟）**
> "需要考虑：  
> 1. 空链表：返回null  
> 2. 单节点：返回自身  
> 3. 两个节点：正常反转"

## 📝 总结

### 核心要点
1. **反转链表**：prev、current、next三指针
2. **环检测**：快慢指针
3. **删除节点**：双指针间隔
4. **合并链表**：双指针同步

### 双指针技巧
- 快慢指针 → 找中间、检测环
- 间隔指针 → 删除倒数第n个
- 同步指针 → 合并、相交
