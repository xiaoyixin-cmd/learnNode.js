## 💻 O(n²) 排序算法

### 1. 冒泡排序

#### 原理
相邻元素两两比较，大的往后移，每轮确定一个最大值。

#### 实现

```javascript
/**
 * 冒泡排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n²)
 * 空间复杂度：O(1)
 * 稳定性：✅
 */
function bubbleSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const n = arr.length;
  
  // 外层循环：需要n-1轮
  for (let i = 0; i < n - 1; i++) {
    let swapped = false;  // 优化：检测是否发生交换
    
    // 内层循环：每轮比较n-i-1次
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        // 交换
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }
    
    // 优化：如果某轮没有交换，说明已有序
    if (!swapped) break;
  }
  
  return arr;
}

// 测试
console.log(bubbleSort([64, 34, 25, 12, 22, 11, 90]));
// [11, 12, 22, 25, 34, 64, 90]
```

#### 复杂度分析

```javascript
// 最好情况：O(n) - 数组已有序
// 最坏情况：O(n²) - 数组逆序
// 平均情况：O(n²)

// 空间复杂度：O(1) - 原地排序

// 稳定性：✅ - 相等元素不交换
```

### 2. 选择排序

#### 原理
每轮选择未排序部分的最小值，放到已排序部分的末尾。

#### 实现

```javascript
/**
 * 选择排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n²)
 * 空间复杂度：O(1)
 * 稳定性：❌
 */
function selectionSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const n = arr.length;
  
  for (let i = 0; i < n - 1; i++) {
    let minIndex = i;
    
    // 找到未排序部分的最小值
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[minIndex]) {
        minIndex = j;
      }
    }
    
    // 交换
    if (minIndex !== i) {
      [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
    }
  }
  
  return arr;
}

// 测试
console.log(selectionSort([64, 25, 12, 22, 11]));
// [11, 12, 22, 25, 64]
```

#### 不稳定性示例

```javascript
// 为什么选择排序不稳定？
const arr = [5, 8, 5, 2, 9];
//           ↑     ↑
//          第1个5 第2个5

selectionSort(arr);
// [2, 5, 5, 8, 9]
//      ↑  ↑
//   第2个5在前面了！
```

### 3. 插入排序

#### 原理
类似整理扑克牌，将未排序元素插入到已排序部分的正确位置。

#### 实现

```javascript
/**
 * 插入排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n²)
 * 空间复杂度：O(1)
 * 稳定性：✅
 */
function insertionSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const n = arr.length;
  
  for (let i = 1; i < n; i++) {
    const key = arr[i];
    let j = i - 1;
    
    // 将大于key的元素向后移动
    while (j >= 0 && arr[j] > key) {
      arr[j + 1] = arr[j];
      j--;
    }
    
    // 插入key
    arr[j + 1] = key;
  }
  
  return arr;
}

// 测试
console.log(insertionSort([12, 11, 13, 5, 6]));
// [5, 6, 11, 12, 13]
```

#### 优化：二分插入排序

```javascript
/**
 * 二分插入排序
 * 优化：用二分查找插入位置
 */
function binaryInsertionSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const n = arr.length;
  
  for (let i = 1; i < n; i++) {
    const key = arr[i];
    
    // 二分查找插入位置
    let left = 0;
    let right = i - 1;
    
    while (left <= right) {
      const mid = Math.floor((left + right) / 2);
      
      if (arr[mid] > key) {
        right = mid - 1;
      } else {
        left = mid + 1;
      }
    }
    
    // 移动元素
    for (let j = i - 1; j >= left; j--) {
      arr[j + 1] = arr[j];
    }
    
    // 插入
    arr[left] = key;
  }
  
  return arr;
}
```

## 💻 O(n log n) 排序算法

### 1. 快速排序

#### 原理
分治思想：选择pivot，将小于pivot的放左边，大于的放右边，递归排序。

#### 实现

```javascript
/**
 * 快速排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n log n)
 * 空间复杂度：O(log n)
 * 稳定性：❌
 */
function quickSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  return quickSortHelper(arr, 0, arr.length - 1);
}

function quickSortHelper(arr, left, right) {
  if (left >= right) return arr;
  
  // 分区
  const pivotIndex = partition(arr, left, right);
  
  // 递归排序左右两部分
  quickSortHelper(arr, left, pivotIndex - 1);
  quickSortHelper(arr, pivotIndex + 1, right);
  
  return arr;
}

function partition(arr, left, right) {
  // 选择最右边元素作为pivot
  const pivot = arr[right];
  let i = left - 1;
  
  for (let j = left; j < right; j++) {
    if (arr[j] <= pivot) {
      i++;
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
  }
  
  // 将pivot放到正确位置
  [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
  
  return i + 1;
}

// 测试
console.log(quickSort([10, 7, 8, 9, 1, 5]));
// [1, 5, 7, 8, 9, 10]
```

#### 优化：三路快排

```javascript
/**
 * 三路快排
 * 优化：处理大量重复元素
 */
function quickSort3Way(arr, left = 0, right = arr.length - 1) {
  if (left >= right) return arr;
  
  const pivot = arr[left];
  let lt = left;      // arr[left+1...lt] < pivot
  let gt = right;     // arr[gt...right] > pivot
  let i = left + 1;   // arr[lt+1...i-1] == pivot
  
  while (i <= gt) {
    if (arr[i] < pivot) {
      [arr[lt + 1], arr[i]] = [arr[i], arr[lt + 1]];
      lt++;
      i++;
    } else if (arr[i] > pivot) {
      [arr[i], arr[gt]] = [arr[gt], arr[i]];
      gt--;
    } else {
      i++;
    }
  }
  
  [arr[left], arr[lt]] = [arr[lt], arr[left]];
  
  quickSort3Way(arr, left, lt - 1);
  quickSort3Way(arr, gt + 1, right);
  
  return arr;
}

// 测试：大量重复元素
console.log(quickSort3Way([3, 1, 2, 3, 3, 1, 2]));
// [1, 1, 2, 2, 3, 3, 3]
```

### 2. 归并排序

#### 原理
分治思想：将数组分成两半，递归排序，然后合并。

#### 实现

```javascript
/**
 * 归并排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n log n)
 * 空间复杂度：O(n)
 * 稳定性：✅
 */
function mergeSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));
  
  return merge(left, right);
}

function merge(left, right) {
  const result = [];
  let i = 0;
  let j = 0;
  
  // 合并两个有序数组
  while (i < left.length && j < right.length) {
    if (left[i] <= right[j]) {  // <=保证稳定性
      result.push(left[i]);
      i++;
    } else {
      result.push(right[j]);
      j++;
    }
  }
  
  // 添加剩余元素
  return result.concat(left.slice(i)).concat(right.slice(j));
}

// 测试
console.log(mergeSort([38, 27, 43, 3, 9, 82, 10]));
// [3, 9, 10, 27, 38, 43, 82]
```

#### 优化：原地归并

```javascript
/**
 * 原地归并排序
 * 优化：减少空间复杂度
 */
function mergeSortInPlace(arr, left = 0, right = arr.length - 1) {
  if (left >= right) return arr;
  
  const mid = Math.floor((left + right) / 2);
  
  mergeSortInPlace(arr, left, mid);
  mergeSortInPlace(arr, mid + 1, right);
  
  mergeInPlace(arr, left, mid, right);
  
  return arr;
}

function mergeInPlace(arr, left, mid, right) {
  const temp = [];
  let i = left;
  let j = mid + 1;
  
  while (i <= mid && j <= right) {
    if (arr[i] <= arr[j]) {
      temp.push(arr[i++]);
    } else {
      temp.push(arr[j++]);
    }
  }
  
  while (i <= mid) temp.push(arr[i++]);
  while (j <= right) temp.push(arr[j++]);
  
  for (let k = 0; k < temp.length; k++) {
    arr[left + k] = temp[k];
  }
}
```

### 3. 堆排序

#### 原理
利用堆的性质：父节点 >= 子节点（大顶堆）。

#### 实现

```javascript
/**
 * 堆排序
 * @param {number[]} arr - 待排序数组
 * @returns {number[]} 排序后数组
 * 时间复杂度：O(n log n)
 * 空间复杂度：O(1)
 * 稳定性：❌
 */
function heapSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const n = arr.length;
  
  // 1. 建堆（从最后一个非叶子节点开始）
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }
  
  // 2. 排序（每次将堆顶元素与末尾元素交换）
  for (let i = n - 1; i > 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]];
    heapify(arr, i, 0);
  }
  
  return arr;
}

function heapify(arr, n, i) {
  let largest = i;
  const left = 2 * i + 1;
  const right = 2 * i + 2;
  
  if (left < n && arr[left] > arr[largest]) {
    largest = left;
  }
  
  if (right < n && arr[right] > arr[largest]) {
    largest = right;
  }
  
  if (largest !== i) {
    [arr[i], arr[largest]] = [arr[largest], arr[i]];
    heapify(arr, n, largest);
  }
}

// 测试
console.log(heapSort([12, 11, 13, 5, 6, 7]));
// [5, 6, 7, 11, 12, 13]
```

## 💻 特殊排序算法

### 1. 计数排序

```javascript
/**
 * 计数排序
 * 适用场景：数据范围不大（如0-100）
 * 时间复杂度：O(n + k)
 * 空间复杂度：O(k)
 */
function countingSort(arr) {
  if (!arr || arr.length <= 1) return arr;
  
  const max = Math.max(...arr);
  const min = Math.min(...arr);
  const range = max - min + 1;
  
  // 计数数组
  const count = new Array(range).fill(0);
  
  // 统计每个元素出现次数
  for (const num of arr) {
    count[num - min]++;
  }
  
  // 累加
  for (let i = 1; i < range; i++) {
    count[i] += count[i - 1];
  }
  
  // 输出
  const output = new Array(arr.length);
  
  for (let i = arr.length - 1; i >= 0; i--) {
    const num = arr[i];
    const index = count[num - min] - 1;
    output[index] = num;
    count[num - min]--;
  }
  
  return output;
}

// 测试
console.log(countingSort([4, 2, 2, 8, 3, 3, 1]));
// [1, 2, 2, 3, 3, 4, 8]
```

### 2. 桶排序

```javascript
/**
 * 桶排序
 * 适用场景：均匀分布的数据
 * 时间复杂度：O(n)
 * 空间复杂度：O(n + k)
 */
function bucketSort(arr, bucketSize = 5) {
  if (!arr || arr.length <= 1) return arr;
  
  const max = Math.max(...arr);
  const min = Math.min(...arr);
  
  // 桶的数量
  const bucketCount = Math.floor((max - min) / bucketSize) + 1;
  const buckets = Array.from({ length: bucketCount }, () => []);
  
  // 分配到桶
  for (const num of arr) {
    const bucketIndex = Math.floor((num - min) / bucketSize);
    buckets[bucketIndex].push(num);
  }
  
  // 对每个桶排序，然后合并
  const result = [];
  
  for (const bucket of buckets) {
    if (bucket.length > 0) {
      insertionSort(bucket);  // 桶内用插入排序
      result.push(...bucket);
    }
  }
  
  return result;
}

// 测试
console.log(bucketSort([29, 25, 3, 49, 9, 37, 21, 43]));
// [3, 9, 21, 25, 29, 37, 43, 49]
```

## 🎯 面试标准答题

### 问题："实现快速排序"

**标准答案（15-20分钟）：**

**1. 算法原理（3分钟）**
> "快速排序是分治算法：  
> 1. 选择pivot（通常选最右边元素）  
> 2. 分区：小于pivot的放左边，大于的放右边  
> 3. 递归排序左右两部分"

**2. 代码实现（10分钟）**
```javascript
function quickSort(arr, left = 0, right = arr.length - 1) {
  if (left >= right) return arr;
  
  const pivotIndex = partition(arr, left, right);
  
  quickSort(arr, left, pivotIndex - 1);
  quickSort(arr, pivotIndex + 1, right);
  
  return arr;
}

function partition(arr, left, right) {
  const pivot = arr[right];
  let i = left - 1;
  
  for (let j = left; j < right; j++) {
    if (arr[j] <= pivot) {
      i++;
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
  }
  
  [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
  return i + 1;
}
```

**3. 复杂度分析（3分钟）**
> "时间复杂度：  
> - 最好/平均：O(n log n)  
> - 最坏：O(n²)（数组已有序）  
>   
> 空间复杂度：O(log n)（递归栈）  
>   
> 稳定性：❌（分区时可能改变相等元素顺序）"

**4. 优化方向（2分钟）**
> "优化方法：  
> 1. 随机选择pivot（避免最坏情况）  
> 2. 三路快排（处理大量重复元素）  
> 3. 小数组用插入排序  
> 4. 尾递归优化"

## 📝 总结

### 核心要点
1. **O(n²)**：冒泡、选择、插入
2. **O(n log n)**：快排、归并、堆排
3. **特殊**：计数排序、桶排序

### 选择建议
- 通用 → 快速排序
- 稳定 → 归并排序
- 节省空间 → 堆排序
- 小数据 → 插入排序
- 范围小 → 计数排序
