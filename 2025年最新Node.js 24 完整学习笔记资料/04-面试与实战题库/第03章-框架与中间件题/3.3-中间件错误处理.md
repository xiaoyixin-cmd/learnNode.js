## 📚 核心知识点

### 1. 同步错误 vs 异步错误

**同步错误：**
```javascript
// 可以被 try/catch 捕获
try {
  throw new Error('Sync error');
} catch (err) {
  console.error(err.message);  // "Sync error"
}
```

**异步错误：**
```javascript
// ❌ try/catch 捕获不了（错误在下一个 tick 抛出）
try {
  setTimeout(() => {
    throw new Error('Async error');
  }, 0);
} catch (err) {
  console.error(err.message);  // 捕获不到
}

// ✅ 使用 Promise 或 async/await
async function test() {
  try {
    await new Promise((resolve, reject) => {
      setTimeout(() => reject(new Error('Async error')), 0);
    });
  } catch (err) {
    console.error(err.message);  // "Async error"
  }
}
```

---

### 2. Express 错误处理机制

#### 同步错误（可自动捕获）
```javascript
const express = require('express');
const app = express();

// 同步错误（Express 自动捕获）
app.get('/sync-error', (req, res) => {
  throw new Error('Sync error');  // ✅ 会被错误处理中间件捕获
});

// 错误处理中间件（4个参数）
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});

app.listen(3000);
```

---

#### 异步错误（需手动传递）
```javascript
// ❌ 异步错误（Express 捕获不了）
app.get('/async-error', (req, res) => {
  setTimeout(() => {
    throw new Error('Async error');  // 未捕获，导致进程崩溃
  }, 0);
});

// ✅ 手动传递给 next()
app.get('/async-error', (req, res, next) => {
  setTimeout(() => {
    next(new Error('Async error'));  // 传递给错误处理中间件
  }, 0);
});

// ✅ 使用 Promise
app.get('/async-error', (req, res, next) => {
  Promise.reject(new Error('Async error'))
    .catch(next);  // 捕获并传递
});

// ✅ 使用 async/await（推荐）
app.get('/async-error', async (req, res, next) => {
  try {
    await someAsyncFunction();
  } catch (err) {
    next(err);
  }
});
```

### 3. Koa 错误处理机制

**优势：自动捕获同步和异步错误**
```javascript
const Koa = require('koa');
const app = new Koa();

// 全局错误处理中间件（放在最前面）
app.use(async (ctx, next) => {
  try {
    await next();  // ✅ 自动捕获所有后续中间件的错误
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = { error: err.message };
    ctx.app.emit('error', err, ctx);
  }
});

// 同步错误（自动捕获）
app.use(async ctx => {
  throw new Error('Sync error');  // ✅ 被上面的 try/catch 捕获
});

// 异步错误（自动捕获）
app.use(async ctx => {
  await Promise.reject(new Error('Async error'));  // ✅ 被捕获
});

app.listen(3000);
```

## 💻 Express 错误处理详解

### 1. 同步错误处理

```javascript
const express = require('express');
const app = express();

// 路由（同步错误）
app.get('/users/:id', (req, res) => {
  const user = db.users.findById(req.params.id);
  
  if (!user) {
    throw new Error('User not found');  // ✅ 自动传递给错误处理中间件
  }
  
  res.json(user);
});

// 错误处理中间件（必须 4 个参数）
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ error: err.message });
});

app.listen(3000);
```

---

### 2. 异步错误处理（包装器）

```javascript
// 异步包装器（自动捕获错误）
function asyncHandler(fn) {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
}

// 使用
app.get('/users/:id', asyncHandler(async (req, res) => {
  const user = await db.users.findById(req.params.id);
  
  if (!user) {
    throw new Error('User not found');  // ✅ 自动捕获并传递给 next()
  }
  
  res.json(user);
}));

// 错误处理中间件
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});
```

---

### 3. express-async-errors（自动捕获）

```javascript
// 安装：npm install express-async-errors
require('express-async-errors');  // 导入后自动生效

const express = require('express');
const app = express();

// 异步路由（自动捕获错误）
app.get('/users/:id', async (req, res) => {
  const user = await db.users.findById(req.params.id);
  
  if (!user) {
    throw new Error('User not found');  // ✅ 自动捕获
  }
  
  res.json(user);
});

// 错误处理中间件
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});
```

## 💻 Koa 错误处理详解

### 1. 全局错误处理

```javascript
const Koa = require('koa');
const app = new Koa();

// 方式 1：try/catch（推荐）
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      error: err.message,
      stack: process.env.NODE_ENV === 'development' ? err.stack : undefined
    };
    
    // 触发 'error' 事件（用于日志记录）
    ctx.app.emit('error', err, ctx);
  }
});

// 方式 2：监听 'error' 事件
app.on('error', (err, ctx) => {
  console.error('Server error:', err);
  // 记录到日志系统
});

// 业务逻辑
app.use(async ctx => {
  if (!ctx.query.name) {
    ctx.throw(400, 'Name is required');
  }
  ctx.body = { name: ctx.query.name };
});

app.listen(3000);
```

### 2. 自定义错误类

```javascript
// 自定义错误类
class AppError extends Error {
  constructor(message, status = 500) {
    super(message);
    this.status = status;
    this.name = this.constructor.name;
  }
}

class ValidationError extends AppError {
  constructor(message) {
    super(message, 400);
  }
}

class NotFoundError extends AppError {
  constructor(message) {
    super(message, 404);
  }
}

// 使用
app.use(async ctx => {
  const user = await db.users.findById(ctx.params.id);
  
  if (!user) {
    throw new NotFoundError('User not found');
  }
  
  ctx.body = user;
});

// 错误处理
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    if (err instanceof AppError) {
      ctx.status = err.status;
      ctx.body = { error: err.message };
    } else {
      ctx.status = 500;
      ctx.body = { error: 'Internal Server Error' };
    }
  }
});
```

## 💻 实战案例

### 案例 1：统一错误响应格式

```javascript
const Koa = require('koa');
const app = new Koa();

// 错误处理中间件
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    const status = err.status || 500;
    const message = err.message || 'Internal Server Error';
    
    // 统一响应格式
    ctx.status = status;
    ctx.body = {
      success: false,
      error: {
        code: err.code || 'INTERNAL_ERROR',
        message,
        details: process.env.NODE_ENV === 'development' ? err.stack : undefined
      },
      timestamp: new Date().toISOString()
    };
    
    // 记录日志
    console.error(`[${status}] ${message}`, {
      url: ctx.url,
      method: ctx.method,
      ip: ctx.ip,
      stack: err.stack
    });
  }
});

// 业务逻辑
app.use(async ctx => {
  if (ctx.url === '/error') {
    const err = new Error('Something went wrong');
    err.code = 'CUSTOM_ERROR';
    throw err;
  }
  
  ctx.body = { success: true, data: { msg: 'Hello' } };
});

app.listen(3000);
```

const Koa = require('koa');
const winston = require('winston');

const app = new Koa();

// 配置日志
const logger = winston.createLogger({

### 案例 3：错误通知（Slack/邮件）

```javascript
const Koa = require('koa');
const axios = require('axios');

const app = new Koa();

// 发送 Slack 通知
async function notifySlack(err, ctx) {
  const message = {
    text: `🚨 Server Error: ${err.message}`,
    attachments: [{
      fields: [
        { title: 'URL', value: ctx.url },
        { title: 'Method', value: ctx.method },
        { title: 'IP', value: ctx.ip },
        { title: 'Stack', value: err.stack.substring(0, 500) }
      ]
    }]
  };
  
  await axios.post(process.env.SLACK_WEBHOOK_URL, message);
}

// 错误处理 + 通知
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    // 严重错误通知（5xx）
    if (err.status >= 500) {
      await notifySlack(err, ctx);
    }
    
    ctx.status = err.status || 500;
    ctx.body = { error: err.message };
  }
});

app.listen(3000);
```

## 🎓 面试标准答题

### 问题 1："Express 和 Koa 的错误处理机制有什么区别？"

**标准答案：**
> **核心区别：自动捕获 vs 手动传递**
> 
> **Express：**
> - ✅ **同步错误**：自动捕获
> - ❌ **异步错误**：需手动传递给 `next(err)`
> 
> ```javascript
> // Express
> app.get('/async', (req, res, next) => {
>   Promise.reject(new Error('Async error'))
>     .catch(next);  // 手动传递
> });
> ```
> 
> **Koa：**
> - ✅ **同步错误**：自动捕获
> - ✅ **异步错误**：自动捕获（洋葱模型 + async/await）
> 
> ```javascript
> // Koa
> app.use(async (ctx, next) => {
>   try {
>     await next();  // 自动捕获所有错误
>   } catch (err) {
>     ctx.status = 500;
>     ctx.body = { error: err.message };
>   }
> });
> ```
> 
> **为什么有区别？**
> - Express：基于回调，`next()` 不返回 Promise
> - Koa：基于 async/await，`await next()` 自动捕获 Promise rejection
> 
> **最佳实践：**
> - Express：使用 `express-async-errors` 或 `asyncHandler` 包装器
> - Koa：在最外层中间件使用 `try/catch`

**加分项：**
> 在实际项目中，我们使用 Koa 的洋葱模型，在最外层统一处理错误，配合自定义错误类，实现了错误分类、日志记录、Slack 通知的完整错误处理体系。

---

### 问题 2："如何捕获异步中间件的错误？"

**标准答案：**
> **方案 1：async/await + try/catch（推荐）**
> ```javascript
> app.use(async (ctx, next) => {
>   try {
>     await next();
>   } catch (err) {
>     ctx.status = 500;
>     ctx.body = { error: err.message };
>   }
> });
> ```
> 
> **方案 2：Promise.catch()**
> ```javascript
> app.get('/users', (req, res, next) => {
>   getUsers()
>     .then(users => res.json(users))
>     .catch(next);  // 传递给错误处理中间件
> });
> ```
> 
> **方案 3：包装器函数**
> ```javascript
> function asyncHandler(fn) {
>   return (req, res, next) => {
>     Promise.resolve(fn(req, res, next)).catch(next);
>   };
> }
> 
> app.get('/users', asyncHandler(async (req, res) => {
>   const users = await getUsers();
>   res.json(users);
> }));
> ```
> 
> **方案 4：express-async-errors（自动）**
> ```javascript
> require('express-async-errors');
> 
> app.get('/users', async (req, res) => {
>   const users = await getUsers();  // 自动捕获错误
>   res.json(users);
> });
> ```

## ⚠️ 常见误区

### 误区 1：错误处理中间件位置错误

```javascript
// ❌ 错误：错误处理中间件放在前面
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});

app.get('/users', (req, res) => {
  throw new Error('Error');  // 捕获不到
});

// ✅ 正确：放在最后
app.get('/users', (req, res) => {
  throw new Error('Error');
});

app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});
```

---

### 误区 2：忘记 4 个参数

```javascript
// ❌ 错误：只有 3 个参数（不是错误处理中间件）
app.use((err, req, res) => {
  res.status(500).json({ error: err.message });
});

// ✅ 正确：必须 4 个参数
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});
```

---

### 误区 3：异步错误未捕获

```javascript
// ❌ 错误：异步错误未捕获
app.get('/users', (req, res) => {
  setTimeout(() => {
    throw new Error('Async error');  // 进程崩溃
  }, 0);
});

// ✅ 正确：捕获并传递
app.get('/users', (req, res, next) => {
  setTimeout(() => {
    next(new Error('Async error'));
  }, 0);
});
```
