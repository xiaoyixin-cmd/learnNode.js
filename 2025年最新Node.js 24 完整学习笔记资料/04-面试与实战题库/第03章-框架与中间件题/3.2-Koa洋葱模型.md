## 📚 核心知识点

### 1. 什么是洋葱模型？

**定义：**
> 中间件按照**先进后出**的顺序执行，形成类似洋葱的层层嵌套结构。

**执行流程：**
```
请求 →
  ┌─ M1 before ─┐
  │ ┌─ M2 before ─┐
  │ │ ┌─ M3 before ─┐
  │ │ │   处理逻辑   │
  │ │ └─ M3 after ──┘
  │ └─ M2 after ──┘
  └─ M1 after ──┘
→ 响应
```

**为什么叫洋葱？**
- 每层中间件包裹下一层（like 洋葱的层）
- 请求从外层进入，响应从内层返回
- `await next()` 是**分界线**（进入下一层 vs 返回上一层）

---

### 2. 洋葱模型 vs 传统回调

**传统回调（Express）：**
```javascript
app.use((req, res, next) => {
  console.log('1. Before');
  next();
  console.log('2. After');  // ❌ 立即执行，不等待下一个中间件
});

app.use((req, res, next) => {
  setTimeout(() => {
    console.log('3. Middleware 2');
    next();
  }, 100);
});

// 输出：1. Before → 2. After → 3. Middleware 2
// After 不等待 Middleware 2 完成
```

**洋葱模型（Koa）：**
```javascript
app.use(async (ctx, next) => {
  console.log('1. Before');
  await next();  // ✅ 等待所有后续中间件完成
  console.log('4. After');
});

app.use(async (ctx, next) => {
  await new Promise(resolve => setTimeout(resolve, 100));
  console.log('2. Middleware 2');
  await next();
  console.log('3. After Middleware 2');
});

// 输出：1. Before → 2. Middleware 2 → 3. After Middleware 2 → 4. After
// 层层回溯
```

## 💻 手写 koa-compose

### 版本 1：基础实现（核心原理）

```javascript
function compose(middlewares) {
  // 返回一个函数，接收 context 对象
  return function(context) {
    // 从第一个中间件开始执行
    return dispatch(0);
    
    function dispatch(i) {
      // 获取当前中间件
      const fn = middlewares[i];
      
      // 如果没有中间件了，返回 resolved Promise
      if (!fn) return Promise.resolve();
      
      try {
        // 执行当前中间件，传入 context 和 next 函数
        return Promise.resolve(fn(context, () => dispatch(i + 1)));
      } catch (err) {
        return Promise.reject(err);
      }
    }
  };
}

// 使用示例
const middlewares = [
  async (ctx, next) => {
    console.log('1');
    await next();
    console.log('4');
  },
  async (ctx, next) => {
    console.log('2');
    await next();
    console.log('3');
  }
];

const fn = compose(middlewares);
fn({});  // 输出：1 → 2 → 3 → 4
```

**核心机制：**
1. `dispatch(i)` 执行第 `i` 个中间件
2. `next()` 实际上是 `dispatch(i + 1)`（执行下一个中间件）
3. `await next()` 等待所有后续中间件完成
4. 返回 Promise，支持异步操作

### 版本 2：防止多次调用 next()

```javascript
function compose(middlewares) {
  return function(context) {
    let index = -1;  // 记录当前执行到的中间件索引
    
    return dispatch(0);
    
    function dispatch(i) {
      // 防止多次调用 next()
      if (i <= index) {
        return Promise.reject(new Error('next() called multiple times'));
      }
      
      index = i;
      
      const fn = middlewares[i];
      if (!fn) return Promise.resolve();
      
      try {
        return Promise.resolve(fn(context, () => dispatch(i + 1)));
      } catch (err) {
        return Promise.reject(err);
      }
    }
  };
}

// 测试多次调用 next()
const middlewares = [
  async (ctx, next) => {
    await next();
    await next();  // ❌ 错误：next() called multiple times
  }
];

const fn = compose(middlewares);
fn({}).catch(err => console.error(err.message));
```

### 版本 3：完整实现（koa-compose 源码）

```javascript
function compose(middlewares) {
  // 参数校验
  if (!Array.isArray(middlewares)) {
    throw new TypeError('Middleware stack must be an array!');
  }
  
  for (const fn of middlewares) {
    if (typeof fn !== 'function') {
      throw new TypeError('Middleware must be composed of functions!');
    }
  }
  
  return function(context, next) {
    let index = -1;
    
    return dispatch(0);
    
    function dispatch(i) {
      if (i <= index) {
        return Promise.reject(new Error('next() called multiple times'));
      }
      
      index = i;
      
      let fn = middlewares[i];
      
      // 最后一个中间件，调用传入的 next（如果有）
      if (i === middlewares.length) fn = next;
      
      if (!fn) return Promise.resolve();
      
      try {
        return Promise.resolve(fn(context, dispatch.bind(null, i + 1)));
      } catch (err) {
        return Promise.reject(err);
      }
    }
  };
}
```

## 💻 实战案例

### 案例 1：日志中间件（计算响应时间）

```javascript
const Koa = require('koa');
const app = new Koa();

// 日志中间件（洋葱模型典型应用）
app.use(async (ctx, next) => {
  const start = Date.now();
  
  await next();  // 执行后续所有中间件
  
  const ms = Date.now() - start;
  console.log(`${ctx.method} ${ctx.url} - ${ms}ms`);
  ctx.set('X-Response-Time', `${ms}ms`);
});

// 业务逻辑
app.use(async ctx => {
  await new Promise(resolve => setTimeout(resolve, 100));
  ctx.body = { msg: 'Hello' };
});

app.listen(3000);

// 请求：GET /
// 输出：GET / - 102ms
// 响应头：X-Response-Time: 102ms
```

### 案例 2：错误处理中间件

```javascript
const Koa = require('koa');
const app = new Koa();

// 全局错误处理（放在最外层）
app.use(async (ctx, next) => {
  try {
    await next();  // 执行后续中间件
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      error: err.message
    };
    ctx.app.emit('error', err, ctx);
  }
});

// 业务逻辑
app.use(async ctx => {
  if (!ctx.query.name) {
    ctx.throw(400, 'Name is required');
  }
  ctx.body = { name: ctx.query.name };
});

app.listen(3000);

// 请求：GET /?name=Alice → {"name":"Alice"}
// 请求：GET /             → {"error":"Name is required"}
```

### 案例 4：缓存中间件

```javascript
const Koa = require('koa');
const Redis = require('ioredis');

const app = new Koa();
const redis = new Redis();

// 缓存中间件
app.use(async (ctx, next) => {
  const key = `cache:${ctx.url}`;
  
  // 检查缓存
  const cached = await redis.get(key);
  if (cached) {
    console.log('Cache hit');
    ctx.body = JSON.parse(cached);
    return;
  }
  
  // 执行后续中间件
  await next();
  
  // 缓存响应（洋葱模型：在响应阶段缓存）
  if (ctx.body) {
    await redis.set(key, JSON.stringify(ctx.body), 'EX', 60);
    console.log('Cache set');
  }
});

// 业务逻辑
app.use(async ctx => {
  ctx.body = { data: 'expensive query result' };
});

app.listen(3000);
```

### 问题 2："洋葱模型的执行流程是什么？"

**标准答案：**
> **洋葱模型的执行流程：先进后出**
> 
> **1. 执行阶段（向内）**
> ```javascript
> app.use(async (ctx, next) => {
>   console.log('M1 before');  // 1. 执行
>   await next();              // 2. 进入下一层
>   console.log('M1 after');   // 6. 回溯
> });
> 
> app.use(async (ctx, next) => {
>   console.log('M2 before');  // 3. 执行
>   await next();              // 4. 进入下一层
>   console.log('M2 after');   // 5. 回溯
> });
> ```
> 
> **2. 流程图**
> ```
> 请求 →
>   M1 before (1)
>     ↓
>   M2 before (2)
>     ↓
>   M3 before (3)
>     ↓
>   业务逻辑 (4)
>     ↓
>   M3 after (5)
>     ↓
>   M2 after (6)
>     ↓
>   M1 after (7)
> ← 响应
> ```
> 
> **3. 关键特性**
> - `await next()` 是分界线（进入 vs 回溯）
> - 在 `await next()` 之前：请求阶段（向内）
> - 在 `await next()` 之后：响应阶段（向外）
> - 适合：日志、性能监控、错误处理

## ⚠️ 常见误区

### 误区 1：忘记 await next()

```javascript
// ❌ 错误：忘记 await
app.use(async (ctx, next) => {
  console.log('Before');
  next();  // 不等待
  console.log('After');
});

// 输出：Before → After（立即执行，不等待下一个中间件）

// ✅ 正确：await next()
app.use(async (ctx, next) => {
  console.log('Before');
  await next();  // 等待
  console.log('After');
});
```

---

### 误区 2：多次调用 next()

```javascript
// ❌ 错误：多次调用 next()
app.use(async (ctx, next) => {
  await next();
  await next();  // Error: next() called multiple times
});

// ✅ 正确：只调用一次
app.use(async (ctx, next) => {
  await next();
  // 只能调用一次
});
```

---

### 误区 3：在 next() 之后返回响应

```javascript
// ❌ 错误：在 next() 之后修改 ctx.body（可能被覆盖）
app.use(async (ctx, next) => {
  await next();
  ctx.body = { msg: 'Modified' };  // 可能被后续中间件覆盖
});

// ✅ 正确：在 next() 之前返回响应
app.use(async (ctx, next) => {
  if (ctx.url === '/api') {
    ctx.body = { msg: 'API' };
    return;  // 不调用 next()
  }
  await next();
});
```
