## 📚 核心知识点

### 1. 并发控制的必要性

**问题场景：**
```javascript
// ❌ 问题：同时发起 1000 个请求
const tasks = Array.from({ length: 1000 }, (_, i) => 
  () => fetch(`https://api.example.com/data/${i}`)
);

await Promise.all(tasks.map(task => task()));
// 可能导致：
// 1. 服务器拒绝连接（Too Many Requests 429）
// 2. 浏览器/Node.js 连接数限制
// 3. 内存占用过高
```

**解决方案：**
- 限制并发数量（如：同时最多 5 个请求）
- 使用队列管理任务
- 失败重试机制

---

### 2. p-limit 核心思想

```
待执行队列: [Task6, Task7, Task8, ...]
               ↓
运行中: [Task1, Task2, Task3, Task4, Task5]  ← 并发数 = 5
               ↓
完成后: 从队列取出 Task6 执行
```

**关键点：**
- 维护一个**运行计数器**（running）
- 维护一个**等待队列**（queue）
- 任务完成后，从队列取出下一个任务

## 💻 手写 p-limit 实现

### 版本 1：基础实现

```javascript
function pLimit(concurrency) {
  const queue = [];
  let running = 0;
  
  const next = () => {
    running--;
    
    if (queue.length > 0) {
      const run = queue.shift();
      run();
    }
  };
  
  const run = async (fn, resolve, reject) => {
    running++;
    
    try {
      const result = await fn();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      next();
    }
  };
  
  return (fn) => {
    return new Promise((resolve, reject) => {
      if (running < concurrency) {
        run(fn, resolve, reject);
      } else {
        queue.push(() => run(fn, resolve, reject));
      }
    });
  };
}

// 使用示例
const limit = pLimit(3);  // 最多 3 个并发

const tasks = Array.from({ length: 10 }, (_, i) => 
  limit(() => fetch(`https://api.example.com/${i}`))
);

await Promise.all(tasks);
```

### 版本 2：支持优先级

```javascript
function pLimit(concurrency) {
  const queue = [];
  let running = 0;
  
  const next = () => {
    running--;
    
    if (queue.length > 0) {
      // 按优先级排序（数字越大优先级越高）
      queue.sort((a, b) => b.priority - a.priority);
      const { run } = queue.shift();
      run();
    }
  };
  
  const run = async (fn, resolve, reject) => {
    running++;
    
    try {
      const result = await fn();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      next();
    }
  };
  
  return (fn, priority = 0) => {
    return new Promise((resolve, reject) => {
      if (running < concurrency) {
        run(fn, resolve, reject);
      } else {
        queue.push({
          priority,
          run: () => run(fn, resolve, reject)
        });
      }
    });
  };
}

// 使用
const limit = pLimit(3);

limit(() => fetch('/low'), 1);      // 低优先级
limit(() => fetch('/high'), 10);    // 高优先级（优先执行）
limit(() => fetch('/medium'), 5);   // 中优先级
```

### 版本 3：支持取消

```javascript
class PLimitWithCancel {
  constructor(concurrency) {
    this.concurrency = concurrency;
    this.queue = [];
    this.running = 0;
    this.cancelled = new Set();
  }
  
  next() {
    this.running--;
    
    while (this.queue.length > 0 && this.running < this.concurrency) {
      const { id, run } = this.queue.shift();
      
      if (!this.cancelled.has(id)) {
        run();
        break;
      }
    }
  }
  
  async run(id, fn, resolve, reject) {
    if (this.cancelled.has(id)) {
      reject(new Error('Cancelled'));
      return;
    }
    
    this.running++;
    
    try {
      const result = await fn();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      this.cancelled.delete(id);
      this.next();
    }
  }
  
  add(fn) {
    const id = Math.random().toString(36).slice(2);
    
    const promise = new Promise((resolve, reject) => {
      if (this.running < this.concurrency) {
        this.run(id, fn, resolve, reject);
      } else {
        this.queue.push({
          id,
          run: () => this.run(id, fn, resolve, reject)
        });
      }
    });
    
    promise.cancel = () => {
      this.cancelled.add(id);
    };
    
    return promise;
  }
}

// 使用
const limiter = new PLimitWithCancel(3);

const task1 = limiter.add(() => fetch('/api/1'));
const task2 = limiter.add(() => fetch('/api/2'));

task1.cancel();  // 取消任务
```

### 案例 1：批量下载文件

```javascript
const pLimit = require('p-limit');
const fs = require('fs');
const https = require('https');

async function downloadFile(url, dest) {
  return new Promise((resolve, reject) => {
    const file = fs.createWriteStream(dest);
    
    https.get(url, (response) => {
      response.pipe(file);
      file.on('finish', () => {
        file.close();
        resolve();
      });
    }).on('error', (err) => {
      fs.unlink(dest, () => {});
      reject(err);
    });
  });
}

async function batchDownload(urls) {
  const limit = pLimit(5);  // 最多 5 个并发下载
  
  const tasks = urls.map((url, i) => 
    limit(() => {
      console.log(`Downloading ${i + 1}/${urls.length}...`);
      return downloadFile(url, `./file-${i}.txt`);
    })
  );
  
  await Promise.all(tasks);
  console.log('All downloads complete!');
}

// 使用
const urls = [
  'https://example.com/file1.txt',
  'https://example.com/file2.txt',
  // ... 100 个文件
];

batchDownload(urls);
```

### 案例 3：爬虫并发控制

```javascript
const pLimit = require('p-limit');
const axios = require('axios');
const cheerio = require('cheerio');

async function crawlPage(url) {
  const { data } = await axios.get(url);
  const $ = cheerio.load(data);
  
  const links = [];
  $('a').each((i, el) => {
    links.push($(el).attr('href'));
  });
  
  return links;
}

async function crawlWebsite(startUrl, maxPages = 100) {
  const limit = pLimit(5);  // 最多 5 个并发请求
  const visited = new Set();
  const queue = [startUrl];
  
  while (queue.length > 0 && visited.size < maxPages) {
    const batch = queue.splice(0, 5);  // 每次处理 5 个
    
    const tasks = batch.map((url) => 
      limit(async () => {
        if (visited.has(url)) return [];
        visited.add(url);
        
        console.log(`Crawling: ${url} (${visited.size}/${maxPages})`);
        
        try {
          const links = await crawlPage(url);
          return links.filter(link => !visited.has(link));
        } catch (error) {
          console.error(`Failed to crawl ${url}:`, error.message);
          return [];
        }
      })
    );
    
    const results = await Promise.all(tasks);
    const newLinks = results.flat();
    queue.push(...newLinks);
  }
  
  console.log(`Crawled ${visited.size} pages`);
}

// 使用
crawlWebsite('https://example.com', 100);
```

### 案例 4：失败重试机制

```javascript
function pLimitWithRetry(concurrency, maxRetries = 3) {
  const limit = pLimit(concurrency);
  
  return (fn) => {
    return limit(async () => {
      let lastError;
      
      for (let i = 0; i <= maxRetries; i++) {
        try {
          return await fn();
        } catch (error) {
          lastError = error;
          console.log(`Retry ${i + 1}/${maxRetries}:`, error.message);
          
          if (i < maxRetries) {
            // 指数退避
            await new Promise(resolve => 
              setTimeout(resolve, Math.pow(2, i) * 1000)
            );
          }
        }
      }
      
      throw lastError;
    });
  };
}

// 使用
const limit = pLimitWithRetry(3, 3);

const tasks = urls.map(url => 
  limit(() => fetch(url).then(r => r.json()))
);

await Promise.all(tasks);
```

### 问题 1："如何限制并发请求数量？请手写 p-limit 实现"

**标准答案：**
> 我会实现一个并发控制函数，核心思路是：
> 
> **1. 数据结构**
> - `running`：当前运行的任务数
> - `queue`：等待队列
> - `concurrency`：并发限制
> 
> **2. 执行逻辑**
> ```javascript
> function pLimit(concurrency) {
>   const queue = [];
>   let running = 0;
>   
>   const next = () => {
>     running--;
>     if (queue.length > 0) {
>       queue.shift()();  // 执行下一个任务
>     }
>   };
>   
>   return (fn) => {
>     return new Promise((resolve, reject) => {
>       const run = async () => {
>         running++;
>         try {
>           const result = await fn();
>           resolve(result);
>         } catch (error) {
>           reject(error);
>         } finally {
>           next();
>         }
>       };
>       
>       if (running < concurrency) {
>         run();
>       } else {
>         queue.push(run);
>       }
>     });
>   };
> }
> ```
> 
> **3. 工作流程**
> - 任务数 < 并发数：立即执行
> - 任务数 ≥ 并发数：加入等待队列
> - 任务完成：running--，从队列取出下一个
> 
> **4. 使用示例**
> ```javascript
> const limit = pLimit(3);
> 
> const tasks = urls.map(url => 
>   limit(() => fetch(url))
> );
> 
> await Promise.all(tasks);
> ```

**加分项：**
> 在实际项目中，我基于这个原理实现了：
> 1. **优先级队列**：高优先级任务优先执行
> 2. **取消机制**：支持取消等待中的任务
> 3. **失败重试**：自动重试失败的任务
> 4. **进度追踪**：实时显示完成进度

### 问题 2："Promise.all 如何改造成并发控制？"

**标准答案：**
> Promise.all 是**无限并发**，需要改造成**有限并发**：
> 
> **原始 Promise.all：**
> ```javascript
> await Promise.all([task1(), task2(), ..., task1000()]);
> // 同时执行 1000 个任务
> ```
> 
> **改造方案 1：使用 p-limit**
> ```javascript
> const limit = pLimit(5);
> const tasks = [task1, task2, ..., task1000].map(task => limit(task));
> await Promise.all(tasks);
> ```
> 
> **改造方案 2：手动分批**
> ```javascript
> async function batchPromiseAll(tasks, batchSize) {
>   const results = [];
>   
>   for (let i = 0; i < tasks.length; i += batchSize) {
>     const batch = tasks.slice(i, i + batchSize);
>     const batchResults = await Promise.all(batch.map(task => task()));
>     results.push(...batchResults);
>   }
>   
>   return results;
> }
> 
> await batchPromiseAll(tasks, 5);  // 每批 5 个
> ```
> 
> **改造方案 3：递归控制**
> ```javascript
> async function concurrentMap(tasks, concurrency) {
>   const results = [];
>   const executing = [];
>   
>   for (const task of tasks) {
>     const p = Promise.resolve().then(task);
>     results.push(p);
>     
>     if (concurrency <= tasks.length) {
>       const e = p.then(() => executing.splice(executing.indexOf(e), 1));
>       executing.push(e);
>       
>       if (executing.length >= concurrency) {
>         await Promise.race(executing);
>       }
>     }
>   }
>   
>   return Promise.all(results);
> }
> ```

## ⚠️ 常见误区

### 误区 1：以为 map + Promise.all 会限制并发

```javascript
// ❌ 错误理解：以为这样会限制并发
const tasks = urls.map(async (url) => {
  return await fetch(url);
});
await Promise.all(tasks);

// 实际：所有任务同时执行（无限并发）
```

---

### 误区 2：使用 for...of 串行执行

```javascript
// ❌ 错误：完全串行，效率低
for (const url of urls) {
  await fetch(url);  // 一个一个执行
}

// ✅ 正确：并发控制
const limit = pLimit(5);
const tasks = urls.map(url => limit(() => fetch(url)));
await Promise.all(tasks);
```

---

### 误区 3：忘记处理错误

```javascript
// ❌ 错误：一个失败全部失败
await Promise.all(tasks);

// ✅ 正确：使用 allSettled
const results = await Promise.allSettled(tasks);
results.forEach((result, i) => {
  if (result.status === 'fulfilled') {
    console.log(`Task ${i} success:`, result.value);
  } else {
    console.error(`Task ${i} failed:`, result.reason);
  }
});
```

```javascript
async function pMap(items, mapper, options = {}) {
  const { concurrency = Infinity } = options;
  const results = new Array(items.length);
  const executing = [];
  
  for (const [index, item] of items.entries()) {
    const p = Promise.resolve().then(() => mapper(item, index));
    results[index] = p;
    
    if (concurrency < items.length) {
      const e = p.then(() => executing.splice(executing.indexOf(e), 1));
      executing.push(e);
      
      if (executing.length >= concurrency) {
        await Promise.race(executing);
      }
    }
  }
  
  return Promise.all(results);
}
```
