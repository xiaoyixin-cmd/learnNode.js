## 📚 核心知识点

### 1. Worker Threads 适用场景

| 任务类型 | 使用工具 | 原因 |
|---------|---------|------|
| **CPU 密集**（加密、压缩、图片处理） | Worker Threads | 避免阻塞主线程 |
| **I/O 密集**（文件、网络） | 异步 I/O | libuv 自动处理 |
| **负载均衡** | Cluster | 多进程共享端口 |

**判断标准：**
```javascript
// ❌ 不需要 Worker（I/O 密集）
fs.readFile('./file.txt', callback);  // 异步 I/O，不阻塞

// ✅ 需要 Worker（CPU 密集）
for (let i = 0; i < 1e9; i++) {  // 阻塞主线程
  // 复杂计算
}
```

---

### 2. Worker Threads vs Cluster

| 特性 | Worker Threads | Cluster |
|------|---------------|---------|
| **用途** | CPU 密集型任务 | 负载均衡 |
| **内存** | 共享内存（SharedArrayBuffer） | 独立内存 |
| **通信** | MessageChannel | IPC |
| **创建成本** | 低（线程） | 高（进程） |
| **适用场景** | 计算密集 | HTTP 服务 |

## 💻 基础用法

### 1. 创建 Worker

```javascript
// main.js
const { Worker } = require('worker_threads');

const worker = new Worker('./worker.js', {
  workerData: { value: 42 }  // 传递初始数据
});

worker.on('message', (result) => {
  console.log('Result from worker:', result);
});

worker.on('error', (error) => {
  console.error('Worker error:', error);
});

worker.on('exit', (code) => {
  console.log(`Worker exited with code ${code}`);
});

// worker.js
const { parentPort, workerData } = require('worker_threads');

// 接收初始数据
console.log('Worker data:', workerData.value);  // 42

// 执行计算
const result = workerData.value * 2;

// 发送结果
parentPort.postMessage(result);
```

---

### 2. 双向通信

```javascript
// main.js
const { Worker } = require('worker_threads');

const worker = new Worker('./worker.js');

worker.postMessage({ type: 'start', data: 100 });

worker.on('message', (msg) => {
  if (msg.type === 'result') {
    console.log('Result:', msg.data);
  }
});

// worker.js
const { parentPort } = require('worker_threads');

parentPort.on('message', (msg) => {
  if (msg.type === 'start') {
    const result = msg.data * 2;
    parentPort.postMessage({ type: 'result', data: result });
  }
});
```

## 💻 实战案例

### 案例 1：斐波那契数列（CPU 密集）

```javascript
// main.js
const { Worker } = require('worker_threads');

function fibonacci(n) {
  return new Promise((resolve, reject) => {
    const worker = new Worker(`
      const { parentPort, workerData } = require('worker_threads');
      
      function fib(n) {
        if (n <= 1) return n;
        return fib(n - 1) + fib(n - 2);
      }
      
      const result = fib(workerData.n);
      parentPort.postMessage(result);
    `, {
      eval: true,
      workerData: { n }
    });
    
    worker.on('message', resolve);
    worker.on('error', reject);
  });
}

// 使用
(async () => {
  console.log('Main thread continues...');
  
  const result = await fibonacci(40);
  console.log('Fibonacci(40):', result);
})();

// 输出：
// Main thread continues...  ← 主线程不阻塞
// Fibonacci(40): 102334155
```

### 案例 2：图片处理（并行）

```javascript
// image-processor.js
const { Worker } = require('worker_threads');
const sharp = require('sharp');

async function processImages(images) {
  const workers = images.map((imagePath) => {
    return new Promise((resolve, reject) => {
      const worker = new Worker('./image-worker.js', {
        workerData: { imagePath }
      });
      
      worker.on('message', resolve);
      worker.on('error', reject);
    });
  });
  
  return Promise.all(workers);
}

// image-worker.js
const { workerData, parentPort } = require('worker_threads');
const sharp = require('sharp');

(async () => {
  const { imagePath } = workerData;
  
  // CPU 密集：图片压缩
  await sharp(imagePath)
    .resize(800, 600)
    .jpeg({ quality: 80 })
    .toFile(imagePath.replace('.jpg', '-thumb.jpg'));
  
  parentPort.postMessage({ success: true, path: imagePath });
})();

// 使用
const images = ['img1.jpg', 'img2.jpg', ..., 'img100.jpg'];
const results = await processImages(images);
console.log('All images processed:', results.length);
```

### 案例 3：加密计算（crypto）

```javascript
// main.js
const { Worker } = require('worker_threads');

async function hashPassword(password) {
  return new Promise((resolve, reject) => {
    const worker = new Worker('./crypto-worker.js', {
      workerData: { password }
    });
    
    worker.on('message', resolve);
    worker.on('error', reject);
  });
}

// crypto-worker.js
const { workerData, parentPort } = require('worker_threads');
const crypto = require('crypto');

const { password } = workerData;

// CPU 密集：pbkdf2（10万次迭代）
crypto.pbkdf2(password, 'salt', 100000, 64, 'sha512', (err, hash) => {
  if (err) throw err;
  parentPort.postMessage(hash.toString('hex'));
});

// 使用
const users = ['user1', 'user2', ..., 'user1000'];

const promises = users.map(user => 
  hashPassword(user).then(hash => ({ user, hash }))
);

const results = await Promise.all(promises);
console.log('All passwords hashed');
```

### 案例 4：Worker Pool（线程池）

```javascript
// worker-pool.js
const { Worker } = require('worker_threads');

class WorkerPool {
  constructor(workerPath, poolSize = 4) {
    this.workerPath = workerPath;
    this.poolSize = poolSize;
    this.workers = [];
    this.queue = [];
    
    for (let i = 0; i < poolSize; i++) {
      this.workers.push({ busy: false, worker: null });
    }
  }
  
  async exec(data) {
    return new Promise((resolve, reject) => {
      const task = { data, resolve, reject };
      
      const freeWorker = this.workers.find(w => !w.busy);
      
      if (freeWorker) {
        this.runTask(freeWorker, task);
      } else {
        this.queue.push(task);
      }
    });
  }
  
  runTask(workerSlot, task) {
    workerSlot.busy = true;
    
    const worker = new Worker(this.workerPath, {
      workerData: task.data
    });
    
    worker.on('message', (result) => {
      task.resolve(result);
      workerSlot.busy = false;
      worker.terminate();
      
      if (this.queue.length > 0) {
        this.runTask(workerSlot, this.queue.shift());
      }
    });
    
    worker.on('error', (error) => {
      task.reject(error);
      workerSlot.busy = false;
      worker.terminate();
    });
  }
}

// 使用
const pool = new WorkerPool('./worker.js', 4);

const tasks = Array.from({ length: 100 }, (_, i) => 
  pool.exec({ value: i })
);

const results = await Promise.all(tasks);
console.log('All tasks completed:', results.length);
```

## 🎯 SharedArrayBuffer（共享内存）

### 1. 基础用法

```javascript
// main.js
const { Worker } = require('worker_threads');

// 创建共享内存（4 字节 * 10）
const sharedBuffer = new SharedArrayBuffer(40);
const sharedArray = new Int32Array(sharedBuffer);

sharedArray[0] = 42;

const worker = new Worker('./worker.js', {
  workerData: { sharedBuffer }
});

setTimeout(() => {
  console.log('Main thread:', sharedArray[0]);  // 84（Worker 修改了）
}, 1000);

// worker.js
const { workerData } = require('worker_threads');

const sharedArray = new Int32Array(workerData.sharedBuffer);

console.log('Worker thread:', sharedArray[0]);  // 42
sharedArray[0] = 84;  // 修改共享内存
```

---

### 2. 原子操作（Atomics）

```javascript
// main.js
const { Worker } = require('worker_threads');

const sharedBuffer = new SharedArrayBuffer(4);
const sharedArray = new Int32Array(sharedBuffer);

const worker = new Worker('./worker.js', {
  workerData: { sharedBuffer }
});

// 原子加 1
Atomics.add(sharedArray, 0, 1);
console.log('Main:', sharedArray[0]);  // 1

// 等待 Worker 修改
Atomics.wait(sharedArray, 0, 1);
console.log('After wait:', sharedArray[0]);  // 2

// worker.js
const { workerData } = require('worker_threads');

const sharedArray = new Int32Array(workerData.sharedBuffer);

setTimeout(() => {
  Atomics.add(sharedArray, 0, 1);  // 原子加 1
  Atomics.notify(sharedArray, 0);   // 通知主线程
}, 1000);
```

## 🎓 面试标准答题

### 问题 1："什么样的任务需要用 Worker Threads？"

**标准答案：**
> **需要 Worker Threads 的场景：CPU 密集型任务**
> 
> **1. 判断标准**
> - 任务执行时间 > 10ms
> - 会阻塞事件循环
> - 同步计算密集
> 
> **2. 典型场景**
> - **加密/解密**：crypto.pbkdf2、scrypt
> - **压缩/解压**：zlib、gzip
> - **图片处理**：sharp、jimp
> - **复杂计算**：科学计算、AI 推理
> - **大数据处理**：排序、聚合
> 
> **3. 不需要 Worker 的场景**
> - **I/O 密集**：文件读写、网络请求（异步 I/O 自动处理）
> - **简单计算**：< 1ms 的计算
> - **事件驱动**：定时器、事件监听
> 
> **4. 示例对比**
> ```javascript
> // ❌ 不需要 Worker（I/O 密集）
> fs.readFile('./file.txt', callback);
> 
> // ✅ 需要 Worker（CPU 密集）
> crypto.pbkdf2(password, salt, 100000, 64, 'sha512', callback);
> ```
> 
> **5. 性能对比**
> - 主线程执行：阻塞事件循环，响应时间 > 1s
> - Worker 执行：不阻塞，响应时间 < 10ms

**加分项：**
> 在实际项目中，我们将视频转码（ffmpeg）移到 Worker Threads，避免了主线程阻塞，API 响应时间从 5s 降到 50ms。

---

### 问题 2："Worker Threads 和 Cluster 的区别？"

**标准答案：**
> **核心区别：线程 vs 进程**
> 
> | 特性 | Worker Threads | Cluster |
> |------|---------------|---------|
> | **本质** | 线程（轻量） | 进程（重量） |
> | **用途** | CPU 密集型任务 | 负载均衡 |
> | **内存** | 共享内存 | 独立内存 |
> | **通信** | MessageChannel（快） | IPC（慢） |
> | **创建成本** | 低（1-5ms） | 高（50-100ms） |
> | **端口共享** | ❌ 不支持 | ✅ 支持 |
> 
> **使用场景：**
> 
> **Worker Threads：**
> - 图片处理（sharp）
> - 加密计算（crypto.pbkdf2）
> - 数据压缩（zlib）
> - 不需要网络服务
> 
> **Cluster：**
> - HTTP 服务器（多进程监听同一端口）
> - 充分利用多核 CPU
> - 进程隔离（一个崩溃不影响其他）
> 
> **示例：**
> ```javascript
> // Worker Threads（计算）
> const worker = new Worker('./calc.js');
> worker.postMessage({ data: largeArray });
> 
> // Cluster（HTTP 服务）
> if (cluster.isMaster) {
>   for (let i = 0; i < cpus().length; i++) {
>     cluster.fork();
>   }
> } else {
>   app.listen(3000);
> }
> ```

## ⚠️ 常见误区

### 误区 1：滥用 Worker Threads

```javascript
// ❌ 错误：简单任务用 Worker（创建线程的开销 > 任务本身）
const worker = new Worker('./simple.js');
worker.postMessage({ a: 1, b: 2 });

// simple.js
const { a, b } = workerData;
parentPort.postMessage(a + b);  // 简单加法，不值得用 Worker

// ✅ 正确：直接计算
const result = 1 + 2;
```

---

### 误区 2：忘记终止 Worker

```javascript
// ❌ 错误：Worker 不终止，内存泄漏
const worker = new Worker('./worker.js');
worker.on('message', (result) => {
  console.log(result);
  // 忘记 worker.terminate()
});

// ✅ 正确
worker.on('message', (result) => {
  console.log(result);
  worker.terminate();  // 终止 Worker
});
```

---

### 误区 3：传递不可序列化的数据

```javascript
// ❌ 错误：传递函数（不可序列化）
worker.postMessage({ fn: () => console.log('hello') });
// DataCloneError: function could not be cloned

// ✅ 正确：传递数据
worker.postMessage({ type: 'hello' });
```
