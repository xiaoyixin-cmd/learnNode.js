## 📚 核心知识点

### 1. Stream vs fs.readFile 对比

| 特性 | fs.readFile | Stream |
|------|-------------|--------|
| **内存占用** | 整个文件大小（1GB 文件 = 1GB 内存） | 固定缓冲区（默认 64KB） |
| **启动时间** | 慢（等待全部读取） | 快（立即开始处理） |
| **适用场景** | 小文件（< 10MB） | 大文件（> 100MB） |
| **背压支持** | ❌ 无 | ✅ 有 |
| **内存安全** | ❌ 可能 OOM | ✅ 恒定内存 |

---

### 2. Stream 工作原理

```
文件（1GB）
   ↓
┌──────────────────────────────────┐
│   Readable Stream                │
│   ┌──────┐ ┌──────┐ ┌──────┐    │
│   │ 64KB │→│ 64KB │→│ 64KB │... │
│   └──────┘ └──────┘ └──────┘    │
└──────────────────────────────────┘
   ↓ data event
┌──────────────────────────────────┐
│   处理逻辑（边读边处理）          │
└──────────────────────────────────┘
   ↓
┌──────────────────────────────────┐
│   Writable Stream（可选）         │
└──────────────────────────────────┘
```

**关键参数：**
- `highWaterMark`：缓冲区大小（默认 64KB）
- `encoding`：字符编码（默认 Buffer）

### 案例 1：内存占用对比

```javascript
const fs = require('fs');

// ❌ fs.readFile（内存占用 = 文件大小）
console.log('Before readFile:', process.memoryUsage().heapUsed / 1024 / 1024, 'MB');

fs.readFile('./large-file.txt', (err, data) => {
  console.log('After readFile:', process.memoryUsage().heapUsed / 1024 / 1024, 'MB');
  console.log('File size:', data.length / 1024 / 1024, 'MB');
});

// 输出（100MB 文件）：
// Before readFile: 5.23 MB
// After readFile: 105.45 MB  ← 增加了 100MB

// ✅ Stream（内存占用恒定）
console.log('Before Stream:', process.memoryUsage().heapUsed / 1024 / 1024, 'MB');

const stream = fs.createReadStream('./large-file.txt');

stream.on('data', (chunk) => {
  console.log('During Stream:', process.memoryUsage().heapUsed / 1024 / 1024, 'MB');
});

stream.on('end', () => {
  console.log('After Stream:', process.memoryUsage().heapUsed / 1024 / 1024, 'MB');
});

// 输出（100MB 文件）：
// Before Stream: 5.23 MB
// During Stream: 5.45 MB  ← 恒定在 5-6 MB
// During Stream: 5.47 MB
// After Stream: 5.50 MB
```

### 案例 2：处理大文件（GB 级）

```javascript
const fs = require('fs');
const { Transform } = require('stream');

// 统计行数（不加载整个文件到内存）
function countLines(filePath) {
  return new Promise((resolve, reject) => {
    let lineCount = 0;
    
    const stream = fs.createReadStream(filePath, {
      encoding: 'utf8',
      highWaterMark: 64 * 1024,  // 64KB 缓冲区
    });
    
    let lastChunk = '';
    
    stream.on('data', (chunk) => {
      // 合并上次未处理完的部分
      chunk = lastChunk + chunk;
      
      // 统计换行符
      const lines = chunk.split('\n');
      lineCount += lines.length - 1;
      
      // 保存最后一行（可能不完整）
      lastChunk = lines[lines.length - 1];
    });
    
    stream.on('end', () => {
      if (lastChunk) lineCount++;  // 最后一行
      resolve(lineCount);
    });
    
    stream.on('error', reject);
  });
}

// 使用
countLines('./10gb-log.txt').then((count) => {
  console.log('Total lines:', count);
});

// 内存占用：始终 < 10MB（无论文件多大）
```

### 案例 3：文件复制（Stream vs readFile）

```javascript
const fs = require('fs');
const { performance } = require('perf_hooks');

// ❌ readFile 方式（慢 + 占内存）
async function copyWithReadFile(src, dest) {
  const start = performance.now();
  const startMem = process.memoryUsage().heapUsed;
  
  const data = await fs.promises.readFile(src);
  await fs.promises.writeFile(dest, data);
  
  const duration = performance.now() - start;
  const memUsed = (process.memoryUsage().heapUsed - startMem) / 1024 / 1024;
  
  console.log(`readFile: ${duration.toFixed(2)}ms, Memory: ${memUsed.toFixed(2)}MB`);
}

// ✅ Stream 方式（快 + 省内存）
async function copyWithStream(src, dest) {
  return new Promise((resolve, reject) => {
    const start = performance.now();
    const startMem = process.memoryUsage().heapUsed;
    
    const readStream = fs.createReadStream(src);
    const writeStream = fs.createWriteStream(dest);
    
    readStream.pipe(writeStream);
    
    writeStream.on('finish', () => {
      const duration = performance.now() - start;
      const memUsed = (process.memoryUsage().heapUsed - startMem) / 1024 / 1024;
      
      console.log(`Stream: ${duration.toFixed(2)}ms, Memory: ${memUsed.toFixed(2)}MB`);
      resolve();
    });
    
    writeStream.on('error', reject);
  });
}

// 测试（复制 500MB 文件）
copyWithReadFile('./500mb.txt', './500mb-copy1.txt');
copyWithStream('./500mb.txt', './500mb-copy2.txt');

// 输出：
// readFile: 1234.56ms, Memory: 512.34MB  ← 慢 + 占内存
// Stream: 456.78ms, Memory: 0.45MB      ← 快 + 省内存
```

### 案例 4：Transform Stream（数据转换）

```javascript
const fs = require('fs');
const { Transform } = require('stream');

// 大写转换流
class UpperCaseTransform extends Transform {
  _transform(chunk, encoding, callback) {
    this.push(chunk.toString().toUpperCase());
    callback();
  }
}

// 读取 → 转换 → 写入
fs.createReadStream('./input.txt')
  .pipe(new UpperCaseTransform())
  .pipe(fs.createWriteStream('./output.txt'));

// 内存占用：始终 < 1MB（无论文件多大）
```

### 案例 5：CSV 解析（大文件）

```javascript
const fs = require('fs');
const { Transform } = require('stream');

// CSV 解析流
class CSVParser extends Transform {
  constructor() {
    super({ objectMode: true });
    this.headers = null;
    this.buffer = '';
  }
  
  _transform(chunk, encoding, callback) {
    this.buffer += chunk.toString();
    const lines = this.buffer.split('\n');
    
    // 保留最后一行（可能不完整）
    this.buffer = lines.pop() || '';
    
    lines.forEach((line) => {
      if (!this.headers) {
        this.headers = line.split(',');
      } else {
        const values = line.split(',');
        const obj = {};
        this.headers.forEach((header, i) => {
          obj[header] = values[i];
        });
        this.push(obj);  // 输出对象
      }
    });
    
    callback();
  }
  
  _flush(callback) {
    if (this.buffer) {
      const values = this.buffer.split(',');
      const obj = {};
      this.headers.forEach((header, i) => {
        obj[header] = values[i];
      });
      this.push(obj);
    }
    callback();
  }
}

// 使用
fs.createReadStream('./large.csv')
  .pipe(new CSVParser())
  .on('data', (row) => {
    console.log(row);  // { name: 'Alice', age: '25', ... }
  });

// 处理 10GB CSV 文件，内存 < 10MB
```

## 🎓 面试标准答题

### 问题 1："为什么大文件读取要用 Stream 而不是 fs.readFile？"

**标准答案：**
> **主要原因是内存占用的差异：**
> 
> **1. 内存占用对比：**
> - **fs.readFile**：将整个文件加载到内存
>   - 100MB 文件 → 占用 100MB 内存
>   - 1GB 文件 → 占用 1GB 内存
>   - 可能触发 OOM（内存溢出）
> 
> - **Stream**：固定大小缓冲区（默认 64KB）
>   - 100MB 文件 → 占用 64KB 内存
>   - 1GB 文件 → 占用 64KB 内存
>   - 内存安全，不会溢出
> 
> **2. 处理速度对比：**
> - **fs.readFile**：必须等待全部读取完成
>   - 启动慢（等待 I/O）
>   - 不能提前处理
> 
> - **Stream**：边读边处理
>   - 启动快（立即开始）
>   - 可以并行处理
> 
> **3. 背压支持：**
> - **fs.readFile**：无背压控制，可能导致内存暴涨
> - **Stream**：支持背压（Backpressure），自动调节读取速度
> 
> **示例对比：**
> ```javascript
> // ❌ readFile（1GB 文件 = 1GB 内存）
> fs.readFile('./1gb.txt', (err, data) => {
>   console.log(data.length);  // 1073741824 字节
> });
> 
> // ✅ Stream（1GB 文件 = 64KB 内存）
> fs.createReadStream('./1gb.txt')
>   .on('data', (chunk) => {
>     console.log(chunk.length);  // 65536 字节（64KB）
>   });
> ```

**加分项：**
> 在实际项目中，我们处理日志文件时，从 fs.readFile 改为 Stream，成功处理了 10GB 的日志文件，内存占用从 10GB 降到 10MB，避免了服务器崩溃。

---

### 问题 2："Stream 的工作原理是什么？"

**标准答案：**
> Stream 基于**分块读取 + 事件驱动**的原理：
> 
> **1. 分块读取（Chunking）**
> - 文件分成固定大小的块（chunk）
> - 默认大小：64KB（可通过 highWaterMark 调整）
> - 每次只读取一个块到内存
> 
> **2. 事件驱动（Event-Driven）**
> - **data 事件**：每读取一个块触发
> - **end 事件**：读取完成触发
> - **error 事件**：读取错误触发
> 
> **3. 背压机制（Backpressure）**
> - 如果处理速度 < 读取速度
> - Stream 自动暂停读取（pause）
> - 等待处理完成后恢复（resume）
> 
> **4. 工作流程：**
> ```
> 文件系统
>   ↓ read 64KB
> 内部缓冲区（Buffer）
>   ↓ emit('data', chunk)
> 事件监听器（处理逻辑）
>   ↓ 处理完成
> 继续 read 下一个 64KB
> ```
> 
> **5. 核心优势：**
> - **内存可控**：始终占用固定内存
> - **并行处理**：读取和处理并行
> - **自动调节**：背压机制防止内存溢出

## ⚠️ 常见误区

### 误区 1：以为 Stream 总是更快

```javascript
// ❌ 错误理解：Stream 总是比 readFile 快
// 实际：小文件时 readFile 更快

// 小文件（< 10MB）
fs.readFile('./small.txt', (err, data) => {
  // 更快（一次性读取，无事件开销）
});

// ✅ 正确理解
// 小文件（< 10MB）：readFile 更快
// 大文件（> 100MB）：Stream 更快（内存安全）
```

---

### 误区 2：忘记处理错误

```javascript
// ❌ 错误：忘记处理 error 事件
const stream = fs.createReadStream('./not-exist.txt');
stream.on('data', (chunk) => {
  console.log(chunk);
});
// 文件不存在时，程序崩溃

// ✅ 正确
const stream = fs.createReadStream('./not-exist.txt');
stream.on('data', (chunk) => {
  console.log(chunk);
});
stream.on('error', (err) => {
  console.error('Error:', err);
});
```

---

### 误区 3：不关注背压

```javascript
// ❌ 错误：不处理背压
const readStream = fs.createReadStream('./input.txt');
const writeStream = fs.createWriteStream('./output.txt');

readStream.on('data', (chunk) => {
  writeStream.write(chunk);  // 可能导致内存暴涨
});

// ✅ 正确：使用 pipe 自动处理背压
readStream.pipe(writeStream);
```

### 技巧 4：多流合并

```javascript
const fs = require('fs');
const { PassThrough } = require('stream');

// 合并多个文件
function mergeFiles(files, output) {
  const writeStream = fs.createWriteStream(output);
  
  function processNext() {
    if (files.length === 0) {
      writeStream.end();
      return;
    }
    
    const file = files.shift();
    const readStream = fs.createReadStream(file);
    
    readStream.pipe(writeStream, { end: false });
    readStream.on('end', processNext);
  }
  
  processNext();
}

mergeFiles(['part1.txt', 'part2.txt', 'part3.txt'], 'merged.txt');
```

```javascript
const fs = require('fs');
const zlib = require('zlib');

function compressFile(input, output) {
  return new Promise((resolve, reject) => {
    const gzip = zlib.createGzip();
    const source = fs.createReadStream(input);
    const destination = fs.createWriteStream(output);
    
    source.pipe(gzip).pipe(destination);
    
    destination.on('finish', resolve);
    destination.on('error', reject);
  });
}

  console.log('Compression complete');
```
