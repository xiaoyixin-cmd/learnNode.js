## 📚 核心知识点

### 1. Buffer vs TypedArray

| 特性 | Buffer | TypedArray |
|------|--------|-----------|
| **环境** | Node.js 专用 | 浏览器 + Node.js |
| **底层** | ArrayBuffer | ArrayBuffer |
| **编码** | 支持（UTF-8、Base64） | 不支持 |
| **API** | 丰富（slice、concat） | 基础 |
| **性能** | 高（C++ 优化） | 高 |
| **用途** | 文件、网络 I/O | 数值计算 |

**核心：**
> Buffer 是基于 `Uint8Array` 的 Node.js 扩展，增加了编码解码功能。

---

### 2. Buffer 的内存分配

```javascript
// 方式 1：安全分配（自动填充 0）
const buf1 = Buffer.alloc(10);
console.log(buf1);  // <Buffer 00 00 00 00 00 00 00 00 00 00>

// 方式 2：不安全分配（未初始化，性能更高）
const buf2 = Buffer.allocUnsafe(10);
console.log(buf2);  // <Buffer 78 9a bc de ...>（随机数据）

// 方式 3：从已有数据创建
const buf3 = Buffer.from('hello');
const buf4 = Buffer.from([0x68, 0x65, 0x6c, 0x6c, 0x6f]);
```

**性能对比：**
```javascript
console.time('alloc');
for (let i = 0; i < 1e6; i++) {
  Buffer.alloc(1024);
}
console.timeEnd('alloc');  // ~150ms

console.time('allocUnsafe');
for (let i = 0; i < 1e6; i++) {
  Buffer.allocUnsafe(1024);
}
console.timeEnd('allocUnsafe');  // ~50ms（快 3 倍）
```

## 💻 内存共享机制

### 1. Buffer 和 Uint8Array 共享内存

```javascript
// Buffer → Uint8Array
const buf = Buffer.from('hello');
const uint8 = new Uint8Array(buf.buffer, buf.byteOffset, buf.byteLength);

console.log(buf);    // <Buffer 68 65 6c 6c 6f>
console.log(uint8);  // Uint8Array(5) [ 104, 101, 108, 108, 111 ]

// 修改 Uint8Array，Buffer 也改变（共享内存）
uint8[0] = 0x48;  // 'H'
console.log(buf.toString());  // "Hello"
```

---

### 2. ArrayBuffer 共享

```javascript
// 创建共享内存
const arrayBuffer = new ArrayBuffer(16);

// 不同视图操作同一块内存
const uint8View = new Uint8Array(arrayBuffer);
const uint16View = new Uint16Array(arrayBuffer);
const float32View = new Float32Array(arrayBuffer);

uint8View[0] = 255;
console.log(uint16View[0]);  // 255（共享内存）

float32View[0] = 3.14;
console.log(uint8View[0]);  // 195（浮点数的第一个字节）
```

---

### 3. 零拷贝（Zero-Copy）

```javascript
// ❌ 传统方式：拷贝数据
const data = Buffer.from('A'.repeat(1024 * 1024));  // 1 MB
const copy = Buffer.from(data);  // 拷贝 1 MB（耗时）

// ✅ 零拷贝：共享内存
const slice = data.slice(0, 1024);  // 不拷贝，仅引用
slice[0] = 0x42;  // 'B'
console.log(data[0]);  // 0x42（修改影响原数据）

// Node.js 14+ 新 API（真正的切片，不共享内存）
const subarray = data.subarray(0, 1024);
subarray[0] = 0x42;
console.log(data[0]);  // 0x42（仍然共享内存）
```

## 💻 实战案例

### 案例 1：文件读取（零拷贝）

```javascript
const fs = require('fs');

// ❌ 传统方式：多次拷贝
const data = fs.readFileSync('./file.txt');  // 拷贝 1
const str = data.toString('utf-8');          // 拷贝 2
console.log(str);

// ✅ 零拷贝：直接操作 Buffer
const fd = fs.openSync('./file.txt', 'r');
const buffer = Buffer.allocUnsafe(1024);

const bytesRead = fs.readSync(fd, buffer, 0, 1024, 0);
const str = buffer.toString('utf-8', 0, bytesRead);  // 仅解码

fs.closeSync(fd);
```

### 案例 2：网络传输（Buffer Pool）

```javascript
const net = require('net');

// Buffer 池（复用内存）
const bufferPool = {
  buffers: [],
  
  allocate(size) {
    const buf = this.buffers.find(b => b.length >= size && !b.inUse);
    if (buf) {
      buf.inUse = true;
      return buf.slice(0, size);
    }
    
    const newBuf = Buffer.allocUnsafe(size);
    newBuf.inUse = true;
    this.buffers.push(newBuf);
    return newBuf;
  },
  
  release(buf) {
    const original = this.buffers.find(b => 
      b.buffer === buf.buffer && b.byteOffset === buf.byteOffset
    );
    if (original) {
      original.inUse = false;
    }
  }
};

const server = net.createServer((socket) => {
  socket.on('data', (data) => {
    const buf = bufferPool.allocate(data.length);
    data.copy(buf);
    
    // 处理数据...
    
    bufferPool.release(buf);
  });
});

server.listen(3000);
```

### 案例 3：二进制协议解析

```javascript
// 自定义二进制协议：
// [Header: 4 bytes][Length: 4 bytes][Data: N bytes]

function parsePacket(buffer) {
  // 使用 DataView 解析多字节数据
  const view = new DataView(buffer.buffer, buffer.byteOffset, buffer.byteLength);
  
  const header = view.getUint32(0);    // 4 字节（大端序）
  const length = view.getUint32(4);    // 4 字节
  const data = buffer.slice(8, 8 + length);
  
  return { header, length, data };
}

// 使用
const packet = Buffer.from([
  0x00, 0x00, 0x00, 0x01,  // Header: 1
  0x00, 0x00, 0x00, 0x05,  // Length: 5
  0x68, 0x65, 0x6c, 0x6c, 0x6f  // Data: "hello"
]);

const parsed = parsePacket(packet);
console.log(parsed);
// { header: 1, length: 5, data: <Buffer 68 65 6c 6c 6f> }
```

### 案例 4：图片像素操作

```javascript
const fs = require('fs');
const { createCanvas, loadImage } = require('canvas');

async function convertToGrayscale(inputPath, outputPath) {
  const image = await loadImage(inputPath);
  const canvas = createCanvas(image.width, image.height);
  const ctx = canvas.getContext('2d');
  
  ctx.drawImage(image, 0, 0);
  
  // 获取像素数据（Uint8ClampedArray）
  const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
  const pixels = imageData.data;  // [R, G, B, A, R, G, B, A, ...]
  
  // 灰度化：gray = 0.299*R + 0.587*G + 0.114*B
  for (let i = 0; i < pixels.length; i += 4) {
    const gray = Math.round(
      pixels[i] * 0.299 +      // R
      pixels[i + 1] * 0.587 +  // G
      pixels[i + 2] * 0.114    // B
    );
    
    pixels[i] = gray;      // R
    pixels[i + 1] = gray;  // G
    pixels[i + 2] = gray;  // B
    // pixels[i + 3] 是 Alpha，保持不变
  }
  
  ctx.putImageData(imageData, 0, 0);
  
  const buffer = canvas.toBuffer('image/png');
  fs.writeFileSync(outputPath, buffer);
}

// 使用
convertToGrayscale('./input.png', './output.png');
```

## 🎓 面试标准答题

### 问题 1："Buffer 和 TypedArray 的区别是什么？"

**标准答案：**
> **Buffer 是 Node.js 专用的二进制数据容器，基于 `Uint8Array` 实现。**
> 
> **核心区别：**
> 
> | 特性 | Buffer | TypedArray |
> |------|--------|-----------|
> | **环境** | Node.js | 浏览器 + Node.js |
> | **底层** | Uint8Array | ArrayBuffer |
> | **编码** | ✅ 支持（UTF-8、Base64） | ❌ 不支持 |
> | **API** | 丰富（slice、concat、indexOf） | 基础 |
> | **性能** | 高（C++ 优化） | 高 |
> 
> **示例：**
> ```javascript
> // Buffer（Node.js）
> const buf = Buffer.from('hello', 'utf-8');
> console.log(buf);  // <Buffer 68 65 6c 6c 6f>
> 
> // TypedArray（通用）
> const uint8 = new Uint8Array([104, 101, 108, 108, 111]);
> console.log(uint8);  // Uint8Array(5) [ 104, 101, 108, 108, 111 ]
> ```
> 
> **内存共享：**
> ```javascript
> const buf = Buffer.from('hello');
> const uint8 = new Uint8Array(buf.buffer);
> uint8[0] = 0x48;  // 修改 TypedArray
> console.log(buf.toString());  // "Hello"（Buffer 也改变）
> ```

**加分项：**
> Buffer 继承自 `Uint8Array`，可以使用所有 TypedArray 的方法（`map`、`filter`、`reduce`），但 Buffer 额外提供了 `toString()`、`indexOf()` 等方法。

---

### 问题 2："Buffer 如何实现零拷贝（Zero-Copy）？"

**标准答案：**
> **零拷贝（Zero-Copy）** 是指多个对象共享同一块内存，避免数据复制。
> 
> **实现方式：**
> 
> **1. slice()（共享内存）**
> ```javascript
> const buf = Buffer.from('A'.repeat(1024 * 1024));  // 1 MB
> const slice = buf.slice(0, 1024);  // 不拷贝，仅引用
> 
> slice[0] = 0x42;  // 'B'
> console.log(buf[0]);  // 0x42（修改影响原数据）
> ```
> 
> **2. Buffer 和 TypedArray 共享**
> ```javascript
> const buf = Buffer.from('hello');
> const uint8 = new Uint8Array(buf.buffer);  // 共享内存
> 
> uint8[0] = 0x48;
> console.log(buf.toString());  // "Hello"
> ```
> 
> **3. 应用场景**
> - **文件读取**：避免多次拷贝
> - **网络传输**：复用 Buffer Pool
> - **二进制协议**：直接操作原始数据
> 
> **4. 注意事项**
> - `slice()` 共享内存，修改会影响原数据
> - `Buffer.from(buf)` 会拷贝数据，不共享
> - `subarray()` 和 `slice()` 行为相同

## ⚠️ 常见误区

### 误区 1：以为 slice() 会拷贝数据

```javascript
// ❌ 错误理解：以为 slice() 会拷贝
const buf = Buffer.from('hello');
const slice = buf.slice(0, 2);  // 以为创建了新数据

slice[0] = 0x48;  // 'H'
console.log(buf.toString());  // "Hello"（原数据被修改）

// ✅ 正确：真正拷贝
const copy = Buffer.from(buf);
copy[0] = 0x48;
console.log(buf.toString());  // "hello"（原数据不变）
```

---

### 误区 2：Buffer.allocUnsafe() 的安全性

```javascript
// ❌ 错误：直接使用未初始化的内存（可能泄露敏感数据）
const buf = Buffer.allocUnsafe(10);
console.log(buf);  // <Buffer 78 9a bc de ...>（旧内存数据）

// ✅ 正确：使用前填充 0
const buf2 = Buffer.allocUnsafe(10).fill(0);
console.log(buf2);  // <Buffer 00 00 00 00 00 00 00 00 00 00>
```

---

### 误区 3：忘记字节序（Endianness）

```javascript
// 小端序 vs 大端序
const buf = Buffer.allocUnsafe(4);

buf.writeUInt32LE(0x12345678, 0);  // 小端序
console.log(buf);  // <Buffer 78 56 34 12>

buf.writeUInt32BE(0x12345678, 0);  // 大端序
console.log(buf);  // <Buffer 12 34 56 78>
```

```javascript
function myConcat(buffers) {
  const totalLength = buffers.reduce((sum, buf) => sum + buf.length, 0);
  const result = Buffer.allocUnsafe(totalLength);
  
  let offset = 0;
  for (const buf of buffers) {
    buf.copy(result, offset);
    offset += buf.length;
  }
  
  return result;
}
```
